001package org.nasdanika.html.model.app.gen;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import org.eclipse.emf.common.util.EList;
007import org.eclipse.emf.ecore.EObject;
008import org.eclipse.emf.ecore.util.EcoreUtil;
009import org.nasdanika.common.BiSupplier;
010import org.nasdanika.common.ConsumerFactory;
011import org.nasdanika.common.Context;
012import org.nasdanika.common.Function;
013import org.nasdanika.common.ListCompoundSupplierFactory;
014import org.nasdanika.common.ProgressMonitor;
015import org.nasdanika.common.SupplierFactory;
016import org.nasdanika.emf.EObjectAdaptable;
017import org.nasdanika.exec.content.ContentFactory;
018import org.nasdanika.exec.content.Text;
019import org.nasdanika.html.HTMLElement;
020import org.nasdanika.html.Tag;
021import org.nasdanika.html.bootstrap.BootstrapFactory;
022import org.nasdanika.html.bootstrap.Navs;
023import org.nasdanika.html.model.app.Header;
024import org.nasdanika.html.model.app.Label;
025import org.nasdanika.html.model.bootstrap.Appearance;
026
027public class HeaderConsumerFactoryAdapter extends PagePartConsumerFactoryAdapter<Header> {
028
029        protected HeaderConsumerFactoryAdapter(Header header) {
030                super(header);
031        }
032        
033        @Override
034        protected List<EObject> getContent() {          
035                // A bit convoluted, refactor later to use a function and add class to the title tag - similar to NavBar
036                List<EObject> ret = new ArrayList<>();
037                Label title = getTarget().getTitle();
038                if (title != null) {
039                        Label cTitle = EcoreUtil.copy(title);
040                        cTitle.eAdapters().add(AppAdapterFactory.INSTANCE.adapt(cTitle, SupplierFactory.Provider.class));
041                        
042                        Appearance titleAppearance = org.nasdanika.html.model.bootstrap.BootstrapFactory.eINSTANCE.createAppearance();
043                        titleAppearance.eAdapters().add(AppAdapterFactory.INSTANCE.adapt(titleAppearance, ConsumerFactory.class));
044                        cTitle.setAppearance(titleAppearance);
045                        
046                        Text cText = ContentFactory.eINSTANCE.createText();
047                        cText.eAdapters().add(AppAdapterFactory.INSTANCE.adapt(cText, SupplierFactory.class));
048                        cText.setContent("nsd-app-header-title");
049                        titleAppearance.getAttributes().put("class", cText);
050                        ret.add(cTitle);
051                }
052                ret.addAll(super.getContent());         
053                return ret;
054        }
055        
056        private Function<BiSupplier<HTMLElement<?>, List<Object>>, HTMLElement<?>> createNavsFunction(Context context) {
057                return new Function<BiSupplier<HTMLElement<?>,List<Object>>, HTMLElement<?>>() {
058                        
059                        @Override
060                        public double size() {
061                                return 1;
062                        }
063                        
064                        @Override
065                        public String name() {
066                                return "Navs";
067                        }
068                        
069                        @Override
070                        public HTMLElement<?> execute(BiSupplier<HTMLElement<?>, List<Object>> input, ProgressMonitor progressMonitor) throws Exception {
071                                Tag ret = (Tag) input.getFirst();
072                                Tag navs = Util.navs(input.getSecond(), context.get(BootstrapFactory.class, BootstrapFactory.INSTANCE));
073                                navs.addClass("nsd-app-header-navs");
074                                ret.content(navs);                              
075                                return ret;
076                        }
077                };
078        }
079        
080        /**
081         * Adapts items to suppliers, builds {@link Navs} from items and adds them to the header.
082         */
083        @Override
084        protected Function<HTMLElement<?>, HTMLElement<?>> createConfigureFunction(Context context) throws Exception {
085                EList<EObject> items = getTarget().getItems();
086                if (items.isEmpty()) {
087                        return super.createConfigureFunction(context);
088                }
089                Function<HTMLElement<?>, BiSupplier<HTMLElement<?>, List<Object>>> itemsFunction = new ListCompoundSupplierFactory<>("Items", EObjectAdaptable.adaptToSupplierFactoryNonNull(items, Object.class)).<HTMLElement<?>>asFunctionFactory().create(context);
090                return super.createConfigureFunction(context).then(itemsFunction).then(createNavsFunction(context));
091        }
092
093}
