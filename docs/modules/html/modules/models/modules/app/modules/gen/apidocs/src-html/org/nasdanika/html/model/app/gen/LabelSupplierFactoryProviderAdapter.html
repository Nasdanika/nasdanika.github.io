001package org.nasdanika.html.model.app.gen;
002
003import org.eclipse.emf.common.notify.impl.AdapterImpl;
004import org.nasdanika.common.SupplierFactory;
005import org.nasdanika.html.jstree.JsTreeNode;
006import org.nasdanika.html.model.app.Label;
007import org.nasdanika.html.model.html.Tag;
008
009public class LabelSupplierFactoryProviderAdapter<M extends Label> extends AdapterImpl implements SupplierFactory.Provider {
010
011        public LabelSupplierFactoryProviderAdapter(M label) {
012                setTarget(label);
013        }
014        
015        @Override
016        public boolean isAdapterForType(Object type) {
017                return type == SupplierFactory.Provider.class;
018        }
019
020        @SuppressWarnings("unchecked")
021        @Override
022        public <T> SupplierFactory<T> getFactory(Class<T> type) {
023                if (type.isAssignableFrom(Tag.class)) {
024                        return (SupplierFactory<T>) new LabelTagSupplierFactoryAdapter<>((M) getTarget());
025                }
026                // TODO Card
027                
028                if (type.isAssignableFrom(JsTreeNode.class)) {
029                        return (SupplierFactory<T>) new LabelJsTreeNodeSupplierFactoryAdapter<>((M) getTarget());
030                }
031                
032                return null;
033        }
034
035}
