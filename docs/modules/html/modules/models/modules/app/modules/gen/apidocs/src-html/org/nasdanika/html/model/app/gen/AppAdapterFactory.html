001package org.nasdanika.html.model.app.gen;
002
003import org.nasdanika.common.ConsumerFactory;
004import org.nasdanika.common.SupplierFactory;
005import org.nasdanika.emf.FunctionAdapterFactory;
006import org.nasdanika.exec.gen.ExecutionParticpantAdapterFactory;
007import org.nasdanika.html.HTMLElement;
008import org.nasdanika.html.model.app.AppPackage;
009import org.nasdanika.html.model.app.ContentPanel;
010import org.nasdanika.html.model.app.Footer;
011import org.nasdanika.html.model.app.Header;
012import org.nasdanika.html.model.app.Label;
013import org.nasdanika.html.model.app.Link;
014import org.nasdanika.html.model.app.NavigationBar;
015import org.nasdanika.html.model.app.NavigationPanel;
016import org.nasdanika.html.model.app.Page;
017import org.nasdanika.html.model.bootstrap.gen.BootstrapAdapterFactory;
018import org.nasdanika.html.model.html.gen.HtmlAdapterFactory;
019
020/**
021 * @author Pavel
022 *
023 */
024public class AppAdapterFactory extends BootstrapAdapterFactory {
025        
026        public static AppAdapterFactory INSTANCE = new AppAdapterFactory();
027        
028        @SuppressWarnings({ "unchecked", "rawtypes" })
029        public AppAdapterFactory() {
030                ClassLoader classLoader = getClassLoader();
031                
032                registerAdapterFactory(
033                                new FunctionAdapterFactory<SupplierFactory<HTMLElement<?>>, Page>(
034                                        AppPackage.Literals.PAGE, 
035                                        (Class) SupplierFactory.class, 
036                                        classLoader, 
037                                        PageSupplierFactoryAdapter::new));
038                
039                registerAdapterFactory(
040                                new FunctionAdapterFactory<SupplierFactory.Provider, Label>(
041                                        AppPackage.Literals.LABEL, 
042                                        SupplierFactory.Provider.class, 
043                                        classLoader, 
044                                        LabelSupplierFactoryProviderAdapter::new));
045                
046                registerAdapterFactory(
047                                new FunctionAdapterFactory<SupplierFactory.Provider, Link>(
048                                        AppPackage.Literals.LINK, 
049                                        SupplierFactory.Provider.class, 
050                                        classLoader, 
051                                        LinkSupplierFactoryProviderAdapter::new));
052                
053                registerAdapterFactory(
054                                new FunctionAdapterFactory<ConsumerFactory<HTMLElement<?>>, ContentPanel>(
055                                        AppPackage.Literals.CONTENT_PANEL, 
056                                        (Class) ConsumerFactory.class, 
057                                        classLoader, 
058                                        ContentPanelConsumerFactoryAdapter::new));              
059                
060                registerAdapterFactory(
061                                new FunctionAdapterFactory<ConsumerFactory<HTMLElement<?>>, Footer>(
062                                        AppPackage.Literals.FOOTER, 
063                                        (Class) ConsumerFactory.class, 
064                                        classLoader, 
065                                        FooterConsumerFactoryAdapter::new));            
066                
067                registerAdapterFactory(
068                                new FunctionAdapterFactory<ConsumerFactory<HTMLElement<?>>, Header>(
069                                        AppPackage.Literals.HEADER, 
070                                        (Class) ConsumerFactory.class, 
071                                        classLoader, 
072                                        HeaderConsumerFactoryAdapter::new));            
073                
074                registerAdapterFactory(
075                                new FunctionAdapterFactory<ConsumerFactory<HTMLElement<?>>, NavigationBar>(
076                                        AppPackage.Literals.NAVIGATION_BAR, 
077                                        (Class) ConsumerFactory.class, 
078                                        classLoader, 
079                                        NavigationBarConsumerFactoryAdapter::new));             
080                
081                registerAdapterFactory(
082                                new FunctionAdapterFactory<ConsumerFactory<HTMLElement<?>>, NavigationPanel>(
083                                        AppPackage.Literals.NAVIGATION_PANEL, 
084                                        (Class) ConsumerFactory.class, 
085                                        classLoader, 
086                                        NavigationPanelConsumerFactoryAdapter::new));   
087                
088                // Lower-level factories
089                registerAdapterFactory(new ExecutionParticpantAdapterFactory());
090                registerAdapterFactory(new HtmlAdapterFactory());
091                registerAdapterFactory(new BootstrapAdapterFactory());                          
092        }
093        
094}
