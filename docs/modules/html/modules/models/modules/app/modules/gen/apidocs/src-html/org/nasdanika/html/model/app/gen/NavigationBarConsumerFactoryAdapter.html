001package org.nasdanika.html.model.app.gen;
002
003import java.util.List;
004import java.util.Map;
005
006import org.eclipse.emf.common.util.EList;
007import org.eclipse.emf.ecore.EObject;
008import org.eclipse.emf.ecore.EStructuralFeature;
009import org.nasdanika.common.BiSupplier;
010import org.nasdanika.common.Context;
011import org.nasdanika.common.Function;
012import org.nasdanika.common.ListCompoundSupplierFactory;
013import org.nasdanika.common.MapCompoundSupplierFactory;
014import org.nasdanika.common.ProgressMonitor;
015import org.nasdanika.emf.EObjectAdaptable;
016import org.nasdanika.html.HTMLElement;
017import org.nasdanika.html.Tag;
018import org.nasdanika.html.bootstrap.BootstrapFactory;
019import org.nasdanika.html.bootstrap.Navs;
020import org.nasdanika.html.model.app.AppPackage;
021import org.nasdanika.html.model.app.Label;
022import org.nasdanika.html.model.app.NavigationBar;
023
024public class NavigationBarConsumerFactoryAdapter extends PagePartConsumerFactoryAdapter<NavigationBar> {
025
026        protected NavigationBarConsumerFactoryAdapter(NavigationBar navigationBar) {
027                super(navigationBar);
028        }
029        
030        private Function<BiSupplier<HTMLElement<?>, Map<EStructuralFeature, Object>>, HTMLElement<?>> createNavbarFunction(Context context) {
031                return new Function<BiSupplier<HTMLElement<?>,Map<EStructuralFeature, Object>>, HTMLElement<?>>() {
032                        
033                        @Override
034                        public double size() {
035                                return 1;
036                        }
037                        
038                        @Override
039                        public String name() {
040                                return "Navs";
041                        }
042                        
043                        @Override
044                        public HTMLElement<?> execute(BiSupplier<HTMLElement<?>, Map<EStructuralFeature, Object>> input, ProgressMonitor progressMonitor) throws Exception {
045                                Tag ret = (Tag) input.getFirst();
046                                Tag brand = (Tag) input.getSecond().get(AppPackage.Literals.NAVIGATION_BAR__BRAND);
047                                @SuppressWarnings("unchecked")
048                                List<Object> items = (List<Object>) input.getSecond().get(AppPackage.Literals.PAGE_PART__ITEMS);        
049                                NavigationBar semanticElement = getTarget();
050                                ret.content(Util.navbar(
051                                                brand, 
052                                                items, 
053                                                semanticElement.getExpand(), 
054                                                semanticElement.isDark(),
055                                                semanticElement.getBackground(),                                                
056                                                context.get(BootstrapFactory.class, BootstrapFactory.INSTANCE)));       
057                                
058                                return ret;
059                        }
060                };
061        }
062        
063        /**
064         * Adapts items to suppliers, builds {@link Navs} from items and adds them to the header.
065         */
066        @Override
067        protected Function<HTMLElement<?>, HTMLElement<?>> createConfigureFunction(Context context) throws Exception {
068                Label brand = getTarget().getBrand();
069                EList<EObject> items = getTarget().getItems();
070                if (brand == null && items.isEmpty()) {
071                        return super.createConfigureFunction(context);
072                }
073                
074                MapCompoundSupplierFactory<EStructuralFeature, Object> featuresSupplierFactory = new MapCompoundSupplierFactory<>("Brand and items");
075                
076                if (brand != null) {
077                        featuresSupplierFactory.put(AppPackage.Literals.NAVIGATION_BAR__BRAND, EObjectAdaptable.adaptToSupplierFactoryNonNull(brand, Object.class));
078                }
079                
080                if (!items.isEmpty()) {
081                        featuresSupplierFactory.put(AppPackage.Literals.PAGE_PART__ITEMS, new ListCompoundSupplierFactory<>("Items", EObjectAdaptable.adaptToSupplierFactoryNonNull(items, Object.class)));
082                }
083                
084                Function<HTMLElement<?>, BiSupplier<HTMLElement<?>, Map<EStructuralFeature, Object>>> featuresFunction = featuresSupplierFactory.<HTMLElement<?>>asFunctionFactory().create(context);                   
085                return super.createConfigureFunction(context).then(featuresFunction).then(createNavbarFunction(context));
086        }
087        
088}
