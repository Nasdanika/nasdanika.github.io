001package org.nasdanika.html.model.app.gen;
002
003import java.util.LinkedHashMap;
004import java.util.Map;
005
006import org.eclipse.emf.common.notify.AdapterFactory;
007import org.eclipse.emf.ecore.EStructuralFeature;
008import org.nasdanika.common.BiSupplier;
009import org.nasdanika.common.Context;
010import org.nasdanika.common.FunctionFactory;
011import org.nasdanika.common.MapCompoundConsumerFactory;
012import org.nasdanika.common.ProgressMonitor;
013import org.nasdanika.common.Supplier;
014import org.nasdanika.common.SupplierFactory;
015import org.nasdanika.emf.EObjectAdaptable;
016import org.nasdanika.html.HTMLElement;
017import org.nasdanika.html.Tag;
018import org.nasdanika.html.bootstrap.BootstrapElement;
019import org.nasdanika.html.bootstrap.BootstrapFactory;
020import org.nasdanika.html.bootstrap.Container;
021import org.nasdanika.html.bootstrap.Container.Row;
022import org.nasdanika.html.bootstrap.Container.Row.Col;
023import org.nasdanika.html.model.app.AppPackage;
024import org.nasdanika.html.model.app.ContentPanel;
025import org.nasdanika.html.model.app.Footer;
026import org.nasdanika.html.model.app.Header;
027import org.nasdanika.html.model.app.NavigationBar;
028import org.nasdanika.html.model.app.NavigationPanel;
029import org.nasdanika.html.model.app.Page;
030import org.nasdanika.html.model.bootstrap.Appearance;
031import org.nasdanika.html.model.bootstrap.gen.BootstrapElementSupplierFactoryAdapter;
032
033public class PageSupplierFactoryAdapter extends BootstrapElementSupplierFactoryAdapter<Page, BootstrapElement<?,?>> {
034        
035        public PageSupplierFactoryAdapter(Page page, AdapterFactory adapterFactory) {
036                super(page, adapterFactory);
037        }
038                
039        protected Supplier<BiSupplier<Map<EStructuralFeature, HTMLElement<?>>, HTMLElement<?>>> createContainerSupplier(Context context) {
040                return new Supplier<BiSupplier<Map<EStructuralFeature, HTMLElement<?>>, HTMLElement<?>>>() {
041
042                        @Override
043                        public double size() {
044                                return 1;
045                        }
046
047                        @Override
048                        public String name() {
049                                return "Container";
050                        }
051
052                        @Override
053                        public BiSupplier<Map<EStructuralFeature, HTMLElement<?>>, HTMLElement<?>> execute(ProgressMonitor progressMonitor) throws Exception {
054                                BootstrapFactory bootstrapFactory = context.get(BootstrapFactory.class, BootstrapFactory.INSTANCE);
055                                Page semanticElement = getTarget();
056                                Container container = semanticElement.isFluid() ? bootstrapFactory.fluidContainer() : bootstrapFactory.container();
057                                
058                                Map<EStructuralFeature, HTMLElement<?>> parts = new LinkedHashMap<>();
059                                
060                                if (semanticElement.getHeader() != null) {
061                                        Col header = container.row().col();                             
062                                        header.toHTMLElement().addClass("nsd-app-header");                                      
063                                        parts.put(AppPackage.Literals.PAGE__HEADER, header.toHTMLElement());
064                                }
065                                
066                                if (semanticElement.getNavigationBar() != null) {
067                                        Col navigationBar =  container.row().col();
068                                        navigationBar.toHTMLElement().addClass("nsd-app-navbar");
069                                        parts.put(AppPackage.Literals.PAGE__NAVIGATION_BAR, navigationBar.toHTMLElement());
070                                }
071
072                                if (semanticElement.getNavigationPanel() != null || semanticElement.getContentPanel() != null) {
073                                        Row contentRow = container.row();
074                                        contentRow.toHTMLElement().addClass("nsd-app-content-row");
075                                        parts.put(AppPackage.Literals.PAGE__CONTENT_ROW_APPEARANCE, contentRow.toHTMLElement());
076
077                                        if (semanticElement.getNavigationPanel() != null) {
078                                                Col navigationPanel = contentRow.col();
079                                                navigationPanel.toHTMLElement().addClass("nsd-app-navigation-panel");
080                                                parts.put(AppPackage.Literals.PAGE__NAVIGATION_PANEL, navigationPanel.toHTMLElement());
081                                        }
082                                
083                                        if (semanticElement.getContentPanel() != null) {
084                                                Col contentPanel = contentRow.col();
085                                                contentPanel.toHTMLElement().addClass("nsd-app-content-panel");
086                                                parts.put(AppPackage.Literals.PAGE__CONTENT_PANEL, contentPanel.toHTMLElement());
087                                        }
088                                }
089
090                                if (semanticElement.getFooter() != null) {
091                                        Col footer =  container.row().col();
092                                        footer.toHTMLElement().addClass("nsd-app-footer");
093                                        parts.put(AppPackage.Literals.PAGE__FOOTER, footer.toHTMLElement());
094                                }
095                                
096                                Tag containerTag = container.toHTMLElement();
097                                containerTag.addClass("nsd-app-container");
098                                
099                                return new BiSupplier<Map<EStructuralFeature,HTMLElement<?>>, HTMLElement<?>>() {
100
101                                        @Override
102                                        public Map<EStructuralFeature, HTMLElement<?>> getFirst() {
103                                                return parts;
104                                        }
105
106                                        @Override
107                                        public HTMLElement<?> getSecond() {
108                                                return containerTag;
109                                        }
110                                };
111                        }
112                        
113                };
114        }
115        
116        @Override
117        public Supplier<HTMLElement<?>> createHTMLElementSupplier(Context context) throws Exception {
118                MapCompoundConsumerFactory<EStructuralFeature, HTMLElement<?>> partsFactory = new MapCompoundConsumerFactory<>("Page parts");
119                Page semanticElement = (Page) getTarget();
120                
121                Header header = semanticElement.getHeader();
122                if (header != null) {
123                        partsFactory.put(AppPackage.Literals.PAGE__HEADER, EObjectAdaptable.adaptToConsumerFactoryNonNull(header, HTMLElement.class));                  
124                }
125                NavigationBar navigationBar = semanticElement.getNavigationBar();
126                if (navigationBar != null) {
127                        partsFactory.put(AppPackage.Literals.PAGE__NAVIGATION_BAR, EObjectAdaptable.adaptToConsumerFactoryNonNull(navigationBar, HTMLElement.class));                   
128                }
129                Appearance conentRowAppearance = semanticElement.getContentRowAppearance();
130                if (conentRowAppearance != null) {
131                        partsFactory.put(AppPackage.Literals.PAGE__CONTENT_ROW_APPEARANCE, EObjectAdaptable.adaptToConsumerFactoryNonNull(conentRowAppearance, HTMLElement.class));                     
132                }
133                NavigationPanel navigationPanel = semanticElement.getNavigationPanel();
134                if (navigationPanel != null) {
135                        partsFactory.put(AppPackage.Literals.PAGE__NAVIGATION_PANEL, EObjectAdaptable.adaptToConsumerFactoryNonNull(navigationPanel, HTMLElement.class));                       
136                }
137                ContentPanel contentPanel = semanticElement.getContentPanel();
138                if (contentPanel != null) {
139                        partsFactory.put(AppPackage.Literals.PAGE__CONTENT_PANEL, EObjectAdaptable.adaptToConsumerFactoryNonNull(contentPanel, HTMLElement.class));                     
140                }
141                Footer footer = semanticElement.getFooter();
142                if (footer != null) {
143                        partsFactory.put(AppPackage.Literals.PAGE__FOOTER, EObjectAdaptable.adaptToConsumerFactoryNonNull(footer, HTMLElement.class));                  
144                }       
145                
146                SupplierFactory<BiSupplier<Map<EStructuralFeature,HTMLElement<?>>, HTMLElement<?>>> containerSupplierFactory = this::createContainerSupplier;           
147                FunctionFactory<BiSupplier<Map<EStructuralFeature, HTMLElement<?>>, HTMLElement<?>>, HTMLElement<?>> partsFunctionFactory = partsFactory.asBiSupplierFunctionFactory();
148                return containerSupplierFactory.then(partsFunctionFactory).create(context);
149        }       
150        
151}
