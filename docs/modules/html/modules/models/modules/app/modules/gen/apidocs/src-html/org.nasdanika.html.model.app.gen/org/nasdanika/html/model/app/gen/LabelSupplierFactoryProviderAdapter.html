001package org.nasdanika.html.model.app.gen;
002
003import org.eclipse.emf.common.notify.AdapterFactory;
004import org.eclipse.emf.common.notify.impl.AdapterImpl;
005import org.nasdanika.common.SupplierFactory;
006import org.nasdanika.html.jstree.JsTreeNode;
007import org.nasdanika.html.model.app.Label;
008import org.nasdanika.html.model.html.Tag;
009
010public class LabelSupplierFactoryProviderAdapter<M extends Label> extends AdapterImpl implements SupplierFactory.Provider {
011
012        protected AdapterFactory adapterFactory;
013
014        public LabelSupplierFactoryProviderAdapter(M label, AdapterFactory adapterFactory) {
015                setTarget(label);
016                this.adapterFactory = adapterFactory;
017        }
018        
019        @Override
020        public boolean isAdapterForType(Object type) {
021                return type == SupplierFactory.Provider.class;
022        }
023
024        @SuppressWarnings("unchecked")
025        @Override
026        public <T> SupplierFactory<T> getFactory(Class<T> type) {
027                if (type.isAssignableFrom(Tag.class)) {
028                        return (SupplierFactory<T>) new LabelTagSupplierFactoryAdapter<>((M) getTarget(), adapterFactory);
029                }
030                // TODO Card
031                
032                if (type.isAssignableFrom(JsTreeNode.class)) {
033                        return (SupplierFactory<T>) new LabelJsTreeNodeSupplierFactoryAdapter<>((M) getTarget());
034                }
035                
036                return null;
037        }
038
039}
