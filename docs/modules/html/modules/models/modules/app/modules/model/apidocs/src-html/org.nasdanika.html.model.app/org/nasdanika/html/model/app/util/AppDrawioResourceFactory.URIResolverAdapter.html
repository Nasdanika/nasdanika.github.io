073        protected Map<EReference, List<Action>> createSemanticElement(
074                        Resource resource, 
075                        Element element,
076                        Map<Element, ElementEntry<Map<EReference, List<Action>>>> childEntries) {
077                
078                EReference containmentReference = getContainmentReference(resource, element);
079                Action action = null;
080                
081                if (element instanceof Document) {
082                        action = createDocumentAction(resource, (Document) element, childEntries);
083                        if (action == null) {
084                                for (Entry<Element, ElementEntry<Map<EReference, List<Action>>>> childEntry: childEntries.entrySet()) {
085                                        for (Entry<EReference, List<Action>> referenceEntry: childEntry.getValue().getSemanticElement().entrySet()) {                                           
086                                                for (Action rootAction: referenceEntry.getValue()) {
087                                                        if (isRootAction(childEntry.getKey(), rootAction)) {
088                                                                resource.getContents().add(rootAction);
089                                                        }
090                                                        addURIResolverAdapters(rootAction);
091                                                }
092                                        }
093                                }                               
094                        } else {
095                                resource.getContents().add(action);
096                                addURIResolverAdapters(action);
097                        }
098                } else if (element instanceof Page) {
099                        action = createPageAction(resource, (Page) element, childEntries);
100                } else if (element instanceof Node) {
101                        action = createNodeAction(resource, (Node) element, childEntries, containmentReference);
102                } else if (element instanceof Layer) {
103                        action = createLayerAction(resource, (Layer) element, childEntries, containmentReference);
104                } else if (element instanceof Connection) {
105                        action = createConnectionAction(resource, (Connection) element, childEntries, containmentReference);
106                }
107                
108                if (action == null) {
109                        // Pass-through with grouping
110                        Map<EReference, List<Action>> ret = new HashMap<>();                            
111                        for (ElementEntry<Map<EReference, List<Action>>> childEntry: childEntries.values()) {
112                                for (Entry<EReference, List<Action>> referenceEntry: childEntry.getSemanticElement().entrySet()) {                                              
113                                        ret.computeIfAbsent(referenceEntry.getKey(), ref -> new ArrayList<>()).addAll(referenceEntry.getValue());       
114                                }
115                        }
116                        return ret;
117                }
118                        
119                Map<EReference, List<Action>> accumulator = new HashMap<>();
120                for (ElementEntry<Map<EReference, List<Action>>> childEntry: childEntries.values()) {
121                        for (Entry<EReference, List<Action>> referenceEntry: childEntry.getSemanticElement().entrySet()) {
122                                accumulator.computeIfAbsent(referenceEntry.getKey(), r -> new ArrayList<>()).addAll(referenceEntry.getValue());
123                        }
124                }
125                
126                for (Entry<EReference, List<Action>> re: accumulator.entrySet()) {
127                        ((Collection<Action>) action.eGet(re.getKey())).addAll(re.getValue());
128                }
129                
130                return Collections.singletonMap(containmentReference, Collections.singletonList(action));
131        }

133        protected boolean isRootAction(Element element, Action action) {
134                if (action == null) {
135                        return false;
136                }
137                Object elementAdapter = EcoreUtil.getRegisteredAdapter(action, ElementAdapter.class);                                   
138                if (elementAdapter instanceof ElementAdapter) {
139                         Element actionElement = ((ElementAdapter) elementAdapter).getElement();
140                         if (actionElement instanceof ModelElement) {
141                                 String rootActionFlag = ((ModelElement) actionElement).getProperty(getRootActionFlagProperty());
142                                 return Util.isBlank(rootActionFlag) || !"false".equals(rootActionFlag);
143                         }
144                }
145                
146                return true;
147        }

153        protected void addURIResolverAdapters(Action root) {
154                BiFunction<Label, URI, URI> uriResolver = org.nasdanika.html.model.app.util.Util.uriResolver(root);
155                class URIResolverAdapterImpl extends AdapterImpl implements URIResolverAdapter {
156                                                
157                        @Override
158                        public boolean isAdapterForType(Object type) {
159                                return URIResolverAdapter.class == type;
160                        }
161
162                        @Override
163                        public URI resolve(Action base) {
164                                Action targetAction = (Action) getTarget();
165                                if (targetAction == base) {
166                                        return null;
167                                }                               
168                                URI baseURI = base == null ? null : uriResolver.apply(base, null);
169                                return uriResolver.apply(targetAction, baseURI);
170                        }
171
172                }
173                
174                root.eAdapters().add(new URIResolverAdapterImpl());
175                TreeIterator<EObject> cit = root.eAllContents();
176                while (cit.hasNext()) {
177                        EObject next = cit.next();
178                        if (next instanceof Action /* && EcoreUtil.getRegisteredAdapter(next, URIResolverAdapter.class) == null */) {
179                                next.eAdapters().add(new URIResolverAdapterImpl());
180                        }
181                }
182                        
183        }

229        protected boolean hasDocumentation(Resource resource, Element element) {
230                if (element instanceof Page) {
231                        return hasDocumentation(resource, ((Page) element).getModel().getRoot());
232                }
233                if (element instanceof ModelElement) {
234                        String documentationProperty = getDocumentationProperty();
235                        if (Util.isBlank(documentationProperty)) {
236                                return false;
237                        }
238                        
239                        String docProp = ((ModelElement) element).getProperty(documentationProperty);
240                        return !Util.isBlank(docProp);
241                }
242                
243                return false;
244        }

247        protected EObject getDocumentation(Resource resource, Element element, String embeddedDiagram, String tableOfContents) {
248                if (element instanceof Page) {
249                        return getDocumentation(resource, ((Page) element).getModel().getRoot(), embeddedDiagram, tableOfContents);
250                }
251                if (element instanceof ModelElement) {
252                        String documentationProperty = getDocumentationProperty();
253                        if (Util.isBlank(documentationProperty)) {
254                                return null;
255                        }
256                        
257                        String docProp = ((ModelElement) element).getProperty(documentationProperty);
258                        if (Util.isBlank(docProp)) {
259                                return null;
260                        }
261                        
262                        Markdown markdown = ContentFactory.eINSTANCE.createMarkdown();
263                        URI docURI = URI.createURI(docProp);
264                        docURI = docURI.resolve(resource.getURI());
265                        org.nasdanika.exec.content.Resource docResource = ContentFactory.eINSTANCE.createResource();
266                        docResource.setLocation(docURI.toString());
267                        markdown.setSource(docResource);                                                
268                        markdown.setStyle(true);
269
270                        Interpolator interpolator = ContentFactory.eINSTANCE.createInterpolator();                      
271                        interpolator.setSource(markdown);
272                        
273                        if (Util.isBlank(embeddedDiagram) && Util.isBlank(tableOfContents)) {
274                                return interpolator;
275                        }
276                        
277                        Configurator configurator = ExecFactory.eINSTANCE.createConfigurator();
278                        configurator.setTarget(interpolator);
279                        
280                        EMap<String, EObject> properties = configurator.getProperties();
281                        if (!Util.isBlank(embeddedDiagram)) {
282                                Text diagramText = ContentFactory.eINSTANCE.createText();
283                                diagramText.setContent(embeddedDiagram);
284                                properties.put("diagram", diagramText);
285                        }
286                        
287                        if (!Util.isBlank(tableOfContents)) {
288                                Text tocText = ContentFactory.eINSTANCE.createText();
289                                tocText.setContent(tableOfContents);
290                                properties.put("toc", tocText);
291                        }
292                        
293                        return configurator;                    
294                }
295                
296                return null;
297        }

299        protected Action createPageAction(Resource resource, Page page, Map<Element, ElementEntry<Map<EReference, List<Action>>>> childEntries) {
300                Entry<ModelElement, Map<Element, ElementEntry<Map<EReference, List<Action>>>>> pageElementEntry = getPageElementEntry(resource, page, childEntries);
301                Action ret =  pageElementEntry == null ? createAction(resource, page) : createModelElementAction(resource, pageElementEntry.getKey(), pageElementEntry.getValue(), null);               
302                ret.setText(page.getName());    
303                ret.setLocation("pages/" + page.getId() + "/index.html");
304                return ret;
305        }

388        protected Action createModelElementAction(Resource resource, ModelElement modelElement, Map<Element, ElementEntry<Map<EReference, List<Action>>>> childEntries, EReference containmentReference) {
389                if (!shallCreateAction(resource, modelElement, childEntries, containmentReference)) {
390                        return null;
391                }
392                Action ret = createAction(resource, modelElement);
393                ret.setId(modelElement.getId());
394                ret.setText(getModelElementLabel(modelElement));
395                String link = modelElement.getLink();
396                if (Util.isBlank(link) || modelElement.isPageLink()) {
397                        String path = Objects.requireNonNull(getPath(modelElement), "Path is null for " + modelElement.getLabel() + " / " + modelElement.getId());
398                        if (containmentReference == null) {
399                                ret.setLocation(path + "/index.html");
400                        } else {
401                                ret.setLocation(containmentReference.getName() + "/" + path + "/index.html");
402                        }
403                } else {
404                        ret.setLocation(link);
405                }
406                ret.setDescription(modelElement.getTooltip());
407                String icon = getIcon(modelElement);
408                if (!Util.isBlank(icon)) {
409                        ret.setIcon(icon);
410                }
411                
412                return ret;
413        }

415        /**
416         * Creates actions for documented connections.
417         * @param resource
418         * @param connection
419         * @param childEntries
420         * @param containmentReference
421         * @return
422         */
423        protected Action createConnectionAction(Resource resource, Connection connection, Map<Element, ElementEntry<Map<EReference, List<Action>>>> childEntries, EReference containmentReference) {
424                if (!shallCreateAction(resource, connection, childEntries, containmentReference)) {
425                        return null;                    
426                }
427                Action ret = createAction(resource, connection);
428                ret.setId(connection.getId());
429                String label = getModelElementLabel(connection);
430                if (Util.isBlank(label)) {
431                        Node source = connection.getSource();
432                        String sourceLabel = getModelElementLabel(source);
433                        
434                        Node target = connection.getTarget();
435                        String targetLabel = getModelElementLabel(target);
436                        
437                        if (Util.isBlank(sourceLabel) && Util.isBlank(targetLabel)) {
438                                ret.setText("(unlabeled)");                             
439                        } else {
440                                StringBuilder labelBuilder = new StringBuilder();
441                                if (!Util.isBlank(sourceLabel)) {
442                                        labelBuilder.append(sourceLabel).append(" ");
443                                }
444                                labelBuilder.append("->");
445                                if (!Util.isBlank(targetLabel)) {
446                                        labelBuilder.append(" ").append(targetLabel);
447                                }
448                                ret.setText(labelBuilder.toString());
449                        }
450                } else {
451                        ret.setText(label);
452                }
453                String link = connection.getLink();
454                if (Util.isBlank(link) || connection.isPageLink()) {
455                        String path = Objects.requireNonNull(getPath(connection), "Path is null for " + connection.getLabel() + " / " + connection.getId());
456                        if (containmentReference == null) {
457                                ret.setLocation(path + "/index.html");
458                        } else {
459                                ret.setLocation(containmentReference.getName() + "/" + path + "/index.html");
460                        }
461                } else {
462                        ret.setLocation(link);
463                }
464                ret.setDescription(connection.getTooltip());
465                String icon = getIcon(connection);
466                if (!Util.isBlank(icon)) {
467                        ret.setIcon(icon);
468                }
469                
470                return ret;                             
471        }

493        @SuppressWarnings("unchecked")
494        @Override
495        protected void resolve(
496                        Resource resource, 
497                        Element element, 
498                        Map<EReference, List<Action>> semanticElement,
499                        Map<Element, ElementEntry<Map<EReference, List<Action>>>> childEntries,
500                        Function<Predicate<Element>, Map<EReference, List<Action>>> resolver) {
501                
502                if (element instanceof Page) {                  
503                        Model model = ((Page) element).getModel();
504                        Map<Element, ElementEntry<Map<EReference, List<Action>>>> modelChildEntries = childEntries.get(model).getChildEntries();
505                        Root root = model.getRoot();
506                        Map<Element, ElementEntry<Map<EReference, List<Action>>>> rootChildEntries = modelChildEntries.get(root).getChildEntries();
507
508                        for (Action action: semanticElement.values().stream().flatMap(Collection::stream).filter(a -> isActionForElement(root, a)).collect(Collectors.toList())) {
509                                // Embedded diagram
510                                String embeddedDiagram = generateEmbeddedDiagram(element, action, resolver);
511                                if (isEmbedDiagram(resource, element)) {                                                                                                                        
512                                        addContent(action, embeddedDiagram);                                            
513                                }
514
515                                // TOC
516                                StringBuilder tocBuilder = new StringBuilder(getListOpenTag(element)).append(System.lineSeparator());
517                                for (Entry<Element, ElementEntry<Map<EReference, List<Action>>>> layerEntry: rootChildEntries.entrySet()) { 
518                                        tocBuilder.append(generateTableOfContents(action, (ModelElement) layerEntry.getKey(), layerEntry.getValue().getChildEntries(), resolver));
519                                }                                       
520                                tocBuilder.append(getListCloseTag(element)).append(System.lineSeparator());
521                                
522                                if (isGenerateTableOfContents(resource, element)) {
523                                        addContent(action, tocBuilder.toString());
524                                }
525                                
526                                EObject doc = getDocumentation(resource, element, embeddedDiagram, tocBuilder.toString());
527                                if (doc == null) {
528                                        String rootTooltip = root.getTooltip();
529                                        if (!Util.isBlank(rootTooltip)) {
530                                                addContent(action, rootTooltip);
531                                        }
532                                } else {
533                                        action.getContent().add(doc);
534                                }                                       
535                        }                       
536                } else if (element instanceof ModelElement) {
537                        for (Action action: semanticElement.values().stream().flatMap(Collection::stream).filter(a -> isActionForElement(element, a)).collect(Collectors.toList())) {
538                                if (element instanceof Connection) {
539                                        // Source and target
540                                        Table table = BootstrapFactory.eINSTANCE.createTable();
541                                        table.getAttributes().put("style", createText("width:auto"));
542                                        
543                                        TableRow sourceRow = BootstrapFactory.eINSTANCE.createTableRow();
544                                        table.getRows().add(sourceRow);
545                                        
546                                        TableCell sourceHeader = BootstrapFactory.eINSTANCE.createTableCell();
547                                        sourceHeader.setHeader(true);
548                                        sourceRow.getCells().add(sourceHeader);
549                                        sourceHeader.getContent().add(createText("Source"));
550                                                                                
551                                        TableCell sourceCell = BootstrapFactory.eINSTANCE.createTableCell();
552                                        sourceRow.getCells().add(sourceCell);
553                                        
554                                        Node source = ((Connection) element).getSource();
555                                        String sourceLabel = getModelElementLabel(source);
556                                        if (Util.isBlank(sourceLabel)) {
557                                                sourceLabel = "(unlabeled)";
558                                        } 
559                                        String sourceLink = getModelElementLink(source, action, resolver);
560                                        if (sourceLink == null) {
561                                                sourceCell.getContent().add(createText(sourceLabel));                                           
562                                        } else {
563                                                sourceCell.getContent().add(createText("<a href=\"" + sourceLink + "\">" + sourceLabel + "</a>"));                                                                                              
564                                        }
565                                        
566                                        TableRow targetRow = BootstrapFactory.eINSTANCE.createTableRow();
567                                        table.getRows().add(targetRow);
568                                        
569                                        TableCell targetHeader = BootstrapFactory.eINSTANCE.createTableCell();
570                                        targetHeader.setHeader(true);
571                                        targetRow.getCells().add(targetHeader);
572                                        targetHeader.getContent().add(createText("Target"));
573                                                                                
574                                        TableCell targetCell = BootstrapFactory.eINSTANCE.createTableCell();
575                                        targetRow.getCells().add(targetCell);
576                                        
577                                        Node target = ((Connection) element).getTarget();
578                                        String targetLabel = getModelElementLabel(target);
579                                        if (Util.isBlank(targetLabel)) {
580                                                targetLabel = "(unlabeled)";
581                                        } 
582                                        String targetLink = getModelElementLink(target, action, resolver);
583                                        if (targetLink == null) {
584                                                targetCell.getContent().add(createText(targetLabel));                                           
585                                        } else {
586                                                targetCell.getContent().add(createText("<a href=\"" + targetLink + "\">" + targetLabel + "</a>"));                                                                                              
587                                        }                                       
588
589                                        action.getContent().add(table);
590                                } else if (element instanceof Node) {
591                                        Node node = (Node) element;
592                                        EReference connectionsActionContainmentReference = getConnectionsActionContainmentReference(node);
593                                        if (connectionsActionContainmentReference != null) {
594                                                List<Connection> inboundConnections = node.getInboundConnections();
595                                                if (!inboundConnections.isEmpty()) {
596                                                        Table table = BootstrapFactory.eINSTANCE.createTable();
597                                                        table.setBordered(true);
598                                                        table.getAttributes().put("style", createText("width:auto"));
599                                                        
600                                                        TableRow headerRow = BootstrapFactory.eINSTANCE.createTableRow();
601                                                        table.getRows().add(headerRow);
602                                                                                                
603                                                        TableCell labelHeader = BootstrapFactory.eINSTANCE.createTableCell();
604                                                        labelHeader.setHeader(true);
605                                                        headerRow.getCells().add(labelHeader);
606                                                        labelHeader.getContent().add(createText("Label"));
607        
608                                                        TableCell sourceHeader = BootstrapFactory.eINSTANCE.createTableCell();
609                                                        sourceHeader.setHeader(true);
610                                                        headerRow.getCells().add(sourceHeader);
611                                                        sourceHeader.getContent().add(createText("Source"));
612        
613                                                        TableCell tooltipHeader = BootstrapFactory.eINSTANCE.createTableCell();
614                                                        tooltipHeader.setHeader(true);
615                                                        headerRow.getCells().add(tooltipHeader);
616                                                        tooltipHeader.getContent().add(createText("Tooltip"));

791        protected String generateEmbeddedDiagram(
792                        Element element, 
793                        Action action,                  
794                        Function<Predicate<Element>, Map<EReference, List<Action>>> resolver) {
795                try {
796                        Document toEmbed = Document.create(true, null);
797                        Page page = (Page) element;
798                        toEmbed.getPages().add(page);
799                        
800                        Consumer<Element> visitor = e -> {
801                                if (e instanceof ModelElement) {
802                                        String link = getModelElementLink((ModelElement) e, action, resolver);
803                                        if (!Util.isBlank(link)) {
804                                                ((ModelElement) e).setLink(link);
805                                        }
806                                }
807                        };
808                        
809                        toEmbed.accept(visitor, null);                  
810                        return toEmbed.toHtml(null, getDiagramViewer());
811                } catch (Exception  e) {
812                        return "Error embedding diagram: " + e;
813                }
814        }

824        protected String generateTableOfContents(
825                        Action pageAction, 
826                        ModelElement modelElement, 
827                        Map<Element, ElementEntry<Map<EReference, List<Action>>>> childEntries,
828                        Function<Predicate<Element>, Map<EReference, List<Action>>> resolver) {         
829                
830                String label = getModelElementLabel(modelElement);
831                StringBuilder ret = new StringBuilder();
832                if (Util.isBlank(label)) {
833                        if (modelElement instanceof Layer && childEntries != null) {
834                                for (Entry<Element, ElementEntry<Map<EReference, List<Action>>>> childEntry: childEntries.entrySet()) {
835                                        ret.append(generateTableOfContents(pageAction, (ModelElement) childEntry.getKey(), childEntry.getValue().getChildEntries(), resolver));
836                                }
837                        }
838                } else {                
839                        ret.append("<li>").append(System.lineSeparator());
840                        String link = getModelElementLink(modelElement, pageAction, resolver);
841                        if (Util.isBlank(link)) {
842                                ret.append(label);
843                        } else {
844                                ret
845                                        .append("<a href=\"")
846                                        .append(link)
847                                        .append("\">")
848                                        .append(label)
849                                        .append("</a>");                                
850                        }                       
851                        
852                        String tooltip = modelElement.getTooltip();
853                        if (!Util.isBlank(tooltip)) {
854                                ret.append(" - ").append(tooltip).append(System.lineSeparator());
855                        }
856
857                        if (modelElement instanceof Layer) {
858                                if (childEntries != null && !childEntries.isEmpty()) {
859                                        ret.append(getListOpenTag(modelElement)).append(System.lineSeparator());
860                                        for (Entry<Element, ElementEntry<Map<EReference, List<Action>>>> childEntry: childEntries.entrySet()) {
861                                                ret.append(generateTableOfContents(pageAction, (ModelElement) childEntry.getKey(), childEntry.getValue().getChildEntries(), resolver));
862                                        }
863                                        ret.append(getListCloseTag(modelElement)).append(System.lineSeparator());
864                                }
865                        }
866                        ret.append("</li>").append(System.lineSeparator());
867                }
868                
869                return ret.toString();
870        }

872        /**
873         * Resolves link to the model element from this action.
874         * @param modelElement Model element
875         * @param action Source action 
876         * @param resolver Resolver
877         * @return
878         */
879        protected String getModelElementLink(ModelElement modelElement, Action action, Function<Predicate<Element>, Map<EReference, List<Action>>> resolver) {
880                if (shallCreateLink(modelElement)) {
881                        // Link by cross-reference
882                        String crossReferenceProperty = getCrossReferenceProperty();
883                        if (!Util.isBlank(crossReferenceProperty)) {
884                                String crossReferencePropertyValue = modelElement.getProperty(crossReferenceProperty);
885                                if (!Util.isBlank(crossReferencePropertyValue)) {
886                                        int colonIdx = crossReferencePropertyValue.indexOf(":");
887                                        if (colonIdx != -1) {
888                                                Predicate<Element> predicate = e -> {
889                                                        if (e instanceof ModelElement) {
890                                                                String prop = ((ModelElement) e).getProperty(crossReferencePropertyValue.substring(0, colonIdx));
891                                                                if (!Util.isBlank(prop)) {
892                                                                        return prop.equals(crossReferencePropertyValue.substring(colonIdx + 1));
893                                                                }
894                                                        }
895                                                        return false;
896                                                };
897                                                Map<EReference, List<Action>> actionMap = resolver.apply(predicate);
898                                                for (Action eAction: actionMap.values().stream().flatMap(Collection::stream).collect(Collectors.toList())) {
899                                                        Object uriResolverAdapter = EcoreUtil.getRegisteredAdapter(eAction, URIResolverAdapter.class);
900                                                        if (uriResolverAdapter instanceof URIResolverAdapter) {
901                                                                URI eURI = ((URIResolverAdapter) uriResolverAdapter).resolve(action);
902                                                                if (eURI != null) {
903                                                                        return eURI.toString();
904                                                                }
905                                                        }
906                                                }
907                                        }
908                                }
909                        }
910                        
911                        // Link by equality
912                        Map<EReference, List<Action>> actionMap = resolver.apply(modelElement::equals);
913                        for (Action eAction: actionMap.values().stream().flatMap(Collection::stream).filter(a -> isActionForElement(modelElement, a)).collect(Collectors.toList())) {
914                                Object uriResolverAdapter = EcoreUtil.getRegisteredAdapter(eAction, URIResolverAdapter.class);
915                                if (uriResolverAdapter instanceof URIResolverAdapter) {
916                                        URI eURI = ((URIResolverAdapter) uriResolverAdapter).resolve(action);
917                                        if (eURI != null) {
918                                                return eURI.toString();
919                                        }
920                                }
921                        }
922                }
923                return null;
924        }

926        /**
927         * @param modelElement
928         * @return true if an action shall be created for a given model element. This implementation returns true for elements without external links and without cross-references.
929         * False is returned for undocumented connections.
930         */
931        protected boolean shallCreateAction(Resource resource, ModelElement modelElement, Map<Element, ElementEntry<Map<EReference, List<Action>>>> childEntries, EReference containmentReference) {
932                return (Util.isBlank(modelElement.getLink()) || modelElement.isPageLink())
933                                && (Util.isBlank(getCrossReferenceProperty()) || Util.isBlank(modelElement.getProperty(getCrossReferenceProperty())))
934                                && !(modelElement instanceof Connection && !hasDocumentation(resource, modelElement)); // No actions for undocumented connections
935        }

963        /**
964         * Creates a new action using factory. Override to, say, use a prototype action.
965         * @param resource
966         * @param element
967         * @return
968         */
969        protected Action createAction(Resource resource, Element element) {
970                if (element instanceof ModelElement) {
971                        String actionProperty = ((ModelElement) element).getProperty("action");
972                        if (!Util.isBlank(actionProperty)) {
973                                URI actionURI = URI.createURI(actionProperty);
974                                actionURI = actionURI.resolve(resource.getURI());
975                                Action prototype = EcoreUtil.copy((Action) resourceSet.getEObject(actionURI, true));
976                                prototype.setId(UUID.randomUUID().toString());
977                                prototype.eAdapters().add(createElementAdapter(element));
978                                return prototype;
979                        }
980                }
981                Action action = AppFactory.eINSTANCE.createAction();
982                action.eAdapters().add(createElementAdapter(element));          
983                return action;
984        }
