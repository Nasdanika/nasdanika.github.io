001package org.nasdanika.html.model.app.gen;
002
003import org.nasdanika.common.SupplierFactory;
004import org.nasdanika.html.jstree.JsTreeNode;
005import org.nasdanika.html.model.app.Link;
006import org.nasdanika.html.model.html.Tag;
007
008public class LinkSupplierFactoryProviderAdapter<M extends Link> extends LabelSupplierFactoryProviderAdapter<M> {
009
010        public LinkSupplierFactoryProviderAdapter(M link) {
011                super(link);
012        }
013
014        @SuppressWarnings("unchecked")
015        @Override
016        public <T> SupplierFactory<T> getFactory(Class<T> type) {
017                if (type.isAssignableFrom(Tag.class)) {
018                        return (SupplierFactory<T>) new LinkTagSupplierFactoryAdapter<>((M) getTarget());
019                }
020
021                // TODO - Card
022                
023                if (type.isAssignableFrom(JsTreeNode.class)) {
024                        return (SupplierFactory<T>) new LinkJsTreeNodeSupplierFactoryAdapter<>((M) getTarget());
025                }
026                
027                return super.getFactory(type);
028        }
029
030}
