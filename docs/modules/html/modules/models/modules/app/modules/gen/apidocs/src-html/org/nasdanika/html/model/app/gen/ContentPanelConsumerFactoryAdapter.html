054        /**
055         * Builds the content panel. Creates elements for navigation panels and sections and returns them for subsequent processing by functions which follow. 
056         * @param context
057         * @return
058         */
059        private Function<BiSupplier<HTMLElement<?>, Map<EStructuralFeature, Object>>, BiSupplier<Map<EStructuralFeature, HTMLElement<?>>, BiSupplier<List<HTMLElement<?>>, HTMLElement<?>>>> createContentPanelFunction(Context context) {
060                return new Function<BiSupplier<HTMLElement<?>,Map<EStructuralFeature, Object>>, BiSupplier<Map<EStructuralFeature, HTMLElement<?>>, BiSupplier<List<HTMLElement<?>>, HTMLElement<?>>>>() {
061                        
062                        @Override
063                        public double size() {
064                                return 1;
065                        }
066                        
067                        @Override
068                        public String name() {
069                                return "Content panel";
070                        }
071                        
072                        @Override
073                        @SuppressWarnings("unchecked")
074                        public BiSupplier<Map<EStructuralFeature, HTMLElement<?>>, BiSupplier<List<HTMLElement<?>>, HTMLElement<?>>> execute(BiSupplier<HTMLElement<?>, Map<EStructuralFeature, Object>> input, ProgressMonitor progressMonitor) throws Exception {
075                                HTMLElement<?> ret = (HTMLElement<?>) input.getFirst();
076                                
077                                BootstrapFactory bootstrapFactory = context.get(BootstrapFactory.class, BootstrapFactory.INSTANCE);
078                                Map<EStructuralFeature, HTMLElement<?>> navigationPanels = new LinkedHashMap<>();
079                                
080                                // Floating navs - left and right
081                                ContentPanel semanticElement = getTarget();
082                                if (semanticElement.getFloatLeftNavigation() != null) {
083                                        Tag floatLeftNavigation = bootstrapFactory.getHTMLFactory().div();
084                                        floatLeftNavigation.addClass("nsd-app-content-panel-float-left-navigation");
085                                        ((java.util.function.Consumer<Object>) ret).accept(floatLeftNavigation);
086                                        navigationPanels.put(AppPackage.Literals.CONTENT_PANEL__FLOAT_LEFT_NAVIGATION, floatLeftNavigation);
087                                }
088                                if (semanticElement.getFloatRightNavigation() != null) {
089                                        Tag floatRightNavigation = bootstrapFactory.getHTMLFactory().div();
090                                        floatRightNavigation.addClass("nsd-app-content-panel-float-right-navigation");
091                                        ((java.util.function.Consumer<Object>) ret).accept(floatRightNavigation);
092                                        navigationPanels.put(AppPackage.Literals.CONTENT_PANEL__FLOAT_RIGHT_NAVIGATION, floatRightNavigation);
093                                }
094                                                                
095                                Container container = bootstrapFactory.fluidContainer();
096                                ((java.util.function.Consumer<Object>) ret).accept(container.toHTMLElement());
097                                
098                                List<Object> breadcrumb = (List<Object>) input.getSecond().get(AppPackage.Literals.CONTENT_PANEL__BREADCRUMB);
099                                if (breadcrumb != null) {
100                                        Breadcrumb bc = bootstrapFactory.breadcrumb();
101                                        for (Object bce: breadcrumb) {
102                                                boolean active = false;
103                                                if (bce instanceof Tag) {
104                                                        Object bcd = ((Tag) bce).getData();
105                                                        if (bcd instanceof Item) {
106                                                                active = ((Item) bcd).isActive();
107                                                        }
108                                                }
109                                                bc.item(active, bce);
110                                        }
111                                        bc.toHTMLElement().addClass("nsd-app-content-panel-breadcrumb");
112                                        Row bcRow = container.row();
113                                        bcRow.toHTMLElement().addClass("nsd-app-content-panel-breadcrumb-row");
114                                        bcRow.col(bc);
115                                }               
116                                
117                                SectionStyle effectiveSemanticContainerSectionStyle = SectionStyle.HEADER;
118                                EObject semanticContainer = semanticElement.eContainer();
119                                if (semanticContainer instanceof ContentPanel && semanticElement.eContainmentFeature() == AppPackage.Literals.CONTENT_PANEL__SECTIONS) {
120                                        effectiveSemanticContainerSectionStyle = effectiveSectionStyle((ContentPanel) semanticContainer);
121                                }
122                                
123                                Tag title = (Tag) input.getSecond().get(AppPackage.Literals.CONTENT_PANEL__TITLE);
124                                List<Object> items = (List<Object>) input.getSecond().get(AppPackage.Literals.PAGE_PART__ITEMS);
125                                if (title != null || items != null) {
126                                        Row titleAndItemsRow = container.row();
127                                        titleAndItemsRow.toHTMLElement().addClass("nsd-app-content-panel-title-and-items-row");
128                                        boolean withTitle = effectiveSemanticContainerSectionStyle == SectionStyle.HEADER && title != null;
129                                        if (withTitle) {
130                                                title.addClass("nsd-app-content-panel-title");
131                                                Tag titleHeader = bootstrapFactory.getHTMLFactory().tag("H" + Math.min(headerDepth(semanticElement) + 1, 6), title);
132                                                titleAndItemsRow.col(titleHeader).width(Breakpoint.DEFAULT, Size.AUTO);
133                                        }
134        
135                                        if (items != null) {
136                //                              ContentPanel semanticElement = getTarget();
137                                                Tag navs = Util.navs(items, context.get(BootstrapFactory.class, BootstrapFactory.INSTANCE));
138                                                if (withTitle) {
139                                                        navs.addClass("nsd-app-content-panel-navs");
140                                                }
141                                                titleAndItemsRow.col(navs);
142                                        }
143                                }
144                                
145                                // Passing the title to title consumer stored in data
146                                if (effectiveSemanticContainerSectionStyle != SectionStyle.HEADER) {
147                                        BiConsumer<ContentPanel, Tag> titleConsumer = (BiConsumer<ContentPanel, Tag>) ret.getData(TITLE_CONSUMER_KEY);
148                                        if (titleConsumer != null) {
149                                                titleConsumer.accept(semanticElement, title);
150                                        }
151                                }
152                                
153                                Row contentRow = container.row();                               
154                                contentRow.toHTMLElement().addClass("nsd-app-content-panel-content-row");
155                                
156                                // Content nav left
157                                if (semanticElement.getLeftNavigation() != null) {
158                                        Tag leftNavigationColumn = contentRow.col().toHTMLElement();
159                                        leftNavigationColumn.addClass("nsd-app-content-panel-left-navigation");
160                                        navigationPanels.put(AppPackage.Literals.CONTENT_PANEL__LEFT_NAVIGATION, leftNavigationColumn);
161                                }
162                                
163                                Tag contentCol = contentRow.col().toHTMLElement();
164                                Container contentFloatsAndSectionsContainer = bootstrapFactory.fluidContainer();
165                                contentCol.accept(contentFloatsAndSectionsContainer);
166                                Tag ownContentCol = contentFloatsAndSectionsContainer.row().col().toHTMLElement();                              
167                                
168                                List<Object> content = (List<Object>) input.getSecond().get(HtmlPackage.Literals.HTML_ELEMENT__CONTENT);
169                                if (content != null) {
170                                        content.forEach(ownContentCol::accept);
171                                }
172                                
173                                
174                                List<HTMLElement<?>> sections;                          
175                                if (semanticElement.getSections().isEmpty()) {
176                                        sections = Collections.emptyList();
177                                } else {
178                                        // Creating containers for sections
179                                        SectionStyle effectivSectionStyle = effectiveSectionStyle(semanticElement);
180                                        switch (effectivSectionStyle) {
181                                        case ACTION_GROUP:
182                                                sections = actionGroupSections(contentFloatsAndSectionsContainer, bootstrapFactory);                            
183                                                break;
184                                        case CARD:
185                                                sections = cardSections(contentFloatsAndSectionsContainer, bootstrapFactory);                           
186                                                break;
187                                        case CARD_PILL:
188                                                sections = cardPillSections(contentFloatsAndSectionsContainer, bootstrapFactory);                               
189                                                break;
190                                        case CARD_TAB:
191                                                sections = cardTabSections(contentFloatsAndSectionsContainer, bootstrapFactory);                                
192                                                break;
193                                        case HEADER:
194                                                // Cols in rows
195                                                sections = headerSections(contentFloatsAndSectionsContainer);                           
196                                                break;
197                                        case PILL:
198                                                sections = tabSections(contentFloatsAndSectionsContainer, true, bootstrapFactory);                              
199                                                break;
200                                        case TAB:
201                                                sections = tabSections(contentFloatsAndSectionsContainer, false, bootstrapFactory);                             
202                                                break;
203                                        case TABLE:
204                                                sections = tableSections(contentFloatsAndSectionsContainer, bootstrapFactory);                          
205                                                break;
206                                        default:
207                                                throw new UnsupportedOperationException("Unsupported section style: " + effectivSectionStyle);
208                                        }
209                                }
210                                
211                                if (semanticElement.getRightNavigation() != null) {
212                                        Tag rightNavigationColumn = contentRow.col().toHTMLElement();
213                                        rightNavigationColumn.addClass("nsd-app-content-panel-right-navigation");
214                                        navigationPanels.put(AppPackage.Literals.CONTENT_PANEL__RIGHT_NAVIGATION, rightNavigationColumn);
215                                }
216                                
217                                return new BiSupplier<Map<EStructuralFeature, HTMLElement<?>>, BiSupplier<List<HTMLElement<?>>, HTMLElement<?>>>() {
218
219                                        // Navigation panels map
220                                        @Override
221                                        public Map<EStructuralFeature, HTMLElement<?>> getFirst() {
222                                                return navigationPanels;
223                                        }
224
225                                        // Sections and content panel
226                                        @Override
227                                        public BiSupplier<List<HTMLElement<?>>, HTMLElement<?>> getSecond() {
228                                                return new BiSupplier<List<HTMLElement<?>>, HTMLElement<?>>() {
229
230                                                        @Override
231                                                        public List<HTMLElement<?>> getFirst() {
232                                                                return sections;
233                                                        }
234
235                                                        @Override
236                                                        public HTMLElement<?> getSecond() {
237                                                                return ret;
238                                                        }
239                                                        
240                                                };
241                                        }
242                                        
243                                };
244                        }
245                };
246        }

248        private List<HTMLElement<?>> tabSections(Container contentFloatsAndSectionsContainer, boolean pills, BootstrapFactory bootstrapFactory) {
249                Row sectionRow = contentFloatsAndSectionsContainer.row();
250                sectionRow.toHTMLElement().addClass("nsd-app-content-panel-section-row");
251                Tag tabsCol = sectionRow.col().toHTMLElement();
252                HTMLFactory htmlFactory = bootstrapFactory.getHTMLFactory();
253                Tag navList = htmlFactory.tag(TagName.ul).addClass("nav").addClass(pills ? "nav-pills" : "nav-tabs");
254                tabsCol.accept(navList);
255                Tag contentDiv = htmlFactory.div().id(htmlFactory.nextId()).addClass("tab-content");
256                tabsCol.accept(contentDiv);
257                List<HTMLElement<?>> sections = new ArrayList<>();
258                
259                for (ContentPanel section: getTarget().getSections()) {
260                        Tag sectionContentContainer = htmlFactory.div().addClass("tab-pane", "fade").id(htmlFactory.nextId());
261                        Label sTitle = section.getTitle();
262                        boolean isActive = sTitle != null && sTitle.isActive();
263                        if (isActive) {
264                                sectionContentContainer.addClass("show", "active");
265                        }
266                        contentDiv.accept(sectionContentContainer);
267                        sections.add(sectionContentContainer);
268                        BiConsumer<ContentPanel, Tag> titleConsumer = (sectionSemanticElement, title) -> {
269                                navList.accept(htmlFactory.tag(TagName.li, title).addClass("nav-item"));
270                                Object sectionContentContainerId = sectionContentContainer.getId();
271                                title
272                                        .addClass("nav-link")
273                                        .addClassConditional(isActive, "active")
274                                        .id(sectionContentContainerId + "-tab")
275                                        .attribute("data-toggle", "tab")
276                                        .attribute("href", "#" + sectionContentContainerId)
277                                        .attribute("role", "tab")
278                                        .attribute("aria-controls", sectionContentContainerId)
279                                        .attribute("aria-selected", String.valueOf(isActive));
280                        };
281                        sectionContentContainer.setData(TITLE_CONSUMER_KEY, titleConsumer);
282                }
283                return sections;
284        }

286        private List<HTMLElement<?>> actionGroupSections(Container contentFloatsAndSectionsContainer, BootstrapFactory bootstrapFactory) {
287                Row sectionRow = contentFloatsAndSectionsContainer.row();
288                sectionRow.toHTMLElement().addClass("nsd-app-content-panel-section-row", "nsd-app-content-panel-section-action-group");
289                HTMLFactory htmlFactory = bootstrapFactory.getHTMLFactory();
290                Tag actionList = htmlFactory.div().addClass("list-group", "list-group-flush").attribute("role",  "tablist");
291                sectionRow.col().width(Breakpoint.DEFAULT, Size.AUTO).toHTMLElement().accept(actionList);
292                Tag contentDiv = htmlFactory.div().addClass("tab-content");
293                sectionRow.col().toHTMLElement().accept(contentDiv);
294                List<HTMLElement<?>> sections = new ArrayList<>();
295                
296                boolean active = true;
297                for (ContentPanel section: getTarget().getSections()) {
298                        Tag sectionContentContainer = htmlFactory.div().addClass("tab-pane").id(htmlFactory.nextId());
299                        if (active) {
300                                sectionContentContainer.addClass("active");
301                        }
302                        boolean isActive = active;
303                        contentDiv.accept(sectionContentContainer);
304                        sections.add(sectionContentContainer);
305                        BiConsumer<ContentPanel, Tag> titleConsumer = (sectionSemanticElement, title) -> {
306                                actionList.accept(title);
307                                Object sectionContentContainerId = sectionContentContainer.getId();
308                                title
309                                        .addClass("list-group-item", "list-group-item-action")
310                                        .addClassConditional(isActive, "active")
311                                        .id(sectionContentContainerId + "-tab")
312                                        .attribute("data-toggle", "list")
313                                        .attribute("href", "#" + sectionContentContainerId)
314                                        .attribute("role", "tab");
315                        };
316                        sectionContentContainer.setData(TITLE_CONSUMER_KEY, titleConsumer);
317                        sectionContentContainer.setData(section);
318                        active = false;
319                }
320                return sections;
321        }

323        private List<HTMLElement<?>> headerSections(Container contentFloatsAndSectionsContainer) {
324                List<HTMLElement<?>> sections = new ArrayList<>();
325                for (ContentPanel section: getTarget().getSections()) {
326                        Tag sectionCol = contentFloatsAndSectionsContainer.row().col().toHTMLElement();
327                        sectionCol.setData(section);
328                        sections.add(sectionCol);
329                }
330                return sections;
331        }

333        private List<HTMLElement<?>> tableSections(Container contentFloatsAndSectionsContainer, BootstrapFactory bootstrapFactory) {
334                Row sectionRow = contentFloatsAndSectionsContainer.row();
335                sectionRow.toHTMLElement().addClass("nsd-app-content-panel-section-row", "nsd-app-content-panel-section-table");
336                Table table = bootstrapFactory.table().bordered();
337                sectionRow.col(table);
338                List<HTMLElement<?>> sections = new ArrayList<>();
339
340                TableBody body = table.body();
341                for (ContentPanel section: getTarget().getSections()) {
342                        org.nasdanika.html.bootstrap.RowContainer.Row row = body.row();
343                        Cell titleCell = row.cell().toHTMLElement();
344                        Cell contentCell = row.cell().toHTMLElement();
345                        contentCell.setData(section);
346                        sections.add(contentCell);
347                        BiConsumer<ContentPanel, Tag> titleConsumer = (sectionSemanticElement, title) -> {
348                                titleCell.accept(title);
349                        };
350                        contentCell.setData(TITLE_CONSUMER_KEY, titleConsumer);
351                }
352                return sections;
353        }

367        private static SectionStyle effectiveSectionStyle(ContentPanel semanticElement) {
368                SectionStyle style = semanticElement.getSectionStyle();
369                if (style == null) {
370                        style = SectionStyle.AUTO;
371                }
372                
373                if (style == SectionStyle.AUTO) {
374                        switch (depth(semanticElement)) {
375                        case 0:
376                                return SectionStyle.TAB;
377                        case 1:
378                                return SectionStyle.ACTION_GROUP;
379                        default:
380                                return SectionStyle.HEADER;
381                        }
382                }
383                
384                return style;
385        }

387        private static int depth(ContentPanel semanticElement) {
388                Object semanticContainer = semanticElement.eContainer();
389                if (semanticContainer instanceof ContentPanel && semanticElement.eContainmentFeature() == AppPackage.Literals.CONTENT_PANEL__SECTIONS) {
390                        return depth((ContentPanel) semanticContainer) + 1;
391                }
392                return 0;
393        }
394        
395        private static int headerDepth(ContentPanel semanticElement) {
396                Object semanticContainer = semanticElement.eContainer();
397                if (semanticContainer instanceof ContentPanel && semanticElement.eContainmentFeature() == AppPackage.Literals.CONTENT_PANEL__SECTIONS) {
398                        ContentPanel containerPanel = (ContentPanel) semanticContainer;
399                        int containerHeaderDepth = headerDepth(containerPanel);
400                        return effectiveSectionStyle(containerPanel) == SectionStyle.HEADER ? containerHeaderDepth + 1 : containerHeaderDepth;
401                }
402                return 0;
403        }

410        /**
411         * Adapts items to suppliers, builds {@link Navs} from items and adds them to the header.
412         */
413        @Override
414        protected Function<HTMLElement<?>, HTMLElement<?>> createConfigureFunction(Context context) throws Exception {
415                ContentPanel semanticElement = getTarget();
416                MapCompoundSupplierFactory<EStructuralFeature, Object> featuresSupplierFactory = new MapCompoundSupplierFactory<>("Features");                          
417                
418                EList<EObject> content = semanticElement.getContent();
419                if (!content.isEmpty()) {
420                        featuresSupplierFactory.put(HtmlPackage.Literals.HTML_ELEMENT__CONTENT, new ListCompoundSupplierFactory<>("Content", EObjectAdaptable.adaptToSupplierFactoryNonNull(content, Object.class)));
421                }
422                
423                EList<EObject> items = semanticElement.getItems();
424                if (!items.isEmpty()) {
425                        featuresSupplierFactory.put(AppPackage.Literals.PAGE_PART__ITEMS, new ListCompoundSupplierFactory<>("Navigation items", EObjectAdaptable.adaptToSupplierFactoryNonNull(items, Object.class)));
426                }
427                
428                Label title = semanticElement.getTitle();
429                if (title != null) {
430                        featuresSupplierFactory.put(AppPackage.Literals.CONTENT_PANEL__TITLE, EObjectAdaptable.adaptToSupplierFactoryNonNull(title, Object.class));
431                }
432                
433                EList<Label> breadcrumb = semanticElement.getBreadcrumb();
434                if (!breadcrumb.isEmpty()) {
435                        featuresSupplierFactory.put(AppPackage.Literals.CONTENT_PANEL__BREADCRUMB, new ListCompoundSupplierFactory<>("Breadcrumb", EObjectAdaptable.adaptToSupplierFactoryNonNull(breadcrumb, Object.class)));
436                }
437                                
438                Function<HTMLElement<?>, BiSupplier<HTMLElement<?>, Map<EStructuralFeature, Object>>> featureSuppliersFunction = featuresSupplierFactory.<HTMLElement<?>>asFunctionFactory().create(context);                           
439                Function<HTMLElement<?>, BiSupplier<Map<EStructuralFeature, HTMLElement<?>>, BiSupplier<List<HTMLElement<?>>, HTMLElement<?>>>> contentPanelFunction = super.createConfigureFunction(context).then(featureSuppliersFunction).then(createContentPanelFunction(context));
440                
441                // Navigation panels and sections
442                MapCompoundConsumerFactory<EStructuralFeature, HTMLElement<?>> navigationPanelsFactory = new MapCompoundConsumerFactory<>("Navigation panels");
443                
444                NavigationPanel floatLeftNavigation = semanticElement.getFloatLeftNavigation();
445                if (floatLeftNavigation != null) {
446                        navigationPanelsFactory.put(AppPackage.Literals.CONTENT_PANEL__FLOAT_LEFT_NAVIGATION, EObjectAdaptable.adaptToConsumerFactoryNonNull(floatLeftNavigation, HTMLElement.class));                  
447                }
448                NavigationPanel leftNavigation = semanticElement.getLeftNavigation();
449                if (leftNavigation != null) {
450                        navigationPanelsFactory.put(AppPackage.Literals.CONTENT_PANEL__LEFT_NAVIGATION, EObjectAdaptable.adaptToConsumerFactoryNonNull(leftNavigation, HTMLElement.class));                     
451                }
452                NavigationPanel floatRightNavigation = semanticElement.getFloatRightNavigation();
453                if (floatRightNavigation != null) {
454                        navigationPanelsFactory.put(AppPackage.Literals.CONTENT_PANEL__FLOAT_RIGHT_NAVIGATION, EObjectAdaptable.adaptToConsumerFactoryNonNull(floatRightNavigation, HTMLElement.class));                        
455                }
456                NavigationPanel rightNavigation = semanticElement.getRightNavigation();
457                if (rightNavigation != null) {
458                        navigationPanelsFactory.put(AppPackage.Literals.CONTENT_PANEL__RIGHT_NAVIGATION, EObjectAdaptable.adaptToConsumerFactoryNonNull(rightNavigation, HTMLElement.class));                   
459                }
460                
461                FunctionFactory<BiSupplier<Map<EStructuralFeature, HTMLElement<?>>, BiSupplier<List<HTMLElement<?>>, HTMLElement<?>>>, BiSupplier<List<HTMLElement<?>>, HTMLElement<?>>> navigationPanelsFunctionFactory = navigationPanelsFactory.asBiSupplierFunctionFactory();
462                Function<HTMLElement<?>, BiSupplier<List<HTMLElement<?>>, HTMLElement<?>>> contentPanelWithNavigationsFunction = contentPanelFunction.then(navigationPanelsFunctionFactory.create(context));
463
464                // Sections
465                ListCompoundConsumerFactory<HTMLElement<?>> sectionsFactory = new ListCompoundConsumerFactory<>("Sections", EObjectAdaptable.adaptToConsumerFactoryNonNull(semanticElement.getSections(), HTMLElement.class));
466                Function<BiSupplier<List<HTMLElement<?>>, HTMLElement<?>>, HTMLElement<?>> sectionsFunction = sectionsFactory.<HTMLElement<?>>asBiSupplierFunctionFactory().create(context);
467                Function<HTMLElement<?>, HTMLElement<?>> ret = contentPanelWithNavigationsFunction.then(sectionsFunction);
468                
469                return ret;
470        }
