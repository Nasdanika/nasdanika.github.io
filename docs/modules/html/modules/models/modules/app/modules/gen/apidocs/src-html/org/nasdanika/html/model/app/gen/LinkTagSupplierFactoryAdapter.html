001package org.nasdanika.html.model.app.gen;
002
003import java.util.List;
004
005import org.nasdanika.common.Context;
006import org.nasdanika.common.ProgressMonitor;
007import org.nasdanika.common.SupplierFactory;
008import org.nasdanika.common.Util;
009import org.nasdanika.emf.EObjectAdaptable;
010import org.nasdanika.html.Event;
011import org.nasdanika.html.HTMLFactory;
012import org.nasdanika.html.Tag;
013import org.nasdanika.html.TagName;
014import org.nasdanika.html.model.app.Action;
015import org.nasdanika.html.model.app.Link;
016
017public class LinkTagSupplierFactoryAdapter<M extends Link> extends LabelTagSupplierFactoryAdapter<M> {
018        
019        public LinkTagSupplierFactoryAdapter(M link) {
020                super(link);
021                if (link instanceof Action) {
022                        throw new IllegalArgumentException("Actions must be converted to links first");
023                }
024        }
025        
026        @Override
027        protected Tag createTextAndIconTag(Tag icon, String text, Tag modal, Context context, ProgressMonitor progressMonitor) {
028                M semanticElement = getTarget();
029                HTMLFactory htmlFactory = context.get(HTMLFactory.class, HTMLFactory.INSTANCE);
030
031                Tag anchor = htmlFactory.tag(TagName.a);
032                if (icon != null) {
033                        icon.addClass("nsd-app-label-icon");
034                        anchor.accept(icon);
035                }
036                if (!Util.isBlank(text)) {
037                        anchor.accept(text);
038                }
039                
040                if (!semanticElement.isDisabled()) {
041                        if (modal != null) {
042                                if (modal.getId() == null) {
043                                        modal.id(htmlFactory.nextId());
044                                }
045                                anchor.attribute("data-toggle", "modal");
046                                anchor.attribute("data-target", "#" + modal.getId());
047                                anchor.attribute("href", "#");
048        
049                                @SuppressWarnings("unchecked")
050                                List<Object> pageBody = context.get(org.nasdanika.html.model.html.gen.PageSupplierFactoryAdapter.PAGE_BODY_PROPERTY, List.class);
051                                pageBody.add(modal);
052                        } else {        
053                                String name = context.interpolateToString(semanticElement.getName());
054                                if (!Util.isBlank(name)) {
055                                        anchor.attribute("name", name);
056                                } else {                                
057                                        String location = context.interpolateToString(semanticElement.getLocation());
058                                        String confirmation = context.interpolateToString(semanticElement.getConfirmation());
059                                        if (!Util.isBlank(location)) {
060                                                anchor.attribute("href", location);                                     
061                                                if (!Util.isBlank(confirmation)) {
062                                                        anchor.on(Event.click, "return confirm('" + confirmation + "');");
063                                                }
064                                                String target = context.interpolateToString(semanticElement.getTarget());
065                                                if (!Util.isBlank(target)) {
066                                                        anchor.attribute("target", target);
067                                                }
068                                        } else { 
069                                                String script = context.interpolateToString(semanticElement.getScript());
070                                                if (!Util.isBlank(script)) {
071                                                        if (!Util.isBlank(confirmation)) {
072                                                                script = "if (confirm('" + confirmation + "')) { "+ script +" } return false;";
073                                                        }
074                                                        anchor.on(Event.click, script);
075                                                        anchor.style("cursor", "pointer");
076                                                        anchor.attribute("href", "#");
077                                                } 
078                                        }
079                                }
080                        }
081                }
082                
083                return anchor;
084        }
085
086        @Override
087        protected SupplierFactory<Tag> getModalFactory() {
088                M semanticElement = getTarget();
089                org.nasdanika.html.model.bootstrap.Modal modal = semanticElement.getModal();
090                SupplierFactory<Tag> modalFactory = modal == null ? SupplierFactory.empty() : EObjectAdaptable.adaptToSupplierFactoryNonNull(modal, Tag.class);
091                return modalFactory;
092        }
093        
094}
