022public class AppAdapterFactory extends BootstrapAdapterFactory {
023        
024        public static AppAdapterFactory INSTANCE = new AppAdapterFactory();
025        
026        @SuppressWarnings({ "unchecked", "rawtypes" })
027        public AppAdapterFactory() {
028                ClassLoader classLoader = getClassLoader();
029                
030                registerAdapterFactory(
031                                new FunctionAdapterFactory<SupplierFactory<HTMLElement<?>>, Page>(
032                                        AppPackage.Literals.PAGE, 
033                                        (Class) SupplierFactory.class, 
034                                        classLoader, 
035                                        e -> new PageSupplierFactoryAdapter(e, this)));
036                
037                registerAdapterFactory(
038                                new FunctionAdapterFactory<SupplierFactory.Provider, Label>(
039                                        AppPackage.Literals.LABEL, 
040                                        SupplierFactory.Provider.class, 
041                                        classLoader, 
042                                        e -> new LabelSupplierFactoryProviderAdapter(e, this)));
043                
044                registerAdapterFactory(
045                                new FunctionAdapterFactory<SupplierFactory.Provider, Link>(
046                                        AppPackage.Literals.LINK, 
047                                        SupplierFactory.Provider.class, 
048                                        classLoader, 
049                                        e -> new LinkSupplierFactoryProviderAdapter(e, this)));
050                
051                registerAdapterFactory(
052                                new FunctionAdapterFactory<ConsumerFactory<HTMLElement<?>>, ContentPanel>(
053                                        AppPackage.Literals.CONTENT_PANEL, 
054                                        (Class) ConsumerFactory.class, 
055                                        classLoader, 
056                                        e -> new ContentPanelConsumerFactoryAdapter(e, this)));         
057                
058                registerAdapterFactory(
059                                new FunctionAdapterFactory<ConsumerFactory<HTMLElement<?>>, Footer>(
060                                        AppPackage.Literals.FOOTER, 
061                                        (Class) ConsumerFactory.class, 
062                                        classLoader, 
063                                        e -> new FooterConsumerFactoryAdapter(e, this)));               
064                
065                registerAdapterFactory(
066                                new FunctionAdapterFactory<ConsumerFactory<HTMLElement<?>>, Header>(
067                                        AppPackage.Literals.HEADER, 
068                                        (Class) ConsumerFactory.class, 
069                                        classLoader, 
070                                        e -> new HeaderConsumerFactoryAdapter(e, this)));               
071                
072                registerAdapterFactory(
073                                new FunctionAdapterFactory<ConsumerFactory<HTMLElement<?>>, NavigationBar>(
074                                        AppPackage.Literals.NAVIGATION_BAR, 
075                                        (Class) ConsumerFactory.class, 
076                                        classLoader, 
077                                        e -> new NavigationBarConsumerFactoryAdapter(e, this)));                
078                
079                registerAdapterFactory(
080                                new FunctionAdapterFactory<ConsumerFactory<HTMLElement<?>>, NavigationPanel>(
081                                        AppPackage.Literals.NAVIGATION_PANEL, 
082                                        (Class) ConsumerFactory.class, 
083                                        classLoader, 
084                                        e -> new NavigationPanelConsumerFactoryAdapter(e, this)));      
085                
086                // Lower-level factories - through inheritance
087//              registerAdapterFactory(new ExecutionParticpantAdapterFactory());
