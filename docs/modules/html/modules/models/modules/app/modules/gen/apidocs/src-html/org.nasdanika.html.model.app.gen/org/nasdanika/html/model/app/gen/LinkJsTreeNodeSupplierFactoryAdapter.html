001package org.nasdanika.html.model.app.gen;
002
003import java.util.List;
004
005import org.nasdanika.common.Consumer;
006import org.nasdanika.common.Context;
007import org.nasdanika.common.ProgressMonitor;
008import org.nasdanika.common.Supplier;
009import org.nasdanika.common.SupplierFactory;
010import org.nasdanika.common.Util;
011import org.nasdanika.html.HTMLFactory;
012import org.nasdanika.html.Tag;
013import org.nasdanika.html.jstree.JsTreeNode;
014import org.nasdanika.html.model.app.Action;
015import org.nasdanika.html.model.app.AppPackage;
016import org.nasdanika.html.model.app.Link;
017
018public class LinkJsTreeNodeSupplierFactoryAdapter<M extends Link> extends LabelJsTreeNodeSupplierFactoryAdapter<M> {
019        
020        public LinkJsTreeNodeSupplierFactoryAdapter(M link) {
021                super(link);
022                if (link instanceof Action) {
023                        throw new IllegalArgumentException("Actions must be converted to links first");
024                }
025        }
026        
027        /**
028         * For links. Placing here to simplify code. This implementation returns empty supplier.
029         * @return
030         */
031        protected SupplierFactory<Tag> getModalFactory() {
032                return SupplierFactory.empty();
033        }
034        
035        @Override
036        public Supplier<JsTreeNode> create(Context context) throws Exception {
037                @SuppressWarnings("resource")
038                Consumer<JsTreeNode> configurator = new Consumer<JsTreeNode>() {
039
040                        @Override
041                        public double size() {
042                                return 1;
043                        }
044
045                        @Override
046                        public String name() {
047                                return "Binding jsTree node";
048                        }
049
050                        @Override
051                        public void execute(JsTreeNode jsTreeNode, ProgressMonitor progressMonitor) throws Exception {
052                                M semanticElement = getTarget();
053                                jsTreeNode.disabled(semanticElement.isDisabled());
054                                jsTreeNode.id(semanticElement.getId());
055                                
056                                if (!semanticElement.isDisabled()) {
057                                        Tag modal = (Tag) jsTreeNode.getData(AppPackage.Literals.LINK__MODAL);
058                                        if (modal != null) {
059                                                if (modal.getId() == null) {
060                                                        modal.id(context.get(HTMLFactory.class, HTMLFactory.INSTANCE).nextId());
061                                                }
062                                                jsTreeNode.anchorAttribute("data-toggle", "modal");
063                                                jsTreeNode.anchorAttribute("data-target", "#" + modal.getId());
064                        
065                                                @SuppressWarnings("unchecked")
066                                                List<Object> pageBody = context.get(org.nasdanika.html.model.html.gen.PageSupplierFactoryAdapter.PAGE_BODY_PROPERTY, List.class);
067                                                pageBody.add(modal);
068                                        } else {                                                                        
069                                                String location = context.interpolateToString(semanticElement.getLocation());
070                                                String confirmation = context.interpolateToString(semanticElement.getConfirmation());
071                                                if (!Util.isBlank(location)) {
072                                                        String activatorScript = "window.location='" + location + "';";
073                                                        if (Util.isBlank(confirmation)) {
074                                                                jsTreeNode.anchorAttribute("onclick", activatorScript);
075                                                        } else {
076                                                                jsTreeNode.anchorAttribute("onclick", "if (confirm('" + confirmation + "')) { " + activatorScript + " } return false;");                                                                
077                                                        }
078                                                } else { 
079                                                        String script = context.interpolateToString(semanticElement.getScript());
080                                                        if (!Util.isBlank(script)) {
081                                                                if (!Util.isBlank(confirmation)) {
082                                                                        script = "if (confirm('" + confirmation + "')) { "+ script +" } return false;";
083                                                                }
084                                                                jsTreeNode.anchorAttribute("onclick", script);
085//                                                              anchor.style("cursor", "pointer");
086                                                        }
087                                                }
088                                        }
089                                }
090                        }
091                        
092                };
093                
094                return super.create(context).then(configurator.asFunction());
095        }       
096                
097}
