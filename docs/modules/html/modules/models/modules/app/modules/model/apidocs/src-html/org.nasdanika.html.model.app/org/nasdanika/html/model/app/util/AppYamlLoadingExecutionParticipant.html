001package org.nasdanika.html.model.app.util;
002
003import java.util.Collection;
004
005import org.eclipse.emf.ecore.EPackage;
006import org.nasdanika.common.Context;
007import org.nasdanika.html.model.app.AppPackage;
008import org.nasdanika.html.model.bootstrap.util.BootstrapYamlLoadingExecutionParticipant;
009
010/**
011 * {@link YamlLoadingSupplier} for Bootstrap.
012 * Registers exec- loader. 
013 * @author Pavel
014 *
015 */
016public abstract class AppYamlLoadingExecutionParticipant extends BootstrapYamlLoadingExecutionParticipant {
017
018        public AppYamlLoadingExecutionParticipant(Context context) {
019                super(context);
020        }
021
022        @Override
023        protected Collection<EPackage> getEPackages() {
024                Collection<EPackage> ret = super.getEPackages(); 
025                ret.add(AppPackage.eINSTANCE);
026                return ret;
027        }
028        
029}
