001package org.nasdanika.html.model.app.gen;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import org.eclipse.emf.common.notify.AdapterFactory;
007import org.eclipse.emf.common.util.EList;
008import org.eclipse.emf.ecore.EObject;
009import org.nasdanika.common.BiSupplier;
010import org.nasdanika.common.Context;
011import org.nasdanika.common.Function;
012import org.nasdanika.common.ListCompoundSupplierFactory;
013import org.nasdanika.common.ProgressMonitor;
014import org.nasdanika.emf.EObjectAdaptable;
015import org.nasdanika.html.HTMLElement;
016import org.nasdanika.html.Tag;
017import org.nasdanika.html.bootstrap.BootstrapFactory;
018import org.nasdanika.html.bootstrap.Navs;
019import org.nasdanika.html.model.app.Footer;
020
021public class FooterConsumerFactoryAdapter extends PagePartConsumerFactoryAdapter<Footer> {
022
023        protected FooterConsumerFactoryAdapter(Footer footer, AdapterFactory adapterFactory) {
024                super(footer, adapterFactory);
025        }
026        
027        private Function<BiSupplier<HTMLElement<?>, List<Object>>, HTMLElement<?>> createNavsFunction(Context context) {
028                return new Function<BiSupplier<HTMLElement<?>,List<Object>>, HTMLElement<?>>() {
029                        
030                        @Override
031                        public double size() {
032                                return 1;
033                        }
034                        
035                        @Override
036                        public String name() {
037                                return "Navs";
038                        }
039                        
040                        @Override
041                        public HTMLElement<?> execute(BiSupplier<HTMLElement<?>, List<Object>> input, ProgressMonitor progressMonitor) throws Exception {
042                                Tag ret = (Tag) input.getFirst();                               
043                                List<Object> items = new ArrayList<>();
044                                BootstrapFactory bootstrapFactory = context.get(BootstrapFactory.class, BootstrapFactory.INSTANCE);
045                                boolean isFirst = true;
046                                for (Object item: input.getSecond()) {
047                                        if (isFirst) {
048                                                isFirst = false;
049                                        } else {
050                                                items.add(bootstrapFactory.getHTMLFactory().span("&nbsp;&bull;&nbsp;"));
051                                        }
052                                        items.add(item);
053                                }
054                                Tag navs = Util.navs(items, bootstrapFactory);
055                                navs.addClass("nsd-app-footer-navs");
056                                ret.content(navs);                              
057                                return ret;
058                        }
059                };
060        }
061        
062        /**
063         * Adapts items to suppliers, builds {@link Navs} from items and adds them to the header.
064         */
065        @Override
066        protected Function<HTMLElement<?>, HTMLElement<?>> createConfigureFunction(Context context) throws Exception {
067                EList<EObject> items = getTarget().getItems();
068                if (items.isEmpty()) {
069                        return super.createConfigureFunction(context);
070                }
071                Function<HTMLElement<?>, BiSupplier<HTMLElement<?>, List<Object>>> itemsFunction = new ListCompoundSupplierFactory<>("Items", EObjectAdaptable.adaptToSupplierFactoryNonNull(items, Object.class)).<HTMLElement<?>>asFunctionFactory().create(context);
072                return super.createConfigureFunction(context).then(itemsFunction).then(createNavsFunction(context));
073        }
074
075}
