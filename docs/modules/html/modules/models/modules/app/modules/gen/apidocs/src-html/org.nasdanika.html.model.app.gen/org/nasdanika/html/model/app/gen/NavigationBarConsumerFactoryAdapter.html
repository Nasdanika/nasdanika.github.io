001package org.nasdanika.html.model.app.gen;
002
003import java.util.List;
004import java.util.Map;
005
006import org.eclipse.emf.common.notify.AdapterFactory;
007import org.eclipse.emf.common.util.EList;
008import org.eclipse.emf.ecore.EObject;
009import org.eclipse.emf.ecore.EStructuralFeature;
010import org.nasdanika.common.BiSupplier;
011import org.nasdanika.common.Context;
012import org.nasdanika.common.Function;
013import org.nasdanika.common.ListCompoundSupplierFactory;
014import org.nasdanika.common.MapCompoundSupplierFactory;
015import org.nasdanika.common.ProgressMonitor;
016import org.nasdanika.emf.EObjectAdaptable;
017import org.nasdanika.html.HTMLElement;
018import org.nasdanika.html.Tag;
019import org.nasdanika.html.bootstrap.BootstrapFactory;
020import org.nasdanika.html.bootstrap.Navs;
021import org.nasdanika.html.model.app.AppPackage;
022import org.nasdanika.html.model.app.Label;
023import org.nasdanika.html.model.app.NavigationBar;
024
025public class NavigationBarConsumerFactoryAdapter extends PagePartConsumerFactoryAdapter<NavigationBar> {
026
027        protected NavigationBarConsumerFactoryAdapter(NavigationBar navigationBar, AdapterFactory adapterFactory) {
028                super(navigationBar, adapterFactory);
029        }
030        
031        private Function<BiSupplier<HTMLElement<?>, Map<EStructuralFeature, Object>>, HTMLElement<?>> createNavbarFunction(Context context) {
032                return new Function<BiSupplier<HTMLElement<?>,Map<EStructuralFeature, Object>>, HTMLElement<?>>() {
033                        
034                        @Override
035                        public double size() {
036                                return 1;
037                        }
038                        
039                        @Override
040                        public String name() {
041                                return "Navs";
042                        }
043                        
044                        @Override
045                        public HTMLElement<?> execute(BiSupplier<HTMLElement<?>, Map<EStructuralFeature, Object>> input, ProgressMonitor progressMonitor) throws Exception {
046                                Tag ret = (Tag) input.getFirst();
047                                Tag brand = (Tag) input.getSecond().get(AppPackage.Literals.NAVIGATION_BAR__BRAND);
048                                @SuppressWarnings("unchecked")
049                                List<Object> items = (List<Object>) input.getSecond().get(AppPackage.Literals.PAGE_PART__ITEMS);        
050                                NavigationBar semanticElement = getTarget();
051                                ret.content(Util.navbar(
052                                                brand, 
053                                                items, 
054                                                semanticElement.getExpand(), 
055                                                semanticElement.isDark(),
056                                                semanticElement.getBackground(),                                                
057                                                context.get(BootstrapFactory.class, BootstrapFactory.INSTANCE)));       
058                                
059                                return ret;
060                        }
061                };
062        }
063        
064        /**
065         * Adapts items to suppliers, builds {@link Navs} from items and adds them to the header.
066         */
067        @Override
068        protected Function<HTMLElement<?>, HTMLElement<?>> createConfigureFunction(Context context) throws Exception {
069                Label brand = getTarget().getBrand();
070                EList<EObject> items = getTarget().getItems();
071                if (brand == null && items.isEmpty()) {
072                        return super.createConfigureFunction(context);
073                }
074                
075                MapCompoundSupplierFactory<EStructuralFeature, Object> featuresSupplierFactory = new MapCompoundSupplierFactory<>("Brand and items");
076                
077                if (brand != null) {
078                        featuresSupplierFactory.put(AppPackage.Literals.NAVIGATION_BAR__BRAND, EObjectAdaptable.adaptToSupplierFactoryNonNull(brand, Object.class));
079                }
080                
081                if (!items.isEmpty()) {
082                        featuresSupplierFactory.put(AppPackage.Literals.PAGE_PART__ITEMS, new ListCompoundSupplierFactory<>("Items", EObjectAdaptable.adaptToSupplierFactoryNonNull(items, Object.class)));
083                }
084                
085                Function<HTMLElement<?>, BiSupplier<HTMLElement<?>, Map<EStructuralFeature, Object>>> featuresFunction = featuresSupplierFactory.<HTMLElement<?>>asFunctionFactory().create(context);                   
086                return super.createConfigureFunction(context).then(featuresFunction).then(createNavbarFunction(context));
087        }
088        
089}
