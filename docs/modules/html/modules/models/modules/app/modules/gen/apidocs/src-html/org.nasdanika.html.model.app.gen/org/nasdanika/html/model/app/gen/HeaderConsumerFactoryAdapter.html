001package org.nasdanika.html.model.app.gen;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import org.eclipse.emf.common.notify.AdapterFactory;
007import org.eclipse.emf.common.util.EList;
008import org.eclipse.emf.ecore.EObject;
009import org.eclipse.emf.ecore.util.EcoreUtil;
010import org.nasdanika.common.BiSupplier;
011import org.nasdanika.common.ConsumerFactory;
012import org.nasdanika.common.Context;
013import org.nasdanika.common.Function;
014import org.nasdanika.common.ListCompoundSupplierFactory;
015import org.nasdanika.common.ProgressMonitor;
016import org.nasdanika.common.SupplierFactory;
017import org.nasdanika.emf.EObjectAdaptable;
018import org.nasdanika.exec.content.ContentFactory;
019import org.nasdanika.exec.content.Text;
020import org.nasdanika.html.HTMLElement;
021import org.nasdanika.html.Tag;
022import org.nasdanika.html.bootstrap.BootstrapFactory;
023import org.nasdanika.html.bootstrap.Navs;
024import org.nasdanika.html.model.app.Header;
025import org.nasdanika.html.model.app.Label;
026import org.nasdanika.html.model.bootstrap.Appearance;
027
028public class HeaderConsumerFactoryAdapter extends PagePartConsumerFactoryAdapter<Header> {
029
030        protected HeaderConsumerFactoryAdapter(Header header, AdapterFactory adapterFactory) {
031                super(header, adapterFactory);
032        }
033        
034        @Override
035        protected List<EObject> getContent() {          
036                // A bit convoluted, refactor later to use a function and add class to the title tag - similar to NavBar
037                List<EObject> ret = new ArrayList<>();
038                Label title = getTarget().getTitle();
039                if (title != null) {
040                        Label cTitle = EcoreUtil.copy(title);
041                        cTitle.eAdapters().add(AppAdapterFactory.INSTANCE.adapt(cTitle, SupplierFactory.Provider.class));
042                        
043                        Appearance titleAppearance = org.nasdanika.html.model.bootstrap.BootstrapFactory.eINSTANCE.createAppearance();
044                        titleAppearance.eAdapters().add(AppAdapterFactory.INSTANCE.adapt(titleAppearance, ConsumerFactory.class));
045                        cTitle.setAppearance(titleAppearance);
046                        
047                        Text cText = ContentFactory.eINSTANCE.createText();
048                        cText.eAdapters().add(AppAdapterFactory.INSTANCE.adapt(cText, SupplierFactory.class));
049                        cText.setContent("nsd-app-header-title");
050                        titleAppearance.getAttributes().put("class", cText);
051                        ret.add(cTitle);
052                }
053                ret.addAll(super.getContent());         
054                return ret;
055        }
056        
057        private Function<BiSupplier<HTMLElement<?>, List<Object>>, HTMLElement<?>> createNavsFunction(Context context) {
058                return new Function<BiSupplier<HTMLElement<?>,List<Object>>, HTMLElement<?>>() {
059                        
060                        @Override
061                        public double size() {
062                                return 1;
063                        }
064                        
065                        @Override
066                        public String name() {
067                                return "Navs";
068                        }
069                        
070                        @Override
071                        public HTMLElement<?> execute(BiSupplier<HTMLElement<?>, List<Object>> input, ProgressMonitor progressMonitor) throws Exception {
072                                Tag ret = (Tag) input.getFirst();
073                                Tag navs = Util.navs(input.getSecond(), context.get(BootstrapFactory.class, BootstrapFactory.INSTANCE));
074                                navs.addClass("nsd-app-header-navs");
075                                ret.content(navs);                              
076                                return ret;
077                        }
078                };
079        }
080        
081        /**
082         * Adapts items to suppliers, builds {@link Navs} from items and adds them to the header.
083         */
084        @Override
085        protected Function<HTMLElement<?>, HTMLElement<?>> createConfigureFunction(Context context) throws Exception {
086                EList<EObject> items = getTarget().getItems();
087                if (items.isEmpty()) {
088                        return super.createConfigureFunction(context);
089                }
090                Function<HTMLElement<?>, BiSupplier<HTMLElement<?>, List<Object>>> itemsFunction = new ListCompoundSupplierFactory<>("Items", EObjectAdaptable.adaptToSupplierFactoryNonNull(items, Object.class)).<HTMLElement<?>>asFunctionFactory().create(context);
091                return super.createConfigureFunction(context).then(itemsFunction).then(createNavsFunction(context));
092        }
093
094}
