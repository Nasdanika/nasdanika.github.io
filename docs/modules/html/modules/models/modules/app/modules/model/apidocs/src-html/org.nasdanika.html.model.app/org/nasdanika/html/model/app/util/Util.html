030        /**
031         * Resolves URI's by traversing containment references from the root Action taking {@link ActionReference} into account.
032         * @param context
033         * @return A function resolving {@link URI} for the argument {@link Action}. Caches results.
034         */
035        public static BiFunction<Label,URI,URI> uriResolver(Label root) {               
036                return uriResolver(root, null);
037        }
038        
039        /**
040         * Resolves URI's by traversing containment references from the root Action taking {@link ActionReference} into account.
041         * @param context
042         * @return A function resolving {@link URI} for the argument {@link Action}. Caches results.
043         */
044        public static BiFunction<Label,URI,URI> uriResolver(Label root, Context context) {              
045                Map<String, URI> cache = new HashMap<>();
046                if (context == null) {
047                        context = Context.EMPTY_CONTEXT;
048                }
049                URI baseURI = context.get(Context.BASE_URI_PROPERTY, URI.class);
050                if (baseURI == null) {
051                        context = context.fork();
052                        baseURI = URI.createURI("temp://" + UUID.randomUUID() + "/" + UUID.randomUUID() + "/");
053                        ((MutableContext) context).put(Context.BASE_URI_PROPERTY, baseURI);
054                }
055                
056                traverse(root, baseURI, context, cache);
057                
058                return new BiFunction<Label, URI, URI>() {
059                        
060                        @Override
061                        public URI apply(Label label, URI base) {
062                                URI uri = cache.get(label.getUuid());           
063                                return base == null || uri == null ? uri : uri.deresolve(base, true, true, true);
064                        }
065                
066                };
067        }
068        
069        private static void traverse(Label label, URI base, Context context, Map<String, URI> cache) {
070                URI linkURI = compute(label, base, context);
071                cache.put(label.getUuid(), linkURI);
072                
073                for (EObject child: resolveActionReferences(label.getChildren())) {
074                        if (child instanceof Action) {
075                                traverse((Action) child, linkURI == null ? base : linkURI, context, cache);
076                        }
077                }
078                
079                if (label instanceof Action) {
080                        Action action = (Action) label;
081                        for (EObject item: resolveActionReferences(action.getNavigation())) {
082                                if (item instanceof Link) {
083                                        traverse((Link) item, linkURI == null ? base : linkURI, context, cache);
084                                }
085                        }
086                        for (Action section: action.getSections()) {
087                                traverse(section, linkURI == null ? base : linkURI, context, cache);
088                        }
089                        for (Action anonymous: action.getAnonymous()) {
090                                traverse(anonymous, linkURI == null ? base : linkURI, context, cache);
091                        }
092                }
093                
094                // TODO - nav panels?
095        }
096
097        private static URI compute(Label label, URI base, Context context) {
098                if (label instanceof Link) {
099                        Link link = (Link) label;
100                        String uriString;
101                        if (link.eContainmentFeature() == AppPackage.Literals.ACTION__SECTIONS) {
102                                String aName = context.interpolateToString(link.getLocation());
103                                if (isBlank(aName)) {
104                                        return null;
105                                }
106                                uriString = "#" + aName;
107                        } else {                                
108                                uriString = context.interpolateToString(link.getLocation());
109                                if (isBlank(uriString)) {
110                                        return null;
111                                }
112                        }
113                        URI uri = URI.createURI(uriString);
114                        if (uri.isRelative() && base != null && base.isHierarchical() && !base.isRelative()) {
115                                return uri.resolve(base);
116                        }
117                        return uri;
118                } 
119                
120                return null;
121        }
122        
123        public static EObject resolveActionReference(EObject obj) {
124                return obj instanceof ActionReference ? ((ActionReference) obj).getTarget() : obj;
125        }
126        
127        public static List<EObject> resolveActionReferences(EList<EObject> objs) {
128                return objs.stream().map((Function<EObject, EObject>) Util::resolveActionReference).collect(Collectors.toList());
129        }
