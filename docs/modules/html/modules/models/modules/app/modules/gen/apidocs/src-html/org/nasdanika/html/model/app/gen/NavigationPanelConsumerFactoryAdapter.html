042        private Function<BiSupplier<HTMLElement<?>, List<JsTreeNode>>, HTMLElement<?>> createTreeFunction(Context context) {
043                return new Function<BiSupplier<HTMLElement<?>, List<JsTreeNode>>, HTMLElement<?>>() {
044                        
045                        @Override
046                        public double size() {
047                                return 1;
048                        }
049                        
050                        @Override
051                        public String name() {
052                                return "JsTree";
053                        }
054                        
055                        @Override
056                        public HTMLElement<?> execute(BiSupplier<HTMLElement<?>, List<JsTreeNode>> input, ProgressMonitor progressMonitor) throws Exception {
057                                Tag ret = (Tag) input.getFirst();
058                                
059                                Tag panel;
060                                if (getTarget().isCollapsible()) {
061                                        Tag triggerDiv = ret.getFactory().div().addClass("nsd-collapse-panel", "bg-light", "text-center");
062                                        ret.content(triggerDiv);
063                                        HTMLFactory htmlFactory = ret.getFactory();
064                                        panel = htmlFactory.div();
065                                        panel.id("nsd-nav-panel-" + getTarget().getUuid() + "-collapsible");
066                                        
067                                        Tag iTag = htmlFactory.tag(TagName.i).addClass("fa").attribute("aria-hidden", "true");
068
069                                        Tag collapsibleTrigger = htmlFactory.span(iTag)
070                                                        .addClass("nsd-collapsible-trigger")
071                                                        .attribute("data-toggle", "collapse")
072                                                        .attribute("data-target", "#" + panel.getId())
073                                                        .attribute("aria-expanded", "true")
074                                                        .attribute("aria-controls", panel.getId())
075                                                        .attribute("title", "Toggle panel")
076                                                        .id(panel.getId()+"-trigger");
077                                        
078                                        triggerDiv.content(collapsibleTrigger);
079                                        panel.addClass("nsd-collapsible", "collapse", "show");
080                                        panel.style("clear", "right");
081
082                                        ret.content(panel);
083                                } else {
084                                        panel = ret;
085                                }
086                                
087                                JsTreeFactory jsTreeFactory = context.get(JsTreeFactory.class, JsTreeFactory.INSTANCE);
088                                String treeId = getTarget().getId();
089                                HTMLFactory htmlFactory = jsTreeFactory.getHTMLFactory();
090                                if (org.nasdanika.common.Util.isBlank(treeId)) {
091                                        treeId = htmlFactory.nextId();
092                                }
093                                Tag container = htmlFactory.div().id(treeId);
094                                panel.accept(container);
095                                
096                                JSONObject jsTree = jsTreeFactory.buildJsTree(input.getSecond());
097
098                                // Configures jsTree. TODO - from the model.
099                                jsTree.put("plugins", Arrays.asList("state", "types"));                 
100                                jsTree.put("state", Collections.singletonMap("key", treeId));
101                                jsTree.put("types", Collections.singletonMap("leaf", Collections.singletonMap("icon", "jstree-file"))); // File leaf icon.
102                                
103                                // Deletes selection from state
104                                String filter = "tree.state.filter = function(state) { delete state.core.selected; return state; };";
105                                
106                                Tag script = jsTreeFactory.bind(container, jsTree, filter);
107                                panel.accept(script);
108                                
109                                return ret;
110                        }
111                                                
112                };
113        }

115        private Function<BiSupplier<HTMLElement<?>, Map<Class<?>, Object>>, HTMLElement<?>> createCardsFunction(Context context) {
116                return new Function<BiSupplier<HTMLElement<?>, Map<Class<?>, Object>>, HTMLElement<?>>() {
117                        
118                        @Override
119                        public double size() {
120                                return 1;
121                        }
122                        
123                        @Override
124                        public String name() {
125                                return "Cards";
126                        }
127                        
128                        @SuppressWarnings("unchecked")
129                        @Override
130                        public HTMLElement<?> execute(BiSupplier<HTMLElement<?>, Map<Class<?>, Object>> input, ProgressMonitor progressMonitor) throws Exception {
131                                Tag ret = (Tag) input.getFirst();
132                                Tag panel;
133                                if (getTarget().isCollapsible()) {
134                                        HTMLFactory htmlFactory = ret.getFactory();
135                                        Tag triggerDiv = htmlFactory.div().addClass("nsd-collapse-panel", "bg-light", "text-center");
136                                        ret.content(triggerDiv);
137                                        panel = htmlFactory.div();
138                                        panel.id("nsd-nav-panel-" + getTarget().getUuid() + "-collapsible");
139                                        
140                                        Tag iTag = htmlFactory.tag(TagName.i).addClass("fa").attribute("aria-hidden", "true");
141
142                                        Tag collapsibleTrigger = htmlFactory.span(iTag)
143                                                        .addClass("nsd-collapsible-trigger")
144                                                        .attribute("data-toggle", "collapse")
145                                                        .attribute("data-target", "#" + panel.getId())
146                                                        .attribute("aria-expanded", "true")
147                                                        .attribute("aria-controls", panel.getId())
148                                                        .attribute("title", "Toggle panel")                                                     
149                                                        .id(panel.getId()+"-trigger");
150                                        
151                                        triggerDiv.content(collapsibleTrigger);
152                                        panel.addClass("nsd-collapsible", "collapse", "show");
153                                        panel.style("clear", "right");
154
155                                        ret.content(panel);
156                                } else {
157                                        panel = ret;
158                                }
159                                
160                                List<Object> items = (List<Object>) input.getSecond().get(Object.class);
161                                
162                                JsTreeFactory jsTreeFactory = context.get(JsTreeFactory.class, JsTreeFactory.INSTANCE);
163                                String treeId = getTarget().getId();
164                                HTMLFactory htmlFactory = jsTreeFactory.getHTMLFactory();
165                                
166                                Tag currentActionGroup = null; // Add items without children to an action group
167                                for (Object item: items) {
168                                        if (item instanceof Tag) {
169                                                Tag itemTag = (Tag) item;
170                                                Object data = itemTag.getData();
171                                                if (data instanceof Label) {
172                                                        Label semanticElement = (Label) data; 
173                                                        if (semanticElement.getChildren().isEmpty()) {
174                                                                if (currentActionGroup == null) {
175                                                                        currentActionGroup = htmlFactory.div().addClass("list-group", "list-group-flush");
176                                                                }
177                                                                itemTag.addClass("list-group-item", "list-group-action");
178                                                                itemTag.addClassConditional(semanticElement.isActive(), "active");
179                                                                itemTag.addClassConditional(semanticElement.isDisabled(), "disabled");
180                                                                currentActionGroup.accept(itemTag);
181                                                        } else {
182                                                                if (currentActionGroup != null) {
183                                                                        panel.accept(currentActionGroup);
184                                                                        currentActionGroup = null;
185                                                                }
186                                                                
187                                                                Card card = context.get(BootstrapFactory.class, BootstrapFactory.INSTANCE).card();
188                                                                panel.accept(card);
189                                                                TagBootstrapElement cardHeader = card.getHeader();
190                                                                Color color = semanticElement.getColor();
191                                                                if (color != null) {
192                                                                        card.border(color);
193                                                                        cardHeader.background(color);
194                                                                }
195                                                                Tag title = (Tag) itemTag.getContent().get(0);
196                                                                
197                                                                // Remove dropdown classes and attributes
198                                                                title
199                                                                        .removeClass("nav-link", "dropdown-toggle")
200                                                                        .attribute("role", null)
201                                                                        .attribute("data-toggle", null);                                                                
202                                                                
203                                                                if (color != null) {
204                                                                        title.removeClass("text-" + color.code);
205                                                                }
206                                                                
207                                                                Tag cardHeaderTag = cardHeader.toHTMLElement();
208                                                                Tag collapsible;
209                                                                // Tree or items
210                                                                TagBootstrapElement cardBody = card.getBody();                                                                                                                          
211                                                                if (depth(semanticElement.getChildren()) == 1) {
212                                                                        Tag childrenActionGroup = htmlFactory.div().addClass("list-group", "list-group-flush");
213                                                                        for (Object itemChild: (List<Object>) itemTag.getData(AppPackage.Literals.LABEL__CHILDREN)) {
214                                                                                if (itemChild instanceof Tag) {
215                                                                                        Tag itemChildTag = (Tag) itemChild;
216                                                                                        Label childSemanticElement = (Label) itemChildTag.getData();
217                                                                                        itemChildTag.addClass("list-group-item", "list-group-action").removeClass("dropdown-item");
218                                                                                        itemChildTag.addClassConditional(childSemanticElement.isActive(), "active");
219                                                                                        itemChildTag.addClassConditional(childSemanticElement.isDisabled(), "disabled");
220                                                                                        childrenActionGroup.accept(itemChildTag);
221                                                                                } else {
222                                                                                        cardBody.toHTMLElement().accept(itemChild);
223                                                                                }
224                                                                        }
225                                                                        card.toHTMLElement().accept(childrenActionGroup);
226                                                                        collapsible = childrenActionGroup;
227                                                                } else {                                                                        
228                                                                        cardBody.padding().all(Breakpoint.DEFAULT, Size.S1);
229                                                                        collapsible = cardBody.toHTMLElement();
230                                                                        for (JsTreeNode node: (List<JsTreeNode>) input.getSecond().get(JsTreeNode.class)) {
231                                                                                if (node.getData() == semanticElement) {
232                                                                                        String nodeTreeId;
233                                                                                        if (org.nasdanika.common.Util.isBlank(treeId)) {
234                                                                                                nodeTreeId = "nsd-nav-tree-" + semanticElement.getId();
235                                                                                        } else {
236                                                                                                nodeTreeId = treeId + "-" + semanticElement.getId();
237                                                                                        }
238                                                                                        
239                                                                                        Tag container = htmlFactory.div().id(nodeTreeId);
240                                                                                        cardBody.toHTMLElement().accept(container);
241                                                                                        
242                                                                                        JSONObject jsTree = jsTreeFactory.buildJsTree(node.children());
243                                                        
244                                                                                        // Configures jsTree. TODO - from the model.
245                                                                                        jsTree.put("plugins", Arrays.asList("state", "types"));                 
246                                                                                        jsTree.put("state", Collections.singletonMap("key", treeId));
247                                                                                        jsTree.put("types", Collections.singletonMap("leaf", Collections.singletonMap("icon", "jstree-file"))); // File leaf icon.
248                                                                                        
249                                                                                        // Deletes selection from state
250                                                                                        String filter = "tree.state.filter = function(state) { delete state.core.selected; return state; };";
251                                                                                        
252                                                                                        Tag script = jsTreeFactory.bind(container, jsTree, filter);
253                                                                                        panel.accept(script);
254                                                                                        break;
255                                                                                }
256                                                                        }
257                                                                }
258
259                                                                // Collapse button
260                                                                if (getTarget().getStyle() == NavigationPanelStyle.COLLAPSIBLE_CARDS) {
261                                                                        Tag iTag = htmlFactory.tag(TagName.i).addClass("fa").attribute("aria-hidden", "true");
262                                                                        collapsible.id(semanticElement.getId()+"-collapsible");
263
264                                                                        Tag collapsibleTrigger = htmlFactory.span(iTag)
265                                                                                        .addClass("nsd-collapsible-trigger")
266                                                                                        .attribute("data-toggle", "collapse")
267                                                                                        .attribute("data-target", "#" + collapsible.getId())
268                                                                                        .attribute("aria-expanded", "true")
269                                                                                        .attribute("aria-controls", collapsible.getId())
270                                                                                        .attribute("title", "Toggle card")                                                                                      
271                                                                                        .id(semanticElement.getId()+"-collapsible-trigger");
272                                                                        
273                                                                        cardHeaderTag.content(collapsibleTrigger);
274                                                                        collapsible.addClass("nsd-collapsible", "collapse", "show");
275                                                                }
276                                                                
277                                                                cardHeaderTag.content(title);                                                           
278                                                        }
279                                                } else {
280                                                        panel.accept(item);
281                                                }                                               
282                                        } else {
283                                                panel.accept(item);
284                                        }
285                                }
286                                if (currentActionGroup != null) {
287                                        panel.accept(currentActionGroup);
288                                }
289                                
290                                return ret;
291                        }
292                };
293        }
294                
295        private static int depth(Collection<EObject> items) {
296                int ret = 0;
297                for (EObject item: items) {
298                        if (item instanceof Label) {
299                                ret = Math.max(ret, depth(((Label) item).getChildren()) + 1);
300                        }
301                }
302                return ret;
303        }

305        /**
306         * Adapts items to {@link JsTreeNode} suppliers.
307         */
308        @Override
309        protected Function<HTMLElement<?>, HTMLElement<?>> createConfigureFunction(Context context) throws Exception {
310                NavigationPanel semanticElement = getTarget();
311                NavigationPanelStyle style = semanticElement.getStyle();
312                
313                EList<EObject> items = semanticElement.getItems();
314                if (style == NavigationPanelStyle.AUTO) {
315                        style = depth(items) > 2 ?  NavigationPanelStyle.TREE : NavigationPanelStyle.CARDS; 
316                }
317                
318                ListCompoundSupplierFactory<JsTreeNode> nodesFactory = new ListCompoundSupplierFactory<>("Nodes", EObjectAdaptable.adaptToSupplierFactoryNonNull(items, JsTreeNode.class));
319                if (style == NavigationPanelStyle.TREE) {
320                        Function<HTMLElement<?>, BiSupplier<HTMLElement<?>, List<JsTreeNode>>> nodesFunction = nodesFactory.<HTMLElement<?>>asFunctionFactory().create(context);
321                        return super.createConfigureFunction(context).then(nodesFunction).then(createTreeFunction(context));
322                }
323                
324                ListCompoundSupplierFactory<Object> tagsFactory = new ListCompoundSupplierFactory<>("Tags", EObjectAdaptable.adaptToSupplierFactoryNonNull(items, Object.class));
325                
326                MapCompoundSupplierFactory<Class<?>, Object> tagsAndNodesSupplierFactory = new MapCompoundSupplierFactory<>("Tags and nodes");
327                tagsAndNodesSupplierFactory.put(JsTreeNode.class, nodesFactory);
328                tagsAndNodesSupplierFactory.put(Object.class, tagsFactory);
329                Function<HTMLElement<?>, BiSupplier<HTMLElement<?>, Map<Class<?>, Object>>> tagsAndNodesFunction = tagsAndNodesSupplierFactory.<HTMLElement<?>>asFunctionFactory().create(context);
330                
331                return super.createConfigureFunction(context).then(tagsAndNodesFunction).then(createCardsFunction(context));
332        }
