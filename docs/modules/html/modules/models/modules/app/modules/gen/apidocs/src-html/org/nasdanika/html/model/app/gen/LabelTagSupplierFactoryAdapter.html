030public class LabelTagSupplierFactoryAdapter<M extends Label> extends BootstrapElementSupplierFactoryAdapter<M, BootstrapElement<?,?>> {
031        
032        public LabelTagSupplierFactoryAdapter(M label) {
033                super(label);
034        }
035                
036        @Override
037        public Supplier<HTMLElement<?>> createHTMLElementSupplier(Context context) throws Exception {
038                Function<Map<EStructuralFeature, Object>, HTMLElement<?>> labelFunction = new Function<Map<EStructuralFeature, Object>, HTMLElement<?>>() {
039
040                        @Override
041                        public double size() {
042                                return 1;
043                        }
044
045                        @Override
046                        public String name() {
047                                return "Label";
048                        }
049
050                        @Override
051                        public HTMLElement<?> execute(Map<EStructuralFeature, Object> features, ProgressMonitor progressMonitor) throws Exception {
052                                BootstrapFactory bootstrapFactory = context.get(BootstrapFactory.class, BootstrapFactory.INSTANCE);
053                                        
054                                Tag ret = (Tag) features.get(AppPackage.Literals.LABEL__TEXT);
055                                
056                                Label semanticElement = getTarget();
057                                Color color = semanticElement.getColor();                                                               
058                                String notification = semanticElement.getNotification();
059                                if (!Util.isBlank(notification)) {
060                                        Color notificationColor = Color.PRIMARY;
061                                        if (color == Color.PRIMARY) {
062                                                notificationColor = Color.SECONDARY;
063                                        } else if (color == Color.DANGER) {
064                                                notificationColor = Color.DANGER;                               
065                                        } else if (color == Color.WARNING) {
066                                                notificationColor = Color.WARNING;                              
067                                        } 
068                                        Tag badge = bootstrapFactory.badge(true, notificationColor, notification);
069                                        badge.addClass("nsd-label-notification");
070                                        ret.accept(badge);
071                                }
072                                
073                                String tooltip = semanticElement.getTooltip();
074                                if (!Util.isBlank(tooltip)) {
075                                        ret.attribute("title", tooltip);
076                                }                               
077                                
078                                Tag help = (Tag) features.get(AppPackage.Literals.LABEL__HELP);
079                                HTMLFactory htmlFactory = bootstrapFactory.getHTMLFactory();
080                                if (help != null) {
081                                        @SuppressWarnings("unchecked")
082                                        List<Object> pageBody = context.get(org.nasdanika.html.model.html.gen.PageSupplierFactoryAdapter.PAGE_BODY_PROPERTY, List.class);
083                                        pageBody.add(help);
084                                        
085                                        Tag trigger = htmlFactory.tag(TagName.sup).addClass("far fa-question-circle", "nsd-label-help").style("cursor", "pointer");
086                                        if (!Util.isBlank(tooltip)) {
087                                                trigger.attribute("title", tooltip);
088                                        }
089                                        if (help.getId() == null) {
090                                                help.id(htmlFactory.nextId());
091                                        }
092                                        trigger.attribute("data-toggle", "modal");
093                                        trigger.attribute("data-target", "#" + help.getId());
094                                        
095                                        if (TagName.span.name().equalsIgnoreCase(ret.getTagName())) {
096                                                ret.accept(trigger);
097                                        } else {
098                                                ret = htmlFactory.span(ret, trigger);
099                                        }
100                                }
101                                                                
102                                @SuppressWarnings("unchecked")
103                                List<Object> children = (List<Object>) features.get(AppPackage.Literals.LABEL__CHILDREN);
104                                if (children == null || semanticElement.eContainer() instanceof Label) {                                
105                                        setData(ret, semanticElement, features);
106                                        return ret;
107                                }
108                                
109                                // Navigation drop-down
110                                Tag dropdown = htmlFactory.tag(TagName.li).addClass("dropdown");
111                                ret
112                                        .addClass("nav-link", "dropdown-toggle")
113                                        .attribute("role", "button")
114                                        .attribute("data-toggle", "dropdown");
115                                
116                                dropdown.accept(ret);
117                                                                
118                                Tag dropdownMenu = htmlFactory.div().addClass("dropdown-menu");
119                                dropdown.accept(dropdownMenu);                          
120                                addDropdownItems(dropdownMenu, children);
121                                
122                                setData(dropdown, semanticElement, features);                           
123                                return dropdown;
124                        }
125
126                        protected void setData(Tag ret, Label semanticElement, Map<EStructuralFeature, Object> features) {
127                                ret.setData(semanticElement);
128                                
129                                /**
130                                 * Storing feature data to use downstream.
131                                 */
132                                for (Entry<EStructuralFeature, Object> fe: features.entrySet()) {
133                                        ret.setData(fe.getKey(), fe.getValue());
134                                }
135                        }
136                        
137                        @SuppressWarnings("unchecked")
138                        private void addDropdownItems(Tag dropdownMenu, List<Object> items) {
139                                for (Object item: items) {                                      
140                                        dropdownMenu.accept(item);
141                                        if (item instanceof Tag) {
142                                                Tag itemTag = (Tag) item;
143                                                switch (itemTag.getTagName().toLowerCase()) {
144                                                case "a": 
145                                                        itemTag.addClass("dropdown-item");
146                                                        break;
147                                                case "div": 
148                                                        itemTag.addClass("dropdown-divider");
149                                                        break;
150                                                case "h6": 
151                                                        itemTag.addClass("dropdown-header");
152                                                        break;
153                                                }
154                                                List<Object> children = (List<Object>) itemTag.getData(AppPackage.Literals.LABEL__CHILDREN);
155                                                if (children != null) {
156                                                        addDropdownItems(dropdownMenu, children);
157                                                }                                       
158                                        }
159                                }
160                        }
161                        
162                };
163                
164                MapCompoundSupplierFactory<EStructuralFeature, Object> featuresFactory = new MapCompoundSupplierFactory<>("Features");
165                
166                M semanticElement = getTarget();
167                org.nasdanika.html.model.bootstrap.Modal help = semanticElement.getHelp();
168                if (help != null) {
169                        featuresFactory.put(AppPackage.Literals.LABEL__HELP, EObjectAdaptable.adaptToSupplierFactoryNonNull(help, Tag.class));
170                }
171                
172                List<EObject> children = semanticElement.getChildren();
173                if (!children.isEmpty()) {
174                        featuresFactory.put(AppPackage.Literals.LABEL__CHILDREN, new ListCompoundSupplierFactory<>("Children", EObjectAdaptable.adaptToSupplierFactoryNonNull(children, Object.class)));                        
175                }
176                
177                SupplierFactory<Tag> textSupplierFactory = this::createTextAndIconSupplier;
178                featuresFactory.put(AppPackage.Literals.LABEL__TEXT, textSupplierFactory);
179                
180                return featuresFactory.create(context).then(labelFunction);
181        }

183        /**
184         * Override in Link to create a link.
185         * @param context
186         * @param progressMonitor
187         * @return
188         * @throws Exception 
189         */
190        protected Supplier<Tag> createTextAndIconSupplier(Context context) throws Exception {
191                Function<Tag,Tag> textAndIconFunction = new Function<Tag, Tag>() {
192                                
193                        @Override
194                        public double size() {
195                                return 1;
196                        }
197
198                        @Override
199                        public String name() {
200                                return "Label text and icon";
201                        }
202
203                        @Override
204                        public Tag execute(Tag modal, ProgressMonitor progressMonitor) throws Exception {
205                                BootstrapFactory bootstrapFactory = context.get(BootstrapFactory.class, BootstrapFactory.INSTANCE);
206                                HTMLFactory htmlFactory = context.get(HTMLFactory.class, HTMLFactory.INSTANCE);
207                                Label semanticElement = getTarget();
208                                
209                                String text = context.interpolateToString(getTarget().getText());
210                                for (EObject semanticAncestor = semanticElement.eContainer(); semanticAncestor != null; semanticAncestor = semanticAncestor.eContainer()) {
211                                        if (semanticAncestor instanceof NavigationPanel) {
212                                                int labelTrimLength = ((NavigationPanel) semanticAncestor).getLabelTrimLength();
213                                                if (labelTrimLength > 0 && text.length() > labelTrimLength) {
214                                                        text = text.substring(0, labelTrimLength - 3) + "...";
215                                                }
216                                                break;
217                                        }                                               
218                                }
219                                
220                                String icon = semanticElement.getIcon();
221                                Tag iconTag = null;
222                                if (icon != null) {
223                                        if (icon.contains("/")) {
224                                                // Image
225                                                iconTag = htmlFactory.tag(TagName.img).attribute("src", icon); //.style().height("1em");
226                                        } else {
227                                                // Class
228                                                iconTag = htmlFactory.span().addClass(icon);
229                                        }
230                                }
231                                
232                                Tag ret = createTextAndIconTag(iconTag, text, modal, context, progressMonitor);
233
234                                Color color = semanticElement.getColor();                               
235                                if (color != null) {
236                                        bootstrapFactory.wrap(ret).text().color(color);
237                                }       
238                                                                                                
239                                return ret;
240                        }
241                        
242                };
243                
244                return getModalFactory().create(context).then(textAndIconFunction);                             
245        }
246        
247        /**
248         * @param icon Icon
249         * @param text text
250         * @param modal Modal for binding to links, not applicalbe to labels.
251         * @param context
252         * @param progressMonitor
253         * @return
254         */
255        protected Tag createTextAndIconTag(Tag icon, String text, Tag modal, Context context, ProgressMonitor progressMonitor) {
256                HTMLFactory htmlFactory = context.get(HTMLFactory.class, HTMLFactory.INSTANCE);
257                TagName tagName;
258                if (getTarget().getChildren().isEmpty()) {
259                        tagName = TagName.span; // A regular label.
260                } else {
261                        // Drop-down activator or drop-down header if contained in a label
262                        tagName = getTarget().eContainer() instanceof Label ? TagName.h6 : TagName.a;
263                }
264                
265                if (Util.isBlank(text)) {
266                        if (icon == null) {
267                                // Drop-down divider
268                                return htmlFactory.div();
269                        }
270                        
271                        // Wrap into a header for drop-down headers
272                        return tagName == TagName.h6 ? htmlFactory.tag(tagName, icon) : icon;
273                }
274                if (icon == null) {
275                        return htmlFactory.tag(tagName, text);
276                }
277                
278                icon.addClass("nsd-app-label-icon");
279                return htmlFactory.tag(tagName, icon, text);    
280        }
281        
282        /**
283         * For links. Placing here to simplify code. This implementation returns empty supplier.
284         * @return
285         */
286        protected SupplierFactory<Tag> getModalFactory() {
287                return SupplierFactory.empty();
288        }
