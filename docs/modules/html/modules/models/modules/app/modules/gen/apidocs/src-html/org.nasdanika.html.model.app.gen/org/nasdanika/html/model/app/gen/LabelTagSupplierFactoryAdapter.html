031public class LabelTagSupplierFactoryAdapter<M extends Label> extends BootstrapElementSupplierFactoryAdapter<M, BootstrapElement<?,?>> {
032        
033        public LabelTagSupplierFactoryAdapter(M label, AdapterFactory adapterFactory) {
034                super(label, adapterFactory);
035        }
036                
037        @Override
038        public Supplier<HTMLElement<?>> createHTMLElementSupplier(Context context) throws Exception {
039                Function<Map<EStructuralFeature, Object>, HTMLElement<?>> labelFunction = new Function<Map<EStructuralFeature, Object>, HTMLElement<?>>() {
040
041                        @Override
042                        public double size() {
043                                return 1;
044                        }
045
046                        @Override
047                        public String name() {
048                                return "Label";
049                        }
050
051                        @Override
052                        public HTMLElement<?> execute(Map<EStructuralFeature, Object> features, ProgressMonitor progressMonitor) throws Exception {
053                                BootstrapFactory bootstrapFactory = context.get(BootstrapFactory.class, BootstrapFactory.INSTANCE);
054                                        
055                                Tag ret = (Tag) features.get(AppPackage.Literals.LABEL__TEXT);
056                                
057                                Label semanticElement = getTarget();
058                                Color color = semanticElement.getColor();                                                               
059                                String notification = semanticElement.getNotification();
060                                if (!Util.isBlank(notification)) {
061                                        Color notificationColor = Color.PRIMARY;
062                                        if (color == Color.PRIMARY) {
063                                                notificationColor = Color.SECONDARY;
064                                        } else if (color == Color.DANGER) {
065                                                notificationColor = Color.DANGER;                               
066                                        } else if (color == Color.WARNING) {
067                                                notificationColor = Color.WARNING;                              
068                                        } 
069                                        Tag badge = bootstrapFactory.badge(true, notificationColor, notification);
070                                        badge.addClass("nsd-label-notification");
071                                        ret.accept(badge);
072                                }
073                                
074                                String tooltip = semanticElement.getTooltip();
075                                if (!Util.isBlank(tooltip)) {
076                                        ret.attribute("title", tooltip);
077                                }                               
078                                
079                                Tag help = (Tag) features.get(AppPackage.Literals.LABEL__HELP);
080                                HTMLFactory htmlFactory = bootstrapFactory.getHTMLFactory();
081                                if (help != null) {
082                                        @SuppressWarnings("unchecked")
083                                        List<Object> pageBody = context.get(org.nasdanika.html.model.html.gen.PageSupplierFactoryAdapter.PAGE_BODY_PROPERTY, List.class);
084                                        pageBody.add(help);
085                                        
086                                        Tag trigger = htmlFactory.tag(TagName.sup).addClass("far fa-question-circle", "nsd-label-help").style("cursor", "pointer");
087                                        if (!Util.isBlank(tooltip)) {
088                                                trigger.attribute("title", tooltip);
089                                        }
090                                        if (help.getId() == null) {
091                                                help.id(htmlFactory.nextId());
092                                        }
093                                        trigger.attribute("data-toggle", "modal");
094                                        trigger.attribute("data-target", "#" + help.getId());
095                                        
096                                        if (TagName.span.name().equalsIgnoreCase(ret.getTagName())) {
097                                                ret.accept(trigger);
098                                        } else {
099                                                ret = htmlFactory.span(ret, trigger);
100                                        }
101                                }
102                                                                
103                                @SuppressWarnings("unchecked")
104                                List<Object> children = (List<Object>) features.get(AppPackage.Literals.LABEL__CHILDREN);
105                                if (children == null || semanticElement.eContainer() instanceof Label) {                                
106                                        setData(ret, semanticElement, features);
107                                        return ret;
108                                }
109                                
110                                // Navigation drop-down
111                                Tag dropdown = htmlFactory.tag(TagName.li).addClass("dropdown");
112                                ret
113                                        .addClass("nav-link", "dropdown-toggle")
114                                        .attribute("role", "button")
115                                        .attribute("data-toggle", "dropdown");
116                                
117                                dropdown.accept(ret);
118                                                                
119                                Tag dropdownMenu = htmlFactory.div().addClass("dropdown-menu");
120                                dropdown.accept(dropdownMenu);                          
121                                addDropdownItems(dropdownMenu, children);
122                                
123                                setData(dropdown, semanticElement, features);                           
124                                return dropdown;
125                        }
126
127                        protected void setData(Tag ret, Label semanticElement, Map<EStructuralFeature, Object> features) {
128                                ret.setData(semanticElement);
129                                
130                                /**
131                                 * Storing feature data to use downstream.
132                                 */
133                                for (Entry<EStructuralFeature, Object> fe: features.entrySet()) {
134                                        ret.setData(fe.getKey(), fe.getValue());
135                                }
136                        }
137                        
138                        @SuppressWarnings("unchecked")
139                        private void addDropdownItems(Tag dropdownMenu, List<Object> items) {
140                                for (Object item: items) {                                      
141                                        dropdownMenu.accept(item);
142                                        if (item instanceof Tag) {
143                                                Tag itemTag = (Tag) item;
144                                                switch (itemTag.getTagName().toLowerCase()) {
145                                                case "a": 
146                                                        itemTag.addClass("dropdown-item");
147                                                        break;
148                                                case "div": 
149                                                        itemTag.addClass("dropdown-divider");
150                                                        break;
151                                                case "h6": 
152                                                        itemTag.addClass("dropdown-header");
153                                                        break;
154                                                }
155                                                List<Object> children = (List<Object>) itemTag.getData(AppPackage.Literals.LABEL__CHILDREN);
156                                                if (children != null) {
157                                                        addDropdownItems(dropdownMenu, children);
158                                                }                                       
159                                        }
160                                }
161                        }
162                        
163                };
164                
165                MapCompoundSupplierFactory<EStructuralFeature, Object> featuresFactory = new MapCompoundSupplierFactory<>("Features");
166                
167                M semanticElement = getTarget();
168                org.nasdanika.html.model.bootstrap.Modal help = semanticElement.getHelp();
169                if (help != null) {
170                        featuresFactory.put(AppPackage.Literals.LABEL__HELP, EObjectAdaptable.adaptToSupplierFactoryNonNull(help, Tag.class));
171                }
172                
173                List<EObject> children = semanticElement.getChildren();
174                if (!children.isEmpty()) {
175                        featuresFactory.put(AppPackage.Literals.LABEL__CHILDREN, new ListCompoundSupplierFactory<>("Children", EObjectAdaptable.adaptToSupplierFactoryNonNull(children, Object.class)));                        
176                }
177                
178                SupplierFactory<Tag> textSupplierFactory = this::createTextAndIconSupplier;
179                featuresFactory.put(AppPackage.Literals.LABEL__TEXT, textSupplierFactory);
180                
181                return featuresFactory.create(context).then(labelFunction);
182        }

184        /**
185         * Override in Link to create a link.
186         * @param context
187         * @param progressMonitor
188         * @return
189         * @throws Exception 
190         */
191        protected Supplier<Tag> createTextAndIconSupplier(Context context) throws Exception {
192                Function<Tag,Tag> textAndIconFunction = new Function<Tag, Tag>() {
193                                
194                        @Override
195                        public double size() {
196                                return 1;
197                        }
198
199                        @Override
200                        public String name() {
201                                return "Label text and icon";
202                        }
203
204                        @Override
205                        public Tag execute(Tag modal, ProgressMonitor progressMonitor) throws Exception {
206                                BootstrapFactory bootstrapFactory = context.get(BootstrapFactory.class, BootstrapFactory.INSTANCE);
207                                HTMLFactory htmlFactory = context.get(HTMLFactory.class, HTMLFactory.INSTANCE);
208                                Label semanticElement = getTarget();
209                                
210                                String text = context.interpolateToString(getTarget().getText());
211                                for (EObject semanticAncestor = semanticElement.eContainer(); semanticAncestor != null; semanticAncestor = semanticAncestor.eContainer()) {
212                                        if (semanticAncestor instanceof NavigationPanel) {
213                                                int labelTrimLength = ((NavigationPanel) semanticAncestor).getLabelTrimLength();
214                                                if (labelTrimLength > 0 && text.length() > labelTrimLength) {
215                                                        text = text.substring(0, labelTrimLength - 3) + "...";
216                                                }
217                                                break;
218                                        }                                               
219                                }
220                                
221                                String icon = semanticElement.getIcon();
222                                Tag iconTag = null;
223                                if (icon != null) {
224                                        if (icon.contains("/")) {
225                                                // Image
226                                                iconTag = htmlFactory.tag(TagName.img).attribute("src", icon); //.style().height("1em");
227                                        } else {
228                                                // Class
229                                                iconTag = htmlFactory.span().addClass(icon);
230                                        }
231                                }
232                                
233                                Tag ret = createTextAndIconTag(iconTag, text, modal, context, progressMonitor);
234
235                                Color color = semanticElement.getColor();                               
236                                if (color != null) {
237                                        bootstrapFactory.wrap(ret).text().color(color);
238                                }       
239                                                                                                
240                                return ret;
241                        }
242                        
243                };
244                
245                return getModalFactory().create(context).then(textAndIconFunction);                             
246        }
247        
248        /**
249         * @param icon Icon
250         * @param text text
251         * @param modal Modal for binding to links, not applicalbe to labels.
252         * @param context
253         * @param progressMonitor
254         * @return
255         */
256        protected Tag createTextAndIconTag(Tag icon, String text, Tag modal, Context context, ProgressMonitor progressMonitor) {
257                HTMLFactory htmlFactory = context.get(HTMLFactory.class, HTMLFactory.INSTANCE);
258                TagName tagName;
259                if (getTarget().getChildren().isEmpty()) {
260                        tagName = TagName.span; // A regular label.
261                } else {
262                        // Drop-down activator or drop-down header if contained in a label
263                        tagName = getTarget().eContainer() instanceof Label ? TagName.h6 : TagName.a;
264                }
265                
266                if (Util.isBlank(text)) {
267                        if (icon == null) {
268                                // Drop-down divider
269                                return htmlFactory.div();
270                        }
271                        
272                        // Wrap into a header for drop-down headers
273                        return tagName == TagName.h6 ? htmlFactory.tag(tagName, icon) : icon;
274                }
275                if (icon == null) {
276                        return htmlFactory.tag(tagName, text);
277                }
278                
279                icon.addClass("nsd-app-label-icon");
280                return htmlFactory.tag(tagName, icon, text);    
281        }
282        
283        /**
284         * For links. Placing here to simplify code. This implementation returns empty supplier.
285         * @return
286         */
287        protected SupplierFactory<Tag> getModalFactory() {
288                return SupplierFactory.empty();
289        }
