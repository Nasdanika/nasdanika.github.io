001package org.nasdanika.html.model.app.gen;
002
003import java.util.List;
004import java.util.Map;
005import java.util.Map.Entry;
006
007import org.eclipse.emf.common.notify.impl.AdapterImpl;
008import org.eclipse.emf.ecore.EObject;
009import org.eclipse.emf.ecore.EStructuralFeature;
010import org.nasdanika.common.Context;
011import org.nasdanika.common.Function;
012import org.nasdanika.common.ListCompoundSupplierFactory;
013import org.nasdanika.common.MapCompoundSupplierFactory;
014import org.nasdanika.common.ProgressMonitor;
015import org.nasdanika.common.Supplier;
016import org.nasdanika.common.SupplierFactory;
017import org.nasdanika.common.Util;
018import org.nasdanika.emf.EObjectAdaptable;
019import org.nasdanika.exec.content.Text;
020import org.nasdanika.html.Tag;
021import org.nasdanika.html.jstree.JsTreeFactory;
022import org.nasdanika.html.jstree.JsTreeNode;
023import org.nasdanika.html.model.app.AppPackage;
024import org.nasdanika.html.model.app.Label;
025import org.nasdanika.html.model.app.NavigationPanel;
026
027public class LabelJsTreeNodeSupplierFactoryAdapter<M extends Label> extends AdapterImpl implements SupplierFactory<JsTreeNode> {
028        
029        public LabelJsTreeNodeSupplierFactoryAdapter(M label) {
030                setTarget(label);
031        }
032        
033        @SuppressWarnings("unchecked")
034        @Override
035        public M getTarget() {
036                return (M) super.getTarget();
037        }
038
039        @Override
040        public Supplier<JsTreeNode> create(Context context) throws Exception {
041                Function<Map<EStructuralFeature, Object>, JsTreeNode> jsTreeNodeFunction = new Function<Map<EStructuralFeature, Object>, JsTreeNode>() {
042
043                        @Override
044                        public double size() {
045                                return 1;
046                        }
047
048                        @Override
049                        public String name() {
050                                return "jsTree Node";
051                        }
052
053                        @Override
054                        public JsTreeNode execute(Map<EStructuralFeature, Object> features, ProgressMonitor progressMonitor) throws Exception {
055                                JsTreeFactory jsTreeFactory = context.get(JsTreeFactory.class, JsTreeFactory.INSTANCE);
056                                JsTreeNode ret = jsTreeFactory.jsTreeNode();
057                                M semanticElement = getTarget();
058                                
059                                String icon = context.interpolateToString(semanticElement.getIcon());
060                                if (!Util.isBlank(icon)) {
061                                        ret.icon(icon);
062                                }
063                                String text = context.interpolateToString(semanticElement.getText());
064                                if (!Util.isBlank(text)) {
065                                        for (EObject semanticAncestor = semanticElement.eContainer(); semanticAncestor != null; semanticAncestor = semanticAncestor.eContainer()) {
066                                                if (semanticAncestor instanceof NavigationPanel) {
067                                                        int labelTrimLength = ((NavigationPanel) semanticAncestor).getLabelTrimLength();
068                                                        if (labelTrimLength > 0 && text.length() > labelTrimLength) {
069                                                                ret.anchorAttribute("title", text);
070                                                                text = text.substring(0, labelTrimLength - 3) + "...";
071                                                        }
072                                                        break;
073                                                }                                               
074                                        }
075                                        
076                                        ret.text(text);
077                                }
078                                String tooltip = context.interpolateToString(semanticElement.getTooltip());
079                                if (!Util.isBlank(tooltip)) {
080                                        ret.anchorAttribute("title", tooltip);
081                                }
082                                for (Entry<String, EObject> attr: semanticElement.getAttributes().entrySet()) {
083                                        String attrName = attr.getKey();
084                                        if (attrName.startsWith("data-nsd-")) { // Pass through Nasdanika attributes 
085                                                EObject attrValue = attr.getValue();
086                                                if (attrValue instanceof Text) { // Text only
087                                                        ret.attribute(attrName, ((Text) attrValue).getContent());                                                       
088                                                }
089                                        }
090                                }
091                                ret.id(jsTreeFactory.getHTMLFactory().nextId()); // TODO - label ID 
092                                ret.setData(semanticElement);
093                                
094                                /**
095                                 * Storing feature data to use downstream.
096                                 */
097                                for (Entry<EStructuralFeature, Object> fe: features.entrySet()) {
098                                        ret.setData(fe.getKey(), fe.getValue());
099                                }
100                                
101                                ret.selected(semanticElement.isActive());
102                                
103                                Tag help = (Tag) features.get(AppPackage.Literals.LABEL__HELP);
104                                if (help != null) {
105                                        // TODO - Context menu item.
106                                }
107                                
108                                @SuppressWarnings("unchecked")
109                                List<JsTreeNode> children = (List<JsTreeNode>) features.get(AppPackage.Literals.LABEL__CHILDREN);
110                                if (children != null) {
111                                        ret.children().addAll(children);
112                                }
113                                
114                                if (ret.children().isEmpty()) {
115                                        ret.attribute("type", "leaf"); // Set node type to leaf if there are no children.
116                                }
117                                                                
118                                return ret;
119                        }
120                        
121                };
122                
123                MapCompoundSupplierFactory<EStructuralFeature, Object> featuresFactory = new MapCompoundSupplierFactory<>("Features");
124                
125                M semanticElement = getTarget();
126                org.nasdanika.html.model.bootstrap.Modal help = semanticElement.getHelp();
127                if (help != null) {
128                        featuresFactory.put(AppPackage.Literals.LABEL__HELP, EObjectAdaptable.adaptToSupplierFactoryNonNull(help, Tag.class));
129                }
130                
131                List<EObject> children = semanticElement.getChildren();
132                if (!children.isEmpty()) {
133                        featuresFactory.put(AppPackage.Literals.LABEL__CHILDREN, new ListCompoundSupplierFactory<>("Children", EObjectAdaptable.adaptToSupplierFactoryNonNull(children, JsTreeNode.class)));                    
134                }
135                
136                SupplierFactory<Tag> modalFactory = getModalFactory();
137                if (modalFactory != null) {
138                        featuresFactory.put(AppPackage.Literals.LINK__MODAL, modalFactory);
139                }
140                                
141                return featuresFactory.create(context).then(jsTreeNodeFunction);
142        }
143        
144        /**
145         * For links. Placing here to simplify code. This implementation returns empty supplier.
146         * @return
147         */
148        protected SupplierFactory<Tag> getModalFactory() {
149                return null;
150        }
151        
152}
