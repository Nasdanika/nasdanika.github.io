041        public static final String CLEAR_STATE_FILTER = "tree.state.filter = function(state) { delete state.core.selected; return state; };";
042        private static final String SEARCH_FILTER = " tree.search.search_callback = function(searchStr, node) { if (typeof window.nsdJsTreeSearchCallback === 'function') return window.nsdJsTreeSearchCallback(searchStr, node); var sf = new $.vakata.search(searchStr, true, { caseSensitive : false, fuzzy : false }); return sf.search(node.text).isMatch; };";
043        private static final String SEARCH_INPUT_SUFFIX = "_searchInput";
044        private static final String SEARCH_INPUT_TOOLTIP = "Full-text search. You can use wildcards, e.g. 'foo*' or 'f*o'; title or content fields, e.g. 'title:foo* bar'; boosts, e.g. 'foo^10 bar'; fuzzy matches, e.g. 'foo~1'; and term presence, e.g. '+foo bar -baz'";

050        private Function<BiSupplier<HTMLElement<?>, List<JsTreeNode>>, HTMLElement<?>> createTreeFunction(NavigationPanelStyle effectiveStyle, Context context) {
051                return new Function<BiSupplier<HTMLElement<?>, List<JsTreeNode>>, HTMLElement<?>>() {
052                        
053                        @Override
054                        public double size() {
055                                return 1;
056                        }
057                        
058                        @Override
059                        public String name() {
060                                return "JsTree";
061                        }
062                        
063                        @Override
064                        public HTMLElement<?> execute(BiSupplier<HTMLElement<?>, List<JsTreeNode>> input, ProgressMonitor progressMonitor) throws Exception {
065                                Tag ret = (Tag) input.getFirst();
066                                
067                                Tag panel;
068                                if (getTarget().isCollapsible()) {
069                                        Tag triggerDiv = ret.getFactory().div().addClass("nsd-collapse-panel", "bg-light", "text-center");
070                                        ret.content(triggerDiv);
071                                        HTMLFactory htmlFactory = ret.getFactory();
072                                        panel = htmlFactory.div();
073                                        panel.id("nsd-nav-panel-" + getTarget().getUuid() + "-collapsible");
074                                        
075                                        Tag iTag = htmlFactory.tag(TagName.i).addClass("fa").attribute("aria-hidden", "true");
076
077                                        Tag collapsibleTrigger = htmlFactory.span(iTag)
078                                                        .addClass("nsd-collapsible-trigger")
079                                                        .attribute("data-toggle", "collapse")
080                                                        .attribute("data-target", "#" + panel.getId())
081                                                        .attribute("aria-expanded", "true")
082                                                        .attribute("aria-controls", panel.getId())
083                                                        .attribute("title", "Toggle panel")
084                                                        .id(panel.getId()+"-trigger");
085                                        
086                                        triggerDiv.content(collapsibleTrigger);
087                                        panel.addClass("nsd-collapsible", "collapse", "show");
088                                        panel.style("clear", "right");
089
090                                        ret.content(panel);
091                                } else {
092                                        panel = ret;
093                                }
094                                
095                                JsTreeFactory jsTreeFactory = context.get(JsTreeFactory.class, JsTreeFactory.INSTANCE);
096                                HTMLFactory htmlFactory = jsTreeFactory.getHTMLFactory();
097                                String treeId = getTarget().getId();
098                                if (org.nasdanika.common.Util.isBlank(treeId)) {
099                                        treeId = htmlFactory.nextId();
100                                }
101                                
102                                String searchInputId = null;
103                                if (effectiveStyle == NavigationPanelStyle.SEARCHABLE_TREE) {
104                                        Input searchInput = htmlFactory.input(InputType.text);
105                                        searchInput.addClass("form-control", "mt-1");
106                                        searchInputId = treeId + SEARCH_INPUT_SUFFIX;
107                                        searchInput.id(searchInputId);
108                                        searchInput.attribute("title", SEARCH_INPUT_TOOLTIP);
109                                        panel.accept(searchInput);
110                                }
111                                
112                                Tag container = htmlFactory.div().id(treeId);
113                                panel.accept(container);
114                                
115                                JSONObject jsTree = jsTreeFactory.buildJsTree(input.getSecond());
116
117                                // Configures jsTree. TODO - from the model.
118                                List<String> plugins = new ArrayList<>();
119                                plugins.add("state");
120                                plugins.add("type");
121                                if (effectiveStyle == NavigationPanelStyle.SEARCHABLE_TREE) {
122                                        plugins.add("search");
123                                        JSONObject searchConfig = new JSONObject();
124                                        searchConfig.put("show_only_matches", true);
125                                        searchConfig.put("show_only_matches_children", true);
126                                        jsTree.put("search", searchConfig);
127                                        
128                                        // TODO - use lunrjs search.js
129                                }
130                                
131                                jsTree.put("plugins", plugins);                 
132                                jsTree.put("state", Collections.singletonMap("key", treeId));
133                                jsTree.put("types", Collections.singletonMap("leaf", Collections.singletonMap("icon", "jstree-file"))); // File leaf icon.
134                                
135                                // Deletes selection from state
136                                String filter = CLEAR_STATE_FILTER;
137                                
138                                if (effectiveStyle == NavigationPanelStyle.SEARCHABLE_TREE) {
139                                        filter += " " + SEARCH_FILTER;
140                                }
141                                
142                                Tag script = jsTreeFactory.bind(container, jsTree, filter, searchInputId == null ? null : "#" + searchInputId);
143                                panel.accept(script);
144                                
145                                return ret;
146                        }
147                                                
148                };
149        }

151        private Function<BiSupplier<HTMLElement<?>, Map<Class<?>, Object>>, HTMLElement<?>> createCardsFunction(Context context) {
152                return new Function<BiSupplier<HTMLElement<?>, Map<Class<?>, Object>>, HTMLElement<?>>() {
153                        
154                        @Override
155                        public double size() {
156                                return 1;
157                        }
158                        
159                        @Override
160                        public String name() {
161                                return "Cards";
162                        }

164                        @SuppressWarnings("unchecked")
165                        @Override
166                        public HTMLElement<?> execute(BiSupplier<HTMLElement<?>, Map<Class<?>, Object>> input, ProgressMonitor progressMonitor) throws Exception {
167                                Tag ret = (Tag) input.getFirst();
168                                Tag panel;
169                                if (getTarget().isCollapsible()) {
170                                        HTMLFactory htmlFactory = ret.getFactory();
171                                        Tag triggerDiv = htmlFactory.div().addClass("nsd-collapse-panel", "bg-light", "text-center");
172                                        ret.content(triggerDiv);
173                                        panel = htmlFactory.div();
174                                        panel.id("nsd-nav-panel-" + getTarget().getUuid() + "-collapsible");
175                                        
176                                        Tag iTag = htmlFactory.tag(TagName.i).addClass("fa").attribute("aria-hidden", "true");
177
178                                        Tag collapsibleTrigger = htmlFactory.span(iTag)
179                                                        .addClass("nsd-collapsible-trigger")
180                                                        .attribute("data-toggle", "collapse")
181                                                        .attribute("data-target", "#" + panel.getId())
182                                                        .attribute("aria-expanded", "true")
183                                                        .attribute("aria-controls", panel.getId())
184                                                        .attribute("title", "Toggle panel")                                                     
185                                                        .id(panel.getId()+"-trigger");
186                                        
187                                        triggerDiv.content(collapsibleTrigger);
188                                        panel.addClass("nsd-collapsible", "collapse", "show");
189                                        panel.style("clear", "right");
190
191                                        ret.content(panel);
192                                } else {
193                                        panel = ret;
194                                }
195                                
196                                List<Object> items = (List<Object>) input.getSecond().get(Object.class);
197                                
198                                JsTreeFactory jsTreeFactory = context.get(JsTreeFactory.class, JsTreeFactory.INSTANCE);
199                                String treeId = getTarget().getId();
200                                HTMLFactory htmlFactory = jsTreeFactory.getHTMLFactory();
201                                
202                                Tag currentActionGroup = null; // Add items without children to an action group
203                                for (Object item: items) {
204                                        if (item instanceof Tag) {
205                                                Tag itemTag = (Tag) item;
206                                                Object data = itemTag.getData();
207                                                if (data instanceof Label) {
208                                                        Label semanticElement = (Label) data; 
209                                                        if (semanticElement.getChildren().isEmpty()) {
210                                                                if (currentActionGroup == null) {
211                                                                        currentActionGroup = htmlFactory.div().addClass("list-group", "list-group-flush");
212                                                                }
213                                                                itemTag.addClass("list-group-item", "list-group-action");
214                                                                itemTag.addClassConditional(semanticElement.isActive(), "active");
215                                                                itemTag.addClassConditional(semanticElement.isDisabled(), "disabled");
216                                                                currentActionGroup.accept(itemTag);
217                                                        } else {
218                                                                if (currentActionGroup != null) {
219                                                                        panel.accept(currentActionGroup);
220                                                                        currentActionGroup = null;
221                                                                }
222                                                                
223                                                                Card card = context.get(BootstrapFactory.class, BootstrapFactory.INSTANCE).card();
224                                                                panel.accept(card);
225                                                                TagBootstrapElement cardHeader = card.getHeader();
226                                                                Color color = semanticElement.getColor();
227                                                                if (color != null) {
228                                                                        card.border(color);
229                                                                        cardHeader.background(color);
230                                                                }
231                                                                Tag title = (Tag) itemTag.getContent().get(0);
232                                                                
233                                                                // Remove dropdown classes and attributes
234                                                                title
235                                                                        .removeClass("nav-link", "dropdown-toggle")
236                                                                        .attribute("role", null)
237                                                                        .attribute("data-toggle", null);                                                                
238                                                                
239                                                                if (color != null) {
240                                                                        title.removeClass("text-" + color.code);
241                                                                }
242                                                                
243                                                                Tag cardHeaderTag = cardHeader.toHTMLElement();
244                                                                Tag collapsible;
245                                                                // Tree or items
246                                                                TagBootstrapElement cardBody = card.getBody();                                                                                                                          
247                                                                if (depth(semanticElement.getChildren()) == 1) {
248                                                                        Tag childrenActionGroup = htmlFactory.div().addClass("list-group", "list-group-flush");
249                                                                        for (Object itemChild: (List<Object>) itemTag.getData(AppPackage.Literals.LABEL__CHILDREN)) {
250                                                                                if (itemChild instanceof Tag) {
251                                                                                        Tag itemChildTag = (Tag) itemChild;
252                                                                                        Label childSemanticElement = (Label) itemChildTag.getData();
253                                                                                        itemChildTag.addClass("list-group-item", "list-group-action").removeClass("dropdown-item");
254                                                                                        itemChildTag.addClassConditional(childSemanticElement.isActive(), "active");
255                                                                                        itemChildTag.addClassConditional(childSemanticElement.isDisabled(), "disabled");
256                                                                                        childrenActionGroup.accept(itemChildTag);
257                                                                                } else {
258                                                                                        cardBody.toHTMLElement().accept(itemChild);
259                                                                                }
260                                                                        }
261                                                                        card.toHTMLElement().accept(childrenActionGroup);
262                                                                        collapsible = childrenActionGroup;
263                                                                } else {                                                                        
264                                                                        cardBody.padding().all(Breakpoint.DEFAULT, Size.S1);
265                                                                        collapsible = cardBody.toHTMLElement();
266                                                                        List<JsTreeNode> nodes = (List<JsTreeNode>) input.getSecond().get(JsTreeNode.class);
267                                                                        for (JsTreeNode node: nodes) {
268                                                                                if (node.getData() == semanticElement) {
269                                                                                        String nodeTreeId;
270                                                                                        if (org.nasdanika.common.Util.isBlank(treeId)) {
271                                                                                                nodeTreeId = "nsd-nav-tree-" + semanticElement.getId();
272                                                                                        } else {
273                                                                                                nodeTreeId = treeId + "-" + semanticElement.getId();
274                                                                                        }
275                                                                                        
276                                                                                        boolean isSearchable = treeSize(node.children()) >= getSearchThreshold();                                                                       
277                                                                                        String searchInputId = null;
278                                                                                        if (isSearchable) {
279                                                                                                Input searchInput = htmlFactory.input(InputType.text);
280                                                                                                searchInput.addClass("form-control");
281                                                                                                searchInputId = cardBody.getFactory().getHTMLFactory().nextId() + SEARCH_INPUT_SUFFIX;
282                                                                                                searchInput.id(searchInputId);
283                                                                                                searchInput.attribute("title", SEARCH_INPUT_TOOLTIP);                                                                                           
284                                                                                                cardBody.toHTMLElement().accept(searchInput);
285                                                                                        }
286                                                                                        
287                                                                                        Tag container = htmlFactory.div().id(nodeTreeId);
288                                                                                        cardBody.toHTMLElement().accept(container);
289                                                                                        
290                                                                                        JSONObject jsTree = jsTreeFactory.buildJsTree(node.children());
291                                                        
292                                                                                        // Configures jsTree. TODO - from the model.
293                                                                                        List<String> plugins = new ArrayList<>();
294                                                                                        plugins.add("state");
295                                                                                        plugins.add("type");
296                                                                                        if (isSearchable) {
297                                                                                                plugins.add("search");
298                                                                                                JSONObject searchConfig = new JSONObject();
299                                                                                                searchConfig.put("show_only_matches", true);
300                                                                                                searchConfig.put("show_only_matches_children", true);
301                                                                                                jsTree.put("search", searchConfig);
302                                                                                                
303                                                                                                // TODO - use lunrjs search.js
304                                                                                        }
305                                                                                        
306                                                                                        jsTree.put("plugins", plugins);                 
307                                                                                        jsTree.put("state", Collections.singletonMap("key", treeId));
308                                                                                        jsTree.put("types", Collections.singletonMap("leaf", Collections.singletonMap("icon", "jstree-file"))); // File leaf icon.
309                                                                                        
310                                                                                        // Deletes selection from state
311                                                                                        String filter = CLEAR_STATE_FILTER;
312                                                                                        
313                                                                                        if (isSearchable) {
314                                                                                                filter += " " + SEARCH_FILTER;
315                                                                                        }
316                                                                                        
317                                                                                        Tag script = jsTreeFactory.bind(container, jsTree, filter, searchInputId == null ? null : "#" + searchInputId);
318                                                                                        panel.accept(script);
319                                                                                        break;
320                                                                                }
321                                                                        }
322                                                                }
323
324                                                                // Collapse button
325                                                                if (getTarget().getStyle() == NavigationPanelStyle.COLLAPSIBLE_CARDS) {
326                                                                        Tag iTag = htmlFactory.tag(TagName.i).addClass("fa").attribute("aria-hidden", "true");
327                                                                        collapsible.id(semanticElement.getId()+"-collapsible");
328
329                                                                        Tag collapsibleTrigger = htmlFactory.span(iTag)
330                                                                                        .addClass("nsd-collapsible-trigger")
331                                                                                        .attribute("data-toggle", "collapse")
332                                                                                        .attribute("data-target", "#" + collapsible.getId())
333                                                                                        .attribute("aria-expanded", "true")
334                                                                                        .attribute("aria-controls", collapsible.getId())
335                                                                                        .attribute("title", "Toggle card")                                                                                      
336                                                                                        .id(semanticElement.getId()+"-collapsible-trigger");
337                                                                        
338                                                                        cardHeaderTag.content(collapsibleTrigger);
339                                                                        collapsible.addClass("nsd-collapsible", "collapse", "show");
340                                                                }
341                                                                
342                                                                cardHeaderTag.content(title);                                                           
343                                                        }
344                                                } else {
345                                                        panel.accept(item);
346                                                }                                               
347                                        } else {
348                                                panel.accept(item);
349                                        }
350                                }
351                                if (currentActionGroup != null) {
352                                        panel.accept(currentActionGroup);
353                                }
354                                
355                                return ret;
356                        }
357                };
358        }

360        protected int getSearchThreshold() {
361                return getTarget().getJsTreeSearchThreshold(); 
362        }
363                
364        private static int depth(Collection<EObject> items) {
365                int ret = 0;
366                for (EObject item: items) {
367                        if (item instanceof Label) {
368                                ret = Math.max(ret, depth(((Label) item).getChildren()) + 1);
369                        }
370                }
371                return ret;
372        }
373        
374        private static int treeSize(Collection<JsTreeNode> nodes) {
375                int ret = 0;
376                for (JsTreeNode node: nodes) {
377                        ++ret;
378                        ret += treeSize(node.children());
379                }
380                return ret;
381        }       
382        
383        private static int size(Collection<EObject> items) {
384                int ret = 0;
385                for (EObject item: items) {
386                        ++ret;
387                        if (item instanceof Label) {
388                                ret += size(((Label) item).getChildren());
389                        }
390                }
391                return ret;
392        }

394        /**
395         * Adapts items to {@link JsTreeNode} suppliers.
396         */
397        @Override
398        protected Function<HTMLElement<?>, HTMLElement<?>> createConfigureFunction(Context context) throws Exception {
399                NavigationPanel semanticElement = getTarget();
400                NavigationPanelStyle style = semanticElement.getStyle();
401                
402                EList<EObject> items = semanticElement.getItems();
403                if (style == NavigationPanelStyle.AUTO) {
404                        style = depth(items) > 2 ?  size(items) >= getSearchThreshold() ? NavigationPanelStyle.SEARCHABLE_TREE : NavigationPanelStyle.TREE : NavigationPanelStyle.CARDS; 
405                }
406                
407                ListCompoundSupplierFactory<JsTreeNode> nodesFactory = new ListCompoundSupplierFactory<>("Nodes", EObjectAdaptable.adaptToSupplierFactoryNonNull(items, JsTreeNode.class));
408                if (style == NavigationPanelStyle.TREE || style == NavigationPanelStyle.SEARCHABLE_TREE) {
409                        Function<HTMLElement<?>, BiSupplier<HTMLElement<?>, List<JsTreeNode>>> nodesFunction = nodesFactory.<HTMLElement<?>>asFunctionFactory().create(context);
410                        return super.createConfigureFunction(context).then(nodesFunction).then(createTreeFunction(style, context));
411                }
412                
413                ListCompoundSupplierFactory<Object> tagsFactory = new ListCompoundSupplierFactory<>("Tags", EObjectAdaptable.adaptToSupplierFactoryNonNull(items, Object.class));
414                
415                MapCompoundSupplierFactory<Class<?>, Object> tagsAndNodesSupplierFactory = new MapCompoundSupplierFactory<>("Tags and nodes");
416                tagsAndNodesSupplierFactory.put(JsTreeNode.class, nodesFactory);
417                tagsAndNodesSupplierFactory.put(Object.class, tagsFactory);
418                Function<HTMLElement<?>, BiSupplier<HTMLElement<?>, Map<Class<?>, Object>>> tagsAndNodesFunction = tagsAndNodesSupplierFactory.<HTMLElement<?>>asFunctionFactory().create(context);
419                
420                return super.createConfigureFunction(context).then(tagsAndNodesFunction).then(createCardsFunction(context));
421        }
