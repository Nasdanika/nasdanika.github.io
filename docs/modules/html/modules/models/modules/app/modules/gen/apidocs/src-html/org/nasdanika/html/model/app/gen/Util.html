080        /**
081         * Creates Bootstrap Navs from a list of items.
082         * @param items
083         * @param bootstrapFactory
084         * @return
085         */
086        public static Tag navs(List<Object> items, BootstrapFactory bootstrapFactory) {
087                HTMLFactory htmlFactory = bootstrapFactory.getHTMLFactory();
088                Tag navs = htmlFactory.tag(TagName.ul).addClass("nav");
089                for (Object item: items) {
090                        Tag li = htmlFactory.tag(TagName.li, item).addClass("nav-item");
091                        navs.accept(li);
092                        if (item instanceof Tag) {
093                                Tag itemTag = (Tag) item;
094                                if (TagName.a.name().equalsIgnoreCase(itemTag.getTagName())) {
095                                        itemTag.addClass("nav-link");
096                                        Object data = itemTag.getData();
097                                        if (data instanceof Item) {
098                                                Item itemData = (Item) data;
099                                                if (itemData.isActive()) {
100                                                        itemTag.addClass("active");
101                                                } else if (itemData.isDisabled()) {
102                                                        itemTag.addClass("disabled");
103                                                }
104                                        }
105                                } else if (TagName.span.name().equalsIgnoreCase(itemTag.getTagName())) {
106                                        itemTag.addClass("navbar-text");
107                                }
108                        }
109                }
110                return navs;
111        }

113        /**
114         * Creates Bootstrap Navs from a list of items.
115         * @param items
116         * @param bootstrapFactory
117         * @return
118         */
119        public static Tag navbar(
120                        Tag brand, 
121                        List<Object> items,
122                        Breakpoint expand, 
123                        boolean dark,
124                        Color background,
125                        BootstrapFactory bootstrapFactory) {
126                HTMLFactory htmlFactory = bootstrapFactory.getHTMLFactory();
127                Tag navbar = htmlFactory.tag(TagName.ul).addClass("navbar", dark ? "navbar-dark" : "navbar-light");
128                if (expand != null) {
129                        navbar.addClass("navbar-expand-" + expand.code);
130                }
131                if (background != null) {
132                        navbar.addClass("bg-" + background.code);
133                }
134                if (brand != null) {
135                        brand.addClass("navbar-brand");
136                        navbar.accept(brand);
137                }
138                String navbarContentId = htmlFactory.nextId();
139                Button toggler = htmlFactory.button(htmlFactory.span().addClass("navbar-toggler-icon"));
140                toggler
141                        .addClass("navbar-toggler")
142                        .attribute("type", "button")
143                        .attribute("data-toggle", "collapse")
144                        .attribute("data-target", "#" + navbarContentId)
145                        .attribute("aria-expanded", "false")
146                        .attribute("aria-label", "Toggle navigation");
147                navbar.accept(toggler);
148                
149                Tag content = htmlFactory.div().addClass("collapse", "navbar-collapse").id(navbarContentId);
150                navbar.accept(content);
151                if (items != null && !items.isEmpty()) {
152                        Tag navs = navs(items, bootstrapFactory);
153                        navs.removeClass("nav").addClass("navbar-nav", "mr-auto");
154                        content.accept(navs);
155                }
156                return navbar;
157        }

160         * Generates application site from an {@link Action} model root. 
161         * @param root Root action to generate header and footer. Can be null.
162         * @param pageTemplate Page template.
163         * @param context Context used for uri resolution - interpolation of action locations and names. can be null.
164         * @param container Receiver of generated resources.
165         * @param progressMonitor Progress monitor.
166         */
167        public static void generateSite(
168                        Action root, 
169                        org.nasdanika.html.model.bootstrap.Page pageTemplate,
170                        Container container,
171                        Context context,
172                        ProgressMonitor progressMonitor) {
173        
174                Action principal = null;
175                EList<EObject> rootChildren = root.getChildren();
176                if (rootChildren.size() > 0) {
177                        EObject firstChild = root.getChildren().get(0);
178                        if (firstChild instanceof Action) {
179                                principal = (Action) firstChild;
180                        } else if (firstChild instanceof ActionReference) {
181                                principal = ((ActionReference) firstChild).getTarget();
182                        }
183                }
184                
185                generateSite(root, principal, root, Collections.emptyList(), pageTemplate, container, context, progressMonitor);
186        }

189         * Generates application site from an {@link Action} model using a random base URI and a caching URI resolver created from the argument context with base-uri injected.
190         * @param root Root action to generate header and footer. Can be null.
191         * @param principal Principal action to generate navigation bar and navigation panel. Can be null.
192         * @param activeAction Active action to generate the content panel.
193         * @param actionPath Action path to show in breadcrumbs.
194         * @param pageTemplate Page template.
195         * @param context Context used for uri resolution - interpolation of action locations and names. can be null.
196         * @param container Receiver of generated resources.
197         * @param progressMonitor Progress monitor.
198         */
199        public static void generateSite(
200                        Action root, 
201                        Action principal, 
202                        Action activeAction, 
203                        List<Action> actionPath,
204                        org.nasdanika.html.model.bootstrap.Page pageTemplate,
205                        Container container,
206                        Context context,
207                        ProgressMonitor progressMonitor) {
208        
209                if (context == null) {
210                        context = Context.EMPTY_CONTEXT;
211                }
212                if (context.get(Context.BASE_URI_PROPERTY) == null) {
213                        context = context.fork();
214                        ((MutableContext) context).put(Context.BASE_URI_PROPERTY, URI.createURI("temp://" + UUID.randomUUID() + "/" + UUID.randomUUID() + "/"));
215                }
216                BiFunction<Action, URI, URI> uriResolver = uriResolver(root, context);          
217                generateSite(
218                                root, 
219                                principal, 
220                                activeAction,
221                                actionPath,
222                                pageTemplate,
223                                uriResolver, 
224                                (URI) context.get(Context.BASE_URI_PROPERTY), 
225                                container, 
226                                progressMonitor);
227        }

229        /**
230         * Generates application site from an {@link Action} model
231         * @param root Root action to generate header and footer. Can be null.
232         * @param principal Principal action to generate navigation bar and navigation panel. Can be null.
233         * @param activeAction Active action to generate the content panel.
234         * @param pageTemplate Page template.
235         * @param uriResolver Resolves {@link Action} to {@link URI}.
236         * @param baseURI Base URI for resolution, specifically to create relative URI's.
237         * @param container Receiver of generated resources.
238         * @param progressMonitor Progress monitor.
239         */
240        public static void generateSite(
241                        Action root, 
242                        Action principal, 
243                        Action activeAction, 
244                        List<Action> actionPath,
245                        org.nasdanika.html.model.bootstrap.Page pageTemplate,
246                        BiFunction<Action, URI, URI> uriResolver, 
247                        URI baseURI,
248                        Container container,
249                        ProgressMonitor progressMonitor) {
250                
251                if (activeAction.eContainmentFeature() != AppPackage.Literals.ACTION__SECTIONS) {
252                        URI uri = uriResolver.apply(activeAction, baseURI);                             
253                        if (uri != null && uri.isRelative()) {
254                                org.nasdanika.exec.resources.File file = container.getFile(uri.toString());
255                                org.nasdanika.html.model.bootstrap.Page bootstrapPage = EcoreUtil.copy(pageTemplate);
256                                bootstrapPage.setName(activeAction.getText());
257                                if (bootstrapPage.getBody().isEmpty()) {
258                                        bootstrapPage.getBody().add(AppFactory.eINSTANCE.createPage());
259                                }
260                                for (EObject be: bootstrapPage.getBody()) {
261                                        if (be instanceof org.nasdanika.html.model.app.Page) {
262                                                buildAppPage(root, principal, activeAction, actionPath, (org.nasdanika.html.model.app.Page) be, uriResolver, progressMonitor);                                          
263                                        }
264                                }
265                                file.getContents().add(bootstrapPage);
266                                
267                                for (org.nasdanika.exec.resources.Resource res: activeAction.getResources()) {
268                                        ((Container) file.eContainer()).getContents().add(EcoreUtil.copy(res));                                 
269                                }                               
270                        }
271                }
272                
273                List<Action> subActionPath = new ArrayList<>(actionPath);
274                if (activeAction != root && activeAction != principal) {
275                        subActionPath.add(activeAction);
276                }
277                
278                for (EObject child: resolveActionReferences(activeAction.getChildren())) {
279                        if (child instanceof Action) {
280                                generateSite(root, principal, (Action) child, subActionPath, pageTemplate, uriResolver, baseURI, container, progressMonitor);
281                        }
282                }
283                
284                for (Action section: activeAction.getSections()) {
285                        generateSite(root, principal, (Action) section, subActionPath, pageTemplate, uriResolver, baseURI, container, progressMonitor);
286                }
287                                
288                for (EObject navigation: resolveActionReferences(activeAction.getNavigation())) {
289                        if (navigation instanceof Action) {
290                                generateSite(root, principal, (Action) navigation, subActionPath, pageTemplate, uriResolver, baseURI, container, progressMonitor);
291                        }
292                }
293                
294                for (Action anonymous: activeAction.getAnonymous()) {
295                        generateSite(root, principal, anonymous, subActionPath, pageTemplate, uriResolver, baseURI, container, progressMonitor);
296                }
297                
298                generateNavigationPanel(root, principal, actionPath, activeAction.getFloatLeftNavigation(), pageTemplate, uriResolver, baseURI, container, progressMonitor);
299                generateNavigationPanel(root, principal, actionPath, activeAction.getFloatRightNavigation(), pageTemplate, uriResolver, baseURI, container, progressMonitor);
300                generateNavigationPanel(root, principal, actionPath, activeAction.getLeftNavigation(), pageTemplate, uriResolver, baseURI, container, progressMonitor);
301                generateNavigationPanel(root, principal, actionPath, activeAction.getRightNavigation(), pageTemplate, uriResolver, baseURI, container, progressMonitor);
302        }

304        private static void buildAppPage(
305                        Action root, 
306                        Action principal, 
307                        Action activeAction, 
308                        List<Action> actionPath,
309                        org.nasdanika.html.model.app.Page appPage,
310                        BiFunction<Action, URI, URI> uriResolver, 
311                        ProgressMonitor progressMonitor) {
312                
313                if (root != null) {
314                        Label title = createLabel(root, activeAction, uriResolver, null, "header/title", false, false);
315                        EList<EObject> rootChildren = root.getChildren();
316                        if (title != null || rootChildren.size() > 1) {
317                                // Header
318                                Header header = appPage.getHeader();
319                                if (header == null) {
320                                        header = AppFactory.eINSTANCE.createHeader();
321                                        appPage.setHeader(header);
322                                }
323                                if (title != null) {
324                                        header.setTitle(title);
325                                }
326                                List<EObject> headerItems = header.getItems();
327                                rootChildren.listIterator(1).forEachRemaining(rac -> {
328                                        if (rac instanceof Action) {
329                                                headerItems.add(createLabel((Action) rac, activeAction, uriResolver, null, "header/navigation", true, false));
330                                        } else {
331                                                headerItems.add(EcoreUtil.copy(rac));
332                                        }
333                                });
334                        }
335                        
336                        List<EObject> rootNavigation = resolveActionReferences(root.getNavigation());
337                        if (!rootNavigation.isEmpty()) {
338                                Footer footer = appPage.getFooter();
339                                if (footer == null) {
340                                        footer = AppFactory.eINSTANCE.createFooter();
341                                        appPage.setFooter(footer);
342                                }
343                                EList<EObject> footerItems = footer.getItems();
344                                rootNavigation.forEach(ran -> {
345                                        if (ran instanceof Action) {
346                                                footerItems.add(createLabel((Action) ran, activeAction, uriResolver, null, "footer/navigation", true, false));
347                                        } else {
348                                                footerItems.add(EcoreUtil.copy(ran));
349                                        }
350                                });
351                                
352                        }
353                }
354                
355                if (principal != null) {
356                        // Navbar 
357                        Label brand = createLabel(principal, activeAction, uriResolver, null, "navbar/brand", false, false);
358                        List<EObject> principalNavigation = resolveActionReferences(principal.getNavigation());
359                        if (brand != null || !principalNavigation.isEmpty()) {
360                                NavigationBar navBar = appPage.getNavigationBar();
361                                if (navBar == null) {
362                                        navBar = AppFactory.eINSTANCE.createNavigationBar();
363                                        appPage.setNavigationBar(navBar);
364                                }
365                                if (brand != null) {
366                                        navBar.setBrand(brand);
367                                }
368                                EList<EObject> navBarItems = navBar.getItems();
369                                principalNavigation.forEach(principalNavigationElement -> {
370                                        if (principalNavigationElement instanceof Action) {
371                                                navBarItems.add(createLabel((Action) principalNavigationElement, activeAction, uriResolver, null, "navbar/item", true, false));
372                                        } else {
373                                                navBarItems.add(EcoreUtil.copy(principalNavigationElement));
374                                        }
375                                });
376                        }
377                        
378                        // Navigation panel
379                        List<EObject> principalChildren = resolveActionReferences(principal.getChildren());
380                        if (!principalChildren.isEmpty()) {
381                                NavigationPanel navPanel = appPage.getNavigationPanel();
382                                if (navPanel == null) {
383                                        navPanel = AppFactory.eINSTANCE.createNavigationPanel();
384                                        appPage.setNavigationPanel(navPanel);
385                                }
386                                if (isBlank(navPanel.getId())) {
387                                        navPanel.setId(principal.getId() + "-navigation-panel");
388                                }
389                                Function<Action, String> navItemIdProvider = na -> isBlank(na.getId()) ? null : "nsd-app-nav-item-" + na.getId();
390                                List<EObject> navPanelItems = navPanel.getItems();
391                                principalChildren.forEach(principalChild -> {
392                                        if (principalChild instanceof Action) {
393                                                navPanelItems.add(createLabel((Action) principalChild, activeAction, uriResolver, navItemIdProvider, "nav-panel", true, true));
394                                        } else {
395                                                navPanelItems.add(EcoreUtil.copy(principalChild));
396                                        }
397                                });
398                                
399                        }
400                }
401                
402                // Content panel
403                ContentPanel contentPanel = appPage.getContentPanel();
404                if (contentPanel == null) {
405                        contentPanel = AppFactory.eINSTANCE.createContentPanel();
406                        appPage.setContentPanel(contentPanel);
407                }
408                
409                buildContentPanel(activeAction, actionPath, activeAction == root || activeAction == principal, contentPanel, uriResolver, progressMonitor);     
410        }

412        private static void buildContentPanel(
413                        Action action, 
414                        List<Action> path,
415                        boolean rootOrPrincipal,
416                        ContentPanel contentPanel,
417                        BiFunction<Action, URI, URI> uriResolver, 
418                        ProgressMonitor progressMonitor) {
419                
420                if (!rootOrPrincipal) {
421                        if (path != null && !path.isEmpty()) {
422                                EList<Label> breadcrumb = contentPanel.getBreadcrumb();
423                                path.forEach(pathElement -> {
424                                        Label element = createLabel(pathElement, action, uriResolver, null, "content-panel/breadcrumb", false, false);
425                                        if (element != null) {
426                                                breadcrumb.add(element);
427                                        }
428                                });
429                                Label tail = createLabel(action, action, uriResolver, null, "content-panel/breadcrumb", false, false);          
430                                tail.setActive(true);
431                                breadcrumb.add(tail);
432                        }
433        
434                        if (!isBlank(action.getText()) || !isBlank(action.getIcon())) {
435                                Label title = org.nasdanika.common.Util.isBlank(action.getName()) ? AppFactory.eINSTANCE.createLabel() : AppFactory.eINSTANCE.createLink(); 
436                                configureLabel(action, action, uriResolver, null, "content-panel/title", title, false, false);
437                                contentPanel.setTitle(title);
438                        }
439                                        
440                        List<EObject> navigation = resolveActionReferences(action.getNavigation());
441                        EList<EObject> navigationItems = contentPanel.getItems();
442                        navigation.forEach(navigationElement -> {
443                                if (navigationElement instanceof Action) {
444                                        navigationItems.add(createLabel((Action) navigationElement, action, uriResolver, null, "content-panel/navigation-item", true, false));
445                                } else {
446                                        navigationItems.add(EcoreUtil.copy(navigationElement));
447                                }
448                        });
449                }
450                
451                int sectionColumns = action.getSectionColumns();
452                if (sectionColumns > 0) {
453                        contentPanel.setSectionColumns(sectionColumns);
454                }
455                SectionStyle sectionStyle = action.getSectionStyle();
456                if (sectionStyle != null) {
457                        contentPanel.setSectionStyle(sectionStyle);
458                }
459                
460                EList<ContentPanel> cpSections = contentPanel.getSections();
461                for (Action section: action.getSections()) {
462                        ContentPanel sectionPanel = AppFactory.eINSTANCE.createContentPanel();
463                        cpSections.add(sectionPanel);
464                        buildContentPanel(section, null, false, sectionPanel, uriResolver, progressMonitor);
465                }
466
467                contentPanel.setFloatLeftNavigation(createNavigationPanel(action.getFloatLeftNavigation()));
468                contentPanel.setFloatRightNavigation(createNavigationPanel(action.getFloatRightNavigation()));
469                contentPanel.setLeftNavigation(createNavigationPanel(action.getLeftNavigation()));              
470                contentPanel.setRightNavigation(createNavigationPanel(action.getRightNavigation()));
471                
472                contentPanel.getContent().addAll(EcoreUtil.copyAll(action.getContent()));
473        }

475        /**
476         * Creates a copy of a panel if not null and replaces action references with links.
477         * @param panel
478         * @return
479         */
480        private static NavigationPanel createNavigationPanel(NavigationPanel panel) {
481                if (panel == null) {
482                        return null;
483                }
484                NavigationPanel ret = EcoreUtil.copy(panel);
485                ListIterator<EObject> iit = ret.getItems().listIterator();
486                while (iit.hasNext()) {
487                        EObject next = iit.next();
488                        if (next instanceof ActionReference) {
489                                iit.set(EcoreUtil.copy(((ActionReference) next).getTarget()));
490                        }
491                }
492                return ret;
493        }

499        /**
500         * @param action
501         * @param activeAction
502         * @return true if this action is active or one of its anonymous or navigation descendants is active.
503         */
504        private static boolean isActiveAction(Action action, Action activeAction, boolean inspectChildren) {
505                if (action == activeAction) {
506                        return true;
507                }
508                
509                for (Action ac: action.getAnonymous()) {
510                        if (isActiveAction(ac, activeAction, inspectChildren)) {
511                                return true;
512                        }
513                }
514                
515                for (EObject nc: action.getNavigation()) {
516                        if (nc instanceof Action && isActiveAction((Action) nc, activeAction, inspectChildren)) {
517                                return true;
518                        }
519                }
520                
521                if (inspectChildren) {
522                        for (EObject child: action.getChildren()) {
523                                if (child instanceof Action && isActiveAction((Action) child, activeAction, inspectChildren)) {
524                                        return true;
525                                }
526                        }
527                }
528                
529                return false;
530        }

532        private static void configureLabel(
533                        Action action, 
534                        Action activeAction, 
535                        BiFunction<Action, URI, URI> uriResolver, 
536                        Function<Action, String> idProvider, 
537                        String appearancePath, 
538                        Label label, 
539                        boolean recursive,
540                        boolean inNavPanel) {
541                Appearance aa = action.getAppearance();
542                if (aa != null) {
543                        label.setAppearance(aa.effectiveAppearance(appearancePath));
544                }
545
546                label.setColor(action.getColor());
547                label.setDescription(action.getDescription());
548                label.setDisabled(action.isDisabled());
549                Modal help = action.getHelp();
550                if (help != null) {
551//              label.setHelp(value); TODO - make links in help relative.
552                        throw new UnsupportedOperationException("Help modals not supported yet");
553                }
554                label.setIcon(action.getIcon());
555                if (idProvider != null) {
556                        label.setId(idProvider.apply(action));
557                }
558//              label.setId(action.getId());            
559                label.setNotification(action.getNotification());
560                label.setOutline(action.isOutline());
561                label.setText(action.getText());
562                label.setTooltip(action.getTooltip());
563                
564                for (Entry<String, EObject> ae: action.getAttributes().entrySet()) {            
565                        label.getAttributes().put(ae.getKey(), EcoreUtil.copy(ae.getValue()));
566                }
567                
568                if (label instanceof Link) {
569                        configureLink(action, activeAction, uriResolver, (Link) label);
570                }
571                
572                if (recursive) {
573                        EList<EObject> labelChildren = label.getChildren();
574                        for (EObject actionChild: action.getChildren()) {
575                                if (actionChild instanceof ActionReference) {
576                                        actionChild = ((ActionReference) actionChild).getTarget();
577                                }
578                                if (actionChild instanceof Action) {
579                                        Action childAction = (Action) actionChild;
580                                        labelChildren.add(createLabel(childAction, activeAction, uriResolver, idProvider, "header/navigation", recursive, inNavPanel));
581                                        
582                                        // Second level - headers, separators.
583                                } else {
584                                        labelChildren.add(EcoreUtil.copy(actionChild));
585                                }                                                       
586                        }                       
587                }
588                
589                label.setActive(inNavPanel ? isActiveAction(action, activeAction, !hasActiveChildren(label)) : action.isActive());              
590        }

592        private static boolean hasActiveChildren(Label label) {
593                for (EObject child: label.getChildren()) {
594                        if (child instanceof Label) {
595                                Label cl = (Label) child;
596                                if (cl.isActive() || hasActiveChildren(cl)) {
597                                        return true;
598                                }
599                        }
600                }
601                return false;
602        }

604        private static void configureLink(
605                        Action action, 
606                        Action activeAction, 
607                        BiFunction<Action, URI, URI> uriResolver, 
608                        Link link) {
609                
610                URI activeActionURI = uriResolver.apply(activeAction, null);
611                URI uri = uriResolver.apply(action, activeActionURI);
612                
613                if (uri != null) {
614                        link.setLocation(uri.toString());
615                }
616                link.setConfirmation(action.getConfirmation());
617                if (action.getModal() != null) {
618//              link.setModal(value); TODO - contextualization of links - relativization of URI's. Token expansion with a property computer? something like ${relative-uri/<uri here>}? 
619//              Support of sub-tokens e.g. ${{relative-uri/${token}}} recognizes ${token} as a sub-token to be expanded to result in ${relative-uri/<token value>}.
620                        throw new UnsupportedOperationException("Modals are not supported yet");
621                }
622                if (action.isModalActivator()) {
623                        throw new UnsupportedOperationException("Modals are not supported yet");
624                }
625                if (action.isInline()) {
626                        throw new UnsupportedOperationException("Inline actions are not supported yet");                        
627                }
628                link.setName(action.getName());
629                link.setScript(action.getScript());
630                link.setTarget(action.getTarget());             
631        }

682        /**
683         * Resolves URI's by traversing containment references from the root Action taking {@link ActionReference} into account.
684         * @param context
685         * @return A function resolving {@link URI} for the argument {@link Action}. Caches results.
686         */
687        public static BiFunction<Action,URI,URI> uriResolver(Action root, Context context) {
688                Map<Action, URI> cache = new HashMap<>();
689                traverse(root, context.get(Context.BASE_URI_PROPERTY, URI.class), context, cache);
690                
691                return new BiFunction<Action, URI, URI>() {
692                        
693                        @Override
694                        public URI apply(Action action, URI base) {
695                                URI uri = cache.get(action);                            
696                                return base == null || uri == null ? uri : uri.deresolve(base, true, true, true);
697                        }
698                
699                };
700        }

702        private static void traverse(Action action, URI base, Context context, Map<Action, URI> cache) {
703                URI actionURI = compute(action, base, context);
704                cache.put(action, actionURI);
705//              if (actionURI != null) {
706//                      System.out.println(">>> " + action.getText() + " " + action.getLocation() + " > " + actionURI);
707//              }
708                
709                for (EObject child: resolveActionReferences(action.getChildren())) {
710                        if (child instanceof Action) {
711                                traverse((Action) child, actionURI == null ? base : actionURI, context, cache);
712                        }
713                }
714                for (EObject item: resolveActionReferences(action.getNavigation())) {
715                        if (item instanceof Action) {
716                                traverse((Action) item, actionURI == null ? base : actionURI, context, cache);
717                        }
718                }
719                for (Action section: action.getSections()) {
720                        traverse(section, actionURI == null ? base : actionURI, context, cache);
721                }
722                for (Action anonymous: action.getAnonymous()) {
723                        traverse(anonymous, actionURI == null ? base : actionURI, context, cache);
724                }
725                
726                // TODO - nav panels?
727        }

729        private static URI compute(Action action, URI base, Context context) {
730                String uriString;
731                if (action.eContainmentFeature() == AppPackage.Literals.ACTION__SECTIONS) {
732                        String aName = context.interpolateToString(action.getLocation());
733                        if (isBlank(aName)) {
734                                return null;
735                        }
736                        uriString = "#" + aName;
737                } else {                                
738                        uriString = context.interpolateToString(action.getLocation());
739                        if (isBlank(uriString)) {
740                                return null;
741                        }
742                }
743                URI uri = URI.createURI(uriString);
744                if (uri.isRelative() && base != null) {
745                        return uri.resolve(base);
746                }
747                return uri;
748        }

762        /**
763         * For Nasdanika App pages extracts page title, breadcrumbs and content text into a JSONObject to add to a collector and then
764         * to use for constructing search indices. 
765         * @param path
766         * @param file
767         * @param contentConsumer Consumer of content elements which can be used to analyze page contents, e.g. find dead links. Can be null.  
768         * @return Search document object for non-empty Nasdanika App pages, null otherwise.
769         * @throws IOException
770         */
771        public static JSONObject createSearchDocument(String path, File file, Consumer<? super Element> contentConsumer) throws IOException {
772                Document document = Jsoup.parse(file, "UTF-8");
773                Elements contentPanelQuery = document.select("body > div > div.row.nsd-app-content-row > div.col.nsd-app-content-panel");                                                                                                     
774                if (contentPanelQuery.isEmpty()) {
775                        return null;
776                }
777                Elements contentQuery = contentPanelQuery.select("div > div.row.nsd-app-content-panel-content-row");
778                if (contentQuery.isEmpty()) {
779                        return null;
780                }
781                if (contentConsumer != null) {
782                        contentQuery.forEach(contentConsumer);
783                }
784                
785                StringBuilder contentText = new StringBuilder(contentQuery.text());
786                
787                // Drawio diagrams labels
788                for (Element diagramDiv: contentQuery.select("div.mxgraph")) {
789                        traverseMxGraphModel(diagramDiv, cell -> {
790                                Object cellValue = cell.getValue();
791                                if (cellValue instanceof org.w3c.dom.Element) {
792                                        String label = ((org.w3c.dom.Element) cellValue).getAttribute("label");
793                                        if (!org.nasdanika.common.Util.isBlank(label)) {
794                                                String labelText = Jsoup.parse(label).text();
795                                                contentText.append(" ").append(labelText);
796                                        }
797                                } else if (cellValue != null) {
798                                        contentText.append(" ").append(cellValue);                                      
799                                }
800                        });
801                }
802                
803                if (org.nasdanika.common.Util.isBlank(contentText.toString())) {
804                        return null;
805                }
806                
807                JSONObject searchDocument = new JSONObject();
808                searchDocument.put("content", StringEscapeUtils.escapeHtml4(contentText.toString()));
809                Elements titleQuery = contentPanelQuery.select("div > div.row.nsd-app-content-panel-title-and-items-row > div.col-auto > h1");
810                if (titleQuery.size() == 1) {
811                        searchDocument.put("title", StringEscapeUtils.escapeHtml4(titleQuery.get(0).text()));
812                } else {
813                        searchDocument.put("title", document.title());
814                }
815                Elements breadcrumbQuery = contentPanelQuery.select("div > div.row.nsd-app-content-panel-breadcrumb-row > div > nav > ol > li");
816                if (breadcrumbQuery.size() > 0) {
817                        searchDocument.put("path", String.join("/", breadcrumbQuery.stream().map(e -> StringEscapeUtils.escapeHtml4(e.text())).collect(Collectors.toList())));
818                }
819                return searchDocument;
820        }

822        /**
823         * Loads {@link mxGraphModel} from mxgraph div.
824         * @param mxGraphDiv
825         * @return
826         */
827        public static mxGraphModel loadMxGraphModel(Element mxGraphDiv) {
828                String data = mxGraphDiv.attr("data-mxgraph");
829                if (data != null) {
830                        JSONObject jsonData = new JSONObject(data);
831                        Object xml = jsonData.get("xml");
832                        if (xml instanceof String) {
833                                org.w3c.dom.Document xmlDocument = mxXmlUtils.parseXml((String) xml);
834                                org.w3c.dom.Element documentElement = xmlDocument.getDocumentElement();
835                                String compressedFlag = documentElement.getAttribute("compressed");
836                                if ("true".equals(compressedFlag)) {
837                                        try {
838                                                String textContent = documentElement.getTextContent();
839                                                if (!Base64.isBase64(textContent)) {
840                                                        throw new NasdanikaException("Compressed diagram is not Base64 encoded");
841                                                }
842                                            byte[] compressed = Base64.decodeBase64(textContent);
843                                            byte[] decompressed = inflate(compressed);
844                                            String decompressedStr = new String(decompressed, StandardCharsets.UTF_8);
845                                            String decodedStr = URLDecoder.decode(decompressedStr, StandardCharsets.UTF_8.name());
846                                            org.w3c.dom.Document modelDoc = mxXmlUtils.parseXml(decodedStr);
847                                                mxCodec codec = new mxCodec(modelDoc);
848                                                return (mxGraphModel) codec.decode(modelDoc.getDocumentElement());
849                                        } catch (IOException e) {
850                                                throw new NasdanikaException("Error decoding a drwaio diagram: " + e, e);
851                                        }
852                                }
853                                
854                                mxCodec codec = new mxCodec(xmlDocument);
855                                org.w3c.dom.Element diagramElement = (org.w3c.dom.Element) xmlDocument.getElementsByTagName("diagram").item(0);
856                                org.w3c.dom.Element graphModelElement = (org.w3c.dom.Element) diagramElement.getElementsByTagName("mxGraphModel").item(0);
857                                return (mxGraphModel) codec.decode(graphModelElement);
858                        }
859                }
860                return null;
861        }

863        private static  byte[] inflate(byte[] content) throws IOException {
864                ByteArrayOutputStream baos = new ByteArrayOutputStream();
865                try (ByteArrayInputStream source = new ByteArrayInputStream(content); OutputStream target = new InflaterOutputStream(baos, new Inflater(true))) {
866                byte[] buf = new byte[8192];
867                int length;
868                while ((length = source.read(buf)) > 0) {
869                    target.write(buf, 0, length);
870                }
871                }
872                
873                return baos.toByteArray();
874        }

886        /**
887         * Creates an inspector consumer which finds and reports broken links, missing images and elements with nsd-error class.
888         * @param file
889         * @param path
890         * @param siteDir
891         * @param errorConsumer
892         * @return
893         */
894        public static Consumer<? super Element> createInspector(Predicate<String> linkPredicate, Consumer<String> errorConsumer) {              
895                return element -> {
896                        Elements anchors = element.getElementsByTag("a");
897                        for (Element anchor: anchors) {
898                                String target = anchor.attr("href");
899                                if (target != null) {
900                                        String id = anchor.attr("id");
901                                        boolean selfReferencing = id != null && target.equals("#" + id); // Ignoring header tags generated by markdown.
902                                        if (!selfReferencing && !linkPredicate.test(target)) {
903                                                errorConsumer.accept("Broken link: " + anchor);
904                                        }
905                                }
906                        }
907                        
908                        // Images
909                        Elements images = element.getElementsByTag("img");
910                        for (Element img: images) {
911                                String src = img.attr("src");
912                                if (src != null && !linkPredicate.test(src)) {
913                                        errorConsumer.accept("Missing image: " + img);
914                                }
915                        }
916                        
917                        // NSD errors
918                        Elements errors = element.select("div.nsd-error");
919                        for (Element error: errors) {
920                                errorConsumer.accept("Error: " + error.text());
921                        }
922                        
923                        // Image maps
924                        for (Element area: element.select("area")) {
925                                String target = area.attr("href");
926                                if (target != null && !linkPredicate.test(target)) {
927                                        errorConsumer.accept("Broken link: " + area);                                   
928                                }
929                        }
930                        
931                        // Drawio diagrams
932                        for (Element diagramDiv: element.select("div.mxgraph")) {
933                                traverseMxGraphModel(diagramDiv, cell -> {
934                                        Object cellValue = cell.getValue();
935                                        if (cellValue instanceof org.w3c.dom.Element) {
936                                                String link = ((org.w3c.dom.Element) cellValue).getAttribute("link");
937                                                if (!org.nasdanika.common.Util.isBlank(link) && !linkPredicate.test(link)) {
938                                                        String label = ((org.w3c.dom.Element) cellValue).getAttribute("label");
939                                                        if (org.nasdanika.common.Util.isBlank(label)) {
940                                                                errorConsumer.accept("Broken diagram link: " + link);
941                                                        } else {
942                                                                errorConsumer.accept("Broken diagram link on " + label + ": " + link);                                                                                  
943                                                        }
944                                                }
945                                        }
946                                });
947                        }                       
948                };
949        }

958        /**
959         * Creates a predicate checking for links relative to the argument file in the argument directory. 
960         * @param file Base file
961         * @param dir Directory within which check for relative links
962         * @return
963         */
964        public static Predicate<String> createRelativeLinkPredicate(File file, File dir) {
965                return target -> {
966                        File targetFile;
967                        String fragment;
968                        if (target.startsWith("#")) {
969                                targetFile = file;
970                                fragment = target;
971                        } else {
972                                int hashIdx = target.indexOf("#");
973                                if (hashIdx == -1) {
974                                        fragment = null;
975                                } else {
976                                        fragment = target.substring(hashIdx);
977                                        target = target.substring(0, hashIdx);
978                                }
979                                if (target.contains(":") || target.contains("?")) { // Absolute or with a query string, not checking
980                                        return true;
981                                }
982                                targetFile = new File(file.getParentFile(), target);
983                                for (File parent = targetFile.getParentFile(); parent != null; parent = parent.getParentFile()) {
984                                        if (parent.equals(dir)) {
985                                                if (!targetFile.exists()) {
986                                                        return false;
987                                                }
988                                        }
989                                }
990                        }
991                        if (fragment == null || fragment.trim().length() == 0) {
992                                return true;
993                        }
994                        
995                        try {
996                                Document document = Jsoup.parse(targetFile, StandardCharsets.UTF_8.name());
997                                Elements namedAnchors = document.select("[name='" + fragment.substring(1) + "']");
998                                Elements idAnchors = document.select("[id='" + fragment.substring(1) + "']");
999                                return namedAnchors.size() + idAnchors.size() == 1;
1000                        } catch (IOException e) {
1001                                throw new NasdanikaException("Error parsing " + targetFile.getAbsolutePath(), e);
1002                        }
1003                        
1004                };
