001package org.nasdanika.html.bootstrap;
002
003import org.nasdanika.html.Tag;
004
005public interface Navbar extends BootstrapElement<Tag,Navbar> {
006        
007        /**
008         * Creates a navbar items - a tag. Returns the tag so it can be additionally styled or
009         * more content can be added.
010         * @param href
011         * @param active
012         * @param disabled
013         * @param content
014         * @return
015         */
016        Tag item(Object href, boolean active, boolean disabled, Object... content);
017        
018        Dropdown dropdown(boolean active, Object... name); 
019
020        Navbar form(org.nasdanika.html.Form form);
021        
022        /**
023         * Creates navbar text (span) and returns it.
024         * @param content
025         * @return
026         */
027        Tag navbarText(Object... content);
028
029}
