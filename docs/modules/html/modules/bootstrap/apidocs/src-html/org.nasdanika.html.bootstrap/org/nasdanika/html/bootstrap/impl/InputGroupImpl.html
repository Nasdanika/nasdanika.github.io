001package org.nasdanika.html.bootstrap.impl;
002
003import org.nasdanika.html.Fragment;
004import org.nasdanika.html.HTMLFactory;
005import org.nasdanika.html.InputBase;
006import org.nasdanika.html.Tag;
007import org.nasdanika.html.bootstrap.BootstrapFactory;
008import org.nasdanika.html.bootstrap.InputGroup;
009
010public class InputGroupImpl extends DivWrappingBootstrapElementImpl<InputGroup> implements InputGroup  {
011
012        private Tag prepend;
013        private Fragment inputs;
014        private Tag append;
015        
016        public InputGroupImpl(BootstrapFactory factory) {
017                super(factory);
018                htmlElement.addClass("input-group");
019                HTMLFactory htmlFactory = getFactory().getHTMLFactory();
020                prepend = htmlFactory.nonEmptyDiv().addClass("input-group-prepend");
021                inputs = htmlFactory.fragment();
022                append = htmlFactory.nonEmptyDiv().addClass("input-group-append");
023                htmlElement.content(prepend, inputs, append);
024        }
025        
026        protected Object wrapContent(Object content) {
027                return content instanceof CharSequence ? getFactory().getHTMLFactory().span(content).addClass("input-group-text") : content;
028        }
029
030        @Override
031        public InputGroup prepend(Object content) {
032                prepend.content(wrapContent(content));
033                return this;
034        }
035
036        @Override
037        public InputGroup input(InputBase<?> input) {
038                input.addClass("form-control");
039                inputs.content(input);
040                return this;
041        }
042
043        @Override
044        public InputGroup append(Object content) {
045                append.content(wrapContent(content));
046                return this;
047        }
048
049        @Override
050        public InputGroup large() {
051                return large(true);
052        }
053
054        @Override
055        public InputGroup large(boolean large) {
056                htmlElement.addClassConditional(large, "input-group-lg");
057                return this;
058        }
059
060        @Override
061        public InputGroup small() {
062                return small(true);
063        }
064
065        @Override
066        public InputGroup small(boolean small) {
067                htmlElement.addClassConditional(small, "input-group-sm");
068                return this;
069        }
070
071}
