001package org.nasdanika.html.bootstrap.impl;
002
003import org.nasdanika.html.bootstrap.BootstrapElement;
004import org.nasdanika.html.bootstrap.Breakpoint;
005import org.nasdanika.html.bootstrap.Size;
006import org.nasdanika.html.bootstrap.Spacing;
007
008public class SpacingImpl<B extends BootstrapElement<?, ?>> implements Spacing<B> {
009        
010        private B bootstrapElement;
011        private String prefix;
012
013        SpacingImpl(B bootstrapElement, String prefix) {
014                this.bootstrapElement = bootstrapElement;
015                this.prefix = prefix;
016        }
017
018        @Override
019        public B toBootstrapElement() {
020                return bootstrapElement;
021        }
022        
023        private void suffix(String suffix) {
024                bootstrapElement.toHTMLElement().addClass(prefix+suffix);
025        }
026
027        @Override
028        public Spacing<B> top(Breakpoint breakpoint, Size size) {
029                suffix(breakpoint.size("t", size));
030                return this;
031        }
032
033        @Override
034        public Spacing<B> bottom(Breakpoint breakpoint, Size size) {
035                suffix(breakpoint.size("b", size));
036                return this;
037        }
038
039        @Override
040        public Spacing<B> left(Breakpoint breakpoint, Size size) {
041                suffix(breakpoint.size("l", size));
042                return this;
043        }
044
045        @Override
046        public Spacing<B> right(Breakpoint breakpoint, Size size) {
047                suffix(breakpoint.size("r", size));
048                return this;
049        }
050
051        @Override
052        public Spacing<B> x(Breakpoint breakpoint, Size size) {
053                suffix(breakpoint.size("x", size));
054                return this;
055        }
056
057        @Override
058        public Spacing<B> y(Breakpoint breakpoint, Size size) {
059                suffix(breakpoint.size("y", size));
060                return this;
061        }
062
063        @Override
064        public Spacing<B> all(Breakpoint breakpoint, Size size) {
065                bootstrapElement.toHTMLElement().addClass(breakpoint.size(prefix, size));
066                // TODO Auto-generated method stub
067                return this;
068        }
069
070}
