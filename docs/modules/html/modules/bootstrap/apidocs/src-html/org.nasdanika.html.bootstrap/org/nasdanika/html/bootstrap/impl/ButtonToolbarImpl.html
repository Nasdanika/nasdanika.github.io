001package org.nasdanika.html.bootstrap.impl;
002
003import org.nasdanika.html.bootstrap.BootstrapFactory;
004import org.nasdanika.html.bootstrap.ButtonToolbar;
005
006public class ButtonToolbarImpl extends DivWrappingBootstrapElementImpl<ButtonToolbar> implements ButtonToolbar {
007        
008        public ButtonToolbarImpl(BootstrapFactory factory) {
009                super(factory);
010                htmlElement.addClass("btn-toolbar").attribute("role", "toolbar");
011        }
012
013        @Override
014        public ButtonToolbar add(Object content) {
015                htmlElement.content(content);
016                return this;
017        }
018
019}
