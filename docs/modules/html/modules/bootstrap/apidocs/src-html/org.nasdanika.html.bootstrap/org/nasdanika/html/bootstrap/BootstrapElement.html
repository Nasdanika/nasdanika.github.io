001package org.nasdanika.html.bootstrap;
002
003import org.nasdanika.html.HTMLElement;
004import org.nasdanika.html.Markup;
005import org.nasdanika.html.Producer;
006
007public interface BootstrapElement<H extends HTMLElement<?>, B extends BootstrapElement<H,?>> extends Producer, Markup {
008        
009        H toHTMLElement();
010        
011        BootstrapFactory getFactory();
012        
013        /**
014         * Adds border classes
015         * @param color Optional color, may be null.
016         * @param placement Optionial placements, all borders if none a specified.
017         * @return
018         */
019        B border(Color color, Placement... placement);
020        
021        B background(Color color);
022        
023        Text<B> text();
024        
025        Spacing<B> margin();
026        
027        Spacing<B> padding();
028        
029        Float<B> _float();
030
031}
