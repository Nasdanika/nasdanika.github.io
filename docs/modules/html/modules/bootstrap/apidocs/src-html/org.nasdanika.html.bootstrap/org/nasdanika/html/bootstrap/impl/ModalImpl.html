001package org.nasdanika.html.bootstrap.impl;
002
003import org.nasdanika.html.HTMLElement;
004import org.nasdanika.html.Tag;
005import org.nasdanika.html.bootstrap.BootstrapFactory;
006import org.nasdanika.html.bootstrap.Breakpoint;
007import org.nasdanika.html.bootstrap.Modal;
008import org.nasdanika.html.bootstrap.TagBootstrapElement;
009
010public class ModalImpl extends WrappingBootstrapElementImpl<Tag, Modal> implements Modal {
011
012        private TagBootstrapElement content;    
013        private TagBootstrapElement header;
014        private TagBootstrapElement body;
015        private TagBootstrapElement footer;
016        private Tag dialog;
017
018        @Override
019        public TagBootstrapElement getContent() {
020                return content;
021        }
022        
023        public ModalImpl(BootstrapFactory factory) {
024                super(factory, factory.getHTMLFactory().nonEmptyDiv());
025                htmlElement.addClass("modal");
026                htmlElement.attribute("tabindex", "-1");
027                dialog = factory.getHTMLFactory().div().addClass("modal-dialog");
028                htmlElement.content(dialog);
029                Tag contentDiv = factory.getHTMLFactory().div().addClass("modal-content");
030                dialog.content(contentDiv);
031                content = getFactory().wrap(contentDiv);
032                
033                header = getFactory().wrap(getFactory().getHTMLFactory().nonEmptyDiv());
034                header.toHTMLElement().addClass("modal-header");
035                content.toHTMLElement().content(header.toHTMLElement());
036                
037                body = getFactory().wrap(getFactory().getHTMLFactory().nonEmptyDiv());
038                body.toHTMLElement().addClass("modal-body");
039                content.toHTMLElement().content(body.toHTMLElement());                                  
040                
041                footer = getFactory().wrap(getFactory().getHTMLFactory().nonEmptyDiv());
042                footer.toHTMLElement().addClass("modal-footer");
043                content.toHTMLElement().content(footer.toHTMLElement());
044        }
045
046        @Override
047        public TagBootstrapElement getHeader() {
048                return header;
049        }
050
051        @Override
052        public TagBootstrapElement getBody() {
053                return body;
054        }
055
056        @Override
057        public TagBootstrapElement getFooter() {
058                return footer;
059        }
060
061        @Override
062        public void bindTrigger(HTMLElement<?> trigger) {
063                if (htmlElement.getId() == null) {
064                        htmlElement.id(getFactory().getHTMLFactory().nextId());
065                }
066                trigger.attribute("data-toggle", "modal");
067                trigger.attribute("data-target", "#" + htmlElement.getId());
068        }
069        
070        @Override
071        public String activatorScript() {
072                if (htmlElement.getId() == null) {
073                        htmlElement.id(getFactory().getHTMLFactory().nextId());
074                }
075                return "jQuery('#" + htmlElement.getId() + "').modal('show')";
076        }
077
078        @Override
079        public void bindDismisser(HTMLElement<?> dismisser) {
080                dismisser.attribute("data-dismiss", "modal");
081        }
082
083        @Override
084        public Modal scrollable(boolean scrollable) {
085                if (scrollable) {
086                        dialog.addClass("modal-dialog-scrollable");
087                } else {
088                        dialog.removeClass("modal-dialog-scrollable");
089                }
090                return this;
091        }
092
093        @Override
094        public Modal centered(boolean centered) {
095                if (centered) {
096                        dialog.addClass("modal-dialog-centered");
097                } else {
098                        dialog.removeClass("modal-dialog-centered");
099                }
100                return this;
101        }
102
103        @Override
104        public Modal size(Breakpoint size) {
105                dialog.removeClass("modal-sm");
106                dialog.removeClass("modal-lg");
107                dialog.removeClass("modal-xl");
108                if (size != null) {
109                        switch (size) {
110                        case EXTRA_LARGE:
111                        case LARGE:
112                        case SMALL:
113                                dialog.addClass("modal-" + size.code);
114                                break;
115                        default:
116                                break;                  
117                        }
118                }
119                return this;
120        }
121
122}
