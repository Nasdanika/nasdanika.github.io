001package org.nasdanika.html.bootstrap;

003/**
004 * Bootstrap spacing - padding and margins - https://getbootstrap.com/docs/4.5/utilities/spacing/.
005 * Integer space can take values from 0 to 5.
006 * @author Pavel Vlasov
007 *
008 */
009public interface Spacing<B extends BootstrapElement<?, ?>> {
010        
011        B toBootstrapElement();
012        
013        Spacing<B> top(Breakpoint breakpoint, Size size);
014        
015        Spacing<B> bottom(Breakpoint breakpoint, Size size);
016        
017        Spacing<B> left(Breakpoint breakpoint, Size size);
018        
019        Spacing<B> right(Breakpoint breakpoint, Size size);
020        
021        Spacing<B> x(Breakpoint breakpoint, Size size);
022        
023        Spacing<B> y(Breakpoint breakpoint, Size size);
024        
025        Spacing<B> all(Breakpoint breakpoint, Size size);
026        
027}
