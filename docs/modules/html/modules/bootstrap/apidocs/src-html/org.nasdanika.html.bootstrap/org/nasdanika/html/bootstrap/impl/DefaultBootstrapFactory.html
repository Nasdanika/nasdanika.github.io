040/**
041 * HTML factory which relies on Bootstrap styles and scripts.
042 * @author Pavel
043 *
044 */
045public class DefaultBootstrapFactory implements BootstrapFactory {
046        
047        private HTMLFactory htmlFactory;
048
049        public DefaultBootstrapFactory(HTMLFactory htmlFactory) {
050                this.htmlFactory = htmlFactory;
051        }
052        
053        /**
054         * For the service loader
055         */
056        public DefaultBootstrapFactory() {
057                this(ServiceLoader.load(HTMLFactory.class).findFirst().get());
058        }
059        
060        @Override
061        public HTMLFactory getHTMLFactory() {
062                return htmlFactory;
063        }
064
065        @Override
066        public Tag alert(Color color, Object... content) {
067                return getHTMLFactory().div(content)
068                                .addClass("alert")
069                                .addClassConditional(color != null && color.code != null, "alert-"+color.code)
070                                .attribute("role", "alert");
071        }
072
073        @Override
074        public Tag badge(boolean pill, Color color, Object... content) {
075                Tag ret = getHTMLFactory().span(content)
076                                .addClass("badge")
077                                .addClassConditional(color != null && color.code != null, "badge-"+color.code)
078                                .addClassConditional(pill, "badge-pill");
079                return ret;
080        }
081
082        @Override
083        public Tag badge(boolean pill, Object... content) {
084                Tag ret = getHTMLFactory().span(content)
085                                .addClass("badge")
086                                .addClassConditional(pill, "badge-pill");
087                return ret;
088        }
089
090        @Override
091        public Tag badgeLink(Object href, boolean pill, Color color, Object... content) {
092                Tag ret = getHTMLFactory().link(href, content)
093                                .addClass("badge")
094                                .addClassConditional(color != null && color.code != null, "badge-"+color.code)
095                                .addClassConditional(pill, "badge-pill");
096                return ret;
097        }

154        @Override
155        public <B extends BootstrapElement<?,?>> B tooltip(B bootstrapElement, Object tooltip, boolean html, Placement placement) {
156                tooltip(bootstrapElement.toHTMLElement(), tooltip, html, placement);
157                return bootstrapElement;
158        }
159        
160        @Override
161        public <H extends HTMLElement<?>> H tooltip(H htmlElement, Object tooltip, boolean html, Placement placement) {
162                htmlElement
163                        .attribute("data-toggle", "tooltip")
164                        .attribute("data-placement", placement.name().toLowerCase())                    
165                        .attribute("data-html", "true", html)
166                        .attribute("title", tooltip);           
167                return htmlElement;
168        }
169        
170        @Override
171        public Tag initTooltipScript() {
172                return htmlFactory.nonEmptyTag(TagName.script, "$(function () { $('[data-toggle=\"tooltip\"]').tooltip(); });");
173        }

210        @Override
211        public <P extends HTMLPage> P bootstrapHTMLPage(P page) {
212                page
213                        .head(getHTMLFactory().tag(TagName.meta).attribute("charset", "utf-8"))
214                        .head(getHTMLFactory().tag(TagName.meta).attribute("name", "viewport").attribute("content", "width=device-width, initial-scale=1, shrink-to-fit=no"));
215                return page;
216        }

224        public HTMLPage bootstrapCdnHTMLPage(Theme theme) {
225                return bootstrapCdnHTMLPage(getHTMLFactory().page(), theme);
226        }
227        
228        @Override
229        public <P extends HTMLPage> P bootstrapCdnHTMLPage(P page, Theme theme) {
230                theme.cdn(page);
231                P bootstrapPage = bootstrapHTMLPage(page);
232                bootstrapPage.script("https://code.jquery.com/jquery-3.3.1.min.js");
233                bootstrapPage.script("https://cdn.jsdelivr.net/npm/popper.js@1.16.1/dist/umd/popper.min.js");
234                bootstrapPage.script("https://cdn.jsdelivr.net/npm/bootstrap@4.5.2/dist/js/bootstrap.min.js");
235                return page;
236        }
237        
238        @Override
239        public <P extends HTMLPage> P bootstrapCdnHTMLPage(P page) {
240                return bootstrapCdnHTMLPage(page, Theme.Default);
241        }
242        
243        @Override
244        public HTMLPage bootstrapHTMLPage() {
245                return bootstrapHTMLPage(getHTMLFactory().page());
246        }

248        @Override
249        public Select themeSelect(Theme selected) {
250                Select select = getHTMLFactory().select();
251                select.on(Event.change, "document.getElementById('"+Theme.STYLESHEET_ID+"').href = this.value;");
252                for (Theme theme: Theme.values()) {
253                        select.option(theme.stylesheetCdnURL, theme.name(), theme == selected, false);
254                }
255                return select;
256        }

299        @Override
300        public TagBootstrapElement progressBar(int percentage) {
301                HTMLFactory htmlFactory = getHTMLFactory();                                             
302                Tag progressDiv = htmlFactory.div(percentage + "%").addClass("progress-bar").style().width(percentage + "%");
303                return wrap(htmlFactory.div(progressDiv).addClass("progress"));
