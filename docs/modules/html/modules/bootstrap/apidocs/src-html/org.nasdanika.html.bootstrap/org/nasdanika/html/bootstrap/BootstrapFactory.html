004

013/**
014 * Factory for creating bootstrap UI elements. 
015 * @author Pavel
016 *
017 */
018public interface BootstrapFactory {
019        
020        BootstrapFactory INSTANCE = new DefaultBootstrapFactory(HTMLFactory.INSTANCE);
021        
022        /**
023         * @return Underlying HTML factory.
024         */
025        HTMLFactory getHTMLFactory();
026        
027        Tag alert(Color color, Object... content);
028        
029        Tag badge(boolean pill, Color color, Object... content);
030        Tag badge(boolean pill, Object... content);
031        Tag badgeLink(Object href, boolean pill, Color color, Object... content);
032        
033        /**
034         * Wraps HTML element for applying Bootstrap styling.
035         * @param htmlElement
036         * @return
037         */
038        <H extends HTMLElement<?>> BootstrapElement<H, BootstrapElement<H, ?>> wrap(H htmlElement);
039        
040        /**
041         * Wraps tag.
042         * @param tag
043         * @return
044         */
045        TagBootstrapElement wrap(Tag tag);
046                        
047        Breadcrumb breadcrumb();
048        
049        /**
050         * Wraps HTML element into a Bootstrap button.
051         * @param htmlElement
052         * @return
053         */
054        <H extends HTMLElement<?>> Button<H> button(H htmlElement, Color color, boolean outline);
055        
056        ButtonGroup buttonGroup(boolean vertical);
057        
058        ButtonToolbar buttonToolbar();
059        
060        Dropdown dropdown(Button<?> button, boolean split, Direction direction);
061
062        InputGroup inputGroup();
063        
064        Table table(org.nasdanika.html.Table htmlTable);
065        
066        /**
067         * Wraps HTML row.
068         * @param row
069         * @return
070         */
071        RowContainer.Row row(org.nasdanika.html.RowContainer.Row htmlRow);
072
073        /**
074         * Wraps HTML cell.
075         * @param htmlCell
076         * @return
077         */
078        RowContainer.Row.Cell cell(org.nasdanika.html.RowContainer.Row.Cell htmlCell);
079        
080        
081        /**
082         * Creates a new HTML Table, wraps it into Bootstrap table and returns.
083         * @return
084         */
085        Table table();
086        
087        TagBootstrapElement progressBar(int percentage);
088        
089        /**
090         * 
091         * @param label
092         * @param input
093         * @param hint
094         * @param horizontalLabelWidths If not null and not empty, a horizontal form group is created as explained here - https://getbootstrap.com/docs/4.5/components/forms/#horizontal-form with
095         * label widths taken from the map and control widths computed as 12 - label width.  
096         * @return
097         */
098        FormGroup formGroup(Object label, Object input, Object hint, Map<Breakpoint, Size> horizontalLabelWidths);
099        
100        <H extends HTMLElement<?>> H tooltip(H htmlElement, Object tooltip, boolean html, Placement placement);
101        
102        <B extends BootstrapElement<?,?>> B tooltip(B bootstrapElement, Object tooltip, boolean html, Placement placement);
103        
104        /**
105         * @return Script to initialize tooltips.
106         */
107        Tag initTooltipScript();
108        
109        Navs navs();
110        
111        /**
112         * @deprecated Use navs().tabs()
113         * @return
114         */
115        @Deprecated
116        Navs tabs();
117
118        /**
119         * @deprecated Use navs().pills() 
120         * @return
121         */
122        @Deprecated
123        Navs pills();
124        
125        Container container();
126        
127        Container container(Tag div);
128        
129        Container fluidContainer();
130        
131        Container fluidContainer(Tag div);
132        
133        /**
134         * Adds bootstrap required meta tags to the page.
135         * @param htmlPage
136         * @return
137         */
138        <P extends HTMLPage> P bootstrapHTMLPage(P htmlPage);
139
140        /**
141         * Adds bootstrap required meta tags and stylesheet and script tags to load
142         * resources from CDN.
143         * @param htmlPage
144         * @return
145         */
146        <P extends HTMLPage> P bootstrapCdnHTMLPage(P htmlPage);
147        
148        /**
149         * Adds bootstrap required meta tags, default or Bootswatch stylesheet and script tags to load
150         * resources from CDN.
151         * @param htmlPage
152         * @param theme Bootswatch theme, uses default theme is null or Default.
153         * @return
154         */
155        <P extends HTMLPage> P bootstrapCdnHTMLPage(P htmlPage, Theme theme);           
156        
157        /**
158         * Creates an HTML page with meta tags required by Bootstrap.
159         * @return
160         */
161        HTMLPage bootstrapHTMLPage();
162        
163        /**
164         * Creates an HTML page with Bootstrap stylesheet and scripts served from CDN.
165         * @return
166         */
167        HTMLPage bootstrapCdnHTMLPage();
168        
169        /**
170         * Creates an HTML page with Bootstrap stylesheet and scripts served from CDN.
171         * @param theme Bootswatch theme. Uses the default Bootstrap theme if null or Default.
172         * @return
173         */
174        HTMLPage bootstrapCdnHTMLPage(Theme theme);             
175        
176        /**
177         * Creates a select with themes as options and an event handler to update the application theme on change.  
178         * @return
179         */
180        Select themeSelect(Theme selected);
181        
182        /**
183         * Adds "display-&lt;level&gt;" class.
184         * @param element
185         * @param level Display level from 1 (largest) to 4 (smallest).
186         * @return
187         */
188        <H extends HTMLElement<?>> H display(H element, int level);
189        
190        /**
191         * Creates a navbar
192         * @param expandSize
193         * @param dark
194         * @param background
195         * @param brand
196         * @return
197         */
198        Navbar navbar(Breakpoint expandSize, boolean dark, Color background, HTMLElement<?> brand);
199        
200        ListGroup listGroup(boolean flush);
201        
202        ActionGroup actionGroup(boolean flush);
203        
204        Card card();
205        
206        Modal modal();
