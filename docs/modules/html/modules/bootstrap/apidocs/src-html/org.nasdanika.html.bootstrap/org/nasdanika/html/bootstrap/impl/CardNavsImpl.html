003import org.nasdanika.html.Tag;
004import org.nasdanika.html.bootstrap.Card;
005
006public class CardNavsImpl implements Card.Navs {
007        
008        private static final String NAV_TABS = "nav-tabs";
009        private static final String NAV_PILLS = "nav-pills";
010        
011        private static final String CARD_HEADER_TABS = "card-header-tabs";
012        private static final String CARD_HEADER_PILLS = "card-header-pills";
013        
014        private Tag contentDiv;
015        private CardImpl card;
016        private Tag navsDiv;
017
018        protected CardNavsImpl(CardImpl card) {
019                this.card = card;
020                
021                navsDiv = card.getFactory().getHTMLFactory().nonEmptyDiv();
022                navsDiv.addClass("nav");
023                card.getHeader().toHTMLElement().content(navsDiv);
024                
025                contentDiv = card.getFactory().getHTMLFactory().nonEmptyDiv().addClass("tab-content");
026                card.getBody().toHTMLElement().content(contentDiv);
027        }
028
029        @Override
030        public Card.Navs item(Object name, boolean active, boolean disabled, Object contentId, Object... content) {
031                if (contentId == null) {
032                        contentId = "card-nav-"+card.getFactory().getHTMLFactory().nextId();
033                }
034                Tag navLink = card.getFactory().getHTMLFactory().link("#"+contentId, name)
035                                .addClass("nav-item", "nav-link")
036                                .addClassConditional(active, "active")
037                                .attribute("data-toggle", "tab")
038                                .addClassConditional(disabled, "disabled")
039                                .attribute("role", "tab");
040                navsDiv.content(navLink);
041
042                Tag cDiv = card.getFactory().getHTMLFactory().nonEmptyDiv(content)
043                                .id(contentId)
044                                .attribute("role", "tabpanel")
045                                .addClass("tab-pane", "fade")
046                                .addClassConditional(active, "show", "active");
047                
048                if (!cDiv.isEmpty()) {
049                        navsDiv.attribute("role", "tablist");
050                }
051                                
052                contentDiv.content(cDiv);               
053                return this;
054        }

071        @Override
072        public String toString() {
073                return card.toString();
074        }
075
076        @Override
077        public Card.Navs tabs(boolean tabs) {
078                navsDiv.addClassConditional(tabs, NAV_TABS);
079                navsDiv.addClassConditional(tabs, CARD_HEADER_TABS);
080                return this;
081        }
082
083        @Override
084        public Card.Navs pills(boolean pills) {
085                navsDiv.addClassConditional(pills, NAV_PILLS);
086                navsDiv.addClassConditional(pills, CARD_HEADER_PILLS);
087                return this;
088        }
