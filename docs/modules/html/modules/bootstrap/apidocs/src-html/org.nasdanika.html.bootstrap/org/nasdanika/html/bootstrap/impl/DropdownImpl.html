001package org.nasdanika.html.bootstrap.impl;
002
003import org.nasdanika.html.Form;
004import org.nasdanika.html.HTMLElement;
005import org.nasdanika.html.HTMLFactory;
006import org.nasdanika.html.Tag;
007import org.nasdanika.html.bootstrap.BootstrapElement;
008import org.nasdanika.html.bootstrap.BootstrapFactory;
009import org.nasdanika.html.bootstrap.Button;
010import org.nasdanika.html.bootstrap.Direction;
011import org.nasdanika.html.bootstrap.Dropdown;
012
013public class DropdownImpl extends DivWrappingBootstrapElementImpl<Dropdown> implements Dropdown {
014        
015        protected DropdownMenu menu;
016
017        public DropdownImpl(BootstrapFactory factory, Button<?> button, boolean split, Direction direction) {
018                super(factory);
019                htmlElement.addClass("btn-group");
020                HTMLFactory htmlFactory = getFactory().getHTMLFactory();
021                htmlElement.content(button);
022
023                switch (direction) {
024                case DOWN:
025                        // Default behavior
026                        break;
027                case LEFT:
028                case RIGHT:
029                case UP:
030                        htmlElement.addClass("drop"+direction.name().toLowerCase());
031                        break;
032                default:
033                        break;
034                
035                }
036                
037                HTMLElement<?> toggleHtmlElement = button.toHTMLElement();
038                
039                if (split) {
040                        org.nasdanika.html.Button splitHtmlButton = htmlFactory.button(htmlFactory.span("Toggle Dropdown").addClass("sr-only"))
041                                        .addClass("dropdown-toggle-split");
042                        Button<?> splitButton = getFactory().button(
043                                        splitHtmlButton, 
044                                        ((ButtonImpl<?>) button).getColor(), 
045                                        ((ButtonImpl<?>) button).isOutline()); 
046                        
047                        toggleHtmlElement = splitHtmlButton;
048                        htmlElement.content(splitButton);
049                }
050                
051                toggleHtmlElement
052                        .addClass("dropdown-toggle")
053                        .attribute("data-toggle", "dropdown")
054                        .attribute("aria-haspopup", "true")
055                        .attribute("aria-expanded", "false");
056
057                if (toggleHtmlElement.getId() == null) {
058                        toggleHtmlElement.id(htmlFactory.nextId());
059                }               
060                
061                menu = new DropdownMenu(factory, factory.wrap(toggleHtmlElement));
062                htmlElement.content(menu.toHTMLElement());
063        }
064        
065        @Override
066        public Dropdown item(HTMLElement<?> item, boolean active, boolean disabled) {
067                menu.item(item, active, disabled);
068                return this;
069        }
070
071        @Override
072        public Tag header(Object... content) {
073                return menu.header(content);
074        }
075
076        @Override
077        public Tag divider() {
078                return menu.divider();
079        }
080
081        @Override
082        public Dropdown form(Form form) {
083                menu.form(form);
084                return this;
085        }
086
087        @Override
088        public BootstrapElement<?, ?> getToggle() {
089                return menu.getToggle();
090        }
091        
092}
