001package org.nasdanika.html.bootstrap.impl;
002
003import org.nasdanika.html.Tag;
004import org.nasdanika.html.TagName;
005import org.nasdanika.html.bootstrap.ActionGroup;
006import org.nasdanika.html.bootstrap.BootstrapFactory;
007import org.nasdanika.html.bootstrap.Color;
008import org.nasdanika.html.bootstrap.Container;
009import org.nasdanika.html.bootstrap.Container.Row;
010import org.nasdanika.html.bootstrap.Breakpoint;
011import org.nasdanika.html.bootstrap.Size;
012
013public class ActionGroupImpl extends DivWrappingBootstrapElementImpl<ActionGroup> implements ActionGroup {
014
015        private Tag contentDiv;
016        
017        protected ActionGroupImpl(BootstrapFactory factory, boolean flush) {
018                super(factory);
019                htmlElement.addClass("list-group").addClassConditional(flush, "list-group-flush");
020                
021                contentDiv = factory.getHTMLFactory().nonEmptyDiv().addClass("tab-content");            
022        }
023
024        @Override
025        public boolean isEmpty() {
026                return htmlElement.isEmpty();
027        }
028        
029        @Override
030        public void item(Object name, Object content) {
031                contentAction(name, isEmpty(), false, Color.DEFAULT, null, content);            
032        }
033
034        @Override
035        public Tag action(boolean active, boolean disabled, Color color, Object href, Object... name) {
036                Tag a = configureLink(getFactory().getHTMLFactory().tag(TagName.a, name).attribute("href", href), active, disabled, color);
037                htmlElement.content(a);
038                return a;
039        }
040        
041        protected Tag configureLink(Tag a, boolean active, boolean disabled, Color color) {
042                boolean hasColor = color != null && color.code != null;
043                return a.addClass("list-group-item", "list-group-action")                               
044                                .addClassConditional(active, "active")
045                                .addClassConditional(disabled, "disabled")
046                                .addClassConditional(hasColor, hasColor ? "list-group-item-"+color.code : null);
047        }
048
049        @Override
050        public Tag contentAction(
051                        Object name, 
052                        boolean active, 
053                        boolean disabled, 
054                        Color color, 
055                        Object contentId,
056                        Object... content) {
057                htmlElement.attribute("role", "tablist");
058                if (contentId == null) {
059                        contentId = "list-group-action-"+getFactory().getHTMLFactory().nextId();
060                }
061                Tag a = configureLink(getFactory().getHTMLFactory().tag(TagName.a, name).attribute("href", "#"+contentId), active, disabled, color);
062                a.attribute("data-toggle", "list").attribute("role", "tab");
063                htmlElement.content(a);
064                
065                Tag cDiv = getFactory().getHTMLFactory().nonEmptyDiv(content)
066                                .id(contentId)
067                                .attribute("role", "tabpanel")
068                                .addClass("tab-pane", "fade")
069                                .addClassConditional(active, "show", "active");                         
070                                
071                contentDiv.content(cDiv);                               
072                
073                return a;
074        }
075
076        @Override
077        public Tag getContentDiv() {
078                return contentDiv;
079        }
080
081        @Override
082        public Container asContainer(boolean fluid) {
083                BootstrapFactory factory = getFactory();
084                Container ret = fluid ? factory.fluidContainer() : factory.container();
085                Row row = ret.row();
086                row.col(htmlElement).width(Breakpoint.DEFAULT, Size.AUTO);
087                row.col(contentDiv).width(Breakpoint.DEFAULT, Size.NONE);
088                return ret;
089        }
090        
091}
