001package org.nasdanika.html.bootstrap.impl;
002
003import org.nasdanika.html.Tag;
004import org.nasdanika.html.bootstrap.BootstrapFactory;
005import org.nasdanika.html.bootstrap.Card;
006import org.nasdanika.html.bootstrap.TagBootstrapElement;
007
008public class CardImpl extends WrappingBootstrapElementImpl<Tag, Card> implements Card {
009
010        private TagBootstrapElement header;
011        private TagBootstrapElement body;
012        private TagBootstrapElement footer;
013        private CardNavsImpl navs;
014
015        public CardImpl(BootstrapFactory factory) {
016                super(factory, factory.getHTMLFactory().nonEmptyDiv());
017                htmlElement.addClass("card");
018                
019                header = getFactory().wrap(getFactory().getHTMLFactory().nonEmptyDiv());
020                header.toHTMLElement().addClass("card-header");
021                htmlElement.content(header.toHTMLElement());                                                            
022                
023                body = getFactory().wrap(getFactory().getHTMLFactory().nonEmptyDiv());
024                body.toHTMLElement().addClass("card-body");
025                htmlElement.content(body.toHTMLElement());
026                
027                footer = getFactory().wrap(getFactory().getHTMLFactory().nonEmptyDiv());
028                footer.toHTMLElement().addClass("card-footer");
029                htmlElement.content(footer.toHTMLElement());            
030        }
031
032        @Override
033        public TagBootstrapElement getHeader() {
034                return header;
035        }
036
037        @Override
038        public TagBootstrapElement getBody() {
039                return body;
040        }
041
042        @Override
043        public TagBootstrapElement getFooter() {
044                return footer;
045        }
046
047        @Override
048        public Navs asNavs() {
049                if (navs == null) {
050                        navs = new CardNavsImpl(this);
051                }
052                return navs;
053        }
054
055}
