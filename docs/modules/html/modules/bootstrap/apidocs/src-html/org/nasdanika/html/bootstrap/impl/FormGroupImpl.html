001package org.nasdanika.html.bootstrap.impl;
002
003import java.util.Map;
004
005import org.nasdanika.html.Container;
006import org.nasdanika.html.HTMLElement;
007import org.nasdanika.html.Input;
008import org.nasdanika.html.InputType;
009import org.nasdanika.html.Tag;
010import org.nasdanika.html.TagName;
011import org.nasdanika.html.bootstrap.BootstrapFactory;
012import org.nasdanika.html.bootstrap.Breakpoint;
013import org.nasdanika.html.bootstrap.FormGroup;
014import org.nasdanika.html.bootstrap.Size;
015
016public class FormGroupImpl extends DivWrappingBootstrapElementImpl<FormGroup> implements FormGroup {
017
018        private Object input;
019        private Container<?> inputContainer;
020
021        @SuppressWarnings("unchecked")
022        public FormGroupImpl(BootstrapFactory factory, Object label, Object input, Object hint, Map<Breakpoint, Size> horizontalLabelWidths) {
023                super(factory);
024                this.input = input;
025                boolean isHorizontal = horizontalLabelWidths != null && !horizontalLabelWidths.isEmpty();
026                htmlElement.addClass("form-group").addClassConditional(isHorizontal, "row");
027                // Possibly keep and expose the label
028                Tag labelTag = label==null ? null : factory.getHTMLFactory().nonEmptyTag(TagName.label, label).addClassConditional(isHorizontal, "col-form-label");
029                if (isHorizontal && labelTag != null) {
030                        horizontalLabelWidths.entrySet().forEach(horizontalLabelWidth -> {
031                                labelTag.addClass(horizontalLabelWidth.getKey().size("col", horizontalLabelWidth.getValue()));
032                        });
033                }
034                
035                if (isHorizontal) {
036                        if (label != null) {
037                                htmlElement.content(labelTag);
038                        }                               
039                        Tag controlDiv = getFactory().getHTMLFactory().div();
040                        horizontalLabelWidths.entrySet().forEach(horizontalLabelWidth -> {
041                                controlDiv.addClass(horizontalLabelWidth.getKey().size("col", horizontalLabelWidth.getValue().complementary()));
042                        });
043                        
044                        htmlElement.content(controlDiv);
045                        if (input instanceof Input && (((Input) input).getType() == InputType.checkbox || ((Input) input).getType() == InputType.radio)) {                              
046                                ((Input) input).addClass("form-check-input");   
047                                Tag formCheckDiv = getFactory().getHTMLFactory().div();
048                                controlDiv.content(formCheckDiv);
049                                formCheckDiv.addClass("form-check");
050                                formCheckDiv.content(input);
051                                inputContainer = formCheckDiv;
052                                if (label == null) {
053                                        ((Input) input).addClass("position-static");
054                                }                               
055                                if (hint != null) {
056                                        formCheckDiv.content(factory.getHTMLFactory().nonEmptyTag(TagName.label, hint).addClass("form-check-label"));                   
057                                }
058                        } else {
059                                inputContainer = controlDiv;
060                                if (input instanceof HTMLElement) {
061                                        ((HTMLElement<Input>) input).addClass("form-control");
062                                }
063                                controlDiv.content(input);                              
064                                
065                                if (hint != null) {
066                                        controlDiv.content(factory.getHTMLFactory().nonEmptyTag(TagName.small, hint).addClass("form-text", "text-muted"));                      
067                                }
068                        }
069                } else {        
070                        inputContainer = htmlElement;
071                        if (input instanceof Input && (((Input) input).getType() == InputType.checkbox || ((Input) input).getType() == InputType.radio)) {
072                                ((HTMLElement<Input>) input).addClass("form-check-input");                      
073                                htmlElement.addClass("form-check");
074                                if (label == null) {
075                                        ((HTMLElement<Input>) input).addClass("position-static");
076                                } else {
077                                        htmlElement.content(labelTag.addClass("form-check-label"));
078                                }
079                        } else {
080                                if (label != null) {
081                                        htmlElement.content(labelTag);
082                                }
083                                if (input instanceof HTMLElement<?>) {
084                                        ((HTMLElement<Input>) input).addClass("form-control");
085                                }
086                                htmlElement.content(input);
087                        }
088                        
089                        if (hint != null) {
090                                htmlElement.content(factory.getHTMLFactory().nonEmptyTag(TagName.small, hint).addClass("form-text", "text-muted"));                     
091                        }
092                }
093        }
094
095        @Override
096        public FormGroup large() {
097                return large(true);
098        }
099
100        @SuppressWarnings("unchecked")
101        @Override
102        public FormGroup large(boolean large) {
103                if (input instanceof HTMLElement) {
104                        ((HTMLElement<Input>) input).addClassConditional(large, "form-control-lg");
105                }
106                return this;
107        }
108
109        @Override
110        public FormGroup small() {
111                return small(true);
112        }
113
114        @SuppressWarnings("unchecked")
115        @Override
116        public FormGroup small(boolean small) {
117                if (input instanceof HTMLElement) {
118                        ((HTMLElement<Input>) input).addClassConditional(small, "form-control-sm");
119                }
120                return this;
121        }
122
123        @Override
124        public FormGroup plainText() {
125                return plainText(true);
126        }
127
128        @SuppressWarnings("unchecked")
129        @Override
130        public FormGroup plainText(boolean plainText) {
131                if (input instanceof HTMLElement) {
132                        ((HTMLElement<Input>) input).addClassConditional(plainText, "form-control-plaintext");
133                }
134                return this;
135        }
136
137        @Override
138        public FormGroup inline() {
139                return inline(true);
140        }
141
142        @Override
143        public FormGroup inline(boolean inline) {
144                htmlElement.addClassConditional(inline, "form-check-inline");
145                return this;
146        }
147
148        @SuppressWarnings("unchecked")
149        @Override
150        public FormGroup valid(Object... feedback) {
151                if (input instanceof HTMLElement) {
152                        ((HTMLElement<Input>) input).addClass("is-valid");
153                }
154                inputContainer.content(getFactory().getHTMLFactory().nonEmptyDiv(feedback).addClass("valid-feedback"));
155                return this;
156        }
157
158        @SuppressWarnings("unchecked")
159        @Override
160        public FormGroup invalid(Object... feedback) {
161                if (input instanceof HTMLElement) {
162                        ((HTMLElement<Input>) input).addClass("is-invalid");
163                }
164                inputContainer.content(getFactory().getHTMLFactory().nonEmptyDiv(feedback).addClass("invalid-feedback"));
165                return this;
166        }
167
168}
