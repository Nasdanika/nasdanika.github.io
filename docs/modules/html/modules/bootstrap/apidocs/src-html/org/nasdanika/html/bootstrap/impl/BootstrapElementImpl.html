001package org.nasdanika.html.bootstrap.impl;
002
003import org.nasdanika.html.HTMLElement;
004import org.nasdanika.html.bootstrap.BootstrapElement;
005import org.nasdanika.html.bootstrap.BootstrapFactory;
006import org.nasdanika.html.bootstrap.Color;
007import org.nasdanika.html.bootstrap.Breakpoint;
008import org.nasdanika.html.bootstrap.Placement;
009import org.nasdanika.html.bootstrap.Spacing;
010import org.nasdanika.html.bootstrap.Text;
011
012public abstract class BootstrapElementImpl<H extends HTMLElement<?>,B extends BootstrapElement<H,?>> implements BootstrapElement<H,B> {
013        
014        private BootstrapFactory factory;
015
016        protected BootstrapElementImpl(BootstrapFactory factory) {
017                this.factory = factory;
018        }
019        
020        @Override
021        public BootstrapFactory getFactory() {
022                return factory;
023        }
024        
025        @Override
026        public String toString() {
027                return toHTMLElement().toString();
028        }
029        
030        @Override
031        public Object produce(int indent) {
032                return toHTMLElement().produce(indent);
033        }
034        
035        @SuppressWarnings("unchecked")
036        @Override
037        public B border(Color color, Placement... placement) {
038                H htmlElement = toHTMLElement();
039                if (htmlElement != null && color != Color.NONE) {
040                        for (Placement p: placement) {
041                                htmlElement.addClass("border-"+p.name().toLowerCase());
042                        }
043                        htmlElement.addClassConditional(placement.length == 0, "border");
044                        if (color != null && color.code != null) {
045                                htmlElement.addClass("border-"+color.code);
046                        }
047                }               
048                return (B) this;
049        }
050        
051        @SuppressWarnings("unchecked")
052        public B background(Color color) {
053                toHTMLElement().addClassConditional(color != null && color.code != null, "bg-"+color.code);
054                return (B) this;
055        }
056        
057        @Override
058        public Text<B> text() {
059                return new Text<B>() {
060
061                        @SuppressWarnings("unchecked")
062                        @Override
063                        public B toBootstrapElement() {
064                                return (B) BootstrapElementImpl.this;
065                        }
066
067                        @Override
068                        public Text<B> color(Color color) {
069                                toHTMLElement().addClassConditional(color != null && color.code != null, "text-"+color.code);
070                                return this;
071                        }
072
073                        @Override
074                        public Text<B> alignment(Alignment alignment) {
075                                toHTMLElement().addClass("text-"+alignment.name().toLowerCase());
076                                return this;
077                        }
078
079                        @Override
080                        public Text<B> alignment(Breakpoint breakpoint, Alignment alignment) {
081                                String code = breakpoint.code;
082                                String prefix = code.length() == 0 ? "text" : "text-"+code;                             
083                                toHTMLElement().addClass(prefix+"-"+alignment.name().toLowerCase());
084                                return this;
085                        }
086
087                        @Override
088                        public Text<B> transform(Transform transform) {
089                                toHTMLElement().addClass("text-"+transform.name().toLowerCase());
090                                return this;
091                        }
092
093                        @Override
094                        public Text<B> weight(Weight weight) {
095                                toHTMLElement().addClass("font-weight-"+weight.name().toLowerCase());
096                                return this;
097                        }
098
099                        @Override
100                        public Text<B> monospace() {
101                                return monospace(true);
102                        }
103
104                        @Override
105                        public Text<B> monospace(boolean monospace) {
106                                toHTMLElement().addClassConditional(monospace, "text-monospace");
107                                return this;
108                        }
109
110                        @Override
111                        public Text<B> italic() {
112                                return italic(true);
113                        }
114
115                        @Override
116                        public Text<B> italic(boolean italic) {
117                                toHTMLElement().addClassConditional(italic, "font-italic");
118                                return this;
119                        }
120
121                        @Override
122                        public Text<B> nowrap() {
123                                return nowrap(true);
124                        }
125
126                        @Override
127                        public Text<B> nowrap(boolean nowrap) {
128                                toHTMLElement().addClassConditional(nowrap, "text-nowrap");
129                                return this;
130                        }
131
132                        @Override
133                        public Text<B> truncate() {
134                                return truncate(true);
135                        }
136
137                        @Override
138                        public Text<B> truncate(boolean truncate) {
139                                toHTMLElement().addClassConditional(truncate, "text-truncate");
140                                return this;
141                        }
142                        
143                };
144        }
145        
146        @SuppressWarnings("unchecked")
147        @Override
148        public org.nasdanika.html.bootstrap.Float<B> _float() {
149                return new FloatImpl<B>((B) this);
150        }
151        
152        @SuppressWarnings("unchecked")
153        @Override
154        public Spacing<B> margin() {
155                return new SpacingImpl<B>((B) this, "m");
156        }
157
158        @SuppressWarnings("unchecked")
159        @Override
160        public Spacing<B> padding() {
161                return new SpacingImpl<B>((B) this, "p");
162        }
163
164}
