001package org.nasdanika.html.bootstrap;
002
003import org.nasdanika.common.Util;
004
005public enum Breakpoint {
006        
007        DEFAULT("", ""),
008        SMALL("sm", "Small"),
009        MEDIUM("md", "Medium"),
010        LARGE("lg", "Large"),
011        EXTRA_LARGE("xl", "Extra large");
012        
013        Breakpoint(String code, String label) {
014                this.code = code;
015                this.label = label;
016        }
017        
018        public final String code;
019        public final String label;
020        
021        /**
022         * @return width string
023         */
024        public String size(String prefix, Size size) {
025                StringBuilder ret = new StringBuilder(prefix);
026                if (code.length() > 0) {
027                        if (ret.length() > 0) {
028                                ret.append("-");
029                        }
030                        ret.append(code);
031                }
032                if (size.code.length() > 0) {
033                        if (ret.length() > 0) {
034                                ret.append("-");
035                        }
036                        ret.append(size.code);
037                }
038                return ret.toString();
039        };
040                
041        public static Breakpoint fromCode(String code) {
042                if (Util.isBlank(code)) {
043                        return DEFAULT;
044                }
045                for (Breakpoint breakpoint: values()) {
046                        if (breakpoint.code.equals(code)) {
047                                return breakpoint;
048                        }
049                }
050                throw new IllegalArgumentException("No breakpoint value for code "+code);
051        }
052                
053        public static Breakpoint fromLabel(String label) {
054                if (label == null || label.trim().length() == 0) {
055                        return DEFAULT;
056                }
057                
058                for (Breakpoint breakpoint: values()) {
059                        if (breakpoint.label.equals(label)) {
060                                return breakpoint;
061                        }
062                }
063                throw new IllegalArgumentException("No breakpoint value for label "+label);
064        }
065
066}
