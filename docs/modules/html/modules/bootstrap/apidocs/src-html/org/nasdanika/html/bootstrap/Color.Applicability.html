001package org.nasdanika.html.bootstrap;
002
003import java.util.ArrayList;
004import java.util.Arrays;
005import java.util.List;
006import java.util.stream.Collectors;
007
008import org.nasdanika.common.Util;
009
010/**
011 * Bootstrap colors. Not all colors are applicable in all situations. E.g. TRANSPARENT is applicable for backgrounds.
012 * @author Pavel Vlasov
013 *
014 */
015public enum Color {
016                        
017        /**
018         * No color - for borders means that there should be no border. 
019         * For other uses is the same as DEFAULT
020         */
021        NONE(null, ""),
022        /**
023         * Default color - no specific.
024         */
025        DEFAULT(null, ""),
026        PRIMARY("primary", "Primary"),
027        SECONDARY("secondary", "Secondary"),
028        SUCCESS("success", "Success"),
029        DANGER("danger", "Danger"),
030        WARNING("warning", "Warning"),
031        INFO("info", "Info"),
032        LIGHT("light", "Light"),
033        DARK("dark", "Dark"),
034        BODY("body", "Body", Applicability.TEXT),
035        MUTED("muted", "Muted", Applicability.TEXT),
036        WHITE("white", "White"),
037        BLACK50("black-50", "Black 50%", Applicability.TEXT),
038        WHITE50("white-50", "White 50%", Applicability.TEXT),
039        TRANSPARENT("transparent", "Transparent", Applicability.BACKGROUND);
040        
041        /**
042         * Defines where the color can be applied.
043         * @author Pavel
044         *
045         */
046        public static enum Applicability {
047                BACKGROUND,
048                BORDER,
049                TEXT;
050                
051                /**
052                 * @return Applicable colors;
053                 */
054                public List<Color> colors() {
055                        return Arrays.stream(Color.values()).filter(c -> c.isApplicable(this)).collect(Collectors.toList());
056                }
057                
058                /**
059                 * @return Labels of applicable colors with suppressed duplicates - only one empty string for both NONE and DEFAULT.
060                 */
061                public List<Object> colorLabels() {
062                        List<Object> ret = new ArrayList<>();
063                        for (Color c: colors()) {
064                                if (!ret.contains(c.label)) {
065                                        ret.add(c.label);
066                                }
067                        }               
068                        return ret;
069                }
070                
071        }
072        
073        Color(String code, String label, Applicability... applicability) {
074                this.code = code;
075                this.label = label;
076                this.applicabilities = applicability;
077        }
078        
079        public final String code;
080        public final String label;
081        private final Applicability[] applicabilities; 
082                
083        public static Color fromLabel(String label) {
084                for (Color color: values()) {
085                        if (color.label.equals(label)) {
086                                return color;
087                        }
088                }
089                throw new IllegalArgumentException("No color value for label "+label);
090        }
091        
092        public static Color fromCode(String code) {
093                if (Util.isBlank(code)) {
094                        return NONE;
095                }
096                for (Color color: values()) {
097                        if (color.code != null && color.code.equals(code)) {
098                                return color;
099                        }
100                }
101                throw new IllegalArgumentException("No color value for code "+code);
102        }
103        
104        public boolean isApplicable(Applicability applicability) {
105                if (applicabilities.length == 0) {
106                        return true;
107                }
108                for (Applicability a: applicabilities) {
109                        if (a == applicability) {
110                                return true;
111                        }
112                }
113                return false;
114        }
115
116}
