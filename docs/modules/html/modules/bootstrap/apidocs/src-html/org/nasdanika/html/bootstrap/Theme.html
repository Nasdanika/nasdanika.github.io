004
005/**
006 * Enumeration for out-of-the-box and Bootswatch themes.
007 * @author Pavel Vlasov
008 *
009 */
010public enum Theme {
011        /** Default Bootstrap theme **/
012        Default("https://cdn.jsdelivr.net/npm/bootstrap@4.5.2/dist/css/bootstrap.min.css"),
013        /** Bootswatch theme **/
014        Cerulean,
015        /** Bootswatch theme **/
016        Cosmo,
017        /** Bootswatch theme **/
018        Cyborg,
019        /** Bootswatch theme **/
020        Darkly,
021        /** Bootswatch theme **/
022        Flatly,
023        /** Bootswatch theme **/
024        Journal,
025        /** Bootswatch theme **/
026        Litera,
027        /** Bootswatch theme **/
028        Lumen,
029        /** Bootswatch theme **/
030        Lux,
031        /** Bootswatch theme **/
032        Materia,
033        /** Bootswatch theme **/
034        Minty,  
035        /** Bootswatch theme **/
036        Pulse,
037        /** Bootswatch theme **/
038        Sandstone,
039        /** Bootswatch theme **/
040        Simplex,
041        /** Bootswatch theme **/
042        Sketchy,
043        /** Bootswatch theme **/
044        Slate,
045        /** Bootswatch theme **/
046        Solar,
047        /** Bootswatch theme **/
048        Spacelab,
049        /** Bootswatch theme **/
050        Superhero,
051        /** Bootswatch theme **/
052        United,
053        /** Bootswatch theme **/
054        Yeti;
055        
056        /**
057         * id of bootstrap stylesheet link head tag.
058         */
059        public static final String STYLESHEET_ID = "nsd-bootstrap-theme-stylesheet";
060        
061        /**
062         * CDN location of theme stylesheet.
063         */
064        public final String stylesheetCdnURL;
065        
066        /**
067         * Adds CDN stylesheet declaration to the page.
068         * @param page
069         * @return
070         */
071        public <P extends HTMLPage> P cdn(P page) {
072                page.stylesheet(stylesheetCdnURL).id(STYLESHEET_ID);
073                return page;
074        }
075        private Theme() {
076                stylesheetCdnURL = "https://stackpath.bootstrapcdn.com/bootswatch/4.5.2/"+name().toLowerCase()+"/bootstrap.min.css";
077        }
078        
079        private Theme(String stylesheetCdnURL) {
080                this.stylesheetCdnURL = stylesheetCdnURL;
081        }
082}
