001package org.nasdanika.html.bootstrap;
002
003/**
004 * Size enumeration for widths and spacings. Spacings valid values are S1..S5.
005 * @author Pavel
006 *
007 */
008public enum Size {
009        
010        NONE("", true),
011        S0("0", true),
012        S1("1", true),
013        S2("2", true),
014        S3("3", true),
015        S4("4", true),
016        S5("5", true),
017        S6("6", false),
018        S7("7", false),
019        S8("8", false),
020        S9("9", false),
021        S10("10", false),
022        S11("11", false),
023        S12("12", false),
024        AUTO("auto", true);
025        
026        Size(String code, boolean spacing) {
027                this.code = code;
028                this.spacing = spacing;
029        }               
030        
031        /**
032         * Size code.
033         */
034        public final String code;
035        
036        /**
037         * If true, this size is applicable for {@link Spacing}.
038         */
039        private final boolean spacing;
040        
041        /**
042         * @return true if this size is applicable for {@link Spacing} - 0..5, auto.
043         */
044        public boolean isSpacing() {
045                return spacing;
046        }
047        
048        public static Size fromCode(String code) {
049                if (code == null) {
050                        return NONE;
051                }
052                for (Size size: values()) {
053                        if (size.code.equals(code)) {
054                                return size;
055                        }
056                }
057                throw new IllegalArgumentException("No size value for code "+code);
058        }
059        
060        /**
061         * For numeric sizes returns size complementing to 12
062         * @param size
063         * @return
064         */
065        public Size complementary() {
066                switch (this) {
067                case S0:
068                        return S12;
069                case S1:
070                        return S11;
071                case S2:
072                        return S10;
073                case S3:
074                        return S9;
075                case S4:
076                        return S8;
077                case S5:
078                        return S7;
079                case S6:
080                        return S6;
081                case S7:
082                        return S5;
083                case S8:
084                        return S4;
085                case S9:
086                        return S3;
087                case S10:
088                        return S2;
089                case S11:
090                        return S1;
091                case S12:
092                        return S0;
093                case AUTO:
094                case NONE:
095                default:
096                        return AUTO;
097                }
098        }
099
100}
