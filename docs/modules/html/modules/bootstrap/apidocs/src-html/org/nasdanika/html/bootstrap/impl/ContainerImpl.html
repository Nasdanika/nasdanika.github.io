001package org.nasdanika.html.bootstrap.impl;
002
003import org.nasdanika.html.Tag;
004import org.nasdanika.html.bootstrap.BootstrapFactory;
005import org.nasdanika.html.bootstrap.Container;
006import org.nasdanika.html.bootstrap.Breakpoint;
007import org.nasdanika.html.bootstrap.Size;
008
009public class ContainerImpl extends WrappingBootstrapElementImpl<Tag,Container> implements Container {
010
011        public ContainerImpl(BootstrapFactory factory, Tag htmlElement, boolean fluid) {
012                super(factory, htmlElement);
013                htmlElement
014                        .addClassConditional(!fluid, "container")
015                        .addClassConditional(fluid, "container-fluid");
016                
017        }
018        
019        class RowImpl extends DivWrappingBootstrapElementImpl<Row> implements Row {
020
021                protected RowImpl(BootstrapFactory factory) {
022                        super(factory, true);
023                        htmlElement.addClass("row");
024                }
025                
026                class ColImpl extends DivWrappingBootstrapElementImpl<Col> implements Col {
027
028                        protected ColImpl(BootstrapFactory factory, Object... content) {
029                                super(factory, true);
030                                htmlElement.addClass("col");
031                                htmlElement.content(content);
032                        }
033
034                        @Override
035                        public Col content(Object... content) {
036                                htmlElement.content(content);
037                                return this;
038                        }
039
040                        @Override
041                        public boolean isEmpty() {
042                                return htmlElement.isEmpty();
043                        }
044
045                        @Override
046                        public Col width(Breakpoint breakpoint, Size width) {
047                                htmlElement.removeClass("col");
048                                htmlElement.addClass(breakpoint.size("col", width));
049                                return this;
050                        }                       
051                        
052                }
053
054                @Override
055                public Col col(Object... content) {
056                        Col ret = new ColImpl(getFactory(), content);
057                        htmlElement.content(ret.toHTMLElement());
058                        return ret;
059                }
060                
061        }
062
063        @Override
064        public Row row() {
065                Row ret = new RowImpl(getFactory());
066                htmlElement.content(ret.toHTMLElement());
067                return ret;
068        }
069
070}
