001package org.nasdanika.html.bootstrap.impl;
002
003import org.nasdanika.html.Fragment;
004import org.nasdanika.html.HTMLFactory;
005import org.nasdanika.html.Tag;
006import org.nasdanika.html.TagName;
007import org.nasdanika.html.bootstrap.BootstrapFactory;
008import org.nasdanika.html.bootstrap.Dropdown;
009import org.nasdanika.html.bootstrap.Navs;
010
011public class NavsImpl extends WrappingBootstrapElementImpl<Tag,Navs> implements Navs {
012        
013        private static final String NAV_TABS = "nav-tabs";
014        private static final String NAV_PILLS = "nav-pills";
015        private Fragment fragment;
016        private Tag contentDiv;
017        private boolean hasContent;
018
019        protected NavsImpl(BootstrapFactory factory) {
020                super(factory, factory.getHTMLFactory().nonEmptyDiv());
021                htmlElement.addClass("nav");
022                
023                contentDiv = factory.getHTMLFactory().nonEmptyDiv().addClass("tab-content");
024                fragment = factory.getHTMLFactory().fragment(htmlElement, contentDiv);
025        }
026
027        @Override
028        public void item(Object name, Object content) {
029                item(name, isEmpty(), false, null, content);
030                
031        }
032
033        @Override
034        public boolean isEmpty() {
035                return htmlElement.isEmpty();
036        }
037
038        @Override
039        public Navs item(Object name, boolean active, boolean disabled, Object contentId, Object... content) {
040                if (contentId == null) {
041                        contentId = "nav-"+getFactory().getHTMLFactory().nextId();
042                }
043                Tag navLink = getFactory().getHTMLFactory().link("#"+contentId, name)
044                                .addClass("nav-item", "nav-link")
045                                .addClassConditional(active, "active")
046                                .attribute("data-toggle", "tab")
047                                .addClassConditional(disabled, "disabled")
048                                .attribute("role", "tab");
049                htmlElement.content(navLink);
050
051                Tag cDiv = getFactory().getHTMLFactory().nonEmptyDiv(content)
052                                .id(contentId)
053                                .attribute("role", "tabpanel")
054                                .addClass("tab-pane", "fade")
055                                .addClassConditional(active, "show", "active");
056                
057                if (!cDiv.isEmpty()) {
058                        htmlElement.attribute("role", "tablist");
059                        hasContent = true;
060                }
061                                
062                contentDiv.content(cDiv);               
063                return this;
064        }
065        
066        @Override
067        public Tag item(Object name, Object href, boolean active, boolean disabled) {
068                Tag navLink = getFactory().getHTMLFactory().link(href, name)
069                                .addClass("nav-item", "nav-link")
070                                .addClassConditional(active, "active")
071                                .addClassConditional(disabled, "disabled");
072                htmlElement.content(navLink);
073                return navLink;
074        }
075        
076        @Override
077        public Dropdown dropdown(boolean active, Object... name) {
078                HTMLFactory htmlFactory = getFactory().getHTMLFactory();
079                Tag toggle = htmlFactory.link("#", name)
080                                .addClass("nav-link", "dropdown-toggle")
081                                .attribute("role", "button")
082                                .attribute("data-toggle", "dropdown");          
083                DropdownMenu menu = new DropdownMenu(getFactory(), getFactory().wrap(toggle));
084                Tag li = htmlFactory.tag(TagName.li, toggle, menu.toHTMLElement())
085                                .addClass("nav-item", "dropdown")
086                                .addClassConditional(active, "active");
087                htmlElement.content(li); 
088                return menu;
089        }       
090
091        @Override
092        public Object produce(int indent) {
093                return (hasContent ? fragment : htmlElement).produce(indent);
094        }
095        
096        @Override
097        public String toString() {
098                return (hasContent ? fragment : htmlElement).toString();
099        }
100
101        @Override
102        public Tag getContentDiv() {
103                return contentDiv;
104        }
105
106        @Override
107        public Navs tabs(boolean tabs) {
108                htmlElement.addClassConditional(tabs, NAV_TABS);
109                return this;
110        }
111
112        @Override
113        public Navs pills(boolean pills) {
114                htmlElement.addClassConditional(pills, NAV_PILLS);
115                return this;
116        }
117
118        @Override
119        public Navs content(Object... content) {
120                htmlElement.content(content);
121                return this;
122        }
123
124}
