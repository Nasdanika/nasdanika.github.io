001package org.nasdanika.html.bootstrap.impl;
002
003import org.nasdanika.html.Tag;
004import org.nasdanika.html.TagName;
005import org.nasdanika.html.bootstrap.BootstrapFactory;
006import org.nasdanika.html.bootstrap.Color;
007import org.nasdanika.html.bootstrap.ListGroup;
008
009public class ListGroupImpl extends WrappingBootstrapElementImpl<Tag, ListGroup> implements ListGroup {
010
011        protected ListGroupImpl(BootstrapFactory factory, boolean flush) {
012                super(factory, factory.getHTMLFactory().tag(TagName.ul));
013                htmlElement.addClass("list-group").addClassConditional(flush, "list-group-flush");
014        }
015
016        @Override
017        public boolean isEmpty() {
018                return htmlElement.isEmpty();
019        }
020
021        @Override
022        public Tag item(boolean active, boolean disabled, Color color, Object... content) {
023                Tag li = getFactory().getHTMLFactory().tag(TagName.li, content)
024                                .addClass("list-group-item")
025                                .addClassConditional(active, "active")
026                                .addClassConditional(disabled, "disabled")
027                                .addClassConditional(color != null && color.code != null, "list-group-item-"+color.code);
028                htmlElement.content(li);
029                return li;
030        }
031
032}
