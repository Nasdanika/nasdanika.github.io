001package org.nasdanika.html.bootstrap;
002
003/**
004 * Interface for styling text.
005 * @author Pavel Vlasov
006 *
007 */
008public interface Text<B> {
009        
010        B toBootstrapElement();
011        
012        Text<B> color(Color color);
013        
014        enum Alignment {
015                LEFT, CENTER, RIGHT, JUSTIFY
016        }
017        
018        Text<B> alignment(Alignment alignment);
019
020        Text<B> alignment(Breakpoint breakpoint, Alignment alignment);
021                
022        enum Transform {
023                LOWERCASE, UPPERCASE, CAPITALIZE
024        }
025
026        Text<B> transform(Transform transform);
027        
028        enum Weight {
029                LIGHT, NORMAL, BOLD
030        }
031        
032        Text<B> weight(Weight weight);
033        
034        Text<B> monospace();
035        
036        Text<B> monospace(boolean monospace);
037        
038        Text<B> italic();
039        
040        Text<B> italic(boolean italic);
041        
042        Text<B> nowrap();
043        
044        Text<B> nowrap(boolean nowrap);
045        
046        Text<B> truncate();
047        
048        Text<B> truncate(boolean truncate);
049        
050}
