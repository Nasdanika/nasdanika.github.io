001package org.nasdanika.html.bootstrap.impl;
002
003import org.nasdanika.html.Tag;
004import org.nasdanika.html.TagName;
005import org.nasdanika.html.bootstrap.BootstrapFactory;
006import org.nasdanika.html.bootstrap.Breadcrumb;
007
008public class BreadcrumbImpl extends WrappingBootstrapElementImpl<Tag,Breadcrumb> implements Breadcrumb {
009        
010        private Tag ol;
011
012        public BreadcrumbImpl(BootstrapFactory factory) {
013                super(factory, factory.getHTMLFactory().tag("nav").attribute("aria-label", "breadcrumb"));
014                ol = getFactory().getHTMLFactory().tag(TagName.ol).addClass("breadcrumb");
015                htmlElement.content(ol);
016        }
017
018        @Override
019        public boolean isEmpty() {
020                return ol.isEmpty();
021        }
022
023        @Override
024        public Tag item(boolean active, Object... content) {
025                Tag liTag = getFactory().getHTMLFactory()
026                                .tag(TagName.li, content)
027                                .addClass("breadcrumb-item")
028                                .addClassConditional(active, "active");
029                ol.content(liTag);
030                return liTag;
031        }
032        
033}
