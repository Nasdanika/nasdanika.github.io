001package org.nasdanika.html.bootstrap.impl;
002
003import org.nasdanika.html.HTMLElement;
004import org.nasdanika.html.Tag;
005import org.nasdanika.html.TagName;
006import org.nasdanika.html.bootstrap.BootstrapFactory;
007import org.nasdanika.html.bootstrap.Button;
008import org.nasdanika.html.bootstrap.Color;
009
010public class ButtonImpl<H extends HTMLElement<?>> extends WrappingBootstrapElementImpl<H,Button<H>> implements Button<H> {
011
012        private Color color;
013        private boolean outline;
014        
015        public ButtonImpl(BootstrapFactory factory, H htmlElement, Color color, boolean outline) {
016                super(factory, htmlElement);
017                htmlElement.addClass("btn")
018                        .addClassConditional(color != null && color.code != null && !outline, color ==  null ? null : "btn-"+color.code)
019                        .addClassConditional(color != null && color.code != null && outline,  color ==  null ? null : "btn-outline-"+color.code)
020                        .attribute("role", "button", htmlElement instanceof Tag && ((Tag) htmlElement).is(TagName.a) ); // For <a>
021                
022                this.color = color;
023                this.outline = outline;
024        }
025        
026        Color getColor() {
027                return color;
028        }
029        
030        boolean isOutline() {
031                return outline;
032        }
033
034        @Override
035        public H toHTMLElement() {
036                return htmlElement;
037        }
038
039        @Override
040        public Button<H> large() {
041                return large(true);
042        }
043
044        @Override
045        public Button<H> large(boolean large) {
046                htmlElement.addClassConditional(large, "btn-lg");
047                return this;
048        }
049
050        @Override
051        public Button<H> small() {
052                return small(true);
053        }
054
055        @Override
056        public Button<H> small(boolean small) {
057                htmlElement.addClassConditional(small, "btn-sm");
058                return this;
059        }
060
061        @Override
062        public Button<H> block() {
063                return block(true);
064        }
065
066        @Override
067        public Button<H> block(boolean block) {
068                htmlElement.addClassConditional(block, "btn-block");
069                return this;
070        }
071
072        @Override
073        public Button<H> active() {
074                return active(true);
075        }
076
077        @Override
078        public Button<H> active(boolean active) {
079                htmlElement.addClassConditional(active && !(htmlElement instanceof Button), "active");
080                return this;
081        }
082
083        @Override
084        public Button<H> disabled() {
085                return disabled(true);
086        }
087
088        @Override
089        public Button<H> disabled(boolean disabled) {
090                if (htmlElement instanceof org.nasdanika.html.Button) {
091                        ((org.nasdanika.html.Button) htmlElement).disabled();
092                } else {
093                        htmlElement.addClassConditional(disabled, "disabled");
094                }
095                return this;
096        }
097
098        @Override
099        public Button<H> dataToggle() {
100                return dataToggle(true);
101        }
102
103        @Override
104        public Button<H> dataToggle(boolean dataToggle) {
105                htmlElement.attribute("data-toggle", "button", dataToggle);
106                return null;
107        }
108
109        @Override
110        public Object produce(int indent) {
111                return htmlElement.produce(indent);
112        }
113
114}
