001package org.nasdanika.html.bootstrap.impl;
002
003import org.nasdanika.html.Button;
004import org.nasdanika.html.Button.Type;
005import org.nasdanika.html.Form;
006import org.nasdanika.html.HTMLElement;
007import org.nasdanika.html.HTMLFactory;
008import org.nasdanika.html.Tag;
009import org.nasdanika.html.TagName;
010import org.nasdanika.html.bootstrap.BootstrapFactory;
011import org.nasdanika.html.bootstrap.Color;
012import org.nasdanika.html.bootstrap.Breakpoint;
013import org.nasdanika.html.bootstrap.Dropdown;
014import org.nasdanika.html.bootstrap.Navbar;
015
016public class NavbarImpl extends WrappingBootstrapElementImpl<Tag, Navbar> implements Navbar {
017
018        protected HTMLElement<?> brand;
019        protected Tag items;
020
021        public NavbarImpl(
022                        BootstrapFactory factory,
023                        Breakpoint expandSize, 
024                        boolean dark,
025                        Color background, 
026                        HTMLElement<?> brand) {
027                super(factory, factory.getHTMLFactory().tag(TagName.nav).addClass("navbar"));
028                htmlElement
029                        .addClass("navbar-expand-"+expandSize.code)
030                        .addClass(dark ? "navbar-dark" : "navbar-light")
031                        .addClassConditional(background !=null && background.code != null, "bg-"+background.code);
032                
033                this.brand = brand;
034                if (brand != null) {
035                        brand.addClass("navbar-brand");
036                        htmlElement.content(brand);
037                }
038                
039                HTMLFactory htmlFactory = factory.getHTMLFactory();             
040                String collapseDivId = htmlFactory.nextId();
041                
042                Button toggler = htmlFactory.button(htmlFactory.span().addClass("navbar-toggler-icon"));
043                toggler
044                        .type(Type.BUTTON)
045                        .addClass("navbar-toggler")
046                        .attribute("data-toggle", "collapse")
047                        .attribute("data-target", "#"+collapseDivId);
048                htmlElement.content(toggler);           
049                
050                items = htmlFactory.tag(TagName.ul).addClass("navbar-nav");
051                Tag collapseDiv = htmlFactory.div(items).addClass("collapse", "navbar-collapse").id(collapseDivId);
052                htmlElement.content(collapseDiv);
053        }
054
055        @Override
056        public Tag item(Object href, boolean active, boolean disabled, Object... content) {
057                HTMLFactory htmlFactory = getFactory().getHTMLFactory();
058                Tag link = htmlFactory.link(href, content).addClassConditional(disabled, "disabled").addClass("nav-link"); 
059                Tag li = htmlFactory.tag(TagName.li, link).addClass("nav-item").addClassConditional(active, "active");
060                items.content(li);
061                return li;
062        }
063
064        @Override
065        public Dropdown dropdown(boolean active, Object... name) {
066                HTMLFactory htmlFactory = getFactory().getHTMLFactory();
067                Tag toggle = htmlFactory.link("#", name)
068                                .addClass("nav-link", "dropdown-toggle")
069                                .attribute("role", "button")
070                                .attribute("data-toggle", "dropdown");          
071                DropdownMenu menu = new DropdownMenu(getFactory(), getFactory().wrap(toggle));
072                Tag li = htmlFactory.tag(TagName.li, toggle, menu.toHTMLElement())
073                                .addClass("nav-item", "dropdown")
074                                .addClassConditional(active, "active");
075                items.content(li); 
076                return menu;
077        }
078
079        @Override
080        public Navbar form(Form form) {
081                form.addClass("form-inline");
082                items.content(form);
083                return this;
084        }
085
086        @Override
087        public Tag navbarText(Object... content) {
088                Tag span = getFactory().getHTMLFactory().span(content).addClass("navbar-text");
089                items.content(span);
090                return span;
091        }
092
093}
