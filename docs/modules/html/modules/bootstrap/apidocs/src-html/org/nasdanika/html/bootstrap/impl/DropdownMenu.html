001package org.nasdanika.html.bootstrap.impl;
002
003import org.nasdanika.html.Form;
004import org.nasdanika.html.HTMLElement;
005import org.nasdanika.html.Tag;
006import org.nasdanika.html.TagName;
007import org.nasdanika.html.bootstrap.BootstrapElement;
008import org.nasdanika.html.bootstrap.BootstrapFactory;
009import org.nasdanika.html.bootstrap.Dropdown;
010
011/**
012 * Handles dropdown menu container
013 * @author Pavel Vlasov
014 *
015 */
016public class DropdownMenu extends DivWrappingBootstrapElementImpl<Dropdown> implements Dropdown {
017        
018        private BootstrapElement<?, ?> toggle;
019
020        public DropdownMenu(BootstrapFactory factory, BootstrapElement<?,?> toggle) {
021                super(factory);
022                htmlElement.addClass("dropdown-menu");
023                this.toggle = toggle;
024        }
025
026        @Override
027        public Dropdown item(HTMLElement<?> item, boolean active, boolean disabled) {
028                item
029                        .addClass("dropdown-item")
030                        .addClassConditional(active, "active")
031                        .addClassConditional(disabled, "disabled");
032                htmlElement.content(item);
033                return this;
034        }
035
036        @Override
037        public Tag header(Object... content) {
038                Tag ret = getFactory().getHTMLFactory().tag(TagName.h6, content).addClass("dropdown-header");
039                htmlElement.content(ret);
040                return ret;
041        }
042
043        @Override
044        public Tag divider() {
045                Tag ret = getFactory().getHTMLFactory().div().addClass("dropdown-divider");
046                htmlElement.content(ret);
047                return ret;
048        }
049
050        @Override
051        public Dropdown form(Form form) {
052                htmlElement.content(form);
053                return this;
054        }
055        
056        @Override
057        public BootstrapElement<?, ?> getToggle() {
058                return toggle;
059        }
060
061}
