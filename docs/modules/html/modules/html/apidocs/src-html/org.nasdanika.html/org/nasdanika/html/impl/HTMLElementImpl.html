085        /**
086         * Renders attributes
087         * @param excluded Attributes to exclude (already rendered by subclass).
088         */
089        protected String attributes(String... excluded) {
090                StringBuilder attributeBuilder = new StringBuilder();
091                
092                for (Entry<String, Object> a: attributes.entrySet()) {
093                        if (!STYLE.equals(a.getKey())
094                                        && !CLASS.equals(a.getKey()) 
095                                        && !Arrays.asList(excluded).contains(a.getKey())) {
096                                if (attributeBuilder.length() > 0) {
097                                        attributeBuilder.append(" ");
098                                }                               
099                                Object value = a.getValue();
100                                if (Boolean.TRUE.equals(value)) {
101                                        // boolean attributes
102                                        attributeBuilder.append(a.getKey());                                    
103                                } else {
104                                        attributeBuilder.append(a.getKey()+"=\""+StringEscapeUtils.escapeHtml4(stringify(value, 0))+"\"");
105                                }
106                        }
107                }
108                
109                if (!Arrays.asList(excluded).contains(STYLE)) {
110                        StringBuilder styleBuilder = styles();
111                        if (styleBuilder.length() > 0) {
112                                if (attributeBuilder.length() > 0) {
113                                        attributeBuilder.append(" ");
114                                }
115                                attributeBuilder.append(STYLE+"=\""+styleBuilder+"\"");                         
116                        }
117                }
118                
119                if (!Arrays.asList(excluded).contains(CLASS)) {
120                        StringBuilder classBuilder = classes();
121                        if (classBuilder.length()>0) {
122                                if (attributeBuilder.length()>0) {
123                                        attributeBuilder.append(" ");
124                                }
125                                attributeBuilder.append(CLASS+"=\""+classBuilder+"\"");                         
126                        }
127                }
128                
129                return attributeBuilder.length()==0 ? "" : " "+attributeBuilder.toString();
130        }
131
132        protected StringBuilder classes() {
133                StringBuilder classBuilder = new StringBuilder();
134                for (Object cls: classes) {
135                        if (classBuilder.length()>0) {
136                                classBuilder.append(" ");
137                        }
138                        classBuilder.append(stringify(cls, 0));
139                }
140                if (attributes.containsKey(CLASS)) {
141                        if (classBuilder.length()>0) {
142                                classBuilder.append(" ");
143                        }
144                        classBuilder.append(stringify(attributes.get(CLASS), 0));
145                }
146                return classBuilder;
147        }
148
149        protected StringBuilder styles() {
150                StringBuilder styleBuilder = new StringBuilder();
151                if (attributes.containsKey(STYLE)) {
152                        styleBuilder.append(stringify(attributes.get(STYLE), 0));
153                }
154                for (Entry<String, Object> se: styles.entrySet()) {
155                        if (styleBuilder.length()>0 && !styleBuilder.toString().trim().endsWith(";")) {
156                                styleBuilder.append(";");
157                        }
158                        styleBuilder.append(se.getKey()+":"+stringify(se.getValue(), 0));
159                }
160                return styleBuilder;
161        }
162        
163        /**
164         * Merges value of given attribute, prepended by space, to the attribute definition being
165         * rendered by a subclass
166         * @param attribute
167         */
168        protected String merge(String attribute) {
169                if (STYLE.equals(attribute)) {
170                        StringBuilder styleBuilder = new StringBuilder();
171                        if (attributes.containsKey(STYLE)) {
172                                styleBuilder.append(stringify(attributes.get(STYLE), 0));
173                        }
174                        for (Entry<String, Object> se: styles.entrySet()) {
175                                if (styleBuilder.length() > 0 && !styleBuilder.toString().endsWith(";")) {
176                                        styleBuilder.append(";");
177                                }
178                                styleBuilder.append(se.getKey()+":"+stringify(se.getValue(), 0));
179                        }
180                        return styleBuilder.toString();
181                }
182                
183                if (attributes.containsKey(attribute)) {
184                        return StringEscapeUtils.escapeHtml4(String.valueOf(attributes.get(attribute)));
185                }
186                
187                return "";
188        }
189        
190        /**
191         * Merges attributes from the source element into this element.
192         * @param source
193         */
194        protected void merge(HTMLElementImpl<?> source) {
195                this.attributes.putAll(source.attributes);
196                this.classes.addAll(source.classes);
197                this.styles.putAll(source.styles);
198        }

207        public T attribute(String name, Object value) {
208                if (value==null) {
209                        attributes.remove(name);
210                } else if (value instanceof InputStream || value instanceof Reader) {
211                        attributes.put(name, stringify(value));                 
212                } else {
213                        attributes.put(name, value);
214                }
215                return (T) this;
216        }

287        @Override
288        public T on(String event, Reader handler) throws IOException {
289                StringWriter sw = new StringWriter();
290                char[] cbuf = new char[1024];
291                int l;
292                while ((l=handler.read(cbuf))!=-1) {
293                        sw.write(cbuf, 0, l);
294                }
295                sw.close();
296                handler.close();
297                return on(event, sw.toString());
298        }

327        /**
328         * Special handling for input streams, readers, producers and URL's.
329         * @param content
330         * @return
331         */
332        public static String stringify(Object content, int indent) {
333                try {
334                        if (content == null) {
335                                return "";
336                        }
337                        
338                        if (content instanceof String) {
339                                return (String) content;
340                        }
341                        
342                        if (content instanceof Producer) {
343                                return stringify(((Producer) content).produce(indent), indent);
344                        }
345                        
346                        if (content instanceof InputStream) {
347                                return stringify(new InputStreamReader((InputStream) content, StandardCharsets.UTF_8), indent);
348                        }
349                        if (content instanceof Reader) {
350                                StringWriter sw = new StringWriter();
351                                for (int ch = ((Reader) content).read(); ch!=-1; ch = ((Reader) content).read()) {
352                                        sw.write(ch);
353                                }
354                                ((Reader) content).close();
355                                sw.close();
356                                return sw.toString();
357                        }
358                        
359                        if (content instanceof URL) {
360                                return stringify(((URL) content).openStream(), indent);
361                        }
362        
363                        return content.toString();
364                } catch (RuntimeException e) {
365                        throw e;
366                } catch (Exception e) {
367                        throw new ProducerException(e);
368                }
369        }

379        public static StringBuilder indent(StringBuilder stringBuilder, int indent) {
380                if (indent != -1) {
381                        stringBuilder.append(System.lineSeparator());
382                        for (int i=0; i<indent; ++i) {
383                                stringBuilder.append("  ");
384                        }
385                }
386                return stringBuilder;
387        }

389        public static StringBuilder indent(int indent) {
390                return indent(new StringBuilder(), indent);

405        @Override
406        public String produce(int indent) {             
407                List<Object> theContent = new ArrayList<>();
408                for (Object c: getContent()) {
409                        if (c instanceof FragmentImpl) {
410                                theContent.addAll(((FragmentImpl) c).getAllContent());
411                        } else if (c!=null) {
412                                theContent.add(c);
413                        }                       
414                }
415                if (theContent.isEmpty()) {
416                        if (nonEmpty) {
417                                return "";
418                        }
419                        
420                        if (!forceEndTag()) {
421                                return indent(renderComment(indent), indent).append("<"+getTagName()+attributes()+"/>").toString();
422                        }
423                }
424                boolean hasNonUIElementContent = false;
425                
426                StringBuilder contentBuilder = new StringBuilder(); 
427                for (Object c: theContent) {
428                        contentBuilder.append(stringify(c, indent == -1 ? indent : indent+1));
429                        if (!(c instanceof Markup)) {
430                                hasNonUIElementContent = true;
431                        }
432                }               
433                String strContent = contentBuilder.toString();
434                if (nonEmpty && strContent.trim().length() == 0) {
435                        return "";
436                }
437                StringBuilder sb = indent(renderComment(indent), indent).append("<").append(getTagName()).append(attributes()).append(">").append(strContent);          
438                if (strContent.length() > 0 && !hasNonUIElementContent) {
439                        indent(sb, indent);
440                }
441                return sb.append("</").append(getTagName()).append(">").toString();
442        }
443
444        private boolean forceEndTag() {
445                for (TagName tagName: TagName.values()) {
446                        if (tagName.name().equalsIgnoreCase(this.tagName)) {
447                                return tagName.paired;
448                        }
449                }
450                return false;
451        }

453        /**
454         * Method to add content, make public in container elements.
455         * @param content
456         * @return
457         */
458        @SuppressWarnings("unchecked")
459        protected T content(Object... content) {
460                for (Object c: content) {
461                        if (c instanceof Iterable) {
462                                for (Object cc: (Iterable<?>) c) {
463                                        content(cc);
464                                }
465                        } else if (c != null) {
466                                if (c.getClass().isArray()) {
467                                        for (int i = 0; i < Array.getLength(c); ++i) {
468                                                content(Array.get(c, i));
469                                        }
470                                } else if (c instanceof InputStream || c instanceof Reader) {
471                                        this.content.add(stringify(c));
472                                } else {
473                                        this.content.add(c);
474                                }
475                        }
476                }
477                return (T) this;
478        }
