001package org.nasdanika.html;
002
003import java.util.List;
004
005public interface RowContainer<T extends RowContainer<T>> extends Container<T>, HTMLElement<T> {
006        
007        interface Row extends HTMLElement<Row>, Container<Row> {
008                
009                interface Cell extends HTMLElement<Cell>, Container<Cell> {
010                        
011                        Cell colspan(int colspan);
012                        
013                        Cell rowspan(int rowspan);
014                        
015                }                               
016                
017                Cell cell(Object... content);
018                
019                Cell header(Object... content);
020                
021                List<Cell> cells();
022                
023        }
024                
025        List<Row> rows();
026        
027        /**
028         * Creates a row
029         * @param cells Optional cells content.
030         * @return
031         */
032        Row row(Object...cells);
033        
034        /**
035         * Creates a row
036         * @param headers Optional headers content.
037         * @return
038         */
039        Row headerRow(Object...headers);                        
040
041}
