144        @Override
145        public Function function(final Object... paramArgs) {
146                return new Function() {
147                        
148                        List<Object> params = new ArrayList<>(Arrays.asList(paramArgs));
149                        List<Object> binds = new ArrayList<>();                 
150                        List<Object> code = new ArrayList<>();
151        
152                        @Override
153                        public Function code(Object... code) {
154                                for (Object obj: code) {
155                                        if (obj != null) {
156                                                this.code.add(obj);
157                                        }
158                                }
159                                return this;
160                        }
161        
162                        @Override
163                        public Function bind(Object... bind) {
164                                for (Object obj: bind) {
165                                        if (obj != null) {
166                                                this.binds.add(obj);
167                                        }
168                                }
169                                return this;
170                        }
171                        
172                        @Override
173                        public Function parameter(Object... param) {
174                                for (Object obj: param) {
175                                        if (obj != null) {
176                                                this.params.add(obj);
177                                        }
178                                }
179                                return this;
180                        }
181                        
182                        @Override
183                        public String toString() {
184                                StringBuilder ret = new StringBuilder("function(");
185                                Iterator<Object> pit = params.iterator();
186                                while (pit.hasNext()) {
187                                        ret.append(pit.next());
188                                        if (pit.hasNext()) {
189                                                ret.append(",");
190                                        }
191                                }
192                                ret.append(") {");
193                                for (Object c: code) {
194                                        ret.append(c);
195                                }
196                                ret.append("}");
197                                if (!binds.isEmpty()) {
198                                        ret.append(".bind(");
199                                        Iterator<Object> bit = binds.iterator();
200                                        while (bit.hasNext()) {
201                                                ret.append(bit.next());
202                                                if (bit.hasNext()) {
203                                                        ret.append(",");
204                                                }
205                                        }
206                                        ret.append(")");
207                                }
208                                return ret.toString();
209                        }
210        
211                };
212        }

214        private static final Pattern EXPANDER_PATTERN = Pattern.compile("\\{\\{(.+?)\\}\\}");   
215        
216        /**
217         * Expands tokens in the form of <code>{{token name}}</code> to their values.
218         * If a token is not found expansion is not processed.
219         * @param input
220         * @param env
221         * @return
222         */
223        public static String interpolate(String input, java.util.function.Function<String, Object> tokenSource) {
224                if (tokenSource==null) {
225                        return input;
226                }
227                Matcher matcher = EXPANDER_PATTERN.matcher(input);
228                StringBuilder output = new StringBuilder();
229                int i = 0;
230                while (matcher.find()) {
231                    String token = matcher.group();
232                        Object replacement = tokenSource.apply(token.substring(2, token.length()-2));
233                    if (replacement != null) {
234                            output.append(input.substring(i, matcher.start())).append(replacement);                         
235                            i = matcher.end();
236                    }
237                }
238                output.append(input.substring(i, input.length()));
239                return output.toString();
240        }
241        
242        public static String interpolate(String input, final Map<String, Object> env) {
243                return interpolate(input, token -> env==null ? null : env.get(token));
244        }

246        /**
247         * Expands tokens found in URL, InputStream, Reader or String. 
248         * @param input URL, InputStream, Reader or String.
249         * @param tokenSource Source of tokens.
250         * @return String with expanded tokens.
251         * @throws IOException
252         */
253        @Override
254        public String interpolate(Object input, java.util.function.Function<String, Object> tokenSource) {
255                return interpolate(HTMLElementImpl.stringify(input), tokenSource);
256        }

263         * @throws IOException
264         */

290        @Override
291        public Tag title(final Object title) {
292                Object escapedTitle = new Object() {
293                        public String toString() {
294                                return StringEscapeUtils.escapeEcmaScript(String.valueOf(title));                               
295                        };
296                };
297                return tag("script", "document.title=\"", escapedTitle, "\";");
298        }

300        @Override
301        public Tag overlay(Object... content) {         
302                Tag contentDiv = div(content)
303                                .style("position", "relative")
304                                .style("height", "100%")
305                                .style("width", "100%")
306                                .style("display", "flex")
307                                .style("justify-content", "center")
308                                .style("align-items", "center");
309                
310                return div(contentDiv)
311                                .style("position", "absolute")
312                                .style("top", "0")
313                                .style("left", "0")
314                                .style("width", "100%")
315                                .style("height", "100%")
316                                .style().background().color().color(Color.Silver)
317                                .style("opacity", 0.7)
318                                .style("z-index", 10)
319                                .style("display", "block");
320        }

327        @Override
328        public NamedItemsContainer tagNamedItemsContainer(TagName tagName) {
329                class NamedItemsContainerImpl implements NamedItemsContainer, Producer {
330                        Fragment fragment = fragment();
331
332                        @Override
333                        public Object produce(int indent) {
334                                return fragment.produce(indent);
335                        }
336                        
337                        @Override
338                        public String toString() {
339                                return fragment.toString();
340                        }
341
342                        @Override
343                        public void item(Object name, Object content) {
344                                fragment.content(tag(tagName).content(name), content);
345                        }
346
347                        @Override
348                        public boolean isEmpty() {
349                                return fragment.isEmpty();
350                        }
351                        
352                }
353                return new NamedItemsContainerImpl();
354        }

361        @Override
362        public TokenSource tokenSource() {
363                return new TokenSource() {
364                        
365                        private Map<String, Object> tokens = new HashMap<>();
366                        
367                        @Override
368                        public Object apply(String token) {
369                                return tokens.get(token);
370                        }
371                        
372                        @Override
373                        public TokenSource put(String token, Object value) {
374                                if (value != null) {
375                                        tokens.put(token, value);
376                                }
377                                return this;
378                        }
379
380                        @Override
381                        public String interpolate(Object input) {
382                                return DefaultHTMLFactory.this.interpolate(input, this);
383                        }
384                };
385        }
386        
387        @Override
388        public TokenSource tokenSource(String token, Object value) {
389                return tokenSource().put(token, value);
390        }
