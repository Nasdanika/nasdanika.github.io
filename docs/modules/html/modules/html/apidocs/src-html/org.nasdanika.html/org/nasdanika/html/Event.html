001package org.nasdanika.html;
002
003/**
004 * Enumeration for frequently used HTML events
005 * @author Pavel
006 *
007 */
008public enum Event {
009        // Form Events
010        // Events triggered by actions inside a HTML form (applies to almost all HTML elements, but is most used in form elements):
011
012        /** Fires the moment that the element loses focus **/
013        blur, 
014        
015        /** Fires the moment when the value of the element is changed **/
016        change,
017        
018        /** Script to be run when a context menu is triggered **/
019        contextmenu, 
020        
021        /** Fires the moment when the element gets focus **/
022        focus, 
023        
024        /** Script to be run when a form changes **/
025        formchange,
026        
027        /** Script to be run when a form gets user input **/
028        forminput,
029        
030        /** Script to be run when an element gets user input **/
031        input, 
032        
033        /** Script to be run when an element is invalid **/
034        invalid, 
035        
036        /** Fires when the Reset button in a form is clicked **/
037        reset, 
038        
039        /** Fires after some text has been selected in an element **/
040        select, 
041        
042        /** Fires when a form is submitted **/
043        submit, 
044
045        // Keyboard Events
046        
047        /** Fires when a user is pressing a key **/
048        keydown, 
049        
050        /** Fires when a user presses a key **/
051        keypress,
052        
053        /** Fires when a user releases a key **/
054        keyup, 
055
056        // Mouse Events
057        // Events triggered by a mouse, or similar user actions:
058
059        /** Fires on a mouse click on the element **/
060        click, 
061        
062        /** Fires on a mouse double-click on the element **/
063        dblclick,
064        
065        /** Script to be run when an element is dragged **/
066        drag, 
067        
068        /** Script to be run at the end of a drag operation **/
069        dragend, 
070        
071        /** Script to be run when an element has been dragged to a valid drop target **/
072        dragenter, 
073        
074        /** Script to be run when an element leaves a valid drop target **/
075        dragleave, 
076        
077        /** Script to be run when an element is being dragged over a valid drop target **/
078        dragover, 
079        
080        /** Script to be run at the start of a drag operation **/
081        dragstart, 
082        
083        /** Script to be run when dragged element is being dropped **/
084        drop, 
085        
086        /** Fires when a mouse button is pressed down on an element **/
087        mousedown, 
088        
089        /** Fires when the mouse pointer moves over an element **/
090        mousemove, 
091        
092        /** Fires when the mouse pointer moves out of an element **/
093        mouseout, 
094        
095        /** Fires when the mouse pointer moves over an element **/
096        mouseover, 
097        
098        /** Fires when a mouse button is released over an element **/
099        mouseup, 
100        
101        /** Script to be run when the mouse wheel is being rotated **/
102        mousewheel, 
103        
104        /** Script to be run when an element's scrollbar is being scrolled **/
105        scroll          
106}
