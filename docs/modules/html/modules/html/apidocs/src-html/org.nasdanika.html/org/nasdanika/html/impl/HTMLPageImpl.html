001package org.nasdanika.html.impl;
002
003import java.io.ByteArrayInputStream;
004import java.io.InputStream;
005import java.nio.charset.StandardCharsets;
006
007import org.nasdanika.common.Adaptable;
008import org.nasdanika.html.HTMLFactory;
009import org.nasdanika.html.HTMLPage;
010import org.nasdanika.html.Tag;
011import org.nasdanika.html.TagName;
012
013public class HTMLPageImpl implements HTMLPage, Adaptable {
014
015        private HTMLFactory factory;
016        private Tag html;
017        private Tag head;
018        private Tag body;
019        
020        HTMLPageImpl(HTMLFactory factory) {
021                this.factory = factory;
022                head = factory.tag(TagName.head);
023                body = factory.tag(TagName.body);
024                html = factory.tag(TagName.html, head, body);
025        }
026        
027        @Override
028        public Object produce(int indent) {
029                return "<!DOCTYPE html>"+html.produce(indent);
030        }
031        
032        @Override
033        public HTMLPage body(Object... content) {
034                body.content(content);
035                return this;
036        }
037        
038        /**
039         * Adds external script to the header.
040         * @param url
041         */
042        public Tag script(Object url) {
043                Tag script = factory.tag(TagName.script).attribute("src", url);
044                head(script);
045                return script;
046        }
047        
048        /**
049         * Adds stylesheet to the header.
050         * @param url
051         */
052        public Tag stylesheet(Object url) {
053                Tag stylesheet = factory.tag(TagName.link).attribute("rel", "stylesheet").attribute("href", url);
054                head(stylesheet);
055                return stylesheet;
056        }
057        
058        /**
059         * Adds content to html head element of the page.
060         * @param content
061         */
062        public HTMLPage head(Object... content) {
063                head.content(content);
064                return this;
065        }
066        
067        /**
068         * Adds page title.
069         * @param title
070         */
071        public Tag title(Object title) {
072                Tag ret = factory.tag(TagName.title, title);
073                head(ret);
074                return ret;
075        }
076        
077        @Override
078        public String toString() {
079                return HTMLElementImpl.stringify(produce(0));
080        }
081
082        @Override
083        public HTMLPage lang(Object lang) {
084                html.attribute("lang", lang);
085                return this;
086        }
087
088        @Override
089        public HTMLFactory getFactory() {
090                return factory;
091        }
092        
093        @SuppressWarnings("unchecked")
094        @Override
095        public <TT> TT adaptTo(Class<TT> type) {
096                if (type == InputStream.class) {
097                        String str = toString();
098                        return (TT) new ByteArrayInputStream(str.getBytes(StandardCharsets.UTF_8));
099                }
100                return Adaptable.super.adaptTo(type);
101        }
102
103}
