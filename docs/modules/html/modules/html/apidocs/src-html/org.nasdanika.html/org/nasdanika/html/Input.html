006 *
007 */

012        
013        Input formaction(Object formaction);
014        
015        Input formenctype(Form.EncType formEncType);
016        
017        Input formmethod(Form.Method formMethod);
018        
019        Input formnovalidate(boolean formnovalidate);
020        Input formnovalidate();
021        
022        Input formtarget(Object formTarget);
023        
024        Input dimensions(int width, int height);
025        
026        Input list(Object dataListId);
027        
028        Input min(Object min);
029        Input max(Object max);
030        
031        Input multiple(boolean multiple);
032        Input multiple();
033        
034        Input pattern(Object pattern);
035        
036        Input value(Object value);
037        
038        Input placeholder(Object placeholder);
039        
040        Input step(Object step);        
041        
042        InputType getType();    
043
044}
