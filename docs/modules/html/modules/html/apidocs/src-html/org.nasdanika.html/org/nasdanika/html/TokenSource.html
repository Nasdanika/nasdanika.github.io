001package org.nasdanika.html;
002
003import java.util.function.BiConsumer;
004
005/**
006 * Mutable token source with ability to add tokens in a fluent fashion. 
007 * @author Pavel Vlasov
008 *
009 */
010public interface TokenSource extends java.util.function.Function<String, Object>, BiConsumer<String, Object> {
011        
012        /**
013         * Adds a new token. Null values are ignored.
014         * @param name
015         * @param value
016         * @return
017         */
018        TokenSource put(String token, Object value);
019        
020        @Override
021        default void accept(String token, Object value) {
022                put(token, value);              
023        }
024        
025        /**
026         * Interpolates the object using this token source. 
027         * @param input String, Reader, InputStream or URL.
028         * @return
029         */
030        String interpolate(Object input);
031
032}
