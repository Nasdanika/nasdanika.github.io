014/**
015 * Abstraction of a UI element. toString() produces HTML markup.
016 * @author Pavel
017 *
018 */
019public interface HTMLElement<T extends HTMLElement<T>> extends Producer, Markup {
020

022         *

024         * @param handler Handler script. 
025         * @return

161         * Applications may associate arbitrary objects with the UI element. This data is not used for rendering by UI elements.
162         * @param key
163         * @param data
164         * @return this UI element.
165         */
166        T setData(Object key, Object data);

168        /**

176         * @return Element content.
177         */

180        /**
181         * Loads HTML attributes from a map.
182         * For all top-level keys except ``class``, ``style``, and ``data`` attribute value is produced by converting the value to string for scalars and to JSON string for lists and maps.
183         * "children" key is ignored - it is used to define a hierarchy of attributes.
184         * For class attribute its value is formed by concatenating elements using space as a separator. If elements are hierarchical then class name is formed by concatenation with a dash (``-``) as a separator.
185         * If value of ``data`` attribute is a map then keys of that map get concatenated with ``data`` using dash (``-``) as a separator, the same applies to nested maps. Non-map values become attribute values - scalars are converted to string, 
186         * lists are converted to JSON string.
187         * Style can be defined as a string, list or map. If style is defined as a list, all list values are concatenated with a space as a separator - it is a convent way for long unstructured definitions.
188         * If style value is a map then the value and its contained map values are processed in the following fashion:
189         * a) Keys are concatenated with dash as a separator.
190         * b) List values are concatenated with space as a separator.
191         * 
192         * @param attributes
193         * @return
194         */
195        @SuppressWarnings("unchecked")
196        default T attributes(Map<?,?> attributes) {
197                
198                class Util {
199                        
200                        void setDataAttribute(HTMLElement<?> htmlElement, String key, Object value) {
201                                if (value instanceof Map) {
202                                        for (Entry<?, ?> entry: ((Map<?,?>) value).entrySet()) {
203                                                setDataAttribute(htmlElement, key+"-"+entry.getKey(), value);
204                                        }
205                                } else if (value instanceof Collection) {
206                                        JSONArray jsonArray = new JSONArray((Collection<?>) value);
207                                        htmlElement.attribute(key, jsonArray);                                                                                                                          
208                                } else {
209                                        htmlElement.attribute(key, value);
210                                }                               
211                        }
212
213                        void setStyle(HTMLElement<?> htmlElement, String key, Object value) {
214                                if (value instanceof Map) {
215                                        for (Entry<?, ?> entry: ((Map<?,?>) value).entrySet()) {
216                                                setStyle(htmlElement, key + "-" + entry.getKey(), entry.getValue());
217                                        }
218                                } else if (value instanceof Collection) {
219                                        StringBuilder styleBuilder = new StringBuilder();
220                                        for (Object e: (Collection<?>) value) {
221                                                if (styleBuilder.length() > 0) {
222                                                        styleBuilder.append(" ");
223                                                }
224                                                styleBuilder.append(e);
225                                        }
226                                        htmlElement.style(key, styleBuilder.toString());
227                                } else {
228                                        htmlElement.style(key, value);
229                                }                               
230                        }
231
232                        void addClass(HTMLElement<?> htmlElement, String prefix, Object value) {
233                                if (value instanceof Map) {
234                                        for (Entry<?, ?> entry: ((Map<?,?>) value).entrySet()) {
235                                                addClass(htmlElement, prefix == null ? String.valueOf(entry.getKey()) : prefix + "-" + entry.getKey(), entry.getValue());
236                                        }
237                                } else if (value instanceof Collection) {
238                                        for (Object e: (Collection<?>) value) {
239                                                addClass(htmlElement, prefix, e);
240                                        }
241                                } else {
242                                        if (value instanceof Boolean) {
243                                                if (Boolean.TRUE.equals(value) && prefix != null) {
244                                                        htmlElement.addClass(prefix);
245                                                }
246                                        } else {
247                                                htmlElement.addClass(prefix == null ? String.valueOf(value) : prefix + "-" + value);
248                                        }
249                                }                               
250                        }
251                        
252                        void setAttributes() {
253                                for (Entry<?, ?> entry: attributes.entrySet()) {
254                                        Object value = entry.getValue();
255                                        String key = entry.getKey().toString();
256                                        switch (key) {
257                                        case "children":
258                                                break;
259                                        case "class":
260                                                addClass(HTMLElement.this, null, value);
261                                                break;
262                                        case "style":
263                                                if (value instanceof Collection) {
264                                                        StringBuilder styleBuilder = new StringBuilder();
265                                                        for (Object e: (Collection<?>) value) {
266                                                                if (styleBuilder.length() > 0) {
267                                                                        styleBuilder.append(" ");
268                                                                }
269                                                                styleBuilder.append(e);
270                                                        }
271                                                        HTMLElement.this.attribute(key, styleBuilder.toString());                                                               
272                                                } else if (value instanceof Map) {
273                                                        for (Entry<?, ?> se: ((Map<?,?>) value).entrySet()) {
274                                                                setStyle(HTMLElement.this, se.getKey().toString(), se.getValue());
275                                                        }
276                                                } else {
277                                                        HTMLElement.this.attribute(key, value);                                                         
278                                                }
279                                                break;
280                                        case "data":
281                                                setDataAttribute(HTMLElement.this, key, value);
282                                                break;
283                                        default:
284                                                if (value instanceof Map) {
285                                                        JSONObject jsonObject = new JSONObject((Map<?,?>) value);
286                                                        HTMLElement.this.attribute(key, jsonObject);                                                            
287                                                } else if (value instanceof Collection) {
288                                                        JSONArray jsonArray = new JSONArray((Collection<?>) value);
289                                                        HTMLElement.this.attribute(key, jsonArray);                                                                                                                             
290                                                } else {
291                                                        HTMLElement.this.attribute(key, value);
292                                                }
293                                        }
294                                }
295                        }
296                }
297                
298                new Util().setAttributes();
299        
300                return (T) this;
301        }
