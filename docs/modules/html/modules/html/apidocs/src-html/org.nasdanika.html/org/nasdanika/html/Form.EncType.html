001package org.nasdanika.html;
002
003
004public interface Form extends HTMLElement<Form>, FieldContainer<Form> {
005        
006        // Standard form attributes
007        
008        Form action(Object action);
009        
010        Form autocomplete();
011        Form autocomplete(boolean autocomplete);
012        
013        enum EncType {
014                plain("text/plain"),
015                multipart("multipart/form-data"),
016                urlencoded("application/x-www-form-urlencoded");
017                
018                public final String literal;
019                
020                private EncType(String literal) {
021                        this.literal = literal;         
022                }
023        }
024
025        Form enctype(EncType enctype);
026        
027        Form acceptCharset(String charset);
028
029        enum Method { get, post }
030        
031        Form method(Method method);
032        
033        Form name(String name);
034        
035        Form novalidate();
036        Form novalidate(boolean novalidate);
037        
038        Form target(String target);             
039        
040}
