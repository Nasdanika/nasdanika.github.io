001package org.nasdanika.html;
002
003import java.util.Map;
004
005import org.nasdanika.html.impl.DefaultHTMLFactory;
006
007
008/**
009 * Implementations of this class abstract HTML rendering code from rendering of low-level HTML primitives. 
010 * @author Pavel
011 *
012 */
013public interface HTMLFactory {
014        
015        HTMLFactory INSTANCE = new DefaultHTMLFactory();
016        
017        /**
018         * Generates element ID.
019         * @return
020         */
021        String nextId();
022        
023        /*******************
024         *  Simple API's   *
025         *******************/
026        
027        /**
028         * Creates a tag with a given name which attributes and styles can be manipulated
029         * with UIElement methods.
030         * @param tagName
031         * @param content
032         * @return
033         */
034        Tag tag(String tagName, Object... content);
035        
036        Tag tag(TagName tagName, Object... content);
037        
038        /**
039         * Creates a tag which is rendered only if it has content.
040         * @param tagName
041         * @param content
042         * @return
043         */
044        Tag nonEmptyTag(String tagName, Object... content);
045
046        /**
047         * Creates a tag which is rendered only if it has content.
048         * @param tagName
049         * @param content
050         * @return
051         */
052        Tag nonEmptyTag(TagName tagName, Object... content);
053        
054        
055        Tag div(Object... content);
056        
057        /**
058         * Creates a div which is rendered only if it has content.
059         * @param content
060         * @return
061         */
062        Tag nonEmptyDiv(Object... content);
063        
064        
065        Tag span(Object... content);
066        
067        Fragment fragment(Object... content);
068        
069        /**
070         * A convenience method for creating form inputs
071         * @param type Input type
072         * @param name Input name - optional
073         * @param value Input value - optional
074         * @param id Control id - optional
075         * @param placeholder Placeholder - optional
076         * @return
077         */
078        Input input(InputType type);
079        
080        Select select();
081
082        TextArea textArea();
083
084        Tag link(Object href, Object... content);
085        
086        Tag ul(Iterable<?> items);
087        
088        Tag ol(Iterable<?> items);
089        
090        Table table();
091                        
092        Button button(Object... content);
093        
094        Form form();
095        
096        /**
097         * Generates document.title = title script.
098         * @param title
099         * @return
100         */
101        Tag title(Object title);
102        
103        // --- JavaScript ---
104        
105        Function function(Object... param);
106        
107        /**
108         * Creates an overlay div with vertically and horizontally centered content.
109         * @param content
110         * @return
111         */
112        Tag overlay(Object... content);
113        
114        /**
115         * Generates JavaScript code (without script tag) to size the overlay and then display it.
116         * @param overlaySelector
117         * @param overlayedSelector
118         */
119        String showOverlay(String overlaySelector, String overlayedSelector, int widthAdjustment, int heightAdjustment);
120        
121        /**
122         * Expands tokens in the form of <code>{{token name}}</code> to their values.
123         * If a token is not found expansion is not processed.
124         * @param input String, Reader, InputStream or URL.
125         * @param tokenSource Source of tokens.
126         * @return
127         */
128        String interpolate(Object input, java.util.function.Function<String, Object> tokenSource);
129        
130        /**
131         * Expands tokens in the form of <code>{{token name}}</code> to their values.
132         * If a token is not found expansion is not processed.
133         * @param input String, Reader, InputStream or URL.
134         * @param env
135         * @return
136         */
137        String interpolate(Object input, final Map<String, Object> env);        
138        
139        /**
140         * Expands a single token in the form of <code>{{token name}}</code> to its value.
141         * If a token is not found expansion is not processed.
142         * @param input String, Reader, InputStream or URL.
143         * @param token
144         * @param value
145         * @return
146         */
147        String interpolate(Object input, String token, Object value);
148        
149        /**
150         * Creates a named items container which wraps name in the specified tag, e.g. H2.
151         * @param tagName
152         * @return
153         */
154        NamedItemsContainer tagNamedItemsContainer(TagName tagName);
155        
156        /**
157         * Creates HTML page.
158         * @return
159         */
160        HTMLPage page();
161        
162        /**
163         * Creates a mutable token source with initial token.
164         * @param token
165         * @param value
166         * @return
167         */
168        TokenSource tokenSource(String token, Object value);
169        
170        /**
171         * Creates a new mutable token source.
172         * @return
173         */
174        TokenSource tokenSource();      
175}
