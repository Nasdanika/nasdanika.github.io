001package org.nasdanika.html;
002
003public enum OrderedListType {
004        NUMBER("1", "Number"),
005        UPPERCASE_LETTERS("A", "Uppercase letters"),
006        LOWERCASE_LETTERS("a", "Lowercase letters"),
007        UPPERCASE_ROMAN_NUMBERS("I", "Uppercase roman numbers"),
008        LOWERCASE_ROMAN_NUMBERS("i", "Lowercase roman numbers"),
009        /**
010         * Pseudo-type indicating rotation of types for nested lists. An example of rotated numbering: "1.B.c.IV.v". 
011         */
012        ROTATE(null, "Rotate");
013        
014        OrderedListType(String code, String label) {
015                this.code = code;
016                this.label = label;
017        }
018        
019        public final String code;
020        public final String label;
021                
022        public static OrderedListType fromLabel(String label) {
023                for (OrderedListType breakpoint: values()) {
024                        if (breakpoint.label.equals(label)) {
025                                return breakpoint;
026                        }
027                }
028                throw new IllegalArgumentException("No type value for label "+label);
029        }
030
031
032}
