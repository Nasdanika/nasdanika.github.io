001package org.nasdanika.html;
002
003public interface FieldContainer<T extends FieldContainer<T>> extends Container<T> {
004                        
005        Button button(Object... content);
006        
007        FieldSet fieldset();
008        
009        /**
010         * Creates a new fragment backed by this container configuration. This is a factory method - the new fragment
011         * is not added to this container. This method can be used for wrapping of a group of fields
012         * into a container, e.g. a panel - create fragment, add fields, create a panel from 
013         * the fragment, add panel to this container.
014         * @return
015         */
016        FormFragment formFragment();
017
018}
