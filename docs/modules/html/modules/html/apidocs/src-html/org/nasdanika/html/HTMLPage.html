001package org.nasdanika.html;
002
003/**
004 * An interface for constructing HTML pages.
005 * @author Pavel Vlasov
006 *
007 */
008public interface HTMLPage extends Producer {
009        
010        /**
011         * Adds lang attribute to html tag.
012         * @param lang
013         * @return
014         */
015        HTMLPage lang(Object lang);
016        
017        /**
018         * Adds content to html head element of the page.
019         * @param content
020         */
021        HTMLPage head(Object... content);
022        
023        /**
024         * Adds external script to the header.
025         * @param url
026         */
027        Tag script(Object url);
028        
029        /**
030         * Adds stylesheet to the header.
031         * @param url
032         */
033        Tag stylesheet(Object url);
034                
035        /**
036         * Adds page title.
037         * @param title
038         */
039        Tag title(Object title);
040
041        /**
042         * Adds content to HTML body element of the page.
043         * @param content
044         * @return
045         */
046        HTMLPage body(Object... content);
047        
048        HTMLFactory getFactory();
049        
050}
