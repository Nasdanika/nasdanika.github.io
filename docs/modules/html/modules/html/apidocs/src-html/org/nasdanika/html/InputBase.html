001package org.nasdanika.html;
002
003/**
004 * Base interface for Input, Select, and TextArea
005 * @author Pavel
006 *
007 * @param <T>
008 */
009public interface InputBase<T extends InputBase<T>> extends HTMLElement<T> {
010        
011        T autofocus(boolean autofocus);
012        T autofocus();
013
014        T name(Object name);
015        
016        T required(boolean required);
017        T required();
018        
019        T disabled(boolean disabled);
020        T disabled();
021        
022        T form(Form... form);   
023                        
024}
