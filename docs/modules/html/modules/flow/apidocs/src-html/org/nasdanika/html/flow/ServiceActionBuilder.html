001package org.nasdanika.html.flow;
002
003import java.util.List;
004
005import org.eclipse.emf.ecore.ETypedElement;
006import org.nasdanika.common.Context;
007import org.nasdanika.flow.FlowPackage;
008import org.nasdanika.flow.Service;
009
010public class ServiceActionBuilder extends FlowElementActionBuilder<Service> {
011        
012        public ServiceActionBuilder(Service value, Context context) {
013                super(value, context);
014        }
015        
016        @Override
017        protected List<ETypedElement> getProperties() {
018                List<ETypedElement> properties = super.getProperties();
019                properties.add(FlowPackage.Literals.SERVICE__TARGET);
020                return properties;
021        }
022
023}
