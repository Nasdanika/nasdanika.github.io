001package org.nasdanika.html.emf;

003import java.util.Collection;
004import java.util.Collections;
005
006import org.eclipse.emf.common.notify.Adapter;

008import org.eclipse.emf.common.util.Diagnostic;

012import org.eclipse.emf.ecore.EStructuralFeature;
013import org.nasdanika.common.Context;
014import org.nasdanika.emf.ComposedAdapterFactory;
015import org.nasdanika.emf.DiagnosticProvider;
016import org.nasdanika.emf.DiagnosticProviderAdapter;
017import org.nasdanika.emf.FunctionAdapterFactory;
018import org.nasdanika.html.model.app.util.ActionProvider;
019import org.nasdanika.ncore.NcorePackage;
020import org.nasdanika.ncore.Temporal;
021
022/**
023 * Provides adapters for the Ecore types - {@link EPackage}, {@link EClass}, {@link EStructuralFeature}, {@link EOperation}, ...
024 * @author Pavel
025 *
026 */
027public class ActionProviderAdapterFactory extends ComposedAdapterFactory {
028        
029        public ActionProviderAdapterFactory(Context context) {
030                // Registering adapter factories.
031                registerAdapterFactory(
032                        new FunctionAdapterFactory<ActionProvider, Temporal>(
033                                NcorePackage.Literals.TEMPORAL, 
034                                ActionProvider.class, 
035                                this.getClass().getClassLoader(), 
036                                e -> new TemporalActionBuilder(e, context).asActionProvider()));
037        }
038        
039        @Override
040        public Adapter adaptNew(Notifier target, Object type) {
041                if (type == DiagnosticProvider.class) {
042                        return new DiagnosticProviderAdapter(target, this::getDiagnostic, this::getFeatureDiagnostic);
043                }
044                return super.adaptNew(target, type);
045        }
046        
047        @Override
048        public boolean isFactoryForType(Object type) {
049                return super.isFactoryForType(type) || type == DiagnosticProvider.class;
050        }
051        
052        protected Collection<Diagnostic> getDiagnostic(Notifier target) {
053                return Collections.emptyList();
054        }
055
056        protected Collection<Diagnostic> getFeatureDiagnostic(Notifier target, EStructuralFeature feature) {
057                return Collections.emptyList();
058        }
059        
060}
