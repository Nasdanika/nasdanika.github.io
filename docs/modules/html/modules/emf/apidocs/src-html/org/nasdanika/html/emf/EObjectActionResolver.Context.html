001package org.nasdanika.html.emf;
002
003import org.eclipse.emf.common.util.URI;
004import org.eclipse.emf.ecore.EObject;
005import org.nasdanika.html.model.app.Action;
006
007/**
008 * Marker interface for adapters attached to actions created by
009 * {@link EObjectActionBuilder} instances.
010 * 
011 * This interface is used to perform action configuration which requires action cross-referencing 
012 * such as linking from one action to another. Resolver's execute() is invoked after all actions are created
013 * and the action hierarchy is established. 
014 *  
015 * Parameter function provides mapping of semantic elements to actions created during the adapter execution phase
016 * and collected by the registry.
017 *  
018 * @author Pavel
019 *
020 */
021public interface EObjectActionResolver extends org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context> {
022        
023        /**
024         * Resolver context providing access to semantic element to action mapping and URI resolver.
025         * @author Pavel
026         *
027         */
028        interface Context {
029                
030                Action getAction(EObject semanticElement);
031                
032                URI resolve(Action action, URI base);
033
034                default URI resolve(Action action, Action base) {
035                        URI bURI = resolve(base, (URI) null);
036                        return resolve(action, bURI);
037                }
038                
039        }       
040
041}
