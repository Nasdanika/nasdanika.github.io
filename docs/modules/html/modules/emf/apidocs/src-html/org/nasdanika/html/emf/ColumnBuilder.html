001package org.nasdanika.html.emf;
002
003import org.eclipse.emf.ecore.ETypedElement;
004import org.nasdanika.common.ProgressMonitor;
005import org.nasdanika.html.emf.EObjectActionResolver.Context;
006import org.nasdanika.html.model.app.Action;
007import org.nasdanika.html.model.bootstrap.TableCell;
008
009/**
010 * Interface for building table columns.
011 * @author Pavel
012 *
013 */
014public interface ColumnBuilder<T> {
015
016        /**
017         * Builds a header cell.
018         * @param header
019         * @throws Exception
020         */
021        public void buildHeader(
022                        TableCell header,
023                        Action base, 
024                        ETypedElement typedElement,
025                        Context context, 
026                        ProgressMonitor progressMonitor) throws Exception;
027
028        /**
029         * Builds a value cell
030         * @throws Exception
031         */
032        public void buildCell(
033                        T rowElement, 
034                        TableCell cell,
035                        Action base, 
036                        ETypedElement typedElement,
037                        Context context, 
038                        ProgressMonitor progressMonitor) throws Exception;
039        
040}
