001package org.nasdanika.html.emf;
002
003import java.nio.charset.StandardCharsets;
004import java.security.MessageDigest;
005import java.util.List;
006import java.util.function.BiConsumer;
007
008import org.apache.commons.codec.binary.Hex;
009import org.eclipse.emf.common.util.URI;
010import org.eclipse.emf.ecore.EObject;
011import org.eclipse.emf.ecore.ETypedElement;
012import org.nasdanika.common.BiSupplier;
013import org.nasdanika.common.Context;
014import org.nasdanika.common.ProgressMonitor;
015import org.nasdanika.common.Util;
016import org.nasdanika.html.model.app.Action;
017import org.nasdanika.html.model.app.SectionStyle;
018import org.nasdanika.html.model.bootstrap.Table;
019import org.nasdanika.ncore.NcorePackage;
020import org.nasdanika.ncore.Temporal;
021import org.nasdanika.ncore.util.NcoreUtil;
022
023public class TemporalActionBuilder extends EObjectActionBuilder<Temporal> {
024        
025        public TemporalActionBuilder(Temporal value, Context context) {
026                super(value, context);          
027        }
028        
029        @Override
030        protected Action buildAction(
031                        Action action,
032                        BiConsumer<EObject,Action> registry, 
033                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
034                        ProgressMonitor progressMonitor) throws Exception {
035                Action ret = super.buildAction(action, registry, resolveConsumer, progressMonitor);             
036                Temporal eObj = getTarget();
037                URI uri = NcoreUtil.getUri(eObj);
038                String id = uri == null ? eObj.getUuid() : uri.toString();
039                String digest = Hex.encodeHexString(MessageDigest.getInstance("SHA-256").digest(id.getBytes(StandardCharsets.UTF_8)));
040                ret.setId(digest);
041                
042                String description = eObj.getDescription();
043                addContent(ret, description);
044                ret.setDescription(description);
045
046                BiSupplier<EObject, String> cPath = NcoreUtil.containmentPath(eObj);
047                if (cPath == null || Util.isBlank(cPath.getSecond())) {
048                        ret.setLocation("${base-uri}index.html");
049                } else {
050                        ret.setLocation(cPath.getSecond() + "/index.html");
051                }
052                
053                ret.setText(eObj.normalize().toString()); 
054                ret.setSectionStyle(SectionStyle.HEADER);
055                return ret;
056        }
057
058        @Override
059        public String name() {
060                return getTarget().normalize().toString();
061        }
062        
063        @Override
064        protected List<ETypedElement> getProperties() {
065                List<ETypedElement> propertiew = super.getProperties();
066                propertiew.add(NcorePackage.Literals.TEMPORAL__BASE);
067                propertiew.add(NcorePackage.Literals.TEMPORAL__OFFSET);
068                return propertiew;
069        }
070        
071        @Override
072        protected Table createPropertiesTable(
073                        Action action,
074                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
075                        ProgressMonitor progressMonitor) throws Exception {
076                Table propertiesTable = super.createPropertiesTable(action, context, progressMonitor);
077                propertiesTable.getAttributes().put("style", createText("width:auto"));
078                return propertiesTable;
079        }
080

082        
083}
