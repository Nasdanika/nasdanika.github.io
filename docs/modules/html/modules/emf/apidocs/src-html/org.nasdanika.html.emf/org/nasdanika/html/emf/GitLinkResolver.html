001package org.nasdanika.html.emf;
002
003import org.nasdanika.ncore.GitMarker;
004
005/**
006 * Context service interface for resolving Git web URL links from repository paths.
007 * @author Pavel
008 *
009 */
010public interface GitLinkResolver {
011        
012        /**
013         * Knows how to resolve GitHub links.
014         */
015        static GitLinkResolver INSTANCE = new GitLinkResolver() {
016                
017                @Override
018                public String resolve(GitMarker marker, String remoteURL) {
019                        if (remoteURL.startsWith("https://github.com")) {
020                                return remoteURL.substring(0, remoteURL.length() - 4) + "/blob/" + marker.getHead() + "/" + marker.getPath() + "#L" + marker.getLine(); 
021                        }
022                        return null;
023                }
024        };
025        
026        /**
027         * @param marker
028         * @param remoteURL
029         * @return Link for a given Git marker for a given remote URL.
030         */
031        String resolve(GitMarker marker, String remoteURL);
032
033}
