152        /**
153         * Calls createAction() to creates an action and contained actions, configures it to the extent possible
154         * with the action hierarchy being built and not all action being available. Configuration
155         * which requires action cross-referencing shall be performed in resolve() and be invoked
156         * by a EObjectActionResolver adapter execute() method. The adapter is created and attached in this method.
157         * Resolve adapter is composite and execute() shall be invoked only for the root action's adapter if it is present.
158         * The resolving adapter removes itself upon resolution.  
159         */
160        @Override
161        public Action execute(BiSupplier<Action, BiConsumer<EObject,Action>> actionAndRegistry, ProgressMonitor progressMonitor) throws Exception {
162                EObjectActionResolverAdapter resolver = new EObjectActionResolverAdapter();
163                BiConsumer<EObject, Action> registry = actionAndRegistry.getSecond();
164                Action ret = buildAction(actionAndRegistry.getFirst(), registry, resolver::add, progressMonitor);
165                registry.accept(getTarget(), ret);
166                ret.eAdapters().add(resolver);                          
167                return ret;
168        }

170        /**
171         * Creates an action and contained actions. Configures the action to the extent possible
172         * with the action hierarchy being built and not all action being available. 
173         * Configuration which requires availability of the fully constructed action hierarchy 
174         * shall be performed in commands added to resolveCommandConsumer.
175         * @param resolveConsumer Consumer of consumers to be executed at the resolve phase.
176         * @param progressMonitor
177         * @return
178         */
179        protected Action buildAction(
180                        Action action,
181                        BiConsumer<EObject,Action> registry, 
182                        java.util.function.Consumer<org.nasdanika.common.Consumer<Context>> resolveConsumer, 
183                        ProgressMonitor progressMonitor) throws Exception {
184                Action ret = action == null ? newAction(registry, resolveConsumer, progressMonitor) : action;
185
186                // Diagnostic
187                Collection<Diagnostic> diagnostic = getDiagnostic();
188                int severity = Diagnostic.OK;
189                for (Diagnostic de: diagnostic) {
190                        severity = Math.max(severity, de.getSeverity());
191                }
192                
193                if (severity > 0) {
194                        ret.setColor(getSeverityColor(severity));
195                }
196                
197                for (Diagnostic de: diagnostic) {
198                        Alert diagnosticAlert = BootstrapFactory.eINSTANCE.createAlert();
199                        diagnosticAlert.setColor(getSeverityColor(de.getSeverity()));
200                        diagnosticAlert.getContent().add(createText(StringEscapeUtils.escapeHtml4(de.getMessage())));
201                        ret.getContent().add(diagnosticAlert);
202                }               
203                
204                // Anonymous
205                
206                // Children
207                
208                // Sections
209                
210                return ret;
211        }

213        /**
214         * Called by createAction(). 
215         * If target is and instance of {@link ModelElement} and its <code>actionPrototype</code> reference is set then,
216         * if the prototype is {@link Action}, its copy is created and returned.
217         * Otherwise the prototype is adapted to {@link ActionProvider} and the provider is used to create
218         * a new Action and return. 
219         * 
220         * If the target is not and instance of ModelElement or actionPrototype is not set, then this implementation calls {@link AppFactory} createAction().
221         * 
222         * Override to customize action creation before it gets populated by createAction().
223         * @param registry
224         * @param resolveConsumer
225         * @param progressMonitor
226         * @return
227         * @throws Exception
228         */
229        protected Action newAction(
230                        BiConsumer<EObject,Action> registry, 
231                        java.util.function.Consumer<org.nasdanika.common.Consumer<Context>> resolveConsumer, 
232                        ProgressMonitor progressMonitor) throws Exception {
233                
234                T semanticElement = getTarget();
235                if (semanticElement instanceof ModelElement) {
236                        EObject actionPrototype = ((ModelElement) semanticElement).getActionPrototype();
237                        if (actionPrototype instanceof Action) {
238                                Action copy = EcoreUtil.copy((Action) actionPrototype);
239                                copy.setUuid(UUID.randomUUID().toString());
240                                return copy;
241                        }
242                        if (actionPrototype != null) {
243                                ActionProvider actionProvider = Objects.requireNonNull((ActionProvider) EcoreUtil.getRegisteredAdapter(actionPrototype, ActionProvider.class), "Cannot adapt " + actionPrototype + " to " + ActionProvider.class);
244                                return actionProvider.execute(registry, progressMonitor);
245                        }
246                }
247                return AppFactory.eINSTANCE.createAction();
248        }

297        protected void resolve(Action action, Context context, ProgressMonitor progressMonitor) throws Exception {
298                action.setLeftNavigation(createLeftNavigation(context, progressMonitor));
299                action.setFloatLeftNavigation(createFloatLeftNavigation(context, progressMonitor));
300                action.setRightNavigation(createRightNavigation(context, progressMonitor));
301                action.setFloatRightNavigation(createFloatRightNavigation(context, progressMonitor));
302                
303                // Navigation (context)
304                Action diagnosticAction = createDiagnosticAction(action, context, progressMonitor);
305                if (diagnosticAction != null) {
306                        action.getNavigation().add(diagnosticAction);
307                }
308                
309                // Content
310                
311                // Properties table
312                Table propertiesTable = createPropertiesTable(action, context, progressMonitor);
313                if (propertiesTable != null) {
314                        action.getContent().add(propertiesTable);
315                }
316                
317                // Resources
318        }

320        /**
321         * Creates a properties table from properties
322         * @param context
323         * @param progressMonitor
324         * @return Table or null if there are no properties.
325         * @throws Exception
326         */
327        protected Table createPropertiesTable(Action action, Context context, ProgressMonitor progressMonitor) throws Exception {
328                List<ETypedElement> properties = getProperties();
329                if (properties == null || properties.isEmpty()) {
330                        return null;
331                }
332                
333                Table ret = BootstrapFactory.eINSTANCE.createTable();
334                EList<TableRow> rows = ret.getRows();
335                for (ETypedElement property: properties) {
336                        Object propertyValue = getTypedElementValue(property);
337                        Collection<Diagnostic> featureDiagnostic = property instanceof EStructuralFeature ? getFeatureDiagnostic((EStructuralFeature) property) : Collections.emptyList();
338                        int severity = Diagnostic.OK;
339                        for (Diagnostic diagnostic: featureDiagnostic) {
340                                severity = Math.max(severity, diagnostic.getSeverity());
341                        }
342                        
343                        boolean isSetOrNotEmpty = property instanceof EStructuralFeature && ((EStructuralFeature) property).isChangeable() ? isSet((EStructuralFeature) property) : !isEmptyValue(property, propertyValue);
344                        if (isSetOrNotEmpty || !featureDiagnostic.isEmpty()) {                          
345                                TableRow propertyRow = BootstrapFactory.eINSTANCE.createTableRow();
346                                rows.add(propertyRow);
347                                
348//                              if (severity > 0) {
349//                                      propertyRow.setColor(getSeverityColor(severity));
350//                              }
351                                
352                                EList<TableCell> propertyCells = propertyRow.getCells();
353        
354                                TableCell propertyHeader = BootstrapFactory.eINSTANCE.createTableCell();
355                                propertyHeader.setHeader(true);
356                                propertyCells.add(propertyHeader);
357                                propertyHeader.getContent().add(createETypedElementLabel(property, false));
358                                if (severity > 0) {
359                                        Appearance headerAppearance = BootstrapFactory.eINSTANCE.createAppearance();
360                                        propertyHeader.setAppearance(headerAppearance);
361                                        org.nasdanika.html.model.bootstrap.Text aText = BootstrapFactory.eINSTANCE.createText();
362                                        headerAppearance.setText(aText);
363                                        aText.setColor(getSeverityColor(severity));
364                                }
365                                
366                                TableCell propertyValueCell = BootstrapFactory.eINSTANCE.createTableCell();
367                                propertyCells.add(propertyValueCell);
368                                if (isSetOrNotEmpty) {
369                                        EObject renderedValue = renderValue(action, property, propertyValue, context, progressMonitor);
370                                        if (renderedValue != null) {
371                                                propertyValueCell.getContent().add(renderedValue);
372                                        }
373                                }
374                                for (Diagnostic diagnostic: featureDiagnostic) {
375                                        Alert diagnosticAlert = BootstrapFactory.eINSTANCE.createAlert();
376                                        diagnosticAlert.setColor(getSeverityColor(diagnostic.getSeverity()));
377                                        diagnosticAlert.getContent().add(createText(StringEscapeUtils.escapeHtml4(diagnostic.getMessage())));
378                                        propertyValueCell.getContent().add(diagnosticAlert);
379                                }
380                                
381                        }
382                }
383                                
384                return ret;
385        }

468                        BiConsumer<EObject, Action> registry,
469                        Consumer<org.nasdanika.common.Consumer<Context>> resolveConsumer,
470                        ProgressMonitor progressMonitor) throws Exception {
471                
472                ActionProvider provider = adaptChild(child);
473                if (provider == null) {
474                        return null;

476                
477                Action ret = provider.execute(registry, progressMonitor);
478                
479                if (ret != null && resolveConsumer != null) {
480                        Adapter childResolver = EcoreUtil.getExistingAdapter(ret, EObjectActionResolver.class);
481                        if (childResolver instanceof EObjectActionResolver) {
482                                resolveConsumer.accept((EObjectActionResolver) childResolver);
483                        }
484                }
485                
486                return ret;
487        }

499        /**
500         * Empty values are not shown in property tables unless there is a diagnostic to show.
501         * This method returns true if value is null, an empty string, an empty collection, false for booleans, or zero for numbers.
502         * @param member
503         * @param value
504         * @return
505         */
506        protected boolean isEmptyValue(ETypedElement member, Object value) {
507                if (value == null) {
508                        return true;
509                }
510                if (value instanceof Collection) {
511                        return ((Collection<?>) value).isEmpty();
512                }
513                if (value instanceof String) {
514                        return Util.isBlank((String) value);            
515                }
516                if (value instanceof Number) {
517                        if (value.equals(0)) {
518                                return true;
519                        }
520                        
521                        if (value instanceof Double) {                          
522                                if (((Double) value).isNaN()) {
523                                        return true;
524                                }
525                                return Math.abs((Double) value) < 2 * Double.MIN_VALUE;
526                        }
527                        
528                        if (value instanceof Float) {
529                                if (((Float) value).isNaN()) {
530                                        return true;
531                                }
532                                return Math.abs((Float) value) < 2 * Float.MIN_VALUE;
533                        }
534                }
535                
536                if (value instanceof Temporal) {
537                        Temporal temporal = (Temporal) value;
538                        return temporal.getInstant() == null 
539                                        && temporal.getBase() == null
540                                        && temporal.getLowerBounds().isEmpty()
541                                        && temporal.getUpperBounds().isEmpty();
542                }
543                
544                return Boolean.FALSE.equals(value);
545        }

551        protected Object getTypedElementValue(EObject eObject, ETypedElement typedElement) throws Exception {           
552                if (typedElement == EcorePackage.Literals.EOBJECT___ECONTAINER) {
553                        return eObject.eContainer();
554                }
555                
556                if (typedElement == NcorePackage.Literals.MODEL_ELEMENT__URI) {
557                        return NcoreUtil.getUri(eObject);
558                }
559                
560                if (typedElement instanceof EStructuralFeature) {
561                        return eObject.eGet((EStructuralFeature) typedElement);
562                }
563                
564                EOperation eOp = (EOperation) typedElement;
565                return eObject.eInvoke(eOp, getArguments(eObject, eOp));
566        }

571                }
572                throw new UnsupportedOperationException();
573        }

597        /**
598         * Renders remote link for GitHub, returns text otherwise.
599         * @param marker
600         * @param remoteUrl
601         * @return
602         */
603        protected EObject gitRemoteLink(GitMarker marker, String remoteUrl) {
604                GitLinkResolver gitLinkResolver = null;
605                if (context != null) {
606                        gitLinkResolver = context.get(GitLinkResolver.class);
607                }
608                if (gitLinkResolver == null) {
609                        gitLinkResolver = GitLinkResolver.INSTANCE;
610                }
611                String remoteLocation = gitLinkResolver.resolve(marker, remoteUrl);
612                EList<String> headRefs = marker.getHeadRefs();
613                if (remoteLocation == null) {                   
614                        org.nasdanika.html.Table table = HTMLFactory.INSTANCE.table();
615                        
616                        Row remoteRow = table.row();
617                        remoteRow.header("Remote");
618                        remoteRow.cell(remoteUrl);
619                        
620                        Row pathRow = table.row();
621                        pathRow.header("Path");
622                        pathRow.cell(marker.getPath());
623                        
624                        Row lineRow = table.row();
625                        lineRow.header("Line");
626                        lineRow.cell(marker.getLine());
627                        
628                        Row colRow = table.row();
629                        colRow.header("Column");
630                        colRow.cell(marker.getColumn());
631                        
632                        if (!headRefs.isEmpty()) {
633                                Row refsRow = table.row();
634                                refsRow.header("Refs");
635                                if (headRefs.size() == 1) {
636                                        refsRow.cell(headRefs.get(0));
637                                } else {
638                                        refsRow.cell(HTMLFactory.INSTANCE.ul(headRefs));
639                                }                               
640                        }
641                        
642                        return createText(table.toString());                    
643                }
644                
645                StringBuilder textBuilder = new StringBuilder(marker.getPath());                        
646                if (marker.getLine() > 0) {
647                        textBuilder.append(" ").append(marker.getLine());
648                        if (marker.getColumn() > 0) {
649                                textBuilder.append(":").append(marker.getColumn());
650                        }
651                }
652                
653                Link link = AppFactory.eINSTANCE.createLink();
654                link.setText(textBuilder.toString());
655                link.setLocation(remoteLocation);
656                if (!headRefs.isEmpty()) {
657                        link.setTooltip(String.join(", ", headRefs));
658                }

665        /**
666         * Renders value.
667         * @param base Context/base action.
668         * @param typedElement Optional typed element for customizing behavior.
669         * @param value Value to render
670         * @param context Context
671         * @param progressMonitor progress monitor.
672         * @return rendered value
673         * @throws Exception
674         */
675        protected EObject renderValue(
676                        Action base, 
677                        ETypedElement typedElement,
678                        Object value, 
679                        Context context, 
680                        ProgressMonitor progressMonitor) throws Exception {             
681                if (isEmptyValue(typedElement, value)) {
682                        return null;
683                }
684                
685                if (value instanceof Instant) {
686                        return renderValue(base, typedElement, new Date(((Instant) value).toEpochMilli()), context, progressMonitor);
687                }
688                
689                if (value == Boolean.TRUE) {
690                        return createText("<i class=\"fas fa-check\"></i>");
691                }
692                
693                if (value instanceof Date) {
694                        Calendar calendar = Calendar.getInstance();
695                        calendar.setTime((Date) value);
696                        String datePart = DateFormat.getDateInstance(DateFormat.DEFAULT, Locale.getDefault()).format((Date) value);
697                        if (calendar.get(Calendar.HOUR_OF_DAY) == 0 && calendar.get(Calendar.MINUTE) == 0 && calendar.get(Calendar.SECOND) == 0) {
698                                return createText(datePart);
699                        }
700                        String timePart = DateFormat.getTimeInstance().format((Date) value);
701                        return createText(datePart + " " + timePart);
702                }
703                
704                if (value instanceof GitMarker) {
705                        GitMarker marker = (GitMarker) value;
706                        EMap<String, String> remotes = marker.getRemotes();
707                        if (remotes.size() == 1) {
708                                return gitRemoteLink(marker, remotes.get(0).getValue());
709                        } else if (!remotes.isEmpty()) {                                
710                                // TODO - a list name -> link
711                                throw new UnsupportedOperationException("Multiple remotes are not yet supported");
712                        }
713                }
714                
715                if (value instanceof Marker) {
716                        Marker marker = (Marker) value;
717                        StringBuilder textBuilder = new StringBuilder(marker.getLocation());
718                        if (marker.getLine() > 0) {
719                                textBuilder.append(" ").append(marker.getLine());
720                                if (marker.getColumn() > 0) {
721                                        textBuilder.append(":").append(marker.getColumn());
722                                }
723                        }
724                        return createText(textBuilder.toString());
725                }
726                
727                if (value instanceof Temporal) {
728                        return temporalValue(base, typedElement, (Temporal) value, context, progressMonitor);
729                }
730                
731                if (value instanceof Duration) {
732                        return createText(Temporal.formatDuration((Duration) value));
733                }               
734                
735                if (value instanceof EObject) {
736                        Action valueAction = context.getAction((EObject) value);
737                        if (valueAction == null) {
738                                return EcoreUtil.copy((EObject) value); // Returning copy to be handled at later stages, e.g. render markdown during site generation. 
739                        }
740                        
741                        URI location = context.resolve(valueAction, base);
742                        if (location == null) { 
743                                throw new IllegalArgumentException("Cannot resolve value action URI: " + value);
744                        }
745                        Link link = AppFactory.eINSTANCE.createLink();
746                        link.setText(valueAction.getText());
747                        link.setIcon(valueAction.getIcon());
748                        link.setTooltip(valueAction.getTooltip());
749                        link.setLocation(location.toString());
750                        return link;
751                } else if (value instanceof EMap) {
752                        return createEMapTable(typedElement, (EMap<?,?>) value, base, context, progressMonitor);
753                } else if (value instanceof Collection) {
754                        Collection<?> vc = (Collection<?>) value;
755                        if (vc.isEmpty()) {
756                                return null;
757                        }
758                        if (vc.size() == 1) {
759                                return renderValue(base, typedElement, vc.iterator().next(), context, progressMonitor);
760                        }                       
761                        
762                        return renderList((Collection<?>) value, true, null, base, typedElement, context, progressMonitor);
763                }
764                
765                if (typedElement == NcorePackage.Literals.MODEL_ELEMENT__URI) {
766                        return createText(value + " <i class='far fa-copy nsd-copy-to-clipboard' style='cursor:pointer' title='Copy URI to clipboard' onclick='navigator.clipboard.writeText(\"" + StringEscapeUtils.escapeHtml4(String.valueOf(value)) + "\")'/>");
767                }
768                
769                return createText(String.valueOf(value));
770        }

873                        if (renderedValue != null) {
874                                valueCell.getContent().add(renderedValue);
875                        }
876                }
877
