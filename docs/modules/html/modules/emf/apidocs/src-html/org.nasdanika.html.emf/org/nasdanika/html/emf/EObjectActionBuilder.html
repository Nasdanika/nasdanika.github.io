153        /**
154         * Calls createAction() to creates an action and contained actions, configures it to the extent possible
155         * with the action hierarchy being built and not all action being available. Configuration
156         * which requires action cross-referencing shall be performed in resolve() and be invoked
157         * by a EObjectActionResolver adapter execute() method. The adapter is created and attached in this method.
158         * Resolve adapter is composite and execute() shall be invoked only for the root action's adapter if it is present.
159         * The resolving adapter removes itself upon resolution.  
160         */
161        @Override
162        public Action execute(BiSupplier<Action, BiConsumer<EObject,Action>> actionAndRegistry, ProgressMonitor progressMonitor) throws Exception {
163                EObjectActionResolverAdapter resolver = new EObjectActionResolverAdapter();
164                BiConsumer<EObject, Action> registry = actionAndRegistry.getSecond();
165                Action ret = buildAction(actionAndRegistry.getFirst(), registry, resolver::add, progressMonitor);
166                registry.accept(getTarget(), ret);
167                ret.eAdapters().add(resolver);                          
168                return ret;
169        }

171        /**
172         * Creates an action and contained actions. Configures the action to the extent possible
173         * with the action hierarchy being built and not all action being available. 
174         * Configuration which requires availability of the fully constructed action hierarchy 
175         * shall be performed in commands added to resolveCommandConsumer.
176         * @param resolveConsumer Consumer of consumers to be executed at the resolve phase.
177         * @param progressMonitor
178         * @return
179         */
180        protected Action buildAction(
181                        Action action,
182                        BiConsumer<EObject,Action> registry, 
183                        java.util.function.Consumer<org.nasdanika.common.Consumer<Context>> resolveConsumer, 
184                        ProgressMonitor progressMonitor) throws Exception {
185                Action ret = action == null ? newAction(registry, resolveConsumer, progressMonitor) : action;
186
187                // Diagnostic
188                Collection<Diagnostic> diagnostic = getDiagnostic();
189                int severity = Diagnostic.OK;
190                for (Diagnostic de: diagnostic) {
191                        severity = Math.max(severity, de.getSeverity());
192                }
193                
194                if (severity > 0) {
195                        ret.setColor(getSeverityColor(severity));
196                }
197                
198                for (Diagnostic de: diagnostic) {
199                        Alert diagnosticAlert = BootstrapFactory.eINSTANCE.createAlert();
200                        diagnosticAlert.setColor(getSeverityColor(de.getSeverity()));
201                        diagnosticAlert.getContent().add(createText(StringEscapeUtils.escapeHtml4(de.getMessage())));
202                        ret.getContent().add(diagnosticAlert);
203                }               
204                
205                // Anonymous
206                
207                // Children
208                
209                // Sections
210                
211                return ret;
212        }

214        /**
215         * Called by createAction(). 
216         * If target is and instance of {@link ModelElement} and its <code>actionPrototype</code> reference is set then,
217         * if the prototype is {@link Action}, its copy is created and returned.
218         * Otherwise the prototype is adapted to {@link ActionProvider} and the provider is used to create
219         * a new Action and return. 
220         * 
221         * If the target is not and instance of ModelElement or actionPrototype is not set, then this implementation calls {@link AppFactory} createAction().
222         * 
223         * Override to customize action creation before it gets populated by createAction().
224         * @param registry
225         * @param resolveConsumer
226         * @param progressMonitor
227         * @return
228         * @throws Exception
229         */
230        protected Action newAction(
231                        BiConsumer<EObject,Action> registry, 
232                        java.util.function.Consumer<org.nasdanika.common.Consumer<Context>> resolveConsumer, 
233                        ProgressMonitor progressMonitor) throws Exception {
234                
235                T semanticElement = getTarget();
236                if (semanticElement instanceof ModelElement) {
237                        EObject actionPrototype = ((ModelElement) semanticElement).getActionPrototype();
238                        if (actionPrototype instanceof Action) {
239                                Action copy = EcoreUtil.copy((Action) actionPrototype);
240                                copy.setUuid(UUID.randomUUID().toString());
241                                TreeIterator<EObject> cit = copy.eAllContents();
242                                while (cit.hasNext()) {
243                                        EObject next = cit.next();
244                                        if (next instanceof ModelElement) {
245                                                ((ModelElement) next).setUuid(UUID.randomUUID().toString());
246                                        }
247                                }
248                                return copy;
249                        }
250                        if (actionPrototype != null) {
251                                ActionProvider actionProvider = Objects.requireNonNull((ActionProvider) EcoreUtil.getRegisteredAdapter(actionPrototype, ActionProvider.class), "Cannot adapt " + actionPrototype + " to " + ActionProvider.class);
252                                Action ret = actionProvider.execute(registry, progressMonitor);
253                                ret.setUuid(UUID.randomUUID().toString());                              
254                                TreeIterator<EObject> cit = ret.eAllContents();
255                                while (cit.hasNext()) {
256                                        EObject next = cit.next();
257                                        if (next instanceof ModelElement) {
258                                                ((ModelElement) next).setUuid(UUID.randomUUID().toString());
259                                        }
260                                }
261                                return ret;
262                        }
263                }
264                return AppFactory.eINSTANCE.createAction();
265        }

314        protected void resolve(Action action, Context context, ProgressMonitor progressMonitor) throws Exception {
315                action.setLeftNavigation(createLeftNavigation(context, progressMonitor));
316                action.setFloatLeftNavigation(createFloatLeftNavigation(context, progressMonitor));
317                action.setRightNavigation(createRightNavigation(context, progressMonitor));
318                action.setFloatRightNavigation(createFloatRightNavigation(context, progressMonitor));
319                
320                // Navigation (context)
321                Action diagnosticAction = createDiagnosticAction(action, context, progressMonitor);
322                if (diagnosticAction != null) {
323                        action.getNavigation().add(diagnosticAction);
324                }
325                
326                // Content
327                
328                // Properties table
329                Table propertiesTable = createPropertiesTable(action, context, progressMonitor);
330                if (propertiesTable != null) {
331                        action.getContent().add(propertiesTable);
332                }
333                
334                // Resources
335        }

337        /**
338         * Creates a properties table from properties
339         * @param context
340         * @param progressMonitor
341         * @return Table or null if there are no properties.
342         * @throws Exception
343         */
344        protected Table createPropertiesTable(Action action, Context context, ProgressMonitor progressMonitor) throws Exception {
345                List<ETypedElement> properties = getProperties();
346                if (properties == null || properties.isEmpty()) {
347                        return null;
348                }
349                
350                Table ret = BootstrapFactory.eINSTANCE.createTable();
351                EList<TableRow> rows = ret.getRows();
352                for (ETypedElement property: properties) {
353                        Object propertyValue = getTypedElementValue(property);
354                        Collection<Diagnostic> featureDiagnostic = property instanceof EStructuralFeature ? getFeatureDiagnostic((EStructuralFeature) property) : Collections.emptyList();
355                        int severity = Diagnostic.OK;
356                        for (Diagnostic diagnostic: featureDiagnostic) {
357                                severity = Math.max(severity, diagnostic.getSeverity());
358                        }
359                        
360                        boolean isSetOrNotEmpty = property instanceof EStructuralFeature && ((EStructuralFeature) property).isChangeable() ? isSet((EStructuralFeature) property) : !isEmptyValue(property, propertyValue);
361                        if (isSetOrNotEmpty || !featureDiagnostic.isEmpty()) {                          
362                                TableRow propertyRow = BootstrapFactory.eINSTANCE.createTableRow();
363                                rows.add(propertyRow);
364                                
365//                              if (severity > 0) {
366//                                      propertyRow.setColor(getSeverityColor(severity));
367//                              }
368                                
369                                EList<TableCell> propertyCells = propertyRow.getCells();
370        
371                                TableCell propertyHeader = BootstrapFactory.eINSTANCE.createTableCell();
372                                propertyHeader.setHeader(true);
373                                propertyCells.add(propertyHeader);
374                                propertyHeader.getContent().add(createETypedElementLabel(property, false));
375                                if (severity > 0) {
376                                        Appearance headerAppearance = BootstrapFactory.eINSTANCE.createAppearance();
377                                        propertyHeader.setAppearance(headerAppearance);
378                                        org.nasdanika.html.model.bootstrap.Text aText = BootstrapFactory.eINSTANCE.createText();
379                                        headerAppearance.setText(aText);
380                                        aText.setColor(getSeverityColor(severity));
381                                }
382                                
383                                TableCell propertyValueCell = BootstrapFactory.eINSTANCE.createTableCell();
384                                propertyCells.add(propertyValueCell);
385                                if (isSetOrNotEmpty) {
386                                        EObject renderedValue = renderValue(action, property, propertyValue, context, progressMonitor);
387                                        if (renderedValue != null) {
388                                                propertyValueCell.getContent().add(renderedValue);
389                                        }
390                                }
391                                for (Diagnostic diagnostic: featureDiagnostic) {
392                                        Alert diagnosticAlert = BootstrapFactory.eINSTANCE.createAlert();
393                                        diagnosticAlert.setColor(getSeverityColor(diagnostic.getSeverity()));
394                                        diagnosticAlert.getContent().add(createText(StringEscapeUtils.escapeHtml4(diagnostic.getMessage())));
395                                        propertyValueCell.getContent().add(diagnosticAlert);
396                                }
397                                
398                        }
399                }
400                                
401                return ret;
402        }

488

495                
496                if (ret != null && resolveConsumer != null) {

516        /**
517         * Empty values are not shown in property tables unless there is a diagnostic to show.
518         * This method returns true if value is null, an empty string, an empty collection, false for booleans, or zero for numbers.
519         * @param member
520         * @param value
521         * @return
522         */
523        protected boolean isEmptyValue(ETypedElement member, Object value) {
524                if (value == null) {
525                        return true;
526                }
527                if (value instanceof Collection) {
528                        return ((Collection<?>) value).isEmpty();
529                }
530                if (value instanceof String) {
531                        return Util.isBlank((String) value);            
532                }
533                if (value instanceof Number) {
534                        if (value.equals(0)) {
535                                return true;
536                        }
537                        
538                        if (value instanceof Double) {                          
539                                if (((Double) value).isNaN()) {
540                                        return true;
541                                }
542                                return Math.abs((Double) value) < 2 * Double.MIN_VALUE;
543                        }
544                        
545                        if (value instanceof Float) {
546                                if (((Float) value).isNaN()) {
547                                        return true;
548                                }
549                                return Math.abs((Float) value) < 2 * Float.MIN_VALUE;
550                        }
551                }
552                
553                if (value instanceof Temporal) {
554                        Temporal temporal = (Temporal) value;
555                        return temporal.getInstant() == null 
556                                        && temporal.getBase() == null
557                                        && temporal.getLowerBounds().isEmpty()
558                                        && temporal.getUpperBounds().isEmpty();
559                }
560                
561                return Boolean.FALSE.equals(value);
562        }

568        protected Object getTypedElementValue(EObject eObject, ETypedElement typedElement) throws Exception {           
569                if (typedElement == EcorePackage.Literals.EOBJECT___ECONTAINER) {
570                        return eObject.eContainer();
571                }
572                
573                if (typedElement == NcorePackage.Literals.MODEL_ELEMENT__URI) {
574                        return NcoreUtil.getUri(eObject);
575                }
576                
577                if (typedElement instanceof EStructuralFeature) {
578                        return eObject.eGet((EStructuralFeature) typedElement);
579                }
580                
581                EOperation eOp = (EOperation) typedElement;
582                return eObject.eInvoke(eOp, getArguments(eObject, eOp));
583        }

588                }
589                throw new UnsupportedOperationException();
590        }

614        /**
615         * Renders remote link for GitHub, returns text otherwise.
616         * @param marker
617         * @param remoteUrl
618         * @return
619         */
620        protected EObject gitRemoteLink(GitMarker marker, String remoteUrl) {
621                GitLinkResolver gitLinkResolver = null;
622                if (context != null) {
623                        gitLinkResolver = context.get(GitLinkResolver.class);
624                }
625                if (gitLinkResolver == null) {
626                        gitLinkResolver = GitLinkResolver.INSTANCE;
627                }
628                String remoteLocation = gitLinkResolver.resolve(marker, remoteUrl);
629                EList<String> headRefs = marker.getHeadRefs();
630                if (remoteLocation == null) {                   
631                        org.nasdanika.html.Table table = HTMLFactory.INSTANCE.table();
632                        
633                        Row remoteRow = table.row();
634                        remoteRow.header("Remote");
635                        remoteRow.cell(remoteUrl);
636                        
637                        Row pathRow = table.row();
638                        pathRow.header("Path");
639                        pathRow.cell(marker.getPath());
640                        
641                        Row lineRow = table.row();
642                        lineRow.header("Line");
643                        lineRow.cell(marker.getLine());
644                        
645                        Row colRow = table.row();
646                        colRow.header("Column");
647                        colRow.cell(marker.getColumn());
648                        
649                        if (!headRefs.isEmpty()) {
650                                Row refsRow = table.row();
651                                refsRow.header("Refs");
652                                if (headRefs.size() == 1) {
653                                        refsRow.cell(headRefs.get(0));
654                                } else {
655                                        refsRow.cell(HTMLFactory.INSTANCE.ul(headRefs));
656                                }                               
657                        }
658                        
659                        return createText(table.toString());                    
660                }
661                
662                StringBuilder textBuilder = new StringBuilder(marker.getPath());                        
663                if (marker.getLine() > 0) {
664                        textBuilder.append(" ").append(marker.getLine());
665                        if (marker.getColumn() > 0) {
666                                textBuilder.append(":").append(marker.getColumn());
667                        }
668                }
669                
670                Link link = AppFactory.eINSTANCE.createLink();
671                link.setText(textBuilder.toString());
672                link.setLocation(remoteLocation);
673                if (!headRefs.isEmpty()) {
674                        link.setTooltip(String.join(", ", headRefs));
675                }
676                if (remoteUrl.startsWith("https://github.com")) {
677                        link.setIcon("fab fa-github");
678                }               
679                return link;

682        /**
683         * Renders value.
684         * @param base Context/base action.
685         * @param typedElement Optional typed element for customizing behavior.
686         * @param value Value to render
687         * @param context Context
688         * @param progressMonitor progress monitor.
689         * @return rendered value
690         * @throws Exception
691         */
692        protected EObject renderValue(
693                        Action base, 
694                        ETypedElement typedElement,
695                        Object value, 
696                        Context context, 
697                        ProgressMonitor progressMonitor) throws Exception {             
698                if (isEmptyValue(typedElement, value)) {
699                        return null;
700                }
701                
702                if (value instanceof Instant) {
703                        return renderValue(base, typedElement, new Date(((Instant) value).toEpochMilli()), context, progressMonitor);
704                }
705                
706                if (value == Boolean.TRUE) {
707                        return createText("<i class=\"fas fa-check\"></i>");
708                }
709                
710                if (value instanceof Date) {
711                        Calendar calendar = Calendar.getInstance();
712                        calendar.setTime((Date) value);
713                        String datePart = DateFormat.getDateInstance(DateFormat.DEFAULT, Locale.getDefault()).format((Date) value);
714                        if (calendar.get(Calendar.HOUR_OF_DAY) == 0 && calendar.get(Calendar.MINUTE) == 0 && calendar.get(Calendar.SECOND) == 0) {
715                                return createText(datePart);
716                        }
717                        String timePart = DateFormat.getTimeInstance().format((Date) value);
718                        return createText(datePart + " " + timePart);
719                }
720                
721                if (value instanceof GitMarker) {
722                        GitMarker marker = (GitMarker) value;
723                        EMap<String, String> remotes = marker.getRemotes();
724                        if (remotes.size() == 1) {
725                                return gitRemoteLink(marker, remotes.get(0).getValue());
726                        } else if (!remotes.isEmpty()) {                                
727                                // TODO - a list name -> link
728                                throw new UnsupportedOperationException("Multiple remotes are not yet supported");
729                        }
730                }
731                
732                if (value instanceof Marker) {
733                        Marker marker = (Marker) value;
734                        StringBuilder textBuilder = new StringBuilder(marker.getLocation());
735                        if (marker.getLine() > 0) {
736                                textBuilder.append(" ").append(marker.getLine());
737                                if (marker.getColumn() > 0) {
738                                        textBuilder.append(":").append(marker.getColumn());
739                                }
740                        }
741                        return createText(textBuilder.toString());
742                }
743                
744                if (value instanceof Temporal) {
745                        return temporalValue(base, typedElement, (Temporal) value, context, progressMonitor);
746                }
747                
748                if (value instanceof Duration) {
749                        return createText(Temporal.formatDuration((Duration) value));
750                }               
751                
752                if (value instanceof EObject) {
753                        Action valueAction = context.getAction((EObject) value);
754                        if (valueAction == null) {
755                                return EcoreUtil.copy((EObject) value); // Returning copy to be handled at later stages, e.g. render markdown during site generation. 
756                        }
757                        
758                        URI location = context.resolve(valueAction, base);
759                        if (location == null) { 
760                                throw new IllegalArgumentException("Cannot resolve value action URI: " + value);
761                        }
762                        Link link = AppFactory.eINSTANCE.createLink();
763                        link.setText(valueAction.getText());
764                        link.setIcon(valueAction.getIcon());
765                        link.setTooltip(valueAction.getTooltip());
766                        link.setLocation(location.toString());
767                        return link;
768                } else if (value instanceof EMap) {
769                        return createEMapTable(typedElement, (EMap<?,?>) value, base, context, progressMonitor);
770                } else if (value instanceof Collection) {
771                        Collection<?> vc = (Collection<?>) value;
772                        if (vc.isEmpty()) {
773                                return null;
774                        }
775                        if (vc.size() == 1) {
776                                return renderValue(base, typedElement, vc.iterator().next(), context, progressMonitor);
777                        }                       
778                        
779                        return renderList((Collection<?>) value, true, null, base, typedElement, context, progressMonitor);
780                }
781                
782                if (typedElement == NcorePackage.Literals.MODEL_ELEMENT__URI) {
783                        return createText(value + " <i class='far fa-copy nsd-copy-to-clipboard' style='cursor:pointer' title='Copy URI to clipboard' onclick='navigator.clipboard.writeText(\"" + StringEscapeUtils.escapeHtml4(String.valueOf(value)) + "\")'/>");
784                }
785                
786                return createText(String.valueOf(value));
787        }

876                for (Object key: sortKeys(eMap.keySet())) {

890                        if (renderedValue != null) {

892                        }
893                }
894
