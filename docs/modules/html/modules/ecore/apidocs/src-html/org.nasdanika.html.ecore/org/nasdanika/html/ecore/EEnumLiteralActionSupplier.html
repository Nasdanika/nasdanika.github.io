001package org.nasdanika.html.ecore;
002
003import org.eclipse.emf.ecore.EClass;
004import org.eclipse.emf.ecore.EEnumLiteral;
005import org.eclipse.emf.ecore.EPackage;
006import org.nasdanika.common.Context;
007import org.nasdanika.common.ProgressMonitor;
008import org.nasdanika.html.model.app.Action;
009
010public class EEnumLiteralActionSupplier extends ENamedElementActionSupplier<EEnumLiteral> {
011
012        public EEnumLiteralActionSupplier(EEnumLiteral value, Context context, java.util.function.Function<EPackage,String> ePackagePathComputer) {
013                super(value, context, ePackagePathComputer);
014        }
015        
016        @Override
017        public Action execute(EClass contextEClass, ProgressMonitor progressMonitor) throws Exception {
018                Action action = super.execute(contextEClass, progressMonitor);
019                action.setId(eObject.eClass().getName() + "-" + encodeEPackage(eObject.getEEnum().getEPackage()) + "-" + eObject.getEEnum().getName() + "-" + eObject.getName());
020                action.setName(eObject.getName());              
021                return action;
022        }
023                
024}
