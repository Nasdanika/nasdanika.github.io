001package org.nasdanika.html.jstree.impl;
002
003import java.util.Arrays;
004import java.util.HashMap;
005import java.util.Map;
006import java.util.ServiceLoader;
007
008import org.json.JSONArray;
009import org.json.JSONObject;
010import org.nasdanika.html.HTMLElement;
011import org.nasdanika.html.HTMLFactory;
012import org.nasdanika.html.HTMLPage;
013import org.nasdanika.html.Tag;
014import org.nasdanika.html.TagName;
015import org.nasdanika.html.TokenSource;
016import org.nasdanika.html.jstree.JsTreeContextMenuItem;
017import org.nasdanika.html.jstree.JsTreeFactory;
018import org.nasdanika.html.jstree.JsTreeNode;
019
020/**
021 * HTML factory which relies on Bootstrap styles and scripts.
022 * @author Pavel
023 *
024 */
025public class DefaultJsTreeFactory implements JsTreeFactory {
026        
027        private HTMLFactory htmlFactory;
028
029        public DefaultJsTreeFactory(HTMLFactory htmlFactory) {
030                this.htmlFactory = htmlFactory;
031        }
032        
033        /**
034         * For the service loader
035         */
036        public DefaultJsTreeFactory() {
037                this(ServiceLoader.load(HTMLFactory.class).findFirst().get());
038        }
039        
040        @Override
041        public HTMLFactory getHTMLFactory() {
042                return htmlFactory;
043        }
044                
045        @Override
046        public JsTreeNode jsTreeNode() {
047                return new JsTreeNodeImpl();
048        }
049        
050        @Override
051        public JsTreeContextMenuItem jsTreeContextMenuItem() {
052                return new JsTreeContextMenuItemImpl();
053        }
054
055        @Override
056        public <P extends HTMLPage> P cdn(P page) {
057                page.stylesheet("https://cdnjs.cloudflare.com/ajax/libs/jstree/3.3.11/themes/default/style.min.css");
058                page.script("https://cdnjs.cloudflare.com/ajax/libs/jstree/3.3.11/jstree.min.js");
059                return page;
060        }
061
062        @Override
063        public JSONObject buildJsTree(Iterable<JsTreeNode> roots) {
064                JSONObject ret = new JSONObject();
065                JSONObject core = new JSONObject();
066                ret.put("core", core);
067                JSONArray jRoots = new JSONArray();
068                for (JsTreeNode root: roots) {
069                        jRoots.put(root.toJSON());
070                }
071                core.put("data", jRoots);
072                return ret;
073        }
074        
075        @Override
076        public JSONObject buildJsTree(JsTreeNode... roots) {
077                return buildJsTree(Arrays.asList(roots));
078        }
079
080        @Override
081        public String buildAjaxJsTree(String nodesUrl, String contextMenuUrl) {
082                TokenSource tokens = getHTMLFactory().tokenSource("nodesUrl", nodesUrl).put("contextMenuUrl", contextMenuUrl);
083                return getHTMLFactory().interpolate(DefaultJsTreeFactory.class.getResource(contextMenuUrl == null ? "ajaxTree.js" : "ajaxTreeWithContextMenu.js"), tokens);
084        }
085
086        @Override
087        public Tag bind(String selector, Object jsTree, Object filter, Object searchInputSelector) {
088                StringBuilder code = new StringBuilder("$(document).ready( function() {").append(System.lineSeparator());
089        
090                if (filter == null) {
091                        code.append("$('"+selector+"').jstree("+jsTree+");");
092                } else {                                
093                        code.append("$('"+selector+"').jstree(function(tree) { " + filter + " return tree; }(" + jsTree + "));");
094                }
095                
096                if (searchInputSelector != null) {
097                        code.append(System.lineSeparator());
098                        Map<String,Object> tokens = new HashMap<>();
099                        tokens.put("searchInputSelector", searchInputSelector);
100                        tokens.put("timer", "window['nsd_jstTreeSearchTimer_" + getHTMLFactory().nextId() + "']");
101                        tokens.put("treeSelector", selector);
102                        code.append(getHTMLFactory().interpolate(DefaultJsTreeFactory.class.getResource("bindSearch.js"), tokens));                     
103                }
104                
105                code.append(System.lineSeparator()).append("});");
106                return htmlFactory.nonEmptyTag(TagName.script, code.toString());
107        }
108
109        @Override
110        public Tag bind(HTMLElement<?> htmlElement, Object jsTree, Object filter, Object searchInputSelector) {
111                if (htmlElement.getId() == null) {
112                        htmlElement.id(htmlFactory.nextId());
113                }
114                return bind("#"+htmlElement.getId(), jsTree, filter, searchInputSelector);
115        }
116        
117}
