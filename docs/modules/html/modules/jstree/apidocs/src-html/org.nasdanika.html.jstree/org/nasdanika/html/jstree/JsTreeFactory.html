001package org.nasdanika.html.jstree;
002
003import org.json.JSONObject;
004import org.nasdanika.html.HTMLElement;
005import org.nasdanika.html.HTMLFactory;
006import org.nasdanika.html.HTMLPage;
007import org.nasdanika.html.Tag;
008import org.nasdanika.html.jstree.impl.DefaultJsTreeFactory;
009
010public interface JsTreeFactory {
011        
012        JsTreeFactory INSTANCE = new DefaultJsTreeFactory(HTMLFactory.INSTANCE);
013        
014        JsTreeNode jsTreeNode();
015        
016        JsTreeContextMenuItem jsTreeContextMenuItem();
017        
018        /**
019         * Add JsTree stylesheet and script elements referencing CDN locations. Does not add jQuery. 
020         * @param page
021         * @return
022         */
023        <P extends HTMLPage> P cdn(P page);
024        
025        /**
026         * Builds JsTree object from a list of root nodes.
027         * @param roots
028         * @return
029         */
030        JSONObject buildJsTree(Iterable<JsTreeNode> roots);
031        
032        /**
033         * Builds JsTree object from a list of root nodes.
034         * @param roots
035         * @return
036         */
037        JSONObject buildJsTree(JsTreeNode... roots);
038        
039        
040        /**
041         * Builds JsTree object for loading nodes and context menus using AJAX from the specified URL's.
042         * Node id is passed to the server side in "id" request parameter.
043         * @param nodesUrl URL to load nodes.
044         * @param contextMenuUrl URL to load context menus. Can be null. 
045         * @return String representation because JSON does not support functions.
046         */
047        String buildAjaxJsTree(String nodesUrl, String contextMenuUrl);
048        
049        /**
050         * Generates script tag for binding jsTree to html element.
051         * @param selector
052         * @param jsTree
053         * @param filter Filter script to manipulate jsTree object before passing it to jsTree. It can be used to add functions to jsTree, which are not supported by JSON. 
054         * The filter script can access jsTree via <code>tree</code> variable.
055         * @param searchInputSelector If not null, binds input to tree search.
056         * @return
057         */
058        Tag bind(String selector, Object jsTree, Object filter, Object searchInputSelector);
059        
060        /**
061         * Generates script tag for binding jsTree to html element.
062         * @param jsTree
063         * @param filter Filter script to manipulate jsTree object before passing it to jsTree. It can be used to add functions to jsTree, which are not supported by JSON. 
064         * The filter script can access jsTree via <code>tree</code> variable.
065         * @param searchInputSelector If not null, binds input to tree search.
066         * @param selector
067         * @return
068         */
069        Tag bind(HTMLElement<?> htmlElement, Object jsTree, Object filter, Object searchInputSelector);
070        
071        HTMLFactory getHTMLFactory();   
072        
073}
