001package org.nasdanika.html.jstree;
002
003import java.util.List;
004import java.util.function.Predicate;
005
006import org.json.JSONObject;
007
008/**
009 * This interface can be used to build JsTree JSON data. ``toString()`` method of the implementation outputs JSON.
010 * @author Pavel Vlasov
011 *
012 */
013public interface JsTreeNode {
014        
015        default JsTreeNode opened() {
016                return opened(true);
017        }
018        
019        JsTreeNode opened(boolean opened);
020        
021        default JsTreeNode selected() {
022                return selected(true);
023        }
024        
025        JsTreeNode selected(boolean selected);
026        
027        default JsTreeNode disabled() {
028                return disabled(true);
029        }
030        
031        JsTreeNode disabled(boolean disabled);
032        
033        /**
034         * Sets icon - image URL or (glyph) icon class.
035         * @param icon
036         * @return
037         */
038        JsTreeNode icon(String icon);   
039        
040        /**
041         * Used to traverse the tree and collect values.
042         * @author Pavel Vlasov
043         *
044         * @param <V>
045         */
046        interface Collector<R> {
047                
048                R visit(JsTreeNode node, List<R> childResults);
049                
050        }
051        
052        <R> R accept(Collector<R> collector);
053        
054        JsTreeNode id(Object id);
055        
056        Object getId();
057        
058        JsTreeNode text(Object text);
059        
060        List<JsTreeNode> children();
061        
062        /**
063         * Creates a new node and adds to the list of children.
064         * @return
065         */
066        JsTreeNode createChild();
067
068        /**
069         * Attribute for the generated LI node.
070         * @param name
071         * @param value
072         * @return
073         */
074        JsTreeNode listItemAttribute(String name, Object value);
075
076        /**
077         * Set custom Attribute/property of the JsTree JSON object.
078         * @param name
079         * @param value
080         * @return
081         */
082        JsTreeNode attribute(String name, Object value);
083        
084        /**
085         * Attribute for the generated A node.
086         * @param name
087         * @param value
088         * @return
089         */
090        JsTreeNode anchorAttribute(String name, Object value);
091        
092        JSONObject toJSON();
093        
094        /**
095         * Outputs to JSON only nodes which match the predicate or contain children matching the predicate.
096         * @param filter predicate or null.
097         * @return JSON object or null if there is no match.
098         */
099        JSONObject toJSON(Predicate<JsTreeNode> filter);        
100        
101        /**
102         * Sets the application defined node data. This data is not used for rendering.
103         * Applications may put arbitrary objects in this field. 
104         * @param data
105         * @return this UI element.
106         */
107        JsTreeNode setData(Object data);
108        
109        /**
110         * Returns the application defined node data, or null if it has not been set. 
111         * @return
112         */
113        Object getData();
114        
115        /**
116         * Sets the application defined property of the node with the specified name to the given value.
117         * Applications may associate arbitrary objects with nodes. This data is not used for rendering.
118         * @param key
119         * @param data
120         * @return this UI element.
121         */
122        JsTreeNode setData(Object key, Object data);
123        
124        /**
125         * If this method is invoked and children collection is empty then toJSON outputs children:true, which can be used in Ajax lazy loading.
126         * @return
127         */
128        JsTreeNode hasChildren();
129        
130        /**
131         * Returns the application defined property of the node with the specified name, or null if it has not been set.
132         * @param Key
133         * @return
134         */
135        Object getData(Object key);
136        
137        
138}
