001package org.nasdanika.html.jstree;
002
003import org.json.JSONObject;
004
005/**
006 * Builds JsTree context menu item - https://www.jstree.com/api/#/?q=$.jstree.defaults.contextmenu&f=$.jstree.defaults.contextmenu.items.
007 * Use toString() to produce JavaScript.
008 * @author Pavel Vlasov
009 *
010 */
011public interface JsTreeContextMenuItem {
012
013        /**
014         * @param separatorBefore a boolean indicating if there should be a separator before this item
015         * @return
016         */
017        JsTreeContextMenuItem separatorBefore(boolean separatorBefore);
018        
019        default JsTreeContextMenuItem separatorBefore() {
020                return separatorBefore(true);
021        }
022
023        /**
024         * 
025         * @param separatorAfter a boolean indicating if there should be a separator after this item
026         * @return
027         */
028        JsTreeContextMenuItem separatorAfter(boolean separatorAfter);
029        
030        default JsTreeContextMenuItem separatorAfter() {
031                return separatorAfter(true);
032        }
033        
034        /**
035         * 
036         * @param disabled a boolean indicating if this action should be disabled
037         * @return
038         */
039        JsTreeContextMenuItem disabled(boolean disabled);
040        
041        default JsTreeContextMenuItem disabled() {
042                return disabled(true);
043        }
044        
045        /**
046         * 
047         * @param label a string - the name of the action (could be a function returning a string)
048         * @return
049         */
050        JsTreeContextMenuItem label(Object label);
051        
052        /**
053         * 
054         * @param title a string - an optional tooltip for the item
055         * @return
056         */
057        JsTreeContextMenuItem title(Object title);
058        
059        /**
060         * 
061         * @param action a function to be executed if this item is chosen
062         * @return
063         */
064        JsTreeContextMenuItem action(Object action);
065        
066        /**
067         * 
068         * @param icon a string, can be a path to an icon or a className, if using an image that is in the current directory use a ./ prefix, otherwise it will be detected as a class
069         * @return
070         */
071        JsTreeContextMenuItem icon(Object icon);
072        
073        /**
074         * 
075         * @param shortcut keyCode which will trigger the action if the menu is open (for example 113 for rename, which equals F2)
076         * @return
077         */
078        JsTreeContextMenuItem shortcut(Object shortcut);
079        
080        /**
081         * 
082         * @param shortcutLabel shortcut label (like for example F2 for rename)
083         * @return
084         */
085        JsTreeContextMenuItem shortcutLabel(Object shortcutLabel);
086        
087        /**
088         * Creates a sub-menu item and returns it. If an item for a given key already exists, returns the existing item.
089         * @param key sub-menu item key.
090         * @return sub-menu item.
091         */
092        JsTreeContextMenuItem createSubMenuItem(String key);
093        
094        /**
095         * Adds a sub-menu item.
096         * @param key
097         * @param item
098         * @return this item.
099         */
100        JsTreeContextMenuItem addSubMenuItem(String key, JsTreeContextMenuItem item);
101
102        /**
103         * Sets sub-menu. If sub-menu set this way is not null, then items added or created with addSubMenuItem and createSubMenuItem are ignored.
104         */
105        JsTreeContextMenuItem subMenu(Object subMenu);
106        
107        /**
108         * JSON representation of the menu object - for AJAX menus.
109         * Action is converted to JSON and shall be converted to a function on the browser side.
110         * @return
111         */
112        JSONObject toJSON();
113}
