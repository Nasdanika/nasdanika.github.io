001package org.nasdanika.html.jstree.impl;
002
003import java.util.Arrays;
004
005import org.json.JSONArray;
006import org.json.JSONObject;
007import org.nasdanika.html.HTMLElement;
008import org.nasdanika.html.HTMLFactory;
009import org.nasdanika.html.HTMLPage;
010import org.nasdanika.html.Tag;
011import org.nasdanika.html.TagName;
012import org.nasdanika.html.TokenSource;
013import org.nasdanika.html.jstree.JsTreeContextMenuItem;
014import org.nasdanika.html.jstree.JsTreeFactory;
015import org.nasdanika.html.jstree.JsTreeNode;
016
017/**
018 * HTML factory which relies on Bootstrap styles and scripts.
019 * @author Pavel
020 *
021 */
022public class DefaultJsTreeFactory implements JsTreeFactory {
023        
024        private HTMLFactory htmlFactory;
025
026        public DefaultJsTreeFactory(HTMLFactory htmlFactory) {
027                this.htmlFactory = htmlFactory;
028        }
029        
030        @Override
031        public HTMLFactory getHTMLFactory() {
032                return htmlFactory;
033        }
034                
035        @Override
036        public JsTreeNode jsTreeNode() {
037                return new JsTreeNodeImpl();
038        }
039        
040        @Override
041        public JsTreeContextMenuItem jsTreeContextMenuItem() {
042                return new JsTreeContextMenuItemImpl();
043        }
044
045        @Override
046        public <P extends HTMLPage> P cdn(P page) {
047                page.stylesheet("https://cdnjs.cloudflare.com/ajax/libs/jstree/3.3.11/themes/default/style.min.css");
048                page.script("https://cdnjs.cloudflare.com/ajax/libs/jstree/3.3.11/jstree.min.js");
049                return page;
050        }
051
052        @Override
053        public JSONObject buildJsTree(Iterable<JsTreeNode> roots) {
054                JSONObject ret = new JSONObject();
055                JSONObject core = new JSONObject();
056                ret.put("core", core);
057                JSONArray jRoots = new JSONArray();
058                for (JsTreeNode root: roots) {
059                        jRoots.put(root.toJSON());
060                }
061                core.put("data", jRoots);
062                return ret;
063        }
064        
065        @Override
066        public JSONObject buildJsTree(JsTreeNode... roots) {
067                return buildJsTree(Arrays.asList(roots));
068        }
069
070        @Override
071        public String buildAjaxJsTree(String nodesUrl, String contextMenuUrl) {
072                TokenSource tokens = getHTMLFactory().tokenSource("nodesUrl", nodesUrl).put("contextMenuUrl", contextMenuUrl);
073                return getHTMLFactory().interpolate(getClass().getResource(contextMenuUrl == null ? "ajaxTree.js" : "ajaxTreeWithContextMenu.js"), tokens);
074        }
075
076        @Override
077        public Tag bind(String selector, Object jsTree, Object filter) {
078                StringBuilder code = new StringBuilder("$(document).ready( function() {").append(System.lineSeparator());
079        
080                if (filter == null) {
081                        code.append("$('"+selector+"').jstree("+jsTree+");");
082                } else {                                
083                        code.append("$('"+selector+"').jstree(function(tree) { " + filter + " return tree; }(" + jsTree + "));");
084                }
085                
086                code.append(System.lineSeparator()).append("});");
087                return htmlFactory.nonEmptyTag(TagName.script, code.toString());
088        }
089
090        @Override
091        public Tag bind(HTMLElement<?> htmlElement, Object jsTree, Object filter) {
092                if (htmlElement.getId() == null) {
093                        htmlElement.id(htmlFactory.nextId());
094                }
095                return bind("#"+htmlElement.getId(), jsTree, filter);
096        }
097        
098}
