001package org.nasdanika.html.jstree.impl;
002
003import java.util.Arrays;
004import java.util.HashMap;
005import java.util.Map;
006
007import org.json.JSONArray;
008import org.json.JSONObject;
009import org.nasdanika.html.HTMLElement;
010import org.nasdanika.html.HTMLFactory;
011import org.nasdanika.html.HTMLPage;
012import org.nasdanika.html.Tag;
013import org.nasdanika.html.TagName;
014import org.nasdanika.html.TokenSource;
015import org.nasdanika.html.jstree.JsTreeContextMenuItem;
016import org.nasdanika.html.jstree.JsTreeFactory;
017import org.nasdanika.html.jstree.JsTreeNode;
018
019/**
020 * HTML factory which relies on Bootstrap styles and scripts.
021 * @author Pavel
022 *
023 */
024public class DefaultJsTreeFactory implements JsTreeFactory {
025        
026        private HTMLFactory htmlFactory;
027
028        public DefaultJsTreeFactory(HTMLFactory htmlFactory) {
029                this.htmlFactory = htmlFactory;
030        }
031        
032        @Override
033        public HTMLFactory getHTMLFactory() {
034                return htmlFactory;
035        }
036                
037        @Override
038        public JsTreeNode jsTreeNode() {
039                return new JsTreeNodeImpl();
040        }
041        
042        @Override
043        public JsTreeContextMenuItem jsTreeContextMenuItem() {
044                return new JsTreeContextMenuItemImpl();
045        }
046
047        @Override
048        public <P extends HTMLPage> P cdn(P page) {
049                page.stylesheet("https://cdnjs.cloudflare.com/ajax/libs/jstree/3.3.11/themes/default/style.min.css");
050                page.script("https://cdnjs.cloudflare.com/ajax/libs/jstree/3.3.11/jstree.min.js");
051                return page;
052        }
053
054        @Override
055        public JSONObject buildJsTree(Iterable<JsTreeNode> roots) {
056                JSONObject ret = new JSONObject();
057                JSONObject core = new JSONObject();
058                ret.put("core", core);
059                JSONArray jRoots = new JSONArray();
060                for (JsTreeNode root: roots) {
061                        jRoots.put(root.toJSON());
062                }
063                core.put("data", jRoots);
064                return ret;
065        }
066        
067        @Override
068        public JSONObject buildJsTree(JsTreeNode... roots) {
069                return buildJsTree(Arrays.asList(roots));
070        }
071
072        @Override
073        public String buildAjaxJsTree(String nodesUrl, String contextMenuUrl) {
074                TokenSource tokens = getHTMLFactory().tokenSource("nodesUrl", nodesUrl).put("contextMenuUrl", contextMenuUrl);
075                return getHTMLFactory().interpolate(DefaultJsTreeFactory.class.getResource(contextMenuUrl == null ? "ajaxTree.js" : "ajaxTreeWithContextMenu.js"), tokens);
076        }
077
078        @Override
079        public Tag bind(String selector, Object jsTree, Object filter, Object searchInputSelector) {
080                StringBuilder code = new StringBuilder("$(document).ready( function() {").append(System.lineSeparator());
081        
082                if (filter == null) {
083                        code.append("$('"+selector+"').jstree("+jsTree+");");
084                } else {                                
085                        code.append("$('"+selector+"').jstree(function(tree) { " + filter + " return tree; }(" + jsTree + "));");
086                }
087                
088                if (searchInputSelector != null) {
089                        code.append(System.lineSeparator());
090                        Map<String,Object> tokens = new HashMap<>();
091                        tokens.put("searchInputSelector", searchInputSelector);
092                        tokens.put("timer", "window['nsd_jstTreeSearchTimer_" + getHTMLFactory().nextId() + "']");
093                        tokens.put("treeSelector", selector);
094                        code.append(getHTMLFactory().interpolate(DefaultJsTreeFactory.class.getResource("bindSearch.js"), tokens));                     
095                }
096                
097                code.append(System.lineSeparator()).append("});");
098                return htmlFactory.nonEmptyTag(TagName.script, code.toString());
099        }
100
101        @Override
102        public Tag bind(HTMLElement<?> htmlElement, Object jsTree, Object filter, Object searchInputSelector) {
103                if (htmlElement.getId() == null) {
104                        htmlElement.id(htmlFactory.nextId());
105                }
106                return bind("#"+htmlElement.getId(), jsTree, filter, searchInputSelector);
107        }
108        
109}
