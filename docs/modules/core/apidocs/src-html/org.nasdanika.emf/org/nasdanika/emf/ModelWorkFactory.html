001package org.nasdanika.emf;
002
003import org.eclipse.emf.common.util.URI;
004import org.eclipse.emf.ecore.EObject;
005import org.eclipse.emf.ecore.EPackage;
006import org.eclipse.emf.ecore.resource.Resource;
007import org.eclipse.emf.ecore.resource.ResourceSet;
008import org.eclipse.emf.ecore.resource.impl.ResourceSetImpl;
009import org.eclipse.emf.ecore.xmi.impl.XMIResourceFactoryImpl;
010import org.nasdanika.common.Context;
011import org.nasdanika.common.Supplier;
012import org.nasdanika.common.SupplierFactory;
013import org.nasdanika.ncore.NcorePackage;
014
015/**
016 * This class loads a model with the root element, adapts it to {@link SupplierFactory}, and delegates its methods to it. 
017 * @author Pavel
018 *
019 */
020public class ModelWorkFactory<T> implements SupplierFactory<T> { 
021
022        protected EObject root; 
023        protected Resource modelResource;
024
025        /**
026         * Creates a generator by loading a generator model specified by the platform URI, e.g. ``org.nasdanika.codegen.tests.models/models/static-text/basic.codegen``, into a new {@link ResourceSet}. 
027         * @param platformPluginUri
028         */
029        public ModelWorkFactory(String platformPluginUri, EPackage... ePackages) {
030                this(URI.createPlatformPluginURI(platformPluginUri, false), ePackages);
031        }
032        
033        /**
034         * Creates a generator by loading a generator model from the specified URI into a new {@link ResourceSet}. 
035         * @param modelUri
036         */
037        public ModelWorkFactory(URI modelUri, EPackage... ePackages) {
038                this(createResourceSet(ePackages), modelUri);
039        }       
040        
041        /**
042         * Creates a generator by loading a generator model specified by the platform URI, e.g. ``org.nasdanika.codegen.tests.models/models/static-text/basic.codegen``, into 
043         * the specified {@link ResourceSet}.
044         * @param resourceSet Resource set to load the model to, see ``ModelGenerator(ResourceSet resourceSet, URI modelUri)`` for details. 
045         * @param platformPluginUri
046         */
047        public ModelWorkFactory(ResourceSet resourceSet, String platformPluginUri) {
048                this(resourceSet, URI.createPlatformPluginURI(platformPluginUri, false));
049        }
050        
051        /**
052         * Creates a new {@link ResourceSet} if it was not passed to a constructor explicitly. This implementation creates a new resource set and configures it to load the model by registering an {@link XMIResourceFactoryImpl}
053         * and putting {@link NcorePackage} to the package registry. If the model uses model elements from other {@link EPackage}s pass  a resource set constructed by the client to one of the 
054         * constructors which accept resource set. The resource set can be constructed by this method and then additional packages may be registered by the client code before passing the resource
055         * set to a constructor.   
056         * @return
057         */
058        public static ResourceSet createResourceSet(EPackage... ePackages) {
059                ResourceSetImpl resourceSet = new ResourceSetImpl();
060                resourceSet.getResourceFactoryRegistry().getExtensionToFactoryMap().put(Resource.Factory.Registry.DEFAULT_EXTENSION, new XMIResourceFactoryImpl());
061                for (EPackage ePackage: ePackages) {
062                        resourceSet.getPackageRegistry().put(ePackage.getNsURI(), ePackage);
063                }
064                return resourceSet;
065        }
066
067        /**
068         * Override to specify result type of supplier.
069         * @return
070         */
071        @SuppressWarnings("unchecked")
072        protected Class<T> getSupplierResultType() {
073                return (Class<T>) Object.class;
074        }
075        
076        /**
077         * Creates a generator by loading a generator model from the specified URI into the specified {@link ResourceSet}.
078         * @param resourceSet Resource set to load the model to. 
079         */
080        public ModelWorkFactory(ResourceSet resourceSet, URI modelUri) {
081                modelResource = resourceSet.getResource(modelUri, true);
082                root = modelResource.getContents().iterator().next();
083        }
084        
085        @Override
086        public Supplier<T> create(Context context) {
087                return EObjectAdaptable.adaptToSupplierFactoryNonNull(root, getSupplierResultType()).create(context);
088        }
089        
090        public Resource getModelResource() {
091                return modelResource;
092        }
093
094}
