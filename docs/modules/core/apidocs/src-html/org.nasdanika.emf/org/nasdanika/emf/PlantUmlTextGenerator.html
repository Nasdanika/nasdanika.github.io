078        /**
079         * Filters the collection retaining only model elements which shall be documented.
080         * @param <T>
081         * @param elements
082         * @return
083         */
084        protected <T extends EModelElement> List<T> retainDocumentable(Collection<T> elements) {
085                return elements.stream().filter(elementPredicate).collect(Collectors.toList());
086        }

089        private static final String IMPLEMENTS_RELATION = "<|..";

093        protected void appendClassStart(String modifiers, String classType, String name, String link, String style) throws IOException {
094                if (modifiers != null) {
095                        collector.append(modifiers);
096                        collector.append(" ");
097                }
098                
099                collector
100                        .append(classType)
101                        .append(" ")
102                        .append(name)
103                        .append(Util.isBlank(link) ? "" : " [[" + link + "]]")
104                        .append(Util.isBlank(style) ? "" : " " + style)
105                        .append(" {\n");
106        }

108        protected String getLabel(ENamedElement namedElement) {
109                return labelProvider.apply(namedElement, namedElement.getName());
110        }
111        
112        protected void appendClassEnd() throws IOException {
113                collector.append("}\n");
114        }

120        protected void appendMember(String modifiers, String visibility, String type, String name, Iterable<String> parameters) throws IOException {
121                collector.append("\t");
122                if (visibility != null) {
123                        collector.append(visibility);
124                }
125                if (modifiers != null) {
126                        collector.append(modifiers);
127                        collector.append(" ");
128                }
129                if (type != null) {
130                        collector.append(type);
131                        collector.append(" ");
132                }
133                if (name != null) {
134                        collector.append(name);
135                }
136                if (parameters != null) {
137                        collector.append("(");
138                        Iterator<String> it = parameters.iterator();
139                        while (it.hasNext()) {
140                                collector.append(it.next());
141                                if (it.hasNext()) {
142                                        collector.append(", ");
143                                }
144                        }
145                        collector.append(")");
146                }
147                collector.append("\n");
148        }
149        
150        protected void appendAttribute(String modifiers, String visibility, String type, String name) throws IOException {
151                appendMember(modifiers, visibility, type, name, null);
152        }
153
154        protected void appendOperation(String modifiers, String visibility, String type, String name, Iterable<String> parameters) throws IOException {
155                appendMember(modifiers, visibility, type, name, parameters);
156        }

158        public void appendGeneralization(EClass subClass, EGenericType superType) throws IOException {
159                if (elementPredicate.test(superType.getEClassifier())) {
160                        EClass superClass = (EClass) superType.getEClassifier();
161                        collector
162                                .append(qualifiedName(superClass))
163                                .append(" ")
164                                .append(superClass.isInterface() && !subClass.isInterface() ? IMPLEMENTS_RELATION : EXTENDS_RELATION)
165                                .append(" ")
166                                .append(qualifiedName(subClass));
167                        
168                        if (!superType.getETypeArguments().isEmpty()) {
169                                collector.append(" : <");
170                                Iterator<EGenericType> it = superType.getETypeArguments().iterator();
171                                while (it.hasNext()) {
172                                        collector.append(genericName(it.next()));
173                                        if (it.hasNext()) {
174                                                collector.append(",");
175                                        }
176                                }
177                                collector.append(">");
178                        }               
179                        
180                        collector.append("\n");
181                }
182        }

186        private DependencyTracer<EClassifier> OUT_DEPENDENCY_TRACER = new DependencyTracer<EClassifier>() {
187
188                @Override
189                protected Iterable<EClassifier> getDependencies(EClassifier obj) {
190                        Collection<EClassifier> ret = new HashSet<>();
191                        if (obj instanceof EClass) {
192                                EClass eClass = (EClass) obj;
193                                ret.addAll(eClass.getESuperTypes());
194                                for (EReference ref: retainDocumentable(eClass.getEReferences())) {                                     
195                                        EClass eReferenceType = ref.getEReferenceType();
196                                        if (elementPredicate.test(eReferenceType)) {
197                                                EClass associationTarget = NcoreUtil.getAssociationTarget(ref);
198                                                if (associationTarget == null) {
199                                                        ret.add(eReferenceType);
200                                                } else if (elementPredicate.test(associationTarget)) {
201                                                        ret.add(eReferenceType);
202                                                        ret.add(associationTarget);                                                     
203                                                }                                               
204                                        }                                       
205                                }
206                                
207                                ret.addAll(EmfUtil.collectTypeDependencies(eClass));                            
208                        }
209                        return retainDocumentable(ret);
210                }
211                
212        };

214        private DependencyTracer<EClassifier> inDependencyTracer = new DependencyTracer<EClassifier>() {
215
216                @Override
217                protected Iterable<EClassifier> getDependencies(EClassifier obj) {
218                        Collection<EClassifier> ret = new HashSet<>();
219                        if (obj instanceof EClass) {
220                                ret.addAll(getSubTypes((EClass) obj));
221                                ret.addAll(getReferrers((EClass) obj));
222                        }
223                        ret.addAll(getUses(obj));
224                        return retainDocumentable(ret);
225                }
226                
227        };

229        private DependencyTracer<EClassifier> bothDependencyTracer = new DependencyTracer<EClassifier>() {
230        
231                @Override
232                protected Iterable<EClassifier> getDependencies(EClassifier obj) {
233                        Collection<EClassifier> ret = new HashSet<>();
234                        if (obj instanceof EClass) {
235                                // In
236                                EClass eClass = (EClass) obj;
237                                ret.addAll(getSubTypes(eClass));
238                                ret.addAll(getReferrers(eClass));
239                                
240                                // Out
241                                ret.addAll(eClass.getESuperTypes());
242                                for (EReference ref: retainDocumentable(eClass.getEReferences())) {
243                                        EClass eReferenceType = ref.getEReferenceType();
244                                        if (elementPredicate.test(eReferenceType)) {
245                                                EClass associationTarget = NcoreUtil.getAssociationTarget(ref);
246                                                if (associationTarget == null) {
247                                                        ret.add(eReferenceType);
248                                                } else if (elementPredicate.test(associationTarget)) {
249                                                        ret.add(eReferenceType);
250                                                        ret.add(associationTarget);                                                     
251                                                }                                               
252                                        }                                       
253                                }
254                                
255                                ret.addAll(EmfUtil.collectTypeDependencies(eClass));                            
256                        }
257
258                        // In
259                        ret.addAll(getUses(obj));
260                        
261                        return retainDocumentable(ret);
262                }
263

266        /**
267         * Appends core classifiers, their related classifiers, and relationships
268         * @param coreClassifiers
269         * @throws IOException 
270         */
271        @Override
272        public void appendWithRelationships(
273                        Collection<? extends EClassifier> coreClassifiers,
274                        RelationshipDirection direction,
275                        int depth) {
276                
277                try {
278                        Set<EClassifier> coreSet = new HashSet<>(retainDocumentable(coreClassifiers));                          
279                        Set<EClassifier> relatedSet = new HashSet<>();
280                        
281                        switch (direction) {
282                        case both:
283                                relatedSet = bothDependencyTracer.trace(coreSet, depth);
284                                break;
285                        case in:
286                                relatedSet = inDependencyTracer.trace(coreSet, depth);
287                                break;
288                        case out:
289                                relatedSet = OUT_DEPENDENCY_TRACER.trace(coreSet, depth);
290                                break;
291                        default:
292                                break;          
293                        }
294                        
295                        Set<EClassifier> allClassifiers = new HashSet<>(coreSet);
296                        
297                        for (EClassifier rc: relatedSet) {
298                                if (!coreSet.contains(rc) && getEClassifierLink(rc) != null) {
299                                        allClassifiers.add(rc);                         
300                                }
301                        }
302                        
303                        for (EClassifier cc: coreSet) {
304                                appendEClassifier(cc, allClassifiers);
305                        }
306                        
307                        for (EClassifier rc: relatedSet) {
308                                if (!coreSet.contains(rc) && getEClassifierLink(rc) != null) {
309                                        String style = NcoreUtil.getNasdanikaAnnotationDetail(rc, DIAGRAM_STYLE_KEY);
310                                        if (style == null) {
311                                                style = RELATED_BACKGROUND;
312                                        } else {
313                                                style = Context.singleton("background", RELATED_BACKGROUND).interpolateToString(style);
314                                        }
315                                        
316                                        appendEClassifier(rc, style, allClassifiers);
317                                }
318                        }               
319                        
320                        for (EClassifier c: allClassifiers) {
321                                if (c instanceof EClass) {
322                                        for (EGenericType gsc: ((EClass) c).getEGenericSuperTypes()) {
323                                                if (allClassifiers.contains(gsc.getEClassifier())) {
324                                                        appendGeneralization((EClass) c, gsc);
325                                                }
326                                        }
327                                }
328                        }
329                        
330                        Set<EReference> processedOpposites = new HashSet<>();
331                        for (EClassifier c: allClassifiers) {
332                                if (c instanceof EClass) {
333                                        for (EReference ref: ((EClass) c).getEReferences()) {
334                                                EClass eReferenceType = ref.getEReferenceType();
335//                                              EClass associationTarget = getAssociationTarget(ref);
336//                                              if (associationTarget == null) { // Regular relationship
337                                                        if (!processedOpposites.contains(ref) && allClassifiers.contains(eReferenceType)) {
338                                                                appendEReference(ref);
339                                                                EReference opposite = NcoreUtil.getOpposite(ref);
340                                                                if (opposite!=null) {
341                                                                        processedOpposites.add(opposite);
342                                                                }
343                                                        }
344//                                              } else { // Association class, no handling of opposites right now
345//                                                      if (allClassifiers.contains(eReferenceType) && allClassifiers.contains(associationTarget)) {
346//                                                              appendAssociation(ref);                                                         
347//                                                      }
348//                                              }
349                                        }
350                                }
351                        }       
352                        
353                        // Type dependencies
354                        for (EClassifier c: allClassifiers) {
355                                for (EClass src: getUses(c)) {
356                                        if (allClassifiers.contains(src)) {
357                                                appendTypeDependency(src, c);
358                                        }                               
359                                }
360                        }
361                } catch (IOException e) {
362                        throw new NasdanikaException(e);
363                }
364        }

366        protected static String getMultiplicity(EStructuralFeature feature) {
367                if (feature.isMany()) {
368                        int lowerBound = feature.getLowerBound();
369                        int upperBound = feature.getUpperBound();
370                        if (lowerBound == upperBound) {
371                                return String.valueOf(lowerBound);
372                        }
373                        if (lowerBound==0 && upperBound==-1) {
374                                return "*";
375                        }
376                        return String.valueOf(lowerBound)+".."+(upperBound == -1 ? "*" : upperBound);
377                }
378                
379                return "";
380        }

382        /**
383         * @param ref
384         * @return Empty string if there are no EKeys, a comma separated list of EKeys in parenthesis otherwise.
385         */
386        protected String eKeys(EReference ref) {
387                StringBuilder ret = new StringBuilder();
388                EList<EAttribute> eKeys = ref.getEKeys();
389                if (!eKeys.isEmpty()) {
390                        ret.append("(");                
391                        ret.append(String.join(",", eKeys.stream().map(this::getLabel).collect(Collectors.toList())));
392                        ret.append(")");
393                }
394                
395                return ret.toString();
396        }

398        protected void appendEReference(EReference ref) throws IOException {
399                if (elementPredicate.test(ref)) {
400                        collector.append(qualifiedName(ref.getEContainingClass()));
401                        collector.append(" ");
402                        EReference opposite = NcoreUtil.getOpposite(ref);
403                        if (ref.isContainment()) {
404                                collector.append("*");
405                        } else if (opposite!=null) {
406                                collector.append("\"");
407                                collector.append(getLabel(opposite));
408                                collector.append(eKeys(opposite));
409                                String multiplicity = getMultiplicity(opposite);
410                                if (!multiplicity.isEmpty()) {
411                                        collector.append("["+multiplicity+"]");
412                                }
413                                collector.append("\" ");
414                        }
415                        
416                        String diagramStyle = NcoreUtil.getNasdanikaAnnotationDetail(ref, DIAGRAM_STYLE_KEY);
417                        String relationLine = Util.isBlank(diagramStyle) ? "--" : "-[" + diagramStyle + "]-";
418                        String associationRelation = relationLine + ">";
419                        
420                        if (opposite == null) {
421                                collector.append(associationRelation);
422                        } else {
423                                collector.append(relationLine);
424                                if (opposite.isContainment()) {
425                                        collector.append("*");
426                                }
427                        }               
428                        collector.append(" ");
429                        
430                        String multiplicity = getMultiplicity(ref);
431                        if (opposite == null) {
432                                if (!multiplicity.isEmpty()) {
433                                        collector
434                                                .append("\"")
435                                                .append(multiplicity)
436                                                .append("\" ");                                         
437                                }
438                        } else {
439                                collector.append("\"");
440                                collector.append(getLabel(ref));                        
441                                collector.append(eKeys(ref));
442                                if (!multiplicity.isEmpty()) {
443                                        collector.append("["+multiplicity+"]");
444                                }
445                                collector.append("\" ");                        
446                        }               
447                        
448                        collector.append(qualifiedName(ref.getEReferenceType()));               
449                        
450                        if (opposite == null) {
451                                collector
452                                        .append(" : ")
453                                        .append(getLabel(ref));
454        
455                                collector.append(eKeys(ref));                   
456                                collector.append(genericTypeArguments(ref.getEGenericType()));
457                        }
458                        
459                        collector.append(System.lineSeparator());
460                }
461        }

510        protected void appendTypeDependency(EClass source, EClassifier target) throws IOException {
511                if (elementPredicate.test(source) && elementPredicate.test(target)) {
512                        collector.append(qualifiedName(source));
513                        collector.append(" ");
514                        collector.append(DEPENDENCY_RELATION);
515                        collector.append(" ");
516                        collector.append(qualifiedName(target));                
517                        collector.append(System.lineSeparator());
518                }
519        }

521        /**
522         * Finds all subtypes in the resourceset. 
523         * @return
524         */
525        protected Collection<EClass> getSubTypes(EClass eClass) {
526                TreeIterator<?> acit;
527                Resource eResource = eClass.eResource();
528                if (eResource == null) {
529                        EPackage ePackage = eClass.getEPackage();
530                        if (ePackage == null) {
531                                return Collections.emptySet();
532                        }
533                        acit = ePackage.eAllContents();
534                } else {
535                        ResourceSet resourceSet = eResource.getResourceSet();
536                        acit = resourceSet == null ? eResource.getAllContents() : resourceSet.getAllContents();
537                }
538                Set<EClass> ret = new HashSet<>();
539                acit.forEachRemaining(obj -> {
540                        if (obj instanceof EClass && ((EClass) obj).getESuperTypes().contains(eClass)) {
541                                ret.add((EClass) obj);
542                        }
543                });
544                return retainDocumentable(ret);
545        }

547        protected Collection<EClass> getReferrers(EClass eClass) {

551        /**
552         * Finds all referrers in the resourceset. 
553         * @return
554         */
555        protected Collection<EClass> getReferrers(EClass eClass, boolean includeAssociations) {
556                TreeIterator<?> acit;
557                Resource eResource = eClass.eResource();
558                if (eResource == null) {
559                        EPackage ePackage = eClass.getEPackage();
560                        if (ePackage == null) {
561                                return Collections.emptySet();
562                        }
563                        acit = ePackage.eAllContents();
564                } else {
565                        ResourceSet resourceSet = eResource.getResourceSet();
566                        acit = resourceSet == null ? eResource.getAllContents() : resourceSet.getAllContents();
567                }
568                Set<EClass> ret = new HashSet<>();
569                acit.forEachRemaining(obj -> {
570                        if (obj instanceof EReference && ((EReference) obj).getEReferenceType() == eClass) {
571                                // Finding association sources. TODO - handle a case when association target does not pass the predicate test, but the association class does.
572                                EClass referrer = ((EReference) obj).getEContainingClass();
573                                if (includeAssociations) {
574                                        for (EClass superReferrer: getReferrers(referrer, false)) {
575                                                for (EReference superReference: superReferrer.getEReferences()) {
576                                                        if (superReference.getEReferenceType() == referrer) {
577                                                                EClass associationTarget = NcoreUtil.getAssociationTarget(superReference);
578                                                                if (associationTarget == eClass) {
579                                                                        ret.add(superReferrer);
580                                                                }
581                                                        }
582                                                }
583                                        }
584                                }
585                                ret.add(referrer);
586                        }
587                });
588                return retainDocumentable(ret);
589        }

591        /**
592         * Finds all type uses in the resourceset. 
593         * @return
594         */
595        protected Collection<EClass> getUses(EClassifier eClassifier) {
596                TreeIterator<?> acit;
597                Resource eResource = eClassifier.eResource();
598                if (eResource == null) {
599                        EPackage ePackage = eClassifier.getEPackage();
600                        if (ePackage == null) {
601                                return Collections.emptySet();
602                        }
603                        acit = ePackage.eAllContents();
604                } else {
605                        ResourceSet resourceSet = eResource.getResourceSet();
606                        acit = resourceSet == null ? eResource.getAllContents() : resourceSet.getAllContents();
607                }
608                Set<EClass> ret = new HashSet<>();
609                acit.forEachRemaining(obj -> {
610                        if (obj instanceof EClass && EmfUtil.collectTypeDependencies((EClass) obj).contains(eClassifier)) {
611                                ret.add((EClass) obj);
612                        }
613                });
614                return retainDocumentable(ret);
615        }

618                String style = Context.EMPTY_CONTEXT.interpolateToString(NcoreUtil.getNasdanikaAnnotationDetail(eClassifier, DIAGRAM_STYLE_KEY));

622        public void appendEClassifier(EClassifier eClassifier, String style, Set<EClassifier> allClassifiers) throws IOException {
623                if (elementPredicate.test(eClassifier)) {
624                        if (eClassifier instanceof EClass) {
625                                appendEClass((EClass) eClassifier, style, allClassifiers);
626                        } else if (eClassifier instanceof EEnum) {
627                                appendEEnum((EEnum) eClassifier, style, allClassifiers);                        
628                        } else if (eClassifier instanceof EDataType) {
629                                appendEDataType((EDataType) eClassifier, style, allClassifiers);
630                        }
631                }
632        }

634        protected String qualifiedName(EClassifier eClassifier) {
635                EPackage ePackage = eClassifier.getEPackage();
636                if (ePackage == null) {
637                        return getLabel(eClassifier);                           
638                }
639                return "\""+getLabel(ePackage)+" ("+ePackage.getNsURI()+")."+getLabel(eClassifier)+"\"";
640        }

642        protected String typeParameters(EClassifier eClassifier) {
643                if (eClassifier.getETypeParameters().isEmpty()) {
644                        return "";
645                }
646                StringBuilder typeParameters = new StringBuilder();
647                for (ETypeParameter typeParameter: eClassifier.getETypeParameters()) {
648                        if (typeParameters.length() > 0) {
649                                typeParameters.append(",");
650                        }
651                        typeParameters.append(genericName(typeParameter));
652                }               
653                
654                return "<" + typeParameters +">";
655        }       
656        
657        protected String genericName(ETypeParameter typeParameter) {
658                StringBuilder ret = new StringBuilder(getLabel(typeParameter));
659                for (EGenericType bound : typeParameter.getEBounds()) {
660                        if (bound.getEUpperBound() != null) {
661                                ret.append(" extends ").append(genericName(bound.getEUpperBound()));
662                        }
663                        if (bound.getELowerBound() != null) {
664                                ret.append(" super ").append(genericName(bound.getELowerBound()));
665                        }
666                }
667                
668                return ret.toString();
669        }
670        
671        protected String genericName(EGenericType eGenericType) {
672                if (eGenericType == null) {
673                        return "void";
674                }
675                StringBuilder ret = new StringBuilder();                
676                if (eGenericType.getETypeParameter() != null) {
677                        ret.append(getLabel(eGenericType.getETypeParameter()));
678                } else if (eGenericType.getEClassifier() != null) {
679                        ret.append(getLabel(eGenericType.getEClassifier()));                    
680                }
681                ret.append(genericTypeArguments(eGenericType));
682                return ret.toString();
683        }
684
685        protected String genericTypeArguments(EGenericType eGenericType) {
686                StringBuilder ret = new StringBuilder();
687                Iterator<EGenericType> it = eGenericType.getETypeArguments().iterator();
688                if (it.hasNext()) {
689                        ret.append("<");
690                        while (it.hasNext()) {
691                                ret.append(genericName(it.next()));
692                                if (it.hasNext()) {
693                                        ret.append(",");
694                                }
695                        }
696                        ret.append(">");
697                }
698                return ret.toString();
699        }

707        public void appendEClass(EClass eClass, String style, Set<EClassifier> allClassifiers) throws IOException {
708                String modifiers = eClass.isAbstract() && !eClass.isInterface() ? "abstract" : null;
709                appendClassStart(modifiers, eClass.isInterface() ? "interface" : "class", qualifiedName(eClass) + typeParameters(eClass), getEClassifierLink(eClass), style);
710                if (isAppendAttributes(eClass)) {
711                        for (EAttribute attribute: eClass.getEAttributes().stream().filter(elementPredicate).sorted(eNamedElementComparator).collect(Collectors.toList())) {                    
712                                EGenericType eGenericType = attribute.getEGenericType();
713                                if (eGenericType != null) {
714                                        appendAttribute(null, null, genericName(eGenericType) + (attribute.isMany() ? "[]" : ""), getLabel(attribute));
715                                }                                               
716                        }
717                }
718                collector.append(CLASS_COMPARTMENT_SEPARATOR_LINE);
719                for (EReference reference: eClass.getEReferences().stream().filter(elementPredicate).sorted(eNamedElementComparator).collect(Collectors.toList())) {
720                        EGenericType eGenericType = reference.getEGenericType();                        
721                        if (eGenericType == null || !allClassifiers.contains(reference.getEReferenceType())) {
722                                appendAttribute(null, null, genericName(eGenericType) + (reference.isMany() ? "[]" : ""), getLabel(reference));
723                        }                                               
724                }
725                
726                collector.append(CLASS_COMPARTMENT_SEPARATOR_LINE);
727                if (isAppendOperations(eClass)) {
728                        for (EOperation op : eClass.getEOperations().stream().filter(elementPredicate).sorted(eNamedElementComparator).collect(Collectors.toList())) {
729                                Collection<String> parameters = new ArrayList<String>();
730                                for (EParameter parameter : op.getEParameters()) {
731                                        String paramString = getLabel(parameter);
732                                        if (parameter.getEGenericType() != null) {
733                                                paramString = genericName(parameter.getEGenericType()) + " " + paramString;
734                                        }
735                                        parameters.add(paramString);
736                                }
737                                appendOperation(null, null, genericName(op.getEGenericType()), getLabel(op), parameters);
738                        }
739                }
740                appendClassEnd();
741        }

743        /**
744         * Constructs {@link EClassifier} with the first documentation sentence as a tooltip. 
745         * @param eClassifier
746         * @return
747         */
748        protected String getEClassifierLink(EClassifier eClassifier) {
749                String link = null;
750                if (eClassifier != null && eClassifierLinkResolver != null) {
751                        link = eClassifierLinkResolver.apply(eClassifier);
752                        if (link != null && eModelElementFirstDocSentenceProvider != null) {
753                                String firstDocSentence = eModelElementFirstDocSentenceProvider.apply(eClassifier);
754                                if (!org.nasdanika.common.Util.isBlank(firstDocSentence)) {
755                                        link += "{" + firstDocSentence + "}";
756                                }
757                        }
758                }
759                return link;
760        }
