001package org.nasdanika.emf;
002
003import java.util.Collections;
004import java.util.List;
005
006import org.eclipse.emf.ecore.EAnnotation;
007import org.eclipse.emf.ecore.EModelElement;
008import org.nasdanika.common.MarkdownHelper;
009
010/**
011 * Class loading annotations from model elements.
012 * @author Pavel
013 *
014 * @param <T>
015 */
016public class AnnotationSource<T extends EModelElement> extends MarkdownHelper {
017
018        protected T modelElement;
019
020        protected AnnotationSource(T modelElement) {
021                this.modelElement = modelElement;
022        }
023
024        /**
025         * Source for Nasdanika annotations.
026         */
027        public static final String NASDANIKA_ANNOTATION_SOURCE = "urn:org.nasdanika";   
028        
029        /**
030         * @return List of sources to use for loading annotations such as icon.
031         */
032        protected List<String> getAnnotationSources() {
033                return Collections.singletonList(NASDANIKA_ANNOTATION_SOURCE);
034        }
035        
036        public String getAnnotation(String key) {
037                for (String annotationSource: getAnnotationSources()) {
038                        EAnnotation annotation = modelElement.getEAnnotation(annotationSource);
039                        if (annotation != null) {
040                                String details = annotation.getDetails().get(key);
041                                if (details != null) {
042                                        return details;
043                                }
044                        }
045                }
046                return null;
047        }               
048
049}
