001package org.nasdanika.emf;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import org.eclipse.emf.common.util.BasicDiagnostic;
007import org.eclipse.emf.common.util.Diagnostic;
008import org.eclipse.emf.common.util.DiagnosticChain;
009import org.eclipse.emf.ecore.EObject;
010import org.eclipse.emf.ecore.EStructuralFeature;
011
012/**
013 * Helper class to reduce amount of diagnostic/validation code.
014 * @author Pavel Vlasov
015 *
016 */
017public class DiagnosticHelper {
018        
019        private String source;
020        private int code;
021        private DiagnosticChain diagnostics;
022        private Object owner;
023
024        /**
025         * DiagnosticHelper shall typically be created at the top of validate() method.
026         * @param diagnostics
027         * @param source
028         * @param code
029         * @param owner
030         */
031        public DiagnosticHelper(DiagnosticChain diagnostics, String source, int code, Object owner) {
032                this.diagnostics = diagnostics;
033                this.source = source;
034                this.code = code;
035                this.owner = owner;
036        }
037        
038        private boolean success = true;
039        
040        /**
041         * @return true if there were no error level diagnostics.
042         */
043        public boolean isSuccess() {
044                return success;
045        }
046        
047        /**
048         * Adds diagnostic.
049         * @param severity
050         * @param message Diagnostic message.
051         * @param messageKey Message key for localization, can be null.
052         * @param feature Feature under diagnostic. Can be null.
053         */
054        public void addDiagnostic(int severity, String message, String messageKey, EStructuralFeature feature) {
055                if (diagnostics != null) {
056                        if (feature != null && owner instanceof EObject && !((EObject) owner).eClass().getEAllStructuralFeatures().contains(feature)) {
057                                throw new IllegalArgumentException("EStructuralFeature "+feature+" does not belong to owner's EClass "+ ((EObject) owner).eClass());
058                        }
059                        List<Object> data = new ArrayList<>();
060                        data.add(owner);
061                        if (feature != null) {
062                                data.add(feature);
063                        }
064                        if (messageKey != null) {
065                                data.add(messageKey);
066                        }
067                        diagnostics.add(new BasicDiagnostic(severity, source, code, message, data.toArray()));
068                
069                        if (Diagnostic.ERROR == severity) {
070                                success = false;
071                        }
072                        
073                }
074        }
075        
076        public void error(String message, String messageKey, EStructuralFeature feature) {
077                addDiagnostic(Diagnostic.ERROR, message, messageKey, feature);
078        }
079        
080        public void error(String message, String messageKey) {
081                error(message, messageKey, null);
082        }
083        
084        public void error(String message, EStructuralFeature feature) {
085                error(message, null, feature);
086        }
087        
088        public void error(String message) {
089                error(message, null, null);
090        }
091        
092        public void warning(String message, String messageKey, EStructuralFeature feature) {
093                addDiagnostic(Diagnostic.WARNING, message, messageKey, feature);
094        }
095        
096        public void warning(String message, String messageKey) {
097                warning(message, messageKey, null);
098        }
099        
100        public void warning(String message, EStructuralFeature feature) {
101                warning(message, null, feature);
102        }
103        
104        public void warning(String message) {
105                warning(message, null, null);
106        }
107        
108        public void info(String message, String messageKey, EStructuralFeature feature) {
109                addDiagnostic(Diagnostic.INFO, message, messageKey, feature);
110        }
111        
112        public void info(String message, String messageKey) {
113                info(message, messageKey, null);
114        }
115        
116        public void info(String message, EStructuralFeature feature) {
117                info(message, null, feature);
118        }
119        
120        public void info(String message) {
121                info(message, null, null);
122        }
123        
124}
