001package org.nasdanika.cli;
002
003import java.io.File;
004import java.io.IOException;
005import java.io.InputStream;
006import java.net.HttpURLConnection;
007import java.net.URL;
008import java.net.URLConnection;
009import java.util.ArrayList;
010import java.util.Collection;
011import java.util.LinkedHashMap;
012import java.util.List;
013import java.util.Map;
014import java.util.Map.Entry;
015import java.util.Properties;
016import java.util.function.Function;
017
018import org.json.JSONObject;
019import org.json.JSONTokener;
020import org.nasdanika.common.Context;
021import org.nasdanika.common.ProgressMonitor;
022import org.yaml.snakeyaml.Yaml;
023
024import picocli.CommandLine.Option;
025
026/**
027 * Base class for commands which build {@link Context} from command line options.
028 * @author Pavel
029 *
030 */
031public abstract class ContextCommand extends CommandBase {
032                        
033        @Option(                        
034                        names = {"-c", "--context-entry"},
035                        description = {
036                                        "Context entries. ",
037                                        "Shadow entries in contexts and mounts."
038                        })
039    private Map<String, String> entries = new LinkedHashMap<>();
040        
041        @Option(                        
042                        names = {"-M", "--context-mount"},
043                        description = {
044                                        "MappingContext resource URL relative to the current directory. "
045                                        + "YAML, JSON, or properties. "
046                                        + "In properties dots are treated as key path separators. "
047                                        + "Type is inferred from the content type header, if it is present, "
048                                        + "or extension. Mounts shadow context entries."
049                        })
050        private Map<String,String> mounts = new LinkedHashMap<>();
051                
052        @Option(                        
053                        names = {"-C", "--context"},
054                        paramLabel = "URL",
055                        description = {
056                                        "Context resource URL relative to the current directory. "
057                                         + "YAML, JSON, or properties. "
058                                        + "In properties dots are treated as key path separators. "
059                                        + "Type is inferred from the content type header, if it is present, "
060                                        + "or extension. Contexts are composed in the order of definition, "
061                                        + "later context entries shadowing the former"
062                        })
063        private List<String> contexts = new ArrayList<>();
064        
065        /**
066         * Called by createContext to obtain an instance of context to be configured.
067         * This implementation creates a new empty context.
068         * Subclasses can override this method to add properties and services into the "primordial" context so they can be used during
069         * context creation. E.g. a URI service can be used by context builders to resolve relative URI's. 
070         * @return
071         */
072        protected Context newContext() {
073                return Context.EMPTY_CONTEXT;                           
074        }
075        
076        /**
077         * Override to contribute to context building. E.g. declare a mix-in and return it from here.
078         * @return
079         */
080        protected Collection<ContextBuilder> getContextBuilders() {
081                return new ArrayList<>();
082        }
083        
084        /**
085         * Creates and configures context by adding mounts, contexts and calling context builders.
086         * @param progressMonitor
087         * @return
088         * @throws Exception
089         */
090        protected Context createContext(ProgressMonitor progressMonitor) throws Exception {
091                Context ret = newContext();             
092                for (Entry<String, String> mountEntry: mounts.entrySet()) {
093                        ret = ret.mount(load(mountEntry.getValue()), mountEntry.getKey());
094                }
095                for (String location: contexts) {
096                        ret = ret.compose(load(location));
097                }
098                
099                ret = Context.wrap(entries::get).compose(ret);
100                for (ContextBuilder contextBuilder: getContextBuilders()) {
101                        ret = contextBuilder.build(ret, progressMonitor);
102                }
103                return ret;
104        }
105
106        /**
107         * Loads context from a location resolved relative to the current directory.
108         * @param location
109         * @return
110         * @throws IOException
111         */
112        protected Context load(String location) throws IOException {
113                URL url = new File(".").toURI().resolve(location).toURL();
114                URLConnection connection = url.openConnection();
115                
116                if (location.endsWith(".json")) {
117                        try (InputStream in = connection.getInputStream()) {
118                                return loadJson(in);
119                        }
120                }
121                
122                if (location.endsWith(".yml") || location.endsWith(".yaml")) {
123                        try (InputStream in = connection.getInputStream()) {
124                                return loadYaml(in);
125                        }                       
126                }
127                if (location.endsWith(".properties")) {
128                        try (InputStream in = connection.getInputStream()) {
129                                return loadProperties(in);
130                        }                       
131                }
132                
133                // Can't deduce content type from extension, attempting to use Content-Type header
134                if (connection instanceof HttpURLConnection && ((HttpURLConnection) connection).getResponseCode() == HttpURLConnection.HTTP_OK) {
135                        String contentType = ((HttpURLConnection) connection).getHeaderField("Content-Type");
136                        if ("application/json".equals(contentType)) {
137                                try (InputStream in = connection.getInputStream()) {
138                                        return loadJson(in);
139                                }                                
140                        } 
141                        
142                        if ("application/x-yaml".equals(contentType)) {
143                                try (InputStream in = connection.getInputStream()) {
144                                        return loadYaml(in);
145                                }                               
146                        } 
147                        
148                        if ("text/x-java-properties".equals(contentType)) {
149                                try (InputStream in = connection.getInputStream()) {
150                                        return loadProperties(in);
151                                }                               
152                        }
153                }
154                
155                throw new IllegalArgumentException("Unknown resource type: " + url);
156        }       
157        
158        protected Context loadJson(InputStream in) throws IOException {
159                JSONObject jo = new JSONObject(new JSONTokener(in));
160                return Context.wrap(jo.toMap()::get);
161        }
162        
163        @SuppressWarnings("unchecked")
164        protected Context loadYaml(InputStream in) throws IOException {
165                Yaml yaml = new Yaml();
166                return Context.wrap(((Map<String,Object>) yaml.load(in))::get);
167        }
168        
169        protected Context loadProperties(InputStream in) throws IOException {
170                Properties props = new Properties();
171                props.load(in);
172                Function<String, Object> source = props::getProperty;            
173                return Context.wrap(source.compose(key -> ((String) key).replace('.', '/')));
174        }
175        
176}
