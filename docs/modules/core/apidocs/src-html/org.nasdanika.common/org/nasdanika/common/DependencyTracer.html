001package org.nasdanika.common;
002
003import java.util.Collections;
004import java.util.HashMap;
005import java.util.Map;
006import java.util.Set;
007
008/**
009 * Base class for dependency tracers. 
010 * @author Pavel Vlasov
011 *
012 * @param <T>
013 */
014public abstract class DependencyTracer<T> {
015
016        /**
017         * Traces dependencies from a set of starting objects
018         * @param from Starting object
019         * @param depth Trace depth. 0 - just the starting objects, -1 - infinite.
020         * @param chain Combine results with the chain.
021         * @return
022         */
023        public Set<T> trace(Set<T> from, int depth) {
024                Map<T, Integer> distanceTracker = createDepthTracker();
025                for (T f: from) {
026                        trace(f, depth, distanceTracker);
027                }
028                return distanceTracker.keySet();
029                
030        }
031        
032        public Set<T> trace(T from, int depth) {
033                return trace(Collections.singleton(from), depth);
034        }
035        
036        /**
037         * Creates depth tracker.
038         * @return HashMap, subclasses may override to return, for example, TreeMap.
039         */
040        protected Map<T,Integer> createDepthTracker() {
041                return new HashMap<T,Integer>();
042        }
043        
044        private void trace(T from, int depth, Map<T, Integer> depthTracker) {
045                Integer objDepth = depthTracker.get(from);
046                if (objDepth == null || objDepth < depth) {
047                        depthTracker.put(from, depth);
048                        if (depth != 0) {
049                                for (T dep: getDependencies(from)) {
050                                        trace(dep, depth == -1 ? depth : depth - 1, depthTracker);
051                                }
052                        }
053                }
054        }                       
055        
056        /**
057         * @param obj
058         * @return Dependencies to trace.
059         */
060        protected abstract Iterable<T> getDependencies(T obj);
061
062}
