001package org.nasdanika.common;
002
003import java.util.AbstractMap;
004import java.util.ArrayList;
005import java.util.List;
006import java.util.Objects;
007import java.util.Set;
008
009public abstract class AbstractSplitJoinMap<S,C,K,V> extends AbstractMap<K, V> {
010        
011        /**
012         * @return State to be split into chunks.
013         */
014        protected abstract S getState();
015        
016        /**
017         * Sets state composed from chunks created from elements.
018         * @param state
019         */
020        protected abstract void setState(S state);
021        
022        /**
023         * Splits state into an array of chunks
024         * @param state
025         * @return
026         */
027        protected abstract List<C> split(S state);
028        
029        /**
030         * Joins an array of chunks into a state.
031         * @param chunks
032         * @return
033         */
034        protected abstract S join(List<C> chunks);
035        
036        /**
037         * Loads a chunk - converts it to an unmodifiable entry which does not have to implement setValue()
038         * as it is not used anyway.
039         * @param chunk
040         * @return
041         */
042        protected abstract Entry<K,V> load(C chunk);
043        
044        /**
045         * Stores an element - converts it into a chunk  
046         * @param element
047         * @return
048         */
049        protected abstract C store(Entry<K,V> element);
050        
051        private AbstractSplitJoinSet<S, C, Entry<K,V>> entrySet = new AbstractSplitJoinSet<S, C, Entry<K,V>>() {
052
053                @Override
054                protected S getState() {
055                        return AbstractSplitJoinMap.this.getState();
056                }
057
058                @Override
059                protected void setState(S state) {
060                        AbstractSplitJoinMap.this.setState(state);
061                }
062
063                @Override
064                protected List<C> split(S state) {
065                        return AbstractSplitJoinMap.this.split(state);
066                }
067
068                @Override
069                protected S join(List<C> chunks) {
070                        return AbstractSplitJoinMap.this.join(chunks);
071                }
072
073                @Override
074                protected Entry<K, V> load(C chunk) {
075                        Entry<K, V> valueEntry = AbstractSplitJoinMap.this.load(chunk);                 
076                        return new Entry<K,V>() {
077                                V value = valueEntry.getValue();
078
079                                @Override
080                                public K getKey() {
081                                        return valueEntry.getKey();
082                                }
083
084                                @Override
085                                public V getValue() {
086                                        return value;
087                                }
088
089                                @Override
090                                public V setValue(V value) {
091                                        if (Objects.equals(this.value, value)) {
092                                                return value;
093                                        }
094                                        V prevValue = this.value;
095                                        this.value = value;
096                                        List<C> chunks = new ArrayList<>();
097                                        for (C chunk: split(getState())) {
098                                                Entry<K, V> e = load(chunk);
099                                                if (Objects.equals(e.getKey(), valueEntry.getKey())) {
100                                                        chunks.add(store(this));
101                                                } else {
102                                                        chunks.add(chunk);
103                                                }
104                                        }
105                                        setState(join(chunks));
106                                        return prevValue;
107                                }
108                                
109                        };
110                }
111
112                @Override
113                protected C store(Entry<K, V> element) {
114                        return AbstractSplitJoinMap.this.store(element);
115                }
116                
117        };
118
119        @Override
120        public Set<Entry<K, V>> entrySet() {
121                return entrySet;
122        }
123        
124        @Override
125        public V put(K key, V value) {
126                Set<Entry<K, V>> entrySet = entrySet();
127                for (Entry<K,V> e: entrySet) {
128                        if (Objects.equals(key, e.getKey())) {
129                                V prev = e.getValue();
130                                e.setValue(value);
131                                return prev;
132                        }
133                }
134                entrySet.add(new Entry<K,V>() {
135
136                        @Override
137                        public K getKey() {
138                                return key;
139                        }
140
141                        @Override
142                        public V getValue() {
143                                return value;
144                        }
145
146                        @Override
147                        public V setValue(V value) {
148                                throw new UnsupportedOperationException();
149                        }
150                        
151                });
152                return null;
153        }
154
155}
