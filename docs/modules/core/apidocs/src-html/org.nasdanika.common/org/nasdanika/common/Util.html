062        /**
063         * Converts name to label text by breaking the name by camel case, capitalizing the first word and lowsercasing the rest.
064         * @return
065         */
066        public static String nameToLabel(String name) {
067                String[] cca = StringUtils.splitByCharacterTypeCamelCase(name);
068                cca[0] = StringUtils.capitalize(cca[0]);
069                for (int i=1; i<cca.length; ++i) {
070                        cca[i] = cca[i].toLowerCase();
071                }
072                return String.join(" ", cca);
073        }

076         * Converts string in camel case, e.g. <code>firstName</code> to kebab case, e.g. <code>first-name</code>
077         * @return
078         */
079        public static String camelToKebab(String str) {
080                String[] ns = StringUtils.splitByCharacterTypeCamelCase(str);
081                for (int i = 0; i < ns.length; ++i) {
082                        ns[i] = ns[i].toLowerCase();
083                }
084                return String.join("-", ns);
085        }

091        /**
092         * Creates a function for context mapping which recognizes path navigation with ..
093         * @param prefix
094         * @return
095         */
096        public static java.util.function.Function<String,String> hierarchicalMapper(String prefix) {            
097                return key -> {
098                        StringTokenizer st = new StringTokenizer(prefix + key, "/", true);
099                        LinkedList<String> collector = new LinkedList<>();
100                        while (st.hasMoreTokens()) {
101                                String token = st.nextToken();
102                                if ("..".equals(token) && collector.size() > 1 && !"..".equals(collector.get(collector.size() - 2))) {
103                                        collector.removeLast();
104                                        collector.removeLast();
105                                        if (st.hasMoreTokens()) {
106                                                st.nextToken(); 
107                                        }
108                                } else {
109                                        collector.add(token);
110                                }
111                        }
112                        StringBuilder sb = new StringBuilder();
113                        collector.forEach(sb::append);
114                        return sb.toString();                   
115                };
116        }

118        public static InputStream join(Stream<InputStream> streams) throws IOException {
119                ByteArrayOutputStream baos = new ByteArrayOutputStream();
120                streams.forEach(in -> {
121                        if (in != null) {                               
122                                try (BufferedInputStream bin = new BufferedInputStream(in)) {
123                                        int b;
124                                        while ((b = bin.read()) != -1) {
125                                                baos.write(b);
126                                        }                                               
127                                } catch (IOException e) {
128                                        throw new NasdanikaException("Error joining streams: " + e, e);
129                                }
130                        }
131                });
132                baos.close();
133                return new ByteArrayInputStream(baos.toByteArray());
134        }

136        public static InputStream join(InputStream... streams) throws IOException {
137                if (streams.length == 1) {
138                        return streams[0];
139                }
140                ByteArrayOutputStream baos = new ByteArrayOutputStream();
141                for (InputStream in: streams) {
142                        if (in != null) {
143                                try (BufferedInputStream bin = new BufferedInputStream(in)) {
144                                        int b;
145                                        while ((b = bin.read()) != -1) {
146                                                baos.write(b);
147                                        }                                               
148                                }
149                        }
150                }
151                baos.close();
152                return new ByteArrayInputStream(baos.toByteArray());
153        }

155        public static InputStream toStream(Context context, String text) throws IOException {
156                Charset charset = context.get(Charset.class, StandardCharsets.UTF_8);
157                ByteArrayOutputStream baos = new ByteArrayOutputStream();
158                if (text != null) {
159                        try (OutputStreamWriter writer = new OutputStreamWriter(baos, charset)) {
160                                writer.write(text);
161                        }
162                }
163                baos.close();
164                return new ByteArrayInputStream(baos.toByteArray());
165        }

217        public static String toString(Context context, InputStream in) throws IOException {
218                if (in == null) {
219                        return "";
220                }
221                StringWriter sw = new StringWriter();
222                Charset charset = context.get(Charset.class, StandardCharsets.UTF_8);
223                try (Reader reader = new InputStreamReader(new BufferedInputStream(in), charset)) {
224                        int ch;
225                        while ((ch = reader.read()) != -1) {
226                                sw.write(ch);
227                        }
228                }
229                sw.close();
230                return sw.toString();
231        }

454        /**
455         * Extracts first sentence from text
456         * @param context
457         * @param text
458         * @return
459         */
460        public static String firstSentence(String text, int minSentenceLength, int maxSentenceLength, String... abbreviations) {
461                if (text == null || text.length() < minSentenceLength) {
462                        return text;
463                }
464                Matcher matcher = SENTENCE_PATTERN.matcher(text);               
465                Z: while (matcher.find()) {
466                        String group = matcher.group();
467                        for (String abbr: abbreviations) {
468                                if (group.trim().endsWith(abbr)) {
469                                        continue Z;
470                                }
471                        }
472                        if (matcher.end() > minSentenceLength && matcher.end() < maxSentenceLength) {
473                                return text.substring(0, matcher.end());
474                        }
475                }
476                
477                return text.length() < maxSentenceLength ? text : text.substring(0, maxSentenceLength)+"...";
478        }

493        /**
494         * Executes full supplier lifecycle - diagnose, execute, commit/rollback, close.
495         * @param monitor
496         * @param diagnosticConsumer TODO
497         * @param context
498         * @param component
499         * @return
500         */
501        public static <T> T call(Supplier<T> supplier, ProgressMonitor monitor, Consumer<Diagnostic> diagnosticConsumer, Status... failOnStatuses) {
502                try (ProgressMonitor progressMonitor = monitor.setWorkRemaining(3).split("Calling supplier", 3)) {
503                        Diagnostic diagnostic = supplier.splitAndDiagnose(progressMonitor);
504                        if (diagnosticConsumer != null) {
505                                diagnosticConsumer.accept(diagnostic);
506                        }
507                        Status status = diagnostic.getStatus();
508                        if (failOnStatuses.length == 0) {
509                                if (status == Status.FAIL) {
510                                        throw new DiagnosticException(diagnostic);
511                                }
512                        } else {
513                                for (Status failStatus: failOnStatuses) {
514                                        if (status == failStatus) {
515                                                throw new DiagnosticException(diagnostic);
516                                        }                                       
517                                }
518                        }
519                        
520                        try {
521                                T result = supplier.splitAndExecute(progressMonitor);
522                                supplier.splitAndCommit(progressMonitor);
523                                return result;
524                        } catch (Exception e) {
525                                supplier.splitAndRollback(progressMonitor);
526                                throw e;
527                        }
528                } finally {
529                        supplier.close();
530                }
531        }

617         * Wraps object into a supplier factory. Handles adapters, and collections.
618         * @param obj
619         * @return
620         */
621        @SuppressWarnings("unchecked")
622        public static <T> SupplierFactory<T> asSupplierFactory(Object obj, Class<T> type, FunctionFactory<List<T>, T> joinFactory) {
623                if (obj instanceof Collection && joinFactory != null) {
624                        ListCompoundSupplierFactory<T> ret = new ListCompoundSupplierFactory<>("Supplier collection");
625                        for (Object e: (Collection<?>) obj) {
626                                SupplierFactory<T> esp = type == InputStream.class ? (SupplierFactory<T>) asInputStreamSupplierFactory(e) : asSupplierFactory(e, type, joinFactory);
627                                if (esp != null) {
628                                        ret.add(esp);
629                                }
630                        }
631                        return ret.then(joinFactory);
632                }
633                
634                if (obj instanceof SupplierFactory) {           
635                        return (SupplierFactory<T>) obj;
636                }
637                
638                if (obj instanceof SupplierFactory.Provider && type != null) {
639                        SupplierFactory<T> factory = ((SupplierFactory.Provider) obj).getFactory(type);
640                        if (factory != null) {
641                                return factory;
642                        }
643                }
644                
645                if (obj instanceof Adaptable) {
646                        Adaptable adaptable = (Adaptable) obj;
647                        if (type != null) {
648                                SupplierFactory.Provider provider = adaptable.adaptTo(SupplierFactory.Provider.class);
649                                if (provider != null) {
650                                        SupplierFactory<T> factory = provider.getFactory(type);
651                                        if (factory != null) {
652                                                return factory;
653                                        }
654                                }
655                        }
656                        
657                        SupplierFactory<T> adapter = adaptable.adaptTo(SupplierFactory.class);
658                        if (adapter != null) {
659                                return adapter;
660                        }
661                }
662                
663                return null;
664        }

758         * Grouping by with support of null keys.
759         * @param <K>
760         * @param <T>
761         * @param elements
762         * @param keyFeature
763         * @return
764         */
765        public static <K, T> Map<K, List<T>> groupBy(Collection<T> elements, java.util.function.Function<? super T, ? extends K> classifier) {
766                Map<K, List<T>> ret = new LinkedHashMap<>();
767                for (T e: elements) {
768                        K k = classifier.apply(e);
769                        ret.computeIfAbsent(k, key -> new ArrayList<>()).add(e);
770                }
771                return ret;
772        }

937        /**
938         * Wraps texts so its line width grows proportionally as the number of lines increases. 
939         * @param text
940         * @param initialLineWidth
941         * @param separator
942         * @return
943         */
944        public static String wrap(String text, int initialLineWidth, int widthIncrementPerLine, String separator) {
945                if (Util.isBlank(text) || text.length() < initialLineWidth) { 
946                        return text;
947                }
948                double lines = quadraticRoot(widthIncrementPerLine, initialLineWidth - widthIncrementPerLine, -text.length()); 
949                return WordUtils.wrap(text, initialLineWidth + (int) (widthIncrementPerLine * Math.ceil(lines)), separator, false);
950        }

952        /**
953         * Inserts separators into a list of elements so when these elements are rendered as text the text is wrapped in such a way that
954         * line width grows proportionally as the number of lines increases. 
955         * @param elements Elements to wrap 
956         * @param initialLineWidth
957         * @param separatorSupplier creates separators to insert into the resulting list.
958         * @return
959         */
960        public static <T> List<T> wrap(
961                        List<T> elements, 
962                        ToIntFunction<T> elementLengthComputer,  
963                        int initialLineWidth, 
964                        int widthIncrementPerLine, 
965                        java.util.function.Supplier<T> wordSeparatorSupplier,                   
966                        java.util.function.Supplier<T> lineSeparatorSupplier) {
967                int length = elements.stream().mapToInt(elementLengthComputer).sum();           
968                int wrapLength = initialLineWidth;
969                if (length > initialLineWidth) { 
970                        wrapLength += (int) (widthIncrementPerLine * Math.ceil(quadraticRoot(widthIncrementPerLine, initialLineWidth - widthIncrementPerLine, -length)));
971                }
972                List<T> ret = new ArrayList<T>();
973                int lineLength = 0;
974                for (T element: elements) {
975                        if (lineLength > wrapLength) {
976                                ret.add(lineSeparatorSupplier.get());
977                                lineLength = 0;
978                        } 
979                        if (lineLength > 0) {
980                                ret.add(wordSeparatorSupplier.get());
981                        }
982                        ret.add(element);
983                        lineLength += elementLengthComputer.applyAsInt(element);
984                }
985                return ret;
986        }

988        private static double quadraticRoot(double a, double b, double c) { 
989                return (-b + Math.sqrt(b*b - 4 * a * c)) / (2 * a);
990        }

992        private static final String PACKAGE_SUMMARY_HTML = "package-summary.html";

994        /**
995         * Java 8 uses package-list, Java 11 uses element-list.
996         */
997        private static final String[] PACKAGE_LIST_LOCATIONS = { "package-list", "element-list" };      
998        
999        private static final String MODULE_PREFIX = "module:";
1000        
1001        /**
1002         * @param urls Mapping of online URL's (to build links) to offline URL's (to load package lists from).  
1003         * @return Function which given a fully qualified class name returns an anchor linking to Javadoc page for the class. The argument may contain a hash symbol (fragment). 
1004         * Everything after the hash gets transfered to the computed URL providing ability to address fields and methods.
1005         * 
1006         */
1007        public static java.util.function.Function<String, String> createJavadocResolver(Map<URL,URL> urls, ProgressMonitor progressMonitor) {
1008                // package -> Base URL.
1009                Map<String,String> unsortedPackageMap = new HashMap<>();                
1010                progressMonitor.setWorkRemaining(urls.size());
1011                Z: for (Map.Entry<URL, URL> urlEntry: urls.entrySet()) {
1012                        for (String packageListLocation: PACKAGE_LIST_LOCATIONS) {
1013                                try (BufferedReader br = new BufferedReader(new InputStreamReader(new URL(urlEntry.getValue(), packageListLocation).openStream()))) {
1014                                        String module = null;
1015                                        String line;
1016                                        while ((line = br.readLine()) != null) {
1017                                                if (line.startsWith(MODULE_PREFIX)) {
1018                                                        module = line.substring(MODULE_PREFIX.length()); 
1019                                                } else {
1020                                                        String packageLocation = urlEntry.getKey().toString() + (module == null ? "index.html?" : module + "/");
1021                                                        unsortedPackageMap.put(line.trim(), packageLocation);
1022                                                }
1023                                        }
1024                                        progressMonitor.worked(Status.SUCCESS, 1, "Loaded package list from " + urlEntry.getValue() + " for " + urlEntry.getKey());
1025                                        continue Z;
1026                                } catch (Exception e) {
1027                                        progressMonitor.worked(Status.ERROR, 1, "Could not download package list from " + urlEntry.getValue() + " - " + e);
1028                                }
1029                        }
1030                }
1031                
1032                Map<String,String> sortedPackageMap = new LinkedHashMap<>();
1033                unsortedPackageMap.keySet().stream().sorted((k1,k2) -> k2.length() - k1.length()).forEach(k -> sortedPackageMap.put(k, unsortedPackageMap.get(k)));
1034                
1035                return new java.util.function.Function<String, String>() {
1036                        
1037                        private String getRef(String spec) {
1038                                int hashIdx = spec.indexOf("#");
1039                                if (hashIdx != -1) {
1040                                        spec = spec.substring(0, hashIdx);
1041                                }
1042                                for (Entry<String, String> location: sortedPackageMap.entrySet()) {
1043                                        String key = location.getKey();
1044                                        String value = location.getValue();
1045                                        if (key.equals(spec)) { // Package
1046                                                return value + spec.replace('.', '/') + "/" + PACKAGE_SUMMARY_HTML;
1047                                        }
1048                                        
1049                                        int idx = spec.lastIndexOf('.');
1050                                        if (idx != -1 && spec.substring(0, idx).equals(key)) { // Class
1051                                                return value + spec.replace('.', '/').replace('$', '.') + ".html";                              
1052                                        }
1053                                }
1054                                return null;                            
1055                        }
1056
1057                        @Override
1058                        public String apply(String key) {
1059                                if (key == null) {
1060                                        return key;
1061                                }
1062                                
1063                                int spaceIdx = key.indexOf(' ');
1064                                String spec = spaceIdx == -1 ? key : key.substring(0, spaceIdx);
1065                                String text = spaceIdx == -1 ? null : key.substring(spaceIdx + 1);
1066                                
1067                                String href = getRef(spec);
1068                                if (href == null) {
1069                                        return spec;
1070                                }
1071                                                                
1072                                int hashIdx = spec.indexOf("#");
1073                                if (hashIdx != -1) {
1074                                        String fragment = spec.substring(hashIdx+1);
1075                                        int firstParenthesisIdx = fragment.indexOf("(");
1076                                        if (firstParenthesisIdx > 0 && fragment.endsWith(")")) {
1077                                                // Convert (type[,type]) to -type-type-                                         
1078                                                StringBuilder fragmentBuilder = new StringBuilder(fragment.substring(0, firstParenthesisIdx)).append("-");
1079                                                String[] pTypes = fragment.substring(firstParenthesisIdx+1, fragment.length()-1).split(",");
1080                                                for (String pType: pTypes) {
1081                                                        pType = pType.trim();
1082                                                        int bIdx = pType.indexOf("[]");
1083                                                        if (bIdx == -1) {
1084                                                                fragmentBuilder.append(pType);
1085                                                        } else {
1086                                                                fragmentBuilder.append(pType.substring(0, bIdx));
1087                                                                for (int bc = bIdx; bc < pType.length(); bc+=2) {
1088                                                                        fragmentBuilder.append(":A");
1089                                                                }
1090                                                        }
1091                                                        fragmentBuilder.append("-");
1092                                                }
1093                                                fragment = fragmentBuilder.toString();
1094                                        } 
1095                                        href += "#" + fragment;
1096                                }
1097                                return new StringBuilder("<a href='")
1098                                                .append(href)
1099                                                .append("'>")
1100                                                .append(Util.isBlank(text) ? spec.replace('#', '.').replace('$', '.') : text)
1101                                                .append("</a>").toString();
1102                        }
1103                        
1104                };
1105                
1106        }
