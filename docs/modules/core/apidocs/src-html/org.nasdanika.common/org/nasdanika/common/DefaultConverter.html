048        /**
049         * Constructor conversion is not working well, using just type and toString value.
050         * @param obj
051         * @return
052         */
053        @ConverterMethod
054        public JSONObject toJSONObject(Object obj) {
055                JSONObject ret = new JSONObject();
056                ret.put("type", obj.getClass().getName());
057                ret.put("value", obj.toString());
058                return ret;
059        }

061        // Reader to String, and JSONObject and JSONArray.
062        
063        @ConverterMethod
064        public String toString(Reader reader) throws IOException {
065                StringWriter sw = new StringWriter();
066                try (BufferedReader br = new BufferedReader(reader)) {
067                        int ch;
068                        while ((ch = br.read()) != -1) {
069                                sw.write(ch);
070                        }
071                }
072                sw.close();
073                return sw.toString();
074        }

088        /**
089         * Reads input stream into a byte array. 
090         * @param in
091         * @return
092         * @throws IOException 
093         */
094        @ConverterMethod
095        public byte[] toByteArray(InputStream in) throws IOException {
096                ByteArrayOutputStream baos = new ByteArrayOutputStream();
097                try (InputStream bin = new BufferedInputStream(in)) {
098                        int b;
099                        while ((b = bin.read()) != -1) {
100                                baos.write(b);
101                        }
102                }
103                baos.close();
104                return baos.toByteArray();                              
105        }

197        /**
198         * Converts character sequence to byte array with UTF_8 encoding.
199         * @param charSequence
200         * @return
201         * @throws IOException
202         */
203        @ConverterMethod
204        public byte[] toByteArray(CharSequence charSequence) throws IOException {
205                ByteArrayOutputStream baos = new ByteArrayOutputStream();
206                try (Writer writer = new OutputStreamWriter(baos, StandardCharsets.UTF_8)) {
207                        writer.write(charSequence.toString());
208                }
209                baos.close();
210                return baos.toByteArray();                              
211        }

231        /**
232         * @param text
233         * @return parsed duration.
234         */
235        @ConverterMethod
236        public Instant toInstant(String text) {
237                if (text == null) {
238                        return null;
239                }
240                if (text.contains("Z")) {
241                        return Instant.parse(text);
242                }
243                // Parsing via Date for convenience
244                Date date = convert(text, Date.class);
245                return date == null ? null : date.toInstant();
246        }
247

250         * @return parsed duration.
251         */
252        @ConverterMethod
253        public Instant toInstant(Date date) {
254                return Instant.ofEpochMilli(date.getTime());
255        }

263        // URI
264        
265        @ConverterMethod
266        public InputStream toInputStream(URI uri) throws IOException {
267                if (Util.CLASSPATH_SCHEME.equals(uri.scheme())) {
268                        String resource = uri.toString().substring(Util.CLASSPATH_URL_PREFIX.length());
269                        return Objects.requireNonNull(getClass().getClassLoader().getResourceAsStream(resource), "ClassLoader resource not found: " + resource);
270                }
271                return toInputStream(new URL(uri.toString()));
272        }
