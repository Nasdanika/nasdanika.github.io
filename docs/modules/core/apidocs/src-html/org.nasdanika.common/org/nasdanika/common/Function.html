001package org.nasdanika.common;
002
003import java.util.ArrayList;
004import java.util.List;
005import java.util.function.BiFunction;
006
007/**
008 * 
009 * @author Pavel Vlasov
010 * @param C context type.
011 * @param T result type.
012 */
013public interface Function<T,R> extends ExecutionParticipant, ExecutionParticipantInfo, BiFunction<T,ProgressMonitor,R> {
014        
015        R execute(T arg, ProgressMonitor progressMonitor);      
016        
017        default R splitAndExecute(T arg, ProgressMonitor progressMonitor) {
018                try (ProgressMonitor subMonitor = split(progressMonitor, "Executing "+name())) {
019                        return execute(arg, subMonitor);
020                }
021        }
022        
023        default R splitAndExecute(T arg, double size, ProgressMonitor progressMonitor) {
024                try (ProgressMonitor subMonitor = split(size, progressMonitor, "Executing "+name())) {
025                        return execute(arg, subMonitor);
026                }
027        }
028        
029        Function<Object,Object> NOP = new Function<Object,Object>() {
030
031                @Override
032                public double size() {
033                        return 0;
034                }
035                
036                @Override
037                public String name() {
038                        return "No operation";
039                }
040
041                @Override
042                public Object execute(Object arg, ProgressMonitor monitor) {
043                        return arg;
044                }
045                
046        };
047        
048        @SuppressWarnings("unchecked")
049        static <V> Function<V,V> nop() {
050                return (Function<V,V>) NOP;
051        }
052        
053        default <V> Function<V,R> before(java.util.function.Function<V,T> before) {
054                return new Function<V,R>() {
055                        
056                        @Override
057                        public R execute(V arg, ProgressMonitor progressMonitor) {
058                                return Function.this.execute(before.apply(arg), progressMonitor);
059                        }
060                        
061                        @Override
062                        public Diagnostic diagnose(ProgressMonitor progressMonitor) {
063                                return Function.this.diagnose(progressMonitor);
064                        }
065                        
066                        @Override
067                        public void close() {
068                                Function.this.close();
069                        }
070                        
071                        @Override
072                        public void commit(ProgressMonitor progressMonitor) {
073                                Function.this.commit(progressMonitor);
074                        }
075                        
076                        @Override
077                        public boolean rollback(ProgressMonitor progressMonitor) {
078                                return Function.this.rollback(progressMonitor);
079                        }
080                        
081                        @Override
082                        public double size() {
083                                return Function.this.size();
084                        }
085                        
086                        @Override
087                        public String name() {
088                                return Function.this.name();
089                        }
090                        
091                };
092        }
093        
094        default <V> Function<T,V> then(java.util.function.Function<? super R, V> then) {
095                return new Function<T,V>() {
096                        
097                        @Override
098                        public V execute(T arg, ProgressMonitor progressMonitor) {
099                                return then.apply(Function.this.execute(arg, progressMonitor));
100                        }
101                        
102                        @Override
103                        public Diagnostic diagnose(ProgressMonitor progressMonitor) {
104                                return Function.this.diagnose(progressMonitor);
105                        }
106                        
107                        @Override
108                        public void close() {
109                                Function.this.close();
110                        }
111                        
112                        @Override
113                        public void commit(ProgressMonitor progressMonitor) {
114                                Function.this.commit(progressMonitor);
115                        }
116                        
117                        @Override
118                        public boolean rollback(ProgressMonitor progressMonitor) {
119                                return Function.this.rollback(progressMonitor);
120                        }
121                        
122                        @Override
123                        public double size() {
124                                return Function.this.size();
125                        }
126                        
127                        @Override
128                        public String name() {
129                                return Function.this.name();
130                        }
131                        
132                };
133        }
134        
135        default <V> Function<T,V> then(Function<? super R, V> then) {
136                List<ExecutionParticipant> elements = new ArrayList<>();
137                elements.add(this);
138                elements.add(then);
139                class Then extends CompoundExecutionParticipant<ExecutionParticipant> implements Function<T,V> {
140
141                        protected Then(String name) {
142                                super(name, true);
143                        }
144                        
145                        @Override
146                        public V execute(T arg, ProgressMonitor progressMonitor) {
147                                return then.splitAndExecute(Function.this.splitAndExecute(arg, progressMonitor), progressMonitor);
148                        }
149
150                        @Override
151                        protected List<ExecutionParticipant> getElements() {
152                                return elements;
153                        }                       
154                }
155                
156                StringBuilder nameBuilder = new StringBuilder();
157                nameBuilder.append(Function.this.name()).append(" => ").append(then.name());
158                return new Then(nameBuilder.toString());
159        }
160                
161        default Consumer<T> then(Consumer<? super R> then) {
162                List<ExecutionParticipant> elements = new ArrayList<>();
163                elements.add(this);
164                elements.add(then);
165                class Then extends CompoundExecutionParticipant<ExecutionParticipant> implements Consumer<T> {
166
167                        protected Then(String name) {
168                                super(name, true);
169                        }
170                        
171                        @Override
172                        public void execute(T arg, ProgressMonitor progressMonitor) {
173                                then.splitAndExecute(Function.this.splitAndExecute(arg, progressMonitor), progressMonitor);
174                        }
175
176                        @Override
177                        protected List<ExecutionParticipant> getElements() {
178                                return elements;
179                        }                       
180                }
181                
182                StringBuilder nameBuilder = new StringBuilder();
183                nameBuilder.append(Function.this.name()).append(" => ").append(then.name());
184                return new Then(nameBuilder.toString());
185        }       
186        
187        static <T,R> Function<T,R> fromBiFunction(BiFunction<T, ProgressMonitor, R> biFunction, String name, double size) {
188                return new Function<T,R>() {
189
190                        @Override
191                        public double size() {
192                                return size;
193                        }
194
195                        @Override
196                        public String name() {
197                                return name;
198                        }
199
200                        @Override
201                        public R execute(T arg, ProgressMonitor progressMonitor) {
202                                return biFunction.apply(arg, progressMonitor);
203                        }
204                        
205                };
206        }
207        
208        static <T,R> Function<T,R> fromFunction(java.util.function.Function<T, R> function, String name, double size) {
209                return fromBiFunction((arg, monitor) -> function.apply(arg), name, size);
210        }
211        
212        default R apply(T arg, ProgressMonitor progressMonitor) {
213                try {
214                        progressMonitor.setWorkRemaining(size()*3); // diagnose, execute, commit or rollback
215                        diagnose(progressMonitor.split("Diagnosing", size())).checkError("Diagnostic failed: "+name());
216                        try {
217                                R result = execute(arg, progressMonitor.split("Executing", size()));
218                                commit(progressMonitor.split("Committing", size()));
219                                return result;
220                        } catch (Exception e) {
221                                try {
222                                        rollback(progressMonitor.split("Rolling back: "+e, size(), e));
223                                } catch (Exception re) {
224                                        e.addSuppressed(re);
225                                }
226                                throw e;
227                        }
228                } finally {
229                        close();
230                }               
231        }
232        
233        /**
234         * Calls asBiFunction() with a {@link NullProgressMonitor}.
235         * @return
236         */
237        default java.util.function.Function<T,R> toFunction(ProgressMonitor progressMonitor) {
238                return arg -> apply(arg, progressMonitor);
239        }
