001package org.nasdanika.common;
002
003/**
004 * Converts object to target type.
005 * @author Pavel
006 *
007 * @param <S> Source type
008 * @param <T> Target type
009 * @param <C> Context type
010 */
011public interface Converter extends Composeable<Converter> {
012        
013        <T> T convert(Object source, Class<T> type);
014        
015        default Converter compose(Converter other) {
016                if (other == null) {
017                        return this;
018                }
019                
020                return new Converter() {
021                        
022                        @Override
023                        public <T> T convert(Object source, Class<T> type) {
024                                T ret = Converter.this.convert(source, type);
025                                return ret == null ? other.convert(source, type) : ret;
026                        }
027                };
028        }
029        
030}
