001package org.nasdanika.common;
002
003import java.util.Arrays;
004import java.util.Collections;
005import java.util.List;
006
007public abstract class DelimitedStringMap extends AbstractSplitJoinMap<String, String, String, String> {
008        
009        private String entryDelimiter;
010        private String keyValueDelimiter;
011        
012        public DelimitedStringMap(String entryDelimiter, String keyValueDelimiter) {
013                this.entryDelimiter = entryDelimiter;
014                this.keyValueDelimiter = keyValueDelimiter;
015        }       
016
017        @Override
018        protected List<String> split(String state) {
019                return Util.isBlank(state) ? Collections.emptyList() : Arrays.asList(state.split(entryDelimiter));
020        }
021
022        @Override
023        protected String join(List<String> chunks) {
024                return String.join(entryDelimiter, chunks.toArray(new String[chunks.size()]));
025        }
026
027        @Override
028        protected Entry<String, String> load(String chunk) {
029                String[] ea = chunk.split(keyValueDelimiter);
030                return new Entry<String, String>() {
031
032                        @Override
033                        public String getKey() {
034                                return ea[0];
035                        }
036
037                        @Override
038                        public String getValue() {
039                                return ea.length > 1 ? ea[1] : null;
040                        }
041
042                        @Override
043                        public String setValue(String value) {
044                                throw new UnsupportedOperationException();
045                        }
046                };
047        }
048
049        @Override
050        protected String store(Entry<String, String> element) {
051                return element.getKey() + keyValueDelimiter + element.getValue();
052        }
053
054}
