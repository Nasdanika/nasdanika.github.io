114        /**
115         * @param markdown Markdown to pre-process
116         * @param replacements Accumulator of post-processing replacements token -&gt; HTML to replace the token with.
117         * @return
118         */
119        protected String preProcessMarkdown(String markdown, Map<String,String> replacements) {
120                if (isProcessFendedBlocks()) {
121                        String ret = processFencedBlocks(markdown, DiagramGenerator.UML_DIALECT, START_UML_PATTERN, replacements, false);
122                        ret = processFencedBlocks(ret, DiagramGenerator.SALT_DIALECT, START_WIREFRAME_PATTERN, replacements, false);
123                        ret = processFencedBlocks(ret, DiagramGenerator.GANTT_DIALECT, START_GANTT_PATTERN, replacements, false);
124                        ret = processFencedBlocks(ret, DiagramGenerator.MINDMAP_DIALECT, START_MINDMAP_PATTERN, replacements, false);
125                        ret = processFencedBlocks(ret, DiagramGenerator.WBS_DIALECT, START_WBS_PATTERN, replacements, false);
126                        ret = processFencedBlocks(ret, DiagramGenerator.DRAWIO_DIALECT, START_DRAWIO_PATTERN, replacements, false);
127                        ret = processFencedBlocks(ret, DiagramGenerator.MERMAID_DIALECT, START_MERMAID_PATTERN, replacements, false);
128                        ret = processFencedBlocks(ret, null, START_PNG_PATTERN, replacements, false);
129                        ret = processFencedBlocks(ret, null, START_JPEG_PATTERN, replacements, false);
130                                        
131                        ret = processFencedBlocks(ret, DiagramGenerator.UML_DIALECT, START_UML_RESOURCE_PATTERN, replacements, true);
132                        ret = processFencedBlocks(ret, DiagramGenerator.SALT_DIALECT, START_WIREFRAME_RESOURCE_PATTERN, replacements, true);
133                        ret = processFencedBlocks(ret, DiagramGenerator.GANTT_DIALECT, START_GANTT_RESOURCE_PATTERN, replacements, true);
134                        ret = processFencedBlocks(ret, DiagramGenerator.MINDMAP_DIALECT, START_MINDMAP_RESOURCE_PATTERN, replacements, true);
135                        ret = processFencedBlocks(ret, DiagramGenerator.WBS_DIALECT, START_WBS_RESOURCE_PATTERN, replacements, true);
136                        ret = processFencedBlocks(ret, DiagramGenerator.DRAWIO_DIALECT, START_DRAWIO_RESOURCE_PATTERN, replacements, true);
137                        ret = processFencedBlocks(ret, DiagramGenerator.MERMAID_DIALECT, START_MERMAID_RESOURCE_PATTERN, replacements, true);
138                        ret = processFencedBlocks(ret, null, START_PNG_RESOURCE_PATTERN, replacements, true);
139                        ret = processFencedBlocks(ret, null, START_JPEG_RESOURCE_PATTERN, replacements, true);
140                                        
141                        return ret;             
142                }
143                                
144                return markdown;
145        }

147        /**
148         * 
149         * @param input
150         * @param dialect
151         * @param startPattern
152         * @param replacements
153         * @param resource Indicates that spec is a URL of a resource resolved relative to the URL returned by getResourceBase(). The URL supports <code>classpath</code> schema for classloader resources.
154         * @return
155         */
156        private String processFencedBlocks(String input, String dialect, Pattern startPattern, Map<String,String> replacements, boolean resource) {
157                StringBuilder output = new StringBuilder();
158                Matcher startMatcher = startPattern.matcher(input);
159                int i = 0;
160                while (startMatcher.find()) {
161                    int startMatcherStart = startMatcher.start();
162                    if (startMatcherStart > i) { // Need to reproduce a situation when i is greater than startMatcherStart
163                                int startMatcherEnd = startMatcher.end();
164                                String endMatcherContent = input.substring(startMatcherEnd);
165                                Matcher endMatcher = FENCED_BLOCK_PATTERN.matcher(endMatcherContent);                   
166                                if (endMatcher.find()) {
167                                        output.append(input.substring(i, startMatcherStart));
168                                        String bareSpec = processSpec(dialect, endMatcherContent.substring(0, endMatcher.start())) ;
169                                        String match = endMatcherContent.substring(endMatcher.start(), endMatcher.end());
170                                    i = startMatcherEnd + endMatcher.start() + match.indexOf("```") + 3; // Just the closing back-ticks, no space or new line characters.
171                                        
172                                        if (resource) {                                 
173                                                try {           
174                                                        URI resourceURI = URI.createURI(bareSpec);
175                                                        URI resourceBase = getResourceBase();
176                                                        if (resourceBase != null && resourceBase.hasAbsolutePath()) {
177                                                                resourceURI = resourceURI.resolve(resourceBase);
178                                                        }
179                                                        bareSpec = loadResource(resourceURI, dialect == null);
180                                                } catch (Exception e) {                                 
181                                                        output
182                                                                .append(System.lineSeparator())
183                                                                .append("<div class=\"nsd-error\">").append(System.lineSeparator())
184                                                                .append("Error loading resource " + bareSpec + ": " + e) 
185                                                                .append(System.lineSeparator()).append("</div>")
186                                                                .append(System.lineSeparator());
187                                                        bareSpec = null;
188                                                }
189                                        }
190                                        
191                                        if (bareSpec != null) {
192                                                String escapedBareSpec = escapeDiagramSpec(dialect, bareSpec);
193                                                
194                                                StringBuilder replacementBuilder = new StringBuilder();
195                                                if (!Util.isBlank(escapedBareSpec)) {
196                                                        replacementBuilder
197                                                                .append(System.lineSeparator())
198                                                                .append("<div style='display:none;white-space:pre-wrap' title='Diagram escaped spec for search'>").append(System.lineSeparator())
199                                                                .append(escapedBareSpec).append(System.lineSeparator()) 
200                                                                .append("</div> ").append(System.lineSeparator());
201                                                }
202                                                
203                                                try {
204                                                        String token = nextToken();
205                                                        
206                                                        if (dialect == null) {
207                                                                replacementBuilder.append("<img src=\"data:image/");
208                                                                if (startPattern == START_PNG_PATTERN || startPattern == START_PNG_RESOURCE_PATTERN) {
209                                                                        replacementBuilder.append("png");
210                                                                } else if (startPattern == START_JPEG_PATTERN || startPattern == START_JPEG_RESOURCE_PATTERN) {
211                                                                        replacementBuilder.append("jpeg");
212                                                                }
213                                                                replacementBuilder
214                                                                        .append(";base64, ")
215                                                                        .append(bareSpec)
216                                                                        .append("\"/>")
217                                                                        .append(System.lineSeparator());
218                                                        } else {
219                                                                replacementBuilder
220                                                                        .append(getDiagramGenerator().generateDiagram(bareSpec, dialect))
221                                                                        .append(System.lineSeparator());
222                                                        }
223                                                        
224                                                        replacements.put(token, replacementBuilder.toString());
225                                                        output
226                                                                .append(System.lineSeparator())
227                                                                .append(System.lineSeparator())
228                                                                .append(token)
229                                                                .append(System.lineSeparator())
230                                                                .append(System.lineSeparator());
231                                                } catch (Exception e) {
232                                                        output.append("<div class=\"nsd-error\">Error during diagram rendering: " + e + "</div>");
233                                                }
234                                        }                               
235                                }
236                    }
237                }
238                output.append(input.substring(i, input.length()));
239                return output.toString();
240        }

252        protected String escapeDiagramSpec(String dialect, String bareSpec) {
253                if (dialect == null) {
254                        return null; // For PNG and JPEG
255                }
256                if (DiagramGenerator.DRAWIO_DIALECT.equals(dialect)) {
257                        // TODO - Extract text from XML, return null for now.
258                        // Not needed with Nasdanika search - it extracts text from Drawio diagrams.
259                        return null;
260                }
261                if (DiagramGenerator.MERMAID_DIALECT.equals(dialect)) {
262                        // No need - spec is available as plain text. 
263                        return null;
264                }
265                return StringEscapeUtils.escapeHtml4(bareSpec).trim();
266        }

272        /**
273         * Loads resource and optionally encodes it to Base64.
274         * @param resource
275         * @param encode
276         * @return
277         */
278        protected String loadResource(URI resource, boolean encode) throws Exception {
279                if (encode) {
280                        byte[] bytes = DefaultConverter.INSTANCE.toByteArray(resource);
281                        return Base64.getEncoder().encodeToString(bytes);
282                }
283                return DefaultConverter.INSTANCE.stringContent(resource);
284        }

316        public String markdownToHtml(String markdown) {
317                if (Util.isBlank(markdown)) {
318                        return "";
319                }
320                Map<String,String> replacements = new HashMap<>();
321                Document document = createMarkdownParser().parse(preProcessMarkdown(markdown, replacements));
322                String html = createMarkdownHtmlRenderer().render(document);
323                for (Entry<String, String> te: replacements.entrySet()) {
324                        html = html.replace(te.getKey(), te.getValue());
325                }
326                return html;
327        }
