001package org.nasdanika.common;
002
003import java.util.function.BiConsumer;
004
005/**
006 * @author Pavel Vlasov
007 */
008public interface Consumer<T> extends ExecutionParticipant, ExecutionParticipantInfo, BiConsumer<T, ProgressMonitor> {
009                
010        /**
011         * Executes Consumer.
012         * @param monitor Monitor to use.
013         * @return
014         */
015        void execute(T arg, ProgressMonitor progressMonitor);   
016        
017        default void splitAndExecute(T arg, ProgressMonitor progressMonitor) {
018                try (ProgressMonitor subMonitor = split(progressMonitor, "Executing "+name())) {
019                        execute(arg, subMonitor);
020                }
021        }       
022        
023        default void splitAndExecute(T arg, double size, ProgressMonitor progressMonitor) {
024                try (ProgressMonitor subMonitor = split(size, progressMonitor, "Executing "+name())) {
025                        execute(arg, subMonitor);
026                }
027        }       
028        
029        Consumer<Object> NOP = new Consumer<Object>() {
030
031                @Override
032                public double size() {
033                        return 0;
034                }
035                
036                @Override
037                public String name() {
038                        return "No operation consumer";
039                }
040
041                @Override
042                public void execute(Object arg, ProgressMonitor monitor) {
043                        // NOP
044                }
045                
046        };
047        
048        @SuppressWarnings("unchecked")
049        static <T> Consumer<T> nop() {
050                return (Consumer<T>) NOP;
051        }
052        
053        @Override
054        default void accept(T arg, ProgressMonitor progressMonitor) {
055                try {
056                        progressMonitor.setWorkRemaining(size()*3); // diagnose, execute, commit or rollback
057                        diagnose(progressMonitor.split("Diagnosing", size())).checkError("Diagnostic failed: "+name());
058                        try {
059                                execute(arg, progressMonitor.split("Executing", size()));
060                                commit(progressMonitor.split("Committing", size()));
061                        } catch (Exception e) {
062                                try {
063                                        rollback(progressMonitor.split("Rolling back: "+e, size(), e));
064                                } catch (Exception re) {
065                                        e.addSuppressed(re);
066                                }
067                                throw e;
068                        }
069                } finally {
070                        close();
071                }               
072        }
073        
074        default <V> Consumer<V> before(java.util.function.Function<V,T> before) {
075                return new Consumer<V>() {
076                        
077                        @Override
078                        public void execute(V arg, ProgressMonitor progressMonitor) {
079                                Consumer.this.execute(before.apply(arg), progressMonitor);
080                        }
081                        
082                        @Override
083                        public Diagnostic diagnose(ProgressMonitor progressMonitor) {
084                                return Consumer.this.diagnose(progressMonitor);
085                        }
086                        
087                        @Override
088                        public void close() {
089                                Consumer.this.close();
090                        }
091                        
092                        @Override
093                        public void commit(ProgressMonitor progressMonitor) {
094                                Consumer.this.commit(progressMonitor);
095                        }
096                        
097                        @Override
098                        public boolean rollback(ProgressMonitor progressMonitor) {
099                                return Consumer.this.rollback(progressMonitor);
100                        }
101                        
102                        @Override
103                        public double size() {
104                                return Consumer.this.size();
105                        }
106                        
107                        @Override
108                        public String name() {
109                                return Consumer.this.name();
110                        }
111                        
112                };
113        }
114                
115        /**
116         * @return Function which executes this consumer and returns the argument. 
117         * This method can be used for chaining consumers.
118         */
119        default Function<T,T> asFunction() {
120                return new Function<T,T>() {
121                        
122                        @Override
123                        public T execute(T arg, ProgressMonitor progressMonitor) {
124                                Consumer.this.execute(arg, progressMonitor);
125                                return arg;
126                        }
127                        
128                        @Override
129                        public Diagnostic diagnose(ProgressMonitor progressMonitor) {
130                                return Consumer.this.diagnose(progressMonitor);
131                        }
132                        
133                        @Override
134                        public void close() {
135                                Consumer.this.close();
136                        }
137                        
138                        @Override
139                        public void commit(ProgressMonitor progressMonitor) {
140                                Consumer.this.commit(progressMonitor);
141                        }
142                        
143                        @Override
144                        public boolean rollback(ProgressMonitor progressMonitor) {
145                                return Consumer.this.rollback(progressMonitor);
146                        }
147                        
148                        @Override
149                        public double size() {
150                                return Consumer.this.size();
151                        }
152                        
153                        @Override
154                        public String name() {
155                                return Consumer.this.name();
156                        }
157                        
158                };
159        }
160        
161        /**
162         * @return Function which executes this consumer passing BiSupplier.getFirst() (T) and returns getSecond() (R). 
163         * This method can be used for chaining consumers.
164         */
165        default <R> Function<BiSupplier<T,R>,R> asBiSupplierFunction() {
166                return new Function<BiSupplier<T,R>,R>() {
167                        
168                        @Override
169                        public R execute(BiSupplier<T,R> input, ProgressMonitor progressMonitor) {
170                                Consumer.this.execute(input.getFirst(), progressMonitor);
171                                return input.getSecond();
172                        }
173                        
174                        @Override
175                        public Diagnostic diagnose(ProgressMonitor progressMonitor) {
176                                return Consumer.this.diagnose(progressMonitor);
177                        }
178                        
179                        @Override
180                        public void close() {
181                                Consumer.this.close();
182                        }
183                        
184                        @Override
185                        public void commit(ProgressMonitor progressMonitor) {
186                                Consumer.this.commit(progressMonitor);
187                        }
188                        
189                        @Override
190                        public boolean rollback(ProgressMonitor progressMonitor) {
191                                return Consumer.this.rollback(progressMonitor);
192                        }
193                        
194                        @Override
195                        public double size() {
196                                return Consumer.this.size();
197                        }
198                        
199                        @Override
200                        public String name() {
201                                return Consumer.this.name();
202                        }
203                        
204                };
205        }
206                                
207        static <T> Consumer<T> fromBiConsumer(BiConsumer<T,ProgressMonitor> biConsumer, String name, double size) {
208                return new Consumer<T>() {
209
210                        @Override
211                        public double size() {
212                                return size;
213                        }
214
215                        @Override
216                        public String name() {
217                                return name;
218                        }
219
220                        @Override
221                        public void execute(T arg, ProgressMonitor progressMonitor) {
222                                biConsumer.accept(arg, progressMonitor);                                
223                        }
224                };
225        }
226        
227        /**
228         * @param <T>
229         * @param supplier
230         * @param name
231         * @param size
232         * @return
233         */
234        static <T> Consumer<T> fromConsumer(java.util.function.Consumer<T> consumer, String name, double size) {
235                return fromBiConsumer((arg,progressMonitor) -> consumer.accept(arg), name, size);               
236        }
237        
238        default java.util.function.Consumer<T> toConsumer(ProgressMonitor progressMonitor) {
239                return arg -> execute(arg, progressMonitor);
240        }
