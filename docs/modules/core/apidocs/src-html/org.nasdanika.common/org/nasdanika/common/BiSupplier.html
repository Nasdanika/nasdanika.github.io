001package org.nasdanika.common;
002
003import java.util.function.BiConsumer;
004
005/**
006 * Supplier of two values of different type.
007 * @author Pavel
008 *
009 * @param <T>
010 * @param <U>
011 */
012public interface BiSupplier<T,U> extends java.util.function.Consumer<BiConsumer<T, U>> {
013        
014        T getFirst();
015        
016        U getSecond();
017        
018        default void accept(BiConsumer<T, U> consumer) {
019                consumer.accept(getFirst(), getSecond());
020        }
021        
022        static <T,U> BiSupplier<T, U> of(T first, U second) {
023                return new BiSupplier<T, U>() {
024
025                        @Override
026                        public T getFirst() {
027                                return first;
028                        }
029
030                        @Override
031                        public U getSecond() {
032                                return second;
033                        }
034                        
035                        @Override
036                        public String toString() {
037                                return BiSupplier.class.getName() + " { first=" + first +", second=" + second + "}";
038                        }
039                };
040        }
041
042}
