001package org.nasdanika.common;
002
003import java.util.Arrays;
004import java.util.List;
005
006/**
007 * For each argument element an element consumer at the same index is invoked if present. 
008 * @author Pavel
009 *
010 * @param <E>
011 */
012public class ListCompoundConsumer<E> extends ListCompoundExecutionParticipant<Consumer<? super E>> implements Consumer<List<E>>  {
013
014        public ListCompoundConsumer(String name, List<Consumer<? super E>> consumers) {
015                super(name);
016                for (Consumer<? super E> consumer: consumers) {
017                        add(consumer);
018                }
019        }
020        
021        @SafeVarargs
022        public ListCompoundConsumer(String name, Consumer<? super E>... suppliers) {
023                this(name, Arrays.asList(suppliers));
024        }
025
026        @Override
027        public void execute(List<E> args, ProgressMonitor progressMonitor) {
028                if (args != null) {
029                        progressMonitor.setWorkRemaining(size());
030                        int idx = 0;
031                        List<Consumer<? super E>> els = getElements();
032                        for (E arg: args) {
033                                if (idx < els.size()) {
034                                        Consumer<? super E> ec = els.get(idx);
035                                        if (ec != null) {
036                                                ec.splitAndExecute(arg, progressMonitor);
037                                        }
038                                }
039                                ++idx;
040                        }
041                }
042        }       
043
044}
