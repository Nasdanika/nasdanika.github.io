001package org.nasdanika.resources;
002
003import java.util.Map;
004
005import org.nasdanika.common.ProgressMonitor;
006
007/**
008 * Container of resources.
009 * @author Pavel
010 *
011 * @param <T> Content type.
012 */
013public interface ResourceContainer<E extends Resource<?>> extends Container<E> {
014                
015        @SuppressWarnings("unchecked")
016        @Override
017        default void copy(Container<? super E> container, String path, ProgressMonitor monitor) {
018                if (exists(monitor.split("Checking existence", 1, this))) {                     
019                        long size = size(monitor.split("Getting size", 1, this));
020                        try (ProgressMonitor subMonitor = monitor.split("Copying "+getName(), size, this)) {
021                                Container<? super E> target = container.getContainer(path, monitor.split("Getting target container "+path, 1, container));
022                                Map<String, Object> children = getChildren(monitor.split("Getting children", 1, this));
023                                if (children != null) {                         
024                                        children.forEach((childName, child) -> ((Resource<E>) child).copy(target, childName, subMonitor.split("Copying child "+path, 1, this)));
025                                }
026                        }
027                }
028        }
029        
030        @SuppressWarnings("unchecked")
031        @Override
032        default void move(Container<? super E> container, String path, ProgressMonitor monitor) {
033                if (exists(monitor.split("Checking existence", 1, this))) {                     
034                        long size = size(monitor.split("Getting size", 1, this));
035                        try (ProgressMonitor subMonitor = monitor.split("Moving "+getName(), size*2 + 1, this)) {
036                                Container<? super E> target = container.getContainer(path, monitor.split("Getting target container "+path, 1, container));
037                                Map<String, Object> children = getChildren(monitor.split("Getting children", 1, this));
038                                if (children != null) {                         
039                                        children.forEach((childName, child) -> ((Resource<E>) child).move(target, childName, subMonitor.split("Copying child "+path, 1, this)));
040                                }
041                                delete(subMonitor.split("Deleting", 1, this));
042                        }
043                }                               
044        }
045        
046}
