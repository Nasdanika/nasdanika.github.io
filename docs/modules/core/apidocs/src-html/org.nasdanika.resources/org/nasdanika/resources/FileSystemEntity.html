001package org.nasdanika.resources;
002
003import java.io.BufferedInputStream;
004import java.io.BufferedOutputStream;
005import java.io.FileInputStream;
006import java.io.FileNotFoundException;
007import java.io.FileOutputStream;
008import java.io.IOException;
009import java.io.InputStream;
010
011import org.nasdanika.common.NasdanikaException;
012import org.nasdanika.common.ProgressMonitor;
013
014/**
015 * Wrapper for a file system file.
016 * @author Pavel
017 *
018 */
019public class FileSystemEntity extends FileSystemResource implements BinaryEntity {
020
021        public FileSystemEntity(java.io.File file) {
022                super(file);
023        }
024
025        @Override
026        public InputStream getState(ProgressMonitor monitor) {
027                if (!file.isFile()) {
028                        throw new UnsupportedOperationException("Not a file: "+file.getAbsolutePath());
029                }
030                if (!canRead()) {
031                        throw new UnsupportedOperationException("Cannot read: "+file.getAbsolutePath());                        
032                }
033                try {
034                        return new FileInputStream(file);
035                } catch (FileNotFoundException e) {
036                        throw new NasdanikaException(e);
037                }
038        }
039
040        @Override
041        public void setState(InputStream contents, ProgressMonitor monitor) {
042                java.io.File parent = file.getParentFile();
043                if (!parent.exists() && !parent.mkdirs()) {
044                        throw new NasdanikaException("Directory does not exist and could not be created: "+parent.getAbsolutePath());
045                }
046                long total = 0;
047                try (BufferedInputStream bin = new BufferedInputStream(contents); BufferedOutputStream bout = new BufferedOutputStream(new FileOutputStream(file))) {
048                        int b;
049                        while ((b = bin.read()) != -1) {
050                                bout.write(b);
051                                ++total;
052                                // Progress reporting at a regular interval?
053                        }
054                } catch (IOException e) {
055                        monitor.worked(100, "Error setting contents: "+e+", stored "+total+" bytes");
056                        throw new NasdanikaException("Could not set contents of "+file.getAbsolutePath(), e);
057                }
058                monitor.worked(100, "Stored "+total+" bytes");
059        }
060
061        @Override
062        public void appendState(InputStream contents, ProgressMonitor monitor) {
063                long total = 0;
064                try (BufferedInputStream bin = new BufferedInputStream(contents); BufferedOutputStream bout = new BufferedOutputStream(new FileOutputStream(file, true))) {
065                        int b;
066                        while ((b = bin.read()) != -1) {
067                                bout.write(b);
068                                ++total;
069                                // Progress reporting at a regular interval?
070                        }
071                } catch (IOException e) {
072                        monitor.worked(100, "Error appending contents: "+e+", appended "+total+" bytes");
073                        throw new NasdanikaException("Could not append contents to "+file.getAbsolutePath(), e);
074                }
075                monitor.worked(100, "Appended "+total+" bytes");
076        }
077
078        @Override
079        public boolean canRead() {
080                return file.canRead();
081        }
082
083        @Override
084        public boolean canWrite() {
085                return file.canWrite();
086        }
087
088        @Override
089        public long size(ProgressMonitor monitor) {
090                return file.isFile() ? file.length() : 0;
091        }
092

094        
095}
