001package org.nasdanika.resources;
002
003import java.util.Collections;
004import java.util.HashMap;
005import java.util.Map;
006
007import org.nasdanika.common.ProgressMonitor;
008
009/**
010 * Container backed by a map. For this container delete() means removal from the map.
011 * @author Pavel
012 *
013 * @param <T>
014 */
015public class MapContainer<E> implements Container<E> {
016                        
017        protected Map<String,Object> children = new HashMap<>();
018
019        /**
020         * Create a new root memory container.
021         */
022        public MapContainer() {
023                
024        }
025        
026        @SuppressWarnings("unchecked")
027        @Override
028        public Object find(String path, ProgressMonitor monitor) {
029                int sPos = path.indexOf(SEPARATOR);
030                if (sPos == -1) {
031                        return children.get(path); 
032                }
033                String firstSegment = path.substring(0, sPos);
034                Object child = children.get(firstSegment);
035                if (child instanceof Container) {
036                        return ((Container<E>) child).find(path.substring(sPos+1), monitor.split("Finding in "+firstSegment, 1, child));
037                }
038                return null;
039        }
040
041        @SuppressWarnings("unchecked")
042        @Override
043        public E get(String path, ProgressMonitor monitor) {
044                Object existing = find(path, monitor.split("Finding existing child at " + path, 1));
045                if (existing instanceof Container) {
046                        // container - can't have another child with the same name.
047                        return null;
048                }
049                if (existing != null) {
050                        return (E) existing;
051                }
052                
053                int sPos = path.indexOf(SEPARATOR);
054                if (sPos == -1) {
055                        E ret = createElement(path, monitor.split("Creating element on access at "+path, 1, this));
056                        children.put(path, ret);
057                        return ret;
058                }
059                
060                String pathHead = path.substring(0, sPos);
061                Container<E> container = getContainer(pathHead, monitor.split("Getting sub-container at "+pathHead, 1, this));
062                String pathTail = path.substring(sPos + 1);
063                return container == null ? null : container.get(pathTail, monitor.split("Getting element at "+pathTail , 1, container));
064        }
065        
066        /**
067         * Override to create elements on access. This method delegates to the parent if the parent is an instance of MapContainer.
068         * @param path
069         * @param monitor
070         * @return
071         */
072        protected E createElement(String path, ProgressMonitor monitor) {
073                Container<E> parent = getParent();
074                if (parent instanceof MapContainer) {
075                        return ((MapContainer<E>) parent).createElement(path, monitor);
076                }
077                return null;
078        }
079
080        @Override
081        public void put(String path, E element, ProgressMonitor monitor) throws IllegalArgumentException {
082                int sPos = path.indexOf(SEPARATOR);
083                if (sPos == -1) {
084                        children.put(path, element);
085                } else {                
086                        String firstSegment = path.substring(0, sPos);
087                        Container<E> container = getContainer(firstSegment, monitor.split("Getting container at "+firstSegment, 1, this));
088                        if (container == null) {
089                                throw new IllegalArgumentException("Unable to put element at "+path+" - no container at firstSe");
090                        }
091                        String pathTail = path.substring(sPos + 1);
092                        container.put(pathTail, element, monitor.split("Putting element at "+pathTail, 1, container));
093                }
094        }
095
096        @SuppressWarnings("unchecked")
097        @Override
098        public Object delete(String path, ProgressMonitor monitor) {
099                Object existing = find(path, monitor.split("Finding existing element at "+path, 1));
100                if (existing == null) {
101                        return null;
102                }
103                
104                int sPos = path.indexOf(SEPARATOR);
105                
106                if (sPos == -1) {
107                        return children.remove(path);
108                }
109                
110                if (existing instanceof Container) {
111                        ((Container<E>) existing).delete(monitor.split("Removing container at "+path, 1, existing));
112                        return existing;
113                }
114                
115                String firstSegment = path.substring(0, sPos);
116                Container<E> container = getContainer(firstSegment, monitor.split("Getting container at "+firstSegment, 1, this));
117                String pathTail = path.substring(sPos + 1);
118                return container == null ? null : container.delete(pathTail, monitor.split("Getting container for "+pathTail, 1, container));
119        }
120
121        @SuppressWarnings("unchecked")
122        @Override
123        public Container<E> getContainer(String path, ProgressMonitor monitor) {
124                Object existing = find(path, monitor.split("Finding existing element at "+path, 1));
125                if (existing instanceof Container) {
126                        return (Container<E>) existing;
127                }
128                if (existing != null) {
129                        // non-container - can't have another resource with the same name.
130                        return null;
131                }
132                int sPos = path.indexOf(SEPARATOR);
133                if (sPos == -1) {
134                        Container<E> ret = createSubContainer(path);
135                        children.put(path, ret);
136                        return ret;
137                }
138                
139                String firstSegment = path.substring(0, sPos);
140                Container<E> container = getContainer(firstSegment, monitor.split("Getting container at "+firstSegment, 1, this));
141                String pathTail = path.substring(sPos + 1);
142                return container == null ? null : container.getContainer(pathTail, monitor.split("Getting container for "+pathTail, 1, container));
143        }
144
145        /**
146         * Creates a sub-container, override to specialize returned container.
147         * @param path
148         * @return
149         */
150        protected MapContainer<E> createSubContainer(String path) {
151                return new MapContainer<E>() {
152
153                        @Override
154                        public String getName() {
155                                return path;
156                        }
157
158                        @Override
159                        public Container<E> getParent() {
160                                return MapContainer.this;
161                        }
162                        
163                };
164        }
165
166        @Override
167        public Map<String, Object> getChildren(ProgressMonitor monitor) {
168                return Collections.unmodifiableMap(children);
169        }
170
171        @Override
172        public Container<E> getParent() {
173                return null;
174        }
175
176        @Override
177        public String getName() {
178                return null;
179        }
180
181        @Override
182        public String toString() {
183                return getClass().getName()+"("+getPath()+")";
184        }
185
186        @SuppressWarnings("unchecked")
187        @Override
188        public void copy(Container<? super E> container, String path, ProgressMonitor monitor) {
189                if (exists(monitor.split("Checking existence", 1, this))) {                     
190                        long size = size(monitor.split("Getting size", 1, this));
191                        try (ProgressMonitor subMonitor = monitor.split("Copying "+getName(), size, this)) {
192                                Container<? super E> target = container.getContainer(path, monitor.split("Getting target container "+path, 1, container));
193                                Map<String, Object> children = getChildren(monitor.split("Getting children", 1, this));
194                                if (children != null) {                         
195                                        children.forEach((childName, child) -> {
196                                                if (child instanceof Resource) {
197                                                        ((Resource<E>) child).copy(target, childName, subMonitor.split("Copying child "+path, 1, this));
198                                                } else {
199                                                        put(childName, (E) child, subMonitor.split("Copying child "+path, 1, this));
200                                                }
201                                        });
202                                }
203                        }
204                }
205        }
206        
207        @SuppressWarnings("unchecked")
208        @Override
209        public void move(Container<? super E> container, String path, ProgressMonitor monitor) {
210                if (exists(monitor.split("Checking existence", 1, this))) {                     
211                        long size = size(monitor.split("Getting size", 1, this));
212                        try (ProgressMonitor subMonitor = monitor.split("Moving "+getName(), size*2 + 1, this)) {
213                                Container<? super E> target = container.getContainer(path, monitor.split("Getting target container "+path, 1, container));
214                                Map<String, Object> children = getChildren(monitor.split("Getting children", 1, this));
215                                if (children != null) {                         
216                                        children.forEach((childName, child) -> {
217                                                if (child instanceof Resource) {
218                                                        ((Resource<E>) child).copy(target, childName, subMonitor.split("Copying child "+path, 1, this));
219                                                } else {
220                                                        put(childName, (E) child, subMonitor.split("Copying child "+path, 1, this));
221                                                }
222                                        });
223                                }
224                                delete(subMonitor.split("Deleting", 1, this));
225                        }
226                }                               
227        }
