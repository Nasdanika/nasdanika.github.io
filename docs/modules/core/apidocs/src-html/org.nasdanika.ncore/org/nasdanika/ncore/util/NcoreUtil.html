071        /**
072         * Flattened inheritance hierarchy from the argument class to all of its supertypes in the order of inheritance.
073         * @param eClass
074         * @return
075         */
076        public static List<EClass> lineage(EClass eClass) {
077                List<EClass> ret = new ArrayList<>();
078                ret.add(eClass);
079                for (EClass sc: eClass.getESuperTypes()) {
080                        for (EClass scLineage: lineage(sc)) {
081                                if (!ret.contains(scLineage)) {
082                                        ret.add(scLineage);
083                                }
084                        }
085                }
086                return ret;
087        }

089        /**
090         * Finds objects referencing the argument object (target) via the specified reference.
091         * @param <T>
092         * @param target Referenced object
093         * @param eReference Reference
094         * @return Object referencing this one via the specified reference.
095         */
096        public static <T extends EObject> EList<T> getReferrers(EObject target, EReference eReference) {
097                EList<T> ret = new BasicInternalEList<>(EObject.class);
098                
099                Resource res = target.eResource(); 
100                TreeIterator<?> cit = null;
101                if (res == null) {
102                        EObject root = target;
103                        EObject rc;
104                        while ((rc = root.eContainer()) != null) {
105                                root = rc;
106                        }
107                        if (root != null) {
108                                NcoreUtil.collect(root, target, eReference, ret);
109                                cit = root.eAllContents();
110                        }                       
111                } else {
112                        ResourceSet rSet = res.getResourceSet();
113                        cit = rSet == null ? res.getAllContents() : rSet. getAllContents();
114                }
115                if (cit != null) {
116                        while (cit.hasNext()) {
117                                NcoreUtil.collect(cit.next(), target, eReference, ret);
118                        }                       
119                }
120                return ret;
121        }

123        @SuppressWarnings("unchecked")
124        private static <T extends EObject> void collect(
125                        Object source, 
126                        EObject target, 
127                        EReference eReference,
128                        EList<T> accumulator) {
129                if (eReference.getEContainingClass().isInstance(source)) {
130                        Object value = ((EObject) source).eGet(eReference);
131                        if (eReference.isMany()) {
132                                if (((Collection<?>) value).contains(target)) {
133                                        accumulator.add((T) source);
134                                }
135                        } else if (value == target) {
136                                accumulator.add((T) source);
137                        }                                       
138                }
139        }

156        /**
157         * Containment path relative to the object container. If the container is a map entry its key is added to path and the entry container is skipped. This happens recursively. 
158         * @param eObject
159         * @return null if there is no container. BiSupplier with first returning the first container which is not EMap entry and reference name concatenated with object key for {@link EMap}'s and with object index for many-references.
160         */
161        public static BiSupplier<EObject, String> containmentPath(EObject eObj) {
162                EObject container = eObj.eContainer();
163                if (container == null) {
164                        return null;
165                }
166                EReference eContainmentFeature = eObj.eContainmentFeature();
167                
168                LinkedList<String> containmentIndex = new LinkedList<>();
169                while (NcoreUtil.isEMapEntry(container)) {
170                        Object key = ((Map.Entry<?,?>) container).getKey();
171                        if (key == null) {
172                                return null;
173                        }
174                        String idx = NcoreUtil.containmentIndex(eObj);
175                        if (idx != null) {
176                                containmentIndex.addFirst(idx);
177                        }
178                        containmentIndex.addFirst(key.toString());
179                        
180                        eContainmentFeature = container.eContainmentFeature();
181                        container = container.eContainer();
182                }
183                
184                if (containmentIndex.isEmpty()) {
185                        String idx = NcoreUtil.containmentIndex(eObj);
186                        if (idx != null) {
187                                containmentIndex.add(idx);
188                        }
189                } 
190        
191                containmentIndex.addFirst(NcoreUtil.getFeatureKey(container.eClass(), eContainmentFeature));
192                return BiSupplier.of(container, String.join("/", containmentIndex));                            
193        }

195        /**
196         * Index part of the containment path. Null for single references. EKeys for containments with EKeys. Position otherwise.
197         * @param eObj
198         * @return
199         */
200        private static String containmentIndex(EObject eObj) {
201                EObject container = eObj.eContainer();
202                if (container == null) {
203                        return null;
204                }
205                EReference eContainmentFeature = eObj.eContainmentFeature();
206                if (eContainmentFeature.isMany()) {
207                        EList<EAttribute> eKeys = eContainmentFeature.getEKeys();
208                        String position = String.valueOf(((List<?>) container.eGet(eContainmentFeature)).indexOf(eObj));
209                        if (eKeys.isEmpty()) {
210                                return position;
211                        }
212                        StringBuilder pathBuilder = new StringBuilder();
213                        for (EAttribute eKey: eKeys) {
214                                if (!eObj.eIsSet(eKey)) {
215                                        return position;
216                                }
217                                if (pathBuilder.length() > 0) {
218                                        pathBuilder.append("/");
219                                }
220                                pathBuilder.append(eObj.eGet(eKey));
221                        }
222                        return pathBuilder.toString();
223                }
224                return null;
225        }

227        /**
228                 * Returns {@link ModelElement}.getUri() if the argument is an instance of ModelElement and getUri() returns a non blank string.
229                 * Computes URI from the containment reference otherwise.
230                 * @param eObj
231                 * @return
232                 */
233                public static URI getUri(EObject eObj) {
234                        if (eObj == null || isEMapEntry(eObj)) {
235                                return null; // EMap entries do not have URI's.
236                        }
237
238                        URI ret = null;
239                        if (eObj instanceof ModelElement) {
240                                String uri = ((ModelElement) eObj).getUri();
241                                if (!Util.isBlank(uri)) {
242                                        ret = URI.createURI(uri);
243                                        if (ret.hasAbsolutePath()) {
244                                                return ret;
245                                        }
246                                }
247                        }                       
248                        
249                        BiSupplier<EObject, String> containmentPath = containmentPath(eObj);
250                        if (containmentPath == null) {
251                                return ret;
252                        }
253                        EObject container = containmentPath.getFirst();
254                        if (container == null) {
255                                if (ret == null && eObj instanceof ModelElement) {
256                                        String uuid = ((ModelElement) eObj).getUuid();
257                                        if (!Util.isBlank(uuid)) {
258                                                return URI.createURI("uuid://" + uuid);
259                                        }
260                                }
261                                return ret;
262                        }
263                        
264                        URI cURI = getUri(container);
265                        if (cURI == null) {
266                                return ret;
267                        }
268                        
269                        URI cpURI = URI.createURI(containmentPath.getSecond());
270                        
271                        if (cURI.hasAbsolutePath()) {
272                                String cLastSegment = cURI.lastSegment();
273                                if (cLastSegment == null || cLastSegment.length() > 0) {
274                                        cURI = cURI.appendSegment("");
275                                }
276                                URI containementURI = cpURI.resolve(cURI);
277                                return ret == null ? containementURI : ret.resolve(containementURI);
278                        }
279                        
280                        // Resolving containment path URI against container URI by combining all segments and treating the first first segment as scheme, second as authority and the rest as segments.                 
281                        List<String> allParts = new ArrayList<>();
282                        if (cURI.scheme() != null) {
283                                allParts.add(cURI.scheme());
284                        }
285                        if (cURI.authority() != null) {
286                                allParts.add(cURI.authority());
287                        }
288                        allParts.addAll(cURI.segmentsList());
289                        
290                        if (cpURI.scheme() != null) {
291                                allParts.add(cpURI.scheme());
292                        }
293                        if (cpURI.authority() != null) {
294                                allParts.add(cpURI.authority());
295                        }
296                        allParts.addAll(cpURI.segmentsList());
297                        
298                        if (allParts.size() == 0) {
299                                return ret;
300                        }
301                        
302                        if (allParts.size() == 1) {
303                                return ret == null ? URI.createURI(allParts.get(0)) : ret;
304                        }
305
306                        if (allParts.size() == 2 && allParts.get(1).length() == 0) {
307                                return URI.createURI(String.join("/", allParts) + (ret == null ? "" : ret));                            
308                        }
309                        
310                        URI allPartsURI = URI.createHierarchicalURI(allParts.get(0), allParts.get(1), null, allParts.subList(2, allParts.size()).toArray(new String[allParts.size() - 2]), null, null);
311                        return ret == null ? allPartsURI : ret.resolve(allPartsURI);
312                }

315         * @param eObj
316         * @return true if the argument is an entry for EMap
317         */
318        public static boolean isEMapEntry(EObject eObj) {
319                if (eObj instanceof Map.Entry) {
320                        EReference cf = eObj.eContainmentFeature();
321                        return cf != null && cf.isMany() && cf.getEType().getInstanceClass() == Map.Entry.class;
322                }
323                return false;
324        }

326        public static EObject commonAncestor(EObject a, EObject b) {
327                if (a == null || b == null) {
328                        return null;
329                }
330                if (EcoreUtil.isAncestor(a, b)) {
331                        return a.eContainer();
332                }
333                if (EcoreUtil.isAncestor(b, a)) {
334                        b.eContainer();
335                }
336                for (EObject ancestor = a.eContainer(); ancestor != null; ancestor = ancestor.eContainer()) {
337                        if (EcoreUtil.isAncestor(ancestor, b)) {
338                                return ancestor;
339                        }
340                }
341                return null;
342        }

366        /**
367         * Feature key is used for feature loading from YAML and other key-value sources and also in construction of containment path by containmentPath() method and, as such, EObject {@link URI} by getUri() method.
368         * This method traverses the class inheritance hierarchy and uses feature-keys class-level annotations to retrieve feature key. If such an annotation is not found then feature-level feature-key annotation is used.
369         * If none is found in the entire hierarchy then feature name is converted to kebab case is returned. 
370         * @param eClass Context class for which feature key is retrieved.
371         * @param feature Feature
372         * @return
373         */
374        public static String getFeatureKey(EClass eClass, EStructuralFeature feature) {
375                String featureKey = NcoreUtil.getNasdanikaAnnotationDetail(feature, FEATURE_KEY);
376                if (!Util.isBlank(featureKey)) {
377                        return featureKey;
378                }
379                
380                for (EClass ec: NcoreUtil.lineage(eClass)) {
381                        String featureKeys = NcoreUtil.getNasdanikaAnnotationDetail(ec, FEATURE_KEYS);
382                        if (!Util.isBlank(featureKeys)) {
383                                Yaml yaml = new Yaml();
384                                Map<String,Object> featureKeyMap = yaml.load(featureKeys);
385                                Object fKey = featureKeyMap.get(feature.getName());
386                                if (fKey instanceof String) {
387                                        return (String) fKey;
388                                }
389                        }
390                }
391                return Util.camelToKebab(feature.getName());
392        }

394        /**
395         * 
396         * @param eReference
397         * @return "Logical" opposite - either the "physical" opposite returned by {@link EReference}.getEOpposite() or a reference with a name specified in "opposite" Nasdanika annotation
398         * or a reference which specifies this reference in its "opposite" Nasdanika annotation.
399         */
400        public static EReference getOpposite(EReference eReference) {
401                EReference opposite = eReference.getEOpposite();
402                if (opposite != null) {
403                        return opposite;
404                }
405                String oName = getNasdanikaAnnotationDetail(eReference, "opposite");
406                EClass refType = eReference.getEReferenceType();
407                for (EReference ref: refType.getEAllReferences()) {
408                        if (ref.getName().equals(oName)) {
409                                return ref;
410                        }
411                        if (ref.getEReferenceType() == eReference.getEContainingClass() && eReference.getName().equals(getNasdanikaAnnotationDetail(ref, "opposite"))) {
412                                return ref;
413                        }
414                }
415                return null;
416        }

418        /**
419         * Computes end and duration from start and previously loaded things.
420         * @param period
421         * @param newStart
422         */
423        public static void onStart(Period period, Temporal newStart) {
424                if (newStart != null) {
425                        LoadTracker loadTracker = (LoadTracker) EcoreUtil.getRegisteredAdapter(period, LoadTracker.class);
426                        if (loadTracker != null && loadTracker.isLoading(NcorePackage.Literals.PERIOD__START)) {
427                                if (loadTracker.isLoaded(NcorePackage.Literals.PERIOD__END)) {
428                                        if (!loadTracker.isLoaded(NcorePackage.Literals.PERIOD__DURATION)) {
429                                                period.setDuration(period.getEnd().minus(newStart));
430                                        }
431                                } else if (loadTracker.isLoaded(NcorePackage.Literals.PERIOD__DURATION)) {
432                                        period.setEnd(newStart.plus(period.getDuration()));
433                                }
434                        }
435                }
436        }

438        public static void onEnd(Period period, Temporal newEnd) {
439                if (newEnd != null) {
440                        LoadTracker loadTracker = (LoadTracker) EcoreUtil.getRegisteredAdapter(period, LoadTracker.class);
441                        if (loadTracker != null && loadTracker.isLoading(NcorePackage.Literals.PERIOD__END)) {
442                                if (loadTracker.isLoaded(NcorePackage.Literals.PERIOD__START)) {
443                                        if (!loadTracker.isLoaded(NcorePackage.Literals.PERIOD__DURATION)) {
444                                                period.setDuration(newEnd.minus(period.getStart()));
445                                        }
446                                } else if (loadTracker.isLoaded(NcorePackage.Literals.PERIOD__DURATION)) {
447                                        period.setStart(newEnd.minus(period.getDuration()));
448                                }
449                        }
450                }
451        }
452
453        public static void onDuration(Period period, Duration newDuration) {
454                if (newDuration != null) {
455                        LoadTracker loadTracker = (LoadTracker) EcoreUtil.getRegisteredAdapter(period, LoadTracker.class);
456                        if (loadTracker != null && loadTracker.isLoading(NcorePackage.Literals.PERIOD__DURATION)) {
457                                if (loadTracker.isLoaded(NcorePackage.Literals.PERIOD__START)) {
458                                        if (!loadTracker.isLoaded(NcorePackage.Literals.PERIOD__END)) {
459                                                period.setEnd(period.getStart().plus(newDuration));
460                                        }
461                                } else if (loadTracker.isLoaded(NcorePackage.Literals.PERIOD__END)) {
462                                        period.setStart(period.getEnd().minus(newDuration));
463                                }
464                        }
465                }
466        }

468        /**
469         * @param eReference
470         * @return Target EClass of association class reference.
471         */
472        public static EClass getAssociationTarget(EReference eReference) {
473                String association = NcoreUtil.getNasdanikaAnnotationDetail(eReference, "association");
474                if (Util.isBlank(association)) {
475                        return null;
476                }
477                EClass refType = eReference.getEReferenceType();
478                EStructuralFeature associationFeature = refType.getEStructuralFeature(association);
479                if (associationFeature instanceof EReference) {
480                        return ((EReference) associationFeature).getEReferenceType();
481                }
482                return null;            
483        }
