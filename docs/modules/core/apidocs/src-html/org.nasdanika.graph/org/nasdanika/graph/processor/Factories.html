001package org.nasdanika.graph.processor;
002
003import static java.lang.annotation.ElementType.FIELD;
004import static java.lang.annotation.ElementType.METHOD;
005import static java.lang.annotation.RetentionPolicy.RUNTIME;
006
007import java.lang.annotation.Retention;
008import java.lang.annotation.Target;
009import java.util.Collection;
010
011/**
012 * For methods and fields this annotation indicates that the method return value or field value shall be a {@link Collection} whose elements are used as introspection targets to create element processors.
013 * @author Pavel
014 *
015 */
016@Retention(RUNTIME)
017@Target({ FIELD, METHOD })
018public @interface Factories {
019        
020        /**
021         * If not blank, the value shall be a <a href="https://docs.spring.io/spring-framework/docs/5.3.22/reference/html/core.html#expressions">Spring boolean expression</a>
022         * which is evaluated in the context of an element. 
023         * @return
024         */
025        String value() default "";
026        
027        /**
028         * @return {@link IntrospectionLevel}, DECLARED by default.
029         */
030        IntrospectionLevel introspect() default IntrospectionLevel.DECLARED;
031        
032        /**
033         * Matching priority. Defaults to 0. Higher priorities are matched before lower priorities.
034         * @return
035         */
036        int priority() default 0;
037
038}
