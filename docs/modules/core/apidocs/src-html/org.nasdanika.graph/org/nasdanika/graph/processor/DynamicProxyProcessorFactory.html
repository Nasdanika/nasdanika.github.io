001package org.nasdanika.graph.processor;
002
003import java.lang.reflect.Method;
004import java.lang.reflect.Proxy;
005import java.util.function.Supplier;
006
007import org.nasdanika.graph.Connection;
008
009/**
010 * Mix-in interface which creates handler proxies as dynamic proxies.
011 * @author Pavel
012 *
013 * @param <P>
014 * @param <H>
015 * @param <E>
016 */
017public interface DynamicProxyProcessorFactory<P, H, E> extends ProcessorFactory<P, H, E> {
018        
019        @SuppressWarnings("unchecked")
020        @Override
021        default H createHandlerProxy(Connection connection, Supplier<H> handlerSupplier, HandlerType type) {
022                return (H) Proxy.newProxyInstance(getClassLoader(), new Class[] { getHandlerInterface(connection, type) }, (Object proxy, Method method, Object[] args) -> {
023                        return method.invoke(handlerSupplier.get(), args);
024                });
025        }
026        
027        /**
028         * @return Handler interface class to create a dynamic proxy.
029         */
030        Class<?> getHandlerInterface(Connection connection, HandlerType type);
031        
032        /**
033         * @return Classloader to create a proxy. This implementation returns current {@link Thread}'s context classloader.
034         */
035        default ClassLoader getClassLoader() {
036                return Thread.currentThread().getContextClassLoader();
037        }
038
039}
