001package org.nasdanika.graph.processor;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.util.Map;
006import java.util.Objects;
007import java.util.function.Supplier;
008import java.util.stream.Stream;
009
010import org.eclipse.emf.common.util.URI;
011import org.eclipse.emf.ecore.EObject;
012import org.eclipse.emf.ecore.resource.impl.ResourceImpl;
013import org.nasdanika.graph.Element;
014
015
016/**
017 * Base class for resource which load their contents by creating a {@link Stream} of {@link Element}'s from a resource {@link URI} and then uses {@link ProcessorFactory} to create {@link Supplier}'s of {@link EObject} with {@link ProcessorFactory}. 
018 * @author Pavel
019 *
020 */
021public abstract class GraphProcessorResource<P> extends ResourceImpl {
022        
023        private ProcessorFactory<P, ?, ?> processorFactory;
024        
025        protected GraphProcessorResource(URI uri, ProcessorFactory<P, ?, ?> processorFactory) {
026                super(uri);
027                this.processorFactory = processorFactory;
028        }
029
030        /**
031         * Loads {@link Element}s from the input stream to pass to 
032         * @param inputStream
033         * @param options
034         * @return
035         * @throws IOException
036         */
037        protected abstract Stream<Element> loadElements(InputStream inputStream, Map<?, ?> options) throws IOException;
038        
039        @Override
040        protected void doLoad(InputStream inputStream, Map<?, ?> options) throws IOException {
041                Map<Element, ProcessorInfo<P>> registry = processorFactory.createProcessors(loadElements(inputStream, options));
042                getRoots(getSemanticElements(registry)).forEach(getContents()::add);
043        }
044
045        /**
046         * Retrieves semantic elements {@link EObject}s from the registry.
047         * @param registry
048         * @return
049         */
050        protected abstract Stream<EObject> getSemanticElements(Map<Element, ProcessorInfo<P>> registry);
051
052        /**
053         * Loads root objects from a stream of semantic elements. This implementation loads objects not contained in other objects. 
054         * @param registry
055         * @return
056         */
057        protected Stream<? extends EObject> getRoots(Stream<EObject> semanticElements) {
058                return semanticElements.filter(Objects::nonNull).filter(eObj -> eObj.eContainer() == null);
059        }
060        
061}
