001package org.nasdanika.graph;
002
003import static java.lang.annotation.ElementType.METHOD;
004import static java.lang.annotation.RetentionPolicy.RUNTIME;
005
006import java.lang.annotation.Retention;
007import java.lang.annotation.Target;
008
009/**
010 * Annotation for methods to which reflective visitor calls are dispatched. 
011 * Methods may have one or two parameters taking the element and a map of children.
012 * @author Pavel
013 *
014 */
015@Retention(RUNTIME)
016@Target({METHOD})
017public @interface Handler {
018        
019        /**
020         * If not blank, the value shall be a <a href="https://docs.spring.io/spring-framework/docs/5.3.22/reference/html/core.html#expressions">Spring boolean expression</a>
021         * which is evaluated in the context of the element being visited. 
022         * @return
023         */
024        String value() default "";
025        
026        /**
027         * Matching priority. Defaults to 0. Higher priorities are matched before lower priorities.
028         * @return
029         */
030        int priority() default 0;
031                
032}
