001package org.nasdanika.graph.processor;
002
003public interface ProcessorInfo<P> {
004        
005        ProcessorConfig<P> getConfig();
006        
007        P getProcessor();
008        
009        static <P> ProcessorInfo<P> of(ProcessorConfig<P> config, P processor) {
010                return new ProcessorInfo<P>() {
011
012                        @Override
013                        public ProcessorConfig<P> getConfig() {
014                                return config;
015                        }
016
017                        @Override
018                        public P getProcessor() {
019                                return processor;
020                        }
021                };
022        }
023
024}
