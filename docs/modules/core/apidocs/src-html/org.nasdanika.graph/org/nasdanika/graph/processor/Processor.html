001package org.nasdanika.graph.processor;
002
003import static java.lang.annotation.ElementType.METHOD;
004import static java.lang.annotation.RetentionPolicy.RUNTIME;
005
006import java.lang.annotation.Retention;
007import java.lang.annotation.Target;
008
009import org.nasdanika.graph.Element;
010
011/**
012 * Annotation for method creating an {@link Element} processor.
013 * The method shall have zero or one parameter. In the second case the parameter type shall be assignable from 
014 * {@link ProcessorConfig} or its subclasses - {@link NodeProcessorConfig} or {@link ConnectionProcessorConfig}.
015 * An annotated method may return an object or null. In the first case the returned object may be introspected to wire handlers and endpoints 
016 * to methods and fields with annotations.
017 * @author Pavel
018 *
019 */
020@Retention(RUNTIME)
021@Target(METHOD)
022public @interface Processor {
023        
024        /**
025         * If not blank, the value shall be a <a href="https://docs.spring.io/spring-framework/docs/5.3.22/reference/html/core.html#expressions">Spring boolean expression</a>
026         * which is evaluated in the context of an element. 
027         * @return
028         */
029        String value() default "";
030        
031        /**
032         * @return {@link IntrospectionLevel}, DECLARED by default.
033         */
034        IntrospectionLevel introspect() default IntrospectionLevel.DECLARED;
035        
036        /**
037         * Matching priority. Defaults to 0. Higher priorities are matched before lower priorities.
038         * @return
039         */
040        int priority() default 0;
041        
042        Class<Element> type() default Element.class; 
043        
044        /**
045         * @return If true, handlers and endpoints wired to annotated methods and fields are removed from {@link ProcessorConfig} returned from {@link ProcessorInfo}.getConfig().
046         */
047        boolean hideWired() default true;
048
049}
