001package org.nasdanika.drawio;
002
003import java.util.Map;
004import java.util.function.BiFunction;
005import java.util.function.Consumer;
006import java.util.stream.Stream;
007
008import org.eclipse.emf.common.util.URI;
009
010/**
011 * {@link Document} element, including the Document itself, backed by org.w3c.dom.Element.
012 * @author Pavel
013 *
014 */
015public interface Element extends org.nasdanika.graph.Element {
016        
017        /**
018         * @return The underlying XML element.
019         */
020        org.w3c.dom.Element getElement();
021        
022        /**
023         * Accepts the visitor in children first way.
024         * @param visitor
025         */
026        default void accept(Consumer<? super Element> visitor, ConnectionBase connectionBase) {
027                accept((e, cr) -> { visitor.accept(e); return null; }, connectionBase);
028        }
029        
030        /**
031         * Accepts a result producing visitor {@link BiFunction}. Passes to it a map of results collected from children, nulls are not included.   
032         * @param <T>
033         * @param visitor
034         * @return result returned by the visitor.
035         */
036        <T> T accept(BiFunction<Element, Map<Element, T>, T> visitor, ConnectionBase connectionBase);
037                
038        @Override
039        default <T> T accept(BiFunction<? super org.nasdanika.graph.Element, Map<? extends org.nasdanika.graph.Element, T>, T> visitor) {
040                BiFunction<Element, Map<Element, T>, T> visitorAdapter = (element, childMappings) -> {
041                        return visitor.apply(element, childMappings);
042                };
043                return accept(visitorAdapter, ConnectionBase.PARENT);
044        }
045        
046        /**
047         * @return Stream containing this element and its children.
048         */
049        default Stream<Element> stream(ConnectionBase connectionBase) {
050                BiFunction<Element, Map<Element, Stream<Element>>, Stream<Element>> visitor = (element, childMappings) -> {
051                        return childMappings.values().stream().reduce(Stream.of(element), (a,b) -> Stream.concat(a, b));
052                };
053                return accept(visitor, connectionBase);
054        }; 
055                
056        URI getURI();   
057
058}
