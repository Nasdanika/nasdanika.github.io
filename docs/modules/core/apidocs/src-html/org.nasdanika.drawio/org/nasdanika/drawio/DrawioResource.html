001package org.nasdanika.drawio;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.io.OutputStream;
006import java.io.OutputStreamWriter;
007import java.io.Writer;
008import java.nio.charset.StandardCharsets;
009import java.util.Map;
010import java.util.stream.Stream;
011
012import javax.xml.parsers.ParserConfigurationException;
013import javax.xml.transform.TransformerException;
014
015import org.eclipse.emf.common.util.URI;
016import org.nasdanika.common.NasdanikaException;
017import org.nasdanika.graph.processor.GraphProcessorResource;
018import org.nasdanika.graph.processor.ProcessorFactory;
019import org.xml.sax.SAXException;
020
021/**
022 * Loads EMF classes from Drawio documents. 
023 * Saving is supported via {@link UpdateAdapter}. All contents is iterated and for each element with UpdateAdapter the adapter's update method is invoked.
024 * The load logic is responsible for attaching adapters to model elements. 
025 * @author Pavel
026 *
027 */
028public abstract class DrawioResource<P> extends GraphProcessorResource<P> {
029        
030        protected Document document;
031        
032        public DrawioResource(URI uri, ProcessorFactory<P, ?, ?> processorFactory) {
033                super(uri, processorFactory);
034        }
035        
036        @Override
037        protected Stream<org.nasdanika.graph.Element> loadElements(InputStream inputStream, Map<?, ?> options)  throws IOException {
038                try {
039                        document = loadDocument(inputStream);
040                        return Stream.of(document);
041                } catch (ParserConfigurationException | SAXException e) {
042                        throw new NasdanikaException(e);
043                }
044        }
045
046        protected Document loadDocument(InputStream inputStream) throws IOException, ParserConfigurationException, SAXException {
047                return Document.load(inputStream, getURI());
048        }
049
050        @Override
051        protected void doSave(OutputStream outputStream, Map<?, ?> options) throws IOException {
052                if (document == null) {
053                        try {
054                                document = Document.create(false, getURI());
055                        } catch (ParserConfigurationException e) {
056                                throw new NasdanikaException(e);
057                        }
058                }
059                update(document);
060                try (Writer writer = new OutputStreamWriter(outputStream, StandardCharsets.UTF_8)) {
061                        try {
062                                writer.write(document.save(null));
063                        } catch (TransformerException e) {
064                                throw new NasdanikaException(e);
065                        }
066                }
067        }
068
069        /**
070         * Override to update document with model data.
071         * @param document Document to update. It may be a previously loaded document or a new empty document for new resources.
072         */
073        protected void update(Document document) {}
074        
075}
