001package org.nasdanika.diagram.gen;
002
003import java.io.File;
004import java.io.IOException;
005import java.nio.file.Files;
006
007import org.eclipse.emf.common.util.URI;
008import org.eclipse.emf.ecore.resource.Resource;
009import org.nasdanika.common.DefaultConverter;
010import org.nasdanika.common.DiagramGenerator;
011import org.nasdanika.common.NasdanikaException;
012import org.nasdanika.common.Util;
013import org.nasdanika.diagram.Diagram;
014import org.nasdanika.drawio.Document;
015import org.nasdanika.ncore.Marker;
016
017/**
018 * Generates diagram HTML delegating to different generators based on diagram type.
019 * @author Pavel
020 *
021 */
022public class Generator {
023        
024        public static Generator INSTANCE = new Generator();
025
026        public static final String PLANTUML_SCHEMA = "plantuml:";
027        public static final String DRAWIO_SCHEMA = "drawio:";
028//      public static final String MERMAID_SCHEMA = "mermaid:";
029
030        public String generate(Diagram diagram) {
031                String type = diagram.getType();
032                try {
033                        if (type.startsWith(PLANTUML_SCHEMA)) {
034                                String dialect = type.substring(PLANTUML_SCHEMA.length()).toLowerCase();
035                                return createPlantumlGenerator().generateDiagram(diagram, dialect);
036                        }
037                        
038        //              if (type.startsWith(MERMAID_SCHEMA)) {
039        //                      String dialect = type.substring(MERMAID_SCHEMA.length()).toLowerCase();
040        //                      return createMermaidGenerator().generateDiagram(diagram, dialect);
041        //              }
042                        
043                        if (type.startsWith(DRAWIO_SCHEMA)) {
044                                String diagramURI = type.substring(DRAWIO_SCHEMA.length());
045                                URI uri = URI.createURI(diagramURI);
046                                
047                                URI markerBase = null;
048                                for (Marker marker: diagram.getMarkers()) { 
049                                        if (marker != null && !Util.isBlank(marker.getLocation())) { // Finds a first marker with location
050                                                markerBase = URI.createURI(marker.getLocation());
051                                                break;
052                                        }                               
053                                }
054        
055                                URI resourceBase = null;
056                                Resource resource = diagram.eResource();
057                                if (resource != null) {
058                                        resourceBase = resource.getURI();
059                                }
060                                
061                                if (markerBase != null && markerBase.hasAbsolutePath() && diagramURI.startsWith("./")) {
062                                        uri = uri.resolve(markerBase);
063                                } else if (resourceBase != null) {
064                                        uri = uri.resolve(resourceBase);
065                                }
066                                
067                                if (uri.isFile()) {
068                                        String fileStr = uri.toFileString();
069                                        File diagramFile = new File(fileStr);
070                                        if (diagramFile.exists()) {
071                                                // TODO - merging - read, merge models, write if result is not null.
072                                        } else {
073                                                File container = diagramFile.getParentFile();
074                                                if (container.isDirectory() || container.mkdirs()) {
075                                                        Document document = createDrawioGenerator().generateModel(diagram);
076                                                        Files.write(diagramFile.toPath(), DefaultConverter.INSTANCE.toByteArray(document.save(true)));
077                                                }
078                                        }
079                                }                                               
080                                
081                                return generateDrawioHtml(uri);
082                        }
083                        
084                        if (type.equals("drawio")) {
085                                return createDrawioGenerator().generateDiagram(diagram);
086                        }
087                } catch (NasdanikaException e) {
088                        throw e;
089                } catch (Exception e) {
090                        throw new NasdanikaException(e);
091                }
092                
093                throw new IllegalArgumentException("Unsupported diagram type: " + type);
094        }
095        
096        /**
097         * Generates draw.io HTML from diagram resource.
098         * @param diagram
099         * @return
100         * @throws IOException
101         */
102        public String generateDrawioHtml(URI diagram) {
103                try {
104                        return getDiagramGenerator().generateDrawioDiagram(DefaultConverter.INSTANCE.stringContent(diagram));
105                } catch (IOException e) {
106                        throw new NasdanikaException(e);
107                }
108        }
109        
110        protected DiagramGenerator getDiagramGenerator() {
111                return DiagramGenerator.INSTANCE;
112        }
113        
114        protected PlantumlGenerator createPlantumlGenerator() {
115                return new PlantumlGenerator(getDiagramGenerator());
116        }
117        
118        protected DrawioGenerator createDrawioGenerator() {
119                return new DrawioGenerator(getDiagramGenerator());
120        }
121
122}
