001package org.nasdanika.exec.gen;
002
003import java.util.List;
004import java.util.Objects;
005import java.util.stream.Collectors;
006
007import org.nasdanika.common.Command;
008import org.nasdanika.common.CommandFactory;
009import org.nasdanika.common.Context;
010import org.nasdanika.common.ProgressMonitor;
011import org.nasdanika.common.Reference;
012import org.nasdanika.common.SupplierFactory;
013import org.nasdanika.common.Util;
014import org.nasdanika.emf.EObjectAdaptable;
015import org.nasdanika.exec.Block;
016
017public class BlockCommandFactoryAdapter extends BlockExecutionParticipantAdapter implements CommandFactory {
018        
019        public BlockCommandFactoryAdapter(Block block) {
020                super(block);
021        }
022        
023        @Override
024        public boolean isAdapterForType(Object type) {
025                return type == SupplierFactory.class;
026        }
027                
028        private static class BlockCommand extends BlockExecutionParticipant<Command> implements Command {
029
030                private Reference<Exception> eRef;
031
032                BlockCommand(
033                                Command tryParticipant,
034                                Command catchParticipant,
035                                Command finallyParticipant,
036                                Reference<Exception> eRef) {
037                        super(tryParticipant, catchParticipant, finallyParticipant);
038                        this.eRef = eRef;
039                }
040
041                @Override
042                public void execute(ProgressMonitor progressMonitor) {
043                        try {
044                                tryParticipant.splitAndExecute(progressMonitor);
045                        } catch (Exception e) {
046                                tryParticipant.splitAndRollback(progressMonitor);
047                                eRef.set(e);
048                                if (catchParticipant == null) {
049                                        throw e;
050                                }
051                                catchParticipant.splitAndExecute(progressMonitor);
052                        } finally {
053                                if (finallyParticipant != null) {
054                                        finallyParticipant.splitAndExecute(progressMonitor);
055                                }
056                        }                       
057                }
058                
059        }       
060        
061        @Override
062        public Command create(Context context) {
063                Reference<Exception> eRef = new Reference<Exception>();
064                Block target = (Block) getTarget();
065                
066                List<CommandFactory> tryBlock = target.getTry().stream().map(e -> Objects.requireNonNull(EObjectAdaptable.adaptTo(e, CommandFactory.class), "Cannot adapt to CommandFactory: " + e)).collect(Collectors.toList());
067                Command tryCommand = Util.asCommandFactory(tryBlock).create(context);
068                
069                Command catchCommand = null;            
070                if (!target.getCatch().isEmpty()) {
071                        List<CommandFactory> catchBlock = target.getCatch().stream().map(e -> Objects.requireNonNull(EObjectAdaptable.adaptTo(e, CommandFactory.class), "Cannot adapt to CommandFactory: " + e)).collect(Collectors.toList());
072                        catchCommand = Util.asCommandFactory(catchBlock).contextify(ctx -> createCatchContextSupplier(ctx, eRef)).create(context);
073                }
074                
075                Command finallyCommand = null;
076                if (!target.getFinally().isEmpty()) {
077                        List<CommandFactory> finallyBlock = target.getFinally().stream().map(e -> Objects.requireNonNull(EObjectAdaptable.adaptTo(e, CommandFactory.class), "Cannot adapt to CommandFactory: " + e)).collect(Collectors.toList());
078                        finallyCommand = Util.asCommandFactory(finallyBlock).contextify(ctx -> createCatchContextSupplier(ctx, eRef)).create(context);
079                }               
080                
081                return new BlockCommand(tryCommand, catchCommand, finallyCommand, eRef);                
082        }
083        
084
085}
