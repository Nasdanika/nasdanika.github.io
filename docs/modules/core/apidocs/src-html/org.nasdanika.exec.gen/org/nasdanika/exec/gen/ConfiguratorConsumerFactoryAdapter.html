001package org.nasdanika.exec.gen;
002
003import java.util.Objects;
004
005import org.nasdanika.common.Consumer;
006import org.nasdanika.common.ConsumerFactory;
007import org.nasdanika.common.Context;
008import org.nasdanika.common.SupplierFactory;
009import org.nasdanika.resources.BinaryEntityContainer;
010import org.nasdanika.emf.EObjectAdaptable;
011import org.nasdanika.exec.Configurator;
012
013public class ConfiguratorConsumerFactoryAdapter extends ConfiguratorExecutionParticipantAdapter implements ConsumerFactory<BinaryEntityContainer> {
014        
015        public ConfiguratorConsumerFactoryAdapter(Configurator configurator) {
016                super(configurator);
017        }
018        
019        @Override
020        public boolean isAdapterForType(Object type) {
021                return type == SupplierFactory.class;
022        }
023        
024        @Override
025        public Consumer<BinaryEntityContainer> create(Context context) {
026                SupplierFactory<Context> contextSupplierFactory = createContextSupplierFactory();
027                Configurator configurator = (Configurator) getTarget();
028                ConsumerFactory<BinaryEntityContainer> consumerFactory = Objects.requireNonNull(EObjectAdaptable.adaptToConsumerFactory(configurator.getTarget(), BinaryEntityContainer.class), "Cannot adapt " + configurator.getTarget() + " to ConsumerFactory");            
029                if (contextSupplierFactory == null) {
030                        return consumerFactory.create(context);
031                }
032                
033                return consumerFactory.contextify(contextSupplierFactory).create(context);
034        }       
035
036}
