001package org.nasdanika.exec.gen.content;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.net.MalformedURLException;
006import java.net.URL;
007import java.util.List;
008import java.util.stream.Collectors;
009
010import org.eclipse.emf.common.notify.impl.AdapterImpl;
011import org.eclipse.emf.common.util.URI;
012import org.eclipse.emf.ecore.EObject;
013import org.nasdanika.common.BasicDiagnostic;
014import org.nasdanika.common.Context;
015import org.nasdanika.common.Diagnostic;
016import org.nasdanika.common.ExecutionException;
017import org.nasdanika.common.ProgressMonitor;
018import org.nasdanika.common.Status;
019import org.nasdanika.common.Supplier;
020import org.nasdanika.common.SupplierFactory;
021import org.nasdanika.common.Util;
022import org.nasdanika.persistence.Marked;
023import org.nasdanika.persistence.Marker;
024import org.nasdanika.emf.EObjectAdaptable;
025import org.nasdanika.exec.content.Resource;
026
027public class ResourceSupplierFactoryAdapter extends AdapterImpl implements SupplierFactory<InputStream> {
028
029        private List<? extends Marker> markers;
030        
031        public ResourceSupplierFactoryAdapter(Resource resource) {
032                setTarget(resource);
033                Marked marked = EObjectAdaptable.adaptTo((EObject) getTarget(), Marked.class);
034                markers = marked == null ? null : marked.getMarkers();
035        }
036        
037        @Override
038        public boolean isAdapterForType(Object type) {
039                return type == SupplierFactory.class;
040        }
041
042        @Override
043        public Supplier<InputStream> create(Context context) {
044                return new Supplier<InputStream>() {
045                        
046                        private URL theURL;
047                        
048                        @Override
049                        public Diagnostic diagnose(ProgressMonitor progressMonitor) {
050                                try {
051                                        Resource resource = (Resource) getTarget();
052                                        URI uri = URI.createURI(resource.isInterpolate() ? context.interpolateToString(resource.getLocation()) : resource.getLocation());
053                                        org.eclipse.emf.ecore.resource.Resource eResource = resource.eResource();
054                                        if (eResource != null) {
055                                                URI base = eResource.getURI();
056                                                if (base != null) {
057                                                        uri = uri.resolve(base);
058                                                }
059                                        }
060                                        
061                                        String iUrl = uri.toString(); 
062                                        
063                                        theURL = iUrl.startsWith(Util.CLASSPATH_URL_PREFIX) ? getClass().getClassLoader().getResource(iUrl.substring(Util.CLASSPATH_URL_PREFIX.length())) : new URL(iUrl);
064                                        if (theURL == null) {
065                                                return new BasicDiagnostic(Status.ERROR, "Classpath resource not found: " + iUrl, markers);                                             
066                                        }
067                                        return Supplier.super.diagnose(progressMonitor);
068                                } catch (MalformedURLException e) {                                     
069                                        return new BasicDiagnostic(Status.ERROR, e.getMessage() + (markers == null ? "" : " at " + markers.stream().map(Object::toString).collect(Collectors.joining(", "))));
070                                }
071                        }
072
073                        @Override
074                        public double size() {
075                                return 1;
076                        }
077
078                        @Override
079                        public String name() {
080                                return "Resource: " + markers == null ? "(undefined)" : markers.stream().map(Object::toString).collect(Collectors.joining(", "));
081                        }
082
083                        @Override
084                        public InputStream execute(ProgressMonitor progressMonitor) {
085                                try {
086                                        return theURL.openStream();
087                                } catch (IOException e) {
088                                        throw new ExecutionException(e, this);
089                                }
090                        }
091                        
092                };
093        }
094
095}
