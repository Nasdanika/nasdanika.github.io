001package org.nasdanika.persistence;
002
003import java.util.List;
004import java.util.Map;
005
006import org.eclipse.emf.common.util.URI;
007import org.nasdanika.common.ProgressMonitor;
008
009public class Attribute<T> implements Feature<T>, ObjectFactory<T> {
010        
011        private List<? extends Marker> markers;
012        private T value;
013        private boolean loaded;
014        private Object key;
015        private boolean required;
016        private Object[] exclusiveWith;
017        private boolean isDefault;
018        private String description;
019        
020        public Attribute(
021                        Object key, 
022                        boolean isDefault, 
023                        boolean required, 
024                        T defaultValue, 
025                        String description, 
026                        Object... exclusiveWith) {
027                this.key = key;
028                this.isDefault = isDefault;
029                this.required = required;
030                this.value = defaultValue;
031                this.description = description;
032                this.exclusiveWith = exclusiveWith;             
033        }
034                        
035        public Object getKey() {
036                return key;
037        }
038        
039        public T getValue() {
040                return value; 
041        }
042        
043        public boolean isLoaded() {
044                return loaded;
045        }
046        
047        @Override
048        public List<? extends Marker> getMarkers() {
049                return markers;
050        }
051
052        @Override
053        public void load(ObjectLoader loader, Map<?,?> source, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) {
054                if (source.containsKey(getKey())) {
055                        for (Object ek: exclusiveWith) {
056                                if (source.containsKey(ek)) {
057                                        throw new ConfigurationException("Features '" + getKey() + " and " + ek + " are mutually exclusive", markers);                                  
058                                }
059                        }
060                        this.markers = Util.getMarkers(source, getKey());
061                        value = create(loader, source.get(getKey()), base, progressMonitor, this.markers);                                              
062                        loaded = true;
063                } else if (isRequired()) {
064                        throw new ConfigurationException("Missing required feature: " + getKey(), markers);
065                }               
066        }       
067        
068        /**
069         * This implementation casts config to the target type. It can be used where not special conversion is required, e.g. config is a String and feature type is String.
070         * @param loader
071         * @param config
072         * @param base
073         * @param progressMonitor
074         * @param marker
075         * @return
076         */
077        @SuppressWarnings("unchecked")
078        @Override
079        public T create(ObjectLoader loader, Object config, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) {
080                return (T) config;
081        }
082
083        public Object[] getExclusiveWith() {
084                return exclusiveWith;
085        }
086        
087        public boolean isRequired() {
088                return required;
089        }
090        
091        @Override
092        public boolean isDefault() {
093                return isDefault;
094        }
095
096        @Override
097        public String getDescription() {
098                return description;
099        }
100}
