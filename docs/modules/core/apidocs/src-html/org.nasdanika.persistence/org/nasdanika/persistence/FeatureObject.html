001package org.nasdanika.persistence;
002
003import java.util.ArrayList;
004import java.util.Collections;
005import java.util.List;
006import java.util.Map;
007import java.util.Optional;
008import java.util.stream.Collectors;
009
010import org.eclipse.emf.common.util.URI;
011import org.nasdanika.common.ProgressMonitor;
012
013/**
014 * Object which loads its configuration to features identified by keys.
015 * @author Pavel
016 *
017 */
018public class FeatureObject implements Marked, Loadable {
019        
020        private List<? extends Marker> markers = new ArrayList<>();
021        
022        @Override
023        public List<? extends Marker> getMarkers() {
024                return markers;
025        }
026        
027        protected List<Feature<?>> features = new ArrayList<>();
028
029        /**
030         * Adds feature to the feature list.
031         * @param <F>
032         * @param feature
033         * @return
034         */
035        protected <F extends Feature<?>> F addFeature(F feature) {
036                if (feature.isDefault() && features.stream().filter(Feature::isDefault).findFirst().isPresent()) {
037                        throw new IllegalArgumentException("Duplicate default feature: " + feature.getKey());
038                }
039                if (features.stream().filter(o -> feature.getKey().equals(o.getKey())).findFirst().isPresent()) {
040                        throw new IllegalArgumentException("Duplicate feature: " + feature.getKey());
041                }
042                features.add(feature);
043                return feature;
044        }
045
046        /**
047         * Loads object
048         * @param loader
049         * @param config
050         * @param base
051         * @param progressMonitor
052         * @param marker
053         * @return this object.
054         */
055        @Override
056        public void load(ObjectLoader loader, Object config, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) {
057                this.markers = markers;                 
058                if (config instanceof Map) {
059                        Map<?,?> configMap = (Map<?,?>) config;
060                        Util.checkUnsupportedKeys(configMap, features.stream().map(Feature::getKey).collect(Collectors.toList()));
061                        for (Feature<?> feature: features) {
062                                feature.load(loader, configMap, base, progressMonitor, markers);
063                        }
064                } else {
065                        Optional<Feature<?>> defaultFeatureOptional = features.stream().filter(Feature::isDefault).findFirst();
066                        if (defaultFeatureOptional.isPresent()) {                               
067                                Feature<?> defaultFeature = defaultFeatureOptional.get();
068                                // Singleton map with a single entry default feature key -> config.
069                                defaultFeature.load(loader, Collections.singletonMap(defaultFeature.getKey(), config), base, progressMonitor, markers);
070                        } else if (!Boolean.TRUE.equals(config)) {
071                                throw new ConfigurationException(getClass().getName() + " configuration shall be a map, got " + config, markers);
072                        }
073                }
074        }                       
075
076}
