001package org.nasdanika.persistence;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.io.Reader;
006import java.net.URL;
007import java.util.List;
008import java.util.Map;
009
010import org.eclipse.emf.common.util.URI;
011import org.nasdanika.common.ConverterMethod;
012import org.nasdanika.common.DefaultConverter;
013
014/**
015 * Reflective converter providing some common conversions, some of them chained
016 * @author Pavel
017 *
018 */
019public class Converter extends DefaultConverter {
020        
021        public static Converter INSTANCE = new Converter();
022        
023        /**
024         * Parses string as YAML.
025         * @param string
026         * @return
027         */
028        @SuppressWarnings("unchecked")
029        @ConverterMethod
030        public Map<String, Object> toYamlMap(String string) {
031                Object data = MarkingYamlConstructor.createMarkingYaml(null).load(string);
032                return data instanceof Map ? (Map<String, Object>) data : null;
033        }
034        
035        @SuppressWarnings("unchecked")
036        @ConverterMethod
037        public Map<String, Object> toYamlMap(Reader reader) {
038                Object data = MarkingYamlConstructor.createMarkingYaml(null).load(reader);
039                return data instanceof Map ? (Map<String, Object>) data : null;         
040        }
041                
042        @SuppressWarnings("unchecked")
043        @ConverterMethod
044        public Map<String, Object> toYamlMap(InputStream inputStream) {
045                Object data = MarkingYamlConstructor.createMarkingYaml(null).load(inputStream);
046                return data instanceof Map ? (Map<String, Object>) data : null;         
047        }
048        
049        @SuppressWarnings("unchecked")
050        @ConverterMethod
051        public Map<String, Object> toYamlMap(URL url) throws IOException {
052                Object data = MarkingYamlConstructor.createMarkingYaml(url.toString()).load(url.openStream());
053                return data instanceof Map ? (Map<String, Object>) data : null;         
054        }
055        
056        @SuppressWarnings("unchecked")
057        @ConverterMethod
058        public Map<String, Object> toYamlMap(URI uri) throws IOException {
059                Object data = MarkingYamlConstructor.createMarkingYaml(uri.toString()).load(toInputStream(uri));
060                return data instanceof Map ? (Map<String, Object>) data : null;         
061        }
062
063        /**
064         * Parses string as YAML.
065         * @param string
066         * @return
067         */
068        @SuppressWarnings("unchecked")
069        @ConverterMethod
070        public List<Object> toYamlList(String string) {
071                Object data = MarkingYamlConstructor.createMarkingYaml(null).load(string);
072                return data instanceof List ? (List<Object>) data : null;
073        }
074        
075        @SuppressWarnings("unchecked")
076        @ConverterMethod
077        public List<Object> toYamlList(Reader reader) {
078                Object data = MarkingYamlConstructor.createMarkingYaml(null).load(reader);
079                return data instanceof List ? (List<Object>) data : null;
080        }
081                
082        @SuppressWarnings("unchecked")
083        @ConverterMethod
084        public List<Object> toYamlList(InputStream inputStream) {
085                Object data = MarkingYamlConstructor.createMarkingYaml(null).load(inputStream);
086                return data instanceof List ? (List<Object>) data : null;
087        }
088        
089        @SuppressWarnings("unchecked")
090        @ConverterMethod
091        public List<Object> toYamlList(URL url) throws IOException {
092                Object data = MarkingYamlConstructor.createMarkingYaml(url.toString()).load(url.openStream());
093                return data instanceof List ? (List<Object>) data : null;
094        }
095        
096        @SuppressWarnings("unchecked")
097        @ConverterMethod
098        public List<Object> toYamlList(URI uri) throws IOException {
099                Object data = MarkingYamlConstructor.createMarkingYaml(uri.toString()).load(toInputStream(uri));
100                return data instanceof List ? (List<Object>) data : null;
101        }
102        
103}
