001package org.nasdanika.persistence;
002
003import java.util.Collections;
004import java.util.List;
005import java.util.Stack;
006import java.util.stream.Collectors;
007
008import org.nasdanika.common.NasdanikaException;
009
010/**
011 * Exception to indicate error in configuration resources.
012 * @author Pavel
013 *
014 */
015@SuppressWarnings("serial")
016public class ConfigurationException extends NasdanikaException {
017        
018        private List<? extends Marker> markers;
019        
020        public ConfigurationException(String message, Marker marker) {
021                this(message, marker == null ? null : Collections.singletonList(marker));
022        }
023
024        public ConfigurationException(String message, Throwable cause, Marker marker) {
025                this(message, cause, marker == null ? null : Collections.singletonList(marker));
026        }       
027
028        public ConfigurationException(String message, List<? extends Marker> markers) {
029                super(markers == null || markers.isEmpty() ? message : message + " at " + markers.stream().map(m -> Marker.toString(m)).collect(Collectors.joining(", ")));
030                this.markers = markers;
031        }
032
033        public ConfigurationException(String message, Throwable cause, List<? extends Marker> markers) {
034                super(markers == null || markers.isEmpty() ? message : message + " at " + markers.stream().map(m -> Marker.toString(m)).collect(Collectors.joining(", ")));
035                this.markers = markers;
036        }
037
038        public ConfigurationException(String message, Marked marked) {
039                this(message, marked == null ? null : marked.getMarkers());
040        }
041
042        public ConfigurationException(String message, Throwable cause, Marked marked) {
043                this(message, cause, marked == null ? null : marked.getMarkers());
044        }
045        
046        public List<? extends Marker> getMarkers() {
047                return markers;
048        }
049        
050        private static ThreadLocal<Stack<List<? extends Marker>>> threadMarkerStack = new ThreadLocal<Stack<List<? extends Marker>>>() {
051
052                protected java.util.Stack<List<? extends Marker>> initialValue() {
053                        return new Stack<>();
054                };
055                
056        };
057        
058        public static void pushThreadMarker(List<? extends Marker> markers) {
059                threadMarkerStack.get().push(markers);
060        }
061        
062        public static void popThreadMarker() {
063                threadMarkerStack.get().pop();
064        }       
065        
066        public static void pushThreadMarked(Marked marked) {
067                pushThreadMarker(marked == null ? null : marked.getMarkers());
068        }
069        
070        public static List<? extends Marker> peekThreadMarker() {
071                Stack<List<? extends Marker>> stack = threadMarkerStack.get();
072                return stack.isEmpty() ? null : stack.peek(); 
073        }
074
075        public ConfigurationException(String message) {
076                this(message, peekThreadMarker());
077        }
078
079        public ConfigurationException(String message, Throwable cause) {
080                this(message, cause, peekThreadMarker());
081        }
082
083}
