001package org.nasdanika.persistence;
002
003import java.util.LinkedHashMap;
004import java.util.List;
005import java.util.Map;
006import java.util.Map.Entry;
007
008import org.eclipse.emf.common.util.URI;
009import org.nasdanika.common.ProgressMonitor;
010
011/**
012 * If config is a map loads each entry value using createValue().
013 * @author Pavel
014 *
015 * @param <V>
016 */
017public class MapAttribute<K, V> extends Attribute<Map<K, V>> {
018
019        public MapAttribute(
020                        Object key, 
021                        boolean isDefault,
022                        boolean required, 
023                        Map<K,V> defaultValue, 
024                        String description, 
025                        Object... exclusiveWith) {
026                super(key, isDefault, required, defaultValue, description, exclusiveWith);
027        }
028        
029        @Override
030        public Map<K,V> create(ObjectLoader loader, Object config, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) {
031                if (config instanceof Map) {
032                        Map<K,V> ret = new LinkedHashMap<>();
033                        for (Entry<?, ?> e: ((Map<?,?>) config).entrySet()) {
034                                K key = createKey(e.getKey());
035                                ret.put(key, createValue(loader, key, e.getValue(), base, progressMonitor, Util.getMarkers((Map<?,?>) config, e.getKey())));
036                        }
037                        return ret;
038                }
039                
040                throw new ConfigurationException("Config should be a map: " + config, markers);
041        }
042        
043        @SuppressWarnings("unchecked")
044        protected K createKey(Object key) {
045                return (K) key;
046        }
047
048        /**
049         * Creates entry value.
050         * @param loader
051         * @param value
052         * @param base
053         * @param progressMonitor
054         * @param marker
055         * @return
056         */
057        @SuppressWarnings("unchecked")
058        protected V createValue(ObjectLoader loader, K key, Object value, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) { 
059                return (V) value; 
060        }
061        
062}
