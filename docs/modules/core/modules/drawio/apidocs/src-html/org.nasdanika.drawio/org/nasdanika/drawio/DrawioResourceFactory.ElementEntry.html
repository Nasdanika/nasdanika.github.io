023/**
024 * Base class for factories which load {@link EObject}'s from Drawio diagrams by performing semantic mapping from diagram elements to model elements. 
025 * @author Pavel
026 *
027 */
028public abstract class DrawioResourceFactory<T> extends ResourceFactoryImpl {

032                /**
033                 * Semantic element created from a given Drawio element. Can be null
034                 * @return

038                Map<Element, ElementEntry<T>> getChildEntries();

040        }

057        @Override
058        public Resource createResource(URI uri) {
059                return new DrawioResource<T>(uri, connectionBase) {
060
061                        @Override
062                        protected ElementEntry<T> createEntry(Element element, Map<Element, ElementEntry<T>> childMappings) {
063                                return DrawioResourceFactory.this.createEntry(this, element, childMappings);
064                        }
065
066                        @Override
067                        protected void resolve(
068                                        Element element, 
069                                        T semanticElement, 
070                                        Map<Element, ElementEntry<T>> childEntries,
071                                        Function<Predicate<Element>, T> resolver) {

073                                DrawioResourceFactory.this.resolve(this, element, semanticElement, childEntries, resolver);                             
074                        }

076                };
077        }

079        /**
080         * Override to add child mappings from a linked page (if any).
081         * @param modelElement
082         * @return
083         */
084        protected Map<Element, ElementEntry<T>> getLinkedPageChildMappings(Resource resource, Page linkedPage, ElementEntry<T> linkedPageEntry) {
085                return Collections.emptyMap();
086        }
087        
088        /**
089         * @param parent
090         * @return Child element comparator for a specific given parent. This implementation orders {@link Layer}s top to bottom 
091         * and {@link ModelElement} children using {@link ElementComparator.Factory} service with a name matching sort property value before colon, if it is present. 
092         * If there is no sort property then elements are sorted by label text by default using "label" comparator.
093         */
094        protected Comparator<Element> getChildComparator(Element parent) {
095                if (parent instanceof Root) {
096                        Comparator<Element> ret = loadChildComparator((ModelElement) parent, null);
097                        if (ret != null) {
098                                return ret;
099                        }
100                        
101                        return new Comparator<Element>() {
102                                
103                                @Override
104                                public int compare(Element l1, Element l2) {
105                                        if (Objects.equals(l1, l2)) {
106                                                return 0;
107                                        }
108                                        return ((Root) parent).getLayers().indexOf(l2) - ((Root) parent).getLayers().indexOf(l1); // Reverse order 
109                                }
110                        };
111                        
112                }
113                
114                String defaultSort = getDefaultSort();
115                if (parent instanceof ModelElement) {
116                        return loadChildComparator((ModelElement) parent, defaultSort);
117                }
118                
119                for (ElementComparator.Factory comparatorFactory: ServiceLoader.load(ElementComparator.Factory.class)) {
120                        if (comparatorFactory.isForType(defaultSort)) {
121                                return comparatorFactory.create(defaultSort, null, parent);
122                        }
123                }                                       
124                
125                return null;
126        }
127        
128        protected String getDefaultSort() {
129                return "label";
130        }
131                
132        /**
133         * Semantic parent is used for resolution of documentation.
134         * Element documentation URI is resolved relative to semantic ancestor documentation URI if it is set and relative to the
135         * document otherwise.
136         * @param element
137         * @return
138         */
139        protected ModelElement getSemanticParent(ModelElement element, Set<Connection> traversed) {
140                if (element instanceof Connection) {
141                        if (connectionBase == ConnectionBase.SOURCE) {
142                                return ((Connection) element).getSource();
143                        }
144                        if (connectionBase == ConnectionBase.TARGET) {
145                                return ((Connection) element).getTarget();
146                        }
147                } 
148                return element.getParent();
149        }
150        
151        protected BiSupplier<String, String> getComparatorConfig(ModelElement parent, String defaultSort) {
152                String sortProperty = getSortProperty(parent);
153                String sort = null;
154                if (!org.nasdanika.common.Util.isBlank(sortProperty)) {
155                        sort = ((ModelElement) parent).getProperty(sortProperty);
156                }
157                if (org.nasdanika.common.Util.isBlank(sort)) {
158                        ModelElement semanticParent = getSemanticParent(parent, new HashSet<>());
159                        if (semanticParent != null) {
160                                return getComparatorConfig(semanticParent, defaultSort);
161                        }
162                        sort = defaultSort;
163                }
164                
165                if (org.nasdanika.common.Util.isBlank(sort) || "none".equals(sort)) {
166                        return null;
167                }
168                
169                String config = null;
170                int colonIdx = sort.indexOf(":");
171                if (colonIdx != -1) {
172                        config = sort.substring(colonIdx + 1);
173                        sort = sort.substring(0, colonIdx);
174                }
175                
176                return BiSupplier.of(sort, config);
177        }
178        
179
180        protected Comparator<Element> loadChildComparator(ModelElement parent, String defaultSort) {
181                BiSupplier<String, String> config = getComparatorConfig(parent, defaultSort);
182                if (config == null) {
183                        return null;
184                }
185                String sortConfig = config.getSecond();
186                if (sortConfig == null) {
187                        sortConfig = getDefaultSortConfig(parent, config.getFirst());
188                }
189                
190                for (ElementComparator.Factory comparatorFactory: ServiceLoader.load(ElementComparator.Factory.class)) {
191                        if (comparatorFactory.isForType(config.getFirst())) {
192                                return comparatorFactory.create(config.getFirst(), sortConfig, parent);
193                        }
194                }                       
195                throw new IllegalArgumentException("Unsupported sort type: " + config.getFirst());
196        }
197        
198        /**
199         * @param parent
200         * @param sort
201         * @return Default configuration for a sort type. For example, for a agle sort (clockwise/counterclockwise) the default configuration can the an angle from the semantic parent of this model element.  
202         */
203        protected String getDefaultSortConfig(ModelElement modelElement, String sort) {
204                if (modelElement instanceof Node && (AngularModelElementComparatorFactory.CLOCKWISE.equals(sort) || AngularModelElementComparatorFactory.COUNTERCLOCKWISE.equals(sort))) {
205                        ModelElement semanticParent = getSemanticParent(modelElement, new HashSet<>());
206                        if (semanticParent instanceof Node) {
207                                return String.valueOf(Math.toDegrees(AngularModelElementComparatorFactory.angle((Node) modelElement, (Node) semanticParent)));
208                        }
209                }
210                return null;
211        }
212
213        protected String getSortProperty(Element parent) {
214                return "sort";
215        }

218         * Create an element, add top level elements to resource content.

237                        Resource resource, 
238                        Element element, 
239                        T semanticElement, 
240                        Map<Element,ElementEntry<T>> childEntries, 
241                        Function<Predicate<Element>, T> resolver);

243        /**
244         * Creates an {@link ElementAdapter}.
245         * @param element
246         * @return
247         */
248        protected ElementAdapter createElementAdapter(Element element) {
249                class ElementAdapterImpl extends AdapterImpl implements ElementAdapter {
250                        
251                        @Override
252                        public Element getElement() {
253                                return element;
254                        }
255                        
256                        @Override
257                        public boolean isAdapterForType(Object type) {
258                                return ElementAdapter.class == type;
259                        }
260
261                }
262
263                return new ElementAdapterImpl();
264        }
265
266        protected ElementEntry<T> createEntry(Resource resource, Element element, Map<Element, ElementEntry<T>> childMappings) {
267                if (childMappings != null && !childMappings.isEmpty()) {
268                        Comparator<Element> childComparator = getChildComparator(element);
269                        if (childComparator != null) {
270                                Map<Element, ElementEntry<T>> orderedChildMappings = new LinkedHashMap<>();
271                                childMappings.entrySet().stream().sorted((a, b) -> childComparator.compare(a.getKey(), b.getKey())).forEachOrdered(e -> orderedChildMappings.put(e.getKey(), e.getValue()));
272                                childMappings = orderedChildMappings;
273                        }
274                }
275                
276                if (element instanceof ModelElement) {
277                        Page linkedPage = ((ModelElement) element).getLinkedPage();
278                        if (linkedPage != null) {
279                                @SuppressWarnings("unchecked")
280                                DrawioResource<T> drawioResource = (DrawioResource<T>) resource;
281                                ElementEntry<T> linkedPageMapping = linkedPage.accept(drawioResource::createEntry, connectionBase);
282                                Map<Element, ElementEntry<T>> linkedPageChildMappings = getLinkedPageChildMappings(resource, linkedPage, linkedPageMapping);
283                                if (linkedPageChildMappings != null && !linkedPageChildMappings.isEmpty()) {
284                                        childMappings = childMappings == null ? new LinkedHashMap<>() : new LinkedHashMap<>(childMappings);
285                                        childMappings.putAll(linkedPageChildMappings);
286                                }                               
287                        }
288                }
289
290                Map<Element, ElementEntry<T>> theChildMappings = childMappings;
291                T semanticElement = createSemanticElement(resource, element, theChildMappings);
292                return new ElementEntry<T>() {
293                        
294                        public T getSemanticElement() {
295                                return semanticElement;
296                        }
297                        
298                        public Map<Element, ElementEntry<T>> getChildEntries() {
299                                return theChildMappings;
300                        }
301                        
302                };
303        }
