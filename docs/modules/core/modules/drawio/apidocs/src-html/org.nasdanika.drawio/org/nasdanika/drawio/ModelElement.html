001package org.nasdanika.drawio;
002
003import java.util.Map;
004import java.util.Set;
005import java.util.function.BiFunction;
006import java.util.function.Function;
007
008import org.eclipse.emf.common.util.URI;
009
010public interface ModelElement extends Element {
011        
012        Model getModel();
013        
014        ModelElement getParent();
015        
016        String getLabel();
017        
018        void setLabel(String label);
019
020        String getLink();
021        
022        void setLink(String link);
023        
024        String getTooltip();
025        
026        void setTooltip(String tooltip);
027        
028        /**
029         * @return style property represented as a map
030         */
031        Map<String,String> getStyle();
032        
033        String getProperty(String name);
034        
035        void setProperty(String name, String value);
036        
037        Set<String> getTags();
038        
039        boolean isVisible();
040        
041        void setVisible(boolean visible);
042        
043        String getId();
044        
045        /**
046         * Resolves some value for the model element by traversing its logical containment hierarchy and resolving that value against logical parents.
047         * For the containment root the value is resolved against the base.
048         * @param <T>
049         * @param base Base value is used to resolve the parent value
050         * @param resolver Resolver {@link BiFunction}
051         * @param connectionBase Indicates logical "mounting" of connections.
052         * @return
053         */
054        <T> T resolve(T base, BiFunction<? super ModelElement,T,T> resolver, ConnectionBase connectionBase);
055        
056        /**
057         * Resolves URI
058         * @param base Base URI for the root object.
059         * @param uriProvider Extracts URI from the model element.
060         * @param connectionBase Connection base (logical parent)
061         * @return
062         */
063        default URI resolveURI(URI base, Function<? super ModelElement,URI> uriProvider, ConnectionBase connectionBase) {
064                return resolve(base, (modelElement, parentURI) -> {                     
065                        URI uri = uriProvider.apply(modelElement);
066                        if (uri == null) {
067                                return parentURI;
068                        }
069                        if (parentURI == null) {
070                                return uri;
071                        }
072                        
073                        return uri.resolve(parentURI);                  
074                }, connectionBase);
075        }
076        
077        Page getLinkedPage();
078        
079        /**
080         * @return True if there is a link and it is in one of page link formats.
081         */
082        boolean isPageLink();
083        
084}
