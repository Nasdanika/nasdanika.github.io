003/**
004 * Base (logical parent) for connections to use, for example, to resolve URI's and also during visiting.
005 * @author Pavel
006 *
007 */
008public enum ConnectionBase {
009        
010        /**
011         * Connection is a logical child of its parent 
012         */
013        PARENT,
014        
015        /**
016         * Connection is a logical child of its source
017         */
018        SOURCE,
019        
020        /**
021         * Connection is a logical child of its target
022         */
023        TARGET,
024
025}
