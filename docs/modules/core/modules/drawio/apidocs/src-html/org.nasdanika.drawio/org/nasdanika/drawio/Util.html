028        /**
029         * Lays out the diagram so nodes do not overlap.
030         * @param root
031         * @param gridSize
032         */
033        public static void layout(Root root, int gridSize) {
034                List<Node> topLevelNodes = root.getLayers().stream().flatMap(layer -> layer.getElements().stream()).filter(Node.class::isInstance).map(Node.class::cast).collect(Collectors.toList());
035                layout(topLevelNodes, new Point(gridSize, gridSize), down -> createOffsetGenerator(gridSize, down));
036        }

038        /**
039         * Computes sizes and positions of a collection of elements so they don't overlap
040         * @param elements
041         * @return
042         */
043        public static Map<Node, Rectangle> layout(
044                        Collection<Node> nodes, 
045                        Point offset,
046                        Function<Boolean, Supplier<Point>> offsetGeneratorProvider /*,                  
047                        BiConsumer<Node, Rectangle> geometryConsumer */) {
048                
049                Map<Node, Rectangle> ret = new HashMap<>();
050                for (Node node: nodes) {
051                        ret.put(node, createElementRectangle(node, offset, offsetGeneratorProvider /*, geometryConsumer */));
052                }
053                
054                // Sort
055                List<Map.Entry<Node, Rectangle>> remaining = new ArrayList<>(ret.entrySet());
056                List<Map.Entry<Node, Rectangle>> sorted = new ArrayList<>();
057                Map<Rectangle,Boolean> downMap = new IdentityHashMap<>();
058                while (!remaining.isEmpty()) {
059                        Entry<Node, Rectangle> candidate = null;
060                        int candidateAffinity = 0;
061                        boolean isCandidateDown = false;
062                        for (Entry<Node, Rectangle> e: remaining) {
063                                Collection<Map.Entry<Node, Rectangle>> els;
064                                if (sorted.isEmpty()) {
065                                        els = new ArrayList<>(remaining);
066                                        els.remove(e);
067                                } else {
068                                        els = sorted;
069                                }
070                                int[] eAffinity = affinity(e.getKey(), els.stream().map(Map.Entry::getKey).collect(Collectors.toList()));
071                                int teAffinity = eAffinity[0] + eAffinity[1];
072                                if (candidate == null || (candidateAffinity < teAffinity)) {
073                                        candidate = e;
074                                        candidateAffinity = teAffinity;
075                                        isCandidateDown = eAffinity[1] > eAffinity[0];
076                                }               
077                        }
078                        sorted.add(candidate);
079                        downMap.put(candidate.getValue(), isCandidateDown);
080                        remaining.remove(candidate);
081                }                               
082                
083                position(sorted, new HashMap<>(), offset, offsetGeneratorProvider, downMap::get);
084                
085                double minX = Integer.MAX_VALUE;
086                double minY = Integer.MAX_VALUE;
087                for (Rectangle rectangle: ret.values()) {                       
088                        minX = Math.min(rectangle.getX(), minX);
089                        minY = Math.min(rectangle.getY(), minY);
090                }
091                
092                for (Rectangle rectangle: ret.values()) {
093                        rectangle.translate((int) (-minX + offset.getX()), (int) (-minY + offset.getY()));
094                }       
095                
096                for (Entry<Node, Rectangle> e: ret.entrySet()) {
097                        Rectangle rectangle = e.getValue();
098                        e.getKey().getGeometry().setBounds(rectangle.x, rectangle.y, rectangle.width, rectangle.height);
099                }       
100                
101                return ret;
102        }

104        private static boolean isSameOrAncestor(ModelElement child, ModelElement ancestor) {
105                if (child == null) {
106                        return false;
107                }
108                if (child.equals(ancestor)) {
109                        return true;
110                }
111                return isSameOrAncestor(child.getParent(), ancestor);
112        }

114        /**
115         * Computes affinity of a given diagram element to a collection of other elements.
116         * Affinity is a two-elements array with the first element containing a total number of 
117         * connections from the first argument element to the second argument elements recursively including child elements on both sides.
118         * The second array element contains a total number of connections from the second argument elements to the first argument element also recursively
119         * including child elements on both sides. Affinity is used to layout elements in a top-down   
120         * @param element
121         * @param elements
122         * @return
123         */
124        private static int[] affinity(ModelElement element, Collection<ModelElement> elements) {
125                int[] outbound = { 0 };
126                element.accept(e -> {
127                        if (e instanceof Connection) {
128                                Node target = ((Connection) e).getTarget();
129                                for (ModelElement oe: elements) {
130                                        if (isSameOrAncestor(target, oe)) {
131                                                ++outbound[0];
132                                        }
133                                }
134                        }
135                }, ConnectionBase.SOURCE);
136                
137                int[] inbound = { 0 };
138                element.accept(e -> {
139                        if (e instanceof Connection) {
140                                Node source = ((Connection) e).getSource();
141                                for (ModelElement oe: elements) {
142                                        if (isSameOrAncestor(source, oe)) {
143                                                ++inbound[0];
144                                        }
145                                }
146                        }
147                }, ConnectionBase.TARGET);
148                
149                return new int[] { outbound[0], inbound[0] };
150        }

152        /**
153         * Sorts rectangles entry set by size and affinity to positioned offset rectangles.
154         * Then positions the first element so it doesn't overlap with the offset rectangles. 
155         * Then passes the remaining elements for positioning.
156         * @param rectangles
157         * @param offsetRectangles
158         */
159        private static void position(
160                        List<Map.Entry<Node, Rectangle>> rectangles, 
161                        Map<Node, Rectangle> offsetRectangles,
162                        Point offset,
163                        Function<Boolean, Supplier<Point>> offsetGeneratorProvider,
164                        Predicate<Rectangle> downPredicate) {
165                if (rectangles.isEmpty()) {
166                        return;
167                }
168                
169                Entry<Node, Rectangle> firstEntry = rectangles.get(0);
170                Rectangle firstRectangle = firstEntry.getValue();
171                position(firstRectangle, offsetRectangles.values(), offsetGeneratorProvider, downPredicate.test(firstRectangle)); 
172                Rectangle offsetRectangle = new Rectangle(firstRectangle);
173                offsetRectangle.translate((int) -offset.getX(), (int) - offset.getY());
174                offsetRectangle.grow( 2 * (int) offset.getX(), 2 * (int) offset.getY());
175                offsetRectangles.put(firstEntry.getKey(), offsetRectangle);                     
176                
177                position(rectangles.subList(1, rectangles.size()), offsetRectangles, offset, offsetGeneratorProvider, downPredicate);
178        }

180        /**
181         * Positions the rectangle so it doesn't overlap with existing rectangles.
182         * @param element
183         * @param geometry
184         * @param rectangle
185         * @param rectangles
186         */
187        private static void position(
188                        Rectangle rectangle, 
189                        Collection<Rectangle> offsetRectangles,
190                        Function<Boolean, Supplier<Point>> offsetGeneratorProvider,
191                        boolean down) {
192                
193                if (offsetRectangles.isEmpty()) {
194                        return;
195                }
196                
197                // Start at the center of offset rectangles union
198                Rectangle union = null;
199                for (Rectangle offsetRectangle: offsetRectangles) {
200                        if (union == null) {
201                                union = new Rectangle(offsetRectangle);
202                        } else {
203                                union.add(offsetRectangle); 
204                        }
205                }
206                double unionCenter = (union.getX() + union.getWidth())/2;
207                double unionMiddle = (union.getY() + union.getHeight())/2;
208
209                double rectangleCenter = (rectangle.getX() + rectangle.getWidth())/2;
210                double rectangleMiddle = (rectangle.getY() + rectangle.getHeight())/2;
211                
212                rectangle.translate((int) (unionCenter - rectangleCenter), (int) (unionMiddle - rectangleMiddle));
213        
214                Supplier<Point> offsetGenerator = offsetGeneratorProvider.apply(down);
215                Z: for (Point offset = offsetGenerator.get(); offset != null; offset = offsetGenerator.get()) {
216                        rectangle.translate((int) offset.getX(), (int) offset.getY());
217                        for (Rectangle offsetRectangle: offsetRectangles) {
218                                if (!rectangle.intersection(offsetRectangle).isEmpty()) {
219                                        continue Z;
220                                }
221                        }
222                        break; // No intersection.
223                }
224        }

226        /**
227         * @param gridSize difference x, y, or both, between offset points in pixels,
228         * @param down Direction of traversing.  
229         * @return generator of a sequence of points which are offset from the initial point for positioning of a rectangle. 
230         * This implementation returns a sequence of offsets which form concentric half-circles up or down.
231         */
232        public static Supplier<Point> createOffsetGenerator(int gridSize, boolean down) {
233                return new Supplier<Point>() {
234
235                        private Point lastPosition = new Point(0, 0);
236                        private Set<Point> visited = new HashSet<>();
237
238                        @Override
239                        public Point get() {
240                                if (visited.add(lastPosition)) {
241                                        return lastPosition;
242                                }
243                                List<Point> newPositions = new ArrayList<>();                                   
244                                newPositions.add(new Point(lastPosition.x + gridSize, lastPosition.y));                                 
245                                newPositions.add(new Point(lastPosition.x, lastPosition.y + gridSize));                                 
246                                newPositions.add(new Point(lastPosition.x + gridSize, lastPosition.y + gridSize));                                      
247                                newPositions.add(new Point(lastPosition.x + gridSize, lastPosition.y - gridSize));                                      
248                                newPositions.add(new Point(lastPosition.x - gridSize, lastPosition.y));                                 
249                                newPositions.add(new Point(lastPosition.x, lastPosition.y - gridSize));                                 
250                                newPositions.add(new Point(lastPosition.x - gridSize, lastPosition.y - gridSize));
251                                newPositions.add(new Point(lastPosition.x - gridSize, lastPosition.y + gridSize));
252                                newPositions.sort((a, b) -> a.x*a.x + a.y*a.y - b.x*b.x - b.y*b.y);
253                                for (Point newPosition: newPositions) {
254                                        if ((down && newPosition.y >= 0 || !down && newPosition.y <=0) && visited.add(newPosition)) {
255                                                Point offset = new Point(newPosition.x - lastPosition.x, newPosition.y - lastPosition.y);
256                                                lastPosition = newPosition;
257                                                return offset;
258                                        }
259                                }
260                                throw new IllegalStateException("Nowhere to move");
261                        }
262                        
263                };
264        }

266        /**
267         * Creates a rectangle for the diagram element and its children at position 0, 0.
268         * @param element
269         * @param offset
270         * @return
271         */
272        private static Rectangle createElementRectangle(
273                        Node node, 
274                        Point offset, 
275                        Function<Boolean, Supplier<Point>> offsetGeneratorProvider) {
276                
277                org.nasdanika.drawio.Rectangle nodeGeometry = node.getGeometry();
278                Rectangle rectangle = new Rectangle(0, 0, (int) nodeGeometry.getWidth(), (int) nodeGeometry.getHeight());
279                List<Node> children = node.getElements().stream().filter(Node.class::isInstance).map(Node.class::cast).collect(Collectors.toList());
280                if (!children.isEmpty()) {
281                        Point childOffset = new Point((int) offset.getX(), (int) offset.getY() + 30);
282                        Map<Node, Rectangle> childGeometry = layout(children, childOffset, offsetGeneratorProvider);
283                        for (ModelElement child: children) {
284                                Rectangle childRectangle = childGeometry.get(child);
285                                rectangle.add(childRectangle);
286                        }
287                        rectangle.grow((int) offset.getX(), (int) offset.getY());
288                }
289                return rectangle;
290        }

294        private static Collection<? extends Element> getChildren(Element element) {
295                if (element instanceof Document) {
296                        return ((Document) element).getPages();
297                }
298                if (element instanceof Page) {
299                        return Collections.singleton(((Page) element).getModel());
300                }
301                if (element instanceof Model) {
302                        return Collections.singleton(((Model) element).getRoot());
303                }
304                if (element instanceof Root) {
305                        return ((Root) element).getLayers();
306                }
307                if (element instanceof Layer) {
308                        return ((Layer) element).getElements();
309                }
310                return Collections.emptySet();
311        }

314        public static Function<Element, Stream<? extends Element>> childrenMapper(Predicate<Element> predicate) {
315                return new Function<Element, Stream<? extends Element>>() {
316                                                
317                        @Override
318                        public Stream<? extends Element> apply(Element element) {
319                                Stream<? extends Element> ret = getChildren(element).stream();
320                                return predicate == null ? ret : ret.filter(predicate);
321                        }
322                        
323                };
324        }
325        
326        public static Function<Element, Stream<? extends Element>> childrenRecursiveMapper() {
327                return new Function<Element, Stream<? extends Element>>() {
328                        
329                        @Override
330                        public Stream<? extends Element> apply(Element element) {
331                                return getChildren(element).stream().flatMap(childrenRecursiveMapper());
332                        }
333                        
334                };
335                
336        }
337        
338        public static Stream<Element> childrenStream(Element element, Predicate<Element> predicate) {
339                return Stream.of(element).flatMap(childrenMapper(predicate));
340        }

342        /**
343         * @param <T>
344         * @param visitor
345         * @param connectionBase Connection base for visiting linked pages.
346         * @return Visitor which passes itself to linked pages and adds linked pages' result to child results.
347         */
348        public static <T> BiFunction<Element, Map<Element, T>, T> withLinkedPages(BiFunction<Element, Map<Element, T>, T> visitor, ConnectionBase connectionBase) {
349                return new BiFunction<Element, Map<Element, T>, T>() {
350
351                        @Override
352                        public T apply(Element element, Map<Element, T> childResults) {                         
353                                if (element instanceof ModelElement) {
354                                        Page linkedPage = ((ModelElement) element).getLinkedPage();
355                                        if (linkedPage != null) {
356                                                Map<Element, T> cr = new LinkedHashMap<>();
357                                                if (childResults != null) {
358                                                        cr.putAll(childResults);
359                                                }
360                                                T linkedPageResult = linkedPage.accept(this, connectionBase);
361                                                if (linkedPageResult != null) {
362                                                        cr.put(linkedPage, linkedPageResult);
363                                                }
364                                                return visitor.apply(element, cr);
365                                        }
366                                }
367                                return visitor.apply(element, childResults);
368                        }
369                        
370                };
371        }

373        /**
374         * @param <T>
375         * @param visitor
376         * @param connectionBase Connection base for visiting linked pages.
377         * @return Visitor which passes itself to linked pages and adds linked pages' result to child results.
378         */
379        public static <T> Consumer<Element> withLinkedPages(Consumer<Element> visitor, ConnectionBase connectionBase) {
380                return new Consumer<Element>() {
381
382                        @Override
383                        public void accept(Element element) {                           
384                                if (element instanceof ModelElement) {
385                                        Page linkedPage = ((ModelElement) element).getLinkedPage();
386                                        if (linkedPage != null) {
387                                                linkedPage.accept(this, connectionBase);
388                                        }
389                                }
390                                visitor.accept(element);
391                        }
392                        
393                };
394        }
