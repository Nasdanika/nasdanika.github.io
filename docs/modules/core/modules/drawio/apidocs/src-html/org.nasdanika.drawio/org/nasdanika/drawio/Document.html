004import java.io.InputStream;
005import java.io.Reader;
006import java.net.URL;
007import java.net.URLDecoder;
008import java.nio.charset.StandardCharsets;
009import java.util.ArrayList;
010import java.util.List;
011
012import javax.imageio.ImageIO;
013import javax.imageio.ImageReader;
014import javax.imageio.metadata.IIOMetadata;
015import javax.xml.parsers.ParserConfigurationException;
016import javax.xml.transform.TransformerException;
017
018import org.eclipse.emf.common.util.URI;
019import org.nasdanika.drawio.impl.DocumentImpl;
020import org.w3c.dom.Node;
021import org.w3c.dom.NodeList;
022import org.xml.sax.SAXException;
023
024/**
025 * Represents the root mxfile element.
026 * 
027 * @author Pavel
028 *
029 */
030public interface Document extends Element {
031        
032        /**
033         * @return a list of pages which represent nested diagram elements
034         */
035        List<Page> getPages();
036        
037        Page createPage();
038        
039        /**
040         * Loads document from an XML string.
041         * @param docStr
042         * @param base
043         * @return
044         * @throws IOException 
045         * @throws SAXException 
046         * @throws ParserConfigurationException 
047         */
048        static Document load(String docStr, URI uri) throws ParserConfigurationException, SAXException, IOException {
049                return new DocumentImpl(docStr, uri);
050        }
051
052        /**
053         * 
054         * @param reader
055         * @param uri
056         * @return
057         * @throws IOException
058         * @throws ParserConfigurationException
059         * @throws SAXException
060         */
061        static Document load(Reader reader, URI uri) throws IOException, ParserConfigurationException, SAXException {
062                return new DocumentImpl(reader, uri);
063        }
064        
065        /**
066         * 
067         * @param compressed
068         * @param base
069         * @return
070         * @throws ParserConfigurationException
071         */
072        static Document create(boolean compressed, URI uri) throws ParserConfigurationException {
073                return new DocumentImpl(compressed, uri);
074        }
075
076        /**
077         * Loads document using UTF-8 encoding.
078         * @param in
079         * @param uri
080         * @return
081         * @throws IOException
082         * @throws SAXException 
083         * @throws ParserConfigurationException 
084         */
085        static Document load(InputStream in, URI uri) throws IOException, ParserConfigurationException, SAXException {
086                return new DocumentImpl(in, uri);
087        }
088        
089        static Document load(URL source) throws IOException, ParserConfigurationException, SAXException {
090                try (InputStream in = source.openStream()) {
091                        return load(in, URI.createURI(source.toString()));
092                }
093        }
094        
095        /**
096         * Loads documents stored in PNG metadata using UTF-8 encoding.
097         * @param in
098         * @return
099         * @throws IOException
100         * @throws SAXException 
101         * @throws ParserConfigurationException 
102         */
103        static List<Document> loadFromPngMetadata(InputStream in, URI uri) throws IOException, ParserConfigurationException, SAXException {
104                ImageReader imageReader = ImageIO.getImageReadersByFormatName("png").next();
105                imageReader.setInput(ImageIO.createImageInputStream(in));
106                IIOMetadata metadata = imageReader.getImageMetadata(0);
107                List<Document> ret = new ArrayList<>();
108                for (int imageCount = imageReader.getNumImages(true), minIndex = imageReader.getMinIndex(), idx = 0; idx < imageCount; ++idx) {
109                        IIOMetadata imageMetadata = imageReader.getImageMetadata(minIndex + idx);
110                        if (imageMetadata != null) {
111                                for (String formatName: imageMetadata.getMetadataFormatNames()) {
112                                        Node tree = metadata.getAsTree(formatName);
113                                        NodeList childNodes = tree.getChildNodes();
114                                        for (int i = 0; i < childNodes.getLength(); ++i) {
115                                                Node child = childNodes.item(i);
116                                                if (child instanceof org.w3c.dom.Element && "Text".equals(((org.w3c.dom.Element) child).getTagName())) {
117                                                        NodeList grandChildNodes = child.getChildNodes();
118                                                        for (int j = 0; j < grandChildNodes.getLength(); ++j) {
119                                                                Node grandChild = grandChildNodes.item(j);
120                                                                if (grandChild instanceof org.w3c.dom.Element 
121                                                                                && "TextEntry".equals(((org.w3c.dom.Element) grandChild).getTagName())
122                                                                                && ((org.w3c.dom.Element) grandChild).hasAttribute("keyword")
123                                                                                && "mxfile".equals(((org.w3c.dom.Element) grandChild).getAttribute("keyword"))) {
124                                                                        String value = ((org.w3c.dom.Element) grandChild).getAttribute("value");
125                                                                        ret.add(load((URLDecoder.decode(value, StandardCharsets.UTF_8)), uri));                                                                 
126                                                                }
127                                                        }
128                                                }
129                                        }
130                                }
131                        }
132                }
133                return ret;
134        }
135        
136        static List<Document> loadFromPngMetadata(URL source) throws IOException, ParserConfigurationException, SAXException {
137                try (InputStream in = source.openStream()) {
138                        return loadFromPngMetadata(in, URI.createURI(source.toString()));
139                }
140        }
141        
142        /**
143         * @param compress If null keeps the original compression setting. If not null forces either compressed (TRUE) or uncompressed (FALSE).
144         * This can be used to generate HTML tags with compressed document while keeping the source document uncompressed so it can be manually inspected, edited, diffed, and merged.
145         * @param viewer URL of the viewer script. Default value is used if null.  
146         * @return div tag with the embedded document which can be inserted into HTML pages.
147         */
148        String toHtml(Boolean compress, String viewer) throws TransformerException, IOException;
149
150        /**
151         * Saves the document to String.
152         * @param compress If null keeps the original compression setting. If not null forces either compressed (TRUE) or uncompressed (FALSE).
153         * This can be used to generate HTML tags with compressed document while keeping the source document uncompressed so it can be manually inspected, edited, diffed, and merged.  
154         * @return Document string.
155         */
156        String save(Boolean compress) throws TransformerException, IOException;
157
158}
