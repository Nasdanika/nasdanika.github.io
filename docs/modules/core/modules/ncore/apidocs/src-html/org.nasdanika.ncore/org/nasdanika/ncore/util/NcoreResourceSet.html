001package org.nasdanika.ncore.util;
002
003import org.eclipse.emf.common.notify.Notifier;
004import org.eclipse.emf.common.util.TreeIterator;
005import org.eclipse.emf.common.util.URI;
006import org.eclipse.emf.ecore.EObject;
007import org.eclipse.emf.ecore.resource.impl.ResourceSetImpl;
008import org.nasdanika.common.Util;
009import org.nasdanika.ncore.ModelElement;
010
011public class NcoreResourceSet extends ResourceSetImpl {
012
013        @Override
014        public EObject getEObject(URI uri, boolean loadOnDemand) {              
015                TreeIterator<Notifier> cit = getAllContents();
016                while (cit.hasNext()) {
017                        Notifier next = cit.next();
018                        if (next instanceof EObject) {
019                                EObject nextEObject = (EObject) next;
020                                if (matchURI(nextEObject, uri)) {
021                                        return nextEObject;
022                                }
023                        }
024                }
025                
026                return super.getEObject(uri, loadOnDemand);
027        }
028        
029        /**
030         * Matches {@link EObject} to {@link URI}. This implementation uses ModelElement.getUri() to 
031         * match containment path URI's and ModelElement.getUuid() to match URI's with <code>uuid</code> scheme.  
032         * uses  
033         * @param eObj
034         * @param uri
035         * @return
036         */
037        protected boolean matchURI(EObject eObj, URI uri) {
038                if (NcoreUtil.isEMapEntry(eObj)) {
039                        return false; // Ignoring EMap entries.
040                }
041                URI objUri = NcoreUtil.getUri(eObj);
042                if (objUri != null && objUri.equals(uri)) {
043                        return true;
044                }
045                if (eObj instanceof ModelElement) {
046                        String uuid = ((ModelElement) eObj).getUuid();
047                        if (!Util.isBlank(uuid)) {
048                                URI uuidUri = URI.createURI("uuid:" + uuid);
049                                if (uuidUri.equals(uri)) {
050                                        return true;
051                                }
052                        }
053                }
054                return false;
055        }
056
057}
