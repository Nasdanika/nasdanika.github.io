071        /**
072         * Flattened inheritance hierarchy from the argument class to all of its supertypes in the order of inheritance.
073         * @param eClass
074         * @return
075         */
076        public static List<EClass> lineage(EClass eClass) {
077                List<EClass> ret = new ArrayList<>();
078                ret.add(eClass);
079                for (EClass sc: eClass.getESuperTypes()) {
080                        for (EClass scLineage: lineage(sc)) {
081                                if (!ret.contains(scLineage)) {
082                                        ret.add(scLineage);
083                                }
084                        }
085                }
086                return ret;
087        }

089        /**
090         * Finds objects referencing the argument object (target) via the specified reference.
091         * @param <T>
092         * @param target Referenced object
093         * @param eReference Reference
094         * @return Object referencing this one via the specified reference.
095         */
096        public static <T extends EObject> EList<T> getReferrers(EObject target, EReference eReference) {
097                EList<T> ret = new BasicInternalEList<>(EObject.class);
098                
099                Resource res = target.eResource(); 
100                TreeIterator<?> cit = null;
101                if (res == null) {
102                        EObject root = target;
103                        EObject rc;
104                        while ((rc = root.eContainer()) != null) {
105                                root = rc;
106                        }
107                        if (root != null) {
108                                NcoreUtil.collect(root, target, eReference, ret);
109                                cit = root.eAllContents();
110                        }                       
111                } else {
112                        ResourceSet rSet = res.getResourceSet();
113                        cit = rSet == null ? res.getAllContents() : rSet. getAllContents();
114                }
115                if (cit != null) {
116                        while (cit.hasNext()) {
117                                NcoreUtil.collect(cit.next(), target, eReference, ret);
118                        }                       
119                }
120                return ret;
121        }

123        @SuppressWarnings("unchecked")
124        private static <T extends EObject> void collect(
125                        Object source, 
126                        EObject target, 
127                        EReference eReference,
128                        EList<T> accumulator) {
129                if (eReference.getEContainingClass().isInstance(source)) {
130                        Object value = ((EObject) source).eGet(eReference);
131                        if (eReference.isMany()) {
132                                if (((Collection<?>) value).contains(target)) {
133                                        accumulator.add((T) source);
134                                }
135                        } else if (value == target) {
136                                accumulator.add((T) source);
137                        }                                       
138                }
139        }

156        /**
157         * Containment path relative to the object container. If the container is a map entry its key is added to path and the entry container is skipped. This happens recursively. 
158         * @param eObject
159         * @return null if there is no container. BiSupplier with first returning the first container which is not EMap entry and reference name concatenated with object key for {@link EMap}'s and with object index for many-references.
160         */
161        public static BiSupplier<EObject, String> containmentPath(EObject eObj) {
162                EObject container = eObj.eContainer();
163                if (container == null) {
164                        return null;
165                }
166                EReference eContainmentFeature = eObj.eContainmentFeature();
167                
168                LinkedList<String> containmentIndex = new LinkedList<>();
169                while (NcoreUtil.isEMapEntry(container)) {
170                        Object key = ((Map.Entry<?,?>) container).getKey();
171                        if (key == null) {
172                                return null;
173                        }
174                        String idx = NcoreUtil.containmentIndex(eObj);
175                        if (idx != null) {
176                                containmentIndex.addFirst(idx);
177                        }
178                        containmentIndex.addFirst(key.toString());
179                        
180                        eContainmentFeature = container.eContainmentFeature();
181                        container = container.eContainer();
182                }
183                
184                if (containmentIndex.isEmpty()) {
185                        String idx = NcoreUtil.containmentIndex(eObj);
186                        if (idx != null) {
187                                containmentIndex.add(idx);
188                        }
189                } 
190        
191                containmentIndex.addFirst(NcoreUtil.getFeatureKey(container.eClass(), eContainmentFeature));
192                return BiSupplier.of(container, String.join("/", containmentIndex));                            
193        }

195        /**
196         * Index part of the containment path. Null for single references. EKeys for containments with EKeys. Position otherwise.
197         * @param eObj
198         * @return
199         */
200        private static String containmentIndex(EObject eObj) {
201                EObject container = eObj.eContainer();
202                if (container == null) {
203                        return null;
204                }
205                EReference eContainmentFeature = eObj.eContainmentFeature();
206                if (eContainmentFeature.isMany()) {
207                        EList<EAttribute> eKeys = eContainmentFeature.getEKeys();
208                        String position = String.valueOf(((List<?>) container.eGet(eContainmentFeature)).indexOf(eObj));
209                        if (eKeys.isEmpty()) {
210                                return position;
211                        }
212                        StringBuilder pathBuilder = new StringBuilder();
213                        for (EAttribute eKey: eKeys) {
214                                if (!eObj.eIsSet(eKey)) {
215                                        return position;
216                                }
217                                if (pathBuilder.length() > 0) {
218                                        pathBuilder.append("/");
219                                }
220                                pathBuilder.append(eObj.eGet(eKey));
221                        }
222                        return pathBuilder.toString();
223                }
224                return null;
225        }

227        /**
228                 * Returns {@link ModelElement}.getUri() if the argument is an instance of ModelElement and getUri() returns a non blank string.
229                 * Computes URI from the containment reference otherwise.
230                 * @param eObj
231                 * @return
232                 */
233                public static URI getUri(EObject eObj) {
234                        if (eObj == null || isEMapEntry(eObj)) {
235                                return null; // EMap entries do not have URI's.
236                        }
237                        
238                        if (eObj instanceof ModelElement) {
239                                String uri = ((ModelElement) eObj).getUri();
240                                if (!Util.isBlank(uri)) {
241                                        return URI.createURI(uri);
242                                }
243                        }
244                        
245                        BiSupplier<EObject, String> containmentPath = containmentPath(eObj);
246                        if (containmentPath == null) {
247                                return null;
248                        }
249                        EObject container = containmentPath.getFirst();
250                        if (container == null) {
251        //                      Leads to stack overflow during loading.                 
252        //                      for (EObject referrer: getReferrers(eObj, NcorePackage.Literals.REFERENCE__TARGET)) {
253        //                              return getUri(referrer);
254        //                      }
255                                if (eObj instanceof ModelElement) {
256                                        String uuid = ((ModelElement) eObj).getUuid();
257                                        if (!Util.isBlank(uuid)) {
258                                                return URI.createURI("uuid://" + uuid);
259                                        }
260                                }
261                                return null;
262                        }
263                        
264                        URI cURI = getUri(container);
265                        if (cURI == null || !cURI.isHierarchical()) {
266                                return null;
267                        }
268                        String cLastSegment = cURI.lastSegment();
269                        if (cLastSegment == null || cLastSegment.length() > 0) {
270                                cURI = cURI.appendSegment("");
271                        }
272                        URI cpURI = URI.createURI(containmentPath.getSecond());
273                        return cpURI.resolve(cURI);
274                }

276        /**
277         * @param eObj
278         * @return true if the argument is an entry for EMap
279         */
280        public static boolean isEMapEntry(EObject eObj) {
281                if (eObj instanceof Map.Entry) {
282                        EReference cf = eObj.eContainmentFeature();
283                        return cf != null && cf.isMany() && cf.getEType().getInstanceClass() == Map.Entry.class;
284                }
285                return false;
286        }

288        public static EObject commonAncestor(EObject a, EObject b) {
289                if (a == null || b == null) {
290                        return null;
291                }
292                if (EcoreUtil.isAncestor(a, b)) {
293                        return a.eContainer();
294                }
295                if (EcoreUtil.isAncestor(b, a)) {
296                        b.eContainer();
297                }
298                for (EObject ancestor = a.eContainer(); ancestor != null; ancestor = ancestor.eContainer()) {
299                        if (EcoreUtil.isAncestor(ancestor, b)) {
300                                return ancestor;
301                        }
302                }
303                return null;
304        }

328        /**
329         * Feature key is used for feature loading from YAML and other key-value sources and also in construction of containment path by containmentPath() method and, as such, EObject {@link URI} by getUri() method.
330         * This method traverses the class inheritance hierarchy and uses feature-keys class-level annotations to retrieve feature key. If such an annotation is not found then feature-level feature-key annotation is used.
331         * If none is found in the entire hierarchy then feature name is converted to kebab case is returned. 
332         * @param eClass Context class for which feature key is retrieved.
333         * @param feature Feature
334         * @return
335         */
336        public static String getFeatureKey(EClass eClass, EStructuralFeature feature) {
337                String featureKey = NcoreUtil.getNasdanikaAnnotationDetail(feature, FEATURE_KEY);
338                if (!Util.isBlank(featureKey)) {
339                        return featureKey;
340                }
341                
342                for (EClass ec: NcoreUtil.lineage(eClass)) {
343                        String featureKeys = NcoreUtil.getNasdanikaAnnotationDetail(ec, FEATURE_KEYS);
344                        if (!Util.isBlank(featureKeys)) {
345                                Yaml yaml = new Yaml();
346                                Map<String,Object> featureKeyMap = yaml.load(featureKeys);
347                                Object fKey = featureKeyMap.get(feature.getName());
348                                if (fKey instanceof String) {
349                                        return (String) fKey;
350                                }
351                        }
352                }
353                return Util.camelToKebab(feature.getName());
354        }

356        /**
357         * 
358         * @param eReference
359         * @return "Logical" opposite - either the "physical" opposite returned by {@link EReference}.getEOpposite() or a reference with a name specified in "opposite" Nasdanika annotation
360         * or a reference which specifies this reference in its "opposite" Nasdanika annotation.
361         */
362        public static EReference getOpposite(EReference eReference) {
363                EReference opposite = eReference.getEOpposite();
364                if (opposite != null) {
365                        return opposite;
366                }
367                String oName = getNasdanikaAnnotationDetail(eReference, "opposite");
368                EClass refType = eReference.getEReferenceType();
369                for (EReference ref: refType.getEAllReferences()) {
370                        if (ref.getName().equals(oName)) {
371                                return ref;
372                        }
373                        if (ref.getEReferenceType() == eReference.getEContainingClass() && eReference.getName().equals(getNasdanikaAnnotationDetail(ref, "opposite"))) {
374                                return ref;
375                        }
376                }
377                return null;
378        }

380        /**
381         * Computes end and duration from start and previously loaded things.
382         * @param period
383         * @param newStart
384         */
385        public static void onStart(Period period, Temporal newStart) {
386                if (newStart != null) {
387                        LoadTracker loadTracker = (LoadTracker) EcoreUtil.getRegisteredAdapter(period, LoadTracker.class);
388                        if (loadTracker != null && loadTracker.isLoading(NcorePackage.Literals.PERIOD__START)) {
389                                if (loadTracker.isLoaded(NcorePackage.Literals.PERIOD__END)) {
390                                        if (!loadTracker.isLoaded(NcorePackage.Literals.PERIOD__DURATION)) {
391                                                period.setDuration(period.getEnd().minus(newStart));
392                                        }
393                                } else if (loadTracker.isLoaded(NcorePackage.Literals.PERIOD__DURATION)) {
394                                        period.setEnd(newStart.plus(period.getDuration()));
395                                }
396                        }
397                }
398        }
399
400        public static void onEnd(Period period, Temporal newEnd) {
401                if (newEnd != null) {
402                        LoadTracker loadTracker = (LoadTracker) EcoreUtil.getRegisteredAdapter(period, LoadTracker.class);
403                        if (loadTracker != null && loadTracker.isLoading(NcorePackage.Literals.PERIOD__END)) {
404                                if (loadTracker.isLoaded(NcorePackage.Literals.PERIOD__START)) {
405                                        if (!loadTracker.isLoaded(NcorePackage.Literals.PERIOD__DURATION)) {
406                                                period.setDuration(newEnd.minus(period.getStart()));
407                                        }
408                                } else if (loadTracker.isLoaded(NcorePackage.Literals.PERIOD__DURATION)) {
409                                        period.setStart(newEnd.minus(period.getDuration()));
410                                }
411                        }
412                }
413        }
414
415        public static void onDuration(Period period, Duration newDuration) {
416                if (newDuration != null) {
417                        LoadTracker loadTracker = (LoadTracker) EcoreUtil.getRegisteredAdapter(period, LoadTracker.class);
418                        if (loadTracker != null && loadTracker.isLoading(NcorePackage.Literals.PERIOD__DURATION)) {
419                                if (loadTracker.isLoaded(NcorePackage.Literals.PERIOD__START)) {
420                                        if (!loadTracker.isLoaded(NcorePackage.Literals.PERIOD__END)) {
421                                                period.setEnd(period.getStart().plus(newDuration));
422                                        }
423                                } else if (loadTracker.isLoaded(NcorePackage.Literals.PERIOD__END)) {
424                                        period.setStart(period.getEnd().minus(newDuration));
425                                }
426                        }
427                }
428        }
