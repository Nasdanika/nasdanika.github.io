058        public static String getNasdanikaAnnotationDetail(EModelElement modelElement, String key, String defaultValue) {
059                EAnnotation na = getNasdanikaAnnotation(modelElement);
060                if (na == null || !na.getDetails().containsKey(key)) {
061                        return defaultValue;
062                }
063                return na.getDetails().get(key);

066        /**
067         * Flattened inheritance hierarchy from the argument class to all of its supertypes in the order of inheritance.
068         * @param eClass
069         * @return
070         */
071        public static List<EClass> lineage(EClass eClass) {
072                List<EClass> ret = new ArrayList<>();
073                ret.add(eClass);
074                for (EClass sc: eClass.getESuperTypes()) {
075                        for (EClass scLineage: lineage(sc)) {
076                                if (!ret.contains(scLineage)) {
077                                        ret.add(scLineage);
078                                }
079                        }
080                }
081                return ret;
082        }

084        /**
085         * Finds objects referencing the argument object (target) via the specified reference.
086         * @param <T>
087         * @param target Referenced object
088         * @param eReference Reference
089         * @return Object referencing this one via the specified reference.
090         */
091        public static <T extends EObject> EList<T> getReferrers(EObject target, EReference eReference) {
092                EList<T> ret = new BasicInternalEList<>(EObject.class);
093                
094                Resource res = target.eResource(); 
095                TreeIterator<?> cit = null;
096                if (res == null) {
097                        EObject root = target;
098                        EObject rc;
099                        while ((rc = root.eContainer()) != null) {
100                                root = rc;
101                        }
102                        if (root != null) {
103                                NcoreUtil.collect(root, target, eReference, ret);
104                                cit = root.eAllContents();
105                        }                       
106                } else {
107                        ResourceSet rSet = res.getResourceSet();
108                        cit = rSet == null ? res.getAllContents() : rSet. getAllContents();
109                }
110                if (cit != null) {
111                        while (cit.hasNext()) {
112                                NcoreUtil.collect(cit.next(), target, eReference, ret);
113                        }                       
114                }
115                return ret;
116        }

118        @SuppressWarnings("unchecked")
119        private static <T extends EObject> void collect(
120                        Object source, 
121                        EObject target, 
122                        EReference eReference,
123                        EList<T> accumulator) {
124                if (eReference.getEContainingClass().isInstance(source)) {
125                        Object value = ((EObject) source).eGet(eReference);
126                        if (eReference.isMany()) {
127                                if (((Collection<?>) value).contains(target)) {
128                                        accumulator.add((T) source);
129                                }
130                        } else if (value == target) {
131                                accumulator.add((T) source);
132                        }                                       
133                }
134        }

142        public static EObject createProxy(EClass type, URI uri) {
143                EObject proxy = type.getEPackage().getEFactoryInstance().create(type);
144                ((InternalEObject) proxy).eSetProxyURI(uri);
145                if (proxy instanceof NamedElement) {
146                        ((NamedElement) proxy).setName(type.getName() + " proxy for " + uri);

151        /**
152         * Containment path relative to the object container. If the container is a map entry its key is added to path and the entry container is skipped. This happens recursively. 
153         * @param eObject
154         * @return null if there is no container. BiSupplier with first returning the first container which is not EMap entry and reference name concatenated with object key for {@link EMap}'s and with object index for many-references.
155         */
156        public static BiSupplier<EObject, String> containmentPath(EObject eObj) {
157                EObject container = eObj.eContainer();
158                if (container == null) {
159                        return null;
160                }
161                EReference eContainmentFeature = eObj.eContainmentFeature();
162                
163                LinkedList<String> containmentIndex = new LinkedList<>();
164                while (NcoreUtil.isEMapEntry(container)) {
165                        Object key = ((Map.Entry<?,?>) container).getKey();
166                        if (key == null) {
167                                return null;
168                        }
169                        String idx = NcoreUtil.containmentIndex(eObj);
170                        if (idx != null) {
171                                containmentIndex.addFirst(idx);
172                        }
173                        containmentIndex.addFirst(key.toString());
174                        
175                        eContainmentFeature = container.eContainmentFeature();
176                        container = container.eContainer();
177                }
178                
179                if (containmentIndex.isEmpty()) {
180                        String idx = NcoreUtil.containmentIndex(eObj);
181                        if (idx != null) {
182                                containmentIndex.add(idx);
183                        }
184                } 
185        
186                containmentIndex.addFirst(NcoreUtil.getFeatureKey(container.eClass(), eContainmentFeature));
187                return BiSupplier.of(container, String.join("/", containmentIndex));                            
188        }

190        /**
191         * Index part of the containment path. Null for single references. EKeys for containments with EKeys. Position otherwise.
192         * @param eObj
193         * @return
194         */
195        private static String containmentIndex(EObject eObj) {
196                EObject container = eObj.eContainer();
197                if (container == null) {
198                        return null;
199                }
200                EReference eContainmentFeature = eObj.eContainmentFeature();
201                if (eContainmentFeature.isMany()) {
202                        EList<EAttribute> eKeys = eContainmentFeature.getEKeys();
203                        String position = String.valueOf(((List<?>) container.eGet(eContainmentFeature)).indexOf(eObj));
204                        if (eKeys.isEmpty()) {
205                                return position;
206                        }
207                        StringBuilder pathBuilder = new StringBuilder();
208                        for (EAttribute eKey: eKeys) {
209                                if (!eObj.eIsSet(eKey)) {
210                                        return position;
211                                }
212                                if (pathBuilder.length() > 0) {
213                                        pathBuilder.append("/");
214                                }
215                                pathBuilder.append(eObj.eGet(eKey));
216                        }
217                        return pathBuilder.toString();
218                }
219                return null;
220        }

222        /**
223                 * Returns {@link ModelElement}.getUri() if the argument is an instance of ModelElement and getUri() returns a non blank string.
224                 * Computes URI from the containment reference otherwise.
225                 * @param eObj
226                 * @return
227                 */
228                public static URI getUri(EObject eObj) {
229                        if (eObj == null || isEMapEntry(eObj)) {
230                                return null; // EMap entries do not have URI's.
231                        }
232                        
233                        if (eObj instanceof ModelElement) {
234                                String uri = ((ModelElement) eObj).getUri();
235                                if (!Util.isBlank(uri)) {
236                                        return URI.createURI(uri);
237                                }
238                        }
239                        
240                        BiSupplier<EObject, String> containmentPath = containmentPath(eObj);
241                        if (containmentPath == null) {
242                                return null;
243                        }
244                        EObject container = containmentPath.getFirst();
245                        if (container == null) {
246        //                      Leads to stack overflow during loading.                 
247        //                      for (EObject referrer: getReferrers(eObj, NcorePackage.Literals.REFERENCE__TARGET)) {
248        //                              return getUri(referrer);
249        //                      }
250                                if (eObj instanceof ModelElement) {
251                                        String uuid = ((ModelElement) eObj).getUuid();
252                                        if (!Util.isBlank(uuid)) {
253                                                return URI.createURI("uuid://" + uuid);
254                                        }
255                                }
256                                return null;
257                        }
258                        
259                        URI cURI = getUri(container);
260                        if (cURI == null || !cURI.isHierarchical()) {
261                                return null;
262                        }
263                        String cLastSegment = cURI.lastSegment();
264                        if (cLastSegment == null || cLastSegment.length() > 0) {
265                                cURI = cURI.appendSegment("");
266                        }
267                        URI cpURI = URI.createURI(containmentPath.getSecond());
268                        return cpURI.resolve(cURI);
269                }

271        /**
272         * @param eObj
273         * @return true if the argument is an entry for EMap
274         */
275        public static boolean isEMapEntry(EObject eObj) {
276                if (eObj instanceof Map.Entry) {
277                        EReference cf = eObj.eContainmentFeature();
278                        return cf != null && cf.isMany() && cf.getEType().getInstanceClass() == Map.Entry.class;
279                }
280                return false;
281        }

283        public static EObject commonAncestor(EObject a, EObject b) {
284                if (a == null || b == null) {
285                        return null;
286                }
287                if (EcoreUtil.isAncestor(a, b)) {
288                        return a.eContainer();
289                }
290                if (EcoreUtil.isAncestor(b, a)) {
291                        b.eContainer();
292                }
293                for (EObject ancestor = a.eContainer(); ancestor != null; ancestor = ancestor.eContainer()) {
294                        if (EcoreUtil.isAncestor(ancestor, b)) {
295                                return ancestor;
296                        }
297                }
298                return null;
299        }

323        /**
324         * Feature key is used for feature loading from YAML and other key-value sources and also in construction of containment path by containmentPath() method and, as such, EObject {@link URI} by getUri() method.
325         * This method traverses the class inheritance hierarchy and uses feature-keys class-level annotations to retrieve feature key. If such an annotation is not found then feature-level feature-key annotation is used.
326         * If none is found in the entire hierarchy then feature name is converted to kebab case is returned. 
327         * @param eClass Context class for which feature key is retrieved.
328         * @param feature Feature
329         * @return
330         */
331        public static String getFeatureKey(EClass eClass, EStructuralFeature feature) {
332                String featureKey = NcoreUtil.getNasdanikaAnnotationDetail(feature, FEATURE_KEY);
333                if (!Util.isBlank(featureKey)) {
334                        return featureKey;
335                }
336                
337                for (EClass ec: NcoreUtil.lineage(eClass)) {
338                        String featureKeys = NcoreUtil.getNasdanikaAnnotationDetail(ec, FEATURE_KEYS);
339                        if (!Util.isBlank(featureKeys)) {
340                                Yaml yaml = new Yaml();
341                                Map<String,Object> featureKeyMap = yaml.load(featureKeys);
342                                Object fKey = featureKeyMap.get(feature.getName());
343                                if (fKey instanceof String) {
344                                        return (String) fKey;
345                                }
346                        }
347                }
348                return Util.camelToKebab(feature.getName());
349        }

351        /**
352         * 
353         * @param eReference
354         * @return "Logical" opposite - either the "physical" opposite returned by {@link EReference}.getEOpposite() or a reference with a name specified in "opposite" Nasdanika annotation
355         * or a reference which specifies this reference in its "opposite" Nasdanika annotation.
356         */
357        public static EReference getOpposite(EReference eReference) {
358                EReference opposite = eReference.getEOpposite();
359                if (opposite != null) {
360                        return opposite;
361                }
362                String oName = getNasdanikaAnnotationDetail(eReference, "opposite");
363                EClass refType = eReference.getEReferenceType();
364                for (EReference ref: refType.getEAllReferences()) {
365                        if (ref.getName().equals(oName)) {
366                                return ref;
367                        }
368                        if (ref.getEReferenceType() == eReference.getEContainingClass() && eReference.getName().equals(getNasdanikaAnnotationDetail(ref, "opposite"))) {
369                                return ref;
370                        }
371                }
372                return null;
373        }
