176         * <!-- begin-user-doc -->
177         * <!-- end-user-doc -->
178         * <!-- begin-model-doc -->
179         * Tests if this temporal is after the specified temporal. Returns null if unknown, e.g. two unrelated events.
180         * <!-- end-model-doc -->
181         * @model
182         * @generated NOT
183         */
184        default Boolean after(Temporal when) {
185                if (when == null) {
186                        return false;
187                }
188                
189                Temporal nThis = normalize();
190                Instant thisInstant = nThis.getInstant();
191                
192                Temporal nWhen = when.normalize();
193                Instant whenInstant = nWhen.getInstant();
194                
195                if (whenInstant != null) {
196                        if (thisInstant == null) {
197                                return null;
198                        }
199                        return thisInstant.isAfter(whenInstant);
200                }
201                if (nWhen.getBase() == nThis.getBase()) {
202                        Duration whenOffset = nWhen.getOffset();
203                        Duration thisOffset = nThis.getOffset();
204                        
205                        if (whenOffset == null) {
206                                whenOffset = Duration.ZERO;
207                        }
208                        if (thisOffset == null) {
209                                thisOffset = Duration.ZERO;
210                        }
211                        return thisOffset.compareTo(whenOffset) > 0;
212                }
213                
214                /*
215                 * Bounds based - when upper bounds shall not be after this or its lower bounds
216                 * If any of when upper bounds is before this or lower bound then this is after when 
217                 * if any of lower bounds are after when then this is after when 
218                 */
219                for (Temporal whenUpperBound: when.getUpperBounds()) {
220                        if (this.after(whenUpperBound) == Boolean.TRUE) {
221                                return true;
222                        }
223                        for (Temporal lowerBound: getLowerBounds()) {
224                                if (lowerBound.after(whenUpperBound) == Boolean.TRUE) {
225                                        return true;
226                                }
227                        }
228                }
229                for (Temporal lowerBound: getLowerBounds()) {
230                        if (lowerBound.after(when) == Boolean.TRUE) {
231                                return true;
232                        }
233                }
234                
235                return null;            
236        }

385         * <!-- begin-user-doc -->
386         * <!-- end-user-doc -->
387         * <!-- begin-model-doc -->
388         * Returns [duration](Duration.html) difference between this temporal and the argument temporal - how much this temporal is after the argument on the time-line, if difference can be computed, e.g. both this temporal and the argument temporal are instant or trace to the same base temporal. Returns null otherwise.
389         * <!-- end-model-doc -->
390         * @model dataType="org.nasdanika.engineering.Duration"
391         * @generated NOT
392         */
393        default Duration minus(Temporal when) {
394                if (when == null) {
395                        return null;
396                }
397                
398                // Instant
399                Instant whenInstant = when.getInstant();
400                Instant thisInstant = getInstant();
401                
402                if (thisInstant != null && whenInstant != null) {
403                        return Duration.ofMillis(thisInstant.toEpochMilli() - whenInstant.toEpochMilli());
404                }
405
406                // Base
407                Temporal base = getBase();
408                if (base == when) {
409                        return getOffset();
410                }
411                
412                Temporal whenBase = when.getBase();             
413                if (whenBase == this) {
414                        Duration whenOffset = when.getOffset();
415                        return whenOffset == null ? whenOffset : whenOffset.negated();
416                }
417                
418                if (whenBase == base) {
419                        Duration whenOffset = when.getOffset();
420                        Duration offset = getOffset();
421                        if (whenOffset == null) {
422                                return offset;
423                        }
424                        if (offset == null) {
425                                return whenOffset.negated();
426                        }
427                        return offset.minus(whenOffset);
428                }
429                
430                if (whenBase == null && base == null) {
431                        return null; // No point in normalization.
432                }
433                
434                if (base == null) {
435                        if (whenBase.getBase() == null) {
436                                return null; // No point in normalization.
437                        }
438                        // There is when super-base - shall normalize.
439                        return minus(when.normalize());
440                }
441                
442                if (whenBase == null) {
443                        if (base.getBase() == null) {
444                                return null; // No point in normalization
445                        }
446                        // There is super-base - shall normalize.
447                        return normalize().minus(when);
448                }
449                
450                // Normalized
451                return normalize().minus(when.normalize()); 
452        }

533        static java.lang.String formatDuration(Duration duration) {
534                if (duration == null || duration.equals(Duration.ZERO)) {
535                        return "Zero";
536                }
537                
538                StringBuilder ret = new StringBuilder();
539                long days = duration.toDays();
540                if (days != 0) {
541                        ret.append(days).append(days == 1 ? " day" : " days");
542                }
543                long hours = duration.toHours() % 24;
544                if (hours != 0) {
545                        if (ret.length() > 0) {
546                                ret.append(" ");
547                        }
548                        ret.append(hours).append(hours == 1 ? " hour" : " hours");                      
549                }               
550                long minutes = duration.toMinutes() % 60;
551                if (minutes != 0) {
552                        if (ret.length() > 0) {
553                                ret.append(" ");
554                        }       
555                        ret.append(minutes).append(minutes == 1 ? " minute" : " minutes");
556                }
557                long seconds = (duration.toMillis() * 1000) % 60;
558                if (seconds != 0) {
559                        if (ret.length() > 0) {
560                                ret.append(" ");
561                        }                       
562                        ret.append(seconds).append(seconds == 1 ? " second" : " seconds");
563                }
564//              long nanos = duration.toNanos();
565//              if (nanos != 0) {
566//                      if (ret.length() > 0) {
567//                              ret.append(" ");
568//                      }                       
569//                      ret.append(nanos).append(nanos == 1 ? " nanosecond" : " nanoseconds");
570//              }
571                
572                return ret.toString();
573        }
