001package org.nasdanika.ncore.util;
002
003import java.util.Comparator;
004
005import org.eclipse.emf.common.util.URI;
006import org.nasdanika.common.Util;
007import org.nasdanika.ncore.NamedElement;
008
009public class NamedElementComparator implements Comparator<NamedElement> {
010        
011        public static NamedElementComparator INSTANCE = new NamedElementComparator();
012
013        @Override
014        public int compare(NamedElement a, NamedElement b) {
015                if (a == b) {
016                        return 0;
017                }
018                if (a == null) {
019                        return 1;
020                }
021                if (b == null) {
022                        return -1;
023                }
024                if (Util.isBlank(a.getName())) {
025                        if (Util.isBlank(b.getName())) {
026                                URI aUri = NcoreUtil.getUri(a);
027                                URI bUri = NcoreUtil.getUri(b);
028                                if (aUri == null) {
029                                        if (bUri == null) {
030                                                return a.hashCode() - b.hashCode();
031                                        }
032                                        return 1;
033                                }
034                                if (bUri == null) {
035                                        return -1;
036                                }
037                                return aUri.toString().compareTo(bUri.toString());
038                        }
039                        
040                        return 1;
041                }
042                
043                if (Util.isBlank(b.getName())) {
044                        return -1;
045                }
046                
047                return a.getName().compareTo(b.getName());
048        }
049
050
051}
