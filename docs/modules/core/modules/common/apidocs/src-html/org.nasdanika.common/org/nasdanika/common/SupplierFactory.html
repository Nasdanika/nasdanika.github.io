001package org.nasdanika.common;
002
003import java.util.function.BiFunction;
004
005/**
006 * @author Pavel Vlasov
007 *
008 * @param <T>
009 */
010public interface SupplierFactory<T> extends ExecutionParticipantFactory<Supplier<T>>, java.util.function.BiFunction<Context,ProgressMonitor,T> {
011                
012        static <T> SupplierFactory<T> empty() {
013                return context -> Supplier.empty();
014        }       
015        
016        /**
017         * Provides {@link SupplierFactory} for a specific type.
018         * @author Pavel
019         *
020         */
021        interface Provider {
022                <T> SupplierFactory<T> getFactory(Class<T> type);
023        }
024                
025        default <V> SupplierFactory<V> then(FunctionFactory<? super T,V> then) {
026                return new SupplierFactory<V>() {
027                        
028                        @Override
029                        public Supplier<V> create(Context context) throws Exception {
030                                return SupplierFactory.this.create(context).then(then.create(context));
031                        }
032                        
033                };
034        }
035        
036        default CommandFactory then(ConsumerFactory<? super T> then) {
037                return new CommandFactory() {
038                        
039                        @Override
040                        public Command create(Context context) throws Exception {
041                                return SupplierFactory.this.create(context).then(then.create(context));
042                        }
043                        
044                };
045        }
046                
047        @Override
048        default T apply(Context context, ProgressMonitor progressMonitor) {
049                try {
050                        return create(context).apply(progressMonitor);
051                } catch (DiagnosticException e) {
052                        throw e;
053                } catch (Exception e) {
054                        throw new NasdanikaException(e);
055                }
056        }
057        
058        default <V> FunctionFactory<V,BiSupplier<V,T>> asFunctionFactory() { 
059                return new FunctionFactory<V, BiSupplier<V,T>>() {
060
061                        @Override
062                        public Function<V, BiSupplier<V, T>> create(Context context) throws Exception {                         
063                                return SupplierFactory.this.create(context).asFunction();
064                        }
065                };
066        }
067        
068        static <T> SupplierFactory<T> from(BiFunction<Context, ProgressMonitor, T> biFunction, String name, double size) {
069                return new SupplierFactory<T>() {
070
071                        @Override
072                        public Supplier<T> create(Context context) throws Exception {
073                                return Supplier.fromFunction(pm -> biFunction.apply(context, pm), name, size);
074                        }
075                        
076                };
077        }
078        
079        static <T> SupplierFactory<T> from(T value, String name) {
080                return from((c,p) -> value, name, 0);           
081        }       
082        
083        /**
084         * If source is adaptable to SupplierFactory then adapter is returned. Otherwise it is wrapped into a supplier factory
085         * @param <T>
086         * @param value
087         * @param name
088         * @return
089         */
090        @SuppressWarnings("unchecked")
091        static <T> SupplierFactory<T> adapt(Object source, String name) {
092                SupplierFactory<?> ret = Adaptable.adaptTo(source, SupplierFactory.class);
093                if (ret == null) {
094                        ret = from((c,p) -> source, name, 0);
095                }
096                return (SupplierFactory<T>) ret;
097        }       
098        
099        /**
100         * @param contextSupplierFactory Factory which creates a context to be passed to this factory to create supplier.
101         * @return Supplier factory which creates a context using the context supplier factory and then uses that context and this factory to create a supplier.
102         */
103        default SupplierFactory<T> contextify(SupplierFactory<Context> contextSupplierFactory) {
104                return new SupplierFactory<T>() {
105
106                        @Override
107                        public Supplier<T> create(Context context) throws Exception {
108                                return new ContextifiedExecutionParticipant.ContextifiedSupplier<T>(contextSupplierFactory.create(context), SupplierFactory.this);
109                        }
110                        
111                };
112        }       
113        
114}
