001package org.nasdanika.common;
002
003import java.util.LinkedHashMap;
004import java.util.Map;
005import java.util.Map.Entry;
006
007public class MapCompoundFunctionFactory<K,T,R> implements FunctionFactory<T,Map<K,R>> {
008
009        private String name;
010        private Map<K,FunctionFactory<? super T, ? extends R>> elements = new LinkedHashMap<>();
011
012        public MapCompoundFunctionFactory(String name, Map<K,FunctionFactory<T,R>> elements) {
013                this.name = name;
014                this.elements.putAll(elements);
015        }
016        
017        public MapCompoundFunctionFactory(String name) {
018                this.name = name;
019        }
020
021        @Override
022        public Function<T,Map<K,R>> create(Context context) throws Exception {
023                MapCompoundFunction<K, T, R> ret = new MapCompoundFunction<K,T,R>(name);
024                for (Entry<K, FunctionFactory<? super T, ? extends R>> e: elements.entrySet()) {
025                        FunctionFactory<? super T, ? extends R> value = e.getValue();
026                        if (value != null) {
027                                ret.put(e.getKey(), value.create(context));
028                        }
029                }
030                return ret;
031        }
032        
033        public void put(K key, FunctionFactory<? super T, ? extends R> element) {
034                elements.put(key, element);
035        }       
036        
037        public void putAll(Map<? extends K, ? extends FunctionFactory<? super T, ? extends R>> elements) {
038                elements.entrySet().forEach(e -> put(e.getKey(), e.getValue()));
039        }
040
041}
