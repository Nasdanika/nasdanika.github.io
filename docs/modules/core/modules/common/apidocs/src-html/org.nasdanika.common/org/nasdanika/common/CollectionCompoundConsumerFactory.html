001package org.nasdanika.common;
002
003import java.util.ArrayList;
004import java.util.Arrays;
005import java.util.Collection;
006import java.util.List;
007
008/**
009 * Creates a {@link CollectionCompoundConsumer} which passes the argument to all element consumers.
010 * @author Pavel
011 *
012 * @param <T>
013 */
014public class CollectionCompoundConsumerFactory<T> implements ConsumerFactory<T> {
015
016        private String name;
017        private List<ConsumerFactory<? super T>> elements = new ArrayList<>();
018
019        public CollectionCompoundConsumerFactory(String name, Collection<? extends ConsumerFactory<? super T>> elements) {
020                this.name = name;
021                this.elements.addAll(elements);
022        }
023        
024        @SafeVarargs
025        public CollectionCompoundConsumerFactory(String name, ConsumerFactory<? super T>... elements) {
026                this(name, Arrays.asList(elements));
027        }
028
029        @Override
030        public Consumer<T> create(Context context) throws Exception {
031                CollectionCompoundConsumer<T> ret = new CollectionCompoundConsumer<>(name);
032                for (ConsumerFactory<? super T> e: elements) {
033                        ret.add(e == null ? null : e.create(context));
034                }
035                return ret;
036        }
037        
038        public void add(ConsumerFactory<? super T> element) {
039                elements.add(element);
040        }
041        
042        public void addAll(Iterable<? extends ConsumerFactory<? super T>> elements) {
043                elements.forEach(this::add);
044        }
045
046}
