074        /**
075         * Converts name to label text by breaking the name by camel case, capitalizing the first word and lowsercasing the rest.
076         * @return
077         */
078        public static String nameToLabel(String name) {
079                String[] cca = StringUtils.splitByCharacterTypeCamelCase(name);
080                cca[0] = StringUtils.capitalize(cca[0]);
081                for (int i=1; i<cca.length; ++i) {
082                        cca[i] = cca[i].toLowerCase();
083                }
084                return String.join(" ", cca);
085        }

088         * Converts string in camel case, e.g. <code>firstName</code> to kebab case, e.g. <code>first-name</code>
089         * @return
090         */
091        public static String camelToKebab(String str) {
092                String[] ns = StringUtils.splitByCharacterTypeCamelCase(str);
093                for (int i = 0; i < ns.length; ++i) {
094                        ns[i] = ns[i].toLowerCase();
095                }
096                return String.join("-", ns);
097        }

103        /**
104         * Creates a function for context mapping which recognizes path navigation with ..
105         * @param prefix
106         * @return
107         */
108        public static java.util.function.Function<String,String> hierarchicalMapper(String prefix) {            
109                return key -> {
110                        StringTokenizer st = new StringTokenizer(prefix + key, "/", true);
111                        LinkedList<String> collector = new LinkedList<>();
112                        while (st.hasMoreTokens()) {
113                                String token = st.nextToken();
114                                if ("..".equals(token) && collector.size() > 1 && !"..".equals(collector.get(collector.size() - 2))) {
115                                        collector.removeLast();
116                                        collector.removeLast();
117                                        if (st.hasMoreTokens()) {
118                                                st.nextToken(); 
119                                        }
120                                } else {
121                                        collector.add(token);
122                                }
123                        }
124                        StringBuilder sb = new StringBuilder();
125                        collector.forEach(sb::append);
126                        return sb.toString();                   
127                };
128        }

130        public static InputStream join(InputStream... streams) throws IOException {
131                if (streams.length == 1) {
132                        return streams[0];
133                }
134                ByteArrayOutputStream baos = new ByteArrayOutputStream();
135                for (InputStream in: streams) {
136                        if (in != null) {
137                                try (BufferedInputStream bin = new BufferedInputStream(in)) {
138                                        int b;
139                                        while ((b = bin.read()) != -1) {
140                                                baos.write(b);
141                                        }                                               
142                                }
143                        }
144                }
145                baos.close();
146                return new ByteArrayInputStream(baos.toByteArray());
147        }

150                Charset charset = context.get(Charset.class, StandardCharsets.UTF_8);
151                ByteArrayOutputStream baos = new ByteArrayOutputStream();
152                if (text != null) {
153                        try (OutputStreamWriter writer = new OutputStreamWriter(baos, charset)) {
154                                writer.write(text);
155                        }
156                }
157                baos.close();
158                return new ByteArrayInputStream(baos.toByteArray());
159        }

208                if (in == null) {
209                        return "";
210                }
211                StringWriter sw = new StringWriter();
212                Charset charset = context.get(Charset.class, StandardCharsets.UTF_8);
213                try (Reader reader = new InputStreamReader(new BufferedInputStream(in), charset)) {
214                        int ch;
215                        while ((ch = reader.read()) != -1) {
216                                sw.write(ch);
217                        }
218                }
219                sw.close();
220                return sw.toString();
221        }

464                        if (markers != null && !markers.isEmpty()) {

568        /**
569         * Convenience method to evaluate Javascript
570         * @param script Script
571         * @param bindings Optional script bindings
572         * @return Evaluation result
573         * @throws ScriptException 
574         */
575        public static Object eval(String script, Map<String,Object> bindings) throws Exception {
576                org.mozilla.javascript.Context scriptContext = org.mozilla.javascript.Context.enter();
577                try {
578                        Scriptable scope = scriptContext.initStandardObjects();
579                        for (Entry<String, Object> be: bindings.entrySet()) {
580                                ScriptableObject.putProperty(scope, be.getKey(), org.mozilla.javascript.Context.javaToJS(be.getValue(), scope));
581                        }
582
583                        return scriptContext.evaluateString(scope, script, "script", 1, null);
584                } finally {
585                        org.mozilla.javascript.Context.exit();
586                }
587        }

623         * A convenience method to check for presence of required configuration keys
624         * @param config Configuration map.
625         * @param supportedKeys Supported keys.
626         * @param marker Map location.
627         * @return config argument
628         */
629        public static Map<String,Object> checkRequiredKeys(Map<String,Object> config, Collection<String> requiredKeys) throws ConfigurationException {
630                StringBuilder missingKeyList = new StringBuilder();
631                for (String rk: requiredKeys) {
632                        if (!config.containsKey(rk)) {
633                                if (missingKeyList.length() > 0) {
634                                        missingKeyList.append(", ");
635                                }
636                                missingKeyList.append(rk);
637                        }
638                }
639                
640                if (missingKeyList.length() == 0) {
641                        return config;
642                }
643                
644                throw new ConfigurationException("Missing required configuration keys: " + missingKeyList, config instanceof Marked ? ((Marked) config).getMarkers() : null);
645        }

647        /**
648         * A convenience method to check for presence of unsupported configuration keys
649         * @param config Configuration map.
650         * @param supportedKeys Supported keys.
651         * @param marker Map location.
652         * @return config argument.
653         */
654        public static Map<?,?> checkUnsupportedKeys(Map<?,?> config, Collection<?> supportedKeys) throws ConfigurationException {
655                if (config == null) {
656                        return config;
657                }
658                Collection<?> unsupportedKeys = new ArrayList<Object>(config.keySet());
659                unsupportedKeys.removeAll(supportedKeys);
660                if (unsupportedKeys.isEmpty()) {
661                        return config;
662                }               
663                
664                if (unsupportedKeys.size() == 1) {
665                        Object unsupportedKey = unsupportedKeys.iterator().next();
666                        throw new ConfigurationException("Unsupported configuration key: " + unsupportedKey, getMarkers(config, unsupportedKey));
667                }
668                
669                StringBuilder keyList = new StringBuilder();
670                for (Object uk: unsupportedKeys) {
671                        if (keyList.length() > 0) {
672                                keyList.append(", ");
673                        }
674                        keyList.append(uk);
675                }
676                throw new ConfigurationException("Unsupported configuration keys: " + keyList, config instanceof Marked ? ((Marked) config).getMarkers() : null);
677        }

690        /**
691         * Extracts first sentence from text
692         * @param context
693         * @param text
694         * @return
695         * @throws Exception
696         */
697        public static String firstSentence(String text, int minSentenceLength, int maxSentenceLength, String... abbreviations) {
698                if (text == null || text.length() < minSentenceLength) {
699                        return text;
700                }
701                Matcher matcher = SENTENCE_PATTERN.matcher(text);               
702                Z: while (matcher.find()) {
703                        String group = matcher.group();
704                        for (String abbr: abbreviations) {
705                                if (group.trim().endsWith(abbr)) {
706                                        continue Z;
707                                }
708                        }
709                        if (matcher.end() > minSentenceLength && matcher.end() < maxSentenceLength) {
710                                return text.substring(0, matcher.end());
711                        }
712                }
713                
714                return text.length() < maxSentenceLength ? text : text.substring(0, maxSentenceLength)+"...";
715        }

724        public static String firstPlainTextSentence(String html, int minSentenceLength, int maxSentenceLength, String... abbreviations) {
725                if (html == null || html.length() < minSentenceLength) {

1260         * Wraps texts so its line width grows proportionally as the number of lines increases. 
1261         * @param text
1262         * @param initialLineWidth
1263         * @param separator
1264         * @return

1266        public static String wrap(String text, int initialLineWidth, int widthIncrementPerLine, String separator) {
1267                if (Util.isBlank(text) || text.length() < initialLineWidth) { 
1268                        return text;
1269                }
1270                double lines = quadraticRoot(widthIncrementPerLine, initialLineWidth - widthIncrementPerLine, -text.length());

1274        /**
1275         * Inserts separators into a list of elements so when these elements are rendered as text the text is wrapped in such a way that
1276         * line width grows proportionally as the number of lines increases. 
1277         * @param elements Elements to wrap 
1278         * @param initialLineWidth
1279         * @param separatorSupplier creates separators to insert into the resulting list.
1280         * @return
1281         */
1282        public static <T> List<T> wrap(
1283                        List<T> elements, 
1284                        ToIntFunction<T> elementLengthComputer,  
1285                        int initialLineWidth, 
1286                        int widthIncrementPerLine, 
1287                        java.util.function.Supplier<T> wordSeparatorSupplier,                   
1288                        java.util.function.Supplier<T> lineSeparatorSupplier) {
1289                int length = elements.stream().mapToInt(elementLengthComputer).sum();           
1290                int wrapLength = initialLineWidth;
1291                if (length > initialLineWidth) { 
1292                        wrapLength += (int) (widthIncrementPerLine * Math.ceil(quadraticRoot(widthIncrementPerLine, initialLineWidth - widthIncrementPerLine, -length)));
1293                }
1294                List<T> ret = new ArrayList<T>();
1295                int lineLength = 0;
1296                for (T element: elements) {
1297                        if (lineLength > wrapLength) {
1298                                ret.add(lineSeparatorSupplier.get());
1299                                lineLength = 0;
1300                        } 
1301                        if (lineLength > 0) {
1302                                ret.add(wordSeparatorSupplier.get());
1303                        }
1304                        ret.add(element);
1305                        lineLength += elementLengthComputer.applyAsInt(element);
1306                }
1307                return ret;
1308        }

1317         * Java 8 uses package-list, Java 11 uses element-list.

1323        /**
1324         * @param urls Mapping of online URL's (to build links) to offline URL's (to load package lists from).  
1325         * @return Function which given a fully qualified class name returns an anchor linking to Javadoc page for the class. The argument may contain a hash symbol (fragment). 
1326         * Everything after the hash gets transfered to the computed URL providing ability to address fields and methods.
1327         * 
1328         */
1329        public static java.util.function.Function<String, String> createJavadocResolver(Map<URL,URL> urls, ProgressMonitor progressMonitor) {
1330                // package -> Base URL.
1331                Map<String,String> unsortedPackageMap = new HashMap<>();                
1332                progressMonitor.setWorkRemaining(urls.size());
1333                Z: for (Map.Entry<URL, URL> urlEntry: urls.entrySet()) {
1334                        for (String packageListLocation: PACKAGE_LIST_LOCATIONS) {
1335                                try (BufferedReader br = new BufferedReader(new InputStreamReader(new URL(urlEntry.getValue(), packageListLocation).openStream()))) {
1336                                        String module = null;
1337                                        String line;
1338                                        while ((line = br.readLine()) != null) {
1339                                                if (line.startsWith(MODULE_PREFIX)) {
1340                                                        module = line.substring(MODULE_PREFIX.length()); 
1341                                                } else {
1342                                                        String packageLocation = urlEntry.getKey().toString() + (module == null ? "index.html?" : module + "/");
1343                                                        unsortedPackageMap.put(line.trim(), packageLocation);
1344                                                }
1345                                        }
1346                                        progressMonitor.worked(Status.SUCCESS, 1, "Loaded package list from " + urlEntry.getValue() + " for " + urlEntry.getKey());
1347                                        continue Z;
1348                                } catch (Exception e) {
1349                                        progressMonitor.worked(Status.ERROR, 1, "Could not download package list from " + urlEntry.getValue() + " - " + e);
1350                                }
1351                        }
1352                }
1353                
1354                Map<String,String> sortedPackageMap = new LinkedHashMap<>();
1355                unsortedPackageMap.keySet().stream().sorted((k1,k2) -> k2.length() - k1.length()).forEach(k -> sortedPackageMap.put(k, unsortedPackageMap.get(k)));
1356                
1357                return new java.util.function.Function<String, String>() {
1358                        
1359                        private String getRef(String spec) {
1360                                int hashIdx = spec.indexOf("#");
1361                                if (hashIdx != -1) {
1362                                        spec = spec.substring(0, hashIdx);
1363                                }
1364                                for (Entry<String, String> location: sortedPackageMap.entrySet()) {
1365                                        String key = location.getKey();
1366                                        String value = location.getValue();
1367                                        if (key.equals(spec)) { // Package
1368                                                return value + spec.replace('.', '/') + "/" + PACKAGE_SUMMARY_HTML;
1369                                        }
1370                                        
1371                                        int idx = spec.lastIndexOf('.');
1372                                        if (idx != -1 && spec.substring(0, idx).equals(key)) { // Class
1373                                                return value + spec.replace('.', '/').replace('$', '.') + ".html";                              
1374                                        }
1375                                }
1376                                return null;                            
1377                        }
1378
1379                        @Override
1380                        public String apply(String key) {
1381                                if (key == null) {
1382                                        return key;
1383                                }
1384                                
1385                                int spaceIdx = key.indexOf(' ');
1386                                String spec = spaceIdx == -1 ? key : key.substring(0, spaceIdx);
1387                                String text = spaceIdx == -1 ? null : key.substring(spaceIdx + 1);
1388                                
1389                                String href = getRef(spec);
1390                                if (href == null) {
1391                                        return spec;
1392                                }
1393                                                                
1394                                int hashIdx = spec.indexOf("#");
1395                                if (hashIdx != -1) {
1396                                        String fragment = spec.substring(hashIdx+1);
1397                                        int firstParenthesisIdx = fragment.indexOf("(");
1398                                        if (firstParenthesisIdx > 0 && fragment.endsWith(")")) {
1399                                                // Convert (type[,type]) to -type-type-                                         
1400                                                StringBuilder fragmentBuilder = new StringBuilder(fragment.substring(0, firstParenthesisIdx)).append("-");
1401                                                String[] pTypes = fragment.substring(firstParenthesisIdx+1, fragment.length()-1).split(",");
1402                                                for (String pType: pTypes) {
1403                                                        pType = pType.trim();
1404                                                        int bIdx = pType.indexOf("[]");
1405                                                        if (bIdx == -1) {
1406                                                                fragmentBuilder.append(pType);
1407                                                        } else {
1408                                                                fragmentBuilder.append(pType.substring(0, bIdx));
1409                                                                for (int bc = bIdx; bc < pType.length(); bc+=2) {
1410                                                                        fragmentBuilder.append(":A");
1411                                                                }
1412                                                        }
1413                                                        fragmentBuilder.append("-");
1414                                                }
1415                                                fragment = fragmentBuilder.toString();
1416                                        } 
1417                                        href += "#" + fragment;
1418                                }
1419                                return new StringBuilder("<a href='")
1420                                                .append(href)
1421                                                .append("'>")
1422                                                .append(Util.isBlank(text) ? spec.replace('#', '.').replace('$', '.') : text)
1423                                                .append("</a>").toString();
1424                        }
1425                        
1426                };
1427                
1428        }

1510         * Walks the directory passing files and their paths to the listener.
1511         * @param source
1512         * @param target
1513         * @param cleanTarget
1514         * @param cleanPredicate
1515         * @param listener
1516         * @throws IOException
1517         */
1518        public static void walk(String path, BiConsumer<File,String> listener, File... files) {
1519                if (files != null) {
1520                        for (File file: files) {
1521                                String filePath = path == null ? file.getName() : path + "/" + file.getName();
1522                                if (file.isDirectory()) {
1523                                        walk(filePath, listener, file.listFiles());
