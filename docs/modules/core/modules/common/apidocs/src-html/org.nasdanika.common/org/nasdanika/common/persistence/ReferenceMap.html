001package org.nasdanika.common.persistence;
002
003import java.util.List;
004import java.util.Map;
005
006import org.eclipse.emf.common.util.URI;
007import org.nasdanika.common.ProgressMonitor;
008
009/**
010 * Loads a map of references using loader.
011 * @author Pavel
012 *
013 */
014public class ReferenceMap<K,V> extends MapAttribute<K,V> {
015
016        public ReferenceMap(Object key, 
017                        boolean isDefault, 
018                        boolean required, 
019                        Map<K,V> defaultValue, 
020                        String description, 
021                        Object... exclusiveWith) {
022                super(key, isDefault, required, defaultValue, description, exclusiveWith);                              
023        }               
024        
025        @SuppressWarnings("unchecked")
026        @Override
027        protected V createValue(ObjectLoader loader, K key, Object value, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) throws Exception {
028                return (V) loader.load(value, base, progressMonitor);
029        }
030
031}
