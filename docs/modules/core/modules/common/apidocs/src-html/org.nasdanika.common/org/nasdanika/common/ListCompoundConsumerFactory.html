001package org.nasdanika.common;
002
003import java.util.ArrayList;
004import java.util.Arrays;
005import java.util.List;
006
007/**
008 * Creates a {@link ListCompoundConsumer} which passes the argument element to element consumer with the same index, if present.
009 * @author Pavel
010 *
011 * @param <T>
012 */
013public class ListCompoundConsumerFactory<T> implements ConsumerFactory<List<T>> {
014
015        private String name;
016        private List<ConsumerFactory<? super T>> elements = new ArrayList<>();
017
018        public ListCompoundConsumerFactory(String name, List<? extends ConsumerFactory<? super T>> elements) {
019                this.name = name;
020                this.elements.addAll(elements);
021        }
022        
023        @SafeVarargs
024        public ListCompoundConsumerFactory(String name, ConsumerFactory<? super T>... elements) {
025                this(name, Arrays.asList(elements));
026        }
027
028        @Override
029        public Consumer<List<T>> create(Context context) throws Exception {
030                ListCompoundConsumer<T> ret = new ListCompoundConsumer<>(name);
031                for (ConsumerFactory<? super T> e: elements) {
032                        ret.add(e == null ? null : e.create(context));
033                }
034                return ret;
035        }
036        
037        public void add(ConsumerFactory<? super T> element) {
038                elements.add(element);
039        }
040        
041        public void addAll(Iterable<? extends ConsumerFactory<? super T>> elements) {
042                elements.forEach(this::add);
043        }
044
045}
