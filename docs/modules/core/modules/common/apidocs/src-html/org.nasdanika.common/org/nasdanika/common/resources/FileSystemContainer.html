001package org.nasdanika.common.resources;
002
003import java.util.Collections;
004import java.util.HashMap;
005import java.util.Map;
006
007import org.nasdanika.common.ProgressMonitor;
008
009/**
010 * Wrapper for a file system directory.
011 * @author Pavel
012 *
013 */
014public class FileSystemContainer extends FileSystemResource implements BinaryEntityContainer {
015
016        public FileSystemContainer(java.io.File file) {
017                super(file);
018        }
019
020        protected FileSystemResource getChild(String name, ProgressMonitor monitor) {
021                return (FileSystemResource) getChildren(monitor).get(name);
022        }
023
024        @Override
025        public FileSystemResource find(String path, ProgressMonitor monitor) {
026                int sPos = path.indexOf(SEPARATOR);
027                if (sPos == -1) {
028                        return getChild(path, monitor); 
029                }
030                String firstSegment = path.substring(0, sPos);
031                FileSystemResource child = getChild(firstSegment, monitor.split("Getting child at "+firstSegment, 1, this));
032                if (child instanceof FileSystemContainer) {
033                        String pathTail = path.substring(sPos+1);
034                        return ((FileSystemContainer) child).find(pathTail, monitor.split("Finding "+pathTail, 1, child));
035                }
036                return null;
037        }
038
039        @Override
040        public FileSystemEntity get(String path, ProgressMonitor monitor) {
041                FileSystemResource existing = find(path, monitor.split("Finding "+path, 1, this));
042                if (existing instanceof FileSystemEntity) {
043                        return (FileSystemEntity) existing;
044                }
045                if (existing instanceof FileSystemContainer) {
046                        // container - can't have an entity with the same name.
047                        return null;
048                }
049                int sPos = path.indexOf(SEPARATOR);
050                if (sPos == -1) {
051                        return new FileSystemEntity(new java.io.File(file, path));
052                }
053                
054                String firstSegment = path.substring(0, sPos);
055                FileSystemContainer container = getContainer(firstSegment, monitor.split("Getting container "+firstSegment, 1, this));
056                String pathTail = path.substring(sPos + 1);
057                return container == null ? null : container.get(pathTail, monitor.split("Getting entity at "+pathTail, 1, container));
058        }
059
060        @Override
061        public FileSystemContainer getContainer(String path, ProgressMonitor monitor) {
062                FileSystemResource existing = find(path, monitor.split("Finding existing container", 1, this));
063                if (existing instanceof FileSystemContainer) {
064                        return (FileSystemContainer) existing;
065                }
066                if (existing instanceof FileSystemEntity) {
067                        // entity - can't have a container with the same name.
068                        return null;
069                }
070                int sPos = path.indexOf(SEPARATOR);
071                if (sPos == -1) {
072                        return new FileSystemContainer(new java.io.File(file, path));
073                }
074                
075                String pathHead = path.substring(0, sPos);
076                FileSystemContainer container = getContainer(pathHead, monitor.split("Getting child container "+pathHead, 1, this));
077                String pathTail = path.substring(sPos + 1);
078                return container == null ? null : container.getContainer(pathTail, monitor.split("Getting child container "+pathHead, 1, container));
079        }
080
081        /**
082         * Returned values are instances of {@link FileSystemResource}, so the return value can be cast to Map&lt;String, FileSystemResource&gt;
083         */
084        @Override
085        public Map<String, Object> getChildren(ProgressMonitor monitor) {
086                Map<String, FileSystemResource> ret = new HashMap<>();          
087                java.io.File[] children = file.listFiles();
088                if (children != null) {
089                        for (java.io.File child: children) {
090                                if (child.isFile()) {
091                                        ret.put(child.getName(), new FileSystemEntity(child));
092                                } else if (child.isDirectory()) {
093                                        ret.put(child.getName(), new FileSystemContainer(child));
094                                }
095                        }
096                }
097                return Collections.unmodifiableMap(ret);
098        }
099
100        @Override
101        public void copy(Container<? super BinaryEntity> container, String path, ProgressMonitor monitor) {
102                throw new UnsupportedOperationException("Implement when needed");               
103        }
104
105        @Override
106        public void move(Container<? super BinaryEntity> container, String path, ProgressMonitor monitor) {
107                throw new UnsupportedOperationException("Implement when needed");               
108        }
109
110        @Override
111        public void put(String path, BinaryEntity element, ProgressMonitor monitor) throws IllegalArgumentException {
112                throw new UnsupportedOperationException("Use get() and getContainer() to operate with file system resources");          
113        }
114
115        @Override
116        public void delete(ProgressMonitor monitor) {
117                for (Object child: getChildren(monitor.split("Getting children", 1, this)).values()) {
118                        ((FileSystemResource) child).delete(monitor.split("Deleting child", 1, child));
119                }
120                super.delete(monitor);
121        }
122
123        @Override
124        public Object delete(String path, ProgressMonitor monitor) {
125                FileSystemResource toDelete = get(path, monitor.split("Getting resource to delete: "+path, 1, this));
126                if (toDelete != null) {
127                        toDelete.delete(monitor.split("Deleting "+path, 1, toDelete));
128                }
129                return toDelete;
130        }

132}
