001package org.nasdanika.common.persistence;
002
003import java.util.List;
004
005import org.eclipse.emf.common.util.URI;
006import org.nasdanika.common.ProgressMonitor;
007
008/**
009 * Delegates feature creation to a factory, e.g. a constructor.
010 * @author Pavel
011 *
012 */
013public class FactoryFeature<T> extends Attribute<T> {
014
015        private ObjectFactory<T> factory;
016
017        public FactoryFeature(
018                        Object key,
019                        boolean isDefault,
020                        boolean required, 
021                        T defaultValue,
022                        String description,
023                        ObjectFactory<T> factory, 
024                        Object... exclusiveWith) {
025                
026                super(key, isDefault, required, defaultValue, description, exclusiveWith);
027                this.factory = factory;
028        }
029        
030        @Override
031        public T create(ObjectLoader loader, Object config, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers)  throws Exception {
032                return factory.create(loader, config, base, progressMonitor, markers);
033        }
034
035}
