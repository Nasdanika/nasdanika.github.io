001package org.nasdanika.common.persistence;
002
003import java.util.List;
004import java.util.function.Supplier;
005
006import org.eclipse.emf.common.util.URI;
007import org.nasdanika.common.ProgressMonitor;
008
009/**
010 * If config is a list loads each element creating elements using element factory and then loading them, otherwise creates a singleton in the same way as explained before.
011 * @author Pavel
012 *
013 * @param <T>
014 */
015public class FeatureObjectListAttribute<T extends FeatureObject> extends ListAttribute<T> {
016
017        private Supplier<T> elementFactory;
018
019        public FeatureObjectListAttribute(
020                        Object key, 
021                        Supplier<T> elementFactory,
022                        boolean isDefault, 
023                        boolean required, 
024                        List<T> defaultValue, 
025                        String description, 
026                        Object... exclusiveWith) {
027                super(key, isDefault, required, defaultValue, description, exclusiveWith);
028                this.elementFactory = elementFactory;
029        }
030
031        /**
032         * Creates element list.
033         * @param loader
034         * @param element
035         * @param base
036         * @param progressMonitor
037         * @param marker
038         * @return
039         * @throws Exception
040         */
041        @Override
042        protected T createElement(ObjectLoader loader, Object element, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers)       throws Exception { 
043                T ret = elementFactory.get();
044                ret.load(loader, element, base, progressMonitor, markers);
045                return ret;
046        }
047        
048}
