001package org.nasdanika.common.resources;
002
003import org.nasdanika.common.ProgressMonitor;
004
005/**
006 * Resource typically represents something with a lifespan longer than the JVM. It is  uniquely identified by its name within its container and its path within the root container.
007 * The resource path separator is ``/``. 
008 * @author Pavel
009 *
010 * @param <E> Element type for the parent container.
011 */
012public interface Resource<E> {
013        
014        public static final String SEPARATOR = "/";
015
016        /**
017         * @return Resource name - the last segment of the path.
018         */
019        String getName();
020        
021        /**
022         * @return true if resource exists.
023         */
024        boolean exists(ProgressMonitor monitor);
025        
026        /**
027         * @return Parent container.
028         */
029        Container<E> getParent();
030        
031        /**
032         * Resource size. -1 for unknown.
033         * @return
034         */
035        long size(ProgressMonitor monitor);
036        
037        /**
038         * @return The root container.
039         */
040        default Container<E> getRoot() {
041                Container<E> parent = getParent();
042                if (parent != null) {
043                        return parent.getRoot();
044                }
045                
046                return this instanceof Container ? (Container<E>) this : null;
047        }
048        
049        /**
050         * Deletes the resource if it exists.
051         * @param monitor Progress monitor.
052         */
053        void delete(ProgressMonitor monitor);
054        
055        /**
056         * Resource path from the root container.
057         * @return
058         */
059        default String getPath() {
060                Container<E> parent = getParent();
061                if (parent == null) {
062                        return getName();
063                }
064                String parentPath = parent.getPath();
065                return parentPath == null || parentPath.isEmpty() ? getName() : parentPath + Container.SEPARATOR + getName();
066        };
067        
068        /**
069         * Copies the resource to the target container under a given path.
070         * @param container Target container, may be the same container.
071         * @param path Path to copy the resource to.
072         * @param monitor Parent progress monitor, the resource splits the monitor and takes size() ticks.
073         */
074        void copy(Container<? super E> container, String path, ProgressMonitor monitor);
075        
076        /**
077         * Moves the resource to the target container under a given path. Equivalent to copying and then deleting.
078         * @param container Target container.
079         * @param path Path to move the resource to.
080         * @param monitor Parent progress monitor, the resource splits the monitor and takes 2*size() ticks - one size for reading and one for writing.
081         */
082        void move(Container<? super E> container, String path, ProgressMonitor monitor);
083        
084//      /**
085//       * Adapts to a different resource type. 
086//       * @param <V> New resource type. 
087//       * Either encoder or decoder can be null, it will make the adapted entities read-only and write-only respectively. Decoder and encoder take the entity as 
088//       * the first argument to allow implement entity-specific operations, e.g. extension-based encoding/decoding such as parsing JSON for entities with ``.json`` extension.
089//       * @param decoder Decodes T to V.
090//       * @param encoder Encodes V to T.
091//       * @param sizeConverter converts size of an entity. Size is passed as-is if null.
092//       * @return
093//       */
094//      <V extends Resource<?>> Resource<V> adapt(FunctionFactory<T,V> adapter, BiFunction<Entity<T>, V, T> encoder, BiFunction<Entity<T>, Long, Long> sizeConverter);
095        
096        /**
097         * @return Resource digest. Entities compute state digest, e.g. SHA256 digest of file contents. Containers derive digest value from their children digests. 
098         */
099        default String getDigest() {
100                return null;
101        }
102        
103        /**
104         * @return Time when this resource was last modified, -1 if unknown. For containers modification means a change in the list of children, not necessarily change in the
105         * children's state.
106         */
107        default long getTimestamp() {
108                return -1;
109        }
110        
111}
