001package org.nasdanika.common.persistence;
002
003import java.util.Collections;
004import java.util.Iterator;
005import java.util.List;
006import java.util.Map;
007import java.util.Map.Entry;
008
009import org.yaml.snakeyaml.DumperOptions;
010import org.yaml.snakeyaml.LoaderOptions;
011import org.yaml.snakeyaml.Yaml;
012import org.yaml.snakeyaml.constructor.Constructor;
013import org.yaml.snakeyaml.nodes.MappingNode;
014import org.yaml.snakeyaml.nodes.NodeTuple;
015import org.yaml.snakeyaml.nodes.ScalarNode;
016import org.yaml.snakeyaml.nodes.SequenceNode;
017import org.yaml.snakeyaml.representer.Representer;
018import org.yaml.snakeyaml.resolver.Resolver;
019
020/**
021 * Snake YAML {@link Constructor} which creates {@link MarkedArrayList} and {@link MarkedLinkedHashMap} which allows to obtain locations of of elements in the source file using {@link Marker}s.
022 * @author Pavel Vlasov
023 *
024 */
025public class MarkingYamlConstructor extends Constructor {
026        
027        private String location;
028
029        public MarkingYamlConstructor(String location) {
030                this.location = location;
031        }
032        
033        @Override
034        protected List<? extends Object> constructSequence(SequenceNode node) {
035                MarkedArrayList<? extends Object> ret = (MarkedArrayList<? extends Object>) super.constructSequence(node);
036                node.getValue().forEach(n -> ret.getElementMarkers().add(Collections.singletonList(new MarkerImpl(location, n.getStartMark()))));
037                Iterator<? extends Object> eit = ret.iterator();
038                Iterator<List<? extends Marker>> mit = ret.getElementMarkers().iterator();
039                while (eit.hasNext() && mit.hasNext()) {
040                        Object e = eit.next();
041                        List<? extends Marker> ml = mit.next();
042                        if (e instanceof Markable) {
043                                ((Markable) e).mark(ml);
044                        }
045                }
046                return ret;
047        }
048        
049        @Override
050        protected List<Object> newList(SequenceNode node) {
051                return new MarkedArrayList<Object>();
052        }
053                                
054        @Override
055        protected Map<Object, Object> constructMapping(MappingNode node) {
056                MarkedLinkedHashMap<Object, Object> ret = (MarkedLinkedHashMap<Object, Object>) super.constructMapping(node);
057                node.getValue()
058                        .stream()
059                        .map(NodeTuple::getKeyNode)
060                        .filter(n -> n instanceof ScalarNode)
061                        .forEach(keyNode -> ret.markEntry(((ScalarNode) keyNode).getValue(), new MarkerImpl(location, keyNode.getStartMark())));
062                
063                for (Entry<Object, Object> e: ret.entrySet()) {
064                        if (e.getValue() instanceof Markable) {
065                                ((Markable) e.getValue()).mark(ret.getEntryMarkers(e.getKey()));
066                        }
067                }
068                return ret;
069        }
070        
071        @Override
072        protected Map<Object, Object> newMap(MappingNode node) {
073                return new MarkedLinkedHashMap<Object, Object>();
074        }                                       
075                
076        /**
077         * @return Yaml which uses this {@link MarkingYamlConstructor} to create collections with markers.
078         */
079        public static Yaml createMarkingYaml(String location) {
080                LoaderOptions loaderOptions = new LoaderOptions();
081                loaderOptions.setAllowDuplicateKeys(false);
082                return new Yaml(new MarkingYamlConstructor(location), new Representer(), new DumperOptions(), loaderOptions, new Resolver());
083        }
084
085}
