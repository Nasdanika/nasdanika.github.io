114        /**
115         * @param markdown Markdown to pre-process
116         * @param replacements Accumulator of post-processing replacements token -&gt; HTML to replace the token with.
117         * @return
118         */
119        protected String preProcessMarkdown(String markdown, Map<String,String> replacements) {
120                if (isProcessFendedBlocks()) {
121                        String ret = processFencedBlocks(markdown, DiagramGenerator.UML_DIALECT, START_UML_PATTERN, replacements, false);
122                        ret = processFencedBlocks(ret, DiagramGenerator.SALT_DIALECT, START_WIREFRAME_PATTERN, replacements, false);
123                        ret = processFencedBlocks(ret, DiagramGenerator.GANTT_DIALECT, START_GANTT_PATTERN, replacements, false);
124                        ret = processFencedBlocks(ret, DiagramGenerator.MINDMAP_DIALECT, START_MINDMAP_PATTERN, replacements, false);
125                        ret = processFencedBlocks(ret, DiagramGenerator.WBS_DIALECT, START_WBS_PATTERN, replacements, false);
126                        ret = processFencedBlocks(ret, DiagramGenerator.DRAWIO_DIALECT, START_DRAWIO_PATTERN, replacements, false);
127                        ret = processFencedBlocks(ret, DiagramGenerator.MERMAID_DIALECT, START_MERMAID_PATTERN, replacements, false);
128                        ret = processFencedBlocks(ret, null, START_PNG_PATTERN, replacements, false);
129                        ret = processFencedBlocks(ret, null, START_JPEG_PATTERN, replacements, false);
130                                        
131                        ret = processFencedBlocks(ret, DiagramGenerator.UML_DIALECT, START_UML_RESOURCE_PATTERN, replacements, true);
132                        ret = processFencedBlocks(ret, DiagramGenerator.SALT_DIALECT, START_WIREFRAME_RESOURCE_PATTERN, replacements, true);
133                        ret = processFencedBlocks(ret, DiagramGenerator.GANTT_DIALECT, START_GANTT_RESOURCE_PATTERN, replacements, true);
134                        ret = processFencedBlocks(ret, DiagramGenerator.MINDMAP_DIALECT, START_MINDMAP_RESOURCE_PATTERN, replacements, true);
135                        ret = processFencedBlocks(ret, DiagramGenerator.WBS_DIALECT, START_WBS_RESOURCE_PATTERN, replacements, true);
136                        ret = processFencedBlocks(ret, DiagramGenerator.DRAWIO_DIALECT, START_DRAWIO_RESOURCE_PATTERN, replacements, true);
137                        ret = processFencedBlocks(ret, DiagramGenerator.MERMAID_DIALECT, START_MERMAID_RESOURCE_PATTERN, replacements, true);
138                        ret = processFencedBlocks(ret, null, START_PNG_RESOURCE_PATTERN, replacements, true);
139                        ret = processFencedBlocks(ret, null, START_JPEG_RESOURCE_PATTERN, replacements, true);
140                                        
141                        return ret;             
142                }
143                                
144                return markdown;
145        }

147        /**
148         * 
149         * @param input
150         * @param dialect
151         * @param startPattern
152         * @param replacements
153         * @param resource Indicates that spec is a URL of a resource resolved relative to the URL returned by getResourceBase(). The URL supports <code>classpath</code> schema for classloader resources.
154         * @return
155         */
156        private String processFencedBlocks(String input, String dialect, Pattern startPattern, Map<String,String> replacements, boolean resource) {
157                StringBuilder output = new StringBuilder();
158                Matcher startMatcher = startPattern.matcher(input);
159                int i = 0;
160                while (startMatcher.find()) {
161                        int startMatcherEnd = startMatcher.end();
162                        String endMatcherContent = input.substring(startMatcherEnd);
163                        Matcher endMatcher = FENCED_BLOCK_PATTERN.matcher(endMatcherContent);                   
164                        if (endMatcher.find()) {
165                            output.append(input.substring(i, startMatcher.start()));
166                                String bareSpec = processSpec(dialect, endMatcherContent.substring(0, endMatcher.start())) ;
167                                String match = endMatcherContent.substring(endMatcher.start(), endMatcher.end());
168                            i = startMatcherEnd + endMatcher.start() + match.indexOf("```") + 3; // Just the closing back-ticks, no space or new line characters.
169                                
170                                if (resource) {                                 
171                                        try {           
172                                                URI resourceURI = URI.createURI(bareSpec);
173                                                URI resourceBase = getResourceBase();
174                                                if (resourceBase != null && resourceBase.hasAbsolutePath()) {
175                                                        resourceURI = resourceURI.resolve(resourceBase);
176                                                }
177                                                bareSpec = loadResource(resourceURI, dialect == null);
178                                        } catch (Exception e) {                                 
179                                                output
180                                                        .append(System.lineSeparator())
181                                                        .append("<div class=\"nsd-error\">").append(System.lineSeparator())
182                                                        .append("Error loading resource " + bareSpec + ": " + e) 
183                                                        .append(System.lineSeparator()).append("</div>")
184                                                        .append(System.lineSeparator());
185                                                bareSpec = null;
186                                        }
187                                }
188                                
189                                if (bareSpec != null) {
190                                        String escapedBareSpec = escapeDiagramSpec(dialect, bareSpec);
191                                        
192                                        StringBuilder replacementBuilder = new StringBuilder();
193                                        if (!Util.isBlank(escapedBareSpec)) {
194                                                replacementBuilder
195                                                        .append(System.lineSeparator())
196                                                        .append("<div style='display:none;white-space:pre-wrap' title='Diagram escaped spec for search'>").append(System.lineSeparator())
197                                                        .append(escapedBareSpec).append(System.lineSeparator()) 
198                                                        .append("</div> ").append(System.lineSeparator());
199                                        }
200                                        
201                                        try {
202                                                String token = nextToken();
203                                                
204                                                if (dialect == null) {
205                                                        replacementBuilder.append("<img src=\"data:image/");
206                                                        if (startPattern == START_PNG_PATTERN || startPattern == START_PNG_RESOURCE_PATTERN) {
207                                                                replacementBuilder.append("png");
208                                                        } else if (startPattern == START_JPEG_PATTERN || startPattern == START_JPEG_RESOURCE_PATTERN) {
209                                                                replacementBuilder.append("jpeg");
210                                                        }
211                                                        replacementBuilder
212                                                                .append(";base64, ")
213                                                                .append(bareSpec)
214                                                                .append("\"/>")
215                                                                .append(System.lineSeparator());
216                                                } else {
217                                                        replacementBuilder
218                                                                .append(getDiagramGenerator().generateDiagram(bareSpec, dialect))
219                                                                .append(System.lineSeparator());
220                                                }
221                                                
222                                                replacements.put(token, replacementBuilder.toString());
223                                                output
224                                                        .append(System.lineSeparator())
225                                                        .append(System.lineSeparator())
226                                                        .append(token)
227                                                        .append(System.lineSeparator())
228                                                        .append(System.lineSeparator());
229                                        } catch (Exception e) {
230                                                output.append("Error during diagram rendering: " + e);
231                                        }
232                                }                               
233                        }
234                }
235                output.append(input.substring(i, input.length()));
236                return output.toString();
237        }

249        protected String escapeDiagramSpec(String dialect, String bareSpec) {
250                if (dialect == null) {
251                        return null; // For PNG and JPEG
252                }
253                if (DiagramGenerator.DRAWIO_DIALECT.equals(dialect)) {
254                        // TODO - Extract text from XML, return null for now.
255                        // Not needed with Nasdanika search - it extracts text from Drawio diagrams.
256                        return null;
257                }
258                if (DiagramGenerator.MERMAID_DIALECT.equals(dialect)) {
259                        // No need - spec is available as plain text. 
260                        return null;
261                }
262                return StringEscapeUtils.escapeHtml4(bareSpec).trim();
263        }

269        /**
270         * Loads resource and optionally encodes it to Base64.
271         * @param resource
272         * @param encode
273         * @return
274         * @throws Exception
275         */
276        protected String loadResource(URI resource, boolean encode) throws Exception {
277                if (encode) {
278                        byte[] bytes = DefaultConverter.INSTANCE.toByteArray(resource);
279                        return Base64.getEncoder().encodeToString(bytes);
280                }
281                return DefaultConverter.INSTANCE.stringContent(resource);
282        }

314        public String markdownToHtml(String markdown) {
315                if (Util.isBlank(markdown)) {
316                        return "";
317                }
318                Map<String,String> replacements = new HashMap<>();
319                Document document = createMarkdownParser().parse(preProcessMarkdown(markdown, replacements));
320                String html = createMarkdownHtmlRenderer().render(document);
321                for (Entry<String, String> te: replacements.entrySet()) {
322                        html = html.replace(te.getKey(), te.getValue());
323                }
324                return html;
325        }
