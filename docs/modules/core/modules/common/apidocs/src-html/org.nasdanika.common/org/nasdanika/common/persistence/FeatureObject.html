001package org.nasdanika.common.persistence;
002
003import java.util.ArrayList;
004import java.util.Collections;
005import java.util.List;
006import java.util.Map;
007import java.util.Optional;
008import java.util.stream.Collectors;
009
010import org.eclipse.emf.common.util.URI;
011import org.nasdanika.common.ProgressMonitor;
012import org.nasdanika.common.Util;
013
014/**
015 * Object which loads its configuration to features identified by keys.
016 * @author Pavel
017 *
018 */
019public class FeatureObject implements Marked, Loadable {
020        
021        private List<? extends Marker> markers = new ArrayList<>();
022        
023        @Override
024        public List<? extends Marker> getMarkers() {
025                return markers;
026        }
027        
028        protected List<Feature<?>> features = new ArrayList<>();
029
030        /**
031         * Adds feature to the feature list.
032         * @param <F>
033         * @param feature
034         * @return
035         */
036        protected <F extends Feature<?>> F addFeature(F feature) {
037                if (feature.isDefault() && features.stream().filter(Feature::isDefault).findFirst().isPresent()) {
038                        throw new IllegalArgumentException("Duplicate default feature: " + feature.getKey());
039                }
040                if (features.stream().filter(o -> feature.getKey().equals(o.getKey())).findFirst().isPresent()) {
041                        throw new IllegalArgumentException("Duplicate feature: " + feature.getKey());
042                }
043                features.add(feature);
044                return feature;
045        }
046
047        /**
048         * Loads object
049         * @param loader
050         * @param config
051         * @param base
052         * @param progressMonitor
053         * @param marker
054         * @return this object.
055         * @throws Exception
056         */
057        @Override
058        public void load(ObjectLoader loader, Object config, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) throws Exception {
059                this.markers = markers;                 
060                if (config instanceof Map) {
061                        Map<?,?> configMap = (Map<?,?>) config;
062                        Util.checkUnsupportedKeys(configMap, features.stream().map(Feature::getKey).collect(Collectors.toList()));
063                        for (Feature<?> feature: features) {
064                                feature.load(loader, configMap, base, progressMonitor, markers);
065                        }
066                } else {
067                        Optional<Feature<?>> defaultFeatureOptional = features.stream().filter(Feature::isDefault).findFirst();
068                        if (defaultFeatureOptional.isPresent()) {                               
069                                Feature<?> defaultFeature = defaultFeatureOptional.get();
070                                // Singleton map with a single entry default feature key -> config.
071                                defaultFeature.load(loader, Collections.singletonMap(defaultFeature.getKey(), config), base, progressMonitor, markers);
072                        } else if (!Boolean.TRUE.equals(config)) {
073                                throw new ConfigurationException(getClass().getName() + " configuration shall be a map, got " + config, markers);
074                        }
075                }
076        }                       
077
078}
