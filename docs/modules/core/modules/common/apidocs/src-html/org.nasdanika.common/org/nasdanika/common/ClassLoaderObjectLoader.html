001package org.nasdanika.common;
002
003import java.lang.reflect.Constructor;
004import java.net.URL;
005import java.util.List;
006
007import org.eclipse.emf.common.util.URI;
008import org.nasdanika.common.persistence.Marker;
009import org.nasdanika.common.persistence.ObjectLoader;
010
011/**
012 * Creates object instances with {@link ClassLoader}. Type shall resolve to fully qualified class name.
013 * Classes shall have a 5-argument constructors taking {@link ObjectLoader}, config object, {@link URL}, {@link ProgressMonitor}, and {@link Marker}. 
014 * @author Pavel
015 *
016 */
017public class ClassLoaderObjectLoader implements ObjectLoader {
018        
019        private ClassLoader classLoader;
020        private java.util.function.Function<String, String> resolver;
021        private ObjectLoader chain;
022
023        /**
024         * 
025         * @param classLoader {@link ClassLoader} for loading classes.
026         * @param resolver Resolver of type to fully qualified class name. If null, then type is used AS-IS. E.g. a map::get for a map of logical names to fully qualified class names.
027         * @param chain Object loader to fall back to if resolver is not null and returned null. 
028         */
029        public ClassLoaderObjectLoader(ClassLoader classLoader, java.util.function.Function<String,String> resolver, ObjectLoader chain) {
030                this.resolver = resolver;
031                this.classLoader = classLoader;
032                this.chain = chain;
033        }
034
035        @Override
036        public Object create(ObjectLoader loader, String type, Object config, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) throws Exception {
037                String fqn = resolver == null ? type : resolver.apply(type);
038                if (fqn == null) {
039                        if (resolver == null) {
040                                throw new NullPointerException("Type name is null");
041                        }
042                        if (chain == null) {
043                                throw new IllegalArgumentException("Type was resolved to null and there is no chain loader: " + type);
044                        }
045                        
046                        return chain.create(loader, type, config, base, progressMonitor, markers);
047                }
048                
049                Class<?> clazz = classLoader.loadClass(fqn);            
050                Constructor<?> constructor = clazz.getConstructor(ObjectLoader.class, Object.class, URL.class, ProgressMonitor.class, Marker.class);            
051                return constructor.newInstance(loader, config, base, progressMonitor, markers);
052        }
053
054}
