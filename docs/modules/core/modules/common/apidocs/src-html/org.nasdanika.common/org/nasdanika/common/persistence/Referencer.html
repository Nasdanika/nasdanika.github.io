001package org.nasdanika.common.persistence;
002
003import java.util.function.Consumer;
004
005/**
006 * Creates a reference from source to target.
007 * @author Pavel Vlasov
008 * 
009 * @param <T> Persistent object type.
010 * @param <R> Reference type.
011 */
012public interface Referencer<T,R> {
013        
014        /**
015         * Creates a reference
016         * @param source Source object.
017         * @param target Target object.
018         * @param referenceConsumer ConsumerFactory of a reference. This consumer may be invoked as part of reference() method
019         * or later, once all objects are stored, in the case of a reference to a not yet stored object if referencing requires an object to be already stored.
020         */
021        void reference(T source, T target, Consumer<R> refernceConsumer);
022
023}
