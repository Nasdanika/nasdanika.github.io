001package org.nasdanika.common.resources;
002
003import org.nasdanika.common.ProgressMonitor;
004
005/**
006 * {@link MapContainer} of {@link EphemeralEntity}ies.
007 * @author Pavel
008 *
009 * @param <T>
010 */
011public class EphemeralBinaryEntityContainer extends MapContainer<BinaryEntity> implements BinaryEntityContainer {
012
013        @Override
014        protected EphemeralBinaryEntity createElement(String path, ProgressMonitor monitor) {
015                return new EphemeralBinaryEntity() {
016
017                        @Override
018                        public String getName() {
019                                return path;
020                        }
021
022                        @Override
023                        public EphemeralBinaryEntityContainer getParent() {
024                                return EphemeralBinaryEntityContainer.this;
025                        }
026                        
027                };
028        }
029
030        @Override
031        protected MapContainer<BinaryEntity> createSubContainer(String path) {
032                return new EphemeralBinaryEntityContainer() {
033
034                        @Override
035                        public String getName() {
036                                return path;
037                        }
038
039                        @Override
040                        public EphemeralBinaryEntityContainer getParent() {
041                                return EphemeralBinaryEntityContainer.this;
042                        }
043                        
044                };
045
046        }
047        
048        
049        @Override
050        public BinaryEntityContainer getParent() {
051                return (BinaryEntityContainer) super.getParent();
052        }
053        
054        @Override
055        public BinaryEntityContainer getContainer(String path, ProgressMonitor monitor) {
056                return (BinaryEntityContainer) super.getContainer(path, monitor);
057        }
058        
059        @Override
060        public BinaryResource find(String path, ProgressMonitor monitor) {
061                return (BinaryResource) super.find(path, monitor);
062        }
063        
064}
