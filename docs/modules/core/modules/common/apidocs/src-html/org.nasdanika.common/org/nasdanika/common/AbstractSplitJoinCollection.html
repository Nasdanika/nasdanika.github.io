001package org.nasdanika.common;
002
003import java.util.AbstractCollection;
004import java.util.ArrayList;
005import java.util.Collections;
006import java.util.Iterator;
007import java.util.List;
008
009/**
010 * Collection which splits the underlying state S into a sequence of chunks C which then get converted to element type E.
011 * The class itself has no state - it always retrieves state, splits, performs an operation and then joins and sets state in case of mutations.
012 * For this reason there is no way to update an element. For sets you may add it again, for lists set at index. 
013 * @author Pavel
014 *
015 * @param <S> State type
016 * @param <C> Chunk type
017 * @param <E> Element type
018 */
019public abstract class AbstractSplitJoinCollection<S,C,E> extends AbstractCollection<E> {
020        
021        /**
022         * @return State to be split into chunks.
023         */
024        protected abstract S getState();
025        
026        /**
027         * Sets state composed from chunks created from elements.
028         * @param state
029         */
030        protected abstract void setState(S state);
031        
032        /**
033         * Splits state into an array of chunks
034         * @param state
035         * @return
036         */
037        protected abstract List<C> split(S state);
038        
039        /**
040         * Joins an array of chunks into a state.
041         * @param chunks
042         * @return
043         */
044        protected abstract S join(List<C> chunks);
045        
046        /**
047         * Loads a chunk - converts it to element.
048         * @param chunk
049         * @return
050         */
051        protected abstract E load(C chunk);
052        
053        /**
054         * Stores an element - converts it into a chunk  
055         * @param element
056         * @return
057         */
058        protected abstract C store(E element);
059
060        /**
061         * The iterator maintains array index - this is the only state it maintains. 
062         * For each operation state is retrieved, split, chunks loaded, ... 
063         */
064        @Override
065        public Iterator<E> iterator() {
066                return new Iterator<E>() {
067                        
068                        // Points to the position of the current elements.
069                        // Initial value is before the first element.
070                        private int idx = -1;
071
072                        @Override
073                        public boolean hasNext() {
074                                S state = getState();
075                                return split(state).size() > idx + 1;
076                        }
077
078                        @Override
079                        public E next() {
080                                return load(split(getState()).get(++idx));
081                        }
082                        
083                        @Override
084                        public void remove() {
085                                List<C> chunks = new ArrayList<>(split(getState())); // In case split returns unmodifiable list.
086                                chunks.remove(idx--);
087                                setState(join(chunks));
088                        }
089                };
090        }
091
092        @Override
093        public int size() {
094                return split(getState()).size();
095        }
096        
097        @Override
098        public void clear() {
099                setState(join(Collections.emptyList()));
100        }
101        
102        @Override
103        public boolean add(E e) {
104                ArrayList<C> chunks = new ArrayList<>(split(getState()));
105                chunks.add(store(e));
106                setState(join(chunks));
107                return true;
108        }
109
110}
