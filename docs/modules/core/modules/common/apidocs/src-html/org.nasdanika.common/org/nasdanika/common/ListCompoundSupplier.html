001package org.nasdanika.common;
002
003import java.util.ArrayList;
004import java.util.Arrays;
005import java.util.Collection;
006import java.util.List;
007
008public class ListCompoundSupplier<E> extends ListCompoundExecutionParticipant<Supplier<? extends E>> implements Supplier<List<E>>  {
009
010        public ListCompoundSupplier(String name, Collection<Supplier<? extends E>> suppliers) {
011                super(name);
012                for (Supplier<? extends E> supplier: suppliers) {
013                        add(supplier);
014                }
015        }
016        
017        @SafeVarargs
018        public ListCompoundSupplier(String name, Supplier<? extends E>... suppliers) {
019                this(name, Arrays.asList(suppliers));
020        }
021
022        @Override
023        public List<E> execute(ProgressMonitor progressMonitor) throws Exception {
024                progressMonitor.setWorkRemaining(size());
025                List<E> result = new ArrayList<>();
026                for (Supplier<? extends E> e: getElements()) {
027                        result.add(e == null ? null : e.splitAndExecute(progressMonitor));                      
028                }
029                return result;
030        }       
031
032}
