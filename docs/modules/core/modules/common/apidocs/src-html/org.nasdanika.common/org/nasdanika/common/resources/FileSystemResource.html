001package org.nasdanika.common.resources;
002
003import org.nasdanika.common.NasdanikaException;
004import org.nasdanika.common.ProgressMonitor;
005
006/**
007 * Wrapper for {@link java.io.File}.
008 * @author Pavel
009 *
010 */
011public abstract class FileSystemResource implements BinaryResource {
012
013        protected java.io.File file;
014
015        public FileSystemResource(java.io.File file) {
016                this.file = file;
017        }
018
019        @Override
020        public String getName() {
021                return file.getName();
022        }
023
024        @Override
025        public boolean exists(ProgressMonitor monitor) {
026                return file.exists();
027        }
028
029        @Override
030        public FileSystemContainer getParent() {
031                java.io.File parent = file.getParentFile();
032                return parent == null ? null : new FileSystemContainer(parent);
033        }
034
035        @Override
036        public void delete(ProgressMonitor monitor) {
037                try (ProgressMonitor deleteMonitor = monitor.split("Deleting "+getName(), 1, file)) {
038                        if (!file.delete()) {
039                                throw new NasdanikaException("Cannot delete "+file.getAbsolutePath());
040                        }
041                }
042        }
043                
044        @Override
045        public long getTimestamp() {
046                return file.lastModified();
047        }       
048
049        @Override
050        public String toString() {
051                return getClass().getName()+"("+getPath()+")";
052        }
053        
054}
