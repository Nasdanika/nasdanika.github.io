001package org.nasdanika.common;
002
003import java.util.Map;
004import java.util.Map.Entry;
005
006/**
007 * Consumes a map and executes each element passing to it a matching input. 
008 * @author Pavel
009 *
010 * @param <K>
011 * @param <T>
012 */
013public class MapCompoundConsumer<K,T> extends MapCompoundExecutionParticipant<K,Consumer<? super T>> implements Consumer<Map<K,T>>  {
014
015        public MapCompoundConsumer(String name) {
016                super(name);
017        }
018
019        public MapCompoundConsumer(String name, Map<K,Consumer<? super T>> suppliers) {
020                super(name);
021                suppliers.forEach(this::put);
022        }
023
024        @Override
025        public void execute(Map<K,T> input, ProgressMonitor progressMonitor) throws Exception {
026                progressMonitor.setWorkRemaining(size());
027                for (Entry<K, Consumer<? super T>> e: elements.entrySet()) {
028                        Consumer<? super T> value = e.getValue();
029                        if (value != null) {
030                                value.splitAndExecute(input.get(e.getKey()), progressMonitor);                  
031                        }
032                }
033        }
034
035}
