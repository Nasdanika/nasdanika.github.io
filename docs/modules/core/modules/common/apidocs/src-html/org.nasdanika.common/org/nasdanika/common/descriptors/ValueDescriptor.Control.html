001package org.nasdanika.common.descriptors;
002
003import org.nasdanika.common.Diagnostic;
004
005/**
006 * Descriptor of a value which can be accessed and modified.
007 * @author Pavel
008 *
009 * @param <T>
010 */
011public interface ValueDescriptor extends Descriptor {
012        
013        /**
014         * Enumeration for giving UI generators a hint regarding which control to create for the value.
015         * Depending on a generator the hint may be ignored.
016         * Radio and check boxes may be used for boolean values or non-boolean values with choices. 
017         * E.g. check boxes can be used for multi-value descriptors. 
018         * @author Pavel
019         *
020         */
021        enum Control {
022                DATE, TIME, DATE_TIME, DROP_DOWN, CHECKBOX, RADIO, FILE, NUMBER, PASSWORD, TEXT, TEXT_AREA              
023        }
024        
025        /**
026         * Sets new value.
027         * @param value
028         * @return Diagnostic, in particular an error diagnostic if value could not be set.
029         */
030        Diagnostic set(Object value);
031        
032//      default boolean isPresent() {
033//              return get() != null;
034//      }
035        
036        Object get();
037        
038//      Class<?> getType();
039        
040        int getLowerBound();
041        
042        int getUpperBound();
043        
044        /**
045         * @return true if value can be set, false if it can only be read.
046         */
047        boolean isEditable();
048        
049        Control getControlHint();
050
051}
