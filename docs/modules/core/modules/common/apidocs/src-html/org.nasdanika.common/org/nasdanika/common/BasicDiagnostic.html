001package org.nasdanika.common;
002
003import java.util.ArrayList;
004import java.util.Collections;
005import java.util.List;
006
007public class BasicDiagnostic implements Diagnostic, DiagnosticChain, Composeable<Diagnostic> {
008        
009        private List<Object> data = new ArrayList<>();
010        private Status status;
011        private String message;
012        private List<Diagnostic> children = new ArrayList<>();
013        
014        public BasicDiagnostic(Status status, String message, Object... data) {
015                this.status = status;
016                this.message = message;
017                for (Object d: data) {
018                        this.data.add(d);
019                }
020        }
021
022        @Override
023        public void add(Diagnostic diagnostic) {
024                if (diagnostic != null) {
025                        children.add(diagnostic);
026                }               
027        }
028
029        @Override
030        public void addAll(Diagnostic diagnostic) {
031                if (diagnostic != null) {
032                        children.addAll(diagnostic.getChildren());
033                }               
034        }
035
036        @Override
037        public Status getStatus() {
038                return children.stream().map(Diagnostic::getStatus).reduce(status, (s1, s2) -> {
039                        if (s1 == null) {
040                                return s2;
041                        }
042                        if (s2 == null) {
043                                return s1;
044                        }
045                        return s1.ordinal() > s2.ordinal() ? s1 : s2;
046                });
047        }
048
049        @Override
050        public String getMessage() {
051                return message;
052        }
053
054        @Override
055        public List<Object> getData() {
056                return data;
057        }
058
059        @Override
060        public List<Diagnostic> getChildren() {
061                return Collections.unmodifiableList(children);
062        }
063                
064        /**
065         * If the diagnostic has {@link Diagnostic#getChildren children}, {@link #addAll add}s those children, otherwise, {@link #add add}s the diagnostic.
066         */
067        @Override
068        public Diagnostic compose(Diagnostic other) {
069                if (other != null) {
070                        if (other.getChildren().isEmpty()) {
071                                add(other);
072                        } else {
073                                addAll(other);
074                        }
075                }
076                return this;
077        }
078                
079}
