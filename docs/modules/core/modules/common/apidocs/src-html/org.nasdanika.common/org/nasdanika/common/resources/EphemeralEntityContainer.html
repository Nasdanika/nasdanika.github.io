001package org.nasdanika.common.resources;
002
003import org.nasdanika.common.ProgressMonitor;
004
005/**
006 * {@link MapContainer} of {@link EphemeralEntity}ies.
007 * @author Pavel
008 *
009 * @param <T>
010 */
011public class EphemeralEntityContainer<T> extends MapContainer<TypedEntity<T>> implements TypedEntityContainer<T> {
012
013        @Override
014        protected EphemeralEntity<T> createElement(String path, ProgressMonitor monitor) {
015                return new EphemeralEntity<T>() {
016
017                        @Override
018                        public void appendState(T state, ProgressMonitor monitor) {
019                                T existingState = getState(monitor.split("Getting existing state", 1, this));
020                                EphemeralEntityContainer.this.appendState(existingState, state, monitor.split("Appending state", 1, this, existingState, state));                               
021                        }
022
023                        @Override
024                        public String getName() {
025                                return path;
026                        }
027
028                        @Override
029                        public TypedEntityContainer<T> getParent() {
030                                return EphemeralEntityContainer.this;
031                        }
032
033                        @Override
034                        public long size(ProgressMonitor monitor) {
035                                return EphemeralEntityContainer.this.stateSize(getState(monitor));
036                        }
037                        
038                };
039        }
040        
041        protected long stateSize(T state) {
042                return 0;
043        }
044        
045        @Override
046        protected MapContainer<TypedEntity<T>> createSubContainer(String path) {
047                return new EphemeralEntityContainer<T>() {
048
049                        @Override
050                        public String getName() {
051                                return path;
052                        }
053
054                        @Override
055                        public TypedEntityContainer<T> getParent() {
056                                return EphemeralEntityContainer.this;
057                        }
058                        
059                };
060        }
061
062        /**
063         * Override to add support of appending state.
064         * @param existingState
065         * @param newState
066         * @param monitor
067         */
068        protected void appendState(T existingState, T newState, ProgressMonitor monitor) {
069                throw new UnsupportedOperationException();
070        }
071        
072        @Override
073        public TypedEntityContainer<T> getContainer(String path, ProgressMonitor monitor) {
074                return (TypedEntityContainer<T>) super.getContainer(path, monitor);
075        }
076        
077        @Override
078        public TypedEntityContainer<T> getParent() {
079                return (TypedEntityContainer<T>) super.getParent();
080        }
081        
082        @SuppressWarnings("unchecked")
083        @Override
084        public TypedResource<T> find(String path, ProgressMonitor monitor) {
085                return (TypedResource<T>) super.find(path, monitor);
086        }
087        
088}
