001package org.nasdanika.common;
002
003import java.util.ArrayList;
004import java.util.Arrays;
005import java.util.Collection;
006import java.util.List;
007import java.util.concurrent.Callable;
008import java.util.concurrent.ExecutorService;
009import java.util.concurrent.Future;
010
011public class CompoundCommand extends ListCompoundExecutionParticipant<Command> implements Command  {
012        
013        private ExecutorService executorService;
014
015        public CompoundCommand(String name, ExecutorService executorService, Collection<Command> commands) {
016                super(name);
017                this.executorService = executorService;
018                for (Command command: commands) {
019                        add(command);
020                }
021        }
022        
023        public CompoundCommand(String name, ExecutorService executorService, Command... commands) {
024                this(name, executorService, Arrays.asList(commands));
025        }
026
027        @Override
028        public void execute(ProgressMonitor progressMonitor) throws Exception {
029                progressMonitor.setWorkRemaining(size());
030                if (executorService == null || elements.size() < 2) {
031                        for (Command e: getElements()) {
032                                e.splitAndExecute(progressMonitor);                     
033                        }
034                } else {
035                        List<Future<Void>> invokeAll = executorService.invokeAll(getTasks(elements.subList(1, elements.size()), progressMonitor));
036                        elements.get(0).splitAndExecute(progressMonitor); // Execute the first element in the current thread while other elements are executed in other threads.
037                        for (Future<?> future: invokeAll) {
038                                future.get();
039                        };                      
040                }
041        }
042        
043        private static List<Callable<Void>> getTasks(List<Command> elements, ProgressMonitor progressMonitor) {
044                List<Callable<Void>> ret = new ArrayList<>();
045                for (Command e: elements) {
046                        ret.add(new Callable<Void>() {
047
048                                @Override
049                                public Void call() throws Exception {
050                                        e.splitAndExecute(progressMonitor);
051                                        return null;
052                                }
053                                
054                        });
055                }               
056                return ret;
057        }
058
059}
