001package org.nasdanika.common;
002
003import java.util.Iterator;
004
005/**
006 * Binding of {@link Iterator} to {@link Context}. The primary purpose of this interface is to use as an adapter type 
007 * in control flow in models. E.g. a model element has an iterator attribute which specifies a named adapter factory
008 * which adapts the model element to ContextIterator.Supplier.Factory. The factory is used to create to create ContextIterator
009 * and the model element is processed (e.g. code is generated) for each iterator element with context provided by the iterator.
010 * @author Pavel
011 *
012 */
013public interface ContextIterator extends Iterator<Context> {
014        
015        interface Factory {
016                
017                ContextIterator create(Context context);
018                
019        }
020
021}
