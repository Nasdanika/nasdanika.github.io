001package org.nasdanika.common.persistence;
002
003import java.util.List;
004import java.util.function.Function;
005
006import org.eclipse.emf.common.util.URI;
007import org.nasdanika.common.ProgressMonitor;
008
009/**
010 * Choice delegates to an object factory returned by the selector function which takes config as an argument.
011 * @author Pavel
012 *
013 * @param <T>
014 */
015public class Choice<T> extends Attribute<T> {
016        
017        private Function<Object, ObjectFactory<T>> selector;
018        
019        public Choice(
020                        Object key, 
021                        Function<Object, ObjectFactory<T>> selector,
022                        boolean isDefault, 
023                        boolean required, 
024                        T defaultValue, 
025                        String description, 
026                        Object... exclusiveWith) {
027                super(key, isDefault, required, defaultValue, description, exclusiveWith);
028                this.selector = selector;
029        }
030        
031        /**
032         * This implementation casts config to the target type. It can be used where not special conversion is required, e.g. config is a String and feature type is String.
033         * @param loader
034         * @param config
035         * @param base
036         * @param progressMonitor
037         * @param marker
038         * @return
039         * @throws Exception
040         */
041        @Override
042        public T create(ObjectLoader loader, Object config, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) throws Exception {
043                return selector.apply(config).create(loader, config, base, progressMonitor, markers);
044        }
045        
046}
