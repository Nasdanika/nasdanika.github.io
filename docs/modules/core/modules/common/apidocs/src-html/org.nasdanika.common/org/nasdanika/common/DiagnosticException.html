001package org.nasdanika.common;
002
003@SuppressWarnings("serial")
004public class DiagnosticException extends RuntimeException {
005
006        private Diagnostic diagnostic;
007
008        public DiagnosticException(Diagnostic diagnostic) {
009                this(diagnostic.getMessage(), diagnostic);
010        }
011
012        public DiagnosticException(String message, Diagnostic diagnostic) {
013                super(message);
014                this.diagnostic = diagnostic;
015        }
016        
017        public Diagnostic getDiagnostic() {
018                return diagnostic;
019        }
020
021}
