001package org.nasdanika.common;
002
003import java.util.function.BinaryOperator;
004
005/**
006 * Something that can be composed to create an aggregate. E.g. {@link AccessController}s can be composed with AND operation,
007 * {@link Context}s can be composed by chaining.
008 * @author Pavel
009 *
010 * @param <T>
011 */
012public interface Composeable<T> {
013        
014        /**
015         * @param <T> Instances of T shall implement {@link Composeable}.
016         * @return Composing operator which can be use in reducing streams of {@link Composeable}s to a single composeable.
017         */
018        @SuppressWarnings("unchecked")
019        static <T> BinaryOperator<T> composer() {
020                return (a,b) -> a == null ? b : ((Composeable<T>) a).compose(b);
021        }
022
023        /**
024         * Returns a composition of this instance and the other with this instance being 
025         * the first if the order of composition is important.
026         * @param other
027         * @return
028         */
029        T compose(T other);
030        
031}
