001package org.nasdanika.common;
002
003import java.io.IOException;
004import java.io.OutputStream;
005import java.net.HttpURLConnection;
006import java.net.URL;
007import java.net.URLConnection;
008import java.nio.charset.StandardCharsets;
009import java.security.MessageDigest;
010
011import org.apache.commons.codec.binary.Hex;
012import org.nasdanika.common.resources.Container;
013
014/**
015 * Generates PlantUML diagrams.
016 * @author Pavel
017 *
018 */
019public interface DiagramGenerator extends Composeable<DiagramGenerator> {
020        
021        public static String UML_DIALECT = "uml";
022        public static String SALT_DIALECT = "salt";
023        public static String GANTT_DIALECT = "gantt";
024        public static String MINDMAP_DIALECT = "mindmap";
025        public static String WBS_DIALECT = "wbs";
026        public static String DRAWIO_DIALECT = "drawio";
027        public static String MERMAID_DIALECT = "mermaid";
028        
029        /**
030         * @param dialect
031         * @return true if given dialect is supported by this generator
032         */
033        boolean isSupported(String dialect);
034        
035        /**
036         * Uses local PlantUML.
037         */
038        DiagramGenerator INSTANCE = new DiagramGeneratorImpl();
039        
040        /**
041         * Creates an instance of {@link DiagramGenerator} which does a POST to the service URL postfixed with /&lt;dialect name&gt; with diagram text spec as payload
042         * and returns the payload as String. UTF-8 is used for character encoding. 
043         * The service shall call the INSTANCE's <code>generateDiagram()</code>. It can also perform caching of generated images. Return code 200 is treated as success. 
044         * Any other response code is treated as an error.
045         * Remoting can be used in situations where GrapViz (https://graphviz.org/) is not available on diagram generating machines or to improve performance via caching.
046         * 
047         * An example of server implementation with SpringBoot can be found here - https://github.com/Nasdanika/spring-exec/blob/main/src/main/java/org/nasdanika/spring/exec/controllers/ExecController.java#L344
048         * @param service Service URL with a trailing slash /. Must be an HTTP(s) URL.
049         * @return Diagram image and image map.
050         */
051        static DiagramGenerator createClient(URL service) {
052                
053                return new DiagramGenerator() {
054
055                        @Override
056                        public String generateDiagram(String spec, String dialect) throws IOException {
057                                URL dialectURL = new URL(service, dialect);
058                                URLConnection connection = dialectURL.openConnection();
059                                if (!(connection instanceof HttpURLConnection)) {
060                                        throw new IllegalArgumentException("Not an HTTP(s) url: "+dialectURL);
061                                }
062                                
063                                HttpURLConnection httpConnection = (HttpURLConnection) connection;
064                                httpConnection.setRequestMethod("POST");
065                                httpConnection.setDoOutput(true);
066                                try (OutputStream out = connection.getOutputStream()) {
067                                        out.write(spec.getBytes(StandardCharsets.UTF_8));
068                                }
069                                
070                                int responseCode = httpConnection.getResponseCode();
071                                if (responseCode == 200) {
072                                        return DefaultConverter.INSTANCE.toString(httpConnection.getInputStream());
073                                }
074                                
075                                throw new IOException("HTTP Call to "+dialectURL+" has failed with response: "+responseCode+" "+httpConnection.getResponseMessage());
076                        }
077
078                        @Override
079                        public boolean isSupported(String dialect) {
080                                try {
081                                        URL dialectURL = new URL(service, dialect);
082                                        URLConnection connection = dialectURL.openConnection();
083                                        if (!(connection instanceof HttpURLConnection)) {
084                                                throw new IllegalArgumentException("Not an HTTP(s) url: "+dialectURL);
085                                        }                               
086                                        HttpURLConnection httpConnection = (HttpURLConnection) connection;                              
087                                        return httpConnection.getResponseCode() == 200; // GET OK indicates that dialect is supported. 
088                                } catch (Exception e) {
089                                        throw new NasdanikaException(e);
090                                }
091                        }
092                        
093                };
094                
095        }
096        
097        /**
098         * Generates a PlantUML diagram with an image map from text definition. The definition shall not contain start and end tags.
099         * @return
100         * @throws IOException 
101         */
102        String generateDiagram(String spec, String dialect) throws Exception;
103                        
104        default String generateUmlDiagram(String spec) throws Exception {
105                return generateDiagram(spec, UML_DIALECT);
106        }
107                
108        default String generateWireframeDiagram(String spec) throws Exception {
109                return generateDiagram(spec, SALT_DIALECT);
110        }
111        
112        default String generateGanttDiagram(String spec) throws Exception {
113                return generateDiagram(spec, GANTT_DIALECT);
114        }
115        
116        default String generateMindmapDiagram(String spec) throws Exception {
117                return generateDiagram(spec, MINDMAP_DIALECT);
118        }
119        
120        default String generateWbsDiagram(String spec) throws Exception {
121                return generateDiagram(spec, WBS_DIALECT);
122        }
123        
124        default String generateDrawioDiagram(String spec) throws Exception {
125                return generateDiagram(spec, DRAWIO_DIALECT);
126        }
127        
128        default String generateMermaidDiagram(String spec) throws Exception {
129                return generateDiagram(spec, MERMAID_DIALECT);
130        }
131        
132        /**
133         * Creates a diagram generator facading this one and cacheing generated diagrams by dialect and diagram text SHA-256 digest.
134         * @param container
135         * @param progressMonitor
136         * @return
137         */
138        default DiagramGenerator cachingDiagramGenerator(Container<String> container, ProgressMonitor progressMonitor) {
139                
140                return new DiagramGenerator() {
141                        
142                        private int hits;
143                        private int misses;
144
145                        @Override
146                        public String generateDiagram(String spec, String dialect) throws Exception {
147                                String cachePath = dialect + "/" + Hex.encodeHexString(MessageDigest.getInstance("SHA-256").digest(spec.getBytes(StandardCharsets.UTF_8))) + ".html";
148                                Object ret = container.find(cachePath, progressMonitor);
149                                if (ret instanceof String) {
150                                        ++hits;
151                                        return (String) ret;
152                                }
153                                String diagram = DiagramGenerator.this.generateDiagram(spec, dialect);
154                                container.put(cachePath, diagram, progressMonitor);
155                                ++misses;
156                                return diagram;
157                        }
158                        
159                        @Override
160                        public String toString() {
161                                return super.toString() + " hits: " + hits + ", misses: " + misses;
162                        }
163
164                        @Override
165                        public boolean isSupported(String dialect) {
166                                return DiagramGenerator.this.isSupported(dialect);
167                        }
168                        
169                };
170        }
171        
172        @Override
173        default DiagramGenerator compose(DiagramGenerator other) {
174                return new DiagramGenerator() {
175
176                        @Override
177                        public boolean isSupported(String dialect) {
178                                return DiagramGenerator.this.isSupported(dialect) || other.isSupported(dialect);
179                        }
180
181                        @Override
182                        public String generateDiagram(String spec, String dialect) throws Exception {
183                                return (DiagramGenerator.this.isSupported(dialect) ? DiagramGenerator.this : other).generateDiagram(spec, dialect);
184                        }
185                        
186                };
187        }
188
