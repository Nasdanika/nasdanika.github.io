001package org.nasdanika.common.persistence;
002
003import java.util.ArrayList;
004import java.util.Collection;
005import java.util.Collections;
006import java.util.List;
007
008import org.eclipse.emf.common.util.URI;
009import org.nasdanika.common.ProgressMonitor;
010import org.nasdanika.common.Util;
011
012/**
013 * If config is a list loads each element using createElement(), otherwise creates a singleton loaded with createElement()
014 * @author Pavel
015 *
016 * @param <T>
017 */
018public class ListAttribute<T> extends Attribute<List<T>> {
019
020        public ListAttribute(Object key, 
021                        boolean isDefault, 
022                        boolean required, 
023                        List<T> defaultValue, 
024                        String description, 
025                        Object... exclusiveWith) {
026                super(key, isDefault, required, defaultValue, description, exclusiveWith);
027        }
028        
029        @Override
030        public List<T> create(ObjectLoader loader, Object config, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers)    throws Exception {
031                if (config instanceof Collection) {
032                        List<T> ret = new ArrayList<>();
033                        int idx = 0;
034                        for (Object element: (Collection<?>) config) {
035                                ret.addAll(createElements(loader, element, base, progressMonitor, Util.getMarkers((Collection<?>) config, idx++)));
036                        }
037                        return ret;
038                }
039                
040                return createElements(loader, config, base, progressMonitor, markers);
041        }
042        
043        /**
044         * Creates list elements from a config element. May create zero or more elements. 
045         * This implementation delegates to createElement(), i.e. creates a single element. 
046         * @param loader
047         * @param element
048         * @param base
049         * @param progressMonitor
050         * @param marker
051         * @return
052         * @throws Exception
053         */
054        protected List<T> createElements(ObjectLoader loader, Object element, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers)        throws Exception { 
055                return Collections.singletonList(createElement(loader, element, base, progressMonitor, markers)); 
056        }
057        
058
059        /**
060         * Creates element list.
061         * @param loader
062         * @param element
063         * @param base
064         * @param progressMonitor
065         * @param marker
066         * @return
067         * @throws Exception
068         */
069        @SuppressWarnings("unchecked")
070        protected T createElement(ObjectLoader loader, Object element, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers)       throws Exception { 
071                return (T) element; 
072        }
073        
074}
