001package org.nasdanika.common;
002
003import java.util.LinkedHashMap;
004import java.util.Map;
005import java.util.Map.Entry;
006
007/**
008 * @author Pavel
009 *
010 * @param <K>
011 * @param <T>
012 */
013public class MapCompoundConsumerFactory<K,T> implements ConsumerFactory<Map<K,T>> {
014
015        private String name;
016        private Map<K,ConsumerFactory<? super T>> elements = new LinkedHashMap<>();
017
018        public MapCompoundConsumerFactory(String name, Map<K,ConsumerFactory<? super T>> elements) {
019                this.name = name;
020                this.elements.putAll(elements);
021        }
022        
023        public MapCompoundConsumerFactory(String name) {
024                this.name = name;
025        }
026
027        @Override
028        public Consumer<Map<K,T>> create(Context context) throws Exception {
029                MapCompoundConsumer<K,T> ret = new MapCompoundConsumer<K,T>(name);
030                for (Entry<K, ConsumerFactory<? super T>> e: elements.entrySet()) {
031                        ConsumerFactory<? super T> value = e.getValue();
032                        if (value != null) {
033                                ret.put(e.getKey(), value.create(context));
034                        }
035                }
036                return ret;
037        }
038        
039        public void put(K key, ConsumerFactory<? super T> element) {
040                elements.put(key, element);
041        }       
042        
043        public void putAll(Map<? extends K, ? extends ConsumerFactory<? super T>> elements) {
044                elements.entrySet().forEach(e -> put(e.getKey(), e.getValue()));
045        }
046
047}
