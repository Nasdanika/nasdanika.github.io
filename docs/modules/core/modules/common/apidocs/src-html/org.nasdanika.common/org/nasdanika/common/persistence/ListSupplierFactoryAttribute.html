001package org.nasdanika.common.persistence;
002
003import java.util.List;
004
005import org.nasdanika.common.Context;
006import org.nasdanika.common.ListCompoundSupplierFactory;
007import org.nasdanika.common.Supplier;
008import org.nasdanika.common.SupplierFactory;
009
010/**
011 * 
012 * @author Pavel
013 *
014 * @param <T>
015 * @param <E> Not really required, but used to eliminate generics errors.
016 */
017public class ListSupplierFactoryAttribute<T,E> extends AbstractFeatureDelegate<Feature<List<E>>> implements SupplierFactoryFeature<List<T>> {
018
019        private boolean interpolate;
020
021        public ListSupplierFactoryAttribute(Feature<List<E>> delegate, boolean interpolate) {
022                super(delegate);
023                this.interpolate = interpolate;
024        }
025
026        @Override
027        public SupplierFactory<List<T>> getValue() {
028                ListCompoundSupplierFactory<T> ret = new ListCompoundSupplierFactory<>("Feature " + getKey());
029                if (delegate.getValue() != null) {
030                        int idx = 0;
031                        for (Object e: delegate.getValue()) {
032                                if (e instanceof String && interpolate) {
033                                        ret.add(new SupplierFactory<T>() {
034
035                                                @Override
036                                                public Supplier<T> create(Context context) throws Exception {
037                                                        Object ie = context.interpolate((String) e);
038                                                        return SupplierFactory.<T>adapt(ie, "Interpolated feature " + getKey() + "[" + e + "]").create(context);                                                        
039                                                }
040                                                
041                                        });
042                                        
043                                } else {
044                                        ret.add(SupplierFactory.<T>adapt(e, "Feature " + getKey() + "[" + (idx) + "]"));
045                                }
046                                ++idx;
047                        }
048                }               
049                return ret;
050        }
051
052}
