001package org.nasdanika.common;
002
003import java.util.LinkedHashMap;
004import java.util.Map;
005import java.util.Map.Entry;
006
007public class MapCompoundSupplier<K,T> extends MapCompoundExecutionParticipant<K,Supplier<? extends T>> implements Supplier<Map<K,T>>  {
008
009        public MapCompoundSupplier(String name) {
010                super(name);
011        }
012
013        public MapCompoundSupplier(String name, Map<K,Supplier<? extends T>> suppliers) {
014                super(name);
015                suppliers.forEach(this::put);
016        }
017
018        @Override
019        public Map<K,T> execute(ProgressMonitor progressMonitor) throws Exception {
020                progressMonitor.setWorkRemaining(size());
021                Map<K,T> result = new LinkedHashMap<>();
022                for (Entry<K, Supplier<? extends T>> e: elements.entrySet()) {
023                        Supplier<? extends T> value = e.getValue();
024                        if (value != null) {
025                                result.put(e.getKey(), value.splitAndExecute(progressMonitor));                 
026                        }
027                }
028                return result;
029        }       
030
031}
