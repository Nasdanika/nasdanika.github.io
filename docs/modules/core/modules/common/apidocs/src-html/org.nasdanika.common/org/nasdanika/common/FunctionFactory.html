001package org.nasdanika.common;
002
003/**
004 * @author Pavel Vlasov
005 *
006 * @param <T>
007 */
008public interface FunctionFactory<T,R> extends ExecutionParticipantFactory<Function<T,R>> {
009        
010        /**
011         * Provides {@link FunctionFactory} for a specific argument and return type.
012         * @author Pavel
013         *
014         */
015        interface Provider {
016                <T,R> FunctionFactory<T,R> getFactory(Class<T> parameterType, Class<R> resultType);
017        }       
018        
019        default <V> FunctionFactory<T,V> then(FunctionFactory<? super R, V> then) {
020                return new FunctionFactory<T, V>() {
021                        
022                        @Override
023                        public Function<T, V> create(Context context) throws Exception {
024                                return FunctionFactory.this.create(context).then(then.create(context));
025                        }
026                        
027                };
028        }
029                
030        default ConsumerFactory<T> then(ConsumerFactory<? super R> then) {
031                return new ConsumerFactory<T>() {
032                        
033                        @Override
034                        public Consumer<T> create(Context context) throws Exception {
035                                return FunctionFactory.this.create(context).then(then.create(context));
036                        }
037                        
038                };
039        }       
040                
041        /**
042         * @param contextSupplierFactory Factory which creates a context to be passed to this factory to create supplier.
043         * @return Function factory which creates a context using the context supplier factory and then uses that context and this factory to create a function.
044         */
045        default FunctionFactory<T,R> contextify(SupplierFactory<Context> contextSupplierFactory) {
046                return new FunctionFactory<T,R>() {
047
048                        @Override
049                        public Function<T,R> create(Context context) throws Exception {
050                                return new ContextifiedExecutionParticipant.ContextifiedFunction<T,R>(contextSupplierFactory.create(context), FunctionFactory.this);
051                        }
052                        
053                };
054        }
055        
056        /**
057         * Creates a function factry which adapts source to target type.
058         * @param <S>
059         * @param <T>
060         * @param type
061         * @return
062         */
063        static <S,T> FunctionFactory<S,T> adapter(Class<T> type) {
064                return new FunctionFactory<S,T>() {
065
066                        @Override
067                        public Function<S,T> create(Context context) throws Exception {
068                                return new Function<S,T>() {
069
070                                        @Override
071                                        public double size() {
072                                                return 1;
073                                        }
074
075                                        @Override
076                                        public String name() {
077                                                return "Adapting to " + type.getName();
078                                        }
079
080                                        @Override
081                                        public T execute(S source, ProgressMonitor progressMonitor) throws Exception {
082                                                return Adaptable.adaptTo(source, type);
083                                        }
084                                        
085                                };
086                        }
087                        
088                };
089                
090        }
091        
092}
