045                 * For types with static INSTANCE field of compatible type returns field value.

050                        for (Field f: type.getFields()) {

055                                                throw new NasdanikaException("Error accessing INSTANCE field of " + type.getName(), e);

060                }

239         * Mounts a context under a specified prefix. get(String) and get(String,Class) methods look up property/source in the mounted context first if the key start with the prefix, removing the prefix.
240         * E.g. if the prefix is <code>resources/</code> then <code>get("resources/myResource")</code> will call <code>get("myResource")</code> in the mounted resource first and will return a value if it is not null.
241         * If the value returned by the mounted context is null, then the value from <code>get("resources/myResource")</code> call to this context will be returned.
242         * In other words, the mounted value shadows this context value, but doesn't hide it.
243         * Typed service calls are always made against this context.
244         * @param context Context to mount.
245         * @param prefix Mount prefix.
246         * @return This context if argument context or argument prefix are null null or a context which delegates get(String) and get(String,Class) to the mounted context if the key start with the prefix.

251                }
252                
253                return new Context() {
254

257                                if (key != null && key.startsWith(prefix)) {
258                                        Object ret = context.get(key.substring(prefix.length()));
259                                        if (ret != null) {
260                                                return ret;
261                                        }
262                                }
263                                return Context.this.get(key);
264                        }
265
266                        @Override
267                        public <T> T get(Class<T> type) {

269                        }
270

272                        public <T> T get(String key, Class<T> type) {
273                                if (key != null && key.startsWith(prefix)) {
274                                        T ret = context.get(key.substring(prefix.length()), type);

277                                        }
278                                }
279                                return Context.this.get(key, type);

391        /**
392         * Creates a new mutable context with a given prefix and service predicate. 
393         * @param mapper Mapper functions, the same as in map();
394         * @param servicePredicate Predicate to use during look up of parent services. One possible scenario is to filter-out {@link Composeable} services from the parent.
395         * @return
396         */
397        default MutableContext fork(Function<String,String> mapper, Predicate<Object> servicePredicate) {
398                Context mapped = map(mapper);
399                if (servicePredicate == null) {
400                        return new SimpleMutableContext(mapped);
401                }
402                Context predicated = new Context() {
403                        
404                        @SuppressWarnings("unchecked")
405                        @Override
406                        public <T> T get(Class<T> type) {
407                                T ret = mapped.get(type);
408                                if (ret == null) {
409                                        return null;
410                                }
411                                if (ret.getClass().isArray()) {
412                                        List<Object> filtered = new ArrayList<Object>(); 
413                                        for (int i = 0; i < Array.getLength(ret); ++i) {
414                                                Object service = Array.get(ret, i);
415                                                if (servicePredicate.test(service)) {
416                                                        filtered.add(service);
417                                                }
418                                        }
419                                        ret = (T) filtered.toArray((Object[]) Array.newInstance(ret.getClass().getComponentType(), filtered.size()));
420                                }
421                                return servicePredicate.test(ret) ? null : ret;
422                        }
423                        
424                        @Override
425                        public Object get(String key) {
426                                return mapped.get(key);
427                        }
428                };
429                return new SimpleMutableContext(predicated);
430        }

432        static Pattern EXPANDER_PATTERN = Pattern.compile("\\$\\{(.+?)\\}");    
433        
434        /**
435         * Expands tokens in the form of ``${token name|default value}`` to their values.
436         * If a token is not found the default value after the pipe is used for expansion.
437         * If there is no default value then the expansion is not processed. Token names shall not contain pipes.
438         * If replacement is not null and there is a {@link Converter} service then the converter is used 
439         * to convert the replacement to {@link String}. If no such conversion can be done then toString() is used 
440         * as default conversion. Add extra open/closing curly brackets to escape the token, e.g. in documentation. 
441         * For example <code>${{xyz}}</code> will be "interpolated" as <code>${xyz}</code>. There may be multiple levels of curly brackets.
442         * This behavior may be leveraged in multi-stage interpolation where early stage interpolators "peel" the token until it comes at the interpolation level where it is supposed to be expanded.    
443         * @param input
444         * @return If the string contains just a single token then the token value is returned as-is. Otherwise tokens in the input are replaced with their values converted to string using context {@link Converter} service if it is present or toString() method.
445         */
446        default Object interpolate(String input) {
447                if (input == null) {
448                        return input;
449                }
450                
451                // Computing properties.
452                Context tokenComputer = computingContext();
453                
454                Matcher matcher = EXPANDER_PATTERN.matcher(input);
455                StringBuilder output = new StringBuilder();
456                int i = 0;
457                while (matcher.find()) {
458                    String token = matcher.group();
459                    String peeledToken = token.substring(2, token.length()-1); // Peeling of ${ and }
460                    if (peeledToken.startsWith("{") && input.length() > matcher.end() && '}' == input.charAt(matcher.end())) {
461                            output
462                                .append(input.substring(i, matcher.start() + 2))
463                                .append(peeledToken.substring(1));
464                            i = matcher.end();
465                    } else {
466                            int pipeIdx = peeledToken.indexOf('|');
467                            String defaultValue = pipeIdx == -1 ? null : peeledToken.substring(pipeIdx + 1);
468                            if (pipeIdx != -1) {
469                                peeledToken = peeledToken.substring(0, pipeIdx);
470                            }
471                            
472                        if (matcher.start() == 0 && matcher.end() == input.length()) {
473                                Object replacement = tokenComputer.get(peeledToken, defaultValue);
474                                        return replacement == null ? input : replacement;
475                        }
476                            
477                                Object replacement = tokenComputer.getString(peeledToken, defaultValue);
478                            if (replacement != null) {
479                                if (!(replacement instanceof String)) {
480                                        Converter converter = get(Converter.class);
481                                        if (converter != null) {
482                                                String str = converter.convert(replacement, String.class);
483                                                if (str != null) {
484                                                        replacement = str;
485                                                }
486                                        }
487                                }
488                                    output.append(input.substring(i, matcher.start())).append(replacement);                         
489                                    i = matcher.end();
490                            }
491                    }
492                }
493                output.append(input.substring(i, input.length()));
494                return output.toString();
495        }

497        /**
498         * Interpolates to {@link String}.
499         * @param input
500         * @return
501         */
502        default String interpolateToString(String input) {
503                Object result = interpolate(input);
504                if (result == null) {
505                        return null;
506                }
507                if (result instanceof String) {
508                        return (String) result;
509                }
510        Converter converter = get(Converter.class);
511        if (converter != null) {
512                String str = converter.convert(result, String.class);
513                if (str != null) {
514                        return str;
515                }
516        }               
517        return String.valueOf(result);
518        }

520        /**
521         * Creates a copy of this map using {@link LinkedHashMap} with {@link String} values interpolated and
522         * {@link Map} and {@link Collection} values passed through interpolate methods with respective parameter types.
523         * @param input
524         * @return A deep copy of the map with string values interpolated.
525         */
526        @SuppressWarnings("unchecked")
527        default <K> Map<K,Object> interpolate(Map<K,?> input) {
528                if (input == null) {
529                        return null;
530                }
531                Map<K,Object> ret = new LinkedHashMap<>();
532                input.forEach((k,v) -> {
533                        Object u;
534                        if (v instanceof String) {
535                                u = interpolate((String) v);
536                        } else if (v instanceof Collection) {
537                                u = interpolate((Collection<Object>) v);
538                        } else if (v instanceof Map) {
539                                u = interpolate((Map<K,Object>) v);
540                        } else {
541                                u = v;
542                        }
543                        ret.put(k,u);
544                });
545                return ret;
546        }

605        /**
606         * Computing context retrieves properties from this context. If a property is found and is of {@link PropertyComputer} type, then the property computer is
607         * invoked to produce value otherwise the value is returned as is. If a property for a given key is not found, the computer traverses the key hierarchy up. Slash (/) is 
608         * a key separator. When it finds a parent property value then is computes the property value in the following way - for a property computer its compute() method is invoked,
609         * for {@link Context} the path from the parent property is used to retrieve value, for {@link Map} its get method is used, for {@link List} an element is retrived if key path is parseable to int,  
610         * for {@link Function} the path is passed to it as an argument and the value is returned as property value. 
611         * @return 
612         */
613        default Context computingContext() {
614                
615                return new Context() {
616
617                        @Override
618                        public Object get(String key) {
619                                return get(key, Object.class);
620                        }
621
622                        @SuppressWarnings("unchecked")
623                        @Override
624                        public <T> T get(String key, Class<T> type) {
625                                Object value = Context.this.get(key, (Class<?>) null);
626                                
627                                if (value instanceof PropertyComputer) {
628                                        return ((PropertyComputer) value).compute(this, key, null, type);
629                                }
630                                
631                                if (value instanceof Function) {
632                                        return ((Function<Context, T>) value).apply(this);
633                                }
634                                
635                                if (value instanceof BiFunction) {
636                                        return ((BiFunction<Context, Class<?>, T>) value).apply(this, type);
637                                }                                                                                                       
638                                
639                                T ret = Context.this.get(key, type);
640                                
641                                // At the spot computing of the property itself
642                                
643                                return ret == null ? computeByParent(key, null, type) : ret;
644                        }
645                        
646                        /**
647                         * Computes value by walking up the property key path.
648                         * @param <T>
649                         * @param key Current path
650                         * @param path Key path after the key
651                         * @param type
652                         * @return
653                         */
654                        @SuppressWarnings("unchecked")
655                        private <T> T computeByParent(String key, String path, Class<T> type) {                         
656                                Converter converter;
657                                if (type == null || type == Object.class) {
658                                        converter = new Converter() {
659                                                
660                                                @Override
661                                                public <TP> TP convert(Object source, Class<TP> type) {
662                                                        return (TP) source;
663                                                }
664                                        };  
665                                } else {
666                                        converter = get(Converter.class, DefaultConverter.INSTANCE);
667                                }
668                                
669                                int lastSlash = key.lastIndexOf('/');
670                                if (lastSlash == -1) {
671                                        return null;
672                                }
673                                
674                                String parentKey = key.substring(0, lastSlash);
675                                String subKey = key.substring(lastSlash + 1);
676                                if (path != null) {
677                                        subKey += "/" + path;
678                                }
679                                Object parentProperty = Context.this.get(parentKey);
680                                if (parentProperty instanceof PropertyComputer) {
681                                        return ((PropertyComputer) parentProperty).compute(this,  parentKey, subKey, type);
682                                }
683                                
684                                if (parentProperty instanceof Context) {
685                                        return ((Context) parentProperty).get(subKey, type);    
686                                }
687                                if (parentProperty instanceof Map) {
688                                        return converter.convert(((Map<?,?>) parentProperty).get(subKey), type);
689                                }
690                                if (parentProperty instanceof List) {
691                                        try {
692                                                int idx = Integer.parseInt(subKey);
693                                                if (idx >= 0 && idx < ((List<?>) parentProperty).size()) {
694                                                        return converter.convert(((List<?>) parentProperty).get(idx), type);
695                                                }
696                                        } catch (NumberFormatException e) {
697                                                // NOP
698                                        }
699                                }
700                                if (parentProperty instanceof Function) {
701                                        return converter.convert(((Function<String, Object>) parentProperty).apply(subKey), type);
702                                }
703
704                                return computeByParent(parentKey, subKey, type);
705                        }
706
707                        @SuppressWarnings("unchecked")
708                        @Override
709                        public <T> T get(Class<T> type) {
710                                T svc = Context.this.get(type);
711                                if (svc instanceof ServiceComputer) {
712                                        return ((ServiceComputer<T>) svc).compute(this, type);
713                                }
714                                if (svc != null) {
715                                        return svc;
716                                }
717                                for (Field f: type.getFields()) {
718                                        if ("INSTANCE".equals(f.getName()) && Modifier.isStatic(f.getModifiers()) && type.isAssignableFrom(f.getType())) {
719                                                try {
720                                                        return (T) f.get(null);
721                                                } catch (IllegalArgumentException | IllegalAccessException e) {
722                                                        throw new NasdanikaException("Error accessing INSTANCE field of " + type.getName(), e);
723                                                }
724                                        }
725                                }
726                                return null;                            
727                        }
728                        
729                        @Override
730                        public String toString() {
731                                return "Computing context " + super.toString();
732                        }
733                        
734                };
735        }

771        /**
772         * Wraps a source function, e.g. Map<String,Object>::get into a context. Performs resolution of hierarchical properties.
773         * @param source Source function.
774         * @return
775         */
776        static Context wrap(Function<String,Object> source) {
777                return new Context() {
778
779                        @SuppressWarnings("unchecked")
780                        @Override
781                        public Object get(String key) {
782                                Object ret = source.apply(key);
783                                if (ret != null) {
784                                        return ret;
785                                }
786                                
787                                int lastSlash = key.lastIndexOf('/');
788                                if (lastSlash == -1) {
789                                        return null;
790                                }
791                                
792                                Object parentProperty = get(key.substring(0, lastSlash));
793                                String subKey = key.substring(lastSlash + 1);
794                                if (parentProperty instanceof Context) {
795                                        return ((Context) parentProperty).get(subKey);  
796                                }
797                                if (parentProperty instanceof Map) {
798                                        return ((Map<?,?>) parentProperty).get(subKey);
799                                }
800                                if (parentProperty instanceof List) {
801                                        try {
802                                                int idx = Integer.parseInt(subKey);
803                                                if (idx >= 0 && idx < ((List<?>) parentProperty).size()) {
804                                                        return ((List<?>) parentProperty).get(idx);
805                                                }
806                                        } catch (NumberFormatException e) {
807                                                // NOP
808                                        }
809                                }
810                                if (parentProperty instanceof Function) {
811                                        return ((Function<String, Object>) parentProperty).apply(subKey);
812                                }
813
814                                return null;
815                        }
816
817                        @Override
818                        public <T> T get(Class<T> type) {
819                                return null;
820                        }
821                        
822                };
823        }
