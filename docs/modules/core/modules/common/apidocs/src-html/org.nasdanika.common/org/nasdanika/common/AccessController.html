001package org.nasdanika.common;
002
003import org.eclipse.emf.ecore.EOperation;
004
005/**
006 * Interface to adapt to for access decisions.
007 * @author Pavel Vlasov
008 *
009 */
010public interface AccessController extends Composeable<AccessController> {
011        
012        public static final String PERMISSION_READ = "read";
013
014        public static final String PERMISSION_CREATE = "create";
015
016        public static final String PERMISSION_DELETE = "delete";
017
018        public static final String PERMISSION_UPDATE = "update";
019
020        public static final String PERMISSION_EXECUTE = "execute";
021
022        /**
023         * Access controller which permits all actions.
024         */
025        AccessController ALLOW_ALL = (action, qualifier, context) -> true; 
026        
027        /**
028         * Access controller which permits nothing.
029         */
030        AccessController DENY_ALL = (action, qualifier, context) -> false; 
031        
032        /**
033         * Checks for a permission to perform an action for a given qualifier with a given context.
034         * @param action Action to perform. E.g. "read" or "execute", "transfer-funds".
035         * @param qualifier Qualifier, e.g. "firstName" for read or "transferFunds" for execute.
036         * @param context Context for context-conditional permissions, e.g. for execute transferFunds it may contain "amount" key with transfer amount as a value, or more generally a map of parameter names to argument 
037         * values for {@link EOperation}. 
038         * @return
039         */
040        boolean hasPermission(String action, String qualifier, Context context);
041        
042        /**
043         * Checks for a permission to perform an action for a given qualifier with empty context.
044         * @param action
045         * @param qualifier
046         * @return
047         */
048        default boolean hasPermission(String action, String qualifier) {
049                return hasPermission(action, qualifier, null);
050        }
051        
052        /**
053         * Checks "read" permissions
054         * @param qualifier
055         * @return
056         */
057        default boolean canRead(String qualifier, Context context) {
058                return hasPermission(PERMISSION_READ, qualifier, context);
059        }       
060        
061        /**
062         * Checks "read" permissions
063         * @param qualifier
064         * @return
065         */
066        default boolean canRead(String qualifier) {
067                return canRead(qualifier, null);
068        }       
069        
070        /**
071         * Checks "create" permission
072         * @param qualifier
073         * @return
074         */
075        default boolean canCreate(String qualifier, Context context) {
076                return hasPermission(PERMISSION_CREATE, qualifier, context);
077        }
078        
079        /**
080         * Checks "create" permissions
081         * @param qualifier
082         * @return
083         */
084        default boolean canCreate(String qualifier) {
085                return canCreate(qualifier, null);
086        }       
087        
088        /**
089         * Checks "delete" permission
090         * @param qualifier
091         * @return
092         */
093        default boolean canDelete(String qualifier, Context context) {
094                return hasPermission(PERMISSION_DELETE, qualifier, context);
095        }
096        
097        /**
098         * Checks "delete" permissions
099         * @param qualifier
100         * @return
101         */
102        default boolean canDelete(String qualifier) {
103                return canDelete(qualifier, null);
104        }       
105        
106        /**
107         * Checks "update" permission
108         * @param qualifier
109         * @return
110         */
111        default boolean canUpdate(String qualifier, Context context) {
112                return hasPermission(PERMISSION_UPDATE, qualifier, context);
113        }
114        
115        /**
116         * Checks "update" permissions
117         * @param qualifier
118         * @return
119         */
120        default boolean canUpdate(String qualifier) {
121                return canUpdate(qualifier, null);
122        }       
123        
124        /**
125         * Checks "execute" permission
126         * @param qualifier
127         * @param context
128         * @return
129         */
130        default boolean canExecute(String qualifier, Context context) {
131                return hasPermission(PERMISSION_EXECUTE, qualifier, context);
132        }
133
134        /**
135         * Checks "execute" permissions
136         * @param qualifier
137         * @return
138         */
139        default boolean canExecute(String qualifier) {
140                return canExecute(qualifier, null);
141        }       
142        
143        /**
144         * @param enclosingContext Context to fall-back to.
145         * @return Access controller which chains the context passed to the <code>hasPermission(action, qualifier, context)</code> with the enclosing context passed to this method. 
146         */
147        default AccessController contextify(Context enclosingContext) {
148                return enclosingContext == null ? this : (action,qualifier,context) -> hasPermission(action, qualifier, context == null ? enclosingContext : context.compose(enclosingContext)); 
149        }
150        
151        /**
152         * Composes with ``and()``.
153         */
154        @Override
155        default AccessController compose(AccessController other) {
156                return and(other);
157        }
158        
159        /**
160         * @param accessControllers
161         * @return An "intersection" access controller which allows a permission only if all elements allow it. In the case of an empty list all permissions are denied.
162         */
163        default AccessController and(AccessController other) {
164                if (other == null) {
165                        return this;
166                }
167                
168                return (action, qualifier, context) -> {
169                        return AccessController.this.hasPermission(action, qualifier, context) && other.hasPermission(action, qualifier, context); 
170                }; 
171        }
172                
173        /**
174         * @param accessControllers
175         * @return A "union" access controller which allows a permission if any of elements allows. In the case of an empty list all permissions are denied.
176         */
177        default AccessController or(AccessController other) {
178                if (other == null) {
179                        return this;
180                }
181                
182                return (action, qualifier, context) -> {
183                        return AccessController.this.hasPermission(action, qualifier, context) || other.hasPermission(action, qualifier, context); 
184                }; 
185        }
186        
187}
