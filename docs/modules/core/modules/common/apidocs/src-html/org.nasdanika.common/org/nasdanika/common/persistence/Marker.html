001package org.nasdanika.common.persistence;
002
003import org.nasdanika.common.Util;
004
005/**
006 * Position in a text file - yaml, json, ...
007 * @author Pavel Vlasov
008 *
009 */
010public interface Marker {
011        
012        int getLine();
013        
014        int getColumn();
015        
016        String getLocation();
017        
018        static String toString(Marker marker) {
019                if (marker == null) {
020                        return "";
021                }
022                return (Util.isBlank(marker.getLocation()) ? "" : marker.getLocation()) + " " + marker.getLine() + ":" + marker.getColumn();
023        }       
024
025}
