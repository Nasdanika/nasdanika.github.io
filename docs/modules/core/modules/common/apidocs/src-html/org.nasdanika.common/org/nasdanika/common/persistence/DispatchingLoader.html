001package org.nasdanika.common.persistence;
002
003import java.util.HashMap;
004import java.util.List;
005import java.util.Map;
006import java.util.Map.Entry;
007
008import org.eclipse.emf.common.util.URI;
009import org.nasdanika.common.ProgressMonitor;
010
011/**
012 * Registers loaders by prefixes and then delegates to the registered loaders removing prefix first.  
013 * @author Pavel
014 *
015 */
016public class DispatchingLoader implements ObjectLoader {
017        
018        private org.nasdanika.common.persistence.ObjectLoader chain;
019
020        public DispatchingLoader(ObjectLoader chain) {
021                this.chain = chain;
022        }
023        
024        public DispatchingLoader() {}
025        
026        protected Map<String,ObjectLoader> registry = new HashMap<>();
027        
028        /**
029         * Registers a loader with given prefix
030         * @param prefix
031         * @param loader
032         */
033        public void register(String prefix, ObjectLoader loader) {
034                registry.put(prefix, loader);
035        }
036
037        @Override
038        public Object create(ObjectLoader loader, String type, Object config, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) throws Exception {
039                for (Entry<String, ObjectLoader> re: registry.entrySet()) {
040                        if (type.startsWith(re.getKey())) {
041                                return re.getValue().create(loader, type.substring(re.getKey().length()), config, base, progressMonitor, markers);
042                        }                       
043                }
044                
045                if (chain == null) {
046                        throw new ConfigurationException("Unsupported type: " + type, markers);
047                }
048                
049                return chain.create(loader, type, config, base, progressMonitor, markers);
050        }
051                
052}
