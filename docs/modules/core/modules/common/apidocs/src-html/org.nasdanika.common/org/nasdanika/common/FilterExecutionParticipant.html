001package org.nasdanika.common;
002
003public class FilterExecutionParticipant<T extends ExecutionParticipant> implements ExecutionParticipant {
004                
005        protected T target;
006
007        public FilterExecutionParticipant(T target) {
008                this.target = target;
009        }
010
011        @Override
012        public double size() {
013                return target.size();
014        }
015
016        @Override
017        public String name() {
018                return target.name();
019        }
020        
021        @Override
022        public Diagnostic diagnose(ProgressMonitor progressMonitor) {
023                return target.diagnose(progressMonitor);
024        }
025        
026        @Override
027        public void commit(ProgressMonitor progressMonitor) throws Exception {
028                target.commit(progressMonitor);
029        }
030        
031        @Override
032        public boolean rollback(ProgressMonitor progressMonitor) throws Exception {
033                return target.rollback(progressMonitor);
034        }
035        
036        @Override
037        public void close() throws Exception {
038                target.close();
039        }
040
041}
