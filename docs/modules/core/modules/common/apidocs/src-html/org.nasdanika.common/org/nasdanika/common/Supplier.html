003import java.util.ArrayList;
004import java.util.List;

007/**
008 * By convention Supplier does not split the monitor for itself - this is the responsibility of the caller.
009 * 
010 * {@link ProgressMonitor} allocation pattern - the caller creates a monitor with work's size and name and passes it to work's execute:
011 * 
012 * ```
013 * try (Supplier supplier = ...; ProgressMonitor workMonitor = monitor.split(supplier.getName(), supplier.getSize(), supplier)) {
014 *     work.execute(workMonitor);
015 *     
016 *     // If success
017 *     work.commit();
018 *     
019 *     // If something went wrong
020 *     work.rollback();
021 *     
022 * }
023 * ```
024 * 
025 * @author Pavel Vlasov
026 * @param C context type.
027 * @param T result type.
028 */
029public interface Supplier<T> extends ExecutionParticipant, ExecutionParticipantInfo, java.util.function.Function<ProgressMonitor,T> {
030        
031        /**
032         * Executes the Supplier.
033         * @param monitor Monitor to use.
034         * @return
035         * @throws Exception
036         */
037        T execute(ProgressMonitor progressMonitor) throws Exception;    
038        
039        default T splitAndExecute(ProgressMonitor progressMonitor) throws Exception {
040                try (ProgressMonitor subMonitor = split(progressMonitor, "Executing " + name())) {
041                        return execute(subMonitor);
042                }
043        }
044        
045        default T splitAndExecute(double size, ProgressMonitor progressMonitor) throws Exception {
046                try (ProgressMonitor subMonitor = split(size, progressMonitor, "Executing " + name())) {
047                        return execute(subMonitor);
048                }
049        }

051        Supplier<Object> EMPTY = new Supplier<Object>() {
052
053                @Override
054                public double size() {
055                        return 0;
056                }
057                
058                @Override
059                public String name() {
060                        return "Empty supplier";
061                }
062
063                @Override
064                public Object execute(ProgressMonitor monitor) throws Exception {
065                        return null;
066                }
067                
068        };
069        
070        @SuppressWarnings("unchecked")
071        static <T> Supplier<T> empty() {
072                return (Supplier<T>) EMPTY;
073        }

075        default <R> Supplier<R> then(java.util.function.Function<? super T,R> then) {
076                return new Supplier<R>() {
077                        
078                        @Override
079                        public R execute(ProgressMonitor progressMonitor) throws Exception {
080                                return then.apply(Supplier.this.execute(progressMonitor));
081                        }
082                        
083                        @Override
084                        public Diagnostic diagnose(ProgressMonitor progressMonitor) {
085                                return Supplier.this.diagnose(progressMonitor);
086                        }
087                        
088                        @Override
089                        public void close() throws Exception {
090                                Supplier.this.close();
091                        }
092                        
093                        @Override
094                        public void commit(ProgressMonitor progressMonitor) throws Exception {
095                                Supplier.this.commit(progressMonitor);
096                        }
097                        
098                        @Override
099                        public boolean rollback(ProgressMonitor progressMonitor) throws Exception {
100                                return Supplier.this.rollback(progressMonitor);
101                        }
102                        
103                        @Override
104                        public double size() {
105                                return Supplier.this.size();
106                        }
107                        
108                        @Override
109                        public String name() {
110                                return Supplier.this.name();
111                        }
112                        
113                };
114        }
115        
116        default <V> Supplier<V> then(Function<? super T,V> then) {
117                List<ExecutionParticipant> elements = new ArrayList<>();
118                elements.add(this);
119                elements.add(then);
120                class Then extends CompoundExecutionParticipant<ExecutionParticipant> implements Supplier<V> {
121
122                        protected Then(String name) {
123                                super(name, true);
124                        }
125                        
126                        @Override
127                        public V execute(ProgressMonitor progressMonitor) throws Exception {
128                                return then.splitAndExecute(Supplier.this.splitAndExecute(progressMonitor), progressMonitor);
129                        }
130
131                        @Override
132                        protected List<ExecutionParticipant> getElements() {
133                                return elements;
134                        }                       
135                }
136                
137                StringBuilder nameBuilder = new StringBuilder();
138                nameBuilder.append(Supplier.this.name()).append(" => ").append(then.name());
139                return new Then(nameBuilder.toString());
140        }
141        
142        default Command then(Consumer<? super T> then) {
143                List<ExecutionParticipant> elements = new ArrayList<>();
144                elements.add(this);
145                elements.add(then);
146                class Then extends CompoundExecutionParticipant<ExecutionParticipant> implements Command {
147
148                        protected Then(String name) {
149                                super(name, true);
150                        }
151                        
152                        @Override
153                        public void execute(ProgressMonitor progressMonitor) throws Exception {
154                                then.splitAndExecute(Supplier.this.splitAndExecute(progressMonitor), progressMonitor);
155                        }
156
157                        @Override
158                        protected List<ExecutionParticipant> getElements() {
159                                return elements;
160                        }                       
161                }
162                
163                StringBuilder nameBuilder = new StringBuilder();
164                nameBuilder.append(Supplier.this.name()).append(" => ").append(then.name());
165                return new Then(nameBuilder.toString());
166        }

168        static <T> Supplier<T> fromFunction(java.util.function.Function<ProgressMonitor, T> function, String name, double size) {
169                return new Supplier<T>() {
170                        
171                        @Override
172                        public double size() {
173                                return size;
174                        }
175                        
176                        @Override
177                        public String name() {
178                                return name;
179                        }
180                        
181                        @Override
182                        public T execute(ProgressMonitor progressMonitor) throws Exception {
183                                return function.apply(progressMonitor);
184                        }
185                };
186        }
187        
188        static <T> Supplier<T> fromSupplier(java.util.function.Supplier<T> supplier, String name, double size) {
189                return fromFunction((p) -> supplier.get(), name, size);         
190        }
191        
192        static <T> Supplier<T> from(T value, String name) {
193                return fromFunction((p) -> value, name, 0);             
194        }
195                
196        static <T> Supplier<T> fromCallable(Callable<T> callable, String name, double size) {
197                return new Supplier<T>() {
198                        
199                        @Override
200                        public double size() {
201                                return size;
202                        }
203                        
204                        @Override
205                        public String name() {
206                                return name;
207                        }
208                        
209                        @Override
210                        public T execute(ProgressMonitor progressMonitor) throws Exception {
211                                return callable.call();
212                        }
213                };
214        }
215        
216        /**
217         * Diagnoses, executes, commits/rollsback, closes.
218         * @param progressMonitor
219         * @return
220         */
221        @Override
222        default public T apply(ProgressMonitor progressMonitor) {
223                try {
224                        progressMonitor.setWorkRemaining(size()*3); // diagnose, execute, commit or rollback
225                        diagnose(progressMonitor.split("Diagnosing", size())).checkError("Diagnostic failed: "+name());
226                        try {
227                                T result = execute(progressMonitor.split("Executing", size()));
228                                commit(progressMonitor.split("Committing", size()));
229                                return result;
230                        } catch (Exception e) {
231                                try {
232                                        rollback(progressMonitor.split("Rolling back: "+e, size(), e));
233                                } catch (Exception re) {
234                                        e.addSuppressed(re);
235                                }
236                                throw e;
237                        }
238                } catch (DiagnosticException e) {
239                        throw e;
240                } catch (Exception e) {
241                        throw new NasdanikaException(e);
242                } finally {
243                        try {
244                                close();
245                        } catch (Exception e) {
246                                throw new NasdanikaException(e);
247                        }
248                }
249        }
250        
251        /**
252         * Calls apply() with a {@link NullProgressMonitor}.
253         * @return
254         */
255        default java.util.function.Supplier<T> asSupplier() {
256                return () -> apply(new NullProgressMonitor());
257        }
258        
259        /**
260         * Creates work, diagnoses it, executes, commits if there is no exception and rolls back if there is, closes, returns result.
261         * Uses  null progress monitor.
262         * @return
263         */
264        default Callable<T> asCallable() {
265                return new Callable<T>() {
266        
267                        @Override
268                        public T call() throws Exception {
269                                try (ProgressMonitor progressMonitor = new NullProgressMonitor()) {
270                                        progressMonitor.setWorkRemaining(size()*3); // diagnose, execute, commit or rollback
271                                        diagnose(progressMonitor.split("Diagnosing", size())).checkError("Diagnostic failed: "+name());
272                                        try {
273                                                T result = execute(progressMonitor.split("Executing", size()));
274                                                commit(progressMonitor.split("Committing", size()));
275                                                return result;
276                                        } catch (Exception e) {
277                                                try {
278                                                        rollback(progressMonitor.split("Rolling back: "+e, size(), e));
279                                                } catch (Exception re) {
280                                                        e.addSuppressed(re);
281                                                }
282                                                throw e;
283                                        }
284                                } catch (DiagnosticException e) {
285                                        throw e;
286                                } catch (Exception e) {
287                                        throw new NasdanikaException(e);
288                                } finally {
289                                        try {
290                                                close();
291                                        } catch (Exception e) {
292                                                throw new NasdanikaException(e);
293                                        }
294                                }
295                        }
296                };
297        }

299        /**
300         * @return Function which executes this supplier and returns the argument and the result as a {@link BiSupplier}.
301         * This method can be used for embedding suppliers into function chains.
302         */
303        default <V> Function<V,BiSupplier<V,T>> asFunction() {
304                return new Function<V,BiSupplier<V,T>>() {
305                        
306                        @Override
307                        public BiSupplier<V,T> execute(V arg, ProgressMonitor progressMonitor) throws Exception {
308                                T result = Supplier.this.execute(progressMonitor);
309                                return new BiSupplier<V, T>() {
310
311                                        @Override
312                                        public V getFirst() {
313                                                return arg;
314                                        }
315
316                                        @Override
317                                        public T getSecond() {
318                                                return result;
319                                        }
320                                };
321                        }
322                        
323                        @Override
324                        public Diagnostic diagnose(ProgressMonitor progressMonitor) {
325                                return Supplier.this.diagnose(progressMonitor);
326                        }
327                        
328                        @Override
329                        public void close() throws Exception {
330                                Supplier.this.close();
331                        }
332                        
333                        @Override
334                        public void commit(ProgressMonitor progressMonitor) throws Exception {
335                                Supplier.this.commit(progressMonitor);
336                        }
337                        
338                        @Override
339                        public boolean rollback(ProgressMonitor progressMonitor) throws Exception {
340                                return Supplier.this.rollback(progressMonitor);
341                        }
342                        
343                        @Override
344                        public double size() {
345                                return Supplier.this.size();
346                        }
347                        
348                        @Override
349                        public String name() {
350                                return Supplier.this.name();
351                        }
352                        
353                };
354        }
