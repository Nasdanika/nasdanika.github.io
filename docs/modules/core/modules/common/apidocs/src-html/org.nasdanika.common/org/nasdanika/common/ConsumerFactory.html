001package org.nasdanika.common;
002
003/**
004 * @author Pavel Vlasov
005 *
006 * @param <T>
007 */
008public interface ConsumerFactory<T> extends ExecutionParticipantFactory<Consumer<T>> {
009        
010        /**
011         * Provides {@link ConsumerFactory} for a specific type.
012         * @author Pavel
013         *
014         */
015        interface Provider {
016                <T> ConsumerFactory<T> getFactory(Class<T> type);
017        }
018        
019        default FunctionFactory<T,T> asFunctionFactory() {
020                return new FunctionFactory<T, T>() {
021
022                        @Override
023                        public Function<T, T> create(Context context) throws Exception {
024                                return ConsumerFactory.this.create(context).asFunction();
025                        }
026                        
027                };
028        }
029        
030        default <R> FunctionFactory<BiSupplier<T,R>,R> asBiSupplierFunctionFactory() {
031                return new FunctionFactory<BiSupplier<T,R>, R>() {
032
033                        @Override
034                        public Function<BiSupplier<T,R>, R> create(Context context) throws Exception {
035                                return ConsumerFactory.this.create(context).asBiSupplierFunction();
036                        }
037                        
038                };
039        }
040        
041        /**
042         * @param contextSupplierFactory Factory which creates a context to be passed to this factory to create supplier.
043         * @return Consumer factory which creates a context using the context supplier factory and then uses that context and this factory to create a function.
044         */
045        default ConsumerFactory<T> contextify(SupplierFactory<Context> contextSupplierFactory) {
046                return new ConsumerFactory<T>() {
047
048                        @Override
049                        public Consumer<T> create(Context context) throws Exception {
050                                return new ContextifiedExecutionParticipant.ContextifiedConsumer<T>(contextSupplierFactory.create(context), ConsumerFactory.this);
051                        }
052                        
053                };
054        }
055        
056}
