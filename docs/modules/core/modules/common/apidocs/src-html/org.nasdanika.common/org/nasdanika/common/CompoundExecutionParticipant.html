001package org.nasdanika.common;
002
003import java.util.ArrayList;
004import java.util.Collection;
005import java.util.Collections;
006import java.util.List;
007import java.util.concurrent.CancellationException;
008
009/**
010 * TODO - support of concurrent execution.
011 * @author Pavel
012 *
013 * @param <E>
014 */
015public abstract class CompoundExecutionParticipant<E extends ExecutionParticipant> implements ExecutionParticipant {
016                
017        private String name;
018        private boolean reverseCommitClose;
019
020        /**
021         * 
022         * @param name
023         * @param reverseCommitClose If true commit and close are executed in reverse order. 
024         * It can be required in participant chaining scenarios where one participant supplies arguments to another participant and
025         * therefore the lifecycle of the second participant shall be enclosed into the lifecycle of the first paricipant.
026         */
027        protected CompoundExecutionParticipant(String name, boolean reverseCommitClose) {
028                this.name = name;
029                this.reverseCommitClose = reverseCommitClose;
030        }
031
032        @Override
033        public String name() {
034                return name;
035        }       
036        
037        protected abstract Collection<E> getElements();
038        
039        protected Double size; // Set to null to clear cached value
040
041        @Override
042        public double size() {
043                if (size == null) {
044                        size = 0.0;
045                        for (E e: getElements()) {
046                                size += e.size();
047                        }
048                }
049                return size;
050        }
051
052        @Override
053        public Diagnostic diagnose(ProgressMonitor progressMonitor) {           
054                if (progressMonitor.isCancelled()) {
055                        progressMonitor.worked(1, "Cancelled");
056                        return new BasicDiagnostic(Status.CANCEL, "Progress monitor is cancelled", this);
057                }
058                BasicDiagnostic ret = new BasicDiagnostic(Status.INFO, name());
059                progressMonitor.setWorkRemaining(size());
060                for (E e: getElements()) {
061                        ret.add(e.splitAndDiagnose(progressMonitor));
062                }
063                return ret;
064        }
065        
066        @Override
067        public void commit(ProgressMonitor progressMonitor) throws Exception {
068                if (progressMonitor.isCancelled()) {
069                        progressMonitor.worked(1, "Cancelled");
070                        throw new CancellationException();
071                }
072                progressMonitor.setWorkRemaining(size());
073                List<E> elements = new ArrayList<E>(getElements());
074                if (reverseCommitClose) {
075                        Collections.reverse(elements);
076                }
077                for (E e: elements) {
078                        e.splitAndCommit(progressMonitor);
079                }
080        }
081        
082        @Override
083        public boolean rollback(ProgressMonitor progressMonitor) throws Exception {
084                if (progressMonitor.isCancelled()) {
085                        progressMonitor.worked(1, "Cancelled");
086                        throw new CancellationException();
087                }
088                progressMonitor.setWorkRemaining(size());
089                boolean result = true;
090                List<E> elements = new ArrayList<E>(getElements());
091                Collections.reverse(elements);
092                for (E e: elements) {
093                        result = e.splitAndRollback(progressMonitor) && result; 
094                }
095                return result;
096        }
097        
098        @Override
099        public void close() throws Exception {
100                List<E> elements = new ArrayList<E>(getElements());
101                if (reverseCommitClose) {
102                        Collections.reverse(elements);
103                }
104                for (E e: elements) {
105                        e.close();
106                }
107        }
108        
109        @Override
110        public String toString() {
111                return super.toString() + " " + name() + " " + size();
112        }
113
114}
