001package org.nasdanika.common;
002
003import java.util.Collection;
004import java.util.LinkedHashMap;
005import java.util.Map;
006
007public abstract class MapCompoundExecutionParticipant<K, E extends ExecutionParticipant> extends CompoundExecutionParticipant<E> {
008
009        protected MapCompoundExecutionParticipant(String name) {
010                super(name, false);
011        }
012
013        protected Map<K,E> elements = new LinkedHashMap<>();
014
015        @Override
016        protected Collection<E> getElements() {
017                return elements.values();
018        }
019        
020        public void put(K key, E element) {
021                elements.put(key, element);
022                size = null;
023        }
024        
025}
