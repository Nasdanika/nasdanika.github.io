001package org.nasdanika.common;
002
003import java.util.Arrays;
004import java.util.Collection;
005
006/**
007 * This consumer invokes element consumers passing them the argument.
008 * @author Pavel
009 *
010 * @param <E>
011 */
012public class CollectionCompoundConsumer<E> extends ListCompoundExecutionParticipant<Consumer<? super E>> implements Consumer<E>  {
013
014        public CollectionCompoundConsumer(String name, Collection<Consumer<? super E>> consumers) {
015                super(name);
016                for (Consumer<? super E> consumer: consumers) {
017                        add(consumer);
018                }
019        }
020        
021        @SafeVarargs
022        public CollectionCompoundConsumer(String name, Consumer<? super E>... suppliers) {
023                this(name, Arrays.asList(suppliers));
024        }
025
026        @Override
027        public void execute(E arg, ProgressMonitor progressMonitor) throws Exception {
028                progressMonitor.setWorkRemaining(size());
029                for (Consumer<? super E> e: getElements()) {
030                        if (e != null) {
031                                e.splitAndExecute(arg, progressMonitor);                        
032                        }
033                }
034        }       
035
036}
