001package org.nasdanika.common;
002
003import java.io.IOException;
004
005import org.apache.commons.text.StringEscapeUtils;
006import org.json.JSONObject;
007
008public class DiagramGeneratorImpl implements DiagramGenerator {
009        
010        @Override
011        public String generateDiagram(String spec, String dialect) throws IOException {
012                if (DRAWIO_DIALECT.equals(dialect)) {
013                        JSONObject data = new JSONObject();
014                        data.put("highlight", "#0000ff");
015                        data.put("nav", true);
016                        data.put("resize",true);
017                        data.put("toolbar", "zoom layers tags lightbox");
018                        data.put("edit","_blank");
019                        
020                        data.put("xml", spec);
021                        
022                        String diagramDiv = "<div class=\"mxgraph\" style=\"max-width:100%;border:1px solid transparent;\" data-mxgraph=\"" + StringEscapeUtils.escapeHtml4(data.toString()) + "\"></div>";
023                        String drawioViewer = getDrawioViewer();
024                        if (Util.isBlank(drawioViewer)) {
025                                return diagramDiv;
026                        }
027                        String script = "<script type=\"text/javascript\" src=\"" + drawioViewer + "\"></script>";
028                        return diagramDiv + System.lineSeparator() + script;            
029                }
030                
031                if (MERMAID_DIALECT.equals(dialect)) {
032                        String mermaidDiv = "<div class=\"mermaid\">"  + System.lineSeparator() + spec + System.lineSeparator() + "</div>";
033                        String mermaidInitialize = getMermaidInitialize();
034                        if (Util.isBlank(mermaidInitialize)) {
035                                return mermaidDiv;
036                        }
037                        return mermaidDiv
038                                        + System.lineSeparator()
039                                        + "<script>"
040                                        + System.lineSeparator()
041                                        + mermaidInitialize
042                                        + System.lineSeparator()
043                                        + "</script>";                                  
044                }
045                
046                throw new UnsupportedOperationException("Unsupported dialect: " + dialect);
047        }
048
049        protected String getDrawioViewer() {
050                return "https://cdn.jsdelivr.net/gh/Nasdanika/drawio@dev/src/main/webapp/js/viewer-static.min.js";
051        }
052        
053        protected String getMermaidInitialize() {
054                return "mermaid.initialize({ startOnLoad: true, securityLevel: 'loose' });";
055        }
056
057        @Override
058        public boolean isSupported(String dialect) {
059                return DRAWIO_DIALECT.equals(dialect) || MERMAID_DIALECT.equals(dialect);
060        }
061}
