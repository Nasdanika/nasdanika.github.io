001package org.nasdanika.common;
002
003import java.util.ArrayList;
004import java.util.Collection;
005import java.util.IdentityHashMap;
006import java.util.List;
007import java.util.Map;
008
009public class ElementIdentityMapCompoundFunctionFactory<T,R> implements FunctionFactory<T, java.util.function.Function<FunctionFactory<T,R>,R>> {
010
011        private String name;
012        private List<FunctionFactory<T,R>> elements = new ArrayList<>();
013
014        public ElementIdentityMapCompoundFunctionFactory(String name, Collection<FunctionFactory<T,R>> functionsFactories) {
015                this.name = name;
016                this.elements.addAll(elements);
017        }
018        
019        public ElementIdentityMapCompoundFunctionFactory(String name) {
020                this.name = name;
021        }
022
023        @Override
024        public Function<T, java.util.function.Function<FunctionFactory<T, R>, R>> create(Context context) throws Exception {
025                Map<FunctionFactory<T,R>, Function<T,R>> functionMap = new IdentityHashMap<>();
026                ElementIdentityMapCompoundFunction<T,R> compoundFunction = new ElementIdentityMapCompoundFunction<>(name);              
027                
028                for (FunctionFactory<T,R> e: elements) {
029                        if (e != null) {
030                                Function<T,R> supplier = e.create(context);
031                                functionMap.put(e, supplier);
032                                compoundFunction.put(supplier);
033                        }
034                }
035                
036                class Filter extends FilterExecutionParticipant<Function<T, java.util.function.Function<Function<T,R>,R>>> implements Function<T,java.util.function.Function<FunctionFactory<T,R>,R>> {
037
038                        public Filter(Function<T,java.util.function.Function<Function<T,R>,R>> target) {
039                                super(target);
040                        }
041
042                        @Override
043                        public java.util.function.Function<FunctionFactory<T, R>, R> execute(T arg, ProgressMonitor progressMonitor) throws Exception {
044                                return ((java.util.function.Function<FunctionFactory<T, R>, Function<T, R>>) functionMap::get).andThen(target.execute(arg, progressMonitor));
045                        }
046                        
047                }
048                
049                return new Filter(compoundFunction);
050        }
051        
052        public void put(FunctionFactory<T,R> element) {
053                if (element != null) {
054                        elements.add(element);
055                }
056        }
057
058}
