001package org.nasdanika.common.persistence;
002
003import java.util.Map;
004
005import org.nasdanika.common.SupplierFactory;
006import org.nasdanika.common.Util;
007
008public class InterpolatedMapSupplierFactoryAttribute extends AbstractFeatureDelegate<SupplierFactoryFeature<Map<?,?>>> implements SupplierFactoryFeature<Map<?,?>> {
009
010        public InterpolatedMapSupplierFactoryAttribute(SupplierFactoryFeature<Map<?,?>> delegate) {
011                super(delegate);
012        }
013
014        @Override
015        public SupplierFactory<Map<?,?>> getValue() {
016                return delegate.getValue().then(Util.INTERPOLATE_MAP);
017        }
018
019}
