001package org.nasdanika.common.persistence;
002
003import org.eclipse.emf.common.util.URI;
004import org.nasdanika.common.ProgressMonitor;
005
006/**
007 * Functional interface for object creation. 
008 * @author Pavel
009 *
010 * @param <T>
011 */
012public interface ObjectFactory<T> {
013        
014        /**
015         * Creates an object of requested type with a given config.
016         * @param ObjectLoader for creating object parts. 
017         * @param config
018         * @param base Base URL for resolving references.
019         * @param progressMonitor 
020         * @param marker Optional source marker for troubleshooting.  
021         * @return Created object
022         * @throws Exception
023         */
024        T create(ObjectLoader loader, Object config, URI base, ProgressMonitor progressMonitor, Marker marker) throws Exception;        
025
026}
