001package org.nasdanika.common;
002
003import java.util.LinkedHashMap;
004import java.util.Map;
005import java.util.Map.Entry;
006
007public class MapCompoundSupplierFactory<K,T> implements SupplierFactory<Map<K,T>> {
008
009        private String name;
010        private Map<K,SupplierFactory<? extends T>> elements = new LinkedHashMap<>();
011
012        public MapCompoundSupplierFactory(String name, Map<K,SupplierFactory<? extends T>> elements) {
013                this.name = name;
014                this.elements.putAll(elements);
015        }
016        
017        public MapCompoundSupplierFactory(String name) {
018                this.name = name;
019        }
020
021        @Override
022        public Supplier<Map<K,T>> create(Context context) throws Exception {
023                MapCompoundSupplier<K,T> ret = new MapCompoundSupplier<K,T>(name);
024                for (Entry<K, SupplierFactory<? extends T>> e: elements.entrySet()) {
025                        SupplierFactory<? extends T> value = e.getValue();
026                        if (value != null) {
027                                ret.put(e.getKey(), value.create(context));
028                        }
029                }
030                return ret;
031        }
032        
033        public void put(K key, SupplierFactory<? extends T> element) {
034                elements.put(key, element);
035        }       
036        
037        public void putAll(Map<? extends K, ? extends SupplierFactory<? extends T>> elements) {
038                elements.entrySet().forEach(e -> put(e.getKey(), e.getValue()));
039        }
040
041}
