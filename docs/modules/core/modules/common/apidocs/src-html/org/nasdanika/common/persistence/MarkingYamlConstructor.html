001package org.nasdanika.common.persistence;
002
003import java.util.Iterator;
004import java.util.List;
005import java.util.Map;
006import java.util.Map.Entry;
007
008import org.yaml.snakeyaml.DumperOptions;
009import org.yaml.snakeyaml.LoaderOptions;
010import org.yaml.snakeyaml.Yaml;
011import org.yaml.snakeyaml.constructor.Constructor;
012import org.yaml.snakeyaml.nodes.MappingNode;
013import org.yaml.snakeyaml.nodes.NodeTuple;
014import org.yaml.snakeyaml.nodes.ScalarNode;
015import org.yaml.snakeyaml.nodes.SequenceNode;
016import org.yaml.snakeyaml.representer.Representer;
017import org.yaml.snakeyaml.resolver.Resolver;
018
019/**
020 * Snake YAML {@link Constructor} which creates {@link MarkedArrayList} and {@link MarkedLinkedHashMap} which allows to obtain locations of of elements in the source file using {@link Marker}s.
021 * @author Pavel Vlasov
022 *
023 */
024public class MarkingYamlConstructor extends Constructor {
025        
026        private String location;
027
028        public MarkingYamlConstructor(String location) {
029                this.location = location;
030        }
031        
032        @Override
033        protected List<? extends Object> constructSequence(SequenceNode node) {
034                MarkedArrayList<? extends Object> ret = (MarkedArrayList<? extends Object>) super.constructSequence(node);
035                node.getValue().forEach(n -> ret.getMarkers().add(new MarkerImpl(location, n.getStartMark())));
036                Iterator<? extends Object> eit = ret.iterator();
037                Iterator<Marker> mit = ret.getMarkers().iterator();
038                while (eit.hasNext() && mit.hasNext()) {
039                        Object e = eit.next();
040                        Marker m = mit.next();
041                        if (e instanceof Markable) {
042                                ((Markable) e).setMarker(m);
043                        }
044                }
045                return ret;
046        }
047        
048        @Override
049        protected List<Object> newList(SequenceNode node) {
050                return new MarkedArrayList<Object>();
051        }
052                                
053        @Override
054        protected Map<Object, Object> constructMapping(MappingNode node) {
055                MarkedLinkedHashMap<Object, Object> ret = (MarkedLinkedHashMap<Object, Object>) super.constructMapping(node);
056                node.getValue()
057                        .stream()
058                        .map(NodeTuple::getKeyNode)
059                        .filter(n -> n instanceof ScalarNode)
060                        .forEach(keyNode -> ret.mark(((ScalarNode) keyNode).getValue(), new MarkerImpl(location, keyNode.getStartMark())));
061                
062                for (Entry<Object, Object> e: ret.entrySet()) {
063                        if (e.getValue() instanceof Markable) {
064                                ((Markable) e.getValue()).setMarker(ret.getMarker(e.getKey()));
065                        }
066                }
067                return ret;
068        }
069        
070        @Override
071        protected Map<Object, Object> newMap(MappingNode node) {
072                return new MarkedLinkedHashMap<Object, Object>();
073        }                                       
074                
075        /**
076         * @return Yaml which uses this {@link MarkingYamlConstructor} to create collections with markers.
077         */
078        public static Yaml createMarkingYaml(String location) {
079                LoaderOptions loaderOptions = new LoaderOptions();
080                loaderOptions.setAllowDuplicateKeys(false);
081                return new Yaml(new MarkingYamlConstructor(location), new Representer(), new DumperOptions(), loaderOptions, new Resolver());
082        }
083
084}
