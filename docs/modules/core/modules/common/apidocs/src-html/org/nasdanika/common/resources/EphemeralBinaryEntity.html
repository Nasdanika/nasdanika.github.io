001package org.nasdanika.common.resources;
002
003import java.io.BufferedInputStream;
004import java.io.ByteArrayInputStream;
005import java.io.ByteArrayOutputStream;
006import java.io.IOException;
007import java.io.InputStream;
008
009import org.nasdanika.common.NasdanikaException;
010import org.nasdanika.common.ProgressMonitor;
011
012public abstract class EphemeralBinaryEntity implements BinaryEntity {
013
014        protected byte[] state;
015
016        @Override
017        public InputStream getState(ProgressMonitor monitor) {
018                return state == null ? null : new ByteArrayInputStream(state);
019        }
020
021        @Override
022        public void setState(InputStream state, ProgressMonitor monitor) {
023                if (state == null) {
024                        this.state = null;
025                } else {
026                        try {
027                                ByteArrayOutputStream baos = new ByteArrayOutputStream();
028                                try (BufferedInputStream bis = new BufferedInputStream(state)) {
029                                        int b;
030                                        while ((b = bis.read()) != -1) {
031                                                baos.write(b);
032                                        }
033                                }
034                                baos.close();
035                                this.state = baos.toByteArray();
036                        } catch (IOException e) {
037                                throw new NasdanikaException(e);
038                        }
039                }
040        }
041        
042        @Override
043        public void appendState(InputStream state, ProgressMonitor monitor) {
044                if (state != null) {
045                        if (this.state == null) {
046                                setState(state, monitor);
047                        } else {
048                                try {
049                                        ByteArrayOutputStream baos = new ByteArrayOutputStream();                                       
050                                        try (InputStream bis = new ByteArrayInputStream(this.state)) {
051                                                int b;
052                                                while ((b = bis.read()) != -1) {
053                                                        baos.write(b);
054                                                }
055                                        }
056                                        try (InputStream bis = new BufferedInputStream(state)) {
057                                                int b;
058                                                while ((b = bis.read()) != -1) {
059                                                        baos.write(b);
060                                                }
061                                        }
062                                        baos.close();
063                                        this.state = baos.toByteArray();
064                                } catch (IOException e) {
065                                        throw new NasdanikaException(e);
066                                }
067                        }
068                }
069        }       
070
071        @Override
072        public boolean canRead() {
073                return true;
074        }
075
076        @Override
077        public boolean canWrite() {
078                return true;
079        }
080        
081        @Override
082        public boolean exists(ProgressMonitor monitor) {
083                return state != null;
084        }
085        
086        @Override
087        public void delete(ProgressMonitor monitor) {
088                setState(null, monitor);        
089        }       
090
091        @Override
092        public String toString() {
093                return getClass().getName()+"("+getPath()+")";
094        }
095        
096        @Override
097        public long size(ProgressMonitor monitor) {
098                return state == null ? 0 : state.length;
099        }
100
101}
