001package org.nasdanika.common.persistence;
002
003/**
004 * Something which knows how to store state into storage and load it from it.
005 * @author Pavel Vlasov
006 * 
007 * @param <T> Referenced object type.
008 * @param <R> Reference type.
009 * @param <S> Storage type.
010 */
011public interface Persistent<T, R, S> {
012        
013        /**
014         * Loads state from the storage.
015         * @param storage Storage.
016         * @param resolver Reference resolver. Can be ``null`` if the object itself "knows" how to resolve references,
017         * e.g. it utilizes object proxies resolved on access.
018         */
019        void load(S storage, Resolver<T, R> resolver);
020        
021        /**
022         * Same as load(storage, null).
023         * @param storage
024         */
025        default void load(S storage) {
026                load(storage, null);
027        }
028        
029        /**
030         * Stores state into the storage.
031         * @param storage
032         */
033        void store(S storage, Referencer<T, R> referencer);
034        
035        /**
036         * Same as store(storage, null).
037         * @param storage
038         */
039        default void store(S storage) {
040                store(storage, null);
041        }
042
043}
