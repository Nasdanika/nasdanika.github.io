026/**
027 * Loads objects from configuration, e.g. YAML or JSON.
028 * Objects are loaded as follows: 
029 * 
030 * * If config value is a map with a single element, then its key is an object type to be instantiated. Type is passed to create() method and is implementation-specific. E.g. a factory may use aliases/logical names for a 
031 * pre-defined set of supported objects/components.  
032 * * If config value is a multi-value map, then the result would be a map with each value instantiated as explained here.
033 * * If config value is a list then the result would be a list with each element instantiated as explained here.
034 * * Otherwise the value is returned as-is. 
035 * 
036 * {@link ExecutionParticipant}'s factory loading methods cast instantiated objects to a specific execution participant type factory and wrap values. Lists, and maps are wrapped into compound execution participants of requested type.
037 *   
038 * @author Pavel
039 *
040 */
041public interface ObjectLoader {
042        
043        /**
044         * Creates an object of requested type with a given config.
045         * @param ObjectLoader for creating object parts. This factory or parent/root loader for chained loaders.
046         * @param type
047         * @param config
048         * @param base Base URL for resolving references.
049         * @param progressMonitor 
050         * @param marker Optional source marker for troubleshooting.  
051         * @return
052         * @throws Exception
053         */
054        Object create(ObjectLoader loader, String type, Object config, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) throws Exception;
055        
056        @SuppressWarnings("unchecked")
057        default Object load(Object spec, URI base, ProgressMonitor progressMonitor) throws Exception {
058                // map
059                if (spec instanceof Map) {
060                        Map<String,Object> map = (Map<String,Object>) spec;
061                        // single
062                        if (map.size() == 1) {
063                                for (Entry<String, Object> e: map.entrySet()) {
064                                        return create(this, e.getKey(), e.getValue(), base, progressMonitor, Util.getMarkers(map, e.getKey()));
065                                }
066                        }
067                        // otherwise
068                        Map<String,Object> ret = new LinkedHashMap<>();
069                        for (Entry<String, Object> e: map.entrySet()) {
070                                ret.put(e.getKey(), load(e.getValue(), base, progressMonitor));
071                        }
072                        return ret;
073                }
074                
075                // list
076                if (spec instanceof Collection) {
077                        List<Object> ret = new ArrayList<>();
078                        for (Object e: (Collection<?>) spec) {
079                                ret.add(load(e, base, progressMonitor));
080                        }
081                        return ret;
082                }
083                
084                // Scalar - return AS-IS
085                return spec;
086        }
087        
088        default Object loadYaml(String yamlString, URI base, ProgressMonitor progressMonitor) throws Exception {
089                Yaml yaml = MarkingYamlConstructor.createMarkingYaml(base == null ? null : base.toString());
090                return load(yaml.load(yamlString), base, progressMonitor);
091        }
092        
093        default Object loadYaml(InputStream in, URI base, ProgressMonitor progressMonitor) throws Exception {
094                try (Reader reader = new InputStreamReader(in, getCharset())) {
095                        return loadYaml(reader, base, progressMonitor);
096                }
097        }
098        
099        default Object loadYaml(Reader reader, URI base, ProgressMonitor progressMonitor) throws Exception {
100                Yaml yaml = MarkingYamlConstructor.createMarkingYaml(base == null ? null : base.toString());
101                return load(yaml.load(reader), base, progressMonitor);
102        }
103        
104        default Object loadYaml(URL url, ProgressMonitor progressMonitor) throws Exception {
105                progressMonitor.worked(1, "Loading YAML from " + url);
106                Yaml yaml = MarkingYamlConstructor.createMarkingYaml(url.toString());
107                try (Reader reader = new InputStreamReader(url.openStream(), getCharset())) {
108                        return load(yaml.load(reader), URI.createURI(url.toString()), progressMonitor);
109                } catch (MarkedYAMLException e) {
110                        throw new ConfigurationException(e.getMessage(), e, new MarkerImpl(url.toString(), e.getProblemMark()));
111                }
112        }
113        
114        default Object loadYaml(File file, ProgressMonitor progressMonitor) throws Exception {
115                return loadYaml(file.toURI().toURL(), progressMonitor);
116        }
117                
118        default Object loadJsonObject(String str, URI base, ProgressMonitor progressMonitor) throws Exception {
119                JSONObject jsonObject = new JSONObject(new JSONTokener(str));
120                return load(jsonObject.toMap(), base, progressMonitor);
121        }
122        
123        default Object loadJsonObject(InputStream in, URI base, ProgressMonitor progressMonitor) throws Exception {
124                try (Reader reader = new InputStreamReader(in, getCharset())) { 
125                        return loadJsonObject(reader, base, progressMonitor);
126                }
127        }
128        
129        default Object loadJsonObject(Reader reader, URI base, ProgressMonitor progressMonitor) throws Exception {
130                JSONObject jsonObject = new JSONObject(new JSONTokener(reader));
131                return load(jsonObject.toMap(), base, progressMonitor);
132        }
133        
134        default Object loadJsonObject(URL url, ProgressMonitor progressMonitor) throws Exception {
135                progressMonitor.worked(1, "Loading JSON object from " + url);
136                return loadJsonObject(url.openStream(), URI.createURI(url.toString()), progressMonitor);
137        }
138                
139        default Object loadJsonArray(String str, URI base, ProgressMonitor progressMonitor) throws Exception {
140                JSONArray jsonArray = new JSONArray(new JSONTokener(str));
141                return load(jsonArray.toList(), base, progressMonitor);
142        }
143        
144        default Object loadJsonArray(InputStream in, URI base, ProgressMonitor progressMonitor) throws Exception {
145                try (Reader reader = new InputStreamReader(in, getCharset())) {
146                        return loadJsonArray(reader,  base, progressMonitor);
147                }
148        }
149        
150        default Object loadJsonArray(Reader reader, URI base, ProgressMonitor progressMonitor) throws Exception {
151                JSONArray jsonArray = new JSONArray(new JSONTokener(reader));
152                return load(jsonArray.toList(), base, progressMonitor);
153        }
154        
155        default Object loadJsonArray(URL url, ProgressMonitor progressMonitor) throws Exception {
156                progressMonitor.worked(1, "Loading JSON array from " + url);
157                return loadJsonArray(url.openStream(), URI.createURI(url.toString()), progressMonitor);
158        }
159        
160        /**
161         * Override to customize charset.
162         * @return
163         */
164        default Charset getCharset() {
165                return Charset.defaultCharset();
166        }
