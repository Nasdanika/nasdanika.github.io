001package org.nasdanika.common;
002
003import java.lang.reflect.Constructor;
004import java.lang.reflect.InvocationTargetException;
005import java.lang.reflect.Method;
006import java.util.Arrays;
007import java.util.Optional;
008
009import org.json.JSONArray;
010import org.json.JSONObject;
011
012public class ReflectiveConverter implements Converter {
013
014        @ConverterMethod
015        public String toString(Object value) {
016                return String.valueOf(value);
017        }
018        
019        public static Class<?> box(Class<?> primitive) {
020                if (primitive == boolean.class) {
021                        return Boolean.class;
022                }
023                if (primitive == byte.class) {
024                        return Byte.class;
025                }
026                if (primitive == char.class) {
027                        return Character.class;
028                }
029                if (primitive == double.class) {
030                        return Double.class;
031                }
032                if (primitive == float.class) {
033                        return Float.class;
034                }
035                if (primitive == int.class) {
036                        return Integer.class;
037                }
038                if (primitive == long.class) {
039                        return Long.class;
040                }
041                if (primitive == short.class) {
042                        return Short.class;
043                }
044                if (primitive == void.class) {
045                        return Void.class;
046                }
047                return primitive;
048        }
049
050        /**
051         * Performs conversion using methods annotated with {@link ConverterMethod} first. A method with a more specific compatible parameter type take precedence over a method with more general type.
052         * If conversion cannot be done with converter methods then constructor conversion is attempted in a similar way - a constructor with a more specific compatible parameter types takes precedence over a less specific.
053         */
054        @SuppressWarnings("unchecked")
055        @Override
056        public <T> T convert(Object source, Class<T> type) {
057                // Adaptable conversion
058                if (source instanceof Adaptable) {
059                        T ret = ((Adaptable) source).adaptTo(type);
060                        if (ret != null) {
061                                return ret;
062                        }
063                }
064                
065                // Converter methods conversion
066                Optional<Method> cm = Arrays.stream(getClass().getMethods())
067                                .filter(m -> m.getAnnotation(ConverterMethod.class) != null 
068                                        && m.getParameterCount() == 1 
069                                        && m.getParameterTypes()[0].isInstance(source)
070                                        && type.isAssignableFrom(m.getReturnType()))
071                                .sorted((m1, m2) -> compare(m1.getParameterTypes()[0], m2.getParameterTypes()[0]))
072                                .findFirst();
073                
074                if (cm.isPresent()) {
075                        try {
076                                return (T) cm.get().invoke(this,source);
077                        } catch (IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
078                                throw new NasdanikaException("Error converting "+source+" to "+type+" using converter method "+cm.get()+": "+e, e);
079                        }
080                }                       
081                
082                if (acceptConstructorConversionTargetType(type)) {
083                        // Constructor conversion
084                        Optional<Constructor<?>> co = Arrays.stream(box(type).getConstructors())
085                                .filter(c -> c.getParameterCount() == 1 && c.getParameterTypes()[0].isInstance(source))
086                                .sorted((c1, c2) -> compare(c1.getParameterTypes()[0], c2.getParameterTypes()[0]))
087                                .findFirst();
088                        
089                        if (co.isPresent()) {
090                                try {
091                                        return (T) co.get().newInstance(source);
092                                } catch (InstantiationException | IllegalAccessException | IllegalArgumentException     | InvocationTargetException e) {
093                                        throw new NasdanikaException("Error converting "+source+" to "+type+" using constructor "+co.get()+": "+e, e);
094                                }
095                        }
096                }                                               
097                return null;
098        }       
099        
100        /**
101         * Allows to filter out constructor conversions. This implementation filters out {@link JSONObject} and {@link JSONArray}.
102         * @param type
103         * @return
104         */
105        protected boolean acceptConstructorConversionTargetType(Class<?> type) {
106                if (JSONObject.class.equals(type) || JSONArray.class.equals(type)) {
107                        return false;
108                }
109                return true;
110        }
111        
112        // Compares two classes for specificity, more specific (subclass) is "less", e.g. if t1 is a superclass of t2 this method returns 1.
113        protected int compare(Class<?> t1, Class<?> t2) {
114                if (t1 == t2) {
115                        return 0;
116                }
117                if (t1.isAssignableFrom(t2)) {
118                        return 1;
119                }
120                if (t2.isAssignableFrom(t1)) {
121                        return -1;
122                }
123                return 0; // Not comparable
124        }
125
126}
