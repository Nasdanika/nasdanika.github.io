001package org.nasdanika.common.persistence;
002
003import org.nasdanika.common.SupplierFactory;
004
005public class DelegatingSupplierFactoryFeature<T> extends AbstractFeatureDelegate<Feature<?>> implements SupplierFactoryFeature<T> {
006
007        public DelegatingSupplierFactoryFeature(Feature<?> delegate) {
008                super(delegate);
009        }
010
011        @SuppressWarnings("unchecked")
012        @Override
013        public SupplierFactory<T> getValue() {
014                return (SupplierFactory<T>) SupplierFactory.adapt(delegate.getValue(), "Feature " + getKey());
015        }
016
017}
