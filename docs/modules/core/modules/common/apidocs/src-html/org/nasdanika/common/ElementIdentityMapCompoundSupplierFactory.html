001package org.nasdanika.common;
002
003import java.util.ArrayList;
004import java.util.Collection;
005import java.util.IdentityHashMap;
006import java.util.List;
007import java.util.Map;
008import java.util.function.Function;
009
010public class ElementIdentityMapCompoundSupplierFactory<T> implements SupplierFactory<java.util.function.Function<SupplierFactory<T>,T>> {
011
012        private String name;
013        private List<SupplierFactory<T>> elements = new ArrayList<>();
014
015        public ElementIdentityMapCompoundSupplierFactory(String name, Collection<SupplierFactory<T>> supplierFactories) {
016                this.name = name;
017                this.elements.addAll(elements);
018        }
019        
020        public ElementIdentityMapCompoundSupplierFactory(String name) {
021                this.name = name;
022        }
023
024        @Override
025        public Supplier<java.util.function.Function<SupplierFactory<T>,T>> create(Context context) throws Exception {
026                Map<SupplierFactory<T>, Supplier<T>> supplierMap = new IdentityHashMap<>();
027                ElementIdentityMapCompoundSupplier<T> compoundSupplier = new ElementIdentityMapCompoundSupplier<>(name);                
028                
029                for (SupplierFactory<T> e: elements) {
030                        if (e != null) {
031                                Supplier<T> supplier = e.create(context);
032                                supplierMap.put(e, supplier);
033                                compoundSupplier.put(supplier);
034                        }
035                }
036                
037                class Filter extends FilterExecutionParticipant<Supplier<java.util.function.Function<Supplier<T>,T>>> implements Supplier<java.util.function.Function<SupplierFactory<T>,T>> {
038
039                        public Filter(Supplier<java.util.function.Function<Supplier<T>,T>> target) {
040                                super(target);
041                        }
042
043                        @Override
044                        public Function<SupplierFactory<T>, T> execute(ProgressMonitor progressMonitor) throws Exception {
045                                return ((Function<SupplierFactory<T>, Supplier<T>>) supplierMap::get).andThen(target.execute(progressMonitor));
046                        }
047                        
048                }
049                
050                return new Filter(compoundSupplier);
051        }
052        
053        public void put(SupplierFactory<T> element) {
054                if (element != null) {
055                        elements.add(element);
056                }
057        }       
058
059}
