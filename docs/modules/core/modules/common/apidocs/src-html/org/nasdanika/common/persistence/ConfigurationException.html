001package org.nasdanika.common.persistence;
002
003import java.util.Stack;
004
005import org.nasdanika.common.NasdanikaException;
006
007/**
008 * Exception to indicate error in configuration resources.
009 * @author Pavel
010 *
011 */
012@SuppressWarnings("serial")
013public class ConfigurationException extends NasdanikaException {
014        
015        private Marker marker;
016
017        public ConfigurationException(String message, Marker marker) {
018                super(marker == null ? message : message + " at " + Marker.toString(marker));
019                this.marker = marker;
020        }
021
022        public ConfigurationException(String message, Throwable cause, Marker marker) {
023                super(marker == null ? message : message + " at " + Marker.toString(marker), cause);
024                this.marker = marker;
025        }
026
027        public ConfigurationException(String message, Marked marked) {
028                this(message, marked == null ? null : marked.getMarker());
029        }
030
031        public ConfigurationException(String message, Throwable cause, Marked marked) {
032                this(message, cause, marked == null ? null : marked.getMarker());
033        }
034        
035        public Marker getMarker() {
036                return marker;
037        }
038        
039        private static ThreadLocal<Stack<Marker>> threadMarkerStack = new ThreadLocal<Stack<Marker>>() {
040
041                protected java.util.Stack<Marker> initialValue() {
042                        return new Stack<>();
043                };
044                
045        };
046        
047        public static void pushThreadMarker(Marker marker) {
048                threadMarkerStack.get().push(marker);
049        }
050        
051        public static void popThreadMarker() {
052                threadMarkerStack.get().pop();
053        }       
054        
055        public static void pushThreadMarked(Marked marked) {
056                pushThreadMarker(marked == null ? null : marked.getMarker());
057        }
058        
059        public static Marker peekThreadMarker() {
060                Stack<Marker> stack = threadMarkerStack.get();
061                return stack.isEmpty() ? null : stack.peek(); 
062        }
063
064        public ConfigurationException(String message) {
065                this(message, peekThreadMarker());
066        }
067
068        public ConfigurationException(String message, Throwable cause) {
069                this(message, cause, peekThreadMarker());
070        }
071
072}
