001package org.nasdanika.common.persistence;
002
003import org.eclipse.emf.common.util.URI;
004import org.nasdanika.common.ProgressMonitor;
005
006/**
007 * Loads feature value using loader.
008 * @author Pavel
009 *
010 */
011public class Reference extends Attribute<Object> {
012
013        public Reference(Object key, boolean isDefault, boolean required, Object defaultValue, String description, Object... exclusiveWith) {
014                super(key, isDefault, required, defaultValue, description, exclusiveWith);
015        }
016        
017        @Override
018        public Object create(ObjectLoader loader, Object config, URI base, ProgressMonitor progressMonitor, Marker marker)      throws Exception {
019                return loader.load(config, base, progressMonitor);
020        }
021
022}
