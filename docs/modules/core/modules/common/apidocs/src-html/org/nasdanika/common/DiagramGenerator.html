001package org.nasdanika.common;
002
003import java.io.IOException;
004import java.io.OutputStream;
005import java.net.HttpURLConnection;
006import java.net.URL;
007import java.net.URLConnection;
008import java.nio.charset.StandardCharsets;
009import java.security.MessageDigest;
010
011import org.apache.commons.codec.binary.Hex;
012import org.nasdanika.common.resources.Container;
013
014/**
015 * Generates PlantUML diagrams.
016 * @author Pavel
017 *
018 */
019public interface DiagramGenerator {
020        
021        /**
022         * PlantUML diagram dialect
023         * @author Pavel
024         *
025         */
026        enum Dialect {
027
028                /**
029                 * For sequence, use case, class, activity, component, state, object, deployment, timing, and network diagrams.
030                 */
031                UML,
032                
033                /**
034                 * For wireframe diagrams - https://plantuml.com/salt
035                 */
036                SALT,
037                
038                /**
039                 * For Gantt charts - https://plantuml.com/gantt-diagram
040                 */
041                GANTT,
042                
043                /**
044                 * For mind maps - https://plantuml.com/mindmap-diagram
045                 */
046                MINDMAP,
047                
048                /**
049                 * For work breakdown structures - https://plantuml.com/wbs-diagram
050                 */
051                WBS,
052                
053                /**
054                 * For generating drawio (https://diagrams.net/) embedded diagram HTML from diagram file contents as a spec. 
055                 */
056                DRAWIO,
057                
058                /**
059                 * For generating Mermaid (https://mermaid-js.github.io/mermaid/#/) diagrams.
060                 */
061                MERMAID
062                
063        }
064        
065        /**
066         * Uses local PlantUML.
067         */
068        DiagramGenerator INSTANCE = new DiagramGeneratorImpl();
069        
070        /**
071         * Creates an instance of {@link DiagramGenerator} which does a POST to the service URL postfixed with /&lt;dialect name&gt; with diagram text spec as payload
072         * and returns the payload as String. UTF-8 is used for character encoding. 
073         * The service shall call the INSTANCE's <code>generateDiagram()</code>. It can also perform caching of generated images. Return code 200 is treated as success. 
074         * Any other response code is treated as an error.
075         * Remoting can be used in situations where GrapViz (https://graphviz.org/) is not available on diagram generating machines or to improve performance via caching.
076         * 
077         * An example of server implementation with SpringBoot can be found here - https://github.com/Nasdanika/spring-exec/blob/main/src/main/java/org/nasdanika/spring/exec/controllers/ExecController.java#L344
078         * @param service Service URL with a trailing slash /. Must be an HTTP(s) URL.
079         * @return Diagram image and image map.
080         */
081        static DiagramGenerator createClient(URL service) {
082                
083                return new DiagramGenerator() {
084
085                        @Override
086                        public String generateDiagram(String spec, Dialect dialect) throws IOException {
087                                URL dialectURL = new URL(service, dialect.name());
088                                URLConnection connection = dialectURL.openConnection();
089                                if (!(connection instanceof HttpURLConnection)) {
090                                        throw new IllegalArgumentException("Not an HTTP(s) url: "+dialectURL);
091                                }
092                                
093                                HttpURLConnection httpConnection = (HttpURLConnection) connection;
094                                httpConnection.setRequestMethod("POST");
095                                httpConnection.setDoOutput(true);
096                                try (OutputStream out = connection.getOutputStream()) {
097                                        out.write(spec.getBytes(StandardCharsets.UTF_8));
098                                }
099                                
100                                int responseCode = httpConnection.getResponseCode();
101                                if (responseCode == 200) {
102                                        return DefaultConverter.INSTANCE.toString(httpConnection.getInputStream());
103                                }
104                                
105                                throw new IOException("HTTP Call to "+dialectURL+" has failed with response: "+responseCode+" "+httpConnection.getResponseMessage());
106                        }
107                        
108                };
109                
110        }
111        
112        /**
113         * Generates a PlantUML diagram with an image map from text definition. The definition shall not contain start and end tags.
114         * @return
115         * @throws IOException 
116         */
117        String generateDiagram(String spec, Dialect dialect) throws Exception;
118                        
119        default String generateUmlDiagram(String spec) throws Exception {
120                return generateDiagram(spec, Dialect.UML);
121        }
122                
123        default String generateWireframeDiagram(String spec) throws Exception {
124                return generateDiagram(spec, Dialect.SALT);
125        }
126        
127        default String generateGanttDiagram(String spec) throws Exception {
128                return generateDiagram(spec, Dialect.GANTT);
129        }
130        
131        default String generateMindmapDiagram(String spec) throws Exception {
132                return generateDiagram(spec, Dialect.MINDMAP);
133        }
134        
135        default String generateWbsDiagram(String spec) throws Exception {
136                return generateDiagram(spec, Dialect.WBS);
137        }
138        
139        default String generateDrawioDiagram(String spec) throws Exception {
140                return generateDiagram(spec, Dialect.DRAWIO);
141        }
142        
143        /**
144         * Creates a diagram generator facading this one and cacheing generated diagrams by dialect and diagram text SHA-256 digest.
145         * @param container
146         * @param progressMonitor
147         * @return
148         */
149        default DiagramGenerator cachingDiagramGenerator(Container<String> container, ProgressMonitor progressMonitor) {
150                
151                return new DiagramGenerator() {
152                        
153                        private int hits;
154                        private int misses;
155
156                        @Override
157                        public String generateDiagram(String spec, Dialect dialect) throws Exception {
158                                String cachePath = dialect.name() + "/" + Hex.encodeHexString(MessageDigest.getInstance("SHA-256").digest(spec.getBytes(StandardCharsets.UTF_8))) + ".html";
159                                Object ret = container.find(cachePath, progressMonitor);
160                                if (ret instanceof String) {
161                                        ++hits;
162                                        return (String) ret;
163                                }
164                                String diagram = DiagramGenerator.this.generateDiagram(spec, dialect);
165                                container.put(cachePath, diagram, progressMonitor);
166                                ++misses;
167                                return diagram;
168                        }
169                        
170                        @Override
171                        public String toString() {
172                                return super.toString() + " hits: " + hits + ", misses: " + misses;
173                        }
174                        
175                };
176        }
177        
178}
