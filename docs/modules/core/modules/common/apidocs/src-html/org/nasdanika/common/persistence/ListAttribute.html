001package org.nasdanika.common.persistence;
002
003import java.net.URL;
004import java.util.ArrayList;
005import java.util.Collection;
006import java.util.Collections;
007import java.util.List;
008
009import org.nasdanika.common.ProgressMonitor;
010import org.nasdanika.common.Util;
011
012/**
013 * If config is a list loads each element using createElement(), otherwise creates a singleton loaded with createElement()
014 * @author Pavel
015 *
016 * @param <T>
017 */
018public class ListAttribute<T> extends Attribute<List<T>> {
019
020        public ListAttribute(Object key, 
021                        boolean isDefault, 
022                        boolean required, 
023                        List<T> defaultValue, 
024                        String description, 
025                        Object... exclusiveWith) {
026                super(key, isDefault, required, defaultValue, description, exclusiveWith);
027        }
028        
029        @Override
030        public List<T> create(ObjectLoader loader, Object config, URL base, ProgressMonitor progressMonitor, Marker marker)     throws Exception {
031                if (config instanceof Collection) {
032                        List<T> ret = new ArrayList<>();
033                        int idx = 0;
034                        for (Object element: (Collection<?>) config) {
035                                ret.add(createElement(loader, element, base, progressMonitor, Util.getMarker((Collection<?>) config, idx++)));
036                        }
037                        return ret;
038                }
039                
040                return Collections.singletonList(createElement(loader, config, base, progressMonitor, marker));
041        }
042
043        /**
044         * Creates element list.
045         * @param loader
046         * @param element
047         * @param base
048         * @param progressMonitor
049         * @param marker
050         * @return
051         * @throws Exception
052         */
053        @SuppressWarnings("unchecked")
054        protected T createElement(ObjectLoader loader, Object element, URL base, ProgressMonitor progressMonitor, Marker marker)        throws Exception { 
055                return (T) element; 
056        }
057        
058}
