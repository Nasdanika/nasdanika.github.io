020/**
021 * Container of entries and other containers provides an abstraction for dealing with hierarchies/trees.
022 * While a container does not necessarily have an out-of-JVM representation it extends {@link Resource} for convenience because these two 
023 * interfaces share common characteristics.
024 * @author Pavel
025 *
026 * @param <E> Element type.
027 */
028public interface Container<E> extends Resource<E> {
029        
030        /**
031         * Finds an existing element or container with a given path.
032         * @param path Child resource path. 
033         * @return Element if it exists or null.
034         */
035        Object find(String path, ProgressMonitor monitor);
036        
037        /**
038         * @param path
039         * @return Element at the path or null if the element does not exist, is not created on access, or it would not be possible to create such an entity, e.g. a container with this path already exists, or one of path elements is not a container.
040         */
041        E get(String path, ProgressMonitor monitor);
042        
043        /**
044         * 
045         * @param path Element path
046         * @param element Element
047         * @throws IllegalArgumentException if it is not possible to put an element at this path. E.g. getContainer(parent path) returns null.
048         */
049        void put(String path, E element, ProgressMonitor monitor) throws IllegalArgumentException;
050        
051        /**
052         * Deletes an element or container at a given path.
053         * @param path
054         * @return Removed element/container or null if it did not exist or the container removed it without retrieving. 
055         */
056        Object delete(String path, ProgressMonitor monitor);
057        
058        /**
059         * Gets an element from the container creating it using a factory if it doesn't already exist.
060         * @param path Path.
061         * @param factory Factory
062         * @throws IllegalArgumentException If it is not possible to put element at the given path.
063         * @return
064         */
065        default E get(String path, BiFunction<String,ProgressMonitor,E> factory, ProgressMonitor monitor) throws IllegalArgumentException {
066                try {
067                        E ret = get(path, monitor.split("Getting existing element", 1));
068                        if (ret == null && factory != null) {                   
069                                ret = factory.apply(path, monitor.split("Creating a new element", 1, path));
070                                if (ret != null) {
071                                        put(path, ret, monitor.split("Putting a new element", 1, path, ret));
072                                }
073                        }
074                        return ret;
075                } finally {
076                        monitor.close();
077                }
078        }
079        
080        /**
081         * Returns a container for a given path. It is a logical operation - the container doesn't have to exist.
082         * @param path
083         * @return Container instance or null if it would not be possible to create such a container, e.g. an element with this path already exists or one of path elements is not a container.
084         */
085        Container<E> getContainer(String path, ProgressMonitor monitor);
086        
087        /**
088         * Lists existing children.
089         * @return A list of existing children. May return null for containers where children can't be retrieved, e.g. write-only containers.
090         */
091        Map<String, Object> getChildren(ProgressMonitor monitor);
092        
093        interface Copier<T,R> {
094                
095                /**
096                 * Makes a copy of an element, merges with the existing value.
097                 * @param element
098                 * @param existingChild Existing child in the target container - element or sub-container.
099                 * @param monitor
100                 * @return
101                 */
102                R copy(String path, T source, Object existingElement, ProgressMonitor monitor); 
103                
104        }
105                
106        /**
107         * Recursively copies children to another container.
108         * @param <F> Element type of the other container.
109         * @param container Target container.
110         * @param copier Copier of elements, converting type, and merging with the existing element if it is present. The first argument is 
111         * element being copied, the second is an existing element or null if there is no such element.
112         * @param monitor Progress monitor
113         */
114        @SuppressWarnings("unchecked")
115        default <F> void copy(Container<F> target, Copier<E,F> copier, ProgressMonitor monitor) {
116                try {
117                        Map<String, Object> children = getChildren(monitor.split("Getting children", 1, this));         
118                        try (ProgressMonitor subMonitor = monitor.split("Copying "+getPath(), children.size(), this)) {
119                                if (children != null) {                         
120                                        children.forEach((path, child) -> {
121                                                if (child instanceof Container) {
122                                                        Container<F> targetChild = target.getContainer(path, monitor.split("Getting container "+path, 1));
123                                                        if (targetChild == null) {
124                                                                throw new IllegalArgumentException("Cannot obtain target container for path "+path);
125                                                        }
126                                                        ((Container<E>) child).copy(targetChild, copier, subMonitor);
127                                                } else {
128                                                        F copy = copier.copy(path, (E) child, target.find(path, subMonitor.split("Finiding", 1, path)), subMonitor.split("Copying "+path, 1));
129                                                        target.put(path, copy, subMonitor.split("Putting copied element", 1, path, copy)); 
130                                                }
131                                        });
132                                }
133                        }
134                } finally {
135                        monitor.close();
136                }
137        }
138        
139        /**
140         * Recursively moves children - copies and then removes.
141         */
142        @SuppressWarnings("unchecked")
143        default <F> void move(Container<F> target, Copier<E,F> copier, ProgressMonitor monitor) {
144                try {
145                        Map<String, Object> children = getChildren(monitor.split("Getting children", 1, this));         
146                        try (ProgressMonitor subMonitor = monitor.split("Copying "+getPath(), children.size(), this)) {
147                                if (children != null) { 
148                                        Collection<String> paths = new ArrayList<>();
149                                        children.forEach((path, child) -> {
150                                                if (child instanceof Container) {
151                                                        Container<F> targetChild = target.getContainer(path, monitor.split("Getting container "+path, 1));
152                                                        if (targetChild == null) {
153                                                                throw new IllegalArgumentException("Cannot obtain target container for path "+path);
154                                                        }
155                                                        ((Container<E>) child).move(targetChild, copier, subMonitor);
156                                                } else {
157                                                        F copy = copier.copy(path, (E) child, target.find(path, subMonitor.split("Finiding", 1, path)), subMonitor.split("Copying "+path, 1));
158                                                        target.put(path, copy, subMonitor.split("Putting copied element", 1, path, copy)); 
159                                                }
160                                                paths.add(path);
161                                        });
162                                        paths.forEach(path -> delete(path, monitor.split("Removing", 1, path)));
163                                }
164                        }
165                } finally {
166                        monitor.close();
167                }
168        }       
169        
170        /**
171         * Adapts to a different element type.
172         * @param <F> Element type to adapt to
173         * @param adapter Adapts element E to target type F. Takes element path and the element itself.
174         * @param converter Converts element F to E on put and get with a factory. Takes element path and the element itself. 
175         * @return
176         */
177        default <F> Container<F> adapt(BiFunction<String,? super E,F> adapter, BiFunction<String,? super F,E> converter) {
178                return new Container<F>() {
179
180                        @SuppressWarnings("unchecked")
181                        @Override
182                        public Object find(String path, ProgressMonitor monitor) {
183                                Object ret = Container.this.find(path, monitor);
184                                if (ret instanceof Container) {
185                                        return ((Container<E>) ret).adapt(adapter, converter);
186                                }
187                                return adapter.apply(path, (E) ret);
188                        }
189
190                        @Override
191                        public F get(String path, ProgressMonitor monitor) {
192                                return adapter.apply(path, Container.this.get(path, monitor));
193                        }
194
195                        @Override
196                        public void put(String path, F element, ProgressMonitor monitor) throws IllegalArgumentException {
197                                Container.this.put(path, converter.apply(path, element), monitor);                              
198                        }
199
200                        @SuppressWarnings("unchecked")
201                        @Override
202                        public Object delete(String path, ProgressMonitor monitor) {
203                                Object deleted = Container.this.delete(path, monitor);
204                                if (deleted instanceof Container) {
205                                        return ((Container<E>) deleted).adapt(adapter, converter);
206                                }
207                                return adapter.apply(path, (E) deleted);
208                        }
209
210                        @Override
211                        public Container<F> getContainer(String path, ProgressMonitor monitor) {
212                                Container<E> ret = Container.this.getContainer(path, monitor);
213                                return ret.adapt(adapter, converter);
214                        }
215
216                        @SuppressWarnings("unchecked")
217                        @Override
218                        public Map<String, Object> getChildren(ProgressMonitor monitor) {
219                                Map<String, Object> ret = new HashMap<>();
220                                Container.this.getChildren(monitor).forEach((path, child) -> {
221                                        if (child instanceof Container) {
222                                                ret.put(path, ((Container<E>) ret).adapt(adapter, converter));
223                                        } else {
224                                                ret.put(path, adapter.apply(path, (E) child));
225                                        }                                       
226                                }); 
227                                return Collections.unmodifiableMap(ret);
228                        }
229
230                        @Override
231                        public Container<F> getParent() {
232                                return Container.this.getParent().adapt(adapter, converter);
233                        }
234
235                        @Override
236                        public String getName() {
237                                return Container.this.getName();
238                        }
239
240                        @Override
241                        public void copy(Container<? super F> container, String path, ProgressMonitor monitor) {
242                                throw new UnsupportedOperationException("Implement when this one is thrown");
243                        }
244
245                        @Override
246                        public void move(Container<? super F> container, String path, ProgressMonitor monitor) {
247                                throw new UnsupportedOperationException("Implement when this one is thrown");
248                        }
249                        
250                        @Override
251                        public String toString() {
252                                return getClass().getName()+"("+getPath()+")";
253                        }
254                        
255                };
256        }
257        
258        /**
259         * Filters this container to return null for getParent() so this container is considered to be a root container.
260         * @return
261         */
262        default Container<E> asRoot() {
263                return new ContainerFilter<E>(this) {
264
265                        @Override
266                        public Container<E> getParent() {
267                                return null;
268                        }
269                        
270                };
271        }

273        /**
274         * TODO - filter -> read predicate, write predicate
275         * Filters this container. 
276         * @param filter Child path filter. 
277         * @return
278         */
279        default Container<E> filter(Predicate<String> filter) {
280                return new ContainerFilter<E>(this) {
281                        
282                        @SuppressWarnings("unchecked")
283                        @Override
284                        public Map<String, Object> getChildren(ProgressMonitor monitor) {
285                                Map<String, Object> ret = new HashMap<>();
286                                Container.this.getChildren(monitor).forEach((path, child) -> {
287                                        if (filter.test(path)) {
288                                                if (child instanceof Container) {
289                                                        ret.put(path, ((Container<E>) ret).filter(filter));
290                                                } else {
291                                                        ret.put(path, child);
292                                                }
293                                        }
294                                }); 
295                                return Collections.unmodifiableMap(ret);
296                        }
297                        
298                        
299                        @Override
300                        public Object find(String path, ProgressMonitor monitor) {
301                                return filter.test(path) ? super.find(path, monitor) : null;
302                        }
303                        
304                        @Override
305                        public Container<E> getContainer(String path, ProgressMonitor monitor) {
306                                return filter.test(path) ? super.getContainer(path, monitor).filter(filter) : null;
307                        }
308                        
309                        @Override
310                        public E get(String path, ProgressMonitor monitor) {
311                                return filter.test(path) ? super.get(path, monitor) : null;
312                        }
313                        
314                        @Override
315                        public void put(String path, E element, ProgressMonitor monitor) throws IllegalArgumentException {
316                                if (!filter.test(path)) {
317                                        throw new IllegalArgumentException("Path doesn't match the filter: "+path);
318                                }
319                                super.put(path, element, monitor);
320                        }
321                        
322                };
323        }

325        /**
326         * Loads entries from a {@link ZipInputStream} into this container. This method used getXXX() methods to obtain container children.
327         * If getXXX() method returns null of if {@link Entity} entry is not null and canWrite() returns false then an exception is thrown.
328         * @param zipInputStream ZipInputStream.
329         * @param filter Entry name filter. Can be null.
330         * @param contentLoader Converts input stream to the element type. The first argument is the path.
331         * @param progressMonitor Progress monitor.
332         * @throws IOException
333         */
334        default void load(
335                        ZipInputStream zipInputStream, 
336                        Predicate<String> filter, 
337                        Copier<InputStream,E> contentLoader, 
338                        ProgressMonitor progressMonitor) throws IOException {
339                
340                try {
341                        ZipEntry zipEntry;
342                        while ((zipEntry = zipInputStream.getNextEntry()) != null) {
343                                try (ProgressMonitor entryMonitor = progressMonitor.split("Entry "+zipEntry.getName(), 1, zipEntry)) {
344                                        String entryName = zipEntry.getName();
345                                        if (filter == null || filter.test(entryName)) {
346                                                if (entryName.endsWith("/")) {
347                                                        if (getContainer(entryName.substring(0, entryName.length() - 1), entryMonitor.split("Getting container "+entryName, 1)) == null) {
348                                                                throw new IOException("Container with path "+entryName+" cannot be created");
349                                                        }
350                                                } else {
351                                                        E element = contentLoader.copy(entryName, zipInputStream, get(entryName, entryMonitor.split("Getting existing", 1, entryName)), entryMonitor.split("Loading "+entryName, 1));
352                                                        if (element != null) {
353                                                                put(entryName, element, entryMonitor.split("Putting copied element", 1, entryName, element));
354                                                        }
355                                                }
356                                        }
357                                        zipInputStream.closeEntry();
358                                }
359                        }
360                } finally {
361                        progressMonitor.close();
362                        zipInputStream.close();
363                }
364        }

366        /**
367         * Stores children of this container to a {@link ZipOutputStream}. 
368         * @param zipOutputStream Zip output stream. This method does not close the stream.
369         * @param prefix Optional path prefix.
370         * @param contentSerializer Converts entity state to the input stream. If null then {@link DefaultConverter}.INSTANCE is used to convert content to {@link InputStream}
371         * @param progressMonitor Progress monitor.
372         * @throws IOException
373         */
374        @SuppressWarnings("unchecked")
375        default void store(
376                ZipOutputStream zipOutputStream, 
377                String prefix,
378                Copier<E,InputStream> contentSerializer, 
379                ProgressMonitor progressMonitor) throws IOException {
380                
381                try {
382                        if (contentSerializer == null) {
383                                contentSerializer = (path, source, existingElement, monitor) -> DefaultConverter.INSTANCE.convert(source, InputStream.class);
384                        }
385                        
386                        if (prefix == null) {
387                                prefix = "";
388                        }
389                        
390                        for (Entry<String, Object> childEntry: getChildren(progressMonitor.split("Getting children", 1, this)).entrySet()) {
391                                try (ProgressMonitor childMonitor = progressMonitor.split("Child "+childEntry.getKey(), 1, childEntry.getValue())) {
392                                        if (childEntry.getValue() instanceof Container) {
393                                                zipOutputStream.putNextEntry(new ZipEntry(prefix+"/"+childEntry.getKey()+"/"));
394                                                zipOutputStream.closeEntry();
395                                                ((Container<E>) childEntry.getValue()).store(zipOutputStream, prefix+childEntry.getKey()+"/", contentSerializer, childMonitor);
396                                        } else {
397                                                zipOutputStream.putNextEntry(new ZipEntry(prefix+childEntry.getKey()));
398                                                try (InputStream contents = contentSerializer.copy(childEntry.getKey(), (E) childEntry.getValue(), null, childMonitor)) {
399                                                        int b;
400                                                        while ((b = contents.read()) != -1) {
401                                                                zipOutputStream.write(b);
402                                                        }
403                                                }
404                                                zipOutputStream.closeEntry();                           
405                                        }
406                                }
407                        }
408                } finally {
409                        progressMonitor.close();
410                }
411        }

413        /**
414         * Deletes all children and then deletes this container from its parent.
415         * @param monitor
416         */
417        default void delete(ProgressMonitor monitor) {
418                new HashMap<>(getChildren(monitor.split("Getting children", 1))).forEach((path, child) -> delete(path, monitor.split("Removing child "+path, 1, path, child)));         
419                Container<E> parent = getParent();
420                if (parent != null) {
421                        parent.delete(getName(), monitor.split("Removing self from the parent", 1, this));
422                }               
423        }
424        
425        default long size(ProgressMonitor monitor) {
426                long[] ret = {0};
427                getChildren(monitor.split("Getting children", 1, this)).forEach((path, child) -> {;
428                        if (child instanceof Resource && ((Resource<?>) child).exists(monitor.split("Checking child existence", 1, child))) {
429                                ret[0] += ((Resource<?>) child).size(monitor.split("Getting child size "+path, 1, child));
430                        }
431                });
432                return ret[0];
433        }
434                
435        /**
436         * Returns true if contains any existing resources.
437         */
438        default boolean exists(ProgressMonitor monitor) {
439                Map<String, Object> children = getChildren(monitor.split("Getting children", 1, this));
440                for (Object child: children.values()) {
441                        if (child instanceof Resource && ((Resource<?>) child).exists(monitor.split("Checking child existence", 1, child))) {
442                                return true;
443                        }
444                };
445                return false;           
446        }
