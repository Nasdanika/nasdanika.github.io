001package org.nasdanika.common.persistence;
002
003/**
004 * Resolves marker to a URL in the source repository.
005 * @author Pavel
006 */
007public interface SourceResolver {
008        
009        /**
010         * Link to source location
011         * @author Pavel
012         * @since 2015.4.2
013         */
014        interface Link {
015                
016                /**
017                 * @return Link text. This method returns location value. 
018                 */
019                default String getText() {
020                        return getLocation();
021                };
022                
023                /**
024                 * @return Source URL. If null then link shall be rendered as plain text returned by getText().
025                 */
026                String getLocation();
027                
028        }
029        
030        /**
031         * @param marker Marker.
032         * @return URL of the marker in the source repository or null.
033         */
034        Link getSource(Marker marker);
035
036}
