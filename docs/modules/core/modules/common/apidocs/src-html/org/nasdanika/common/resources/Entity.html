001package org.nasdanika.common.resources;
002
003import java.util.function.BiFunction;
004import java.util.zip.ZipFile;
005
006import org.nasdanika.common.ProgressMonitor;
007
008/**
009 * Entity is something which has externalizable state. E.g. file is an entity with binary contents/state - a sequence of bytes.
010 * @author Pavel
011 *
012 * @param <T> Entity state type
013 * @param <E> Container element type.
014 */
015public interface Entity<T,E> extends Resource<E> {
016        
017        /**
018         * @param monitor
019         * @return Entity state. 
020         * @throws IllegalStateException if the entity's underlying state storage does not exist or cannot be read.
021         */
022        T getState(ProgressMonitor monitor);
023                
024        /**
025         * 
026         * @param type State type
027         * @param monitor
028         * @return Entity state converted to the the requested content type. E.g. file contents converted to String. This implementation does not perform any conversion and
029         * returns contents AS-IS if it is already of the requested type or throws an {@link UnsupportedOperationException} otherwise.
030         * @throws IllegalStateException if the entity state storage does not exist, cannot be read, or conversion from T to V is not supported.
031         */
032        @SuppressWarnings("unchecked")
033        default <V> V getState(Class<V> type, ProgressMonitor monitor) {
034                T contents = getState(monitor);
035                if (type.isInstance(contents)) {
036                        return (V) contents;
037                }
038                throw new UnsupportedOperationException("Unsupported content type: "+type);
039        }       
040        
041        /**
042         * Sets entity state, e.g. file contents. If the entity storage does not exist this call creates the storage and intermediary containers as needed.
043         * @param state Entity state. 
044         * @param monitor
045         * @throws IllegalStateException if canWrite() returns false.
046         */
047        void setState(T state, ProgressMonitor monitor);
048        
049        /**
050         * Appends state if the entity state exists and calls setState if it doesn't.
051         * @param Entity state.
052         * @param monitor
053         * @throws IllegalStateException if canWrite() returns false.
054         */
055        void appendState(T state, ProgressMonitor monitor);
056
057        /**
058         * @return true if entity state can be read. Some containers may be write-only.
059         */
060        boolean canRead();
061        
062        /**
063         * @return true if entity state can be changed. Some containers may be read-only, e.g. a container backed by a {@link ZipFile} would only support reading resources from it.
064         */
065        boolean canWrite();
066        
067        /**
068         * Copies entity. 
069         * @param container target container, may be the same container.
070         * @param path Path in the target container.
071         * @param monitor Progress monitor.
072         */
073        @SuppressWarnings("unchecked")
074        @Override
075        default void copy(Container<? super E> container, String path, ProgressMonitor monitor) {       
076                if (exists(monitor.split("Checking existence", 1, this))) {
077                        try (
078                                        ProgressMonitor getCopyTargetMonitor = monitor.split("Getting copy target"+getPath(), 1, this); 
079                                        ProgressMonitor readMonitor = monitor.split("Copying "+getPath(), 1, this); 
080                                        ProgressMonitor writeMonitor = monitor.split("Writing "+path, 1, container, path)) {
081                                Object copyTarget = container.get(path, getCopyTargetMonitor);
082                                if (copyTarget instanceof Entity) {
083                                        ((Entity<T,E>) copyTarget).setState(getState(readMonitor), writeMonitor);
084                                } else {
085                                        throw new IllegalArgumentException("Copy target is not an entity: "+copyTarget);
086                                }
087                        }
088                }
089        }
090        
091        /**
092         * Moves entity. This implementation implements move as copy and delete.
093         * @param container target container, may be the same container.
094         * @param path Path in the target container.
095         * @param monitor Progress monitor.
096         */
097        @SuppressWarnings("unchecked")
098        @Override
099        default void move(Container<? super E> container, String path, ProgressMonitor monitor) {
100                if (exists(monitor.split("Checking existence", 1, this))) {
101                        try (
102                                        ProgressMonitor getCopyTargetMonitor = monitor.split("Getting copy target"+getPath(), 1, this); 
103                                        ProgressMonitor readMonitor = monitor.split("Copying "+getPath(), 1, this); 
104                                        ProgressMonitor writeMonitor = monitor.split("Writing "+path, 1, container, path);
105                                        ProgressMonitor deleteMonitor = monitor.split("Deleting"+path, 1, this, path)) {
106                                Object moveTarget = container.get(path, getCopyTargetMonitor);
107                                if (moveTarget instanceof Entity) {
108                                        ((Entity<T,E>) moveTarget).setState(getState(readMonitor), writeMonitor);
109                                        delete(deleteMonitor);
110                                } else {
111                                        throw new IllegalArgumentException("Copy target is not an entity: "+moveTarget);
112                                }
113                        }
114                }
115        }
116        
117        /**
118         * Adapts to a different state type.
119         * @param <V>
120         * @param decoder Decodes T to V.
121         * @param encoder Encodes V to T.
122         * @param sizeConverter converts size of an entity. Size is passed as-is if null.
123         * @return
124         */
125        default <V> Entity<V,E> adapt(BiFunction<Entity<T,E>, T, V> decoder, BiFunction<Entity<T,E>, V, T> encoder, BiFunction<Entity<T,E>, Long, Long> sizeConverter) {
126                return new Entity<V,E>() {
127
128                        @Override
129                        public String getName() {
130                                return Entity.this.getName();
131                        }
132
133                        @Override
134                        public boolean exists(ProgressMonitor monitor) {
135                                return Entity.this.exists(monitor);
136                        }
137
138                        @Override
139                        public Container<E> getParent() {
140                                Container<E> parent = Entity.this.getParent();
141                                @SuppressWarnings("unchecked")
142                                BiFunction<String, E, E> adapter = (path, source) -> {
143                                        return (E) ((Entity<T,E>) source).adapt(decoder, encoder, sizeConverter);
144                                };
145                                
146                                BiFunction<String, E, E> converter = (path, adapted) -> {
147                                        throw new UnsupportedOperationException("Entity put is not supported");
148                                };
149                                return parent == null ? null : parent.adapt(adapter, converter);
150                        }
151
152                        @Override
153                        public void delete(ProgressMonitor monitor) {
154                                Entity.this.delete(monitor);
155                        }
156
157                        @Override
158                        public String getPath() {
159                                return Entity.this.getPath();
160                        }
161
162                        @Override
163                        public V getState(ProgressMonitor monitor) {
164                                return decoder.apply(Entity.this, Entity.this.getState(monitor));
165                        }
166
167                        @Override
168                        public void setState(V contents, ProgressMonitor monitor) {
169                                Entity.this.setState(encoder.apply(Entity.this, contents), monitor);
170                        }
171
172                        @Override
173                        public void appendState(V contents, ProgressMonitor monitor) {
174                                Entity.this.appendState(encoder.apply(Entity.this, contents), monitor);
175                        }
176
177                        @Override
178                        public boolean canRead() {
179                                return decoder != null && Entity.this.canRead();
180                        }
181
182                        @Override
183                        public boolean canWrite() {
184                                return encoder != null && Entity.this.canWrite();
185                        }
186
187                        @Override
188                        public long size(ProgressMonitor monitor) {
189                                return sizeConverter == null ? Entity.this.size(monitor) : sizeConverter.apply(Entity.this, Entity.this.size(monitor));
190                        }
191
192                        @Override
193                        public String toString() {
194                                return getClass().getName()+"("+getPath()+")";
195                        }
196                        
197                };
198        }
