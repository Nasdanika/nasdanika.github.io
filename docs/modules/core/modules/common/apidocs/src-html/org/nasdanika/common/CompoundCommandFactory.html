001package org.nasdanika.common;
002
003import java.util.ArrayList;
004import java.util.Arrays;
005import java.util.Collection;
006import java.util.List;
007import java.util.concurrent.ExecutorService;
008
009public class CompoundCommandFactory implements CommandFactory {
010
011        protected String name;
012        private List<CommandFactory> elements = new ArrayList<>();
013
014        public CompoundCommandFactory(String name, Collection<CommandFactory> elements) {
015                this.name = name;
016                this.elements.addAll(elements);
017        }
018        
019        @SafeVarargs
020        public CompoundCommandFactory(String name, CommandFactory... elements) {
021                this(name, Arrays.asList(elements));
022        }
023
024        @Override
025        public Command create(Context context) throws Exception {
026                CompoundCommand ret = createCompoundCommand(context);
027                for (CommandFactory e: elements) {
028                        ret.add(e.create(context));
029                }
030                return ret;
031        }
032
033        /**
034         * Override to customize {@link CompoundCommand} to which sub-commands will be added.
035         * @param context
036         * @return
037         */
038        protected CompoundCommand createCompoundCommand(Context context) {
039                return new CompoundCommand(name, context.get(ExecutorService.class));
040        }
041        
042        public void add(CommandFactory element) {
043                elements.add(element);
044        }
045
046}
