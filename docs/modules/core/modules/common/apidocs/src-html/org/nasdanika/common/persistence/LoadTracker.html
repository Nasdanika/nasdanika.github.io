001package org.nasdanika.common.persistence;
002
003import org.eclipse.emf.ecore.EStructuralFeature;
004
005/**
006 * Adapter interface to track features which were loaded. 
007 * May be used when some features can be either loaded or computed.
008 * Defined in Common instead of EMF because it needs to be available in Ncore.
009 * @author Pavel
010 *
011 */
012public interface LoadTracker {
013        
014        /**
015         * Feature is already loaded.
016         * This method can be used to decide whether to inject 
017         * a feature value derived from other loaded features 
018         * it a feature is not yet loaded. 
019         * @param feature
020         * @return
021         */
022        boolean isLoaded(EStructuralFeature feature);
023        
024        /**
025         * Feature is being loaded. This method
026         * returns true only when the loader calls the feature setter.
027         * It can be used to decide whether to trigger computations of  
028         * features derived from this feature.
029         * @param feature
030         * @return
031         */
032        boolean isLoading(EStructuralFeature feature);
033        
034        /**
035         * Returns loaded value which was provided to feature setter for changeable features.
036         * This value can be used to derive (compute) feature value for non-changeable derived features which have computed annotation set to true.
037         * @param feature
038         * @return
039         */
040        Object get(EStructuralFeature feature);
041
042}
