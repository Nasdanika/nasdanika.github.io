001package org.nasdanika.common.persistence;
002
003import org.nasdanika.common.Context;
004import org.nasdanika.common.Function;
005import org.nasdanika.common.ProgressMonitor;
006import org.nasdanika.common.SupplierFactory;
007
008public class EnumSupplierFactoryAttribute<T extends Enum<T>> extends AbstractFeatureDelegate<SupplierFactoryFeature<String>> implements SupplierFactoryFeature<T> {
009
010        private Class<T> type;
011        private T defaultValue;
012
013        public EnumSupplierFactoryAttribute(SupplierFactoryFeature<String> delegate, Class<T> type, T defaultValue) {
014                super(delegate);
015                this.type = type;
016                this.defaultValue = defaultValue;
017        }
018
019        @Override
020        public SupplierFactory<T> getValue() {
021                return delegate.getValue().then(this::createFunction);
022        }
023        
024        private Function<String,T> createFunction(Context context) {
025                return new Function<String, T>() {
026
027                        @Override
028                        public double size() {
029                                return 0;
030                        }
031
032                        @Override
033                        public String name() {
034                                return "String to " + type.getName();
035                        }
036
037                        @Override
038                        public T execute(String name, ProgressMonitor progressMonitor) throws Exception {
039                                if (isLoaded()) {
040                                        try {
041                                                return Enum.valueOf(type, name);
042                                        } catch (IllegalArgumentException e) {
043                                                throw new ConfigurationException(e.getMessage(), e, getMarker());
044                                        }
045                                }
046                                return defaultValue;
047                        }
048                        
049                };              
050        }
051
052}
