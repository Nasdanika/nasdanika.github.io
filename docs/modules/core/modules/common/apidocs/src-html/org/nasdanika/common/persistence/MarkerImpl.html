001package org.nasdanika.common.persistence;
002
003import org.yaml.snakeyaml.error.Mark;
004
005/**
006 * Implementation of {@link Marker} with equals() and hashCode() 
007 * @author Pavel Vlasov
008 *
009 */
010public class MarkerImpl implements Marker {
011
012        private int column;
013        private int line;
014        private String location;
015
016        public MarkerImpl(String location, int line, int column) {
017                this.location = location;
018                this.line = line;
019                this.column = column;
020        }
021        
022        public MarkerImpl(String location, Mark mark) {
023                this(location, mark == null ? 0 : mark.getLine() + 1, mark == null ? 0 : mark.getColumn() + 1);
024        }
025
026        @Override
027        public int getLine() {
028                return line;
029        }
030
031        @Override
032        public int getColumn() {
033                return column;
034        }
035        
036        @Override
037        public boolean equals(Object obj) {
038                return obj instanceof Marker && ((Marker) obj).getLine() == getLine() && ((Marker) obj).getColumn() == getColumn() && equal(((Marker) obj).getLocation(), location) ;
039        }
040        
041        private boolean equal(String a, String b) {
042                return a == null ? b == null : a.equals(b);
043        }
044
045        @Override
046        public int hashCode() {
047                return getLine() * 31 + getColumn();
048        }
049        
050        @Override
051        public String toString() {
052                return Marker.toString(this);
053        }
054
055        @Override
056        public String getLocation() {
057                return this.location;
058        }
059                        
060}
