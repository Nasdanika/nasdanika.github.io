001package org.nasdanika.common.persistence;
002
003import java.io.InputStream;
004
005import org.nasdanika.common.SupplierFactory;
006import org.nasdanika.common.Util;
007
008public class StringSupplierFactoryAttribute extends AbstractFeatureDelegate<SupplierFactoryFeature<InputStream>> implements SupplierFactoryFeature<String> {
009
010        private boolean interpolate;
011
012        public StringSupplierFactoryAttribute(SupplierFactoryFeature<InputStream> delegate, boolean interpolate) {
013                super(delegate);
014                this.interpolate = interpolate;
015        }
016
017        public StringSupplierFactoryAttribute(Feature<?> delegate, boolean interpolate) {
018                this(new InputStreamSupplierFactoryAttribute(delegate), interpolate);
019        }
020
021        @Override
022        public SupplierFactory<String> getValue() {
023                if (!delegate.isLoaded()) {
024                        return SupplierFactory.from(null, "Null string from not loaded stream " + getKey()); 
025                }
026                SupplierFactory<String> ret = delegate.getValue().then(Util.TO_STRING);
027                return interpolate ? ret.then(Util.INTERPOLATE_TO_STRING) : ret;
028        }
029
030}
