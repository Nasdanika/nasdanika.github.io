001package org.nasdanika.common.persistence;
002
003import org.nasdanika.common.FunctionFactory;
004import org.nasdanika.common.SupplierFactory;
005
006/**
007 * Applies a function to the delegate value to produce this feature value.
008 * @author Pavel
009 *
010 * @param <T>
011 */
012public class FunctionSupplierFactoryAttribute<T,R> extends AbstractFeatureDelegate<SupplierFactoryFeature<T>> implements SupplierFactoryFeature<R> {
013
014        private FunctionFactory<T, R> functionFactory;
015
016        public FunctionSupplierFactoryAttribute(SupplierFactoryFeature<T> delegate, FunctionFactory<T,R> functionFactory) {
017                super(delegate);
018                this.functionFactory = functionFactory;
019        }
020
021        @Override
022        public SupplierFactory<R> getValue() {
023                return delegate.getValue().then(functionFactory);
024        }
025
026}
