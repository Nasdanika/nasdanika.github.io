001package org.nasdanika.common.persistence;
002
003import java.util.List;
004
005import org.eclipse.emf.common.util.URI;
006import org.nasdanika.common.ProgressMonitor;
007
008public class EnumAttribute<T extends Enum<T>> extends Attribute<T> {
009
010        private Class<T> type;
011
012        public EnumAttribute(
013                        Object key, 
014                        Class<T> type, 
015                        boolean isDefault,
016                        boolean required, 
017                        T defaultValue, 
018                        String description, 
019                        Object... exclusiveWith) {
020                super(key, isDefault, required, defaultValue, description, exclusiveWith);
021                this.type = type;
022        }
023        
024        @Override
025        public T create(ObjectLoader loader, Object config, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) throws Exception {
026                try {
027                        return Enum.valueOf(type, (String) config);
028                } catch (IllegalArgumentException e) {
029                        throw new ConfigurationException(e.getMessage(),  e, markers);
030                }
031        }
032
033}
