001package org.nasdanika.common.persistence;
002
003import java.net.URL;
004
005import org.nasdanika.common.ProgressMonitor;
006
007public class EnumAttribute<T extends Enum<T>> extends Attribute<T> {
008
009        private Class<T> type;
010
011        public EnumAttribute(
012                        Object key, 
013                        Class<T> type, 
014                        boolean isDefault,
015                        boolean required, 
016                        T defaultValue, 
017                        String description, 
018                        Object... exclusiveWith) {
019                super(key, isDefault, required, defaultValue, description, exclusiveWith);
020                this.type = type;
021        }
022        
023        @Override
024        public T create(ObjectLoader loader, Object config, URL base, ProgressMonitor progressMonitor, Marker marker) throws Exception {
025                try {
026                        return Enum.valueOf(type, (String) config);
027                } catch (IllegalArgumentException e) {
028                        throw new ConfigurationException(e.getMessage(),  e, marker);
029                }
030        }
031
032}
