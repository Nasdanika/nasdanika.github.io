001package org.nasdanika.common.persistence;
002
003import java.util.List;
004import java.util.Map;
005
006import org.eclipse.emf.common.util.URI;
007import org.nasdanika.common.ProgressMonitor;
008import org.nasdanika.common.Util;
009
010public class Attribute<T> implements Feature<T>, ObjectFactory<T> {
011        
012        private List<? extends Marker> markers;
013        private T value;
014        private boolean loaded;
015        private Object key;
016        private boolean required;
017        private Object[] exclusiveWith;
018        private boolean isDefault;
019        private String description;
020        
021        public Attribute(
022                        Object key, 
023                        boolean isDefault, 
024                        boolean required, 
025                        T defaultValue, 
026                        String description, 
027                        Object... exclusiveWith) {
028                this.key = key;
029                this.isDefault = isDefault;
030                this.required = required;
031                this.value = defaultValue;
032                this.description = description;
033                this.exclusiveWith = exclusiveWith;             
034        }
035                        
036        public Object getKey() {
037                return key;
038        }
039        
040        public T getValue() {
041                return value; 
042        }
043        
044        public boolean isLoaded() {
045                return loaded;
046        }
047        
048        @Override
049        public List<? extends Marker> getMarkers() {
050                return markers;
051        }
052
053        @Override
054        public void load(ObjectLoader loader, Map<?,?> source, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) throws Exception {
055                if (source.containsKey(getKey())) {
056                        for (Object ek: exclusiveWith) {
057                                if (source.containsKey(ek)) {
058                                        throw new ConfigurationException("Features '" + getKey() + " and " + ek + " are mutually exclusive", markers);                                  
059                                }
060                        }
061                        this.markers = Util.getMarkers(source, getKey());
062                        value = create(loader, source.get(getKey()), base, progressMonitor, this.markers);                                              
063                        loaded = true;
064                } else if (isRequired()) {
065                        throw new ConfigurationException("Missing required feature: " + getKey(), markers);
066                }               
067        }       
068        
069        /**
070         * This implementation casts config to the target type. It can be used where not special conversion is required, e.g. config is a String and feature type is String.
071         * @param loader
072         * @param config
073         * @param base
074         * @param progressMonitor
075         * @param marker
076         * @return
077         * @throws Exception
078         */
079        @SuppressWarnings("unchecked")
080        @Override
081        public T create(ObjectLoader loader, Object config, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) throws Exception {
082                return (T) config;
083        }
084
085        public Object[] getExclusiveWith() {
086                return exclusiveWith;
087        }
088        
089        public boolean isRequired() {
090                return required;
091        }
092        
093        @Override
094        public boolean isDefault() {
095                return isDefault;
096        }
097
098        @Override
099        public String getDescription() {
100                return description;
101        }
102}
