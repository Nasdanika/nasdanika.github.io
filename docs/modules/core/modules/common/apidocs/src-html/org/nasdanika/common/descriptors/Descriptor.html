001package org.nasdanika.common.descriptors;
002
003import org.nasdanika.common.Diagnosable;
004
005/**
006 * Describes something.
007 * @author Pavel
008 *
009 */
010public interface Descriptor extends Diagnosable {
011        
012        String getIcon();
013        
014        String getLabel();
015        
016        String getDescription();
017        
018        /**
019         * @return true if this descriptor can be used to collect information and false otherwise. A descriptor can be disabled if it depends on other descriptors. 
020         * E.g. configuration descriptors for component A can be disabled if the component A is not selected.
021         */
022        boolean isEnabled();
023
024}
