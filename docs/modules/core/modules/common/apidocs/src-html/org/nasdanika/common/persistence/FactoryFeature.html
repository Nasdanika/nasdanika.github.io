001package org.nasdanika.common.persistence;
002
003import java.net.URL;
004
005import org.nasdanika.common.ProgressMonitor;
006
007/**
008 * Delegates feature creation to a factory, e.g. a constructor.
009 * @author Pavel
010 *
011 */
012public class FactoryFeature<T> extends Attribute<T> {
013
014        private ObjectFactory<T> factory;
015
016        public FactoryFeature(
017                        Object key,
018                        boolean isDefault,
019                        boolean required, 
020                        T defaultValue,
021                        String description,
022                        ObjectFactory<T> factory, 
023                        Object... exclusiveWith) {
024                
025                super(key, isDefault, required, defaultValue, description, exclusiveWith);
026                this.factory = factory;
027        }
028        
029        @Override
030        public T create(ObjectLoader loader, Object config, URL base, ProgressMonitor progressMonitor, Marker marker)   throws Exception {
031                return factory.create(loader, config, base, progressMonitor, marker);
032        }
033
034}
