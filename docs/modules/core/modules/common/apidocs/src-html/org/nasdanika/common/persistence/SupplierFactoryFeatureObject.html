001package org.nasdanika.common.persistence;
002
003import java.util.Map;
004
005import org.nasdanika.common.Context;
006import org.nasdanika.common.Function;
007import org.nasdanika.common.MapCompoundSupplierFactory;
008import org.nasdanika.common.Supplier;
009import org.nasdanika.common.SupplierFactory;
010
011/**
012 * Feature object implementing {@link SupplierFactory}. The factory result is constructed from a {@link Map} of feature
013 * keys to feature values. If feature values are SupplierFactories they get executed to produce actual values.
014 * @author Pavel
015 *
016 * @param <T>
017 */
018public abstract class SupplierFactoryFeatureObject<T> extends FeatureObject implements SupplierFactory<T> {
019
020        @Override
021        public Supplier<T> create(Context context) throws Exception {
022                // TODO - handling of prototype here - prototype supplier factory, bi-supplier
023                MapCompoundSupplierFactory<Object, Object> featureMapFactory = new MapCompoundSupplierFactory<>("Feature map factory");
024                for (Feature<?> feature: features) {                    
025                        @SuppressWarnings("unchecked")
026                        SupplierFactory<Object> fsf = feature instanceof SupplierFactoryFeature ? ((SupplierFactoryFeature<Object>) feature).getValue() : SupplierFactory.from(feature.getValue(), "Feature " + feature.getKey());
027                        featureMapFactory.put(feature.getKey(), fsf);
028                }
029                
030                return featureMapFactory.then(this::createResultFunction).create(context);
031        }
032
033        /**
034         * @param context
035         * @return {@link Function} which would create a result object from feature values.
036         */
037        protected abstract Function<Map<Object, Object>, T> createResultFunction(Context context);
038
039}
