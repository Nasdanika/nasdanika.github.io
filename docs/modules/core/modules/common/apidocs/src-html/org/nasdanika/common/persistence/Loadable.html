001package org.nasdanika.common.persistence;
002
003import org.eclipse.emf.common.util.URI;
004import org.nasdanika.common.ProgressMonitor;
005
006/**
007 * Objects which know how to load themselves from configuration may implement this interface.
008 * Also this interface may be used as an adapter interface. In this case an object to be loaded is first created by the object loader based on 
009 * type passed to {@link ObjectLoader}'s create() method and then is adapted to this interface to be loaded.
010 * @author Pavel
011 *
012 */
013public interface Loadable {
014        
015        void load(ObjectLoader loader, Object config, URI base, ProgressMonitor progressMonitor, Marker marker) throws Exception;       
016
017}
