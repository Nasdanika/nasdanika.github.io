001package org.nasdanika.common.persistence;
002
003import java.util.ArrayList;
004import java.util.Collections;
005import java.util.List;
006
007/**
008 * List with markers for elements.
009 * @author Pavel Vlasov
010 *
011 * @param <E>
012 */
013@SuppressWarnings("serial")
014public class MarkedArrayList<E> extends ArrayList<E> implements Marked, Markable {
015        
016        private List<List<? extends Marker>> elementMarkers = new ArrayList<>();
017        private List<? extends Marker> markers;
018        
019        public List<List<? extends Marker>> getElementMarkers() {
020                return elementMarkers;
021        }
022        
023        @Override
024        public List<? extends Marker> getMarkers() {
025                return Collections.unmodifiableList(markers);
026        }
027
028        @Override
029        public void mark(List<? extends Marker> markers) {
030                this.markers = markers;
031        }
032
033}
