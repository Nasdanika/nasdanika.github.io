001package org.nasdanika.common;
002
003import java.util.List;
004import java.util.Map;
005import java.util.function.Function;
006import java.util.stream.Collectors;
007
008/**
009 * This class converts value returned by the source as follows:
010 * * Lists are streamed to lists with converting values as explained below.
011 * * {@link Map}s are wrapped into {@link Context}s using Context.wrap receiving a new instance of {@link InterpolatingSource} constructed from the Map's get.
012 * * {@link String}s are wrapped into interpolating {@link PropertyComputer}s.
013 * @author Pavel
014 *
015 */
016public class InterpolatingSource implements Function<String,Object> {
017
018        private Function<String, Object> source;
019
020        public InterpolatingSource(Function<String,Object> source) {
021                this.source = source;
022        }
023
024        @Override
025        public Object apply(String key) {
026                return convert(source.apply(key));
027        }
028        
029        /**
030         * Converts value. This implementation converts:
031         * 
032         * * {@link Map} to {@link Context}.
033         * * {@link List} to a list of converted values.
034         * * {@link String} to an interpolating {@link PropertyComputer}. 
035         * 
036         * @param value
037         * @return
038         */
039        @SuppressWarnings("unchecked")
040        protected Object convert(Object value) {
041                if (value instanceof Map) {
042                        return Context.wrap(new InterpolatingSource(((Map<String,Object>) value)::get));
043                }
044                if (value instanceof List) {
045                        return ((List<Object>) value).stream().map(this::convert).collect(Collectors.toList());
046                }
047                if (value instanceof String) {
048                        return new PropertyComputer() {
049                                
050                                @Override
051                                public <T> T compute(Context context, String key, String path, Class<T> type) {
052                                        Converter converter = context.get(Converter.class);
053                                        Object interpolated = context.interpolate((String) value);
054                                        return converter == null ? (T) interpolated : converter.convert(interpolated, type);
055                                }
056                                
057                        };
058                }
059                return value;
060        }
061
062}
