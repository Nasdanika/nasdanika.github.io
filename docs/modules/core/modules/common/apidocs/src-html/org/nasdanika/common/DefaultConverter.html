063        // Reader to String, and JSONObject and JSONArray.
064        
065        @ConverterMethod
066        public String toString(Reader reader) throws IOException {
067                StringWriter sw = new StringWriter();
068                try (BufferedReader br = new BufferedReader(reader)) {
069                        int ch;
070                        while ((ch = br.read()) != -1) {
071                                sw.write(ch);
072                        }
073                }
074                sw.close();
075                return sw.toString();
076        }

090        /**
091         * Reads input stream into a byte array. 
092         * @param in
093         * @return
094         * @throws IOException 
095         */
096        @ConverterMethod
097        public byte[] toByteArray(InputStream in) throws IOException {
098                ByteArrayOutputStream baos = new ByteArrayOutputStream();
099                try (InputStream bin = new BufferedInputStream(in)) {
100                        int b;
101                        while ((b = bin.read()) != -1) {
102                                baos.write(b);
103                        }
104                }
105                baos.close();
106                return baos.toByteArray();                              
107        }

199        /**
200         * Converts character sequence to byte array with UTF_8 encoding.
201         * @param charSequence
202         * @return
203         * @throws IOException
204         */
205        @ConverterMethod
206        public byte[] toByteArray(CharSequence charSequence) throws IOException {
207                ByteArrayOutputStream baos = new ByteArrayOutputStream();
208                try (Writer writer = new OutputStreamWriter(baos, StandardCharsets.UTF_8)) {
209                        writer.write(charSequence.toString());
210                }
211                baos.close();
212                return baos.toByteArray();                              
213        }

299        /**
300         * @param text
301         * @return parsed duration.
302         */
303        @ConverterMethod
304        public Instant toInstant(String text) {
305                if (text == null) {
306                        return null;
307                }
308                if (text.contains("Z")) {
309                        return Instant.parse(text);
310                }
311                // Parsing via Date for convenience
312                Date date = convert(text, Date.class);
313                return date == null ? null : date.toInstant();
314        }
