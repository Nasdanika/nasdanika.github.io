001package org.nasdanika.common.resources;
002
003import java.util.Map;
004
005import org.nasdanika.common.ProgressMonitor;
006
007public class ContainerFilter<E> implements Container<E> {
008
009        protected Container<E> target;
010        
011        public ContainerFilter(Container<E> target) {
012                this.target = target;
013        }
014
015        public Object find(String path, ProgressMonitor monitor) {
016                return target.find(path, monitor);
017        }
018
019        public E get(String path, ProgressMonitor monitor) {
020                return target.get(path, monitor);
021        }
022
023        public void put(String path, E element, ProgressMonitor monitor) throws IllegalArgumentException {
024                target.put(path, element, monitor);
025        }
026
027        public Object delete(String path, ProgressMonitor monitor) {
028                return target.delete(path, monitor);
029        }
030
031        public Container<E> getContainer(String path, ProgressMonitor monitor) {
032                return target.getContainer(path, monitor);
033        }
034
035        public Map<String, Object> getChildren(ProgressMonitor monitor) {
036                return target.getChildren(monitor);
037        }
038
039        public Container<E> getParent() {
040                return target.getParent();
041        }
042
043        public String getName() {
044                return target.getName();
045        }
046
047        @Override
048        public void copy(Container<? super E> container, String path, ProgressMonitor monitor) {
049                target.copy(container, path, monitor);          
050        }
051
052        @Override
053        public void move(Container<? super E> container, String path, ProgressMonitor monitor) {
054                target.move(container, path, monitor);          
055        }
056        
057}
