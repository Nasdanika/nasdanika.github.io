001package org.nasdanika.common;
002
003/**
004 * Something which can diagnose/validate itself.
005 * @author Pavel
006 *
007 */
008public interface Diagnosable {
009        
010        /**
011         * Performs self-diagnostic.
012         * @return Diagnostic result.
013         */
014        default Diagnostic diagnose(ProgressMonitor progressMonitor) {
015                return new BasicDiagnostic(Status.SUCCESS, "Diagnostic of " + this, this);
016        };
017
018}
