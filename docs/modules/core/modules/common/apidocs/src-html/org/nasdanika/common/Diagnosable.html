001package org.nasdanika.common;
002
003import java.util.Collections;
004import java.util.List;
005
006/**
007 * Something which can diagnose/validate itself.
008 * @author Pavel
009 *
010 */
011public interface Diagnosable {
012        
013        Diagnostic SUCCESS = new Diagnostic() {
014                
015                @Override
016                public Status getStatus() {
017                        return Status.SUCCESS;
018                }
019                
020                @Override
021                public String getMessage() {
022                        return null;
023                }
024                
025                @Override
026                public List<Object> getData() {
027                        return null;
028                }
029                
030                @Override
031                public List<Diagnostic> getChildren() {
032                        return Collections.emptyList();
033                }
034        }; 
035        
036        /**
037         * Performs self-diagnostic.
038         * @return Diagnostic result.
039         */
040        default Diagnostic diagnose(ProgressMonitor progressMonitor) {
041                return SUCCESS;
042        };
043
044}
