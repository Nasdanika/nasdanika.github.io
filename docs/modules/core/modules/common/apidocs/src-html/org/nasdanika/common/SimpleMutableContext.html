001package org.nasdanika.common;
002
003import java.lang.reflect.Array;
004import java.util.ArrayList;
005import java.util.HashMap;
006import java.util.List;
007import java.util.Map;
008import java.util.function.Function;
009import java.util.function.Predicate;
010import java.util.stream.Collectors;
011
012/**
013 * A simple context context implementation.
014 * This implementation composes {@link Composeable} services into a single service.
015 * 
016 * Also it implements sub-context lookup if property key has one or more slashes (/).
017 * For example, for a property key ``my/important/property`` it will work like explained below:
018 * 
019 * * ``my/important`` property will be looked up. If it is found and is of type {@link Context} then ``property`` will be looked up in the property value and returned if it is not null.
020 * * Otherwise ``my`` property will be looked up. If it is found and is of type Context then ``important/property`` will be looked up in the property value and returned if it is not null.
021 * * And finally ``my/important/property`` key will be looked up in this context.
022 * @author Pavel
023 *
024 */
025public class SimpleMutableContext implements MutableContext {
026        
027        private Map<String, Object> properties = new HashMap<>();
028        
029        private Map<Class<Object>, List<Object>> services = new HashMap<>();
030
031        private Context chain;
032        
033        public SimpleMutableContext() {
034                this(Context.EMPTY_CONTEXT);
035        }
036        
037        /**
038         * Constructs a new mutable context which falls-back to the chain {@link Context}'s.
039         * @param chain
040         */
041        public SimpleMutableContext(Context chain) {
042                this.chain = chain == null ? Context.EMPTY_CONTEXT : chain;
043        }
044        
045        @SuppressWarnings("unchecked")
046        protected <T> List<T> getServices(Class<T> type) {
047                return (List<T>) services.computeIfAbsent((Class<Object>) type, (serviceType) -> new ArrayList<Object>());
048        }
049
050        /**
051         * Returns property by key. If a property at a given key is not found, the property key is split by the last slash (``/``)
052         * and the tail is looked up in the property at the head if the property is {@link Context}, {@link Map}, or {@link List} and the tail can be parsed to an integer.
053         * For example, if ``my/property`` is not present in properties then ``my`` property will be retrieved and if it is of type Context or Map its ``property`` key will be retrieved.
054         * Similarly, ``my/property/list/0`` will retrieve ``my/property/list`` and if it is a list then it will return its first element. 
055         */
056        @SuppressWarnings("unchecked")
057        @Override
058        public Object get(String key) {         
059                Object ret = properties.get(key);
060                if (ret == null) {
061                        ret = chain.get(key);
062                        if (ret != null) {
063                                return ret;
064                        }
065                        
066                        int lastSlash = key.lastIndexOf('/');
067                        if (lastSlash == -1) {
068                                return null;
069                        }
070                        
071                        Object parentProperty = get(key.substring(0, lastSlash));
072                        String subKey = key.substring(lastSlash + 1);
073                        if (parentProperty instanceof Context) {
074                                return ((Context) parentProperty).get(subKey);  
075                        }
076                        if (parentProperty instanceof Map) {
077                                return ((Map<?,?>) parentProperty).get(subKey);
078                        }
079                        if (parentProperty instanceof List) {
080                                try {
081                                        int idx = Integer.parseInt(subKey);
082                                        if (idx >= 0 && idx < ((List<?>) parentProperty).size()) {
083                                                return ((List<?>) parentProperty).get(idx);
084                                        }
085                                } catch (NumberFormatException e) {
086                                        // NOP
087                                }
088                        }
089                        if (parentProperty instanceof Function) {
090                                return ((Function<String, Object>) parentProperty).apply(subKey);
091                        }
092                        
093                        return null;
094                        
095                }
096                
097                return ret;
098        }
099
100        /**
101         * Returns registered services of the type's component type if the type is array, or the first registered service.
102         */
103        @SuppressWarnings({ "unchecked" })
104        @Override
105        public <T> T get(Class<T> type) {
106                if (type.isArray()) {
107                        List<Object> svcs = new ArrayList<>((List<Object>) getServices(type.getComponentType()));
108                        Object chained = chain.get(type); 
109                        if (chained!=null) {
110                                for (int i = 0; i < Array.getLength(chained); ++i) {
111                                        svcs.add(Array.get(chained, i));
112                                }
113                        }
114                        return (T) svcs.toArray((T[]) Array.newInstance(type.getComponentType(), svcs.size()));
115                }
116                
117                if (Composeable.class.isAssignableFrom(type)) {
118                        List<T> svcs = new ArrayList<>(getServices(type));
119                        T chained = chain.get(type); 
120                        if (chained!=null) {
121                                svcs.add(chained);
122                        }
123                        return svcs.stream().reduce(null, Composeable.<T>composer());
124                }
125                
126                List<T> svcs = getServices(type);
127                if (svcs.isEmpty()) {
128                        if (type.isInstance(this)) {
129                                return (T) this;
130                        }
131                        return chain.get(type);
132                }
133                
134                return svcs.get(0);
135        }
136        
137        /**
138         * Returns registered services of the type's component type matching the predicate if the type is array, or the first registered service matching the predicate otherwise.
139         */
140        @SuppressWarnings({ "unchecked" })
141        @Override
142        public <T> T get(Class<T> type, Predicate<? super T> predicate) {
143                if (type.isArray()) {
144                        List<Object> svcs = new ArrayList<Object>((List<Object>) getServices(type.getComponentType()).stream().filter((Predicate<Object>) predicate).collect(Collectors.toList()));
145                        Object chained = chain.get(type, predicate); 
146                        if (chained!=null) {
147                                for (int i = 0; i < Array.getLength(chained); ++i) {
148                                        svcs.add(Array.get(chained, i));
149                                }
150                        }
151                        return (T) svcs.toArray((T[]) Array.newInstance(type.getComponentType(), svcs.size()));
152                }
153                                
154                if (Composeable.class.isAssignableFrom(type)) {
155                        List<T> svcs = new ArrayList<>(getServices(type).stream().filter(predicate).collect(Collectors.toList()));
156                        T chained = chain.get(type, predicate); 
157                        if (chained!=null) {
158                                svcs.add(chained);
159                        }
160                        return svcs.stream().reduce(null, Composeable.<T>composer());
161                }
162                
163                return getServices(type).stream().filter(predicate).findFirst().orElse(chain.get(type, predicate));
164        }
165        
166        /**
167         * Sets property value. If the value is an instance of {@link Context} then it will be used in hierarchical lookups, i.e. it
168         * has the semantics of mounting the context at a particular path. 
169         * It is different from ``mount()`` because mount returns a new context and the original context does not change.
170         * @param key
171         * @param value
172         */
173        public void put(String key, Object value) {
174                properties.put(key, value);
175        }
176        
177        /**
178         * Registers a service. Multiple registrations are allowed.
179         * @param <T>
180         * @param type
181         * @param service
182         */
183        public <T> void register(Class<T> type, T service) {
184                getServices(type).add(service);
185        }       
186        
187        /**
188         * Unregisters a service.
189         * @param <T>
190         * @param type
191         * @param service
192         */
193        public <T> void unregister(Class<T> type, T service) {
194                getServices(type).remove(service);
195        }
196
197        @SuppressWarnings("unchecked")
198        @Override
199        public <T> void register(Class<T> type, ServiceComputer<? super T> serviceComputer) {
200                services.computeIfAbsent((Class<Object>) type, (serviceType) -> new ArrayList<Object>()).add(serviceComputer);
201        }
202
203        @Override
204        public <T> void unregister(Class<T> type, ServiceComputer<? super T> serviceComputer) {
205                getServices(type).remove(serviceComputer);
206        }
207        
208}
