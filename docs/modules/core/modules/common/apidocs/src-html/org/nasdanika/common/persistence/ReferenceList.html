001package org.nasdanika.common.persistence;
002
003import java.util.List;
004
005import org.eclipse.emf.common.util.URI;
006import org.nasdanika.common.ProgressMonitor;
007
008/**
009 * Loads a list of references using loader.
010 * @author Pavel
011 *
012 */
013public class ReferenceList<T> extends ListAttribute<T> {
014
015        public ReferenceList(Object key, 
016                        boolean isDefault, 
017                        boolean required, 
018                        List<T> defaultValue, 
019                        String description, 
020                        Object... exclusiveWith) {
021                super(key, isDefault, required, defaultValue, description, exclusiveWith);                              
022        }
023        
024        @SuppressWarnings("unchecked")
025        @Override
026        protected T createElement(ObjectLoader loader, Object element, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) throws Exception {
027                return (T) loader.load(element, base, progressMonitor);
028        }
029
030}
