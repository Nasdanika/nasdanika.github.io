001package org.nasdanika.common;
002
003import java.io.ByteArrayOutputStream;
004import java.io.IOException;
005import java.util.Base64;
006import java.util.UUID;
007
008import org.apache.commons.text.StringEscapeUtils;
009import org.json.JSONObject;
010
011import net.sourceforge.plantuml.FileFormat;
012import net.sourceforge.plantuml.FileFormatOption;
013import net.sourceforge.plantuml.SourceStringReader;
014
015public class DiagramGeneratorImpl implements DiagramGenerator {
016        
017        @Override
018        public String generateDiagram(String spec, Dialect dialect) throws IOException {
019                if (dialect == Dialect.DRAWIO) {
020                        JSONObject data = new JSONObject();
021                        data.put("highlight", "#0000ff");
022                        data.put("nav", true);
023                        data.put("resize",true);
024                        data.put("toolbar", "zoom layers tags lightbox");
025                        data.put("edit","_blank");
026                        
027                        data.put("xml", spec);
028                        
029                        String diagramDiv = "<div class=\"mxgraph\" style=\"max-width:100%;border:1px solid transparent;\" data-mxgraph=\"" + StringEscapeUtils.escapeHtml4(data.toString()) + "\"></div>";
030                        String script = "<script type=\"text/javascript\" src=\"" + getDrawioViewer() + "\"></script>";
031                        return diagramDiv + System.lineSeparator() + script;            
032                }
033                
034                ByteArrayOutputStream baos = new ByteArrayOutputStream();
035                StringBuilder sb = new StringBuilder("@start")
036                                .append(dialect.name().toLowerCase())
037                                .append(System.lineSeparator())
038                                .append(spec)
039                                .append(System.lineSeparator())
040                                .append("@end")
041                                .append(dialect.name().toLowerCase())
042                                .append(System.lineSeparator());
043                
044                SourceStringReader reader = new SourceStringReader(sb.toString());
045                
046                FileFormatOption fileFormatOption = new FileFormatOption(FileFormat.PNG);
047                reader.outputImage(baos, 0, fileFormatOption);          
048                String diagramCMap = reader.getCMapData(0, fileFormatOption);
049                baos.close();
050
051                StringBuilder ret = new StringBuilder("<img src=\"data:image/png;base64, ");
052                ret
053                        .append(Base64.getEncoder().encodeToString(baos.toByteArray()))
054                        .append("\"");
055                
056                if (Util.isBlank(diagramCMap)) {
057                        ret.append("/>");
058                        return ret.toString();                  
059                }
060                
061                String openingTag = "<map id=\"plantuml_map\" name=\"plantuml_map\">";
062                if (diagramCMap.startsWith(openingTag)) {
063                        String mapId = "plantuml_map_" + UUID.randomUUID().toString();
064                        ret                     
065                        .append(" usemap=\"#")
066                        .append(mapId)
067                        .append("\"/>")
068                        .append(System.lineSeparator())
069                        .append("<map id=\"")
070                        .append(mapId)
071                        .append("\" name=\"")
072                        .append(mapId)
073                        .append("\">")
074                        .append(diagramCMap.substring(openingTag.length()));
075                        
076                } else {                                
077                        ret                     
078                                .append(" usemap=\"#plant_uml_map\"/>")
079                                .append(System.lineSeparator())
080                                .append(diagramCMap);
081                        return ret.toString();
082                }
083                                
084                return ret.toString();
085        }
086
087        protected String getDrawioViewer() {
088                return "https://viewer.diagrams.net/js/viewer-static.min.js";
089        }
090}
