110        /**
111         * @param markdown Markdown to pre-process
112         * @param replacements Accumulator of post-processing replacements token -&gt; HTML to replace the token with.
113         * @return
114         */
115        protected String preProcessMarkdown(String markdown, Map<String,String> replacements) {
116                if (isProcessFendedBlocks()) {
117                        String ret = processFencedBlocks(markdown, DiagramGenerator.Dialect.UML, START_UML_PATTERN, replacements, false);
118                        ret = processFencedBlocks(ret, DiagramGenerator.Dialect.SALT, START_WIREFRAME_PATTERN, replacements, false);
119                        ret = processFencedBlocks(ret, DiagramGenerator.Dialect.GANTT, START_GANTT_PATTERN, replacements, false);
120                        ret = processFencedBlocks(ret, DiagramGenerator.Dialect.MINDMAP, START_MINDMAP_PATTERN, replacements, false);
121                        ret = processFencedBlocks(ret, DiagramGenerator.Dialect.WBS, START_WBS_PATTERN, replacements, false);
122                        ret = processFencedBlocks(ret, DiagramGenerator.Dialect.DRAWIO, START_DRAWIO_PATTERN, replacements, false);
123                        ret = processFencedBlocks(ret, null, START_PNG_PATTERN, replacements, false);
124                        ret = processFencedBlocks(ret, null, START_JPEG_PATTERN, replacements, false);
125                                        
126                        ret = processFencedBlocks(ret, DiagramGenerator.Dialect.UML, START_UML_RESOURCE_PATTERN, replacements, true);
127                        ret = processFencedBlocks(ret, DiagramGenerator.Dialect.SALT, START_WIREFRAME_RESOURCE_PATTERN, replacements, true);
128                        ret = processFencedBlocks(ret, DiagramGenerator.Dialect.GANTT, START_GANTT_RESOURCE_PATTERN, replacements, true);
129                        ret = processFencedBlocks(ret, DiagramGenerator.Dialect.MINDMAP, START_MINDMAP_RESOURCE_PATTERN, replacements, true);
130                        ret = processFencedBlocks(ret, DiagramGenerator.Dialect.WBS, START_WBS_RESOURCE_PATTERN, replacements, true);
131                        ret = processFencedBlocks(ret, DiagramGenerator.Dialect.DRAWIO, START_DRAWIO_RESOURCE_PATTERN, replacements, true);
132                        ret = processFencedBlocks(ret, null, START_PNG_RESOURCE_PATTERN, replacements, true);
133                        ret = processFencedBlocks(ret, null, START_JPEG_RESOURCE_PATTERN, replacements, true);
134                                        
135                        return ret;             
136                }
137                                
138                return markdown;
139        }

141        /**
142         * 
143         * @param input
144         * @param dialect
145         * @param startPattern
146         * @param replacements
147         * @param resource Indicates that spec is a URL of a resource resolved relative to the URL returned by getResourceBase(). The URL supports <code>classpath</code> schema for classloader resources.
148         * @return
149         */
150        private String processFencedBlocks(String input, DiagramGenerator.Dialect dialect, Pattern startPattern, Map<String,String> replacements, boolean resource) {
151                StringBuilder output = new StringBuilder();
152                Matcher startMatcher = startPattern.matcher(input);
153                int i = 0;
154                while (startMatcher.find()) {
155                        int startMatcherEnd = startMatcher.end();
156                        String endMatcherContent = input.substring(startMatcherEnd);
157                        Matcher endMatcher = FENCED_BLOCK_PATTERN.matcher(endMatcherContent);                   
158                        if (endMatcher.find()) {
159                            output.append(input.substring(i, startMatcher.start()));
160                                String bareSpec = endMatcherContent.substring(0, endMatcher.start());
161                                String match = endMatcherContent.substring(endMatcher.start(), endMatcher.end());
162                            i = startMatcherEnd + endMatcher.start() + match.indexOf("```") + 3; // Just the closing back-ticks, no space or new line characters.
163                                
164                                if (resource) {                                 
165                                        try {           
166                                                URI resourceURI = URI.createURI(bareSpec);
167                                                URI resourceBase = getResourceBase();
168                                                if (resourceBase != null && resourceBase.hasAbsolutePath()) {
169                                                        resourceURI = resourceURI.resolve(resourceBase);
170                                                }
171                                                bareSpec = loadResource(resourceURI, dialect == null);
172                                        } catch (Exception e) {                                 
173                                                output
174                                                        .append(System.lineSeparator())
175                                                        .append("```").append(System.lineSeparator())
176                                                        .append("Error loading resource " + bareSpec + ": " + e) 
177                                                        .append(System.lineSeparator()).append("```")
178                                                        .append(System.lineSeparator());
179                                                bareSpec = null;
180                                        }
181                                }
182                                
183                                if (bareSpec != null) {
184                                        String escapedBareSpec = escapeDiagramSpec(dialect, bareSpec);
185                                        
186                                        StringBuilder replacementBuilder = new StringBuilder();
187                                        if (!Util.isBlank(escapedBareSpec)) {
188                                                replacementBuilder
189                                                        .append(System.lineSeparator())
190                                                        .append("<div> ").append(System.lineSeparator())
191                                                        .append("    <div style='display:none;white-space:pre-wrap' title='Diagram escaped spec for search'>").append(System.lineSeparator())
192                                                        .append(escapedBareSpec).append(System.lineSeparator()) 
193                                                        .append("    </div>").append(System.lineSeparator())
194                                                        .append("</div> ").append(System.lineSeparator());
195                                        }
196                                        
197                                        try {
198                                                String token = nextToken();
199                                                
200                                                if (dialect == null) {
201                                                        replacementBuilder.append("<img src=\"data:image/");
202                                                        if (startPattern == START_PNG_PATTERN || startPattern == START_PNG_RESOURCE_PATTERN) {
203                                                                replacementBuilder.append("png");
204                                                        } else if (startPattern == START_JPEG_PATTERN || startPattern == START_JPEG_RESOURCE_PATTERN) {
205                                                                replacementBuilder.append("jpeg");
206                                                        }
207                                                        replacementBuilder
208                                                                .append(";base64, ")
209                                                                .append(bareSpec)
210                                                                .append("\"/>")
211                                                                .append(System.lineSeparator());
212                                                } else {
213                                                        replacementBuilder
214                                                                .append(getDiagramGenerator().generateDiagram(bareSpec, dialect))
215                                                                .append(System.lineSeparator());
216                                                }
217                                                
218                                                replacements.put(token, replacementBuilder.toString());
219                                                output
220                                                        .append(System.lineSeparator())
221                                                        .append(System.lineSeparator())
222                                                        .append(token)
223                                                        .append(System.lineSeparator())
224                                                        .append(System.lineSeparator());
225                                        } catch (Exception e) {
226                                                output.append("Error during diagram rendering: " + e);
227                                        }
228                                }                               
229                        }
230                }
231                output.append(input.substring(i, input.length()));
232                return output.toString();
233        }

235        protected String escapeDiagramSpec(DiagramGenerator.Dialect dialect, String bareSpec) {
236                if (dialect == null) {
237                        return null; // For PNG and JPEG
238                }
239                if (dialect == Dialect.DRAWIO) {
240                        // TODO - Extract text from XML, return null for now.
241                        return null;
242                }
243                return StringEscapeUtils.escapeHtml4(bareSpec).trim();
244        }

250        /**
251         * Loads resource and optionally encodes it to Base64.
252         * @param resource
253         * @param encode
254         * @return
255         * @throws Exception
256         */
257        protected String loadResource(URI resource, boolean encode) throws Exception {
258                if (encode) {
259                        byte[] bytes = DefaultConverter.INSTANCE.toByteArray(resource);
260                        return Base64.getEncoder().encodeToString(bytes);
261                }
262                return DefaultConverter.INSTANCE.stringContent(resource);
263        }

295        public String markdownToHtml(String markdown) {
296                if (Util.isBlank(markdown)) {
297                        return "";
298                }
299                Map<String,String> replacements = new HashMap<>();
300                Document document = createMarkdownParser().parse(preProcessMarkdown(markdown, replacements));
301                String html = createMarkdownHtmlRenderer().render(document);
302                for (Entry<String, String> te: replacements.entrySet()) {
303                        html = html.replace(te.getKey(), te.getValue());
304                }
305                return html;
306        }
