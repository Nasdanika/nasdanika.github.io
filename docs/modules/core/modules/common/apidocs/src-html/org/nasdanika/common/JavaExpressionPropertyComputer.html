001package org.nasdanika.common;
002
003import java.lang.reflect.InvocationTargetException;
004
005import org.codehaus.commons.compiler.CompileException;
006import org.codehaus.janino.ExpressionEvaluator;
007
008/**
009 * Evaluates java expressions passed as sub-keys for this property key. E.g. <code>2+2</code> in <code>eval/2+2</code> if the property key is eval.
010 * Uses the {@link ClassLoader} context service defaulting to the classloader passed to the constructor, which in turn defaults to this class classloader.
011 * @author Pavel
012 *
013 */
014public class JavaExpressionPropertyComputer implements PropertyComputer {
015        
016        private ClassLoader classLoader;
017
018
019        public JavaExpressionPropertyComputer() {
020                this(JavaExpressionPropertyComputer.class.getClassLoader());
021        }
022        
023        public JavaExpressionPropertyComputer(ClassLoader classLoader) {
024                this.classLoader = classLoader;
025        }
026
027        @SuppressWarnings("unchecked")
028        @Override
029        public <T> T compute(Context context, String key, String path, Class<T> type) {
030                ExpressionEvaluator expressionEvaluator = new ExpressionEvaluator();
031                expressionEvaluator.setParentClassLoader(context.get(ClassLoader.class, classLoader));
032                expressionEvaluator.setExpressionType(type);
033                expressionEvaluator.setParameters(new String[] { "context" }, new Class[] { Context.class });
034                try {
035                        expressionEvaluator.cook(path);
036                } catch (CompileException e) {
037                        throw new NasdanikaException("Cannot compile java expression "+path, e);
038                }
039                
040                try {
041                        return (T) expressionEvaluator.evaluate(new Object[] { context });
042                } catch (InvocationTargetException e) {
043                        throw new NasdanikaException("Cannot evaluate java expression "+path, e);
044                }
045        }
046
047}
