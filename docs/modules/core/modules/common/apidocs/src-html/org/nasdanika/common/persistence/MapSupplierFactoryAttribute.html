001package org.nasdanika.common.persistence;
002
003import java.util.Map;
004import java.util.Map.Entry;
005
006import org.nasdanika.common.Context;
007import org.nasdanika.common.MapCompoundSupplierFactory;
008import org.nasdanika.common.Supplier;
009import org.nasdanika.common.SupplierFactory;
010
011public class MapSupplierFactoryAttribute<K,V,U> extends AbstractFeatureDelegate<Feature<Map<K,U>>> implements SupplierFactoryFeature<Map<K,V>> {
012
013        private boolean interpolate;
014
015        public MapSupplierFactoryAttribute(Feature<Map<K,U>> delegate, boolean interpolate) {
016                super(delegate);
017                this.interpolate = interpolate;
018        }
019
020        @Override
021        public SupplierFactory<Map<K,V>> getValue() {
022                MapCompoundSupplierFactory<K,V> ret = new MapCompoundSupplierFactory<>("Feature " + getKey());
023                if (delegate.getValue() != null) {
024                        for (Entry<K, U> e: delegate.getValue().entrySet()) {
025                                if (e.getValue() instanceof String && interpolate) {
026                                        ret.put(e.getKey(), new SupplierFactory<V>() {
027
028                                                @Override
029                                                public Supplier<V> create(Context context) throws Exception {
030                                                        Object ie = context.interpolate((String) e.getValue());
031                                                        return SupplierFactory.<V>adapt(ie, "Interpolated feature " + getKey() + "[" + e.getKey() + "]").create(context);                                                       
032                                                }
033                                                
034                                        });                                     
035                                } else {
036                                        ret.put(e.getKey(), SupplierFactory.<V>adapt(e.getValue(), "Feature " + getKey() + "[" + e.getKey() + "]"));
037                                }
038                        }
039                }               
040                
041                return ret;
042        }
043
044}
