001package org.nasdanika.common;
002
003import java.util.ArrayList;
004import java.util.List;
005
006/**
007 * Compound participant where elements execution results are keyed by the element identity
008 * @author Pavel
009 *
010 * @param <E> Element type
011 * @param <R> Execution result type. 
012 */
013public abstract class ElementIdentityMapCompoundExecutionParticipant<E extends ExecutionParticipant, R> extends CompoundExecutionParticipant<E> {
014
015        protected ElementIdentityMapCompoundExecutionParticipant(String name) {
016                super(name, false);
017        }
018
019        protected List<E> elements = new ArrayList<>();
020        
021        private Double size;
022        
023        @Override
024        public double size() {
025                if (size == null) { 
026                        size = super.size();
027                }
028                return size;
029        }
030
031        @Override
032        protected List<E> getElements() {
033                return elements;
034        }
035        
036        public void put(E element) {
037                if (!elements.contains(element)) {
038                        elements.add(element);
039                        size = null;
040                }
041        }
042        
043}
