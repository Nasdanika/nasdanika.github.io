001package org.nasdanika.common;
002
003public enum Status {
004        
005        INFO,
006        SUCCESS,
007        WARNING,
008        
009        /**
010         * Depending on a situation it might be possible to proceed with errors, e.g. to report them.   
011         */
012        ERROR,
013        
014        /**
015         * This status is more severe than error and generally means that no further processing is possible. FAIL vs ERROR is similar to {@link Error} vs {@link Exception}.
016         * A reason to use FAIL instead of throwing an exception is that an exception would report the first problem, while with FAIL multiple reasons for being unable to proceed might be collected. 
017         */
018        FAIL,
019        
020        CANCEL
021
022}
