001package org.nasdanika.common;
002
003/**
004 * Progress monitor is used for reporting progress in 
005 * the try with resources style:
006 * 
007 * ```
008 * try (ProgressMonitor progressMonitor = ...) {
009 *     while (...) {
010 *         ...
011 *         progressMonitor.worked(work, progressMessage);
012 *     }
013 * }
014 * ```
015 * 
016 * @author Pavel
017 *
018 */
019public interface ProgressMonitor extends AutoCloseable, Composeable<ProgressMonitor> {
020                        
021        /**
022         * Closes this monitor.
023         */
024        @Override
025        void close(); 
026        
027        /**
028         * Indicates that this monitor was cancelled and further operations shall not be performed.
029         * @return
030         */
031        boolean isCancelled();
032        
033        /**
034         * Creates a sub-monitor of this monitor which will consume a given number of ticks from the receiver.
035         * @param Task name
036         * @param ticks Amount of work
037         * @param details optional additional information.
038         * @return
039         */
040        ProgressMonitor split(String taskName, double size, Object... data);
041        
042        /**
043         * Reports progress.
044         * @param work Supplier installment - how much work was done since the last report, not the total work.
045         * @param progressMessage Progress message. 
046         */
047        void worked(Status status, double work, String progressMessage, Object... data);
048        
049        /**
050         * Shortcut for ``worked(SUCCESS, work, progressMessage)``.
051         * @param work
052         * @param progressMessage
053         */
054        default void worked(double work, String progressMessage, Object... data) {
055                worked(Status.SUCCESS, work, progressMessage, data);
056        }
057        
058        /**
059         * Resizes the remaining amount of work.
060         * @param size
061         * @return This instance - fluent API.
062         */
063        ProgressMonitor setWorkRemaining(double size);
064
065        /**
066         * Composes two monitors. The resulting monitor reports isCancelled() as OR, composes splits, 
067         * and invokes other methods on both monitors.
068         */
069        @Override
070        default ProgressMonitor compose(ProgressMonitor other) {
071                if (other == null) {
072                        return this;
073                }
074                return new ProgressMonitor() {
075                        
076                        @Override
077                        public void worked(Status status, double work, String progressMessage, Object... data) {
078                                ProgressMonitor.this.worked(status, work, progressMessage, data);
079                                other.worked(status, work, progressMessage, data);
080                        }
081                        
082                        @Override
083                        public ProgressMonitor split(String taskName, double size, Object... data) {
084                                return  ProgressMonitor.this.split(taskName, size, data).compose(other.split(taskName, size, data));
085                        }
086                        
087                        @Override
088                        public boolean isCancelled() {
089                                return ProgressMonitor.this.isCancelled() || other.isCancelled();
090                        }
091                        
092                        @Override
093                        public void close() {
094                                ProgressMonitor.this.close();
095                                other.close();
096                        }
097
098                        @Override
099                        public ProgressMonitor setWorkRemaining(double size) {
100                                ProgressMonitor.this.setWorkRemaining(size);
101                                other.setWorkRemaining(size);
102                                return this;
103                        }
104                        
105                };
106        }
107        
108        /**
109         * Creates a progress monitor which multiplies work sizes passed to it by the scale. I.e. when
110         * the returned scaled monitor consumes X amount of work, this monitor consumes X / scale amount of work. 
111         * For example you want to consume 1 from this monitor and the task size is 10, then the scale is 10.
112         * @param scale
113         * @return
114         */
115        default ProgressMonitor scale(double scale) {
116                return new ProgressMonitor() {
117                        
118                        @Override
119                        public void worked(Status status, double work, String progressMessage, Object... data) {
120                                ProgressMonitor.this.worked(status, work / scale, progressMessage, data);
121                                
122                        }
123                        
124                        @Override
125                        public ProgressMonitor split(String taskName, double size, Object... data) {
126                                return ProgressMonitor.this.split(taskName, size / scale, data).scale(scale);
127                        }
128                        
129                        @Override
130                        public ProgressMonitor setWorkRemaining(double size) {
131                                ProgressMonitor.this.setWorkRemaining(size / scale);
132                                return this;
133                        }
134                        
135                        @Override
136                        public boolean isCancelled() {
137                                return ProgressMonitor.this.isCancelled();
138                        }
139                        
140                        @Override
141                        public void close() {
142                                ProgressMonitor.this.close();                           
143                        }
144                };
145        }
146                
147}
