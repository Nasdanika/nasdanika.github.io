001package org.nasdanika.common.persistence;
002
003import java.net.URL;
004import java.util.HashMap;
005import java.util.Map;
006import java.util.Map.Entry;
007
008import org.nasdanika.common.ProgressMonitor;
009
010/**
011 * Registers loaders by prefixes and then delegates to the registered loaders removing prefix first.  
012 * @author Pavel
013 *
014 */
015public class DispatchingLoader implements ObjectLoader {
016        
017        private org.nasdanika.common.persistence.ObjectLoader chain;
018
019        public DispatchingLoader(ObjectLoader chain) {
020                this.chain = chain;
021        }
022        
023        public DispatchingLoader() {}
024        
025        protected Map<String,ObjectLoader> registry = new HashMap<>();
026        
027        /**
028         * Registers a loader with given prefix
029         * @param prefix
030         * @param loader
031         */
032        public void register(String prefix, ObjectLoader loader) {
033                registry.put(prefix, loader);
034        }
035
036        @Override
037        public Object create(ObjectLoader loader, String type, Object config, URL base, ProgressMonitor progressMonitor, Marker marker) throws Exception {
038                for (Entry<String, ObjectLoader> re: registry.entrySet()) {
039                        if (type.startsWith(re.getKey())) {
040                                return re.getValue().create(loader, type.substring(re.getKey().length()), config, base, progressMonitor, marker);
041                        }                       
042                }
043                
044                if (chain == null) {
045                        throw new ConfigurationException("Unsupported type: " + type, marker);
046                }
047                
048                return chain.create(loader, type, config, base, progressMonitor, marker);
049        }
050                
051}
