053        /**
054         * Sets property value
055         * @param name
056         * @param value
057         * @return Set diagnostic or null if property was not found.
058         */
059        default Diagnostic setProperty(String name, Object value) {     
060                if (isEnabled()) {
061                        for (Descriptor descriptor: getDescriptors()) {
062                                if (descriptor.isEnabled()) {
063                                        if (descriptor instanceof PropertyDescriptor) {
064                                                PropertyDescriptor propertyDescriptor = (PropertyDescriptor) descriptor;
065                                                if (propertyDescriptor.getName().equals(name) 
066                                                                && propertyDescriptor.isEditable()) {
067                                                        
068                                                        return propertyDescriptor.set(value);                                   
069                                                }
070                                        } else if (descriptor instanceof DescriptorSet) {
071                                                Diagnostic diagnostic = ((DescriptorSet) descriptor).setProperty(name, value);
072                                                if (diagnostic != null) {
073                                                        return diagnostic;
074                                                }
075                                        }
076                                }
077                        }
078                }
079                return null;
080        }
081        
082        /**
083         * Sets property values retrieved from the source
084         * @param source Source of property values
085         */
086        default void setProperties(Function<PropertyDescriptor,Object> source) {        
087                if (isEnabled()) {
088                        for (Descriptor descriptor: getDescriptors()) {
089                                if (descriptor.isEnabled()) {
090                                        if (descriptor instanceof PropertyDescriptor) {
091                                                PropertyDescriptor propertyDescriptor = (PropertyDescriptor) descriptor;
092                                                if (propertyDescriptor.isEditable()) {
093                                                        Object value = source.apply(propertyDescriptor);
094                                                        if (value != null) {
095                                                                propertyDescriptor.set(value);
096                                                        }
097                                                }
098                                        } else if (descriptor instanceof DescriptorSet) {
099                                                ((DescriptorSet) descriptor).setProperties(source);
100                                        }
101                                }
102                        }
103                }
104        }
