001package org.nasdanika.common.persistence;
002
003import java.util.List;
004import java.util.Map;
005
006import org.eclipse.emf.common.util.URI;
007import org.nasdanika.common.ProgressMonitor;
008
009public interface Feature<T> extends Marked {
010        
011        Object getKey();
012        T getValue();
013        boolean isRequired();
014        boolean isLoaded();
015        
016        default Object get(Map<?,?> data) {
017                return data.get(getKey());
018        }
019        
020        /**
021         * @return If true, this feature is loaded if {@link FeatureObject} configuration is not a map. E.g. HTTP call can be configured with just a URL string, i.e. URL is the default feature of HTTP Call. 
022         */
023        boolean isDefault();
024
025        void load(ObjectLoader loader, Map<?,?> source, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) throws Exception;
026        
027        /**
028         * @return Feature description in Markdown.
029         */
030        String getDescription();        
031        
032}
