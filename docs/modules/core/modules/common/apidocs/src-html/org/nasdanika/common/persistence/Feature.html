001package org.nasdanika.common.persistence;
002
003import java.util.Map;
004
005import org.eclipse.emf.common.util.URI;
006import org.nasdanika.common.ProgressMonitor;
007
008public interface Feature<T> extends Marked {
009        
010        Object getKey();
011        T getValue();
012        boolean isRequired();
013        boolean isLoaded();
014        
015        default Object get(Map<?,?> data) {
016                return data.get(getKey());
017        }
018        
019        /**
020         * @return If true, this feature is loaded if {@link FeatureObject} configuration is not a map. E.g. HTTP call can be configured with just a URL string, i.e. URL is the default feature of HTTP Call. 
021         */
022        boolean isDefault();
023
024        void load(ObjectLoader loader, Map<?,?> source, URI base, ProgressMonitor progressMonitor, Marker marker) throws Exception;
025        
026        /**
027         * @return Feature description in Markdown.
028         */
029        String getDescription();        
030        
031}
