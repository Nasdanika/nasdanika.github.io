001package org.nasdanika.exec.gen.content;
002
003import java.io.InputStream;
004import java.net.MalformedURLException;
005import java.util.Base64;
006import java.util.List;
007
008import org.eclipse.emf.common.util.URI;
009import org.eclipse.emf.ecore.resource.Resource;
010import org.nasdanika.common.Context;
011import org.nasdanika.common.Converter;
012import org.nasdanika.common.DefaultConverter;
013import org.nasdanika.common.MarkdownHelper;
014import org.nasdanika.common.MutableContext;
015import org.nasdanika.common.PropertyComputer;
016import org.nasdanika.common.Util;
017import org.nasdanika.common.persistence.ConfigurationException;
018import org.nasdanika.common.persistence.Marked;
019import org.nasdanika.emf.EObjectAdaptable;
020import org.nasdanika.exec.content.Interpolator;
021import org.nasdanika.ncore.Marker;
022
023public class InterpolatorSupplierFactoryAdapter extends FilterSupplierFactoryAdapter<Interpolator> {
024
025        protected InterpolatorSupplierFactoryAdapter(Interpolator interpolator) {
026                super(interpolator);
027        }
028
029        @SuppressWarnings("unchecked")
030        private <T> T computeEmbeddedImage(Context context, String key, String path, Class<T> type) {
031                if (type == null || type.isAssignableFrom(String.class)) {
032                        int idx = path.indexOf("/");
033                        if (idx == -1) {
034                                return null;
035                        }
036
037                        try {
038                                StringBuilder ret = new StringBuilder("<img src=\"data:image/" + path.substring(0, idx) + ";base64, ");
039                                Converter converter = context.get(Converter.class, DefaultConverter.INSTANCE);
040                                byte[] imageBytes = converter.convert(converter.convert(resolve(path.substring(idx + 1)), InputStream.class), byte[].class);
041                                ret
042                                        .append(Base64.getEncoder().encodeToString(imageBytes))
043                                        .append("\"/>");
044                                
045                                return (T) ret.toString();
046                        } catch (Exception e) {
047                                throw new ConfigurationException("Error encoding image '" + path + "' : " + e, e, EObjectAdaptable.adaptTo(getTarget(), Marked.class));
048                        }
049                }
050                return null;
051        }
052        
053        @SuppressWarnings("unchecked")
054        private <T> T computeInclude(Context context, String key, String path, Class<T> type) {
055                if (type == null || type.isAssignableFrom(String.class)) {
056                        try {
057                                Converter converter = context.get(Converter.class, DefaultConverter.INSTANCE);
058                                String includeContent = converter.convert(converter.convert(resolve(path), InputStream.class), String.class);
059                                return (T) context.interpolateToString(includeContent);
060                        } catch (Exception e) {
061                                throw new ConfigurationException("Error including '" + path + "': " + e, e, EObjectAdaptable.adaptTo(getTarget(), Marked.class));
062                        }
063                }
064                return null;
065        }
066        
067        protected URI resolve(String path) throws MalformedURLException {
068                URI uri = URI.createURI(path);
069                
070                URI markerBase = null;
071                List<? extends Marker> markers = getTarget().getMarkers();
072                if (markers != null) {
073                        for (Marker marker: markers) {
074                                if (!Util.isBlank(marker.getLocation())) {
075                                        markerBase = URI.createURI(marker.getLocation());
076                                        break;
077                                }
078                        }
079                }
080
081                URI resourceBase = null;
082                Resource resource = getTarget().eResource();
083                if (resource != null) {
084                        resourceBase = resource.getURI();
085                }
086                
087                if (markerBase != null && markerBase.hasAbsolutePath() && path.startsWith("./")) {
088                        uri = uri.resolve(markerBase);
089                } else if (resourceBase != null) {
090                        uri = uri.resolve(resourceBase);
091                }
092                
093                return uri;                                     
094        }
095        
096        @SuppressWarnings("unchecked")
097        private <T> T computeIncludeMarkdown(Context context, String key, String path, Class<T> type) {
098                if (type == null || type.isAssignableFrom(String.class)) {
099                        try {
100                                Converter converter = context.get(Converter.class, DefaultConverter.INSTANCE);
101                                String markdown = converter.convert(converter.convert(resolve(path), InputStream.class), String.class);
102                                String html = context.computingContext().get(MarkdownHelper.class, MarkdownHelper.INSTANCE).markdownToHtml(markdown);
103                                return (T) context.interpolateToString(html);
104                        } catch (Exception e) {
105                                throw new ConfigurationException("Error including markdown '" + path + "': " + e, e, EObjectAdaptable.adaptTo(getTarget(), Marked.class));
106                        }
107                }
108                return null;
109        }       
110        
111        @SuppressWarnings("unchecked")
112        private <T> T computeIncludeStyledMarkdown(Context context, String key, String path, Class<T> type) {
113                Object html = computeIncludeMarkdown(context, key, path, type);
114                return (T) (html == null ? null : "<div class=\"markdown-body\">" + html + "</div>");
115        }       
116
117        @Override
118        protected String filter(Context context, String input) {
119                if (getTarget().isProcessIncludes()) {
120                        MutableContext mc = context.fork();
121                        mc.put("embedded-image", (PropertyComputer) this::computeEmbeddedImage);
122                        mc.put("include", (PropertyComputer) this::computeInclude);
123                        mc.put("include-markdown", (PropertyComputer) this::computeIncludeMarkdown);
124                        mc.put("include-styled-markdown", (PropertyComputer) this::computeIncludeStyledMarkdown);
125                        
126                        return mc.interpolateToString(input);
127                }
128                return context.interpolateToString(input);
129        }
130
131}
