001package org.nasdanika.exec.gen.content;
002
003import java.io.InputStream;
004import java.util.Objects;
005
006import org.eclipse.emf.common.notify.impl.AdapterImpl;
007import org.eclipse.emf.ecore.EObject;
008import org.nasdanika.common.Context;
009import org.nasdanika.common.Function;
010import org.nasdanika.common.FunctionFactory;
011import org.nasdanika.common.ProgressMonitor;
012import org.nasdanika.common.Supplier;
013import org.nasdanika.common.SupplierFactory;
014import org.nasdanika.common.Util;
015import org.nasdanika.emf.EObjectAdaptable;
016import org.nasdanika.exec.content.Filter;
017
018public abstract class FilterSupplierFactoryAdapter<T extends Filter> extends AdapterImpl implements SupplierFactory<InputStream> {
019        
020        protected FilterSupplierFactoryAdapter(T filter) {
021                setTarget(filter);
022        }
023        
024        @SuppressWarnings("unchecked")
025        @Override
026        public T getTarget() {
027                return (T) super.getTarget();
028        }
029        
030        private FunctionFactory<InputStream,InputStream> filterFactory = ctx -> new Function<InputStream, InputStream>() {
031
032                @Override
033                public double size() {
034                        return 1;
035                }
036
037                @Override
038                public String name() {
039                        return FilterSupplierFactoryAdapter.this.getClass().getName();
040                }
041
042                @Override
043                public InputStream execute(InputStream input, ProgressMonitor progressMonitor) throws Exception {
044                        return Util.filter(ctx, input, str -> FilterSupplierFactoryAdapter.this.filter(ctx, str));
045                }
046                
047        };
048                
049        @Override
050        public Supplier<InputStream> create(Context context) throws Exception {
051                EObject source = ((Filter) getTarget()).getSource();
052                SupplierFactory<InputStream> ssf = Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(source, InputStream.class), "Cannot adapt to SupplierFactory: " + source);
053                return ssf.then(filterFactory).create(context);
054        }
055        
056        protected abstract String filter(Context context, String input);        
057
058}
