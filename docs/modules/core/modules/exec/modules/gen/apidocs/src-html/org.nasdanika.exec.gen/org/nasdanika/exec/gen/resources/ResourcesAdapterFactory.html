001package org.nasdanika.exec.gen.resources;
002
003import org.nasdanika.common.ConsumerFactory;
004import org.nasdanika.common.resources.BinaryEntityContainer;
005import org.nasdanika.emf.ComposedAdapterFactory;
006import org.nasdanika.emf.FunctionAdapterFactory;
007import org.nasdanika.exec.resources.Container;
008import org.nasdanika.exec.resources.File;
009import org.nasdanika.exec.resources.ResourcesPackage;
010
011/**
012 * Provides adapters for the Engineering model elements.
013 * @author Pavel
014 *
015 */
016public class ResourcesAdapterFactory extends ComposedAdapterFactory {
017        
018        public ResourcesAdapterFactory() {
019                
020                registerAdapterFactory(
021                        new FunctionAdapterFactory<ConsumerFactory<BinaryEntityContainer>, File>(
022                                ResourcesPackage.Literals.FILE, 
023                                getBinaryEntityContainerConsumerFactoryClass(), 
024                                this.getClass().getClassLoader(), 
025                                FileConsumerFactoryAdapter::new));
026                
027                registerAdapterFactory(
028                        new FunctionAdapterFactory<ConsumerFactory<BinaryEntityContainer>, Container>(
029                                ResourcesPackage.Literals.CONTAINER, 
030                                getBinaryEntityContainerConsumerFactoryClass(), 
031                                this.getClass().getClassLoader(), 
032                                ContainerConsumerFactoryAdapter::new));
033                                
034        }
035
036        @SuppressWarnings({ "unchecked", "rawtypes" })
037        protected Class<ConsumerFactory<BinaryEntityContainer>> getBinaryEntityContainerConsumerFactoryClass() {
038                return (Class) ConsumerFactory.class;
039        }
040
041        
042}
