001package org.nasdanika.exec.gen;
002
003import java.io.InputStream;
004import java.util.ArrayList;
005import java.util.Collection;
006import java.util.List;
007import java.util.Objects;
008import java.util.stream.Collectors;
009
010import org.eclipse.emf.ecore.EObject;
011import org.nasdanika.common.Context;
012import org.nasdanika.common.ProgressMonitor;
013import org.nasdanika.common.Reference;
014import org.nasdanika.common.Supplier;
015import org.nasdanika.common.SupplierFactory;
016import org.nasdanika.common.Util;
017import org.nasdanika.emf.EObjectAdaptable;
018import org.nasdanika.exec.Block;
019
020public class BlockSupplierFactoryAdapter extends BlockExecutionParticipantAdapter implements SupplierFactory<InputStream> {
021        
022        public BlockSupplierFactoryAdapter(Block block) {
023                super(block);
024        }
025        
026        @Override
027        public boolean isAdapterForType(Object type) {
028                return type == SupplierFactory.class;
029        }
030        
031        private static class BlockSupplier extends BlockExecutionParticipant<Supplier<InputStream>> implements Supplier<InputStream> {
032        
033                private Reference<Exception> eRef;
034        
035                BlockSupplier(
036                                Supplier<InputStream> tryParticipant,
037                                Supplier<InputStream> catchParticipant,
038                                Supplier<InputStream> finallyParticipant,
039                                Reference<Exception> eRef) {
040                        super(tryParticipant, catchParticipant, finallyParticipant);
041                        this.eRef = eRef;
042                }
043        
044                private InputStream executeTryCatch(ProgressMonitor progressMonitor) throws Exception {
045                        try {
046                                return tryParticipant.splitAndExecute(progressMonitor);
047                        } catch (Exception e) {
048                                tryParticipant.splitAndRollback(progressMonitor);
049                                eRef.set(e);
050                                if (catchParticipant == null) {
051                                        throw e;
052                                }
053                                return catchParticipant.splitAndExecute(progressMonitor);
054                        }                       
055                }
056        
057                @Override
058                public InputStream execute(ProgressMonitor progressMonitor) throws Exception {
059                        if (finallyParticipant == null) {
060                                return executeTryCatch(progressMonitor);
061                        } 
062                        
063                        InputStream tcResult = null;
064                        Exception ex = null;
065                        InputStream finallyResult = null;
066                        try {
067                                tcResult = executeTryCatch(progressMonitor);
068                        } catch (Exception e) {
069                                ex = e;
070                        } finally {
071                                finallyResult = finallyParticipant.splitAndExecute(progressMonitor);
072                        }
073                        if (ex == null) {
074                                return Util.join(tcResult, finallyResult);
075                        }
076                        throw ex;                       
077                }
078                
079        }       
080        
081        @Override
082        public Supplier<InputStream> create(Context context) throws Exception {
083                Reference<Exception> eRef = new Reference<Exception>();
084                Block target = (Block) getTarget();
085                
086                Collection<SupplierFactory<InputStream>> tryBlock = new ArrayList<>();
087                for (EObject tbe: target.getTry()) {
088                        tryBlock.add(Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(tbe, InputStream.class), "Cannot adapt to SupplierFactory: " + tbe));
089                }
090                Supplier<InputStream>  trySupplier = Util.asInputStreamSupplierFactory(tryBlock).create(context);
091                
092                Supplier<InputStream> catchSupplier = null;             
093                if (!target.getCatch().isEmpty()) {
094                        List<SupplierFactory<InputStream>> catchBlock = target.getCatch().stream().map(e -> Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(e, InputStream.class), "Cannot adapt to SupplierFactory: " + e)).collect(Collectors.toList());
095                        catchSupplier = Util.asInputStreamSupplierFactory(catchBlock).contextify(ctx -> createCatchContextSupplier(ctx, eRef)).create(context);
096                }
097                
098                Supplier<InputStream>  finallySupplier = null;
099                if (!target.getFinally().isEmpty()) {
100                        List<SupplierFactory<InputStream>> finallyBlock = target.getFinally().stream().map(e -> Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(e, InputStream.class), "Cannot adapt to SupplierFactory: " + e)).collect(Collectors.toList());
101                        finallySupplier = Util.asInputStreamSupplierFactory(finallyBlock).contextify(ctx -> createCatchContextSupplier(ctx, eRef)).create(context);
102                }               
103                
104                return new BlockSupplier(trySupplier, catchSupplier, finallySupplier, eRef);            
105        }       
106
107}
