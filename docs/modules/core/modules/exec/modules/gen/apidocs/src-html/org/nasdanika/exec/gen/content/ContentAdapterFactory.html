001package org.nasdanika.exec.gen.content;
002
003import java.io.InputStream;
004
005import org.nasdanika.common.SupplierFactory;
006import org.nasdanika.emf.ComposedAdapterFactory;
007import org.nasdanika.emf.FunctionAdapterFactory;
008import org.nasdanika.exec.content.Base64;
009import org.nasdanika.exec.content.ContentPackage;
010import org.nasdanika.exec.content.Interpolator;
011import org.nasdanika.exec.content.Markdown;
012import org.nasdanika.exec.content.Resource;
013import org.nasdanika.exec.content.Text;
014
015/**
016 * Provides adapters for the Engineering model elements.
017 * @author Pavel
018 *
019 */
020public class ContentAdapterFactory extends ComposedAdapterFactory {
021        
022        public ContentAdapterFactory() {
023                
024                // Content
025                registerAdapterFactory(
026                        new FunctionAdapterFactory<SupplierFactory<InputStream>, Resource>(
027                                ContentPackage.Literals.RESOURCE, 
028                                getInputStreamSupplierFactoryClass(), 
029                                this.getClass().getClassLoader(), 
030                                ResourceSupplierFactoryAdapter::new));
031                
032                registerAdapterFactory(
033                                new FunctionAdapterFactory<SupplierFactory<InputStream>, Text>(
034                                        ContentPackage.Literals.TEXT, 
035                                        getInputStreamSupplierFactoryClass(), 
036                                        this.getClass().getClassLoader(), 
037                                        TextSupplierFactoryAdapter::new));
038                
039                registerAdapterFactory(
040                                new FunctionAdapterFactory<SupplierFactory<InputStream>, Markdown>(
041                                        ContentPackage.Literals.MARKDOWN, 
042                                        getInputStreamSupplierFactoryClass(), 
043                                        this.getClass().getClassLoader(), 
044                                        MarkdownSupplierFactoryAdapter::new));
045                
046                registerAdapterFactory(
047                                new FunctionAdapterFactory<SupplierFactory<InputStream>, Interpolator>(
048                                        ContentPackage.Literals.INTERPOLATOR, 
049                                        getInputStreamSupplierFactoryClass(), 
050                                        this.getClass().getClassLoader(), 
051                                        InterpolatorSupplierFactoryAdapter::new));
052                
053                registerAdapterFactory(
054                                new FunctionAdapterFactory<SupplierFactory<InputStream>, Base64>(
055                                        ContentPackage.Literals.BASE64, 
056                                        getInputStreamSupplierFactoryClass(), 
057                                        this.getClass().getClassLoader(), 
058                                        Base64SupplierFactoryAdapter::new));
059                                
060        }
061        
062        @SuppressWarnings({ "unchecked", "rawtypes" })
063        protected Class<SupplierFactory<InputStream>> getInputStreamSupplierFactoryClass() {
064                return (Class) SupplierFactory.class;
065        }
066        
067}
