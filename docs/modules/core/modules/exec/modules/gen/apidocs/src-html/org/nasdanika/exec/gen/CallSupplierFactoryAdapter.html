037public class CallSupplierFactoryAdapter extends AdapterImpl implements SupplierFactory<InputStream> {
038        
039        public CallSupplierFactoryAdapter(Call call) {
040                setTarget(call);
041        }
042        
043        @Override
044        public boolean isAdapterForType(Object type) {
045                return type == SupplierFactory.class;
046        }
047        
048        @Override
049        public Supplier<InputStream> create(Context context) throws Exception {
050                MapCompoundSupplierFactory<EStructuralFeature, Object> featureMapFactory = new MapCompoundSupplierFactory<>("Feature map factory");
051                Call call = (Call) getTarget();
052                EList<EObject> init = call.getInit();
053                if (!init.isEmpty()) {
054                        ListCompoundSupplierFactory<Object> initFactory = new ListCompoundSupplierFactory<>("Init"); 
055                        for (EObject initArg: init) {
056                                initFactory.add(Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(initArg, Object.class), "Cannot adapt to SupplierFactory: " + initArg));
057                        }
058                        featureMapFactory.put(ExecPackage.Literals.CALL__INIT, initFactory);
059                }
060                
061                EList<EObject> args = call.getArguments();
062                if (!args.isEmpty()) {
063                        ListCompoundSupplierFactory<Object> argFactory = new ListCompoundSupplierFactory<>("Arguments"); 
064                        for (EObject arg: args) {
065                                argFactory.add(Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(arg, Object.class), "Cannot adapt to SupplierFactory: " + arg));
066                        }
067                        featureMapFactory.put(ExecPackage.Literals.CALL__ARGUMENTS, argFactory);
068                }
069                
070                EMap<String, EObject> properties = call.getProperties();
071                if (!properties.isEmpty()) {
072                        MapCompoundSupplierFactory<String,Object> propertiesFactory = new MapCompoundSupplierFactory<>("Properties");
073                        for (Entry<String, EObject> pe: properties.entrySet()) {
074                                propertiesFactory.put(pe.getKey(), Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(pe.getValue(), Object.class), "Cannot adapt to SupplierFactory: " + pe.getValue()));
075                        }
076                        featureMapFactory.put(ExecPackage.Literals.CALL__PROPERTIES, propertiesFactory);
077                }
078                
079                return featureMapFactory.then(this::createResultFunction).create(context);
080        }
081        
082        protected Function<Map<EStructuralFeature,Object>, InputStream> createResultFunction(Context context) {
083                Call call = (Call) getTarget();
084                
085                return new Function<Map<EStructuralFeature,Object>, InputStream>() {
086                        
087                        private Class<?> clazz;
088                        private Constructor<?> constructor;
089                        private Object instance;                
090                        private Method method;
091
092                        @Override
093                        public double size() {
094                                return 1;
095                        }
096
097                        @Override
098                        public String name() {
099                                String description = call.getDescription();
100                                return Util.isBlank(description) ? "Call" : "Call " + description;
101                        }
102                        
103                        @Override
104                        public Diagnostic diagnose(ProgressMonitor progressMonitor) {
105                                String property = call.getProperty();
106                                if (!Util.isBlank(property)) {
107                                        instance = context.get(property);
108                                        if (instance == null) {
109                                                return new BasicDiagnostic(Status.ERROR, "Call property not found: " + property, call);
110                                        }
111                                        if (!call.getInit().isEmpty()) {
112                                                return new BasicDiagnostic(Status.ERROR, "Cannot specify constructor arguments for property", call);                                                            
113                                        }
114                                        if (!call.getProperties().isEmpty()) {
115                                                return new BasicDiagnostic(Status.ERROR, "Cannot specify properties for property", call);                                                               
116                                        }
117                                } else {
118                                        ClassLoader classLoader = context.get(ClassLoader.class, getClass().getClassLoader());
119                                        String service = call.getService();
120                                        if (!Util.isBlank(service)) {
121                                                try {
122                                                        instance = context.get(classLoader.loadClass(service));
123                                                        if (instance == null) {
124                                                                return new BasicDiagnostic(Status.ERROR, "Call service not found: " + service, call);
125                                                        }
126                                                        if (!call.getInit().isEmpty()) {
127                                                                return new BasicDiagnostic(Status.ERROR, "Cannot specify constructor arguments for service", call);                                                             
128                                                        }
129                                                        if (!call.getProperties().isEmpty()) {
130                                                                return new BasicDiagnostic(Status.ERROR, "Cannot specify properties for service", call);                                                                
131                                                        }
132                                                } catch (ClassNotFoundException e) {
133                                                        return new BasicDiagnostic(Status.ERROR, "Service class not found: " + service, call);
134                                                }                                               
135                                        } else {
136                                                String type = call.getType();
137                                                if (Util.isBlank(type)) {
138                                                        return new BasicDiagnostic(Status.ERROR, "Call must specify one of 'class', 'property', or 'service'", call);                                                   
139                                                }
140                                                try {
141                                                        clazz = classLoader.loadClass(type);
142                                                        if (!BiConsumer.class.isAssignableFrom(clazz) && !call.getProperties().isEmpty()) {
143                                                                return new BasicDiagnostic(Status.ERROR, "In order to inject properties " + clazz.getName() + " must implement " + BiConsumer.class, call);                                                                                                                     
144                                                        }
145                                                        if ((SupplierFactory.class.isAssignableFrom(clazz) || Supplier.class.isAssignableFrom(clazz))
146                                                                        && call.getInit().isEmpty()
147                                                                        && call.getProperties().isEmpty()
148                                                                        && Util.isBlank(call.getMethod())) {
149                                                                // Special case - creating an instance here so can be diagnosed
150                                                                try {
151                                                                        Constructor<?> constr = clazz.getDeclaredConstructor();
152                                                                        Object obj = constr.newInstance();
153                                                                        instance = SupplierFactory.class.isAssignableFrom(clazz) ? ((SupplierFactory<?>) obj).create(context) : (Supplier<?>) obj;
154                                                                        clazz = null;
155                                                                        return ((Supplier<?>) instance).diagnose(progressMonitor);
156                                                                } catch (Exception e) {
157                                                                        return new BasicDiagnostic(Status.ERROR, "Cannot instantiate " + clazz, call, e);                                                                                                                       
158                                                                        
159                                                                }                                                               
160                                                        }
161                                                } catch (ClassNotFoundException e) {
162                                                        return new BasicDiagnostic(Status.ERROR, "Class not found: " + type, call);
163                                                }                                                                                               
164                                        }
165                                }
166                                String methodName = call.getMethod();
167                                if (Util.isBlank(methodName)) {
168                                        if (!call.getArguments().isEmpty()) {
169                                                return new BasicDiagnostic(Status.ERROR, "Cannot specify arguments without method", call);                                                              
170                                        }
171                                } else {
172                                        if (clazz == null) {
173                                                for (Method method: instance.getClass().getMethods()) {
174                                                        if (method.getName().equals(methodName) && method.getParameterCount() == call.getArguments().size() && !Modifier.isStatic(method.getModifiers())) {
175                                                                this.method = method;
176                                                                break;
177                                                        }
178                                                }               
179                                                if (method == null) {
180                                                        return new BasicDiagnostic(Status.ERROR, "Method '" + methodName +"' with " + call.getArguments().size() + " arguments not found in " + instance, call);                                                                                                                                                                                
181                                                }
182                                        } else {
183                                                for (Method method: clazz.getMethods()) {
184                                                        if (method.getName().equals(methodName) && method.getParameterCount() == call.getArguments().size()) {
185                                                                if (Modifier.isStatic(method.getModifiers())) {
186                                                                        if (!call.getInit().isEmpty()) {
187                                                                                return new BasicDiagnostic(Status.ERROR, "Cannot specify constructor arguments for a static method", call);                                                             
188                                                                        }                                                               
189                                                                        if (!call.getProperties().isEmpty()) {
190                                                                                return new BasicDiagnostic(Status.ERROR, "Cannot specify properties for a static method", call);                                                                
191                                                                        }
192                                                                } else if (clazz != null) {
193                                                                        for (Constructor<?> constructor: clazz.getConstructors()) {
194                                                                                if (constructor.getParameterCount() == call.getInit().size()) {
195                                                                                        this.constructor = constructor;
196                                                                                        break;
197                                                                                }
198                                                                        }               
199                                                                        if (constructor == null) {
200                                                                                return new BasicDiagnostic(Status.ERROR, "There is no constructor for class '" + call.getType() + "' which takes " + call.getInit().size() + " arguments", call);
201                                                                        }
202                                                                }
203                                                                this.method = method;
204                                                                break;
205                                                        }
206                                                }               
207                                                if (method == null) {
208                                                        return new BasicDiagnostic(Status.ERROR, "Method '" + methodName +"' with " + call.getArguments().size() + " arguments not found in " + clazz, call);                                                                                                                                                                           
209                                                }                                               
210                                        }
211                                }
212                                return Function.super.diagnose(progressMonitor);
213                        }
214
215                        @SuppressWarnings("unchecked")
216                        @Override
217                        public InputStream execute(Map<EStructuralFeature,Object> data, ProgressMonitor progressMonitor) throws Exception {
218                                List<Object> initList = (List<Object>) data.get(ExecPackage.Literals.CALL__INIT);
219                                
220                                Converter converter = context.get(Converter.class, DefaultConverter.INSTANCE);
221                                Object obj;
222                                if (clazz == null) {
223                                        obj = instance;
224                                } else if (constructor == null) {
225                                        obj = null;
226                                } else {
227                                        Object[] constructorArguments = coerce(context, initList, constructor.getParameterTypes(), progressMonitor);
228                                        obj = constructor.newInstance(constructorArguments);
229                                        Map<String,Object> properties = (Map<String,Object>) data.get(ExecPackage.Literals.CALL__PROPERTIES);                                   
230                                        if (properties != null) {
231                                                for (Entry<String, Object> pe: properties.entrySet()) {
232                                                        ((BiConsumer<String,Object>) obj).accept(pe.getKey(), pe.getValue());
233                                                }
234                                        }
235                                }
236
237                                Object result;
238                                if (method == null) {
239                                        result = obj;
240                                } else {
241                                        List<Object> argList = (List<Object>) data.get(ExecPackage.Literals.CALL__ARGUMENTS);
242                                        Object[] methodArguments = coerce(context, argList, method.getParameterTypes(), progressMonitor);
243                                        result = method.invoke(obj, methodArguments);                                   
244                                }
245
246                                if (result instanceof SupplierFactory) {
247                                        result = ((SupplierFactory<?>) result).create(context); 
248                                } 
249                                
250                                if (result instanceof Supplier) {
251                                        result = Util.call((Supplier<?>) result, progressMonitor, null);
252                                }
253                                
254                                if (result == null) {
255                                        return null;
256                                }
257
258                                InputStream ret = converter.convert(result, InputStream.class);
259                                if (ret == null) {
260                                        throw new ConfigurationException("Cannot convert '" + obj + "' to InputStream", EObjectAdaptable.adaptTo((EObject) getTarget(), Marked.class));
261                                }
262                                return ret;
263                        }
264                };
265        }

267        /**
268         * Converts arguments to parameter types where possible.
269         * @param arguments
270         * @param parameterTypes
271         * @return
272         * @throws Exception
273         */
274        private Object[] coerce(Context context, List<Object> arguments, Class<?>[] parameterTypes, ProgressMonitor progressMonitor) throws Exception {
275                if (arguments == null) {
276                        return null;
277                }
278                Object[] ret = arguments.toArray();
279                for (int i = 0; i < ret.length; ++i) {
280                        ret[i] = coerce(context, ret[i], parameterTypes[i], progressMonitor);
281                }
282                return ret;
283        }
284        
285        /**
286         * Converts argument to parameter type where possible.
287         * @param argument
288         * @param parameterType
289         * @return
290         * @throws Exception
291         */
292        private Object coerce(Context context, Object argument, Class<?> parameterType, ProgressMonitor progressMonitor) throws Exception {
293                if (argument == null || parameterType.isInstance(argument)) {
294                        return argument;
295                }
296                if (parameterType == String.class && argument instanceof InputStream) {
297                        return Util.toString(context, (InputStream ) argument);
298                }
299                Converter converter = context.get(Converter.class, DefaultConverter.INSTANCE);
300                Object ret = converter.convert(argument, parameterType);
301                if (ret == null) {
302                        throw new ConfigurationException("Cannot convert " + argument + " to " + parameterType, EObjectAdaptable.adaptTo((EObject) getTarget(), Marked.class));
303                }
304                return ret;
305        }
