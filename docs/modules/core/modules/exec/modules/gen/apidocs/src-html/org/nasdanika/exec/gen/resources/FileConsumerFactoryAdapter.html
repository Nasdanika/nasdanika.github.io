001package org.nasdanika.exec.gen.resources;
002
003import java.io.InputStream;
004import java.util.Objects;
005import java.util.concurrent.CancellationException;
006
007import org.eclipse.emf.ecore.EObject;
008import org.nasdanika.common.BiSupplier;
009import org.nasdanika.common.Consumer;
010import org.nasdanika.common.ConsumerFactory;
011import org.nasdanika.common.Context;
012import org.nasdanika.common.FunctionFactory;
013import org.nasdanika.common.ListCompoundSupplierFactory;
014import org.nasdanika.common.ProgressMonitor;
015import org.nasdanika.common.SupplierFactory;
016import org.nasdanika.common.Util;
017import org.nasdanika.common.persistence.ConfigurationException;
018import org.nasdanika.common.persistence.Marked;
019import org.nasdanika.common.resources.BinaryEntity;
020import org.nasdanika.common.resources.BinaryEntityContainer;
021import org.nasdanika.common.resources.Merger;
022import org.nasdanika.emf.EObjectAdaptable;
023import org.nasdanika.exec.resources.File;
024
025public class FileConsumerFactoryAdapter extends ResourceConsumerFactoryAdapter<File> {
026
027        public FileConsumerFactoryAdapter(File file) {
028                super(file);
029        }
030        
031        @Override
032        public boolean isAdapterForType(Object type) {
033                return type == ConsumerFactory.class;
034        }
035        
036        protected SupplierFactory<InputStream> getContentsFactory() {
037                ListCompoundSupplierFactory<InputStream> contentsFactory = new ListCompoundSupplierFactory<>("Contents");
038                for (EObject ce: ((File) getTarget()).getContents()) {
039                        contentsFactory.add(Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(ce, InputStream.class), "Cannot adapt to SupplierFactory: " + ce));
040                }
041                return contentsFactory.then(Util.JOIN_STREAMS_FACTORY);
042        }
043                
044        private ConsumerFactory<BiSupplier<BinaryEntityContainer, InputStream>> createFileFactory() {
045                File semanticElement = (File) getTarget();
046                
047                return context -> new Consumer<BiSupplier<BinaryEntityContainer, InputStream>>() {
048
049                        private String finalName = finalName(context.interpolateToString(semanticElement.getName()));
050                
051                        @Override
052                        public double size() {
053                                return 1;
054                        }
055        
056                        @Override
057                        public String name() {
058                                return "Create file " + finalName;
059                        }
060        
061                        @Override
062                        public void execute(BiSupplier<BinaryEntityContainer, InputStream> input, ProgressMonitor progressMonitor) throws Exception {
063                                if (!progressMonitor.isCancelled()) {
064                                        switch (semanticElement.getReconcileAction()) {
065                                        case CANCEL:
066                                                if (input.getFirst().find(finalName, progressMonitor) != null) {
067                                                        throw new CancellationException("Cancelling generation - resource '" + finalName + "' already exists in " + input.getFirst());
068                                                }
069                                        case OVERWRITE: {
070                                                BinaryEntity file = Objects.requireNonNull(input.getFirst().get(finalName, progressMonitor), "Cannot create file " + finalName + " in " + input.getFirst());
071                                                if (input.getSecond() != null) {
072                                                        file.setState(input.getSecond(), progressMonitor);
073                                                }
074                                                break;
075                                        }
076                                        case KEEP: {
077                                                if (input.getFirst().find(finalName, progressMonitor) == null) {
078                                                        BinaryEntity file = Objects.requireNonNull(input.getFirst().get(finalName, progressMonitor), "Cannot create file " + finalName + " in " + input.getFirst());
079                                                        if (input.getSecond() != null) {
080                                                                file.setState(input.getSecond(), progressMonitor);
081                                                        }
082                                                }
083                                                break;
084                                        }
085                                        case APPEND: {
086                                                BinaryEntity file = Objects.requireNonNull(input.getFirst().get(finalName, progressMonitor), "Cannot create file " + finalName + " in " + input.getFirst());
087                                                if (input.getSecond() != null) {
088                                                        file.setState(file.exists(progressMonitor) ? Util.join(file.getState(progressMonitor), input.getSecond()) : input.getSecond(), progressMonitor);
089                                                }
090                                                break;
091                                        }                                                       
092                                        case MERGE: {
093                                                BinaryEntity file = Objects.requireNonNull(input.getFirst().get(finalName, progressMonitor), "Cannot create file " + finalName + " in " + input.getFirst());
094                                                if (input.getSecond() != null) {
095                                                        if (file.exists(progressMonitor)) {
096                                                                file.setState(merge(context, file, file.getState(progressMonitor), input.getSecond(), progressMonitor), progressMonitor);
097                                                        } else {
098                                                                file.setState(input.getSecond(), progressMonitor);
099                                                        }
100                                                }
101                                                break;
102                                        }                                                       
103                                        }
104                                }
105                        }
106                
107                };
108        }
109
110        @Override
111        public Consumer<BinaryEntityContainer> create(Context iContext) throws Exception {
112                FunctionFactory<BinaryEntityContainer, BiSupplier<BinaryEntityContainer, InputStream>> contentFunctionFactory = getContentsFactory().asFunctionFactory();
113                return contentFunctionFactory.then(createFileFactory()).create(iContext);
114        }
115        
116        protected InputStream merge(Context context, BinaryEntity entity, InputStream oldContent, InputStream newContent, ProgressMonitor progressMonitor) throws Exception {
117                File semanticElement = (File) getTarget();
118                EObject semanticMerger = semanticElement.getMerger();
119                Merger merger = semanticMerger == null ? getNativeMerger(context) : Objects.requireNonNull(EObjectAdaptable.adaptTo(semanticMerger, Merger.class));
120                if (merger == null) {                   
121                        throw new ConfigurationException("No merger for reconcile action MERGE", EObjectAdaptable.adaptTo(semanticElement, Marked.class));
122                }
123                return merger.merge(context, entity, oldContent, newContent, progressMonitor);
124        }
125        
126        protected Merger getNativeMerger(Context context) {
127                return null;
128        }       
129        
130}
