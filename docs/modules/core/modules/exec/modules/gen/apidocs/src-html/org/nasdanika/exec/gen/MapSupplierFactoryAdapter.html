001package org.nasdanika.exec.gen;
002
003import java.util.Map.Entry;
004import java.util.Objects;
005
006import org.eclipse.emf.common.notify.impl.AdapterImpl;
007import org.eclipse.emf.ecore.EObject;
008import org.nasdanika.common.Context;
009import org.nasdanika.common.MapCompoundSupplierFactory;
010import org.nasdanika.common.Supplier;
011import org.nasdanika.common.SupplierFactory;
012import org.nasdanika.emf.EObjectAdaptable;
013
014public class MapSupplierFactoryAdapter extends AdapterImpl implements SupplierFactory<java.util.Map<String,Object>> {
015        
016        public MapSupplierFactoryAdapter(org.nasdanika.exec.Map map) {
017                setTarget(map);
018        }
019        
020        @Override
021        public boolean isAdapterForType(Object type) {
022                return type == SupplierFactory.class;
023        }
024                
025        @Override
026        public Supplier<java.util.Map<String,Object>> create(Context context) throws Exception {
027                MapCompoundSupplierFactory<String,Object> entriesFactory = new MapCompoundSupplierFactory<>("Map");
028                for (Entry<String, EObject> e: ((org.nasdanika.exec.Map) getTarget()).getEntries()) {
029                        EObject value = e.getValue();
030                        entriesFactory.put(e.getKey(), Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(value, Object.class), "Cannot adapt to SupplierFactory: " + value));
031                }               
032                
033                return entriesFactory.create(context);
034        }       
035
036}
