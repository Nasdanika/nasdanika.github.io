037public class CallSupplierFactoryAdapter extends AdapterImpl implements SupplierFactory<InputStream> {
038        
039        public CallSupplierFactoryAdapter(Call call) {
040                setTarget(call);
041        }
042        
043        @Override
044        public boolean isAdapterForType(Object type) {
045                return type == SupplierFactory.class;
046        }
047        
048        @Override
049        public Supplier<InputStream> create(Context context) throws Exception {
050                MapCompoundSupplierFactory<EStructuralFeature, Object> featureMapFactory = new MapCompoundSupplierFactory<>("Feature map factory");
051                Call call = (Call) getTarget();
052                EList<EObject> init = call.getInit();
053                if (!init.isEmpty()) {
054                        ListCompoundSupplierFactory<Object> initFactory = new ListCompoundSupplierFactory<>("Init"); 
055                        for (EObject initArg: init) {
056                                initFactory.add(Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(initArg, Object.class), "Cannot adapt to SupplierFactory: " + initArg));
057                        }
058                        featureMapFactory.put(ExecPackage.Literals.CALL__INIT, initFactory);
059                }
060                
061                EList<EObject> args = call.getArguments();
062                if (!args.isEmpty()) {
063                        ListCompoundSupplierFactory<Object> argFactory = new ListCompoundSupplierFactory<>("Arguments"); 
064                        for (EObject arg: args) {
065                                argFactory.add(Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(arg, Object.class), "Cannot adapt to SupplierFactory: " + arg));
066                        }
067                        featureMapFactory.put(ExecPackage.Literals.CALL__ARGUMENTS, argFactory);
068                }
069                
070                EMap<String, EObject> properties = call.getProperties();
071                if (!properties.isEmpty()) {
072                        MapCompoundSupplierFactory<String,Object> propertiesFactory = new MapCompoundSupplierFactory<>("Properties");
073                        for (Entry<String, EObject> pe: properties.entrySet()) {
074                                propertiesFactory.put(pe.getKey(), Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(pe.getValue(), Object.class), "Cannot adapt to SupplierFactory: " + pe.getValue()));
075                        }
076                        featureMapFactory.put(ExecPackage.Literals.CALL__PROPERTIES, propertiesFactory);
077                }
078                
079                return featureMapFactory.then(this::createResultFunction).create(context);
080        }
081        
082        protected Function<Map<EStructuralFeature,Object>, InputStream> createResultFunction(Context context) {
083                Call call = (Call) getTarget();
084                
085                return new Function<Map<EStructuralFeature,Object>, InputStream>() {
086                        
087                        private Class<?> clazz;
088                        private Constructor<?> constructor;
089                        private Object instance;                
090                        private Method method;
091
092                        @Override
093                        public double size() {
094                                return 1;
095                        }
096
097                        @Override
098                        public String name() {
099                                String description = call.getDescription();
100                                return Util.isBlank(description) ? "Call" : "Call " + description;
101                        }
102                        
103                        @Override
104                        public Diagnostic diagnose(ProgressMonitor progressMonitor) {
105                                String property = call.getProperty();
106                                if (!Util.isBlank(property)) {
107                                        instance = context.get(property);
108                                        if (instance == null) {
109                                                return new BasicDiagnostic(Status.ERROR, "Call property not found: " + property, call);
110                                        }
111                                        if (!call.getInit().isEmpty()) {
112                                                return new BasicDiagnostic(Status.ERROR, "Cannot specify constructor arguments for property", call);                                                            
113                                        }
114                                        if (!call.getProperties().isEmpty()) {
115                                                return new BasicDiagnostic(Status.ERROR, "Cannot specify properties for property", call);                                                               
116                                        }
117                                } else {
118                                        ClassLoader classLoader = context.get(ClassLoader.class, getClass().getClassLoader());
119                                        String service = call.getService();
120                                        if (!Util.isBlank(service)) {
121                                                try {
122                                                        instance = context.get(classLoader.loadClass(service));
123                                                        if (instance == null) {
124                                                                return new BasicDiagnostic(Status.ERROR, "Call service not found: " + service, call);
125                                                        }
126                                                        if (!call.getInit().isEmpty()) {
127                                                                return new BasicDiagnostic(Status.ERROR, "Cannot specify constructor arguments for service", call);                                                             
128                                                        }
129                                                        if (!call.getProperties().isEmpty()) {
130                                                                return new BasicDiagnostic(Status.ERROR, "Cannot specify properties for service", call);                                                                
131                                                        }
132                                                } catch (ClassNotFoundException e) {
133                                                        return new BasicDiagnostic(Status.ERROR, "Service class not found: " + service, call);
134                                                }                                               
135                                        } else {
136                                                String type = call.getType();
137                                                if (Util.isBlank(type)) {
138                                                        return new BasicDiagnostic(Status.ERROR, "Call must specify one of 'class', 'property', or 'service'", call);                                                   
139                                                }
140                                                try {
141                                                        clazz = classLoader.loadClass(type);
142                                                        if (!BiConsumer.class.isAssignableFrom(clazz) && !call.getProperties().isEmpty()) {
143                                                                return new BasicDiagnostic(Status.ERROR, "In order to inject properties " + clazz.getName() + " must implement " + BiConsumer.class, call);                                                                                                                     
144                                                        }
145                                                        if ((SupplierFactory.class.isAssignableFrom(clazz) || Supplier.class.isAssignableFrom(clazz))
146                                                                        && call.getInit().isEmpty()
147                                                                        && call.getProperties().isEmpty()
148                                                                        && Util.isBlank(call.getMethod())) {
149                                                                // Special case - creating an instance here so can be diagnosed
150                                                                try {
151                                                                        Constructor<?> constr = clazz.getDeclaredConstructor();
152                                                                        Object obj = constr.newInstance();
153                                                                        instance = SupplierFactory.class.isAssignableFrom(clazz) ? ((SupplierFactory<?>) obj).create(context) : (Supplier<?>) obj;
154                                                                        clazz = null;
155                                                                        return ((Supplier<?>) instance).diagnose(progressMonitor);
156                                                                } catch (Exception e) {
157                                                                        return new BasicDiagnostic(Status.ERROR, "Cannot instantiate " + clazz, call, e);                                                                                                                       
158                                                                        
159                                                                }                                                               
160                                                        }
161                                                } catch (ClassNotFoundException e) {
162                                                        return new BasicDiagnostic(Status.ERROR, "Class not found: " + type, call);
163                                                }                                                                                               
164                                        }
165                                }
166                                String methodName = call.getMethod();
167                                if (Util.isBlank(methodName)) {
168                                        if (!call.getArguments().isEmpty()) {
169                                                return new BasicDiagnostic(Status.ERROR, "Cannot specify arguments without method", call);                                                              
170                                        }
171                                        for (Constructor<?> constructor: clazz.getConstructors()) {
172                                                if (constructor.getParameterCount() == call.getInit().size()) {
173                                                        this.constructor = constructor;
174                                                        break;
175                                                }
176                                        }               
177                                        if (constructor == null) {
178                                                return new BasicDiagnostic(Status.ERROR, "There is no constructor for class '" + call.getType() + "' which takes " + call.getInit().size() + " arguments", call);
179                                        }
180                                } else {
181                                        if (clazz == null) {
182                                                for (Method method: instance.getClass().getMethods()) {
183                                                        if (method.getName().equals(methodName) && method.getParameterCount() == call.getArguments().size() && !Modifier.isStatic(method.getModifiers())) {
184                                                                this.method = method;
185                                                                break;
186                                                        }
187                                                }               
188                                                if (method == null) {
189                                                        return new BasicDiagnostic(Status.ERROR, "Method '" + methodName +"' with " + call.getArguments().size() + " arguments not found in " + instance, call);                                                                                                                                                                                
190                                                }
191                                        } else {
192                                                for (Method method: clazz.getMethods()) {
193                                                        if (method.getName().equals(methodName) && method.getParameterCount() == call.getArguments().size()) {
194                                                                if (Modifier.isStatic(method.getModifiers())) {
195                                                                        if (!call.getInit().isEmpty()) {
196                                                                                return new BasicDiagnostic(Status.ERROR, "Cannot specify constructor arguments for a static method", call);                                                             
197                                                                        }                                                               
198                                                                        if (!call.getProperties().isEmpty()) {
199                                                                                return new BasicDiagnostic(Status.ERROR, "Cannot specify properties for a static method", call);                                                                
200                                                                        }
201                                                                } else if (clazz != null) {
202                                                                        for (Constructor<?> constructor: clazz.getConstructors()) {
203                                                                                if (constructor.getParameterCount() == call.getInit().size()) {
204                                                                                        this.constructor = constructor;
205                                                                                        break;
206                                                                                }
207                                                                        }               
208                                                                        if (constructor == null) {
209                                                                                return new BasicDiagnostic(Status.ERROR, "There is no constructor for class '" + call.getType() + "' which takes " + call.getInit().size() + " arguments", call);
210                                                                        }
211                                                                }
212                                                                this.method = method;
213                                                                break;
214                                                        }
215                                                }               
216                                                if (method == null) {
217                                                        return new BasicDiagnostic(Status.ERROR, "Method '" + methodName +"' with " + call.getArguments().size() + " arguments not found in " + clazz, call);                                                                                                                                                                           
218                                                }                                               
219                                        }
220                                }
221                                return Function.super.diagnose(progressMonitor);
222                        }
223
224                        @SuppressWarnings("unchecked")
225                        @Override
226                        public InputStream execute(Map<EStructuralFeature,Object> data, ProgressMonitor progressMonitor) throws Exception {
227                                List<Object> initList = (List<Object>) data.get(ExecPackage.Literals.CALL__INIT);
228                                
229                                Converter converter = context.get(Converter.class, DefaultConverter.INSTANCE);
230                                Object obj;
231                                if (clazz == null) {
232                                        obj = instance;
233                                } else if (constructor == null) {
234                                        obj = null;
235                                } else {
236                                        Object[] constructorArguments = coerce(context, initList, constructor.getParameterTypes(), progressMonitor);
237                                        obj = constructor.newInstance(constructorArguments);
238                                        Map<String,Object> properties = (Map<String,Object>) data.get(ExecPackage.Literals.CALL__PROPERTIES);                                   
239                                        if (properties != null) {
240                                                for (Entry<String, Object> pe: properties.entrySet()) {
241                                                        ((BiConsumer<String,Object>) obj).accept(pe.getKey(), pe.getValue());
242                                                }
243                                        }
244                                }
245
246                                Object result;
247                                if (method == null) {
248                                        result = obj;
249                                } else {
250                                        List<Object> argList = (List<Object>) data.get(ExecPackage.Literals.CALL__ARGUMENTS);
251                                        Object[] methodArguments = coerce(context, argList, method.getParameterTypes(), progressMonitor);
252                                        result = method.invoke(obj, methodArguments);                                   
253                                }
254
255                                if (result instanceof SupplierFactory) {
256                                        result = ((SupplierFactory<?>) result).create(context); 
257                                } 
258                                
259                                if (result instanceof Supplier) {
260                                        result = Util.call((Supplier<?>) result, progressMonitor, null);
261                                }
262                                
263                                if (result == null) {
264                                        return null;
265                                }
266
267                                InputStream ret = converter.convert(result, InputStream.class);
268                                if (ret == null) {
269                                        throw new ConfigurationException("Cannot convert '" + obj + "' to InputStream", EObjectAdaptable.adaptTo((EObject) getTarget(), Marked.class));
270                                }
271                                return ret;
272                        }
273                };
274        }

276        /**
277         * Converts arguments to parameter types where possible.
278         * @param arguments
279         * @param parameterTypes
280         * @return
281         * @throws Exception
282         */
283        private Object[] coerce(Context context, List<Object> arguments, Class<?>[] parameterTypes, ProgressMonitor progressMonitor) throws Exception {
284                if (arguments == null) {
285                        return null;
286                }
287                Object[] ret = arguments.toArray();
288                for (int i = 0; i < ret.length; ++i) {
289                        ret[i] = coerce(context, ret[i], parameterTypes[i], progressMonitor);
290                }
291                return ret;
292        }
293        
294        /**
295         * Converts argument to parameter type where possible.
296         * @param argument
297         * @param parameterType
298         * @return
299         * @throws Exception
300         */
301        private Object coerce(Context context, Object argument, Class<?> parameterType, ProgressMonitor progressMonitor) throws Exception {
302                if (argument == null || parameterType.isInstance(argument)) {
303                        return argument;
304                }
305                if (parameterType == String.class && argument instanceof InputStream) {
306                        return Util.toString(context, (InputStream ) argument);
307                }
308                Converter converter = context.get(Converter.class, DefaultConverter.INSTANCE);
309                Object ret = converter.convert(argument, parameterType);
310                if (ret == null) {
311                        throw new ConfigurationException("Cannot convert " + argument + " to " + parameterType, EObjectAdaptable.adaptTo((EObject) getTarget(), Marked.class));
312                }
313                return ret;
314        }
