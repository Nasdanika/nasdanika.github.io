001package org.nasdanika.exec.gen;
002
003import java.util.Map;
004import java.util.Map.Entry;
005import java.util.Objects;
006
007import org.eclipse.emf.common.notify.impl.AdapterImpl;
008import org.eclipse.emf.common.util.EMap;
009import org.eclipse.emf.ecore.EObject;
010import org.nasdanika.common.Context;
011import org.nasdanika.common.Function;
012import org.nasdanika.common.FunctionFactory;
013import org.nasdanika.common.MapCompoundSupplierFactory;
014import org.nasdanika.common.SupplierFactory;
015import org.nasdanika.common.Util;
016import org.nasdanika.emf.EObjectAdaptable;
017import org.nasdanika.exec.Configurator;
018
019public abstract class ConfiguratorExecutionParticipantAdapter extends AdapterImpl {
020        
021        protected ConfiguratorExecutionParticipantAdapter(Configurator configurator) {
022                setTarget(configurator);
023        }
024        
025        /**
026         * Creates context supplier factory from configuration for contextification.
027         * 
028         * @return Context supplier factory or null if configuration is empty.
029         */
030        protected org.nasdanika.common.SupplierFactory<Context> createContextSupplierFactory() {
031                Configurator configurator = (Configurator) getTarget();
032                EMap<String, EObject> properties = configurator.getProperties();
033                if (properties.isEmpty()) {
034                        return null;
035                }
036                
037                MapCompoundSupplierFactory<String, Object> entriesFactory = new MapCompoundSupplierFactory<String, Object>("Properties");
038                for (Entry<String, EObject> ce: properties.entrySet()) {
039                        EObject value = ce.getValue();
040                        SupplierFactory<?> sf = Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(value, Object.class), "Cannot adapt " + value + " to SupplierFactory");                          
041                        entriesFactory.put(ce.getKey(), sf.then(Util.OBJECT_TO_STRING_FACTORY));
042                }
043                
044                FunctionFactory<Map<String, Object>, org.nasdanika.common.Context> contextFactory = new FunctionFactory<Map<String, Object>, org.nasdanika.common.Context>() {
045                        
046                        @Override
047                        public Function<Map<String, java.lang.Object>, org.nasdanika.common.Context> create(org.nasdanika.common.Context context) throws Exception {
048                                return Function.fromBiFunction((map, progressMonitor) -> {
049                                        return org.nasdanika.common.Context.wrap(map::get).compose(context);
050                                }, "Contextifier", 1);
051                        }
052                        
053                };
054                
055                return entriesFactory.then(contextFactory);
056        }
057        
058}
