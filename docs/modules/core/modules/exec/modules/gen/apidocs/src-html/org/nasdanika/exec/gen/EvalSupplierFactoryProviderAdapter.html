023public class EvalSupplierFactoryProviderAdapter extends AdapterImpl implements SupplierFactory.Provider {
024        
025        private static final String CONTEXT_BINDING = "context";
026        private static final String PROGRESS_MONITOR_BINDING = "progressMonitor";
027        
028        public EvalSupplierFactoryProviderAdapter(Eval eval) {
029                setTarget(eval);
030        }
031        
032        @Override
033        public boolean isAdapterForType(Object type) {
034                return type == SupplierFactory.class;
035        }

051        @SuppressWarnings("unchecked")
052        @Override
053        public <T> SupplierFactory<T> getFactory(Class<T> type) {
054                Eval eval = (Eval) getTarget();
055                
056                MapCompoundSupplierFactory<String, Object> bindingsFactory = new MapCompoundSupplierFactory<String, Object>("Bindings");
057                EMap<String, EObject> bindings = eval.getBindings();
058                for (Entry<String, EObject> ce: bindings.entrySet()) {
059                        EObject value = ce.getValue();
060                        SupplierFactory<?> sf = Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(value, Object.class), "Cannot adapt " + value + " to SupplierFactory");                          
061                        bindingsFactory.put(ce.getKey(), sf.then(Util.OBJECT_TO_STRING_FACTORY));
062                }
063                
064                SupplierFactory<InputStream> scriptFactory = Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(eval.getScript(), InputStream.class), "Cannot adapt " + eval.getScript() + " to SupplierFactory");          
065                
066                
067                
068                if (type == null || type == Object.class) {
069                        FunctionFactory<BiSupplier<InputStream, Map<String,Object>>, Object> functionFactory = context -> new Function<BiSupplier<InputStream, Map<String,Object>>, Object>() {
070
071                                @Override
072                                public double size() {
073                                        return 1;
074                                }
075
076                                @Override
077                                public String name() {
078                                        return "eval";
079                                }
080
081                                @Override
082                                public Object execute(BiSupplier<InputStream, Map<String,Object>> input, ProgressMonitor progressMonitor) throws Exception {
083                                        Map<String,Object> bindings = new HashMap<>(input.getSecond());
084                                        bindings.put(CONTEXT_BINDING, context);
085                                        bindings.put(PROGRESS_MONITOR_BINDING, progressMonitor);                                        
086                                        return Util.eval(input.getFirst(), bindings);
087                                }
088                        };
089                        
090                        return (SupplierFactory<T>) scriptFactory.then(bindingsFactory.asFunctionFactory()).then(functionFactory);
091                }
092                
093                if (type == Boolean.class) {
094                        FunctionFactory<Object, Boolean> booleanFactory = context -> new Function<Object,Boolean>() {
095
096                                @Override
097                                public double size() {
098                                        return 1;
099                                }
100
101                                @Override
102                                public String name() {
103                                        return "boolean";
104                                }
105
106                                @Override
107                                public Boolean execute(Object obj, ProgressMonitor progressMonitor) throws Exception {
108                                        return Boolean.TRUE.equals(obj);
109                                }
110                        };
111                        
112                        return (SupplierFactory<T>) getFactory(Object.class).then(booleanFactory);                      
113                }
114                
115                if (type == InputStream.class) {
116                        FunctionFactory<Object, InputStream> booleanFactory = context -> new Function<Object,InputStream>() {
117
118                                @Override
119                                public double size() {
120                                        return 1;
121                                }
122
123                                @Override
124                                public String name() {
125                                        return "stream";
126                                }
127
128                                @Override
129                                public InputStream execute(Object obj, ProgressMonitor progressMonitor) throws Exception {
130                                        InputStream ret = DefaultConverter.INSTANCE.convert(obj, InputStream.class);
131                                        if (obj != null && ret == null) {
132                                                ret = Util.toStream(context, obj.toString());
133                                        }
134                                        return ret;
135                                }
136                        };
137                        
138                        return (SupplierFactory<T>) getFactory(Object.class).then(booleanFactory);                      
139                }
140                
141                throw new IllegalArgumentException("Unsupported type: " + type);
142        }
