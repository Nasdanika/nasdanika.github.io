001package org.nasdanika.exec.gen.content;
002
003import java.io.ByteArrayInputStream;
004import java.io.InputStream;
005import java.util.Objects;
006
007import org.eclipse.emf.common.notify.impl.AdapterImpl;
008import org.eclipse.emf.ecore.EObject;
009import org.nasdanika.common.Context;
010import org.nasdanika.common.DefaultConverter;
011import org.nasdanika.common.Function;
012import org.nasdanika.common.ProgressMonitor;
013import org.nasdanika.common.Supplier;
014import org.nasdanika.common.SupplierFactory;
015import org.nasdanika.emf.EObjectAdaptable;
016
017public class Base64SupplierFactoryAdapter extends AdapterImpl implements SupplierFactory<InputStream> {
018        
019        protected Base64SupplierFactoryAdapter(org.nasdanika.exec.content.Base64 base64) {
020                setTarget(base64);
021        }
022        
023        @Override
024        public org.nasdanika.exec.content.Base64 getTarget() {
025                return (org.nasdanika.exec.content.Base64) super.getTarget();
026        }
027        
028        @Override
029        public Supplier<InputStream> create(Context context) throws Exception {
030                EObject source = getTarget().getSource();
031                SupplierFactory<InputStream> ssf = Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(source, InputStream.class), "Cannot adapt to SupplierFactory: " + source);
032                Function<InputStream, InputStream> encoder = new Function<InputStream, InputStream>() {
033
034                        @Override
035                        public double size() {
036                                return 1;
037                        }
038
039                        @Override
040                        public String name() {
041                                return Base64SupplierFactoryAdapter.this.getClass().getName();
042                        }
043
044                        @Override
045                        public InputStream execute(InputStream input, ProgressMonitor progressMonitor) throws Exception {
046                                if (input == null) {
047                                        return null;
048                                }
049                                byte[] data = DefaultConverter.INSTANCE.toByteArray(input);
050                                byte[] encoded = org.apache.commons.codec.binary.Base64.encodeBase64(data);
051                                return new ByteArrayInputStream(encoded);
052                        }
053                        
054                };
055                return ssf.create(context).then(encoder);
056        }
057
058}
