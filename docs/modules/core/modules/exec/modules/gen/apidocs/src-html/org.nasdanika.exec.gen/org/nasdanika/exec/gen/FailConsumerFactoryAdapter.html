001package org.nasdanika.exec.gen;
002
003import org.eclipse.emf.common.notify.impl.AdapterImpl;
004import org.nasdanika.common.Consumer;
005import org.nasdanika.common.ConsumerFactory;
006import org.nasdanika.common.Context;
007import org.nasdanika.common.NasdanikaException;
008import org.nasdanika.common.ProgressMonitor;
009import org.nasdanika.common.SupplierFactory;
010import org.nasdanika.common.resources.BinaryEntityContainer;
011import org.nasdanika.exec.Fail;
012
013public class FailConsumerFactoryAdapter extends AdapterImpl implements ConsumerFactory<BinaryEntityContainer> {
014        
015        public FailConsumerFactoryAdapter(Fail fail) {
016                setTarget(fail);
017        }
018        
019        @Override
020        public boolean isAdapterForType(Object type) {
021                return type == SupplierFactory.class;
022        }
023
024        @Override
025        public Consumer<BinaryEntityContainer> create(Context context) throws Exception {
026                String message = context.interpolateToString(((Fail) getTarget()).getMessage());
027                
028                return new Consumer<BinaryEntityContainer>() {
029
030                        @Override
031                        public double size() {
032                                return 1;
033                        }
034
035                        @Override
036                        public String name() {
037                                return "[Fail] " + message;
038                        }
039
040                        @Override
041                        public void execute(BinaryEntityContainer container, ProgressMonitor progressMonitor) throws Exception {
042                                throw new NasdanikaException(message);
043                        }
044                };
045        }
046        
047        
048
049}
