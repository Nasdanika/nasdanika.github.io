001package org.nasdanika.exec.gen;
002
003import java.util.List;
004import java.util.Objects;
005import java.util.stream.Collectors;
006
007import org.nasdanika.common.Consumer;
008import org.nasdanika.common.ConsumerFactory;
009import org.nasdanika.common.Context;
010import org.nasdanika.common.ProgressMonitor;
011import org.nasdanika.common.Reference;
012import org.nasdanika.common.SupplierFactory;
013import org.nasdanika.common.Util;
014import org.nasdanika.common.resources.BinaryEntityContainer;
015import org.nasdanika.emf.EObjectAdaptable;
016import org.nasdanika.exec.Block;
017
018public class BlockConsumerFactoryAdapter extends BlockExecutionParticipantAdapter implements ConsumerFactory<BinaryEntityContainer> {
019        
020        public BlockConsumerFactoryAdapter(Block block) {
021                super(block);
022        }
023        
024        @Override
025        public boolean isAdapterForType(Object type) {
026                return type == SupplierFactory.class;
027        }
028        
029        private static class BlockConsumer extends BlockExecutionParticipant<Consumer<BinaryEntityContainer>> implements Consumer<BinaryEntityContainer> {
030
031                private Reference<Exception> eRef;
032
033                BlockConsumer(
034                                Consumer<BinaryEntityContainer> tryParticipant,
035                                Consumer<BinaryEntityContainer> catchParticipant,
036                                Consumer<BinaryEntityContainer> finallyParticipant,
037                                Reference<Exception> eRef) {
038                        super(tryParticipant, catchParticipant, finallyParticipant);
039                        this.eRef = eRef;
040                }
041
042                @Override
043                public void execute(BinaryEntityContainer container, ProgressMonitor progressMonitor) throws Exception {
044                        try {
045                                tryParticipant.splitAndExecute(container, progressMonitor);
046                        } catch (Exception e) {
047                                tryParticipant.splitAndRollback(progressMonitor);
048                                eRef.set(e);
049                                if (catchParticipant == null) {
050                                        throw e;
051                                }
052                                catchParticipant.splitAndExecute(container, progressMonitor);
053                        } finally {
054                                if (finallyParticipant != null) {
055                                        finallyParticipant.splitAndExecute(container, progressMonitor);
056                                }
057                        }                       
058                }
059                
060        }       
061        
062        @Override
063        public Consumer<BinaryEntityContainer> create(Context context) throws Exception {
064                Reference<Exception> eRef = new Reference<Exception>();
065                Block target = (Block) getTarget();
066                
067                List<ConsumerFactory<BinaryEntityContainer>> tryBlock = target.getTry().stream().map(e -> Objects.requireNonNull(EObjectAdaptable.adaptToConsumerFactory(e, BinaryEntityContainer.class), "Cannot adapt to ConsumerFactory: " + e)).collect(Collectors.toList());
068                Consumer<BinaryEntityContainer>  tryConsumer = Util.asConsumerFactory(tryBlock).create(context);
069                
070                Consumer<BinaryEntityContainer> catchConsumer = null;           
071                if (!target.getCatch().isEmpty()) {
072                        List<ConsumerFactory<BinaryEntityContainer>> catchBlock = target.getCatch().stream().map(e -> Objects.requireNonNull(EObjectAdaptable.adaptToConsumerFactory(e, BinaryEntityContainer.class), "Cannot adapt to ConsumerFactory: " + e)).collect(Collectors.toList());
073                        catchConsumer = Util.asConsumerFactory(catchBlock).contextify(ctx -> createCatchContextSupplier(ctx, eRef)).create(context);
074                }
075                
076                Consumer<BinaryEntityContainer>  finallyConsumer = null;
077                if (!target.getFinally().isEmpty()) {
078                        List<ConsumerFactory<BinaryEntityContainer>> finallyBlock = target.getFinally().stream().map(e -> Objects.requireNonNull(EObjectAdaptable.adaptToConsumerFactory(e, BinaryEntityContainer.class), "Cannot adapt to ConsumerFactory: " + e)).collect(Collectors.toList());
079                        finallyConsumer = Util.asConsumerFactory(finallyBlock).contextify(ctx -> createCatchContextSupplier(ctx, eRef)).create(context);
080                }               
081                
082                return new BlockConsumer(tryConsumer, catchConsumer, finallyConsumer, eRef);            
083        }       
084
085}
