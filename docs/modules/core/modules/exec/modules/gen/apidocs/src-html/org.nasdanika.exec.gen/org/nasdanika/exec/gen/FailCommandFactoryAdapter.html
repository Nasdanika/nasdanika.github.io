001package org.nasdanika.exec.gen;
002
003import org.eclipse.emf.common.notify.impl.AdapterImpl;
004import org.nasdanika.common.Command;
005import org.nasdanika.common.CommandFactory;
006import org.nasdanika.common.Context;
007import org.nasdanika.common.NasdanikaException;
008import org.nasdanika.common.ProgressMonitor;
009import org.nasdanika.common.SupplierFactory;
010import org.nasdanika.exec.Fail;
011
012public class FailCommandFactoryAdapter extends AdapterImpl implements CommandFactory {
013        
014        public FailCommandFactoryAdapter(Fail fail) {
015                setTarget(fail);
016        }
017        
018        @Override
019        public boolean isAdapterForType(Object type) {
020                return type == SupplierFactory.class;
021        }
022
023        @Override
024        public Command create(Context context) throws Exception {
025                String message = context.interpolateToString(((Fail) getTarget()).getMessage());
026                
027                return new Command() {
028
029                        @Override
030                        public double size() {
031                                return 1;
032                        }
033
034                        @Override
035                        public String name() {
036                                return "[Fail] " + message;
037                        }
038
039                        @Override
040                        public void execute(ProgressMonitor progressMonitor) throws Exception {
041                                throw new NasdanikaException(message);
042                        }
043                };
044        }
045        
046        
047
048}
