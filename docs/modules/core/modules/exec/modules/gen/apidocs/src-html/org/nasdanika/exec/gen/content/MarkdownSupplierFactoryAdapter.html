001package org.nasdanika.exec.gen.content;
002
003import java.util.List;
004
005import org.eclipse.emf.common.util.URI;
006import org.eclipse.emf.ecore.resource.Resource;
007import org.nasdanika.common.Context;
008import org.nasdanika.common.DiagramGenerator;
009import org.nasdanika.common.MarkdownHelper;
010import org.nasdanika.common.Util;
011import org.nasdanika.common.persistence.ConfigurationException;
012import org.nasdanika.exec.content.Markdown;
013import org.nasdanika.ncore.Marker;
014
015public class MarkdownSupplierFactoryAdapter extends FilterSupplierFactoryAdapter<Markdown> {
016
017        protected MarkdownSupplierFactoryAdapter(Markdown markdown) {
018                super(markdown);
019        }
020
021        @Override
022        protected String filter(Context context, String input) {
023                MarkdownHelper markdownHelper = new MarkdownHelper() {
024                        
025                        @Override
026                        protected URI getResourceBase() {
027                                Resource targetResource = getTarget().eResource();
028                                URI resURI = targetResource == null ? null : targetResource.getURI();
029                                List<? extends Marker> markers = getTarget().getMarkers();
030                                if (markers != null) {
031                                        for(Marker marker: markers) {
032                                                if (marker == null || Util.isBlank(marker.getLocation())) {
033                                                        return resURI;
034                                                }
035                                
036                                                try {
037                                                        URI markerLocation = URI.createURI(marker.getLocation());
038                                                        return resURI != null && resURI.isHierarchical() ? markerLocation.resolve(resURI) : markerLocation;
039                                                } catch (Exception e) {
040                                                        throw new ConfigurationException("Invalid location: " + marker.getLocation(), e, marker);
041                                                }
042                                        }
043                                }
044                                return resURI;
045                        }
046                        
047                        @Override
048                        protected DiagramGenerator getDiagramGenerator() {
049                                return context == null ? super.getDiagramGenerator() : context.get(DiagramGenerator.class, super.getDiagramGenerator()); 
050                        }
051                        
052                };
053                String html = /* context.computingContext().get(MarkdownHelper.class, helperComputer) */ markdownHelper.markdownToHtml(input);
054                return getTarget().isStyle() ? "<div class=\"markdown-body\">" + html + "</div>" : html;
055        }
056
057}
