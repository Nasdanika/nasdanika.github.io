001package org.nasdanika.exec.gen.resources;
002
003import org.eclipse.emf.common.notify.impl.AdapterImpl;
004import org.nasdanika.common.ConsumerFactory;
005import org.nasdanika.common.resources.BinaryEntityContainer;
006import org.nasdanika.exec.resources.Resource;
007
008public abstract class ResourceConsumerFactoryAdapter<T extends Resource> extends AdapterImpl implements ConsumerFactory<BinaryEntityContainer> {
009
010        protected ResourceConsumerFactoryAdapter(T resource) {
011                setTarget(resource);
012        }
013        
014        @Override
015        public boolean isAdapterForType(Object type) {
016                return type == ConsumerFactory.class;
017        }
018
019        /**
020         * Produces final (physical) resource name from a a name (logical). E.g. "logical" Java package name is dot separated, but container name is slash separated.
021         * @param name
022         * @return
023         */
024        protected String finalName(String name) {
025                return name;
026        }       
027
028}
