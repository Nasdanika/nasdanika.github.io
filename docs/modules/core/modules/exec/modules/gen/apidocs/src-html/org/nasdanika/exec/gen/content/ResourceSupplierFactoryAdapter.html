001package org.nasdanika.exec.gen.content;
002
003import java.io.InputStream;
004import java.net.MalformedURLException;
005import java.net.URL;
006
007import org.eclipse.emf.common.notify.impl.AdapterImpl;
008import org.eclipse.emf.ecore.EObject;
009import org.nasdanika.common.BasicDiagnostic;
010import org.nasdanika.common.Context;
011import org.nasdanika.common.Diagnostic;
012import org.nasdanika.common.ProgressMonitor;
013import org.nasdanika.common.Status;
014import org.nasdanika.common.Supplier;
015import org.nasdanika.common.SupplierFactory;
016import org.nasdanika.common.Util;
017import org.nasdanika.common.persistence.Marked;
018import org.nasdanika.common.persistence.Marker;
019import org.nasdanika.emf.EObjectAdaptable;
020import org.nasdanika.exec.content.Resource;
021
022public class ResourceSupplierFactoryAdapter extends AdapterImpl implements SupplierFactory<InputStream> {
023
024        private Marker marker;
025        
026        public ResourceSupplierFactoryAdapter(Resource resource) {
027                setTarget(resource);
028                Marked marked = EObjectAdaptable.adaptTo((EObject) getTarget(), Marked.class);
029                marker = marked == null ? null : marked.getMarker();
030        }
031        
032        @Override
033        public boolean isAdapterForType(Object type) {
034                return type == SupplierFactory.class;
035        }
036
037        @Override
038        public Supplier<InputStream> create(Context context) throws Exception {
039                return new Supplier<InputStream>() {
040                        
041                        private URL theURL;
042                        
043                        @Override
044                        public Diagnostic diagnose(ProgressMonitor progressMonitor) {
045                                try {
046                                        Resource resource = (Resource) getTarget();
047                                        String iUrl = resource.isInterpolate() ? context.interpolateToString(resource.getLocation()) : resource.getLocation(); 
048                                        theURL = iUrl.startsWith(Util.CLASSPATH_URL_PREFIX) ? getClass().getClassLoader().getResource(iUrl.substring(Util.CLASSPATH_URL_PREFIX.length())) : new URL(iUrl);
049                                        if (theURL == null) {
050                                                return new BasicDiagnostic(Status.ERROR, "Classpath resource not found: " + iUrl, marker);                                              
051                                        }
052                                        return Supplier.super.diagnose(progressMonitor);
053                                } catch (MalformedURLException e) {                                     
054                                        return new BasicDiagnostic(Status.ERROR, e.getMessage() + (marker == null ? "" : " at " + marker));
055                                }
056                        }
057
058                        @Override
059                        public double size() {
060                                return 1;
061                        }
062
063                        @Override
064                        public String name() {
065                                return "Resource: " + marker;
066                        }
067
068                        @Override
069                        public InputStream execute(ProgressMonitor progressMonitor) throws Exception {
070                                return theURL.openStream();
071                        }
072                        
073                };
074        }
075
076}
