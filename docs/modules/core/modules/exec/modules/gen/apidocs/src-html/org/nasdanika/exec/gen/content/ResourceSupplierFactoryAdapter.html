001package org.nasdanika.exec.gen.content;
002
003import java.io.InputStream;
004import java.net.MalformedURLException;
005import java.net.URL;
006import java.util.List;
007import java.util.stream.Collectors;
008
009import org.eclipse.emf.common.notify.impl.AdapterImpl;
010import org.eclipse.emf.common.util.URI;
011import org.eclipse.emf.ecore.EObject;
012import org.nasdanika.common.BasicDiagnostic;
013import org.nasdanika.common.Context;
014import org.nasdanika.common.Diagnostic;
015import org.nasdanika.common.ProgressMonitor;
016import org.nasdanika.common.Status;
017import org.nasdanika.common.Supplier;
018import org.nasdanika.common.SupplierFactory;
019import org.nasdanika.common.Util;
020import org.nasdanika.common.persistence.Marked;
021import org.nasdanika.common.persistence.Marker;
022import org.nasdanika.emf.EObjectAdaptable;
023import org.nasdanika.exec.content.Resource;
024
025public class ResourceSupplierFactoryAdapter extends AdapterImpl implements SupplierFactory<InputStream> {
026
027        private List<? extends Marker> markers;
028        
029        public ResourceSupplierFactoryAdapter(Resource resource) {
030                setTarget(resource);
031                Marked marked = EObjectAdaptable.adaptTo((EObject) getTarget(), Marked.class);
032                markers = marked == null ? null : marked.getMarkers();
033        }
034        
035        @Override
036        public boolean isAdapterForType(Object type) {
037                return type == SupplierFactory.class;
038        }
039
040        @Override
041        public Supplier<InputStream> create(Context context) throws Exception {
042                return new Supplier<InputStream>() {
043                        
044                        private URL theURL;
045                        
046                        @Override
047                        public Diagnostic diagnose(ProgressMonitor progressMonitor) {
048                                try {
049                                        Resource resource = (Resource) getTarget();
050                                        URI uri = URI.createURI(resource.isInterpolate() ? context.interpolateToString(resource.getLocation()) : resource.getLocation());
051                                        org.eclipse.emf.ecore.resource.Resource eResource = resource.eResource();
052                                        if (eResource != null) {
053                                                URI base = eResource.getURI();
054                                                if (base != null) {
055                                                        uri = uri.resolve(base);
056                                                }
057                                        }
058                                        
059                                        String iUrl = uri.toString(); 
060                                        
061                                        theURL = iUrl.startsWith(Util.CLASSPATH_URL_PREFIX) ? getClass().getClassLoader().getResource(iUrl.substring(Util.CLASSPATH_URL_PREFIX.length())) : new URL(iUrl);
062                                        if (theURL == null) {
063                                                return new BasicDiagnostic(Status.ERROR, "Classpath resource not found: " + iUrl, markers);                                             
064                                        }
065                                        return Supplier.super.diagnose(progressMonitor);
066                                } catch (MalformedURLException e) {                                     
067                                        return new BasicDiagnostic(Status.ERROR, e.getMessage() + (markers == null ? "" : " at " + markers.stream().map(Object::toString).collect(Collectors.joining(", "))));
068                                }
069                        }
070
071                        @Override
072                        public double size() {
073                                return 1;
074                        }
075
076                        @Override
077                        public String name() {
078                                return "Resource: " + markers == null ? "(undefined)" : markers.stream().map(Object::toString).collect(Collectors.joining(", "));
079                        }
080
081                        @Override
082                        public InputStream execute(ProgressMonitor progressMonitor) throws Exception {
083                                return theURL.openStream();
084                        }
085                        
086                };
087        }
088
089}
