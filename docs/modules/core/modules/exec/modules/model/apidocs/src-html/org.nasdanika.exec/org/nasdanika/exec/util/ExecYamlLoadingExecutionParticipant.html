001package org.nasdanika.exec.util;
002
003import java.util.Collection;
004
005import org.eclipse.emf.ecore.EPackage;
006import org.nasdanika.common.Context;
007import org.nasdanika.emf.persistence.NcoreYamlLoadingExecutionParticipant;
008import org.nasdanika.exec.ExecPackage;
009import org.nasdanika.exec.content.ContentPackage;
010import org.nasdanika.exec.resources.ResourcesPackage;
011
012/**
013 * {@link YamlLoadingSupplier} for Engineering {@link EPackage}s.
014 * Registers exec- loader. 
015 * @author Pavel
016 *
017 */
018public abstract class ExecYamlLoadingExecutionParticipant extends NcoreYamlLoadingExecutionParticipant {
019
020        public ExecYamlLoadingExecutionParticipant(Context context) {
021                super(context);
022        }
023
024        @Override
025        protected Collection<EPackage> getEPackages() {
026                Collection<EPackage> ret = super.getEPackages(); 
027                ret.add(ExecPackage.eINSTANCE);
028                ret.add(ContentPackage.eINSTANCE);
029                ret.add(ResourcesPackage.eINSTANCE);
030                return ret;
031        }
032        
033}
