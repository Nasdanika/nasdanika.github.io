001package org.nasdanika.emf;
002
003import java.util.function.Function;
004
005import org.eclipse.emf.common.notify.Notifier;
006import org.eclipse.emf.ecore.EClass;
007import org.eclipse.emf.ecore.EObject;
008import org.eclipse.emf.ecore.EcorePackage;
009
010/**
011 * Creates an adapter using a {@link Function} passing the notifier (target) as an argument.
012 * @param <T> Adapter type.
013 * @param <N> Notifier type.
014 */
015public class FunctionAdapterFactory<T, N> extends DelegatingAdapterFactory<T> {
016        
017        private Function<N, T> function;
018        
019        /**
020         * Uses {@link EObject}'s {@link EClass} as eClass argument
021         * @param type
022         * @param proxyClassLoader
023         * @param function
024         */
025        public FunctionAdapterFactory(Class<T> type, ClassLoader proxyClassLoader, Function<N, T> function) {
026                this(EcorePackage.Literals.EOBJECT, type, proxyClassLoader, function);
027        } 
028        
029        public FunctionAdapterFactory(EClass eClass, Class<T> type, ClassLoader proxyClassLoader, Function<N, T> function) {
030                super(eClass, type, proxyClassLoader);
031                this.function = function;
032        }
033
034        @SuppressWarnings("unchecked")
035        @Override
036        protected T doCreateAdapter(Notifier target) {
037                return function.apply((N) target);
038        }
039
040}
