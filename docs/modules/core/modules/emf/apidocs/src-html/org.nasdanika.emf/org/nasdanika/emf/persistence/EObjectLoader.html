122         * 
123         * @param eClass
124         * @param feature
125         * @return Features in the argument class the argument feature is exclusive with.
126         */
127        public static Object[] getExclusiveWith(EClass eClass, EStructuralFeature feature, BiFunction<EClass,ENamedElement,String> keyProvider) {
128                Set<String> ret = new HashSet<>();
129                for (EStructuralFeature sf: eClass.getEAllStructuralFeatures()) {
130                        String exclusiveWithStr = NcoreUtil.getNasdanikaAnnotationDetail(sf, "exclusive-with"); // A space-separated list of load keys which are mutually exclusive with the annotated feature.
131                        if (!Util.isBlank(exclusiveWithStr)) {
132                                for (String ew: exclusiveWithStr.split("\\s")) {
133                                        if (sf == feature) {
134                                                ret.add(ew);
135                                        } else if (ew.equals(keyProvider.apply(eClass, feature))) {
136                                                ret.add(keyProvider.apply(eClass, sf));
137                                        }
138                                        
139                                }
140                        }
141                }
142
143                return ret.toArray();
144        }

152        public static boolean isDefaultFeature(EClass eClass, EStructuralFeature feature) {
153                if ("true".equals(NcoreUtil.getNasdanikaAnnotationDetail(feature, IS_DEFAULT_FEATURE))) {
154                        return true;
155                }
156                String dfName = NcoreUtil.getNasdanikaAnnotationDetail(eClass, IS_DEFAULT_FEATURE);
157                if (!Util.isBlank(dfName)) {
158                        return feature.getName().equals(dfName);
159                }
160                for (EClass st: eClass.getEAllSuperTypes()) {
161                        dfName = NcoreUtil.getNasdanikaAnnotationDetail(st, IS_DEFAULT_FEATURE);
162                        if (!Util.isBlank(dfName)) {
163                                return feature.getName().equals(dfName);
164                        }                       
165                }
166                return false;
167        }

189        /**
190         * Default implementation of load key provider - gets load key from the Nasdanika annotation (urn:org.nasdanika) <code>load-key</code> details if it is present.
191         * For features delegates to {@link NcoreUtil} getFeatureKey.
192         * Otherwise converts {@link ENamedElement} element name from camel case to lower-cased kebab case. E.g. firstName is converted first-name.
193         */
194        public static final BiFunction<EClass,ENamedElement,String> LOAD_KEY_PROVIDER = (eClass,ne) -> {
195                EAnnotation na = NcoreUtil.getNasdanikaAnnotation(ne);
196                if (na != null) {
197                        if (na.getDetails().containsKey(IS_LOADABLE) && "false".equals(na.getDetails().get(IS_LOADABLE))) {
198                                return null;
199                        }
200                        if (na.getDetails().containsKey(LOAD_KEY)) {
201                                return na.getDetails().get(LOAD_KEY);
202                        }
203                }
204
205                if (eClass != null && ne instanceof EStructuralFeature) {
206                        EStructuralFeature feature = (EStructuralFeature) ne;
207                        String loadKey = getLoadKey(eClass, feature);           
208                        return loadKey == null ? NcoreUtil.getFeatureKey(eClass, feature) : loadKey;
209                }
210                
211                return Util.camelToKebab(ne.getName());
212        };

268        public void register(EPackage ePackage, BiFunction<EClass,ENamedElement,String> keyProvider) {
269                if (keyProvider == null) {
270                        keyProvider = this.keyProvider;
271                }
272                String packageKey = keyProvider.apply(null, ePackage);
273                if (packageKey != null) {
274                        registry.put(packageKey + "-", new EPackageEntry(ePackage, keyProvider == null ? this.keyProvider : keyProvider));
275                }
276        }

294        public Object create(ObjectLoader loader, String type, Object config, java.util.function.Function<EClass, EObject> constructor, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) throws Exception {
295                BiSupplier<EClass,BiFunction<EClass,ENamedElement,String>> result = resolveEClass(type);
296                if (result != null) {
297                        return create(loader, result.getFirst(), config, base, progressMonitor, markers, result.getSecond(), constructor);
298                }
299                
300                return super.create(loader, type, config, base, progressMonitor, markers);
301        }

311        /**
312         * Resolves EClass and key provider from type name.
313         * @param type
314         * @return
315         */
316        public BiSupplier<EClass,BiFunction<EClass,ENamedElement,String>> resolveEClass(String type) {          
317                for (Entry<String, EPackageEntry> re: registry.entrySet()) {
318                        if (type.startsWith(re.getKey())) {
319                                EPackageEntry ePackageEntry = re.getValue();
320                                for (EClassifier eClassifier: ePackageEntry.ePackage.getEClassifiers()) {
321                                        if (eClassifier instanceof EClass && "true".equals(NcoreUtil.getNasdanikaAnnotationDetail(eClassifier, IS_LOADABLE, "true"))) {
322                                                String eClassifierKey = ePackageEntry.keyProvider.apply(null, eClassifier);
323                                                if (eClassifierKey != null && type.equals(re.getKey() + eClassifierKey)) {
324                                                        return new BiSupplier<EClass, BiFunction<EClass,ENamedElement,String>>() {
325
326                                                                @Override
327                                                                public EClass getFirst() {
328                                                                        return (EClass) eClassifier;
329                                                                }
330
331                                                                @Override
332                                                                public BiFunction<EClass,ENamedElement, String> getSecond() {
333                                                                        return ePackageEntry.keyProvider;
334                                                                }
335                                                        }; 
336                                                }
337                                        }
338                                }
339                        }
340                }
341                return null;            
342        }

344        /**
345         * Creates an instance of EClass or a supplier factory
346         * @param loader
347         * @param eClass
348         * @param config
349         * @param base
350         * @param progressMonitor
351         * @param marker
352         * @param ePackageEntry
353         * @param eClassifier
354         * @return
355         * @throws Exception
356         */
357        public Object create(
358                        ObjectLoader loader, 
359                        EClass eClass, 
360                        Object config, 
361                        URI base, 
362                        ProgressMonitor progressMonitor, 
363                        List<? extends Marker> markers, 
364                        BiFunction<EClass,ENamedElement,String> keyProvider,
365                        java.util.function.Function<EClass, EObject> constructor) throws Exception {
366                
367                // Nulls are nulls
368                if (config == null) {
369                        return null;
370                }
371                
372                // Proxy
373                Object proxy = createProxy(eClass, config, base, markers, progressMonitor);
374                if (proxy != null) {
375                        return proxy;
376                }
377                
378                Class<?> instanceClass = eClass.getInstanceClass();
379                EFactory factory = eClass.getEPackage().getEFactoryInstance();
380                
381                if (Loadable.class.isAssignableFrom(instanceClass)) {
382                        EObject eObject = factory.create(eClass);
383                        ((Loadable) eObject).load(loader, config, base, progressMonitor, markers);
384                        return eObject;
385                }
386                if (ContextLoadable.class.isAssignableFrom(instanceClass)) {
387                        return new SupplierFactory<EObject>() {
388
389                                @Override
390                                public Supplier<EObject> create(Context context) throws Exception {
391                                        return new Supplier<EObject>() {
392
393                                                @Override
394                                                public double size() {
395                                                        return 1;
396                                                }
397
398                                                @Override
399                                                public String name() {
400                                                        return "Creating " + eClass.getName();
401                                                }
402
403                                                @Override
404                                                public EObject execute(ProgressMonitor pm) throws Exception {
405                                                        EObject eObject = factory.create(eClass);
406                                                        ((ContextLoadable) eObject).load(context, loader, config, base, pm, markers);
407                                                        return eObject;
408                                                }
409                                                
410                                        };
411                                }                                                                       
412                        };
413                }                                                       
414                EObjectSupplierFactory eObjectSupplierFactory = createEObjectSupplierFactory(eClass, keyProvider, constructor);
415                eObjectSupplierFactory.load(loader, config, base, progressMonitor, markers);
416                return eObjectSupplierFactory;
417        }

432        /**
433         * Creates a proxy object if config is a singleton map with "href" key.
434         * @param eClass
435         * @param config
436         * @param base
437         * @return Proxy object or null if config is not a proxy config.
438         */
439        public EObject createProxy(EClass eClass, Object config, URI base, List<? extends Marker> markers, ProgressMonitor progressMonitor) {
440                if (config instanceof Map) {
441                        @SuppressWarnings("unchecked")
442                        Map<Object,Object> configMap = (Map<Object,Object>) config;
443                        if (configMap.size() == 1 
444                                        && configMap.containsKey(HREF_KEY)) {
445                                EObject eObject = eClass.getEPackage().getEFactoryInstance().create(eClass);
446                                if (eObject instanceof MinimalEObjectImpl) {
447                                        Object href = configMap.get(HREF_KEY);                                  
448                                        URI proxyURI;
449                                        if (href instanceof URI) {
450                                                proxyURI = (URI) href;
451                                        } else if (href instanceof String) {
452                                                proxyURI = URI.createURI((String) configMap.get(HREF_KEY));
453                                        } else {
454                                                return null;
455                                        }
456                                        
457                                        String proxyURIStr = proxyURI.toString();
458                                        if (proxyURIStr.startsWith(LATE_PROXY_RESOLUTION_URI_PREFIX)) {
459                                                proxyURI = URI.createURI(proxyURIStr.substring(LATE_PROXY_RESOLUTION_URI_PREFIX.length()));
460                                        } else if (base != null) {
461                                                proxyURI = proxyURI.resolve(base);
462                                        }
463                                        ((MinimalEObjectImpl) eObject).eSetProxyURI(proxyURI);
464                                        mark(eObject, markers, progressMonitor);
465                                        
466                                        return eObject;
467                                }
468                        }
469                }
470                return null;
471        }

473        /**
474         * If marker is not null adds marked adapter. If eObject is instance of {@link Marked} creates and sets {@link org.nasdanika.ncore.Marker}.
475         * @param eObject
476         * @param marker
477         */
478        public void mark(EObject eObject, List<? extends Marker> markers, ProgressMonitor progressMonitor) {
479                if (markers != null) {
480                        eObject.eAdapters().add(new MarkedAdapter(markers));
481                        if (eObject instanceof Marked) {                
482                                for (Marker marker: markers) {
483                                        org.nasdanika.ncore.Marker mMarker = markerFactory.createMarker(marker.getLocation(), progressMonitor);
484                                        mMarker.setLine(marker.getLine());
485                                        mMarker.setColumn(marker.getColumn());
486                                        ((Marked) eObject).getMarkers().add(0, mMarker);
487                                }
488                        }
489                }
490        }

492        /**
493         * Resolves proxy using the loader's resource set.
494         * Does not suppress exceptions. 
495         */
496        public EObject resolve(EObject proxy) {
497                if (!proxy.eIsProxy()) {
498                        return proxy;
499                }
500
501                URI proxyURI = ((InternalEObject) proxy).eProxyURI();
502                EObject ret = resourceSet.getEObject(proxyURI, true);
503                return ret == null || ret == proxy ? ret : resolve(ret);
504        }

510        /**
511         * @return Load key from <code>load-keys</code> annotation. 
512         */
513        public static String getLoadKey(EClass eClass, EStructuralFeature feature) {
514                for (EClass ec: NcoreUtil.lineage(eClass)) {
515                        String loadKeys = NcoreUtil.getNasdanikaAnnotationDetail(ec, LOAD_KEYS);
516                        if (!Util.isBlank(loadKeys)) {
517                                Yaml yaml = new Yaml();
518                                Map<String,Object> loadKeyMap = yaml.load(loadKeys);
519                                Object loadKey = loadKeyMap.get(feature.getName());
520                                if (loadKey instanceof String) {
521                                        return (String) loadKey;
522                                }
523                        }
524                }
525                return null;
526        }
