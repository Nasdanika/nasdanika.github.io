001package org.nasdanika.emf;
002
003import org.eclipse.emf.ecore.EObject;
004import org.nasdanika.common.CommandFactory;
005
006import picocli.CommandLine.Command;
007
008@Command(
009                description = "Loads a model, adapts to CommandFactory, then creates and executes a command",
010                name = "execute-model")
011public class ExecuteModelCommand extends AdapterModelCommand<EObject,CommandFactory> {
012
013        @Override
014        protected Class<CommandFactory> getAdapterType() {
015                return CommandFactory.class;
016        }
017
018        @Override
019        protected CommandFactory createCommandFactory(CommandFactory adapter) {
020                return adapter;
021        }
022        
023
024}
