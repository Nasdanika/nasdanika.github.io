001package org.nasdanika.emf;
002
003import java.util.ArrayList;
004import java.util.Collection;
005import java.util.HashSet;
006import java.util.List;
007import java.util.Map;
008import java.util.Set;
009import java.util.concurrent.ConcurrentHashMap;
010import java.util.stream.Collectors;
011
012import org.eclipse.emf.common.notify.Adapter;
013import org.eclipse.emf.common.notify.AdapterFactory;
014import org.eclipse.emf.common.notify.Notifier;
015import org.eclipse.emf.ecore.EClass;
016import org.eclipse.emf.ecore.EObject;
017import org.eclipse.emf.ecore.EcorePackage;
018
019/**
020 * Composed factory delegates to its child factories. It uses inheritance ordering to delegate -
021 * delegates to factories registered for more specific classes first. If there are several factories
022 * for the same class then they are compared using factory's compare method if it implements comparable or by inheritance - more specific first.
023 * @author Pavel Vlasov
024 *
025 */
026public class ComposedAdapterFactory implements ComposeableAdapterFactory {
027        
028        private ComposedAdapterFactory parentAdapterFactory;
029        
030        private Map<Object,Boolean> factoryForTypeCache = new ConcurrentHashMap<>();
031
032        @Override
033        public boolean isFactoryForType(Object type) {
034                return factoryForTypeCache.computeIfAbsent(type, this::_isFactoryForType);
035        }
036        
037        private boolean _isFactoryForType(Object type) {
038                for (AdapterFactoryEntry c: children) {
039                        if (c.adapterFactory.isFactoryForType(type)) {
040                                return true;
041                        }
042                }
043                return false;
044        }
045
046        @Override
047        public Object adapt(Object object, Object type) {
048                for (AdapterFactory af: match(object, type)) {
049                        Object ret = af.adapt(object, type);
050                        if (ret != null) {
051                                return ret;
052                        }
053                }
054                return null;
055        }
056
057        @Override
058        public Adapter adapt(Notifier target, Object type) {
059                for (AdapterFactory af: match(target, type)) {
060                        Adapter ret = af.adapt(target, type);
061                        if (ret != null) {
062                                return ret;
063                        }
064                }
065                return null;
066        }
067
068        @Override
069        public Adapter adaptNew(Notifier target, Object type) {
070                for (AdapterFactory af: match(target, type)) {
071                        Adapter ret = af.adaptNew(target, type);
072                        if (ret != null) {
073                                return ret;
074                        }
075                }
076                return null;
077        }
078
079        @Override
080        public void adaptAllNew(Notifier notifier) {
081                for (AdapterFactory af: match(notifier, null)) {
082                        af.adaptAllNew(notifier);
083                }
084        }
085
086        @Override
087        public ComposeableAdapterFactory getRootAdapterFactory() {
088                return parentAdapterFactory == null ? this : parentAdapterFactory.getRootAdapterFactory();
089        }
090
091        @Override
092        public void setParentAdapterFactory(ComposedAdapterFactory parentAdapterFactory) {
093                this.parentAdapterFactory = parentAdapterFactory;               
094        }
095        
096        protected List<AdapterFactory> match(Object obj, Object type) {
097                return children
098                        .stream()
099                        .filter(ae -> ae.match(obj) && (type == null || ae.adapterFactory.isFactoryForType(type)))
100                        .sorted((a,b) -> cmpDistance((EObject) obj, a.eClass, b.eClass))
101                        .map(ae -> ae.adapterFactory)
102                        .collect(Collectors.toList());
103        }
104        
105        protected int cmpDistance(EObject obj, EClass a, EClass b) {
106                if (a == b) {
107                        return 0;
108                }
109                if (a.equals(b)) {
110                        return 0;
111                }
112                if (a.isSuperTypeOf(b)) {
113                        return 1;
114                }
115                if (b.isSuperTypeOf(a)) {
116                        return -1;
117                }
118                if (obj == null) {
119                        return 0; // Does not matter.
120                }
121                EClass eClass = obj.eClass();
122                if (eClass.equals(a)) {
123                        return -1;
124                }
125                if (eClass.equals(b)) {
126                        return 1;
127                }
128                return distance(eClass, a) - distance(eClass, b);
129        }
130        
131        protected int distance(EClass sub, EClass sup) {
132                if (sub.equals(sup)) {
133                        return 0;
134                }
135                int ret = Integer.MAX_VALUE;
136                for (EClass isup: sub.getESuperTypes()) {
137                        if (isup.equals(sup)) {
138                                return 1;
139                        }
140                        if (sup.isSuperTypeOf(isup)) {
141                                ret = Math.min(ret, distance(isup, sup)); 
142                        }
143                }
144                return ret;
145        }
146                
147        private class AdapterFactoryEntry {
148                
149                EClass eClass;
150                AdapterFactory adapterFactory;
151                
152                AdapterFactoryEntry(EClass eClass, AdapterFactory adapterFactory) {
153                        this.eClass = eClass == null ? EcorePackage.Literals.EOBJECT : eClass;
154                        this.adapterFactory = adapterFactory;
155                        if (adapterFactory instanceof ComposeableAdapterFactory) {
156                                ((ComposeableAdapterFactory) adapterFactory).setParentAdapterFactory(ComposedAdapterFactory.this);
157                        }
158                }
159                
160                boolean match(Object obj) {
161                        return obj == null || (obj instanceof EObject && (eClass == null || eClass.isInstance(obj)));
162                }                               
163                
164        }
165        
166        private List<AdapterFactoryEntry> children = new ArrayList<>();
167        
168        public void registerAdapterFactory(ComposeableAdapterFactory child) {
169                for (EClass eClass: child.getEClasses()) {
170                        children.add(new AdapterFactoryEntry(eClass, child));                   
171                }
172        }
173
174        /**
175         * Registers a factory for specified {@link EClass}es in the package.
176         * If no EClasses are specified then the adapter factory is considered to be
177         * applicable to all classes.
178         * @param child
179         * @param eClasses
180         */
181        public void registerAdapterFactory(AdapterFactory child, EClass... eClasses) {
182                if (child instanceof ComposedAdapterFactory) {
183                        throw new IllegalArgumentException("Composeable adapter factories shall be registered with registerAdapterFactory(ComposeableAdapterFactory child) method to avoid ambiguity in eClasses");
184                }
185                if (eClasses.length == 0) {
186                        children.add(new AdapterFactoryEntry(null, child));
187                } else for (EClass eClass: eClasses) {
188                        children.add(new AdapterFactoryEntry(eClass, child));                   
189                }
190                factoryForTypeCache.clear();
191        }
192
193        @Override
194        public Collection<EClass> getEClasses() {
195                Set<EClass> ret = new HashSet<>();
196                for (AdapterFactoryEntry child: children) {
197                        ret.add(child.eClass);
198                }
199                return ret;
200        }
201}
