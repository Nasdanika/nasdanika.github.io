001package org.nasdanika.emf.persistence;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.io.OutputStream;
006import java.io.OutputStreamWriter;
007import java.io.Writer;
008import java.net.URL;
009import java.nio.charset.StandardCharsets;
010import java.util.Collection;
011import java.util.Map;
012
013import org.eclipse.emf.common.util.URI;
014import org.eclipse.emf.ecore.EObject;
015import org.eclipse.emf.ecore.resource.impl.ResourceImpl;
016import org.json.JSONArray;
017import org.nasdanika.common.Context;
018import org.nasdanika.common.NasdanikaException;
019import org.nasdanika.common.ProgressMonitor;
020import org.nasdanika.common.SupplierFactory;
021import org.nasdanika.common.Util;
022import org.nasdanika.common.persistence.ObjectLoader;
023import org.nasdanika.common.persistence.Storable;
024import org.nasdanika.emf.EObjectAdaptable;
025
026/**
027 * Loads EMF classes from JSON arrays using {@link ObjectLoader}s. 
028 * Saving is supported if the resource content implements or is adaptable to {@link Storable}. 
029 * The idea of this resource is to load models from manually created JSON files or JSON files generated 
030 * by some other means, e.g. from MongoDB or REST calls. 
031 * @author Pavel
032 *
033 */
034public class JsonResource extends ResourceImpl {
035        
036        private ObjectLoader loader;
037        private ProgressMonitor progressMonitor;
038        private Context context;
039
040        public JsonResource(URI uri, ObjectLoader loader, Context context, ProgressMonitor progressMonitor) {
041                super(uri);
042                this.loader = loader;
043                this.context = context;
044                this.progressMonitor = progressMonitor;
045        }
046
047        @SuppressWarnings("unchecked")
048        @Override
049        protected void doLoad(InputStream inputStream, Map<?, ?> options) throws IOException {
050                try {
051                        Object data = loader.loadJsonArray(inputStream, getURI(), progressMonitor);
052                        if (data instanceof Collection) {
053                                getContents().addAll((Collection<EObject>) data);
054                        } else {
055                                if (data instanceof SupplierFactory) {
056                                        EObject eObject = Util.call(((SupplierFactory<EObject>) data).create(context), progressMonitor, null);
057                                        getContents().add(eObject);
058                                } else if (data instanceof EObject) {
059                                        getContents().add((EObject) data);
060                                } else {
061                                        throw new IOException("Not an instance of EObject: " + data);
062                                }
063                        }
064                } catch (RuntimeException | IOException e) {
065                        throw e;
066                } catch (Exception e) {
067                        throw new NasdanikaException(e);
068                }
069        }
070        
071        @Override
072        protected void doSave(OutputStream outputStream, Map<?, ?> options) throws IOException {
073                JSONArray data = new JSONArray();
074                for (EObject e: getContents()) {
075                        Storable storable = EObjectAdaptable.adaptTo(e, Storable.class);
076                        if (storable == null) {
077                                throw new IOException("Cannot adapt " + e + " to " + Storable.class.getName());
078                        }
079                        try {
080                                data.put(storable.store(new URL(getURI().toString()), progressMonitor));
081                        } catch (RuntimeException | IOException ex) {
082                                throw ex;
083                        } catch (Exception ex) {
084                                throw new NasdanikaException(ex);
085                        }
086                }
087                
088                try (Writer writer = new OutputStreamWriter(outputStream, StandardCharsets.UTF_8)) {
089                        data.write(writer);
090                }
091        }
092
093        /**
094         * If fragment is null returns the first content element.
095         */
096        @Override
097        public EObject getEObject(String uriFragment) {
098                if (uriFragment == null) {
099                        return getContents().get(0);
100                }
101                return super.getEObject(uriFragment);
102        }
103
104}
