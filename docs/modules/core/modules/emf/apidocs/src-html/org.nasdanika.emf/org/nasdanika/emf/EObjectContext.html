001package org.nasdanika.emf;
002
003import org.eclipse.emf.ecore.EObject;
004import org.eclipse.emf.ecore.EStructuralFeature;
005import org.eclipse.emf.ecore.util.EcoreUtil;
006import org.nasdanika.common.Context;
007
008/**
009 * Wraps {@link EObject} into a context.
010 * @author Pavel
011 *
012 */
013public class EObjectContext implements Context {
014
015        private EObject target;
016
017        public EObjectContext(EObject target) {
018                this.target = target;
019        }
020
021        @Override
022        public Object get(String key) {
023                int sIdx = key.indexOf('/');
024                String featureName = sIdx == -1 ? key : key.substring(0, sIdx);
025                EStructuralFeature feature = target.eClass().getEStructuralFeature(featureName);
026                if (feature == null) {
027                        return null;
028                }
029                Object val = target.eGet(feature);
030                if (val == null || sIdx == -1) {
031                        return val;
032                }
033                if (val instanceof EObject) {
034                        return new EObjectContext((EObject) val).get(key.substring(sIdx + 1));
035                }
036                return null;
037        }
038
039        @SuppressWarnings("unchecked")
040        @Override
041        public <T> T get(Class<T> type) {
042                if (target == null) {
043                        return null;
044                }
045                if (type.isInstance(target)) {
046                        return (T) target;
047                }
048                return (T) EcoreUtil.getRegisteredAdapter(target, type);
049        }
050
051}
