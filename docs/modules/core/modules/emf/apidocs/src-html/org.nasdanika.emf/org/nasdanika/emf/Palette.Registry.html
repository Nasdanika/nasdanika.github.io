003import java.util.ArrayList;
004import java.util.Collection;
005import java.util.HashSet;
006import java.util.List;
007import java.util.Map;
008import java.util.Set;
009import java.util.concurrent.ConcurrentHashMap;
010import java.util.function.Supplier;
011
012import org.eclipse.emf.ecore.EClass;
013import org.eclipse.emf.ecore.EObject;
014import org.eclipse.emf.ecore.EReference;
015import org.eclipse.emf.ecore.util.EcoreUtil;

017/**
018 * Palettes are used for grouping of EObjects, e.g. child candidates for containment references.
019 * Palettes allow to manage reference children in groups and also to have pre-configured children, e.g. objects with pre-populated structural features.
020 * Palette extends {@link EReferencePredicate} so it can be used by control flow (if, for, ...) and reference model elements to filter candidates for a reference. 
021 * 
022 * @author Pavel
023 *
024 */
025public interface Palette extends EReferencePredicate {
026        
027        interface Registry {
028                
029                Palette get(String id);
030                
031                Collection<Palette> getPalettes();
032                
033                Palette create(String id, String name, String description, boolean isWizard);           
034                
035                /**
036                 * Returns a de-dupped list of elements from requested palettes.
037                 * @param palettes
038                 * @return
039                 */
040                default List<EObject> getElements(String... palettes) {
041                        Set<String> ids = new HashSet<>();                      
042                        List<EObject> ret = new ArrayList<>();
043                        for (String pid: palettes) {
044                                Palette palette = get(pid);
045                                if (palette != null) {
046                                        for (Contributor contributor: palette.getContributors()) {
047                                                if (ids.add(contributor.getId())) {
048                                                        ret.add(contributor.get());
049                                                }
050                                        }
051                                }
052                        }
053                        return ret;
054                }
055                
056                static Registry INSTANCE = new Registry() {
057                        
058                        private Map<String,Palette> palettes = new ConcurrentHashMap<>();
059                        
060                        @Override
061                        public Palette get(String id) {
062                                return create(id, null, null, true);
063                        }
064                        
065                        @Override
066                        public Palette create(String id, String name, String description, boolean isWizard) {
067                                return palettes.computeIfAbsent(id, pid -> new Palette() {
068                                        
069                                        private List<Contributor> contributors = new ArrayList<>();
070                                        private List<EReferencePredicate> eReferencePredicates = new ArrayList<>();
071
072                                        @Override
073                                        synchronized public List<Contributor> getContributors() {
074                                                return new ArrayList<Contributor>(contributors);
075                                        }
076
077                                        @Override
078                                        synchronized public void add(Contributor contributor) {
079                                                contributors.add(contributor);                                          
080                                        }
081
082                                        @Override
083                                        public String getId() {
084                                                return id;
085                                        }
086
087                                        @Override
088                                        public String getName() {
089                                                return name;
090                                        }
091
092                                        @Override
093                                        public String getDescription() {
094                                                return description;
095                                        }
096
097                                        @Override
098                                        synchronized public boolean accept(EObject source, EReference eReference, EObject target, boolean direct) {
099                                                for (EReferencePredicate p: eReferencePredicates) {
100                                                        if (p.accept(source, eReference, target, direct)) {
101                                                                return true;
102                                                        }
103                                                }
104                                                return false;
105                                        }
106
107                                        @Override
108                                        synchronized public void add(EReferencePredicate predicate) {
109                                                eReferencePredicates.add(predicate);
110                                        }
111
112                                        @Override
113                                        public boolean isWizard() {
114                                                return isWizard;
115                                        }
116                                        
117                                });
118                        }
119
120                        @Override
121                        public Collection<Palette> getPalettes() {
122                                return palettes.values();
123                        }
124                        
125                };
126                
127        }
128        
129        /**
130         * Palette id.
131         * @return
132         */
133        String getId();
134        
135        interface Contributor extends Supplier<EObject> {
136                
137                /**
138                 * Contributor ID used for de-duplication.
139                 * @return
140                 */
141                String getId();
142                
143        }
144        
145        /**
146         * @return A copy of the internal list of contributors.
147         */
148        List<Contributor> getContributors();
149        
150        void add(Contributor contributor);
151                        
152        default void add(String id, Supplier<EObject> supplier) {
153                add(new Contributor() {
154
155                        @Override
156                        public EObject get() {
157                                return supplier.get();
158                        }
159
160                        @Override
161                        public String getId() {
162                                return id;
163                        }
164                        
165                });
166        }
167        
168        /**
169         * @return Palette elements.
170         */
171        default List<EObject> getElements() {
172                Set<String> ids = new HashSet<>();                      
173                List<EObject> ret = new ArrayList<>();
174                for (Contributor contributor: getContributors()) {
175                        if (ids.add(contributor.getId())) {
176                                ret.add(contributor.get());
177                        }
178                }
179                return ret;
180        }
181                
182        /**
183         * @return Palette elements filtered for a given source object and containing reference.
184         */
185        default List<EObject> getElements(EObject source, EReference eReference) {
186                Set<String> ids = new HashSet<>();                      
187                List<EObject> ret = new ArrayList<>();
188                for (Contributor contributor: getContributors()) {
189                        if (ids.add(contributor.getId())) {
190                                EObject element = contributor.get();
191                                if (!(contributor instanceof EReferencePredicate) || ((EReferencePredicate) contributor).accept(source, eReference, element, true)) {
192                                        ret.add(element);
193                                }
194                        }
195                }
196                return ret;
197        }
198                
199        /**
200         * Adds a contributor which instantiates given EClass and uses EClass name and EPackage namespace URI as ID.
201         * The contributor also implements {@link EReferencePredicate} delegating to the predicate parameter. It allows
202         * to filter palette elements based on the container and containing reference.
203         * Also adds predicates checking that a candidate is instance of one of provided EClass'es. 
204         * @param eClass
205         */
206        default void add(EClass eClass, EReferencePredicate predicate) {
207                abstract class ContributorPredicate implements Contributor, EReferencePredicate { }
208                Contributor contributor = new ContributorPredicate() {
209                        
210                        @Override
211                        public EObject get() {
212                                return eClass.getEPackage().getEFactoryInstance().create(eClass);
213                        }
214                        
215                        @Override
216                        public String getId() {
217                                return eClass.getName()+"@"+eClass.getEPackage().getNsURI();
218                        }
219
220                        @Override
221                        public boolean accept(EObject source, EReference eReference, EObject target, boolean direct) {
222                                return predicate == null ? true : predicate.accept(source, eReference, target, direct);
223                        }
224                        
225                };
226                add((Contributor) contributor);
227                add(new EReferencePredicate() {
228
229                        @Override
230                        public boolean accept(EObject source, EReference eReference, EObject target, boolean direct) {
231                                return eClass.isInstance(target);
232                        }
233                        
234                });
235        };      
236                
237        /**
238         * Adds a contributor which instantiates given EClass and uses EClass name and EPackage namespace URI as ID.
239         * Also adds predicates checking that a candidate is instance of one of provided EClass'es. 
240         * @param eClass
241         */
242        default void add(EClass... eClass) {
243                for (EClass ec: eClass) {
244                        add(new Contributor() {
245                                
246                                @Override
247                                public EObject get() {
248                                        return ec.getEPackage().getEFactoryInstance().create(ec);
249                                }
250                                
251                                @Override
252                                public String getId() {
253                                        return ec.getName()+"@"+ec.getEPackage().getNsURI();
254                                }
255                                
256                        });
257                        add(new EReferencePredicate() {
258
259                                @Override
260                                public boolean accept(EObject source, EReference eReference, EObject target, boolean direct) {
261                                        return ec.isInstance(target);
262                                }
263                                
264                        });
265                }
266        };
267
268        /**
269         * Adds a contributor which creates a copy of a given EObject.
270         * @param eClass
271         * @param id
272         */
273        default void add(String id, EObject eObject) {
274                add(new Contributor() {
275
276                        @Override
277                        public EObject get() {
278                                return EcoreUtil.copy(eObject);
279                        }
280
281                        @Override
282                        public String getId() {
283                                return id;
284                        }
285                        
286                });
287        }
288        
289        /**
290         * Adds a predicate to the palette. The predicates are OR'd.
291         * @param predicate
292         */
293        void add(EReferencePredicate predicate);
294                
295        String getName();
296        
297        String getDescription();
298        
299        /**
300         * @return true if the palette shall be used by the Nasdanika Model Wizard.
301         */
302        boolean isWizard();
303        
304}
