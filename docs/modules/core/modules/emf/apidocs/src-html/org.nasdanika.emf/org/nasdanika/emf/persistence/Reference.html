001package org.nasdanika.emf.persistence;
002
003import java.util.List;
004import java.util.function.BiFunction;
005
006import org.eclipse.emf.common.util.URI;
007import org.eclipse.emf.ecore.EClass;
008import org.eclipse.emf.ecore.ENamedElement;
009import org.eclipse.emf.ecore.EReference;
010import org.nasdanika.common.NasdanikaException;
011import org.nasdanika.common.ProgressMonitor;
012import org.nasdanika.common.persistence.Attribute;
013import org.nasdanika.common.persistence.Marker;
014import org.nasdanika.common.persistence.ObjectLoader;
015
016/**
017 * Single value reference.
018 * @author Pavel
019 *
020 * @param <T>
021 */
022public class Reference<T> extends Attribute<T> {
023        
024        private ReferenceFactory referenceFactory;
025        
026        /**
027         * 
028         * @param key
029         * @param isDefault
030         * @param required
031         * @param defaultValue
032         * @param description
033         * @param resolver
034         * @param referenceType Reference type if the reference is homogenous, i.e. its type is known beforehand.
035         * @param exclusiveWith
036         */
037        public Reference(
038                        Object key, 
039                        boolean isDefault, 
040                        boolean required, 
041                        T defaultValue, 
042                        String description, 
043                        EClass eClass,
044                        EReference eReference,
045                        EObjectLoader resolver,
046                        boolean referenceSupplierFactory,
047                        BiFunction<EClass,ENamedElement,String> keyProvider,
048                        Object... exclusiveWith) {
049                super(key, isDefault, required, defaultValue, description, exclusiveWith);
050                this.referenceFactory = new ReferenceFactory(eClass, eReference, null, resolver, referenceSupplierFactory, keyProvider);
051        }
052        
053        @SuppressWarnings("unchecked")
054        @Override
055        public T create(ObjectLoader loader, Object element, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) throws Exception {
056                List<?> result = referenceFactory.create(loader, element, base, progressMonitor, markers);
057                if (result == null || result.isEmpty()) {
058                        return null;
059                }
060                if (result.size() == 1) {
061                        return (T) result.get(0);
062                }
063                throw new NasdanikaException("Expected result size of 0 or 1, got " + result.size() + ":" + result);
064        }
065
066}
