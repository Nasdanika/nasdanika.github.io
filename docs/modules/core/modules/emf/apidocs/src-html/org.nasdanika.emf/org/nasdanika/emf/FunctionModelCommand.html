001package org.nasdanika.emf;
002
003import org.eclipse.emf.ecore.EObject;
004import org.nasdanika.common.CommandFactory;
005import org.nasdanika.common.ConsumerFactory;
006import org.nasdanika.common.FunctionFactory;
007import org.nasdanika.common.SupplierFactory;
008
009/**
010 * Base class for commands which adapt models to {@link FunctionFactory} and then combine input supplier with function and output consumer to execute.
011 * @author Pavel
012 *
013 */
014public abstract class FunctionModelCommand<T,R> extends AdapterModelCommand<EObject,FunctionFactory<T,R>> {
015
016        @SuppressWarnings({ "unchecked", "rawtypes" })
017        @Override
018        protected Class<FunctionFactory<T,R>> getAdapterType() {
019                return (Class) FunctionFactory.class;
020        }
021
022        @Override
023        protected CommandFactory createCommandFactory(FunctionFactory<T,R> adapter) {
024                return createInputSupplierFactory().then(adapter).then(createOutputConsumerFactory());
025        }
026        
027        /**
028         * @return Factory for input supplier to combine with function and output consumer to create a command.
029         */
030        protected abstract SupplierFactory<T> createInputSupplierFactory();
031        
032        /**
033         * @return Factory for output consumer to combine with input supplier and function to create a command.
034         */
035        protected abstract ConsumerFactory<? super R> createOutputConsumerFactory();
036
037}
