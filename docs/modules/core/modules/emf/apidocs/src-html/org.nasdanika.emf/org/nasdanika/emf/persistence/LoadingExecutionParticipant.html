040/**
041 * {@link ExecutionParticipant} which loads resources using {@link EObjectLoader}, diagnoses them
042 * and stores in a {@link ResourceSet} for use in execute() methods of subclasses.
043 * 
044 * @author Pavel
045 *
046 */
047public abstract class LoadingExecutionParticipant implements ExecutionParticipant {
048        
049        protected Context context;
050        protected List<EObject> roots;
051        protected ResourceSet resourceSet;
052        protected Map<EObject, org.eclipse.emf.common.util.Diagnostic> diagnosticMap = new LinkedHashMap<>();
053//      protected ClassLoader classLoader;
054        
055        public LoadingExecutionParticipant(Context context) {
056                this.context = context;
057        }
058        
059        @Override
060        public double size() {
061                return 1;
062        }
063
064        @Override
065        public String name() {
066                return "Loading resources";
067        }
068        
069        protected boolean matchURI(EObject obj, URI uri) {
070                return false;
071        }
072        
073        protected abstract Collection<URI> getResources();
074        
075        protected ResourceSet createResourceSet(ProgressMonitor progressMonitor) {
076                ResourceSet ret = new NcoreResourceSet() {
077
078                        @Override
079                        protected boolean matchURI(EObject eObj, URI uri) {
080                                return super.matchURI(eObj, uri) || LoadingExecutionParticipant.this.matchURI(eObj, uri);
081                        }
082                        
083                };
084
085                // XMI as default.
086                ret.getResourceFactoryRegistry().getExtensionToFactoryMap().put(Resource.Factory.Registry.DEFAULT_EXTENSION, new XMIResourceFactoryImpl());
087                
088                ret.getURIConverter().getURIHandlers().add(0, new URIHandlerImpl() {
089
090                        @Override
091                        public boolean canHandle(URI uri) {
092                                return uri != null && Util.CLASSPATH_SCHEME.equals(uri.scheme());
093                        }
094
095                        @Override
096                        public InputStream createInputStream(URI uri, Map<?, ?> options) throws IOException {
097                                return DefaultConverter.INSTANCE.toInputStream(uri);
098                        }
099                        
100                });
101                
102                Registry packageRegistry = ret.getPackageRegistry();
103                for (EPackage ePackage: getEPackages()) {
104                        packageRegistry.put(ePackage.getNsURI(), ePackage);
105                }               
106                
107                return ret;
108        }
109        
110        protected abstract Collection<EPackage> getEPackages();
111        
112        /**
113         * Diagnoses the {@link ResourceSet}. This implementation finds unresolved proxies and reports them
114         * with {@link Status} FAIL by calling unresolvedProxyDiagnostic().
115         * @param resourceSet
116         * @return
117         */
118        @SuppressWarnings("unchecked")
119        protected BasicDiagnostic diagnose() {
120                BasicDiagnostic ret = new BasicDiagnostic(Status.SUCCESS, "Diagnostic of " + resourceSet, resourceSet);
121                
122                TreeIterator<Notifier> cit = resourceSet.getAllContents();
123                while (cit.hasNext()) {
124                        Notifier next = cit.next();
125                        if (next instanceof EObject) {
126                                EObject nextEObject = (EObject) next;
127                                if (nextEObject.eIsProxy()) {                                                   
128                                        ret.add(unresolvedProxyDiagnostic(nextEObject, null, null));
129                                } else {
130                                        for (EReference ref: nextEObject.eClass().getEAllReferences()) {
131                                                Object val = nextEObject.eGet(ref);
132                                                if (val instanceof EObject) {
133                                                        if (((EObject) val).eIsProxy()) {
134                                                                ret.add(unresolvedProxyDiagnostic((EObject) val, ref, nextEObject));
135                                                        }
136                                                } else if (val instanceof Collection) {
137                                                        for (EObject ve: (Collection<EObject>) val) {
138                                                                if (ve.eIsProxy()) {
139                                                                        ret.add(unresolvedProxyDiagnostic((EObject) ve, ref, nextEObject));
140                                                                }                                                               
141                                                        }
142                                                }
143                                        }
144                                }
145                        }       
146                }
147                return ret;
148        }
149        
150        /**
151         * Reports unresolved proxies
152         * @param source Unresolved proxy
153         * @param containmentReference Reference containing the proxy. Can be null.
154         * @param container Container of unresolved proxy. Can be null. 
155         * @return
156         */
157        protected Diagnostic unresolvedProxyDiagnostic(EObject source, EReference containmentReference, EObject container) {
158                Marked marked = EObjectAdaptable.adaptTo(source, Marked.class);
159                List<? extends Marker> markers = marked == null ? null : marked.getMarkers();
160                if (markers == null || markers.isEmpty()) {
161                        return new BasicDiagnostic(getUnresolvedProxyStatus(), "Unresolved proxy: " + source, source, containmentReference, container);
162                }
163        
164                return new BasicDiagnostic(getUnresolvedProxyStatus(), "Unresolved proxy at " + markers.stream().map(Object::toString).collect(Collectors.joining(", ")) + ": " + source, source, markers, containmentReference, container);            
165        }
166
167        /**
168         * Override if unresolved proxies are allowed in loaded resources, e.g. they'd be resolvable at some point after the model is loaded. 
169         * This method is called only if isDiagnose() returns true (default). If the diagnostic would result in errors, e.g. due to unresolved
170         * proxies in opposites, then override isDiagnoseModel() instead to return false and then explicitly diagnose the model which is fully loaded and as 
171         * such all proxies are resolvable. 
172         * @return Diagnostic status for unresolved proxies. This method returns FAIL which results in exception during load.
173         */
174        protected Status getUnresolvedProxyStatus() {
175                return Status.FAIL;
176        }
177
178        /**
179         * Loads resources, checks for unresolved proxies and diagnoses.
180         */
181        @Override
182        public Diagnostic diagnose(ProgressMonitor progressMonitor) {
183                resourceSet = createResourceSet(progressMonitor);
184                
185                // Pre-loading
186                for (URI uri: getResources()) {
187                        resourceSet.getResource(uri, true);
188                }
189                
190                // Resolving all proxies and clearing all caches
191                EcoreUtil.resolveAll(resourceSet);
192                resourceSet.getAllContents().forEachRemaining(notifier -> notifier.eNotify(FeatureCacheAdapter.CLEAR_CACHE));
193
194                if (isDiagnoseModel()) {
195                        BasicDiagnostic ret = diagnose();
196                        
197                        Diagnostician diagnostician = new Diagnostician();
198                        if (ret.getStatus() != Status.FAIL) {                           
199                                roots = new ArrayList<>();
200                                Map<Class<Context>, Context> diagnosticContext = Collections.singletonMap(Context.class, context);
201                                for (Resource resource: resourceSet.getResources()) {
202                                        for (EObject e: resource.getContents()) {
203                                                org.eclipse.emf.common.util.Diagnostic diagnostic = diagnostician.validate(e, diagnosticContext);
204                                                diagnosticMap.put(e, diagnostic);
205                                                ret.add(EmfUtil.wrap(diagnostic));
206                                                roots.add(e);
207                                        }
208                                }
209                        }
210                        
211                        return ret;
212                }
213                
214                return ExecutionParticipant.super.diagnose(progressMonitor);
215        }
216        
217        /**
218         * If this method returns <code>true</code> the loaded model is diagnosed by {@link Diagnostician}.
219         * Override to return <code>false</code> if the diagnosis is not required after loading, e.g. the model is knows to contain diagnostic errors. 
220         * E.g. it is a partial model with unresolved proxies.
221         * @return  
222         */
223        protected boolean isDiagnoseModel() {
224                return true;
225        }
226
227        /**
228         * @return Classloader for loading classpath resources (URI's with <code>classpath</code> scheme).
229         * This implementation returns this class classloader.
230         */
231        protected ClassLoader getClassLoader() {
232                return getClass().getClassLoader();
233        }
