001package org.nasdanika.emf;
002
003import java.lang.reflect.InvocationHandler;
004import java.lang.reflect.Method;
005import java.lang.reflect.Proxy;
006import java.util.ArrayList;
007import java.util.Arrays;
008import java.util.Collections;
009import java.util.List;
010import java.util.function.Supplier;
011
012import org.eclipse.emf.common.notify.Adapter;
013import org.eclipse.emf.common.notify.Notifier;
014import org.eclipse.emf.common.notify.impl.AdapterImpl;
015import org.eclipse.emf.ecore.EClass;
016import org.eclipse.emf.ecore.EObject;
017import org.eclipse.emf.ecore.EcorePackage;
018import org.nasdanika.common.FunctionFactory;
019
020/**
021 * Base class for adapter factories delegating to {@link Supplier} or {@link FunctionFactory} to create an adapter and 
022 * wrapping adapter into a dynamic proxy if the adapter does not implement {@link Adapter}.
023 * @author Pavel Vlasov
024 *
025 */
026public abstract class DelegatingAdapterFactory<T> extends ComposeableAdapterFactoryImpl {
027        
028        private Class<T> type;
029        private ClassLoader proxyClassLoader;
030        
031        /**
032         * Uses {@link EObject}'s {@link EClass} as eClass argument
033         * @param type
034         * @param proxyClassLoader
035         */
036        protected DelegatingAdapterFactory(Class<T> type, ClassLoader proxyClassLoader) {
037                this(EcorePackage.Literals.EOBJECT, type, proxyClassLoader);
038        }       
039
040        /**
041         * 
042         * @param type
043         * @param proxyClassLoader Proxy class loader, can be null if created adapter implements {@link Adapter} and no proxy is required.
044         */
045        protected DelegatingAdapterFactory(EClass eClass, Class<T> type, ClassLoader proxyClassLoader) {
046                super(eClass);
047                this.type = type;
048                this.proxyClassLoader = proxyClassLoader;
049        }
050        
051        @Override
052        public boolean isFactoryForType(Object type) {
053                return this.type == type;
054        }
055                
056        private class DelegatingAdapter extends AdapterImpl {
057                
058                private T delegate;
059
060                public DelegatingAdapter(T adapter, Notifier target) {                  
061                        this.delegate = adapter;
062                        this.target = target;
063                }
064                
065                @Override
066                public void setTarget(Notifier newTarget) {                     
067                        if (target != newTarget) {
068                                if (newTarget == null) {
069                                        delegate = null;
070                                } else {
071                                        delegate = doCreateAdapter(newTarget);
072                                }
073                        }
074                        super.setTarget(newTarget);
075                }
076                
077                @Override
078                public boolean isAdapterForType(Object type) {
079                        return DelegatingAdapterFactory.this.isFactoryForType(type);
080                }               
081                                
082        }
083        
084        /**
085         * Creates adapter. This adapter does not have to implement {@link Adapter}. 
086         * If it does not then a proxy will be created implementing both T and adapter.
087         * @param target
088         * @return
089         */
090        protected abstract T doCreateAdapter(Notifier target);
091        
092        public static List<Class<?>> allInterfaces(Class<?> klass) {
093                if (klass==null) {
094                        return Collections.emptyList();
095                }
096                List<Class<?>> ret = new ArrayList<>();
097                if (klass.isInterface()) {
098                        ret.add(klass);
099                } else {
100                        ret.addAll(Arrays.asList(klass.getInterfaces()));
101                        Z: for (Class<?> i: allInterfaces(klass.getSuperclass())) {
102                                for (Class<?> ei: ret) {
103                                        if (i.isAssignableFrom(ei)) {
104                                                continue Z;
105                                        }
106                                }
107                                ret.add(i);
108                        }
109                }
110                return ret;
111        }       
112                
113        @Override
114        protected Adapter createAdapter(Notifier target) {
115                T adapter = doCreateAdapter(target);
116                
117                if (adapter == null) {
118                        return null;
119                }
120                
121                if (adapter instanceof Adapter) {
122                        return (Adapter) adapter;
123                }               
124                
125                DelegatingAdapter da = new DelegatingAdapter(adapter, target);
126                
127                List<Class<?>> adapterInterfaces = allInterfaces(adapter.getClass());
128                Class<?>[] proxyInterfaces = new Class<?>[adapterInterfaces.size() + 1];
129                for (int i = 0; i < adapterInterfaces.size(); ++i) {
130                        proxyInterfaces[i] = adapterInterfaces.get(i);
131                }
132                proxyInterfaces[proxyInterfaces.length -1] = Adapter.class;
133                
134                return (Adapter) Proxy.newProxyInstance(proxyClassLoader, proxyInterfaces, new InvocationHandler() {
135                        
136                        @Override
137                        public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
138                                Class<?> declaringClass = method.getDeclaringClass();
139                                if (declaringClass == Object.class) {
140                                        // Handling equals
141                                        if (method.getName().equals("equals") && method.getParameterCount() == 1 && method.getParameterTypes()[0] == Object.class) {
142                                                return args[0] == proxy;
143                                        }
144                                }
145                                Object target = declaringClass == Adapter.class ? da : da.delegate;
146                                return  method.invoke(target, args);
147                        }
148                        
149                });
150        }
151
152}
