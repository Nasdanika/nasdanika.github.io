001package org.nasdanika.emf;
002
003import java.io.File;
004import java.util.Map;
005
006import org.eclipse.emf.common.util.Diagnostic;
007import org.eclipse.emf.common.util.DiagnosticException;
008import org.eclipse.emf.common.util.URI;
009import org.eclipse.emf.ecore.EObject;
010import org.eclipse.emf.ecore.resource.Resource;
011import org.eclipse.emf.ecore.resource.ResourceSet;
012import org.eclipse.emf.ecore.util.Diagnostician;
013import org.nasdanika.common.CommandFactory;
014import org.nasdanika.common.ConsumerFactory;
015import org.nasdanika.common.Context;
016import org.nasdanika.common.ProgressMonitor;
017import org.nasdanika.common.Supplier;
018
019import picocli.CommandLine.Option;
020import picocli.CommandLine.Parameters;
021
022/**
023 * Loads a model from URL, optionally validates it and then passes to consumer.
024 * @author Pavel
025 *
026 */
027public abstract class ModelCommand<T extends EObject> extends ResourceSetCommand {
028        
029        @Parameters(
030                        paramLabel = "URI", 
031                        description = "Model (object) URI resolved relative to the current directory. May include fragment to address a non-root object")
032        protected String uri;
033        
034        @Option(names = {"-f", "--file"}, description = "URI is a file path")
035        private boolean isFile; 
036        
037        
038        /**
039         * @return Consumer factory which create a consumer to pass the loaded model to.
040         */
041        protected abstract ConsumerFactory<T> getConsumerFactory();
042        
043        protected Supplier<T> createSupplier(Context context) {
044                return new Supplier<T>() {
045
046                        @Override
047                        public double size() {
048                                return 1;
049                        }
050
051                        @Override
052                        public String name() {
053                                return "Loading model";
054                        }
055
056                        @SuppressWarnings("unchecked")
057                        @Override
058                        public T execute(ProgressMonitor progressMonitor) throws Exception {
059                                
060                                // Creating a resource set
061                                ResourceSet resourceSet = createResourceSet();
062
063                                URI modelUri = isFile ? URI.createFileURI(uri) : URI.createURI(new File(".").toURI().resolve(ModelCommand.this.uri).toString());
064                                Resource modelResource = resourceSet.getResource(modelUri, true);
065                                EObject target = modelUri.hasFragment() ? modelResource.getEObject(modelUri.fragment()) : modelResource.getContents().iterator().next();                        
066                                
067                                Diagnostician diagnostician = new Diagnostician() {
068                                        
069                                        public Map<Object,Object> createDefaultContext() {
070                                                Map<Object, Object> ctx = super.createDefaultContext();
071                                                ctx.put(Context.class, context);
072                                                return ctx;
073                                        };
074                                        
075                                };                              
076                                Diagnostic validationResult = diagnostician.validate(target);
077                                if (validationResult.getSeverity() == Diagnostic.ERROR) {
078                                        throw new DiagnosticException(validationResult);
079                                }
080                                return (T) target;
081                        }
082                };
083        }
084
085        /**
086         * Supplier to Consumer. Supplier loads model from URL, consumer takes the model as an argument.
087         */
088        @Override
089        protected CommandFactory getCommandFactory() {
090                return new CommandFactory() {
091                        
092                        @Override
093                        public org.nasdanika.common.Command create(Context context) throws Exception {
094                                return createSupplier(context).then(getConsumerFactory().create(context));
095                        }
096                };
097        }
098        
099        @Override
100        protected void reportException(Exception e) {
101                if (e instanceof DiagnosticException) {
102                        System.err.println("Diagnostic error:");
103                        dumpDiagnostic(((DiagnosticException) e).getDiagnostic(), 0);
104                }
105                super.reportException(e);
106        }
107
108        public static void dumpDiagnostic(Diagnostic d, int indent) {
109                for (int i=0; i < indent; ++i) {
110                        System.err.print("    ");
111                }
112                System.err.println(toString(d));
113            if (d.getChildren() != null) {
114                d.getChildren().forEach(c -> dumpDiagnostic(c, indent + 1));
115            }
116                
117        }
118        
119        static String toString(Diagnostic d) {
120                StringBuilder result = new StringBuilder();
121                switch (d.getSeverity()) {
122                case Diagnostic.OK:
123                        result.append("OK");
124                        break;
125                case Diagnostic.INFO:
126                        result.append("INFO");
127                        break;
128                case Diagnostic.WARNING:
129                        result.append("WARNING");
130                        break;
131                case Diagnostic.ERROR:
132                        result.append("ERROR");
133                        break;
134                case Diagnostic.CANCEL:
135                        result.append("CANCEL");
136                        break;
137                default:
138                        result.append(Integer.toHexString(d.getSeverity()));
139                        break;
140                }
141
142                result.append(" source=");
143                result.append(d.getSource());
144
145                result.append(" code=");
146                result.append(d.getCode());
147
148                result.append(' ');
149                result.append(d.getMessage());
150
151                if (d.getData() != null) {
152                        result.append(" data=");
153                        result.append(d.getData());
154                }
155
156                return result.toString();
157        }       
158        
159}
