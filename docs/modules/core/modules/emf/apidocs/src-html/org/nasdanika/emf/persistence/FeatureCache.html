001package org.nasdanika.emf.persistence;
002
003import java.util.function.BiFunction;
004import java.util.function.Supplier;
005
006import org.eclipse.emf.ecore.EObject;
007import org.eclipse.emf.ecore.EStructuralFeature;
008import org.nasdanika.emf.EObjectAdaptable;
009
010/**
011 * Computes feature value and caches. 
012 * This is an adapter interface.
013 * @author Pavel
014 *
015 */
016public interface FeatureCache {
017        
018        <F extends EStructuralFeature, T> T get(F feature, BiFunction<EObject, F, T> computer);
019        
020        /**
021         * Adapts to FeatureCache and gets from cache if it is present or from the computer otherwise.
022         * @param <F>
023         * @param <T>
024         * @param target
025         * @param feature
026         * @param computer
027         * @param add If true and adapter is not present, adds the adapter.
028         */
029        static <F extends EStructuralFeature, T> T get(EObject target, F feature, BiFunction<EObject, F, T> computer, boolean add) {
030                FeatureCache adapter = EObjectAdaptable.adaptTo(target, FeatureCache.class);
031                if (adapter == null) {
032                        if (add) {
033                                adapter = new FeatureCacheAdapter();
034                                target.eAdapters().add((FeatureCacheAdapter) adapter);
035                                ((FeatureCacheAdapter) adapter).setTarget(target);
036                        } else {
037                                return computer.apply(target, feature);
038                        }
039                }
040                return adapter.get(feature, computer);
041        }
042        
043        /**
044         * Adapts to FeatureCache and gets from cache if it is present or from the computer otherwise.
045         * @param <F>
046         * @param <T>
047         * @param target
048         * @param feature
049         * @param computer
050         * @param add If true and adapter is not present, adds the adapter.
051         */
052        static <F extends EStructuralFeature, T> T get(EObject target, F feature, Supplier<T> computer, boolean add) {
053                return get(target, feature, (t,f) -> computer.get(), add);
054        }
055        
056}
