001package org.nasdanika.emf.persistence;
002
003import java.net.URL;
004import java.util.Map;
005import java.util.function.BiFunction;
006
007import org.eclipse.emf.common.util.EMap;
008import org.eclipse.emf.ecore.EClass;
009import org.eclipse.emf.ecore.ENamedElement;
010import org.eclipse.emf.ecore.EReference;
011import org.eclipse.emf.ecore.EStructuralFeature;
012import org.nasdanika.common.ProgressMonitor;
013import org.nasdanika.common.persistence.MapAttribute;
014import org.nasdanika.common.persistence.Marker;
015import org.nasdanika.common.persistence.ObjectLoader;
016
017/**
018 * Loads {@link EMap} of references using loader.
019 * @author Pavel
020 *
021 */
022public class ReferenceMap<K,V> extends MapAttribute<K,V> {
023
024        private ReferenceFactory<V> valueFactory;
025        
026        /**
027         * @param key
028         * @param isDefault
029         * @param required
030         * @param defaultValue
031         * @param description
032         * @param resolver
033         * @param referenceType Reference type if the reference is homogenous, i.e. its type is known beforehand.
034         * @param exclusiveWith
035         */
036        public ReferenceMap(
037                        Object key, 
038                        boolean isDefault, 
039                        boolean required, 
040                        Map<K,V> defaultValue, 
041                        String description, 
042                        EClass eClass,
043                        EReference eReference,
044                        EObjectLoader resolver,
045                        boolean referenceSupplierFactory,
046                        BiFunction<EClass,ENamedElement,String> keyProvider,
047                        Object... exclusiveWith) {
048                super(key, isDefault, required, defaultValue, description, exclusiveWith);
049                EStructuralFeature valueFeature = eReference.getEReferenceType().getEStructuralFeature("value");
050                if (valueFeature instanceof EReference) {
051                        this.valueFactory = new ReferenceFactory<>(eClass, (EReference) valueFeature, eReference.getName(), resolver, referenceSupplierFactory, keyProvider);
052                }
053        }
054        
055        @Override
056        protected V createValue(ObjectLoader loader, K key, Object value, URL base, ProgressMonitor progressMonitor, Marker marker) throws Exception {
057                return valueFactory == null ? super.createValue(loader, key, value, base, progressMonitor, marker) : valueFactory.create(loader, value, base, progressMonitor, marker);
058        }
059        
060}
