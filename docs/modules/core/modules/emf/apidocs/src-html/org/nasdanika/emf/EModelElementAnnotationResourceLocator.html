001package org.nasdanika.emf;
002
003import java.util.function.Function;
004
005import org.eclipse.emf.ecore.EAnnotation;
006import org.eclipse.emf.ecore.EModelElement;
007import org.nasdanika.common.Context;
008import org.nasdanika.common.ResourceLocator;
009
010/**
011 * Retrieves resource strings from {@link EAnnotation}s for a given {@link EModelElement}.
012 * @author Pavel
013 *
014 */
015public class EModelElementAnnotationResourceLocator implements ResourceLocator {
016
017        private String source;
018        private Function<String, String> keyMapper;
019        private EModelElement target;
020        private Context chain;
021
022        /**
023         * Creates a new instance retrieving resource strings from a given annotation source on a given model element.
024         * @param target {@link EModelElement} to get annotations from.
025         * @param source Source of annotations.
026         * @param keyMapper Key mapping function, can be null. E.g. to load Russian element descriptions the key mapper may replace ``description`` with ``description_ru``.
027         * Key mapper is not applied to the chained context, 
028         * @param chain Chain context to load resource from.
029         */
030        public EModelElementAnnotationResourceLocator(EModelElement target, String source, Function<String,String> keyMapper, Context chain) {
031                this.target = target;
032                this.source = source;
033                this.keyMapper = keyMapper;
034                this.chain = chain == null ? Context.EMPTY_CONTEXT : chain;
035        }
036
037        @Override
038        public Object get(String key) {
039                EAnnotation ann = target.getEAnnotation(source);
040                Object ret = ann == null ? null : ann.getDetails().get(keyMapper == null ? key : keyMapper.apply(key));
041                return ret == null ? chain.get(key) : ret;
042        }
043
044        @Override
045        public <T> T get(Class<T> type) {
046                return null;
047        }
048        
049
050}
