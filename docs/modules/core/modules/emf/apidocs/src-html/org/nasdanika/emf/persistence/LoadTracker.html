001package org.nasdanika.emf.persistence;
002
003import org.eclipse.emf.ecore.EStructuralFeature;
004
005/**
006 * Adapter interface to track features which were loaded. 
007 * May be used when some features can be either loaded or computed.
008 * @author Pavel
009 *
010 */
011public interface LoadTracker {
012        
013        /**
014         * Feature is already loaded.
015         * This method can be used to decide whether to inject 
016         * a feature value derived from other loaded features 
017         * it a feature is not yet loaded. 
018         * @param feature
019         * @return
020         */
021        boolean isLoaded(EStructuralFeature feature);
022        
023        /**
024         * Feature is being loaded. This method
025         * returns true only when the loader calls the feature setter.
026         * It can be used to decide whether to trigger computations of  
027         * features derived from this feature.
028         * @param feature
029         * @return
030         */
031        boolean isLoading(EStructuralFeature feature);
032        
033        /**
034         * Returns loaded value which was provided to feature setter for changeable features.
035         * This value can be used to derive (compute) feature value for non-changeable derived features which have computed annotation set to true.
036         * @param feature
037         * @return
038         */
039        Object get(EStructuralFeature feature);
040
041}
