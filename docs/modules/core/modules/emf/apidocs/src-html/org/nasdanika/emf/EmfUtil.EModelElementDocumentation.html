068        /**
069         * Collects type dependencies for a given class - attribute types, eoperation return and exception types, eparameter types.
070         * Does not collect reference types and supertypes, as they are reported differently, but collects their generic parameter types.
071         * @param eClass
072         * @return
073         */
074        public static Collection<EClassifier> collectTypeDependencies(EClass eClass) {
075                Collection<EClassifier> collector = new HashSet<>();
076                Predicate<EGenericType> predicate = new HashSet<EGenericType>()::add;
077                for (EGenericType st: eClass.getEGenericSuperTypes()) {
078                        collector.addAll(collectGenericTypeDependencies(st, false, predicate));
079                }
080                for (EReference ref: eClass.getEReferences()) {
081                        collector.addAll(collectGenericTypeDependencies(ref.getEGenericType(), false, predicate));
082                }               
083                
084                for (EAttribute attr: eClass.getEAttributes()) {
085                        collector.addAll(collectGenericTypeDependencies(attr.getEGenericType(), true, predicate));
086                }                                       
087                for (EOperation op: eClass.getEOperations()) {
088                        EGenericType opType = op.getEGenericType();
089                        if (opType != null) {
090                                collector.addAll(collectGenericTypeDependencies(opType, true, predicate));
091                        }
092                        for (EGenericType ge: op.getEGenericExceptions()) {
093                                collector.addAll(collectGenericTypeDependencies(ge, true, predicate));
094                        }
095                        for (EParameter ep: op.getEParameters()) {
096                                collector.addAll(collectGenericTypeDependencies(ep.getEGenericType(), true, predicate));
097                        }
098                }                                       
099                return collector;
100        }

102        /**
103         * Collects generic type dependencies, i.e. classifiers used in generic type definition.
104         * @param type Generic type
105         * @param includeEClassifier if true, include EClassifier of the type itself. Otherwise, only EClassifiers of referenced generic types - bounds, type arguments, and type parameter.
106         * @return A set of {@link EClassifier}'s.
107         */
108        public static Collection<EClassifier> collectGenericTypeDependencies(EGenericType type, boolean includeEClassifier, Predicate<EGenericType> predicate) {
109                Set<EClassifier> ret = new HashSet<>();
110                if (type != null && predicate.test(type)) {
111                        EClassifier eClassifier = type.getEClassifier();
112                        if (includeEClassifier && eClassifier != null) {
113                                ret.add(eClassifier);
114                        }
115                        
116                        // Lower bound
117                        ret.addAll(collectGenericTypeDependencies(type.getELowerBound(), true, predicate));
118                        
119                        // Upper bound
120                        ret.addAll(collectGenericTypeDependencies(type.getEUpperBound(), true, predicate));
121                        
122                        // Type arguments
123                        for (EGenericType ta: type.getETypeArguments()) {
124                                ret.addAll(collectGenericTypeDependencies(ta, true, predicate));
125                        }
126                        
127                        // Type parameter
128                        ETypeParameter tp = type.getETypeParameter();
129                        if (tp != null) {
130                                for (EGenericType b: tp.getEBounds()) {
131                                        ret.addAll(collectGenericTypeDependencies(b, true, predicate));                                 
132                                }
133                        }
134                }
135                return ret;
136        }

138        /**
139         * Resolves reference relative to the resource.   
140         * @param resource Used to resolve relative references if not null.
141         * @param reference Reference.
142         * @return
143         * @throws Exception
144         */
145        public static URI resolveReference(Resource resource, String reference) throws Exception {              
146                URI refUri = Util.isBlank(reference) ? null : URI.createURI(reference);
147                if (resource != null) {
148                        URI resUri = resource.getURI();
149                        if (refUri == null) {
150                                return resUri;
151                        }
152                        if (resUri.isHierarchical()) {
153                                refUri = refUri.resolve(resUri);
154                        }                       
155                }
156                return refUri;
157        }

193        /**
194         * Loads element documentation from a documentation resource specified in documentation-reference nasdanika (urn:org.nasdanika) annotation resolved relative to the 
195         * model element resource. 
196         * @param modelElement
197         * @return
198         */
199        public static EModelElementDocumentation getDocumentation(EModelElement modelElement) {
200                String modelDoc = EcoreUtil.getDocumentation(modelElement);
201                if (!Util.isBlank(modelDoc)) {
202                        Resource res = modelElement.eResource();
203                        return EModelElementDocumentation.from(modelDoc, res == null ? null : res.getURI());                    
204                }
205                
206                EAnnotation nasdanikaAnnotation = NcoreUtil.getNasdanikaAnnotation(modelElement);
207                if (nasdanikaAnnotation != null) {
208                        String docRef = nasdanikaAnnotation.getDetails().get("documentation-reference");
209                        if (!Util.isBlank(docRef)) {
210                                try {
211                                        URI refUri = resolveReference(modelElement.eResource(), docRef);
212                                        if (Util.isBlank(refUri.scheme())) {
213                                                return null;                    
214                                        }
215
216                                        String doc = DefaultConverter.INSTANCE.toString(new URL(refUri.toString()).openStream());
217                                        return EModelElementDocumentation.from(doc, refUri);
218                                } catch (Exception e) {
219                                        e.printStackTrace();
220                                        return EModelElementDocumentation.from("<div class='nsd-error'>Error loading documentation: " + e + "</div>", null);
221                                }
222                        }
223                }
224                
225                return null;
226        }

228        /**
229         * Loads element load documentation from load-doc annotation of or a documentation resource specified in load-doc-reference nasdanika (urn:org.nasdanika) annotation resolved relative to the 
230         * model element resource. 
231         * @param modelElement
232         * @return
233         */
234        public static EModelElementDocumentation getLoadDocumentation(EModelElement modelElement) {
235                EAnnotation nasdanikaAnnotation = NcoreUtil.getNasdanikaAnnotation(modelElement);
236                if (nasdanikaAnnotation != null) {
237                        String docRef = nasdanikaAnnotation.getDetails().get(EObjectLoader.LOAD_DOC_REF);
238                        if (!Util.isBlank(docRef)) {
239                                try {
240                                        URI refUri = resolveReference(modelElement.eResource(), docRef);
241                                        if (Util.isBlank(refUri.scheme())) {
242                                                return null;                    
243                                        }
244
245                                        String doc = DefaultConverter.INSTANCE.toString(new URL(refUri.toString()).openStream());
246                                        return EModelElementDocumentation.from(doc, refUri);
247                                } catch (Exception e) {
248                                        e.printStackTrace();
249                                        return EModelElementDocumentation.from("<div class='nsd-error'>Error loading documentation: " + e + "</div>", null);
250                                }
251                        }
252                        
253                        String loadDoc = nasdanikaAnnotation.getDetails().get(EObjectLoader.LOAD_DOC);
254                        if (Util.isBlank(loadDoc)) {
255                                return null;
256                        }
257                        return new EModelElementDocumentation() {
258                                
259                                @Override
260                                public URI getLocation() {
261                                        Resource res = modelElement.eResource();
262                                        return res == null ? null : res.getURI();
263                                }
264                                
265                                @Override
266                                public String getDocumentation() {
267                                        return loadDoc;
268                                }
269                        };
270                }
271                
272                return null;
273        }

276        public static Diagnostic wrap(org.eclipse.emf.common.util.Diagnostic diagnostic) {
277                Status status;
278                switch (diagnostic.getSeverity()) {
279                case org.eclipse.emf.common.util.Diagnostic.CANCEL:
280                        status = Status.CANCEL;
281                        break;
282                case org.eclipse.emf.common.util.Diagnostic.INFO:
283                        status = Status.INFO;
284                        break;
285                case org.eclipse.emf.common.util.Diagnostic.OK:
286                        status = Status.SUCCESS;
287                        break;
288                case org.eclipse.emf.common.util.Diagnostic.WARNING:
289                        status = Status.WARNING;
290                        break;
291                case org.eclipse.emf.common.util.Diagnostic.ERROR:
292                default:
293                        status = Status.ERROR;
294                        break;
295                        
296                }
297                BasicDiagnostic ret = new BasicDiagnostic(status, diagnostic.getMessage(), diagnostic);
298                for (org.eclipse.emf.common.util.Diagnostic child: diagnostic.getChildren()) {
299                        ret.add(wrap(child));
300                }
301                return ret;
302        }

304        public static <K, V extends EObject> Map<K, List<V>> groupBy(Collection<V> elements, EStructuralFeature... keyFeatures) {
305                @SuppressWarnings("unchecked")
306                Function<EObject,K> classifier = e -> {
307                        if (e == null) {
308                                return null;
309                        }
310                        if (keyFeatures.length == 1) {
311                                return (K) e.eGet(keyFeatures[0]);
312                        }
313                        List<Object> key = new ArrayList<>();
314                        for (EStructuralFeature keyFeature: keyFeatures) {
315                                key.add(e.eGet(keyFeature));
316                        }
317                        return (K) key;
318                };
319                return Util.groupBy(elements, classifier);
320        }

333        public static void dumpDiagnostic(org.eclipse.emf.common.util.Diagnostic d, int indent, PrintStream out) {
334                for (int i=0; i < indent; ++i) {
335                        out.print("    ");
336                }
337                out.println(toString(d));
338            if (d.getChildren() != null) {
339                d.getChildren().forEach(c -> dumpDiagnostic(c, indent + 1, out));
340            }           
341        }

343        static String toString(org.eclipse.emf.common.util.Diagnostic d) {
344                StringBuilder result = new StringBuilder();
345                switch (d.getSeverity()) {
346                case org.eclipse.emf.common.util.Diagnostic.OK:
347                        result.append("OK");
348                        break;
349                case org.eclipse.emf.common.util.Diagnostic.INFO:
350                        result.append("INFO");
351                        break;
352                case org.eclipse.emf.common.util.Diagnostic.WARNING:
353                        result.append("WARNING");
354                        break;
355                case org.eclipse.emf.common.util.Diagnostic.ERROR:
356                        result.append("ERROR");
357                        break;
358                case org.eclipse.emf.common.util.Diagnostic.CANCEL:
359                        result.append("CANCEL");
360                        break;
361                default:
362                        result.append(Integer.toHexString(d.getSeverity()));
363                        break;
364                }
365
366                result.append(" source=");
367                result.append(d.getSource());
368
369                result.append(" code=");
370                result.append(d.getCode());
371
372                result.append(' ');
373                result.append(d.getMessage());
374
375                if (d.getData() != null) {
376                        result.append(" data=");
377                        result.append(d.getData());
378                }
379
380                return result.toString();
381        }

383        /**
384         * Resolves all proxies, clears feature caches and then diagnoses the resource set.
385         * @param resourceSet
386         * @param context
387         * @return
388         */
389        public static org.eclipse.emf.common.util.Diagnostic resolveClearCacheAndDiagnose(ResourceSet resourceSet, Context context) {
390                EcoreUtil.resolveAll(resourceSet);
391                resourceSet.getAllContents().forEachRemaining(notifier -> notifier.eNotify(FeatureCacheAdapter.CLEAR_CACHE));
392
393                Diagnostician diagnostician = new Diagnostician();
394                org.eclipse.emf.common.util.BasicDiagnostic ret = new org.eclipse.emf.common.util.BasicDiagnostic();
395                Map<Class<Context>, Context> diagnosticContext = Collections.singletonMap(Context.class, context);
396                for (Resource resource: resourceSet.getResources()) {
397                        org.eclipse.emf.common.util.BasicDiagnostic resourceDiagnostic = new org.eclipse.emf.common.util.BasicDiagnostic(resource.getURI().toString(), 0, "Resource diagnostic", new Object[] {resource});
398                        for (EObject e: resource.getContents()) {
399                                org.eclipse.emf.common.util.Diagnostic diagnostic = diagnostician.validate(e, diagnosticContext);
400                                resourceDiagnostic.add(diagnostic);
401                        }
402                        ret.add(resourceDiagnostic);
403                }
404                return ret;
405        }

407        /**
408         * Diagnoses the resource set.
409         * @param resourceSet
410         * @param context
411         * @param diagnosticConsumer consumer of diagnostics.
412         * @return
413         */
414        public static void diagnose(ResourceSet resourceSet, Context context, BiConsumer<EObject, org.eclipse.emf.common.util.Diagnostic> diagnosticConsumer) {
415                EcoreUtil.resolveAll(resourceSet);
416                resourceSet.getAllContents().forEachRemaining(notifier -> notifier.eNotify(FeatureCacheAdapter.CLEAR_CACHE));
417
418                Diagnostician diagnostician = new Diagnostician();
419                Map<Class<Context>, Context> diagnosticContext = Collections.singletonMap(Context.class, context);
420                for (Resource resource: resourceSet.getResources()) {
421                        for (EObject e: resource.getContents()) {
422                                diagnosticConsumer.accept(e, diagnostician.validate(e, diagnosticContext));
423                        }
424                }
425        }
