001package org.nasdanika.emf.persistence;
002
003import java.io.File;
004import java.net.URI;
005import java.net.URL;
006import java.nio.file.Path;
007import java.nio.file.Paths;
008import java.util.HashMap;
009import java.util.Map;
010import java.util.Map.Entry;
011import java.util.function.BiFunction;
012
013import org.eclipse.emf.ecore.util.EcoreUtil;
014import org.eclipse.jgit.lib.Constants;
015import org.eclipse.jgit.lib.Ref;
016import org.eclipse.jgit.lib.Repository;
017import org.eclipse.jgit.lib.StoredConfig;
018import org.eclipse.jgit.storage.file.FileRepositoryBuilder;
019import org.nasdanika.common.ProgressMonitor;
020import org.nasdanika.common.Status;
021import org.nasdanika.common.Util;
022import org.nasdanika.ncore.GitMarker;
023import org.nasdanika.ncore.Marker;
024import org.nasdanika.ncore.NcoreFactory;
025
026/**
027 * Creates a git marker if location is in a git repository or a regular marker otherwise.
028 * @author Pavel
029 *
030 */
031public class GitMarkerFactory implements BiFunction<String, ProgressMonitor, org.nasdanika.ncore.Marker> {
032        
033        private Map<File, GitMarker> templates = new HashMap<>();
034
035        @Override
036        public Marker apply(String location, ProgressMonitor progressMonitor) {         
037                try {
038                        if (!Util.isBlank(location) && location.startsWith("file:/")) {
039                                URI locationURI = new URL(location).toURI();
040                                Path path = Paths.get(locationURI);
041                                File file = path.toFile().getCanonicalFile();
042                                for (Entry<File, GitMarker> re: templates.entrySet()) {
043                                        if (re.getValue() == null) {
044                                                // We already know that there is no git repo for this location
045                                                org.nasdanika.ncore.Marker marker = NcoreFactory.eINSTANCE.createMarker();
046                                                marker.setLocation(location);
047                                                return marker;
048                                        }
049                                        if (isAncestor(file, re.getKey())) {
050                                                GitMarker marker = EcoreUtil.copy(re.getValue());
051                                                URI repoWorkTreeURI = re.getKey().toURI();                                      
052                                                marker.setPath(repoWorkTreeURI.relativize(locationURI).toString());
053                                                marker.setLocation(location);
054                                                return marker;                                  
055                                        }
056                                }
057                                FileRepositoryBuilder repositoryBuilder = new FileRepositoryBuilder().readEnvironment();
058                                repositoryBuilder.findGitDir(file);
059                                if (repositoryBuilder.getGitDir() == null) {
060                                        templates.put(file.getParentFile(), null);
061                                }                               
062                                try (Repository repository = repositoryBuilder.build()) {
063                                        GitMarker template = NcoreFactory.eINSTANCE.createGitMarker();
064                                        StoredConfig config = repository.getConfig();
065                                        for (String remote: repository.getRemoteNames()) {
066                                                template.getRemotes().put(remote, config.getString("remote", "origin", "url"));
067                                        }
068                                        template.setBranch(repository.getBranch());
069                                        // TODO ref's with commit ID's
070                                        for (Ref ref: repository.getRefDatabase().getRefs()) {
071                                                if (ref.getName().equals(Constants.HEAD)) {
072                                                        template.setHead(ref.getObjectId().getName());
073                                                }
074                                        }
075                                        templates.put(repository.getWorkTree(), template);
076                                        GitMarker marker = EcoreUtil.copy(template);
077                                        URI repoWorkTreeURI = repository.getWorkTree().toURI();                                 
078                                        marker.setPath(repoWorkTreeURI.relativize(locationURI).toString());
079                                        marker.setLocation(location);
080                                        return marker;                                  
081                                }
082                        }
083                } catch (Exception e) {
084                        progressMonitor.worked(Status.ERROR, 1, "Error creating a file from location '" + location +"': " + e, e);
085                }
086                org.nasdanika.ncore.Marker marker = NcoreFactory.eINSTANCE.createMarker();
087                marker.setLocation(location);
088                return marker;  
089        }
090        
091        private boolean isAncestor(File file, File ancestor) {
092                File parent = file.getParentFile();
093                if (parent == null) {
094                        return false;
095                }
096                if (parent.equals(ancestor)) {
097                        return true;
098                }
099                return isAncestor(parent, ancestor);
100        }
101
102}
