001package org.nasdanika.emf;
002
003import java.util.function.Supplier;
004
005import org.eclipse.emf.common.notify.Notifier;
006import org.eclipse.emf.ecore.EClass;
007import org.eclipse.emf.ecore.EcorePackage;
008
009/**
010 * Adapter factory delegating to {@link Supplier}, which can be a constructor.
011 * @author Pavel Vlasov
012 *
013 */
014public class SupplierAdapterFactory<T> extends DelegatingAdapterFactory<T> {
015        
016        private Supplier<T> supplier;
017        
018        public SupplierAdapterFactory(Class<T> type, ClassLoader proxyClassLoader, Supplier<T> supplier) {
019                this(EcorePackage.Literals.EOBJECT, type, proxyClassLoader, supplier);
020        }       
021
022        public SupplierAdapterFactory(EClass eClass, Class<T> type, ClassLoader proxyClassLoader, Supplier<T> supplier) {
023                super(eClass, type, proxyClassLoader);
024                this.supplier = supplier;
025        }
026
027        @Override
028        protected T doCreateAdapter(Notifier target) {
029                return supplier.get();
030        }       
031
032}
