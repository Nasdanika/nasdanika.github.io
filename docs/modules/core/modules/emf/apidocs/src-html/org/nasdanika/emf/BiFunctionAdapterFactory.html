001package org.nasdanika.emf;
002
003import java.util.function.BiFunction;
004
005import org.eclipse.emf.common.notify.Notifier;
006import org.eclipse.emf.ecore.EClass;
007import org.eclipse.emf.ecore.EObject;
008import org.eclipse.emf.ecore.EcorePackage;
009
010/**
011 * Adapter factory which creates adapter using a {@link BiFunction}, e.g. constructor with two arguments.
012 * The first argument is the notifier, the second is the factory. The second argument can be used to get access to the root adapter factory, 
013 * e.g. for retrieval of contextual information.
014 * @param <T> Adapter type.
015 * @param <N> Notifier type.
016 */
017public class BiFunctionAdapterFactory<T, N> extends DelegatingAdapterFactory<T> {
018        
019        private BiFunction<N, BiFunctionAdapterFactory<T, N> ,  T> biFunction;
020        
021        /**
022         * Uses {@link EObject}'s {@link EClass} as eClass argument
023         * @param type
024         * @param proxyClassLoader
025         * @param biFunction
026         */
027        public BiFunctionAdapterFactory(Class<T> type, ClassLoader proxyClassLoader, BiFunction<N, BiFunctionAdapterFactory<T, N> ,  T> biFunction) {
028                this(EcorePackage.Literals.EOBJECT, type, proxyClassLoader, biFunction);
029        } 
030        
031        public BiFunctionAdapterFactory(EClass eClass, Class<T> type, ClassLoader proxyClassLoader, BiFunction<N, BiFunctionAdapterFactory<T, N> ,  T> biFunction) {
032                super(eClass, type, proxyClassLoader);
033                this.biFunction = biFunction;
034        }
035
036        @SuppressWarnings("unchecked")
037        @Override
038        protected T doCreateAdapter(Notifier target) {
039                return biFunction.apply((N) target, this);
040        }
041
042}
