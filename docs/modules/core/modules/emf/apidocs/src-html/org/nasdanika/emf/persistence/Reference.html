001package org.nasdanika.emf.persistence;
002
003import java.util.function.BiFunction;
004
005import org.eclipse.emf.common.util.URI;
006import org.eclipse.emf.ecore.EClass;
007import org.eclipse.emf.ecore.ENamedElement;
008import org.eclipse.emf.ecore.EReference;
009import org.nasdanika.common.ProgressMonitor;
010import org.nasdanika.common.persistence.Attribute;
011import org.nasdanika.common.persistence.Marker;
012import org.nasdanika.common.persistence.ObjectLoader;
013
014/**
015 * If config is a list loads each element creating elements using element factory and then loading them, otherwise creates a singleton in the same way as explained before.
016 * @author Pavel
017 *
018 * @param <T>
019 */
020public class Reference<T> extends Attribute<T> {
021        
022        private ReferenceFactory<T> referenceFactory;
023        
024        /**
025         * 
026         * @param key
027         * @param isDefault
028         * @param required
029         * @param defaultValue
030         * @param description
031         * @param resolver
032         * @param referenceType Reference type if the reference is homogenous, i.e. its type is known beforehand.
033         * @param exclusiveWith
034         */
035        public Reference(
036                        Object key, 
037                        boolean isDefault, 
038                        boolean required, 
039                        T defaultValue, 
040                        String description, 
041                        EClass eClass,
042                        EReference eReference,
043                        EObjectLoader resolver,
044                        boolean referenceSupplierFactory,
045                        BiFunction<EClass,ENamedElement,String> keyProvider,
046                        Object... exclusiveWith) {
047                super(key, isDefault, required, defaultValue, description, exclusiveWith);
048                this.referenceFactory = new ReferenceFactory<>(eClass, eReference, null, resolver, referenceSupplierFactory, keyProvider);
049        }
050        
051        @Override
052        public T create(ObjectLoader loader, Object element, URI base, ProgressMonitor progressMonitor, Marker marker) throws Exception {
053                return referenceFactory.create(loader, element, base, progressMonitor, marker);
054        }
055
056}
