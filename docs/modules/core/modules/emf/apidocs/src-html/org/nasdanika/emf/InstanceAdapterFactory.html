001package org.nasdanika.emf;
002
003import org.eclipse.emf.common.notify.Notifier;
004import org.eclipse.emf.ecore.EClass;
005import org.eclipse.emf.ecore.EcorePackage;
006
007/**
008 * Adapter factory delegating to a single shared adapter instance for all notifiers.
009 * @author Pavel Vlasov
010 *
011 */
012public class InstanceAdapterFactory<T> extends DelegatingAdapterFactory<T> {
013        
014        private T adapter;
015        
016        public InstanceAdapterFactory(Class<T> type, ClassLoader proxyClassLoader, T adapter) {
017                this(EcorePackage.Literals.EOBJECT, type, proxyClassLoader, adapter);
018        }       
019
020        public InstanceAdapterFactory(EClass eClass, Class<T> type, ClassLoader proxyClassLoader, T adapter) {
021                super(eClass, type, proxyClassLoader);
022                this.adapter = adapter;
023        }
024
025        @Override
026        protected T doCreateAdapter(Notifier target) {
027                return adapter;
028        }       
029
030}
