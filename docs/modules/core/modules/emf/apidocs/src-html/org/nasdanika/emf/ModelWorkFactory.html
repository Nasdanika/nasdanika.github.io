001package org.nasdanika.emf;
002
003import org.eclipse.emf.common.util.URI;
004import org.eclipse.emf.ecore.EObject;
005import org.eclipse.emf.ecore.EPackage;
006import org.eclipse.emf.ecore.resource.Resource;
007import org.eclipse.emf.ecore.resource.ResourceSet;
008import org.eclipse.emf.ecore.resource.impl.ResourceSetImpl;
009import org.eclipse.emf.ecore.xmi.impl.XMIResourceFactoryImpl;
010import org.nasdanika.common.Context;
011import org.nasdanika.common.Supplier;
012import org.nasdanika.common.SupplierFactory;
013
014/**
015 * This class loads a model with the root element, adapts it to {@link SupplierFactory}, and delegates its methods to it. 
016 * @author Pavel
017 *
018 */
019public class ModelWorkFactory<T> implements SupplierFactory<T> { 
020
021        protected EObject root; 
022        protected Resource modelResource;
023
024        /**
025         * Creates a generator by loading a generator model specified by the platform URI, e.g. ``org.nasdanika.codegen.tests.models/models/static-text/basic.codegen``, into a new {@link ResourceSet}. 
026         * @param platformPluginUri
027         * @throws Exception 
028         */
029        public ModelWorkFactory(String platformPluginUri, EPackage... ePackages) throws Exception {
030                this(URI.createPlatformPluginURI(platformPluginUri, false), ePackages);
031        }
032        
033        /**
034         * Creates a generator by loading a generator model from the specified URI into a new {@link ResourceSet}. 
035         * @param modelUri
036         * @throws Exception 
037         */
038        public ModelWorkFactory(URI modelUri, EPackage... ePackages) throws Exception {
039                this(createResourceSet(ePackages), modelUri);
040        }       
041        
042        /**
043         * Creates a generator by loading a generator model specified by the platform URI, e.g. ``org.nasdanika.codegen.tests.models/models/static-text/basic.codegen``, into 
044         * the specified {@link ResourceSet}.
045         * @param resourceSet Resource set to load the model to, see ``ModelGenerator(ResourceSet resourceSet, URI modelUri)`` for details. 
046         * @param platformPluginUri
047         * @throws Exception 
048         */
049        public ModelWorkFactory(ResourceSet resourceSet, String platformPluginUri) throws Exception {
050                this(resourceSet, URI.createPlatformPluginURI(platformPluginUri, false));
051        }
052        
053        /**
054         * Creates a new {@link ResourceSet} if it was not passed to a constructor explicitly. This implementation creates a new resource set and configures it to load the model by registering an {@link XMIResourceFactoryImpl}
055         * and putting {@link NcorePackage} to the package registry. If the model uses model elements from other {@link EPackage}s pass  a resource set constructed by the client to one of the 
056         * constructors which accept resource set. The resource set can be constructed by this method and then additional packages may be registered by the client code before passing the resource
057         * set to a constructor.   
058         * @return
059         */
060        public static ResourceSet createResourceSet(EPackage... ePackages) {
061                ResourceSetImpl resourceSet = new ResourceSetImpl();
062                resourceSet.getResourceFactoryRegistry().getExtensionToFactoryMap().put(Resource.Factory.Registry.DEFAULT_EXTENSION, new XMIResourceFactoryImpl());
063                for (EPackage ePackage: ePackages) {
064                        resourceSet.getPackageRegistry().put(ePackage.getNsURI(), ePackage);
065                }
066                return resourceSet;
067        }
068
069        /**
070         * Override to specify result type of supplier.
071         * @return
072         */
073        @SuppressWarnings("unchecked")
074        protected Class<T> getSupplierResultType() {
075                return (Class<T>) Object.class;
076        }
077        
078        /**
079         * Creates a generator by loading a generator model from the specified URI into the specified {@link ResourceSet}.
080         * @param resourceSet Resource set to load the model to. 
081         * @throws Exception 
082         */
083        public ModelWorkFactory(ResourceSet resourceSet, URI modelUri) throws Exception {
084                modelResource = resourceSet.getResource(modelUri, true);
085                root = modelResource.getContents().iterator().next();
086        }
087        
088        @Override
089        public Supplier<T> create(Context context) throws Exception {
090                return EObjectAdaptable.adaptToSupplierFactoryNonNull(root, getSupplierResultType()).create(context);
091        }
092        
093        public Resource getModelResource() {
094                return modelResource;
095        }
096
097}
