001package org.nasdanika.emf;
002
003import org.eclipse.emf.common.util.URI;
004import org.eclipse.emf.ecore.EObject;
005import org.eclipse.emf.ecore.EPackage;
006import org.eclipse.emf.ecore.resource.Resource;
007import org.eclipse.emf.ecore.resource.ResourceSet;
008import org.eclipse.emf.ecore.resource.impl.ResourceSetImpl;
009import org.eclipse.emf.ecore.xmi.impl.XMIResourceFactoryImpl;
010import org.nasdanika.common.Context;
011import org.nasdanika.common.Supplier;
012import org.nasdanika.common.SupplierFactory;
013import org.nasdanika.ncore.NcorePackage;
014
015/**
016 * This class loads a model with the root element, adapts it to {@link SupplierFactory}, and delegates its methods to it. 
017 * @author Pavel
018 *
019 */
020public class ModelWorkFactory<T> implements SupplierFactory<T> { 
021
022        protected EObject root; 
023        protected Resource modelResource;
024
025        /**
026         * Creates a generator by loading a generator model specified by the platform URI, e.g. ``org.nasdanika.codegen.tests.models/models/static-text/basic.codegen``, into a new {@link ResourceSet}. 
027         * @param platformPluginUri
028         * @throws Exception 
029         */
030        public ModelWorkFactory(String platformPluginUri, EPackage... ePackages) throws Exception {
031                this(URI.createPlatformPluginURI(platformPluginUri, false), ePackages);
032        }
033        
034        /**
035         * Creates a generator by loading a generator model from the specified URI into a new {@link ResourceSet}. 
036         * @param modelUri
037         * @throws Exception 
038         */
039        public ModelWorkFactory(URI modelUri, EPackage... ePackages) throws Exception {
040                this(createResourceSet(ePackages), modelUri);
041        }       
042        
043        /**
044         * Creates a generator by loading a generator model specified by the platform URI, e.g. ``org.nasdanika.codegen.tests.models/models/static-text/basic.codegen``, into 
045         * the specified {@link ResourceSet}.
046         * @param resourceSet Resource set to load the model to, see ``ModelGenerator(ResourceSet resourceSet, URI modelUri)`` for details. 
047         * @param platformPluginUri
048         * @throws Exception 
049         */
050        public ModelWorkFactory(ResourceSet resourceSet, String platformPluginUri) throws Exception {
051                this(resourceSet, URI.createPlatformPluginURI(platformPluginUri, false));
052        }
053        
054        /**
055         * Creates a new {@link ResourceSet} if it was not passed to a constructor explicitly. This implementation creates a new resource set and configures it to load the model by registering an {@link XMIResourceFactoryImpl}
056         * and putting {@link NcorePackage} to the package registry. If the model uses model elements from other {@link EPackage}s pass  a resource set constructed by the client to one of the 
057         * constructors which accept resource set. The resource set can be constructed by this method and then additional packages may be registered by the client code before passing the resource
058         * set to a constructor.   
059         * @return
060         */
061        public static ResourceSet createResourceSet(EPackage... ePackages) {
062                ResourceSetImpl resourceSet = new ResourceSetImpl();
063                resourceSet.getResourceFactoryRegistry().getExtensionToFactoryMap().put(Resource.Factory.Registry.DEFAULT_EXTENSION, new XMIResourceFactoryImpl());
064                for (EPackage ePackage: ePackages) {
065                        resourceSet.getPackageRegistry().put(ePackage.getNsURI(), ePackage);
066                }
067                return resourceSet;
068        }
069
070        /**
071         * Override to specify result type of supplier.
072         * @return
073         */
074        @SuppressWarnings("unchecked")
075        protected Class<T> getSupplierResultType() {
076                return (Class<T>) Object.class;
077        }
078        
079        /**
080         * Creates a generator by loading a generator model from the specified URI into the specified {@link ResourceSet}.
081         * @param resourceSet Resource set to load the model to. 
082         * @throws Exception 
083         */
084        public ModelWorkFactory(ResourceSet resourceSet, URI modelUri) throws Exception {
085                modelResource = resourceSet.getResource(modelUri, true);
086                root = modelResource.getContents().iterator().next();
087        }
088        
089        @Override
090        public Supplier<T> create(Context context) throws Exception {
091                return EObjectAdaptable.adaptToSupplierFactoryNonNull(root, getSupplierResultType()).create(context);
092        }
093        
094        public Resource getModelResource() {
095                return modelResource;
096        }
097
098}
