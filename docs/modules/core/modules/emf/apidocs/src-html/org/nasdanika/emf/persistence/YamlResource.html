001package org.nasdanika.emf.persistence;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.io.OutputStream;
006import java.io.OutputStreamWriter;
007import java.net.URL;
008import java.util.ArrayList;
009import java.util.Collection;
010import java.util.List;
011import java.util.Map;
012
013import org.eclipse.emf.common.util.URI;
014import org.eclipse.emf.ecore.EObject;
015import org.eclipse.emf.ecore.resource.impl.ResourceImpl;
016import org.nasdanika.common.Context;
017import org.nasdanika.common.NasdanikaException;
018import org.nasdanika.common.ProgressMonitor;
019import org.nasdanika.common.SupplierFactory;
020import org.nasdanika.common.Util;
021import org.nasdanika.common.persistence.ConfigurationException;
022import org.nasdanika.common.persistence.MarkerImpl;
023import org.nasdanika.common.persistence.ObjectLoader;
024import org.nasdanika.common.persistence.Storable;
025import org.nasdanika.emf.EObjectAdaptable;
026import org.yaml.snakeyaml.DumperOptions;
027import org.yaml.snakeyaml.DumperOptions.FlowStyle;
028import org.yaml.snakeyaml.Yaml;
029import org.yaml.snakeyaml.error.MarkedYAMLException;
030
031/**
032 * Loads EMF classes from YAML using {@link ObjectLoader}s. 
033 * Saving is supported if the resource content implements or is adaptable to {@link Storable}. 
034 * The idea of this resource is to load models from manually created YAML files or YAML files generated 
035 * by some other means. 
036 * @author Pavel
037 *
038 */
039public class YamlResource extends ResourceImpl {
040        
041        private ObjectLoader loader;
042        private ProgressMonitor progressMonitor;
043        private Context context;
044
045        public YamlResource(URI uri, ObjectLoader loader, Context context, ProgressMonitor progressMonitor) {
046                super(uri);
047                this.loader = loader;
048                this.context = context;
049                this.progressMonitor = progressMonitor;
050        }
051
052        @SuppressWarnings("unchecked")
053        @Override
054        protected void doLoad(InputStream inputStream, Map<?, ?> options) throws IOException {
055                try {
056                        Object data = loader.loadYaml(inputStream, getURI(), progressMonitor);
057                        if (data instanceof Collection) {
058                                getContents().addAll((Collection<EObject>) data);
059                        } else {
060                                if (data instanceof SupplierFactory) {
061                                        EObject eObject = Util.call(((SupplierFactory<EObject>) data).create(context), progressMonitor, null);
062                                        getContents().add(eObject);
063                                } else if (data instanceof EObject) {
064                                        getContents().add((EObject) data);
065                                } else {
066                                        throw new IOException("Not an instance of EObject: " + data);
067                                }
068                        }
069                } catch (MarkedYAMLException e) {
070                        throw new ConfigurationException(e.getMessage(), e, new MarkerImpl(getURI().toString(), e.getProblemMark()));
071                } catch (RuntimeException | IOException e) {
072                        throw e;
073                } catch (Exception e) {
074                        throw new NasdanikaException(e);
075                }
076        }
077        
078        @Override
079        protected void doSave(OutputStream outputStream, Map<?, ?> options) throws IOException {
080                List<Object> data = new ArrayList<>();
081                for (EObject e: getContents()) {
082                        Storable storable = EObjectAdaptable.adaptTo(e, Storable.class);
083                        if (storable == null) {
084                                throw new IOException("Cannot adapt " + e + " to " + Storable.class.getName());
085                        }
086                        try {
087                                data.add(storable.store(new URL(getURI().toString()), progressMonitor));
088                        } catch (RuntimeException | IOException ex) {
089                                throw ex;
090                        } catch (Exception ex) {
091                                throw new NasdanikaException(ex);
092                        }
093                }
094                
095                DumperOptions dumperOptions = new DumperOptions();
096                dumperOptions.setDefaultFlowStyle(FlowStyle.BLOCK);
097                dumperOptions.setIndent(4);
098                Yaml yaml = new Yaml(dumperOptions);
099                yaml.dump(data.size() == 1 ? data.get(0) : data, new OutputStreamWriter(outputStream));         
100        }
101
102        /**
103         * If fragment is null returns the first content element.
104         */
105        @Override
106        public EObject getEObject(String uriFragment) {
107                if (uriFragment == null) {
108                        return getContents().get(0);
109                }
110                return super.getEObject(uriFragment);
111        }
112
113}
