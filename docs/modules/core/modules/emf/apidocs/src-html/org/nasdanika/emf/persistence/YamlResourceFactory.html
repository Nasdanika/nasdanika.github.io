001package org.nasdanika.emf.persistence;
002
003import java.net.MalformedURLException;
004import java.net.URL;
005import java.util.function.Function;
006
007import org.eclipse.emf.common.util.URI;
008import org.eclipse.emf.ecore.resource.Resource;
009import org.eclipse.emf.ecore.resource.impl.ResourceFactoryImpl;
010import org.nasdanika.common.Context;
011import org.nasdanika.common.MutableContext;
012import org.nasdanika.common.NasdanikaException;
013import org.nasdanika.common.ProgressMonitor;
014import org.nasdanika.common.Util;
015import org.nasdanika.common.persistence.ObjectLoader;
016
017/**
018 * Creates {@link YamlResource}
019 * @author Pavel
020 *
021 */
022public class YamlResourceFactory extends ResourceFactoryImpl {
023        
024        private ProgressMonitor progressMonitor;
025        private ObjectLoader loader;
026        private Context context;
027
028        public YamlResourceFactory(ObjectLoader loader, Context context, ProgressMonitor progressMonitor) {
029                this.loader = loader;
030                this.context = context;
031                this.progressMonitor = progressMonitor;
032        }
033
034        @Override
035        public Resource createResource(URI uri) {
036                return new YamlResource(uri, loader, resourceContext(uri), progressMonitor);
037        }
038        
039        /**
040         * Override to customize resource context, e.g. add uri-based properties or services.
041         * This imple
042         * @param uri
043         * @return
044         */
045        protected Context resourceContext(URI uri) {
046                MutableContext ret = context.fork();
047
048                Function<String, URL> resolver = path -> {
049                        try {
050                                return new URL(new URL(uri.toString()), path);
051                        } catch (MalformedURLException e) {
052                                throw new NasdanikaException("Cannot resolve '" + path + "' relative to " + uri + ": " + e, e);
053                        }
054                };
055                
056                ret.put("embedded-image", Util.createEmbeddedImagePropertyComputer(resolver));
057                ret.put("embedded-image-data", Util.createEmbeddedImageDataPropertyComputer(resolver));
058                ret.put("include", Util.createIncludePropertyComputer(resolver));
059                ret.put("include-markdown", Util.createIncludeMarkdownPropertyComputer(resolver));
060                
061                return ret;
062        }
063        
064}
