118        public static boolean isDefaultFeature(EClass eClass, EStructuralFeature feature) {
119                if ("true".equals(NcoreUtil.getNasdanikaAnnotationDetail(feature, IS_DEFAULT_FEATURE))) {
120                        return true;
121                }
122                String dfName = NcoreUtil.getNasdanikaAnnotationDetail(eClass, IS_DEFAULT_FEATURE);
123                if (!Util.isBlank(dfName)) {
124                        return feature.getName().equals(dfName);
125                }
126                for (EClass st: eClass.getEAllSuperTypes()) {
127                        dfName = NcoreUtil.getNasdanikaAnnotationDetail(st, IS_DEFAULT_FEATURE);
128                        if (!Util.isBlank(dfName)) {
129                                return feature.getName().equals(dfName);
130                        }                       
131                }
132                return false;
133        }

155        /**
156         * Default implementation of load key provider - gets load key from the Nasdanika annotation (urn:org.nasdanika) <code>load-key</code> details if it is present.
157         * For features delegates to {@link NcoreUtil} getFeatureKey.
158         * Otherwise converts {@link ENamedElement} element name from camel case to lower-cased kebab case. E.g. firstName is converted first-name.
159         */
160        public static final BiFunction<EClass,ENamedElement,String> LOAD_KEY_PROVIDER = (eClass,ne) -> {
161                EAnnotation na = NcoreUtil.getNasdanikaAnnotation(ne);
162                if (na != null) {
163                        if (na.getDetails().containsKey(IS_LOADABLE) && "false".equals(na.getDetails().get(IS_LOADABLE))) {
164                                return null;
165                        }
166                        if (na.getDetails().containsKey(LOAD_KEY)) {
167                                return na.getDetails().get(LOAD_KEY);
168                        }
169                }
170
171                if (eClass != null && ne instanceof EStructuralFeature) {
172                        EStructuralFeature feature = (EStructuralFeature) ne;
173                        String loadKey = getLoadKey(eClass, feature);           
174                        return loadKey == null ? NcoreUtil.getFeatureKey(eClass, feature) : loadKey;
175                }
176                
177                return Util.camelToKebab(ne.getName());
178        };

184                EPackage ePackage;

186                BiFunction<EClass,ENamedElement,String> keyProvider;

190                        this.keyProvider = keyProvider == null ? LOAD_KEY_PROVIDER : keyProvider;

234        public void register(EPackage ePackage, BiFunction<EClass,ENamedElement,String> keyProvider) {
235                if (keyProvider == null) {
236                        keyProvider = this.keyProvider;
237                }
238                String packageKey = keyProvider.apply(null, ePackage);
239                if (packageKey != null) {
240                        registry.put(packageKey + "-", new EPackageEntry(ePackage, keyProvider == null ? this.keyProvider : keyProvider));
241                }
242        }
243        
244        public void register(EPackage ePackage) {
245                register(ePackage, null);
246        }

248        @Override
249        public Object create(ObjectLoader loader, String type, Object config, URI base, ProgressMonitor progressMonitor, Marker marker) throws Exception {
250                BiSupplier<EClass,BiFunction<EClass,ENamedElement,String>> result = resolveEClass(type);
251                if (result != null) {
252                        return create(loader, result.getFirst(), config, base, progressMonitor, marker, result.getSecond());
253                }
254                
255                return super.create(loader, type, config, base, progressMonitor, marker);
256        }

258        /**
259         * Resolves EClass and key provider from type name.
260         * @param type
261         * @return
262         */
263        public BiSupplier<EClass,BiFunction<EClass,ENamedElement,String>> resolveEClass(String type) {          
264                for (Entry<String, EPackageEntry> re: registry.entrySet()) {
265                        if (type.startsWith(re.getKey())) {
266                                EPackageEntry ePackageEntry = re.getValue();
267                                for (EClassifier eClassifier: ePackageEntry.ePackage.getEClassifiers()) {
268                                        if (eClassifier instanceof EClass && "true".equals(NcoreUtil.getNasdanikaAnnotationDetail(eClassifier, IS_LOADABLE, "true"))) {
269                                                String eClassifierKey = ePackageEntry.keyProvider.apply(null, eClassifier);
270                                                if (eClassifierKey != null && type.equals(re.getKey() + eClassifierKey)) {
271                                                        return new BiSupplier<EClass, BiFunction<EClass,ENamedElement,String>>() {
272
273                                                                @Override
274                                                                public EClass getFirst() {
275                                                                        return (EClass) eClassifier;
276                                                                }
277
278                                                                @Override
279                                                                public BiFunction<EClass,ENamedElement, String> getSecond() {
280                                                                        return ePackageEntry.keyProvider;
281                                                                }
282                                                        }; 
283                                                }
284                                        }
285                                }
286                        }
287                }
288                return null;            
289        }

291        /**
292         * Creates an instance of EClass or a supplier factory
293         * @param loader
294         * @param eClass
295         * @param config
296         * @param base
297         * @param progressMonitor
298         * @param marker
299         * @param ePackageEntry
300         * @param eClassifier
301         * @return
302         * @throws Exception
303         */
304        public Object create(
305                        ObjectLoader loader, 
306                        EClass eClass, 
307                        Object config, 
308                        URI base, 
309                        ProgressMonitor progressMonitor, 
310                        Marker marker, 
311                        BiFunction<EClass,ENamedElement,String> keyProvider) throws Exception {
312                
313                // Nulls are nulls
314                if (config == null) {
315                        return null;
316                }
317                
318                // Proxy
319                Object proxy = createProxy(eClass, config, base, marker, progressMonitor);
320                if (proxy != null) {
321                        return proxy;
322                }
323                
324                Class<?> instanceClass = eClass.getInstanceClass();
325                EFactory factory = eClass.getEPackage().getEFactoryInstance();
326                
327                if (Loadable.class.isAssignableFrom(instanceClass)) {
328                        EObject eObject = factory.create(eClass);
329                        ((Loadable) eObject).load(loader, config, base, progressMonitor, marker);
330                        return eObject;
331                }
332                if (ContextLoadable.class.isAssignableFrom(instanceClass)) {
333                        return new SupplierFactory<EObject>() {
334
335                                @Override
336                                public Supplier<EObject> create(Context context) throws Exception {
337                                        return new Supplier<EObject>() {
338
339                                                @Override
340                                                public double size() {
341                                                        return 1;
342                                                }
343
344                                                @Override
345                                                public String name() {
346                                                        return "Creating " + eClass.getName();
347                                                }
348
349                                                @Override
350                                                public EObject execute(ProgressMonitor pm) throws Exception {
351                                                        EObject eObject = factory.create(eClass);
352                                                        ((ContextLoadable) eObject).load(context, loader, config, base, pm, marker);
353                                                        return eObject;
354                                                }
355                                                
356                                        };
357                                }                                                                       
358                        };
359                }                                                       
360                EObjectSupplierFactory eObjectSupplierFactory = new EObjectSupplierFactory(this, eClass, keyProvider);
361                eObjectSupplierFactory.load(loader, config, base, progressMonitor, marker);
362                return eObjectSupplierFactory;
363        }

365        /**
366         * Creates a proxy object if config is a singleton map with "href" key.
367         * @param eClass
368         * @param config
369         * @param base
370         * @return Proxy object or null if config is not a proxy config.
371         */
372        public EObject createProxy(EClass eClass, Object config, URI base, Marker marker, ProgressMonitor progressMonitor) {
373                if (config instanceof Map) {
374                        @SuppressWarnings("unchecked")
375                        Map<Object,Object> configMap = (Map<Object,Object>) config;
376                        if (configMap.size() == 1 
377                                        && configMap.containsKey(HREF_KEY)) {
378                                EObject eObject = eClass.getEPackage().getEFactoryInstance().create(eClass);
379                                if (eObject instanceof MinimalEObjectImpl) {
380                                        Object href = configMap.get(HREF_KEY);                                  
381                                        URI proxyURI;
382                                        if (href instanceof URI) {
383                                                proxyURI = (URI) href;
384                                        } else if (href instanceof String) {
385                                                proxyURI = URI.createURI((String) configMap.get(HREF_KEY));
386                                        } else {
387                                                return null;
388                                        }
389                                        
390                                        String proxyURIStr = proxyURI.toString();
391                                        if (proxyURIStr.startsWith(LATE_PROXY_RESOLUTION_URI_PREFIX)) {
392                                                proxyURI = URI.createURI(proxyURIStr.substring(LATE_PROXY_RESOLUTION_URI_PREFIX.length()));
393                                        } else if (base != null) {
394                                                proxyURI = proxyURI.resolve(base);
395                                        }
396                                        ((MinimalEObjectImpl) eObject).eSetProxyURI(proxyURI);
397                                        mark(eObject, marker, progressMonitor);
398                                        
399                                        return eObject;
400                                }
401                        }
402                }
403                return null;
404        }

406        /**
407         * If marker is not null adds marked adapter. If eObject is instance of {@link Marked} creates and sets {@link org.nasdanika.ncore.Marker}.
408         * @param eObject
409         * @param marker
410         */
411        public void mark(EObject eObject, Marker marker, ProgressMonitor progressMonitor) {
412                if (marker != null) {
413                        eObject.eAdapters().add(new MarkedAdapter(marker));
414                        if (eObject instanceof Marked) {                                
415                                org.nasdanika.ncore.Marker mMarker = markerFactory.apply(marker.getLocation(), progressMonitor);
416                                mMarker.setLine(marker.getLine());
417                                mMarker.setColumn(marker.getColumn());
418                                ((Marked) eObject).setMarker(mMarker);
419                        }
420                }
421        }

423        /**
424         * Resolves proxy using the loader's resource set.
425         * Does not suppress exceptions. 
426         */
427        public EObject resolve(EObject proxy) {
428                if (!proxy.eIsProxy()) {
429                        return proxy;
430                }
431
432                URI proxyURI = ((InternalEObject) proxy).eProxyURI();
433                EObject ret = resourceSet.getEObject(proxyURI, true);
434                return ret == null || ret == proxy ? ret : resolve(ret);
435        }

441        /**
442         * @return Load key from <code>load-keys</code> annotation. 
443         */
444        public static String getLoadKey(EClass eClass, EStructuralFeature feature) {
445                for (EClass ec: NcoreUtil.lineage(eClass)) {
446                        String loadKeys = NcoreUtil.getNasdanikaAnnotationDetail(ec, LOAD_KEYS);
447                        if (!Util.isBlank(loadKeys)) {
448                                Yaml yaml = new Yaml();
449                                Map<String,Object> loadKeyMap = yaml.load(loadKeys);
450                                Object loadKey = loadKeyMap.get(feature.getName());
451                                if (loadKey instanceof String) {
452                                        return (String) loadKey;
453                                }
454                        }
455                }
456                return null;
457        }
