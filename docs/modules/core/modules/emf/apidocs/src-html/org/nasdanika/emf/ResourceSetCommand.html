001package org.nasdanika.emf;
002
003import java.io.File;
004import java.util.ArrayList;
005import java.util.Collection;
006import java.util.List;
007import java.util.stream.Collectors;
008
009import org.eclipse.emf.common.util.URI;
010import org.eclipse.emf.ecore.EPackage;
011import org.eclipse.emf.ecore.EPackage.Registry;
012import org.eclipse.emf.ecore.resource.Resource;
013import org.eclipse.emf.ecore.resource.ResourceSet;
014import org.eclipse.emf.ecore.resource.impl.ResourceSetImpl;
015import org.eclipse.emf.ecore.xmi.impl.XMIResourceFactoryImpl;
016import org.nasdanika.cli.DelegatingCommand;
017
018import picocli.CommandLine.Option;
019
020/**
021 * Base class for command which use {@link ResourceSet}. This command creates a resource set,
022 * registers all known packages, 
023 * loads resources specified with -r option.
024 * @author Pavel
025 *
026 */
027public abstract class ResourceSetCommand extends DelegatingCommand {
028                
029        @Option(                        
030                        names = {"-r", "--resource"},
031                        paramLabel = "URL",
032                        description = "URL of a resource to load to the resource set relative to the current directory. ")
033        private List<String> resources = new ArrayList<>();
034        
035        /**
036         * @return {@link EPackage}'s to register with the resource set for loading the model.
037         * This implementation returns all registered packages.
038         */
039        protected Collection<EPackage> getEPackages() {
040                Registry registry = EPackage.Registry.INSTANCE;
041                return new ArrayList<String>(registry.keySet()).stream().map(nsURI -> registry.getEPackage(nsURI)).collect(Collectors.toList());
042        }
043        
044        /**
045         * Creates an empty resource set with registered packages and {@link XMIResourceFactoryImpl} and loads resource to it.
046         * @return
047         */
048        protected ResourceSet createResourceSet() {
049                ResourceSet resourceSet = createEmptyResourceSet();
050
051                for (String uri: resources) {
052                        URI resourceUri = URI.createURI(new File(".").toURI().resolve(uri).toString());
053                        resourceSet.getResource(resourceUri, true);
054                }
055                return resourceSet;             
056        }
057
058        /**
059         * Creates a {@link ResourceSet} with all known packages registered and with {@link XMIResourceFactoryImpl}
060         * @return
061         */
062        protected ResourceSet createEmptyResourceSet() {
063                ResourceSetImpl resourceSet = new ResourceSetImpl();
064                resourceSet.getResourceFactoryRegistry().getExtensionToFactoryMap().put(Resource.Factory.Registry.DEFAULT_EXTENSION, new XMIResourceFactoryImpl());
065                for (EPackage ePackage: getEPackages()) {
066                        resourceSet.getPackageRegistry().put(ePackage.getNsURI(), ePackage);
067                }
068                return resourceSet;
069        }       
070        
071}
