031/**
032 * Creates reference value/element - {@link EObject} or proxy.
033 * @author Pavel
034 *
035 */
036public class ReferenceFactory<T> implements ObjectFactory<T> {
037        
038        private EObjectLoader resolver;
039        private BiFunction<EClass,ENamedElement, String> keyProvider;
040        private boolean isStrictContainment;
041        private boolean referenceSupplierFactory;
042        private boolean resolveProxies;
043        private boolean isHomogenous;
044        private EReference eReference;
045        private EClass eClass;
046        private Object eReferenceKey;
047        
048        public EReference getEReference() {
049                return eReference;
050        }
051        
052        public boolean isHomogenous() {
053                return isHomogenous;
054        }
055
056        /**
057         * 
058         * @param eClass
059         * @param eReference
060         * @param eReferenceKey Reference key for loading type from annotations. If null, defautls to eReference name. Used by EMap's.
061         * @param resolver
062         * @param referenceSupplierFactory
063         * @param keyProvider
064         */
065        public ReferenceFactory(
066                        EClass eClass,
067                        EReference eReference,
068                        String eReferenceKey,  
069                        EObjectLoader resolver,
070                        boolean referenceSupplierFactory,
071                        BiFunction<EClass,ENamedElement,String> keyProvider) {
072                
073                this.eClass = eClass;
074                this.eReference = eReference;
075                this.eReferenceKey = Util.isBlank(eReferenceKey) ? eReference.getName() : eReferenceKey;
076                this.resolver = resolver;
077                this.referenceSupplierFactory = referenceSupplierFactory;
078                this.resolveProxies = !eReference.isResolveProxies();
079                this.keyProvider = keyProvider;
080                
081                this.isHomogenous = "true".equals(NcoreUtil.getNasdanikaAnnotationDetail(eReference, EObjectLoader.IS_HOMOGENOUS)) || NcoreUtil.getNasdanikaAnnotationDetail(eReference, EObjectLoader.REFERENCE_TYPE) != null;                 
082                this.isStrictContainment = isHomogenous && "true".equals(NcoreUtil.getNasdanikaAnnotationDetail(eReference, EObjectLoader.IS_STRICT_CONTAINMENT));                      
083        }
084        
085        @SuppressWarnings("unchecked")
086        @Override
087        public T create(ObjectLoader loader, Object element, URI base, ProgressMonitor progressMonitor, Marker marker) throws Exception {
088                try {
089                        // Strings are references if not strict containment.
090                        if (element instanceof String && !isStrictContainment) {
091                                if (referenceSupplierFactory) {
092                                        return (T) new SupplierFactory<EObject>() {
093        
094                                                @Override
095                                                public Supplier<EObject> create(Context context) throws Exception {
096                                                        String ref = context.interpolateToString((String) element);
097                                                        return new Supplier<EObject>() {
098        
099                                                                @Override
100                                                                public double size() {
101                                                                        return 1;
102                                                                }
103        
104                                                                @Override
105                                                                public String name() {
106                                                                        return "Loading " + ref;
107                                                                }
108        
109                                                                @Override
110                                                                public EObject execute(ProgressMonitor progressMonitor) throws Exception {
111                                                                        return (EObject) loadReference(ref, base, marker, progressMonitor);
112                                                                }
113                                                                
114                                                        };
115                                                }
116                                                
117                                        };
118                                }
119                                return loadReference((String) element, base, marker, progressMonitor);
120                        }
121                        Object ret = isHomogenous ? resolver.create(loader, effectiveReferenceType(element), element, base, progressMonitor, marker, keyProvider) : loader.load(element, base, progressMonitor);
122                        if (resolveProxies && ret instanceof EObject && ((EObject) ret).eIsProxy()) {
123                                return (T) resolver.resolve((EObject) ret);
124                        }
125                        return (T) ret;
126                } catch (ConfigurationException e) {
127                        throw e;
128                } catch (Exception e) {
129                        if (marker == null) {
130                                throw e;
131                        }
132                        throw new ConfigurationException("Error loading reference: " + e, e, marker);
133                }
134        }
135        
136        /**
137         * Loads reference. Creates a proxy if reference type is not abstract and resolve proxies is true.
138         * @param ref
139         * @param base
140         * @param marker
141         * @return
142         */
143        @SuppressWarnings("unchecked")
144        protected T loadReference(
145                        String ref, 
146                        URI base,
147                        Marker marker, 
148                        ProgressMonitor progressMonitor) {
149                
150                URI refURI = URI.createURI(ref);
151                if (base != null && !ref.startsWith(EObjectLoader.LATE_PROXY_RESOLUTION_URI_PREFIX)) {
152                        refURI = refURI.resolve(URI.createURI(base.toString()));
153                }
154                ConfigurationException.pushThreadMarker(marker);
155                try {
156                        EClass eReferenceType = effectiveReferenceType(ref);
157                        if (!eReferenceType.isAbstract() && !resolveProxies) {
158                                // Can create proxy, if possible, instead of loading object
159                                EObject proxy = resolver.createProxy(eReferenceType, Collections.singletonMap(EObjectLoader.HREF_KEY, refURI), base, marker, progressMonitor);
160                                if (proxy != null) {
161                                        if ( marker != null) {
162                                                proxy.eAdapters().add(new MarkedAdapter(marker));
163                                        }
164                                        return (T) proxy;
165                                }
166                        }
167                        return (T) resolver.getResourceSet().getEObject(refURI, true);
168                } finally {
169                        ConfigurationException.popThreadMarker();
170                }
171        }
172        
173        protected String referenceTypeName(Object element) {
174                if (element instanceof String) {
175                        return "string";
176                }
177                if (element instanceof Integer) {
178                        return "integer";
179                }
180                if (element instanceof Map) {
181                        return "map";
182                }
183                if (element instanceof Collection) {
184                        return "list";
185                }
186                if (element instanceof Boolean) {
187                        return "boolean";
188                }
189                if (element instanceof Date) {
190                        return "date";
191                }
192                // More types?
193                return null;
194        }
195                
196        /**
197         * @return Effective reference type which defaults to the reference type, but can be overridden by the <code>reference-types</code> annotation. 
198         */
199        public EClass effectiveReferenceType(Object element) {
200                EGenericType eGenericReferenceType = eClass.getFeatureType(eReference);
201                EClass eReferenceType = (EClass) eGenericReferenceType.getEClassifier();
202                
203                String rTypes = NcoreUtil.getNasdanikaAnnotationDetail(eReference, EObjectLoader.REFERENCE_TYPE);
204                if (!Util.isBlank(rTypes)) {
205                        Yaml yaml = new Yaml();
206                        Map<String,Object> rTypesMap = yaml.load(rTypes);
207                        String valueFeature = EObjectLoader.getValueFeature(eReference);
208                        Object refType;
209                        if (!Util.isBlank(valueFeature) && element instanceof Map) {
210                                refType = rTypesMap.get(referenceTypeName(((Map<?,?>) element).get(valueFeature)));
211                        } else {
212                                refType = rTypesMap.get(referenceTypeName(element));                            
213                        }
214                        if (refType != null) {
215                                return resolveEClass(refType, eReference.getEContainingClass().getEPackage());
216                        }                               
217                }
218                
219                for (EClass ec: NcoreUtil.lineage(eClass)) {
220                        String refTypes = NcoreUtil.getNasdanikaAnnotationDetail(ec, EObjectLoader.REFERENCE_TYPES);
221                        if (!Util.isBlank(refTypes)) {
222                                Yaml yaml = new Yaml();
223                                Map<String,Object> refTypesMap = yaml.load(refTypes);
224                                Object refType = refTypesMap.get(eReferenceKey);
225                                if (refType != null) {
226                                        return resolveEClass(refType, ec.getEPackage());
227                                }                               
228                        }
229                }
230                return eReferenceType;
231        }       
232        
233        public EStructuralFeature effectiveDefaultFeature(Object element) { 
234                EClass effectiveReferenceType = effectiveReferenceType(element);
235                for (EStructuralFeature feature: effectiveReferenceType.getEAllStructuralFeatures()) {
236                        if (EObjectLoader.isDefaultFeature(effectiveReferenceType, feature)) {
237                                return feature;
238                        }
239                }
240                return null;
241        }
242        
243        protected EClass resolveEClass(Object typeSpec, EPackage ePackage) {
244                String typeName;
245                if (typeSpec instanceof String) {
246                        typeName = (String) typeSpec;
247                } else if (typeSpec instanceof Map) {
248                        @SuppressWarnings("unchecked")
249                        Map<String, Object> refTypeMap = (Map<String,Object>) typeSpec;
250                        Object nsURI = refTypeMap.get("ns-uri");
251                        if (nsURI instanceof String) {
252                                ResourceSet resourceSet = resolver.getResourceSet();
253                                Registry packageRegistry = resourceSet.getPackageRegistry();
254                                ePackage = packageRegistry.getEPackage((String) nsURI); 
255                                if (ePackage == null) {
256                                        throw new IllegalArgumentException("EPackage not found: " + nsURI);                                                                                                     
257                                }
258                                Object className = refTypeMap.get("name");
259                                if (className instanceof String) {
260                                        typeName = (String) className;
261                                } else {
262                                        throw new IllegalArgumentException("Reference type specification name shall be a String, got: " + className);                                                                                                   
263                                }
264                        } else {
265                                throw new IllegalArgumentException("Reference type specification ns-uri shall be a String, got: " + nsURI);                                             
266                        }                                       
267                } else {
268                        throw new IllegalArgumentException("Unsupported reference type specification: " + typeSpec.getClass() + ": " + typeSpec);
269                }                                       
270                
271                EClass ret = (EClass) ePackage.getEClassifier(typeName);
272                if (ret == null) {
273                        throw new IllegalArgumentException("Reference type " + typeName + " not found in EPackage " + ePackage.getNsURI());                                             
274                }
275                return ret;
276        }
