001package org.nasdanika.cli;
002
003import java.io.File;
004import java.io.FilterOutputStream;
005import java.io.IOException;
006import java.io.InputStream;
007import java.io.PrintStream;
008import java.util.ArrayList;
009import java.util.Collections;
010import java.util.List;
011import java.util.stream.Collectors;
012
013import org.fusesource.jansi.HtmlAnsiOutputStream;
014import org.nasdanika.common.DefaultConverter;
015import org.nasdanika.common.MarkdownHelper;
016import org.nasdanika.common.Util;
017
018import picocli.CommandLine;
019import picocli.CommandLine.Command;
020import picocli.CommandLine.Help;
021import picocli.CommandLine.Model.CommandSpec;
022import picocli.CommandLine.Option;
023
024@Command(
025                description = "Outputs usage for all registred commands to console or file as plain text or HTML",
026                name = "help")
027public class HelpCommand extends CommandBase {
028        
029        private CommandLine root;
030        
031        public HelpCommand(CommandLine root) {
032                this.root = root;
033        }
034        
035        @Option(names = {"-o", "--output"}, description = "Output file")
036        private File output;
037        
038        @Option(names = {"-H", "--html"}, defaultValue = "false", description = "Output to HTML with ANSI styling")
039        private boolean html;
040        
041        @Option(names = {"-l", "--header-level"}, defaultValue = "1", description = "Starting level for HTML header tags in HTML output, the default value is ${DEFAULT_VALUE}.")
042        private int level;
043        
044        private static final int WIDTH = 80;
045        
046        private void usage(
047                        List<String> cmdPath, 
048                        CommandLine cmd,
049                        PrintStream out) throws IOException {
050                
051                CommandSpec commandSpec = cmd.getCommandSpec();
052                if (html) {
053                        int hLevel = Math.min(cmdPath.size() + level, 6);
054                        out.println("<h" + hLevel + ">" + cmd.getCommandName() + "</h" + hLevel + ">");
055                        out.print("<table><tr><th valign=\"top\">Version:</th><td> ");
056                        for (String vs: commandSpec.version()) {
057                                out.print(vs);
058                                out.print("<br/>");
059                        }
060                        out.print("</td></tr></table>");
061                        
062                        out.println("<pre style=\"background:black;color:white;padding:5px;width:max-content\">");
063                        try (HtmlAnsiOutputStream haos = new HtmlAnsiOutputStream(new FilterOutputStream(out) { @Override public void close() {} })) {
064                                try (PrintStream ps = new PrintStream(haos)) {
065                                        cmd.usage(ps, Help.Ansi.ON);
066                                }
067                                haos.flush();
068                        }
069                        out.println("</pre>");
070
071                        // Description 
072                        Object userObject = commandSpec.userObject();
073                        if (userObject != null) {
074                                Class<? extends Object> clazz = userObject.getClass();
075                                Description description = clazz.getAnnotation(Description.class);
076                                if (description != null) {
077                                        String dResource = description.value();
078                                        if (Util.isBlank(dResource)) {
079                                                dResource = clazz.getName().substring(clazz.getName().lastIndexOf('.') + 1) + ".md";                                            
080                                        }
081                                        InputStream dStream = clazz.getResourceAsStream(dResource);
082                                        if (dStream != null) {
083                                                String markdown = DefaultConverter.INSTANCE.toString(dStream);
084                                                out.println("<div class=\"markdown-body\">" + MarkdownHelper.INSTANCE.markdownToHtml(markdown) + "</div>");
085                                        }
086                                }
087                        }
088                } else {
089                        out.println();
090                        for (int i = 0; i < WIDTH; ++i) {
091                                out.print("*");
092                        }
093                        out.println();
094                        out.print("* ");
095                        StringBuilder hb = new StringBuilder();
096                        for (String ps: cmdPath) {
097                                hb.append(ps).append(" > ");
098                        }
099                        hb.append(cmd.getCommandName());
100                        out.print(center(hb, WIDTH - 4));
101                        out.println(" *");
102                        
103                        for (String vs: commandSpec.version()) {
104                                out.print("* ");
105                                out.print(center(vs, WIDTH - 4));
106                                out.println(" *");                      
107                        }
108                        
109                        for (int i = 0; i < WIDTH; ++i) {
110                                out.print("*");
111                        }
112                        out.println();
113                        cmd.usage(out);
114                }               
115                
116                List<String> cPath = new ArrayList<>(cmdPath);
117                cPath.add(cmd.getCommandName());
118                for (CommandLine subCommand: cmd.getSubcommands().values().stream().sorted((a,b) -> a.getCommandName().compareTo(b.getCommandName())).collect(Collectors.toList())) {
119                        usage(cPath, subCommand, out);
120                }
121        }
122        
123        private String center(CharSequence text, int width) {
124                StringBuilder sb = new StringBuilder();
125                int toPad = width - text.length();
126                for (int i=0, padding = toPad/2; i < padding; ++i) {
127                        sb.append(" ");
128                }
129                sb.append(text);
130                while (sb.length() < width) {
131                        sb.append(" ");
132                }
133                return sb.toString();
134        }
135        
136        @Override
137        public Integer call() throws Exception {
138                if (output == null) {
139                        usage(Collections.emptyList(), root, System.out);
140                } else {
141                        try (PrintStream out = new PrintStream(output)) {
142                                usage(Collections.emptyList(), root, out);
143                        }
144                }
145                return 0;
146        }
147
148}
