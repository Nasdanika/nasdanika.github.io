001package org.nasdanika.cli;
002
003import picocli.CommandLine.Model.CommandSpec;
004import picocli.CommandLine.ParameterException;
005import picocli.CommandLine.ParentCommand;
006import picocli.CommandLine.Spec;
007
008/**
009 * A base class for commands which are used for grouping and do not provide 
010 * own functionality but may contain options which can be accessed by 
011 * their children via a field annotated with {@link ParentCommand}. 
012 * @author Pavel
013 *
014 */
015public class CommandGroup extends CommandBase {
016        
017        @Spec CommandSpec spec;
018
019        @Override
020        public Integer call() throws Exception {
021                throw new ParameterException(spec.commandLine(), "Missing required subcommand");
022        }
023        
024
025}
