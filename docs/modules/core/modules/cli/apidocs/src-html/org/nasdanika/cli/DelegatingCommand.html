001package org.nasdanika.cli;
002
003import java.util.concurrent.ExecutorService;
004import java.util.concurrent.Executors;
005import java.util.concurrent.TimeUnit;
006
007import org.nasdanika.common.CommandFactory;
008import org.nasdanika.common.Context;
009import org.nasdanika.common.Diagnostic;
010import org.nasdanika.common.DiagnosticException;
011import org.nasdanika.common.ProgressMonitor;
012import org.nasdanika.common.Status;
013
014import picocli.CommandLine.Command;
015import picocli.CommandLine.Mixin;
016import picocli.CommandLine.Option;
017
018/**
019 * Base class for commands which create {@link Context} and {@link ProgressMonitor} and
020 * then delegate execution to {@link Command} created by {@link CommandFactory} obtained from
021 * <code>getCommandFactory()</code> abstract protected method which subclasses shall override.
022 * This command executes all life cycle methods of the target command - diagnose, execute, commit or rollback, close.
023 * Progress can be reported to console of file in text or json format.
024 * 
025 * @author Pavel
026 *
027 */
028@Command(
029                exitCodeListHeading = "Exit codes:\n",
030                exitCodeList = {
031                                "0:Success",
032                                "1:Unhandled exception during execution",
033                                "2:Invalid input",
034                                "3:Diagnostic failed",
035                                "4:Execution failed or was cancelled, successful rollback",
036                                "5:Execution failed or was cancelled, rollback was not successful",
037                                "6:Executor service termination timed out"
038                })
039public abstract class DelegatingCommand extends ContextCommand {
040        
041        protected abstract CommandFactory getCommandFactory();
042        
043        @Mixin
044        private ProgressMonitorMixin progressMonitorMixin;
045                
046        @Option(
047                        names = { "-P", "--parallelism" }, 
048                        defaultValue = "1", 
049                        description = "If the value greater than one then an executor service is created and injected into the context to allow concurrent execution.")
050        private int parallelism;
051        
052        @Option(
053                        names = { "-t", "--timeout" }, 
054                        defaultValue = "60", 
055                        description = "If parallelism is greater than one this option specifies timout in seconds awaiting completion of execution. "
056                                        + "Default value is ${DEFAULT-VALUE}.")
057        private int timeout;            
058        
059        @Override
060        protected Context createContext(ProgressMonitor progressMonitor) throws Exception {
061                Context context = super.createContext(progressMonitor);
062                if (parallelism <= 1) {
063                        return context;
064                }
065                return context.compose(Context.singleton(ExecutorService.class, Executors.newWorkStealingPool(parallelism)));
066        }
067        
068        @Override
069        public Integer call() throws Exception {                
070                try (ProgressMonitor progressMonitor = progressMonitorMixin.createProgressMonitor(4)) {
071                        Context context = createContext(progressMonitor.split("Creating context", 1));
072                        try (org.nasdanika.common.Command delegate = getCommandFactory().create(context)) {
073                                ProgressMonitor commandMonitor = progressMonitor.split("Command monitor", 3).setWorkRemaining(3 * delegate.size());
074                                Diagnostic diagnostic = delegate.splitAndDiagnose(commandMonitor);
075                                if (diagnostic.getStatus() == Status.ERROR) {
076                                        System.err.println("Diagnostic failed");
077                                        diagnostic.dump(System.err, 4);
078                                        return 3;
079                                }
080                                
081                                try {
082                                        delegate.splitAndExecute(commandMonitor);
083                                        delegate.splitAndCommit(commandMonitor);
084                                } catch (Exception e) {
085                                        reportException(e);
086                                        if (e instanceof DiagnosticException) {
087                                                ((DiagnosticException) e).getDiagnostic().dump(System.err, 4);
088                                        }
089                                        return delegate.splitAndRollback(commandMonitor) ? 4 : 5;
090                                }
091                        } finally {
092                                ExecutorService executorService = context.get(ExecutorService.class);
093                                if (executorService != null) {
094                                        executorService.shutdown();
095                                        return executorService.awaitTermination(timeout, TimeUnit.SECONDS) ? 0 : 6;
096                                }
097                        }
098                }
099                return 0;
100        }
101
102        protected void reportException(Exception e) {
103                System.err.println("Exception during command execution or commit: "+e.getMessage());
104                e.printStackTrace();
105        }
106        
107}
