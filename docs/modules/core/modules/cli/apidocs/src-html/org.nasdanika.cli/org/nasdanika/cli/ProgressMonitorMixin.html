001package org.nasdanika.cli;
002
003import java.io.File;
004import java.io.FileWriter;
005import java.io.IOException;
006import java.io.PrintStream;
007import java.io.Writer;
008
009import org.nasdanika.common.PrintStreamProgressMonitor;
010import org.nasdanika.common.ProgressMonitor;
011import org.nasdanika.common.ProgressRecorder;
012
013import picocli.CommandLine.Option;
014
015/**
016 * Mixes in progress monitor.
017 * @author Pavel
018 *
019 */
020public class ProgressMonitorMixin {
021        
022        @Option(names = {"-p", "--progress"}, description = "Output file for progress monitor")
023        private File progressOutput;
024        
025        @Option(names = {"-j", "--json"}, description = "Output progress in JSON")
026        private boolean jsonProgress;
027        
028        public ProgressMonitor createProgressMonitor(double size) throws Exception {
029                if (progressOutput == null) {
030                        if (jsonProgress) {
031                                return new ProgressRecorder() {
032                                        
033                                        @Override
034                                        public void close() {
035                                                System.out.println(toString());
036                                                super.close();
037                                        }
038                                        
039                                };                      
040                        }
041                        return new PrintStreamProgressMonitor(System.out, 0, 4, false);
042                }
043                
044                if (jsonProgress) {
045                        return new ProgressRecorder() {
046                                
047                                @Override
048                                public void close() {
049                                        try (Writer w = new FileWriter(progressOutput)) {
050                                                toJSON(0, true).write(w, 4, 0);
051                                        } catch (IOException e) {
052                                                System.err.println("Error closing progress writer");
053                                                e.printStackTrace();
054                                        }
055                                        super.close();
056                                }
057                                
058                        };                      
059                }
060                return new PrintStreamProgressMonitor(new PrintStream(progressOutput), 0, 4, true);             
061        }       
062
063}
