001package org.nasdanika.cli;
002
003import java.lang.annotation.ElementType;
004import java.lang.annotation.Retention;
005import java.lang.annotation.RetentionPolicy;
006import java.lang.annotation.Target;
007
008/**
009 * Specifies a classloader markdown resource with a detailed description of a command.
010 * The description is output to HTML documentation below the usage information.
011 * By default, if value attribute is not set, a resource with the same name as the annotated type and <code>.md</code> extension is used for description if it exists. 
012 * @author Pavel
013 *
014 */
015@Retention(RetentionPolicy.RUNTIME)
016@Target(ElementType.TYPE)
017public @interface Description {
018
019        /**
020         * Description resource. 
021         * @return
022         */
023        String value() default "";
024        
025        /**
026         * Documentation icon. Treated as URL if there is a slash and as a CSS class otherwise.
027         * @return
028         */
029        String icon() default "";
030        
031        /**
032         * Documentation tooltip.
033         * @return
034         */
035        String tooltip() default "";
036        
037}
