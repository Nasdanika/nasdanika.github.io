001package org.nasdanika.cli;
002
003import java.io.File;
004import java.io.IOException;
005import java.io.InputStream;
006import java.io.PrintStream;
007import java.util.ArrayList;
008import java.util.Collections;
009import java.util.List;
010import java.util.stream.Collectors;
011
012import org.nasdanika.common.DefaultConverter;
013import org.nasdanika.common.MarkdownHelper;
014import org.nasdanika.common.Util;
015
016import picocli.CommandLine;
017import picocli.CommandLine.Command;
018import picocli.CommandLine.Model.CommandSpec;
019import picocli.CommandLine.Option;
020
021@Command(
022                description = "Outputs usage for all registred commands to console or file as plain text or HTML",
023                name = "help")
024public class HelpCommand extends CommandBase {
025        
026        private CommandLine root;
027        
028        public HelpCommand(CommandLine root) {
029                this.root = root;
030        }
031        
032        @Option(names = {"-o", "--output"}, description = "Output file")
033        private File output;
034        
035        @Option(names = {"-H", "--html"}, defaultValue = "false", description = "Output to HTML with ANSI styling")
036        private boolean html;
037        
038        @Option(names = {"-l", "--header-level"}, defaultValue = "1", description = "Starting level for HTML header tags in HTML output, the default value is ${DEFAULT_VALUE}.")
039        private int level;
040        
041        private static final int WIDTH = 80;
042        
043        private void usage(
044                        List<String> cmdPath, 
045                        CommandLine cmd,
046                        PrintStream out) throws IOException {
047                
048                CommandSpec commandSpec = cmd.getCommandSpec();
049                if (html) {
050                        int hLevel = Math.min(cmdPath.size() + level, 6);
051                        out.println("<h" + hLevel + ">" + cmd.getCommandName() + "</h" + hLevel + ">");
052                        out.print("<table><tr><th valign=\"top\">Version:</th><td> ");
053                        for (String vs: commandSpec.version()) {
054                                out.print(vs);
055                                out.print("<br/>");
056                        }
057                        out.print("</td></tr></table>");
058                        
059                        out.println("<pre style=\"background:black;color:white;padding:5px;width:max-content\">");
060                        out.println("TODO - implement, HtmlAnsiOutputStream is not available in jansi anymore");
061//                      try (HtmlAnsiOutputStream haos = new HtmlAnsiOutputStream(new FilterOutputStream(out) { @Override public void close() {} })) {
062//                              try (PrintStream ps = new PrintStream(haos)) {
063//                                      cmd.usage(ps, Help.Ansi.ON);
064//                              }
065//                              haos.flush();
066//                      }
067                        out.println("</pre>");
068
069                        // Description 
070                        Object userObject = commandSpec.userObject();
071                        if (userObject != null) {
072                                Class<? extends Object> clazz = userObject.getClass();
073                                Description description = clazz.getAnnotation(Description.class);
074                                if (description != null) {
075                                        String dResource = description.value();
076                                        if (Util.isBlank(dResource)) {
077                                                dResource = clazz.getName().substring(clazz.getName().lastIndexOf('.') + 1) + ".md";                                            
078                                        }
079                                        InputStream dStream = clazz.getResourceAsStream(dResource);
080                                        if (dStream != null) {
081                                                String markdown = DefaultConverter.INSTANCE.toString(dStream);
082                                                out.println("<div class=\"markdown-body\">" + MarkdownHelper.INSTANCE.markdownToHtml(markdown) + "</div>");
083                                        }
084                                }
085                        }
086                } else {
087                        out.println();
088                        for (int i = 0; i < WIDTH; ++i) {
089                                out.print("*");
090                        }
091                        out.println();
092                        out.print("* ");
093                        StringBuilder hb = new StringBuilder();
094                        for (String ps: cmdPath) {
095                                hb.append(ps).append(" > ");
096                        }
097                        hb.append(cmd.getCommandName());
098                        out.print(center(hb, WIDTH - 4));
099                        out.println(" *");
100                        
101                        for (String vs: commandSpec.version()) {
102                                out.print("* ");
103                                out.print(center(vs, WIDTH - 4));
104                                out.println(" *");                      
105                        }
106                        
107                        for (int i = 0; i < WIDTH; ++i) {
108                                out.print("*");
109                        }
110                        out.println();
111                        cmd.usage(out);
112                }               
113                
114                List<String> cPath = new ArrayList<>(cmdPath);
115                cPath.add(cmd.getCommandName());
116                for (CommandLine subCommand: cmd.getSubcommands().values().stream().sorted((a,b) -> a.getCommandName().compareTo(b.getCommandName())).collect(Collectors.toList())) {
117                        usage(cPath, subCommand, out);
118                }
119        }
120        
121        private String center(CharSequence text, int width) {
122                StringBuilder sb = new StringBuilder();
123                int toPad = width - text.length();
124                for (int i=0, padding = toPad/2; i < padding; ++i) {
125                        sb.append(" ");
126                }
127                sb.append(text);
128                while (sb.length() < width) {
129                        sb.append(" ");
130                }
131                return sb.toString();
132        }
133        
134        @Override
135        public Integer call() throws Exception {
136                if (output == null) {
137                        usage(Collections.emptyList(), root, System.out);
138                } else {
139                        try (PrintStream out = new PrintStream(output)) {
140                                usage(Collections.emptyList(), root, out);
141                        }
142                }
143                return 0;
144        }
145
146}
