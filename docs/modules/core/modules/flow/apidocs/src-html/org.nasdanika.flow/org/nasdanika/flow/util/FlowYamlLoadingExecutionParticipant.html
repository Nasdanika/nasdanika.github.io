001package org.nasdanika.flow.util;
002
003import java.util.Collection;
004
005import org.eclipse.emf.ecore.EPackage;
006import org.eclipse.emf.ecore.resource.ResourceSet;
007import org.nasdanika.common.Context;
008import org.nasdanika.common.ProgressMonitor;
009import org.nasdanika.emf.persistence.YamlResourceFactory;
010import org.nasdanika.exec.gen.ExecutionParticpantAdapterFactory;
011import org.nasdanika.exec.util.ExecYamlLoadingExecutionParticipant;
012import org.nasdanika.flow.FlowPackage;
013
014/**
015 * {@link YamlLoadingSupplier} for Flow {@link EPackage}.
016 * @author Pavel
017 *
018 */
019public abstract class FlowYamlLoadingExecutionParticipant extends ExecYamlLoadingExecutionParticipant {
020
021        public FlowYamlLoadingExecutionParticipant(Context context) {
022                super(context);
023        }
024
025        @Override
026        protected Collection<EPackage> getEPackages() {
027                Collection<EPackage> ret = super.getEPackages();
028                ret.add(FlowPackage.eINSTANCE);
029                return ret;
030        }
031        
032        @Override
033        protected YamlResourceFactory createYamlResorceFactory(ResourceSet resourceSet, ProgressMonitor progressMonitor) {
034                resourceSet.getAdapterFactories().add(new ExecutionParticpantAdapterFactory());
035                return super.createYamlResorceFactory(resourceSet, progressMonitor);
036        }
037        
038}
