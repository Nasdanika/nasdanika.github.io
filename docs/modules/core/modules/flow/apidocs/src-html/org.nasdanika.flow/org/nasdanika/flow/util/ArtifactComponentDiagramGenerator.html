037        public void generateDiagram(Artifact artifact, Diagram diagram) {
038                // Collecting all elements to be included
039                Map<Artifact,Integer> semanticElements = new HashMap<>();
040                int context = diagram.getContext();
041                if (context == -1) {
042                        // Only child elements
043                        for (Artifact child: artifact.getChildren().values()) {
044                                collectRelatedElements(child, semanticElements, 0);                             
045                        }
046                } else {
047                        collectRelatedElements(artifact, semanticElements, context);
048                }
049                
050                // Selecting top-level elements
051                Collection<Artifact> topLevelElements = new HashSet<>();
052                semanticElements.keySet().stream()
053                        .filter(se -> !EcoreUtil.isAncestor(topLevelElements, se))
054                        .forEach(se -> {
055                                topLevelElements.removeIf(topLevelElement -> EcoreUtil.isAncestor(se, topLevelElement));
056                                topLevelElements.add(se);
057                        });
058                
059                Map<Artifact,DiagramElement> semanticMap = new HashMap<>();
060                diagram.getElements().addAll(createDiagramElements(topLevelElements, semanticMap, context == 0 ? null : artifact, diagram.getDepth()));
061                
062                for (Artifact se: topLevelElements) {
063                        wire(se, semanticMap);
064                }
065        }
066
067        private void collectRelatedElements(Artifact semanticElement, Map<Artifact, Integer> accumulator, int depth) {
068                if (semanticElement == null) {
069                        return;
070                }
071                Integer sDepth = accumulator.get(semanticElement);
072                if (sDepth != null && (sDepth < 0 || depth <= sDepth)) {
073                        return; // Do not traverse if existing depth is negative or depth is less than already traversed depth.
074                }
075                accumulator.put(semanticElement, depth);
076                if (depth != 0) {
077                        for (Relationship inboundRelationship: semanticElement.getInboundRelationships()) {
078                                collectRelatedElements((Artifact) inboundRelationship.eContainer().eContainer(), accumulator, depth - 1);
079                        }
080
081                        for (Relationship ouboundRelationship: semanticElement.getOutboundRelationships().values()) {
082                                collectRelatedElements(ouboundRelationship.getTarget(), accumulator, depth - 1);
083                        }
084                }
085                
086                for (Artifact child: semanticElement.getChildren().values()) {
087                        collectRelatedElements(child, accumulator, depth);
088                }
089        }

091        protected DiagramElement createDiagramElement(
092                        Artifact semanticElement, 
093                        Map<Artifact, DiagramElement> semanticMap, 
094                        Artifact contextElement,
095                        int depth) {
096
097                DiagramElement template = semanticElement.getStyle();           
098                org.nasdanika.diagram.DiagramElement ret = template == null ? diagramFactory.createDiagramElement() : EcoreUtil.copy(template);
099                semanticMap.put(semanticElement, ret);
100                if (Util.isBlank(ret.getType())) {
101                        ret.setType("component");
102                }
103                
104                ret.setText(semanticElement.getName());
105                ret.setLocation(getArtifactLocation(semanticElement));
106                ret.setTooltip(getArtifactTooltip(semanticElement));
107                        
108                if (depth != 0 && semanticElement.isPartition()) {
109                        ret.getElements().addAll(createDiagramElements(semanticElement.getChildren().values(), semanticMap, contextElement, depth - 1));
110                }
111        
112                if (template == null) {
113                        EList<String> modifiers = semanticElement.getModifiers();
114                        if (modifiers.contains("final")) {
115                                ret.setBold(true);
116                        } else if (modifiers.contains("abstract")) {
117                                ret.setDashed(true);
118                        }
119                        
120                        if (modifiers.contains("optional")) {
121                                ret.setBorder("grey");
122                        }
123                }
124                
125                if (contextElement == null || semanticElement == contextElement || EcoreUtil.isAncestor(contextElement, semanticElement)) {
126                        if (Util.isBlank(ret.getColor())) {
127                                ret.setColor("FEFECE");                                 
128                        }
129                } else {
130                        ret.setColor("DDDDDD");
131                }
132                
133                setProperties(semanticElement, ret);
134                return ret;
135        }

137        /**
138         * Copies semantic element properties under "representation" property, if it is present, to style properties. 
139         * @param semanticElement
140         * @param style
141         */
142        public static void setProperties(PackageElement<?> semanticElement, Style style) {
143                for (Property property: semanticElement.getProperties()) {
144                        if ("representation".equals(property.getName()) && property instanceof MapProperty) {
145                                style.getProperties().addAll(EcoreUtil.copyAll(((MapProperty) property).getValue()));
146                        }
147                }
148        }

150        /**
151         * Creates diagram for flow elements potentially grouping them.
152         * @param semanticElements 
153         * @param semanticMap
154         * @param contextElement
155         * @return
156         */
157        protected List<DiagramElement> createDiagramElements(
158                        Collection<Artifact> semanticElements,
159                        Map<Artifact, DiagramElement> semanticMap, 
160                        Artifact contextElement,
161                        int depth) {
162                List<DiagramElement> ret = new ArrayList<>();
163                for (Artifact semanticElement: semanticElements) {
164                        ret.add(createDiagramElement(semanticElement, semanticMap, contextElement, depth));
165                }
166                return ret;
167        }

177        /**
178         * Collects relationships from the argument and unmapped children.
179         * @param semanticElement
180         * @param semanticMap
181         * @return
182         */
183        protected Collection<Relationship> collectAllRelationships(Artifact semanticElement, Map<Artifact, DiagramElement> semanticMap) {
184                Collection<Relationship> ret = new ArrayList<>();
185                ret.addAll(semanticElement.getOutboundRelationships().values());
186                for (Artifact child: semanticElement.getChildren().values()) {
187                        if (!semanticMap.containsKey(child)) {
188                                ret.addAll(collectAllRelationships(child, semanticMap));
189                        }
190                }
191                return ret;
192        }

194        protected void wire(Artifact semanticElement, Map<Artifact, DiagramElement> semanticMap) {
195                EList<Connection> connections = semanticMap.get(semanticElement).getConnections();
196                
197                Function<Relationship,Artifact> groupByMappedTarget = relationship -> {
198                        Artifact target = relationship.getTarget(); 
199                        while (!semanticMap.containsKey(target)) {
200                                EObject superContainer = target.eContainer().eContainer();
201                                if (superContainer instanceof Artifact) {
202                                        target = (Artifact) superContainer;
203                                } else {
204                                        return null;
205                                }
206                        }
207                        return target;          
208                };
209                
210                for (Entry<Artifact, List<Relationship>> outboundRelationshipEntry: Util.groupBy(collectAllRelationships(semanticElement, semanticMap), groupByMappedTarget).entrySet()) {                      
211                        Artifact target = outboundRelationshipEntry.getKey();
212                        if (target != null && target != semanticElement) {
213                                DiagramElement connectionTarget = semanticMap.get(target);
214                                if (connectionTarget != null) {
215                                        List<Relationship> relationships = outboundRelationshipEntry.getValue();
216                                        Connection connection;
217                                        if (relationships.size() == 1) {
218                                                // Single relationship
219                                                Relationship outboundRelationship = relationships.get(0);
220                                                Connection template = outboundRelationship.getStyle();
221                                                connection = template == null ? diagramFactory.createConnection() : EcoreUtil.copy(template);
222                                                if (Util.isBlank(connection.getType())) {
223                                                        connection.setType("->");
224                                                }
225                                                String name = outboundRelationship.getName();
226                                                if (!Util.isBlank(name)) {
227                                                        Link link = diagramFactory.createLink();
228                                                        link.setText(name);
229                                                        link.setLocation(getRelationshipLocation(outboundRelationship));
230                                                        link.setTooltip(getRelationshipTooltip(outboundRelationship));
231                                                        connection.getDescription().add(link);
232                                                }
233                                        } else {
234                                                // Grouped relationship
235                                                connection = createGroupConnection(semanticElement, target, relationships);
236                                        }
237                                        connections.add(connection);
238                                        connection.setTarget(connectionTarget);
239                                }
240                        }
241                }
242                
243                for (Artifact child: semanticElement.getChildren().values()) {
244                        if (semanticMap.containsKey(child)) {
245                                wire(child, semanticMap);
246                        }
247                }
248        }

261                return connection;
