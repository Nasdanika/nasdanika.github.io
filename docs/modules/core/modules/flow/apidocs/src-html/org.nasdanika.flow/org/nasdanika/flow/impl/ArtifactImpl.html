618                                        Artifact ext = ((Artifact) cExtends).getChildren().get(key);
619                                        if (ext != null) {
620                                                ret.add(ext);
621                                        }
622                                }
623                        }
624                }
625                
626                return ret;
627        }

629        @Override
630        public void apply(Artifact instance) {
631                super.apply(instance);
632                
633                // Artifacts
634                for (Entry<String, Artifact> ae: getChildren().entrySet()) {
635                        Artifact artifact = ae.getValue();
636                        EMap<String, Artifact> instanceArtifacts = instance.getChildren();
637                        String artifactKey = ae.getKey();
638                        if (artifact == null) {
639                                instanceArtifacts.removeKey(artifactKey);
640                        } else {
641                                Artifact instanceArtifact = artifact.create();
642                                instanceArtifacts.put(artifactKey, instanceArtifact);
643                                artifact.apply(instanceArtifact);
644                        }
645                }
646                
647                // Style
648                DiagramElement style = getStyle();
649                if (style != null) {
650                        instance.setStyle(EcoreUtil.copy(style));
651                }
652
653                // Relationships
654                for (Entry<String, Relationship> outboundRelationship: getOutboundRelationships()) {
655                        Relationship relationship = outboundRelationship.getValue();
656                        EMap<String, Relationship> instanceOutboundRelationships = instance.getOutboundRelationships();
657                        String relKey = outboundRelationship.getKey();
658                        if (relationship == null) {
659                                instanceOutboundRelationships.removeKey(relKey);
660                        } else {
661                                Relationship instanceOutboundRelationship = relationship.create();
662                                instanceOutboundRelationships.put(relKey, instanceOutboundRelationship);
663                                relationship.apply(instanceOutboundRelationship);
664                        }
665                }
666                
667        }
