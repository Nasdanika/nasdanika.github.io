206                // Documentation
207                EList<EObject> documentation = getDocumentation();
208                if (!documentation.isEmpty()) {
209                        EList<EObject> instanceDocumentation = instance.getDocumentation();

212                }

216                if (markers != null) {

220                }

225                        EMap<String, Diagram> instanceRepresentations = instance.getRepresentations();
226                        String representationKey = re.getKey();
227                        if (representation == null) {
228                                instanceRepresentations.removeKey(representationKey);
229                        } else {
230                                instanceRepresentations.put(representationKey, EcoreUtil.copy(representation));
231                        }
232                }
