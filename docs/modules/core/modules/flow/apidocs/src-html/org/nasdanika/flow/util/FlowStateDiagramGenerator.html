052        public void generateDiagram(Collection<FlowElement<?>> flowElements, Diagram diagram, FlowElement<?> contextElement) {
053                // Collecting all elements to be included
054                Map<FlowElement<?>,Integer> semanticElements = new HashMap<>();
055                for (FlowElement<?> flowElement: flowElements) {
056                        collectRelatedElements(flowElement, semanticElements, diagram.getContext());
057                }
058                
059                // Selecting top-level elements
060                Collection<FlowElement<?>> topLevelElements = new HashSet<>();
061                semanticElements.keySet().stream()
062                        .filter(se -> !EcoreUtil.isAncestor(topLevelElements, se))
063                        .forEach(se -> {
064                                topLevelElements.removeIf(topLevelElement -> EcoreUtil.isAncestor(se, topLevelElement));
065                                topLevelElements.add(se);
066                        });
067                
068                Map<FlowElement<?>,DiagramElement> semanticMap = new HashMap<>();
069                diagram.getElements().addAll(createDiagramElements(topLevelElements, semanticMap, contextElement, diagram.getDepth()));
070                
071                for (FlowElement<?> se: topLevelElements) {
072                        wire(se, semanticMap);
073                }
074        }

076        private void collectRelatedElements(FlowElement<?> semanticElement, Map<FlowElement<?>,Integer> accumulator, int depth) { 
077                if (semanticElement == null) {
078                        return;
079                }
080                Integer sDepth = accumulator.get(semanticElement);
081                if (sDepth != null && (sDepth < 0 || depth <= sDepth)) {
082                        return; // Do not traverse if existing depth is negative or depth is less than already traversed depth.
083                }
084                accumulator.put(semanticElement, depth);
085                
086                if (depth != 0) {
087                        for (Transition input: semanticElement.getInputs()) {
088                                FlowElement<?> source = (FlowElement<?>) input.eContainer().eContainer(); 
089                                collectRelatedElements(source, accumulator, source instanceof PseudoState || semanticElement instanceof PseudoState ? depth : depth - 1);
090                        }
091
092                        for (Transition output: semanticElement.getOutputs().values()) {
093                                FlowElement<?> target = output.getTarget(); 
094                                collectRelatedElements(target, accumulator, target instanceof PseudoState || semanticElement instanceof PseudoState ? depth : depth - 1);
095                        }
096
097                        for (Call invocation: semanticElement.getInvocations()) {
098                                FlowElement<?> source = (FlowElement<?>) invocation.eContainer().eContainer(); 
099                                collectRelatedElements(source, accumulator, source instanceof PseudoState || semanticElement instanceof PseudoState ? depth : depth - 1);
100                        }
101
102                        for (Call call: semanticElement.getCalls().values()) {
103                                FlowElement<?> target = call.getTarget(); 
104                                collectRelatedElements(target, accumulator, target instanceof PseudoState || semanticElement instanceof PseudoState ? depth : depth - 1);
105                        }
106                }
107                
108                if (semanticElement instanceof Flow) {
109                        for (FlowElement<?> subElement: ((Flow) semanticElement).getElements().values()) {
110                                collectRelatedElements(subElement, accumulator, depth);
111                        }
112                }
113        }

115        protected DiagramElement createDiagramElement(
116                        FlowElement<?> semanticElement, 
117                        Map<FlowElement<?>, DiagramElement> semanticMap, 
118                        FlowElement<?> contextElement,
119                        int depth) {
120                
121                if (semanticElement instanceof org.nasdanika.flow.Start) {
122                        org.nasdanika.diagram.Start ret = diagramFactory.createStart();
123                        semanticMap.put(semanticElement, ret);
124                        return ret;
125                }
126                
127                if (semanticElement instanceof org.nasdanika.flow.End) {
128                        org.nasdanika.diagram.End ret = diagramFactory.createEnd();
129                        semanticMap.put(semanticElement, ret);
130                        return ret;
131                }
132                
133                org.nasdanika.diagram.DiagramElement ret = diagramFactory.createDiagramElement();
134                semanticMap.put(semanticElement, ret);
135                ret.setType("state");
136                
137                if (semanticElement instanceof PseudoState) {
138                        if (semanticElement instanceof Choice) {
139                                ret.setStereotype("choice");
140                        } else if (semanticElement instanceof EntryPoint) { 
141                                ret.setStereotype("entryPoint");
142                        } else if (semanticElement instanceof ExitPoint) { 
143                                ret.setStereotype("exitPoint");
144                        } else if (semanticElement instanceof ExpansionInput) { 
145                                ret.setStereotype("expansionInput");
146                        } else if (semanticElement instanceof ExpansionOutput) { 
147                                ret.setStereotype("expansioinOutput");
148                        } else if (semanticElement instanceof Fork) { 
149                                ret.setStereotype("fork");
150                        } else if (semanticElement instanceof InputPin) { 
151                                ret.setStereotype("inputPin");
152                        } else if (semanticElement instanceof Join) { 
153                                ret.setStereotype("join");
154                        } else if (semanticElement instanceof OutputPin) { 
155                                ret.setStereotype("outputPin");                         
156                        }
157                        String name = semanticElement.getName();
158                        if (!Util.isBlank(name)) {
159                                Note nameNote = diagramFactory.createNote();
160                                nameNote.setText(name);
161                                ret.getNotes().add(nameNote);                           
162                        }
163                } else {
164                        ret.setText(semanticElement.getName());
165                        ret.setLocation(getFlowElementLocation(semanticElement));
166                        ret.setTooltip(getFlowElementTooltip(semanticElement));
167                        
168                        if (depth != 0 && semanticElement instanceof Flow && ((Flow) semanticElement).isPartition()) {
169                                ret.getElements().addAll(createDiagramElements(((Flow) semanticElement).getElements().values(), semanticMap, contextElement, depth - 1));
170                        }
171                }
172                EList<String> modifiers = semanticElement.getModifiers();
173                if (modifiers.contains("final")) {
174                        ret.setBold(true);
175                } else if (modifiers.contains("abstract")) {
176                        ret.setDashed(true);
177                }
178                
179                if (modifiers.contains("optional")) {
180                        ret.setBorder("grey");
181                }
182                
183                if (contextElement == null || semanticElement == contextElement || EcoreUtil.isAncestor(contextElement, semanticElement)) {
184                        if (Util.isBlank(ret.getColor())) {
185                                ret.setColor("FEFECE");                                 
186                        }
187                } else {
188                        ret.setColor("DDDDDD");
189                }
190                
191                ArtifactComponentDiagramGenerator.setProperties(semanticElement, ret);
192                return ret;
193        }

195        /**
196         * Creates diagram for flow elements potentially grouping them.
197         * @param semanticElements 
198         * @param semanticMap
199         * @param contextElement
200         * @return
201         */
202        protected List<DiagramElement> createDiagramElements(
203                        Collection<FlowElement<?>> semanticElements,
204                        Map<FlowElement<?>, DiagramElement> semanticMap, 
205                        FlowElement<?> contextElement,
206                        int depth) {            
207                List<DiagramElement> ret = new ArrayList<>();
208                if (isGroupByParticipant()) {
209                        java.util.function.Function<FlowElement<?>, Participant> keyExtractor = fe -> {
210                                EList<Participant> participants = fe.getParticipants();
211                                return participants.isEmpty() ? null : participants.get(0);
212                        };
213                        for (Entry<Participant, List<FlowElement<?>>> ge: Util.groupBy(semanticElements, keyExtractor).entrySet()) {
214                                if (ge.getKey() == null) {
215                                        for (FlowElement<?> subElement: ge.getValue()) {
216                                                ret.add(createDiagramElement(subElement, semanticMap, contextElement, depth));
217                                        }                                                       
218                                } else {
219                                        ret.add(createParticipantGroup(ge.getKey(), ge.getValue(), semanticMap, contextElement, depth));
220                                }
221                        }
222                } else {                                
223                        for (FlowElement<?> subElement: semanticElements) {
224                                ret.add(createDiagramElement(subElement, semanticMap, contextElement, depth));
225                        }
226                }
227                return ret;
228        }

238        /**
239         * Collects outputs from the argument and unmapped children.
240         * @param semanticElement
241         * @param semanticMap
242         * @return
243         */
244        protected Collection<Transition> collectAllOutputs(FlowElement<?> semanticElement, Map<FlowElement<?>, DiagramElement> semanticMap) {
245                Collection<Transition> ret = new ArrayList<>();
246                ret.addAll(semanticElement.getOutputs().values());
247                if (semanticElement instanceof Flow) {
248                        for (FlowElement<?> child: ((Flow) semanticElement).getElements().values()) {
249                                if (!semanticMap.containsKey(child)) {
250                                        ret.addAll(collectAllOutputs(child, semanticMap));
251                                }
252                        }
253                }
254                return ret;
255        }

257        /**
258         * Collects outputs from the argument and unmapped children.
259         * @param semanticElement
260         * @param semanticMap
261         * @return
262         */
263        protected Collection<Call> collectAllCalls(FlowElement<?> semanticElement, Map<FlowElement<?>, DiagramElement> semanticMap) {
264                Collection<Call> ret = new ArrayList<>();
265                ret.addAll(semanticElement.getCalls().values());
266                if (semanticElement instanceof Flow) {
267                        for (FlowElement<?> child: ((Flow) semanticElement).getElements().values()) {
268                                if (!semanticMap.containsKey(child)) {
269                                        ret.addAll(collectAllCalls(child, semanticMap));
270                                }
271                        }
272                }
273                return ret;
274        }

276        protected void wire(FlowElement<?> semanticElement, Map<FlowElement<?>, DiagramElement> semanticMap) {
277                EList<Connection> connections = semanticMap.get(semanticElement).getConnections();
278                
279                Function<Transition,FlowElement<?>> groupByMappedTarget = transition -> {
280                        FlowElement<?> target = transition.getTarget(); 
281                        while (!semanticMap.containsKey(target)) {
282                                if (target.eIsProxy()) {
283                                        throw new ConfigurationException("Target is an unresolved proxy: " + target, semanticElement);
284                                }
285                                EObject eContainer = target.eContainer();
286                                if (eContainer == null) {
287                                        throw new NullPointerException("Transition target does not have super-container, cannot group: " + transition);
288                                }
289                                EObject superContainer = eContainer.eContainer();
290                                if (superContainer instanceof FlowElement<?>) {
291                                        target = (FlowElement<?>) superContainer;
292                                } else {
293                                        return null;
294                                }
295                        }
296                        return target;          
297                };
298                
299                // Transitions
300                for (Entry<FlowElement<?>, List<Transition>> outputEntry: Util.groupBy(collectAllOutputs(semanticElement, semanticMap), groupByMappedTarget).entrySet()) {                      
301                        FlowElement<?> target = outputEntry.getKey();
302                        if (target != null && target != semanticElement) {
303                                DiagramElement connectionTarget = semanticMap.get(target);
304                                if (connectionTarget != null) {
305                                        List<Transition> transitions = outputEntry.getValue();
306                                        Connection connection;
307                                        if (transitions.size() == 1) {
308                                                // Single relationship
309                                                Transition output = transitions.get(0);
310                                                connection = diagramFactory.createConnection();
311                                                String name = output.getName();
312                                                if (!Util.isBlank(name)) {
313                                                        Link link = diagramFactory.createLink();
314                                                        link.setText(name);
315                                                        link.setLocation(getTransitionLocation(output));
316                                                        link.setTooltip(getTransitionTooltip(output));
317                                                        connection.getDescription().add(link);
318                                                }
319                                        } else {
320                                                // Grouped relationship
321                                                connection = createGroupConnection(semanticElement, target, transitions);
322                                        }
323                                        connection.setType("-->");
324                                        connections.add(connection);
325                                        connection.setTarget(connectionTarget);
326                                }
327                        }
328                }
329                
330                // Calls
331                for (Entry<FlowElement<?>, List<Call>> callEntry: Util.groupBy(collectAllCalls(semanticElement, semanticMap), groupByMappedTarget).entrySet()) {                        
332                        FlowElement<?> target = callEntry.getKey();
333                        if (target != null && target != semanticElement) {
334                                DiagramElement connectionTarget = semanticMap.get(target);
335                                if (connectionTarget != null) {
336                                        List<Call> calls = callEntry.getValue();
337                                        Connection connection;
338                                        if (calls.size() == 1) {
339                                                // Single relationship
340                                                Call call= calls.get(0);                                                
341                                                connection = diagramFactory.createConnection();
342                                                String name = call.getName();
343                                                if (!Util.isBlank(name)) {
344                                                        Link link = diagramFactory.createLink();
345                                                        link.setText(name);
346                                                        link.setLocation(getCallLocation(call));
347                                                        link.setTooltip(getCallTooltip(call));
348                                                        connection.getDescription().add(link);
349                                                }
350                                        } else {
351                                                // Grouped relationship
352                                                connection = createGroupConnection(semanticElement, target, calls);
353                                        }
354                                        connections.add(connection);
355                                        connection.setTarget(connectionTarget);
356                                        connection.setType("-left${style}->");
357                                        connection.setColor("black");
358                                }
359                        }
360                }
361
362                // Flow elements
363                if (semanticElement instanceof Flow) {
364                        for (FlowElement<?> child: ((Flow) semanticElement).getElements().values()) {
365                                if (semanticMap.containsKey(child)) {
366                                        wire(child, semanticMap);
367                                }
368                        }
369                }
370        }

406        protected DiagramElement createParticipantGroup(
407                        Participant participant, 
408                        Collection<FlowElement<?>> groupElements, 
409                        Map<FlowElement<?>, DiagramElement> semanticMap, 
410                        FlowElement<?> contextElement,
411                        int depth) {
412                
413                org.nasdanika.diagram.DiagramElement ret = diagramFactory.createDiagramElement();
414                ret.setColor("E0E0FF");
415                
416                ret.setText(participant.getName());
417                ret.setType("state");
418                ret.setLocation(getParticipantLocation(participant));
419                ret.setTooltip(getParticipantTooltip(participant));
420                
421                
422                for (FlowElement<?> groupElement: groupElements) {
423                        DiagramElement diagramElement = createDiagramElement(groupElement, semanticMap, contextElement, depth);
424                        if (diagramElement != null) {
425                                ret.getElements().add(diagramElement);
426                        }
427                }
428                
429                return ret;             
430        }
