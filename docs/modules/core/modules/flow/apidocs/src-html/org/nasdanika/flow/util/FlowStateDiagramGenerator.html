048        public void generateDiagram(
049                        Flow flow, 
050                        Diagram diagram,
051                        BiPredicate<FlowElement<?>, Transition> semanticElementPredicate) {
052                Collection<FlowElement<?>> elements = new ArrayList<>(flow.getElements().values());
053                boolean isContext = diagram.getContext() > 0;
054                if (isContext) {
055                        elements.add(flow);
056                }
057                
058                if (semanticElementPredicate == null && isContext) {
059                        semanticElementPredicate = (fe, t) -> {
060                                // Only own calls and invocations.
061                                if (t instanceof Call) {
062                                        boolean isSource = t.eContainer() != null && t.eContainer().eContainer() == flow;
063                                        return isSource || t.getTarget() == flow;
064                                }
065                                
066                                return true;
067                        }; 
068                }               
069                
070                generateDiagram(
071                                elements, 
072                                diagram, 
073                                isContext ? flow : null, 
074                                isContext ? DEFAULT_PARTITION_PREDICATE.or(fe -> fe == flow).and(fe -> EcoreUtil.isAncestor(flow, fe)) : DEFAULT_PARTITION_PREDICATE,
075                                semanticElementPredicate);
076        }

078        public void generateDiagram(
079                        FlowElement<?> flowElement, 
080                        Diagram diagram,
081                        BiPredicate<FlowElement<?>, Transition> semanticElementPredicate) {
082                
083                if (semanticElementPredicate == null && diagram.getContext() > 0) {
084                        semanticElementPredicate = (fe, t) -> {
085                                // Only own calls and invocations.
086                                if (t instanceof Call) {
087                                        boolean isSource = t.eContainer() != null && t.eContainer().eContainer() == flowElement;
088                                        return isSource || t.getTarget() == flowElement;
089                                }
090                                
091                                return true;
092                        }; 
093                }               
094        
095                generateDiagram(
096                                Collections.singleton(flowElement), 
097                                diagram, 
098                                flowElement, 
099                                diagram.getContext() > 0 ? DEFAULT_PARTITION_PREDICATE.and(fe -> EcoreUtil.isAncestor(flowElement, fe)) : DEFAULT_PARTITION_PREDICATE,
100                                semanticElementPredicate);
101        }

103        public void generateDiagram(
104                        Collection<FlowElement<?>> flowElements, 
105                        Diagram diagram, 
106                        FlowElement<?> contextElement,
107                        Predicate<FlowElement<?>> partitionPredicate,
108                        BiPredicate<FlowElement<?>, Transition> semanticElementPredicate) {
109                
110                // Collecting all elements to be included
111                Map<FlowElement<?>,Integer> semanticElements = new HashMap<>();
112                for (FlowElement<?> flowElement: flowElements) {
113                        collectRelatedElements(flowElement, semanticElements, diagram.getContext(), Direction.BOTH, semanticElementPredicate);
114                }
115                
116                // Selecting top-level elements
117                Collection<FlowElement<?>> topLevelElements = new HashSet<>();
118                semanticElements.keySet().stream()
119                        .filter(se -> !EcoreUtil.isAncestor(topLevelElements, se))
120                        .forEach(se -> {
121                                topLevelElements.removeIf(topLevelElement -> EcoreUtil.isAncestor(se, topLevelElement));
122                                topLevelElements.add(se);
123                        });
124                
125                Map<FlowElement<?>,DiagramElement> semanticMap = new HashMap<>();
126                diagram.getElements().addAll(createDiagramElements(topLevelElements, semanticMap, contextElement, partitionPredicate, diagram.getDepth()));
127                
128                for (FlowElement<?> se: topLevelElements) {
129                        wire(se, semanticMap);
130                }
131        }
132        
133        private enum Direction { IN, OUT, BOTH }
134
135        /**
136         * Collects related elements
137         * @param semanticElement Semantic element
138         * @param accumulator Accumulator of related elements
139         * @param depth Collection depth. 
140         * @param in If true, collect in the direction of related inputs/invocations, otherwise in the direction of outputs/calls.
141         */
142        private void collectRelatedElements(
143                        FlowElement<?> semanticElement, 
144                        Map<FlowElement<?>,Integer> accumulator, 
145                        int depth, 
146                        Direction direction,
147                        BiPredicate<FlowElement<?>, Transition> semanticElementPredicate
148                        ) {
149                
150                if (semanticElement == null) {
151                        return;
152                }
153                Integer sDepth = accumulator.get(semanticElement);
154                if (sDepth != null && (sDepth < 0 || depth <= sDepth)) {
155                        return; // Do not traverse if existing depth is negative or depth is less than already traversed depth.
156                }
157                accumulator.put(semanticElement, depth);
158                
159                if (depth != 0) {
160                        if (direction == Direction.BOTH || direction == Direction.IN) {
161                                for (Transition input: semanticElement.getInputs()) {
162                                        FlowElement<?> source = (FlowElement<?>) input.eContainer().eContainer(); 
163                                        boolean isPseudoSource = source instanceof PseudoState;
164                                        if (semanticElementPredicate == null || semanticElementPredicate.test(semanticElement, input)) {
165                                                collectRelatedElements(source, accumulator, isPseudoSource ? depth : depth - 1, isPseudoSource ? Direction.IN : Direction.BOTH, semanticElementPredicate);
166                                        }
167                                }
168        
169                                for (Call invocation: semanticElement.getInvocations()) {
170                                        FlowElement<?> source = (FlowElement<?>) invocation.eContainer().eContainer(); 
171                                        boolean isPseudoSource = source instanceof PseudoState;
172                                        if (semanticElementPredicate == null || semanticElementPredicate.test(semanticElement, invocation)) {
173                                                collectRelatedElements(source, accumulator, isPseudoSource ? depth : depth - 1, isPseudoSource ? Direction.IN : Direction.BOTH, semanticElementPredicate);
174                                        }
175                                }
176                        }
177                        
178                        if (direction == Direction.BOTH || direction == Direction.OUT) {
179                                for (Transition output: semanticElement.getOutputs().values()) {
180                                        FlowElement<?> target = output.getTarget(); 
181                                        boolean isPseudoTarget = target instanceof PseudoState;
182                                        if (semanticElementPredicate == null || semanticElementPredicate.test(semanticElement, output)) {
183                                                collectRelatedElements(target, accumulator, isPseudoTarget ? depth : depth - 1, isPseudoTarget ? Direction.OUT : Direction.BOTH, semanticElementPredicate);
184                                        }
185                                }
186        
187                                for (Call call: semanticElement.getCalls().values()) {
188                                        FlowElement<?> target = call.getTarget(); 
189                                        boolean isPseudoTarget = target instanceof PseudoState;
190                                        if (semanticElementPredicate == null || semanticElementPredicate.test(semanticElement, call)) {
191                                                collectRelatedElements(target, accumulator, isPseudoTarget ? depth : depth - 1, isPseudoTarget ? Direction.OUT : Direction.BOTH, semanticElementPredicate);
192                                        }
193                                }
194                        }
195                }
196                
197                if (semanticElement instanceof Flow) {
198                        for (FlowElement<?> subElement: ((Flow) semanticElement).getElements().values()) {
199                                if (semanticElementPredicate == null || semanticElementPredicate.test(semanticElement, null)) {
200                                        collectRelatedElements(subElement, accumulator, depth, Direction.BOTH, semanticElementPredicate);
201                                }
202                        }
203                }
204        }

206        protected DiagramElement createDiagramElement(
207                        FlowElement<?> semanticElement, 
208                        Map<FlowElement<?>, DiagramElement> semanticMap, 
209                        FlowElement<?> contextElement,
210                        Predicate<FlowElement<?>> partitionPredicate,
211                        int depth) {
212                
213                if (semanticElement instanceof org.nasdanika.flow.Start) {
214                        org.nasdanika.diagram.Start ret = diagramFactory.createStart();
215                        semanticMap.put(semanticElement, ret);
216                        return ret;
217                }
218                
219                if (semanticElement instanceof org.nasdanika.flow.End) {
220                        org.nasdanika.diagram.End ret = diagramFactory.createEnd();
221                        semanticMap.put(semanticElement, ret);
222                        return ret;
223                }
224                
225                org.nasdanika.diagram.DiagramElement ret = diagramFactory.createDiagramElement();
226                semanticMap.put(semanticElement, ret);
227                ret.setType("state");
228                
229                if (semanticElement instanceof PseudoState) {
230                        if (semanticElement instanceof Choice) {
231                                ret.setStereotype("choice");
232                        } else if (semanticElement instanceof EntryPoint) { 
233                                ret.setStereotype("entryPoint");
234                        } else if (semanticElement instanceof ExitPoint) { 
235                                ret.setStereotype("exitPoint");
236                        } else if (semanticElement instanceof ExpansionInput) { 
237                                ret.setStereotype("expansionInput");
238                        } else if (semanticElement instanceof ExpansionOutput) { 
239                                ret.setStereotype("expansioinOutput");
240                        } else if (semanticElement instanceof Fork) { 
241                                ret.setStereotype("fork");
242                        } else if (semanticElement instanceof InputPin) { 
243                                ret.setStereotype("inputPin");
244                        } else if (semanticElement instanceof Join) { 
245                                ret.setStereotype("join");
246                        } else if (semanticElement instanceof OutputPin) { 
247                                ret.setStereotype("outputPin");                         
248                        }
249                        String name = semanticElement.getName();
250                        if (!Util.isBlank(name)) {
251                                Note nameNote = diagramFactory.createNote();
252                                nameNote.setText(name);
253                                ret.getNotes().add(nameNote);                           
254                        }
255                } else {
256                        ret.setText(semanticElement.getName());
257                        ret.setLocation(getFlowElementLocation(semanticElement));
258                        ret.setTooltip(getFlowElementTooltip(semanticElement));
259                        
260                        if (partitionPredicate == null) {
261                                partitionPredicate = DEFAULT_PARTITION_PREDICATE;
262                        }
263                        
264                        if (depth != 0 && partitionPredicate.test(semanticElement)) {
265                                ret.getElements().addAll(createDiagramElements(((Flow) semanticElement).getElements().values(), semanticMap, contextElement, partitionPredicate, depth - 1));
266                        }
267                }
268                EList<String> modifiers = semanticElement.getModifiers();
269                if (modifiers.contains("final")) {
270                        ret.setBold(true);
271                } else if (modifiers.contains("abstract")) {
272                        ret.setDashed(true);
273                }
274                
275                if (modifiers.contains("optional")) {
276                        ret.setBorder("grey");
277                }
278                
279                if (contextElement == null || semanticElement == contextElement || EcoreUtil.isAncestor(contextElement, semanticElement)) {
280                        if (Util.isBlank(ret.getColor())) {
281                                ret.setColor("FEFECE");                                 
282                        }
283                } else {
284                        ret.setColor("DDDDDD");
285                }
286                
287                ArtifactComponentDiagramGenerator.setProperties(semanticElement, ret);
288                return ret;
289        }

291        /**
292         * Creates diagram for flow elements potentially grouping them.
293         * @param semanticElements 
294         * @param semanticMap
295         * @param contextElement
296         * @return
297         */
298        protected List<DiagramElement> createDiagramElements(
299                        Collection<FlowElement<?>> semanticElements,
300                        Map<FlowElement<?>, DiagramElement> semanticMap, 
301                        FlowElement<?> contextElement,
302                        Predicate<FlowElement<?>> partitionPredicate,
303                        int depth) {            
304                List<DiagramElement> ret = new ArrayList<>();
305                if (isGroupByParticipant()) {
306                        java.util.function.Function<FlowElement<?>, Participant> keyExtractor = fe -> {
307                                EList<Participant> participants = fe.getParticipants();
308                                return participants.isEmpty() ? null : participants.get(0);
309                        };
310                        for (Entry<Participant, List<FlowElement<?>>> ge: Util.groupBy(semanticElements, keyExtractor).entrySet()) {
311                                if (ge.getKey() == null) {
312                                        for (FlowElement<?> subElement: ge.getValue()) {
313                                                ret.add(createDiagramElement(subElement, semanticMap, contextElement, partitionPredicate, depth));
314                                        }                                                       
315                                } else {
316                                        ret.add(createParticipantGroup(ge.getKey(), ge.getValue(), semanticMap, contextElement, partitionPredicate, depth));
317                                }
318                        }
319                } else {                                
320                        for (FlowElement<?> subElement: semanticElements) {
321                                ret.add(createDiagramElement(subElement, semanticMap, contextElement, partitionPredicate, depth));
322                        }
323                }
324                return ret;
325        }

335        /**
336         * Collects outputs from the argument and unmapped children.
337         * @param semanticElement
338         * @param semanticMap
339         * @return
340         */
341        protected Collection<Transition> collectAllOutputs(FlowElement<?> semanticElement, Map<FlowElement<?>, DiagramElement> semanticMap) {
342                Collection<Transition> ret = new ArrayList<>();
343                ret.addAll(semanticElement.getOutputs().values());
344                if (semanticElement instanceof Flow) {
345                        for (FlowElement<?> child: ((Flow) semanticElement).getElements().values()) {
346                                if (!semanticMap.containsKey(child)) {
347                                        ret.addAll(collectAllOutputs(child, semanticMap));
348                                }
349                        }
350                }
351                return ret;
352        }

355         * Collects outputs from the argument and unmapped children.
356         * @param semanticElement
357         * @param semanticMap
358         * @return
359         */
360        protected Collection<Call> collectAllCalls(FlowElement<?> semanticElement, Map<FlowElement<?>, DiagramElement> semanticMap) {
361                Collection<Call> ret = new ArrayList<>();
362                ret.addAll(semanticElement.getCalls().values());
363                if (semanticElement instanceof Flow) {
364                        for (FlowElement<?> child: ((Flow) semanticElement).getElements().values()) {
365                                if (!semanticMap.containsKey(child)) {
366                                        ret.addAll(collectAllCalls(child, semanticMap));
367                                }
368                        }
369                }
370                return ret;
371        }

373        protected void wire(FlowElement<?> semanticElement, Map<FlowElement<?>, DiagramElement> semanticMap) {
374                EList<Connection> connections = semanticMap.get(semanticElement).getConnections();
375                
376                Function<Transition,FlowElement<?>> groupByMappedTarget = transition -> {
377                        FlowElement<?> target = transition.getTarget(); 
378                        while (!semanticMap.containsKey(target)) {
379                                if (target.eIsProxy()) {
380                                        throw new ConfigurationException("Target is an unresolved proxy: " + target, semanticElement);
381                                }
382                                EObject eContainer = target.eContainer();
383                                if (eContainer == null) {
384                                        throw new NullPointerException("Transition target does not have super-container, cannot group: " + transition);
385                                }
386                                EObject superContainer = eContainer.eContainer();
387                                if (superContainer instanceof FlowElement<?>) {
388                                        target = (FlowElement<?>) superContainer;
389                                } else {
390                                        return null;
391                                }
392                        }
393                        return target;          
394                };
395                
396                // Transitions
397                for (Entry<FlowElement<?>, List<Transition>> outputEntry: Util.groupBy(collectAllOutputs(semanticElement, semanticMap), groupByMappedTarget).entrySet()) {                      
398                        FlowElement<?> target = outputEntry.getKey();
399                        if (target != null && target != semanticElement) {
400                                DiagramElement connectionTarget = semanticMap.get(target);
401                                if (connectionTarget != null) {
402                                        List<Transition> transitions = outputEntry.getValue();
403                                        Connection connection;
404                                        if (transitions.size() == 1) {
405                                                // Single relationship
406                                                Transition output = transitions.get(0);
407                                                connection = diagramFactory.createConnection();
408                                                String name = output.getName();
409                                                if (!Util.isBlank(name)) {
410                                                        Link link = diagramFactory.createLink();
411                                                        link.setText(name);
412                                                        link.setLocation(getTransitionLocation(output));
413                                                        link.setTooltip(getTransitionTooltip(output));
414                                                        connection.getDescription().add(link);
415                                                }
416                                        } else {
417                                                // Grouped relationship
418                                                connection = createGroupConnection(semanticElement, target, transitions);
419                                        }
420                                        connection.setType("-->");
421                                        connections.add(connection);
422                                        connection.setTarget(connectionTarget);
423                                }
424                        }
425                }
426                
427                // Calls
428                for (Entry<FlowElement<?>, List<Call>> callEntry: Util.groupBy(collectAllCalls(semanticElement, semanticMap), groupByMappedTarget).entrySet()) {                        
429                        FlowElement<?> target = callEntry.getKey();
430                        if (target != null && target != semanticElement) {
431                                DiagramElement connectionTarget = semanticMap.get(target);
432                                if (connectionTarget != null) {
433                                        List<Call> calls = callEntry.getValue();
434                                        Connection connection;
435                                        if (calls.size() == 1) {
436                                                // Single relationship
437                                                Call call= calls.get(0);                                                
438                                                connection = diagramFactory.createConnection();
439                                                String name = call.getName();
440                                                if (!Util.isBlank(name)) {
441                                                        Link link = diagramFactory.createLink();
442                                                        link.setText(name);
443                                                        link.setLocation(getCallLocation(call));
444                                                        link.setTooltip(getCallTooltip(call));
445                                                        connection.getDescription().add(link);
446                                                }
447                                        } else {
448                                                // Grouped relationship
449                                                connection = createGroupConnection(semanticElement, target, calls);
450                                        }
451                                        connections.add(connection);
452                                        connection.setTarget(connectionTarget);
453                                        connection.setType("-right${style}->");
454                                        connection.setColor("black");
455                                }
456                        }
457                }
458
459                // Flow elements
460                if (semanticElement instanceof Flow) {
461                        for (FlowElement<?> child: ((Flow) semanticElement).getElements().values()) {
462                                if (semanticMap.containsKey(child)) {
463                                        wire(child, semanticMap);
464                                }
465                        }
466                }
467        }

503        protected DiagramElement createParticipantGroup(
504                        Participant participant, 
505                        Collection<FlowElement<?>> groupElements, 
506                        Map<FlowElement<?>, DiagramElement> semanticMap, 
507                        FlowElement<?> contextElement,
508                        Predicate<FlowElement<?>> partitionPredicate,
509                        int depth) {
510                
511                org.nasdanika.diagram.DiagramElement ret = diagramFactory.createDiagramElement();
512                ret.setColor("E0E0FF");
513                
514                ret.setText(participant.getName());
515                ret.setType("state");
516                ret.setLocation(getParticipantLocation(participant));
517                ret.setTooltip(getParticipantTooltip(participant));
518                
519                
520                for (FlowElement<?> groupElement: groupElements) {
521                        DiagramElement diagramElement = createDiagramElement(groupElement, semanticMap, contextElement, partitionPredicate, depth);
522                        if (diagramElement != null) {
523                                ret.getElements().add(diagramElement);
524                        }
525                }
526                
527                return ret;             
528        }
