039        public String generateSpec(Diagram diagram) {
040                StringBuilder ret = new StringBuilder();
041                
042                // Modifiers
043                if (!diagram.isVertical()) {
044                        ret.append("left to right direction").append(System.lineSeparator());
045                }
046                if (diagram.isHideEmptyDescription()) {
047                        ret.append("hide empty description").append(System.lineSeparator());
048                }
049                if (diagram.isHideFootbox()) {
050                        ret.append("hide footbox").append(System.lineSeparator());
051                }
052                
053                int noteIdx = 0;
054                for (Note note: diagram.getNotes()) {
055                        ret.append("note as note_").append(++noteIdx);
056                        ret.append(renderNote(note));
057                        ret.append("end note");
058                }
059                for (Layer layer: diagram.getLayers()) {
060                        for (Note note: layer.getNotes()) {
061                                ret.append("note as note_").append(++noteIdx);
062                                ret.append(renderNote(note));
063                                ret.append("end note");
064                        }
065                }
066                
067                List<Connection> connections = new ArrayList<>();
068                for (DiagramElement de: diagram.getElements()) {
069                        ret.append(generate(de, connections, 0));
070                }
071                for (Layer layer: diagram.getLayers()) {
072                        for (DiagramElement de: layer.getElements()) {
073                                ret.append(generate(de, connections, 0));
074                        }
075                }
076                
077                for (Connection c: connections) {
078                        ret.append(generate(c));
079                }
080
081                return ret.toString();
082        }
083        
084        protected String renderLink(Link link) {
085                return renderLink(link.getText(), link.getLocation(), link.getTooltip());
086        }
087
088        protected String renderLink(String text, String location, String tooltip) {
089                if (Util.isBlank(location)) {
090                        return text;
091                }
092                StringBuilder ret = new StringBuilder("[[");
093                ret.append(location);
094                if (!Util.isBlank(tooltip)) {
095                        ret.append("{").append(tooltip).append("}");
096                }
097                if (!Util.isBlank(text)) {
098                        ret.append(" ").append(text);
099                }
100                ret.append("]]");
101                return ret.toString();
102        }
103        
104        protected String render(List<EObject> elements) {
105                StringBuilder ret = new StringBuilder();
106                for (EObject e: elements) {
107                        if (e instanceof Link) {
108                                ret.append(renderLink((Link) e));
109                        } else if (e instanceof Text) {
110                                ret.append(((Text) e).getContent());
111                        } else {
112                                throw new IllegalArgumentException("Unsupported element type: " + e);
113                        }
114                }
115                return ret.toString();
116        }

118        /**
119         * @param diagramElement
120         * @param connectionsCollector
121         * @param depth
122         * @return Diagram element spec with a line separator.
123         */
124        protected String generate(DiagramElement diagramElement, List<Connection> connectionsCollector, int depth) {
125                connectionsCollector.addAll(diagramElement.getConnections());
126                if (diagramElement instanceof Start) {
127                        return "";
128                }
129                if (diagramElement instanceof End) {
130                        return "";
131                }
132                
133                StringBuilder ret = new StringBuilder();
134                
135                for (int i = 0; i < depth; ++i) {
136                        ret.append("  ");
137                }
138                
139                ret.append(diagramElement.getType()).append(" ");               
140                
141                StringBuilder nameBuilder = new StringBuilder();
142                String text = diagramElement.getText();
143                EList<EObject> name = diagramElement.getName();
144                if (!Util.isBlank(text)) {
145                        nameBuilder.append(text);
146                        if (!name.isEmpty()) {
147                                nameBuilder.append(" ");
148                        }
149                }
150                
151                nameBuilder.append(render(name));
152                
153                String renderedName = nameBuilder.toString();
154                if (!Util.isBlank(renderedName)) {
155                        ret
156                                .append("\"")
157                                .append(renderedName)
158                                .append("\"")
159                                .append(" as ");
160                }
161                
162                ret.append(diagramElement.getId());
163                
164                String stereotype = diagramElement.getStereotype();
165                if (!Util.isBlank(stereotype)) {
166                        ret.append(" <<").append(stereotype).append(">>");
167                }
168                
169                String location = diagramElement.getLocation();
170                if (!Util.isBlank(location)) {
171                        ret.append(" ").append(renderLink(null, location, diagramElement.getTooltip()));
172                }
173                
174                StringBuilder styleBuilder = new StringBuilder();
175                String color = diagramElement.getColor();
176                if (!Util.isBlank(color)) {
177                        styleBuilder.append(" #").append(color);
178                        String gradient = diagramElement.getGradient();
179                        if (!Util.isBlank(gradient)) {
180                                styleBuilder.append("|").append(gradient);
181                        }
182                }
183                
184                String borderColor = diagramElement.getBorder();
185                if (!Util.isBlank(borderColor)) {
186                        styleBuilder.append(styleBuilder.length() == 0 ? " #" : ";");
187                        styleBuilder.append("line:").append(borderColor);
188                }
189                                
190                if (diagramElement.isDashed()) {
191                        styleBuilder.append(styleBuilder.length() == 0 ? " #" : ";");
192                        styleBuilder.append("line.dashed");
193                } else if (diagramElement.isDotted()) {
194                        styleBuilder.append(styleBuilder.length() == 0 ? " #" : ";");
195                        styleBuilder.append("line.dotted");
196                } else if (diagramElement.isBold()) {
197                        styleBuilder.append(styleBuilder.length() == 0 ? " #" : ";");
198                        styleBuilder.append("line.bold");
199                }
200                
201                ret.append(styleBuilder);
202                
203                // TODO - styling like background
204                
205                EList<EObject> description = diagramElement.getDescription();
206                if (!description.isEmpty()) {
207                        ret.append(" : ");
208                        ret.append(render(description));
209                }
210                
211                EList<DiagramElement> elements = diagramElement.getElements();
212                if (!elements.isEmpty()) {
213                        ret.append(" {").append(System.lineSeparator());
214
215                        
216                        List<Connection> connections = new ArrayList<>();
217                        for (DiagramElement element: elements) {
218                                ret.append(generate(element, connections, depth + 1)); 
219                        }
220                        
221                        for (Connection c: connections) {
222                                if (EcoreUtil.isAncestor(diagramElement, c.eContainer()) && EcoreUtil.isAncestor(diagramElement, c.getTarget())) {
223                                        ret.append(generate(c));
224                                } else {                                
225                                        connectionsCollector.add(c);
226                                }
227                        }                       
228                        
229                        for (int i = 0; i < depth; ++i) {
230                                ret.append("  ");
231                        }
232                        ret.append("}");
233                }
234                ret.append(System.lineSeparator());
235                
236                for (Note note: diagramElement.getNotes()) {
237                        ret
238                                .append("note ")
239                                .append(note.getPlacement().name().toLowerCase())
240                                .append(" of ")
241                                .append(diagramElement.getId())
242                                .append(System.lineSeparator());
243                        ret.append(renderNote(note));
244                        ret.append("end note").append(System.lineSeparator());
245                }
246                
247                return ret.toString();
248        }
249
250        protected String renderNote(Note note) {
251                StringBuilder ret = new StringBuilder();
252                String noteText = note.getText();
253                EList<EObject> noteContent = note.getContent();
254                if (!Util.isBlank(noteText)) {
255                        ret.append(noteText);
256                        if (!noteContent.isEmpty()) {
257                                ret.append(" ");
258                        }
259                }
260                ret.append(render(noteContent));
261                ret.append(System.lineSeparator());
262                return ret.toString();
263        }
264        
265        protected String generate(Connection connection) {
266                StringBuilder ret = new StringBuilder();
267                
268                DiagramElement source = (DiagramElement) connection.eContainer();
269                
270                String sourceId;
271                if (source instanceof Start) {
272                        sourceId = "[*]";
273                } else {
274                        sourceId = source.getId();
275                }
276                DiagramElement target = connection.getTarget();
277                String targetId;
278                if (target instanceof End) {
279                        targetId = "[*]";
280                } else {
281                        targetId = target.getId();
282                }
283                
284                StringBuilder styleBuilder = new StringBuilder();
285                String color = connection.getColor();
286                if (!Util.isBlank(color)) {
287                        styleBuilder.append("#").append(color);
288                }               
289                if (connection.isDashed()) {
290                        if (styleBuilder.length() > 0) {
291                                styleBuilder.append(",");
292                        }
293                        styleBuilder.append("dashed");
294                } else if (connection.isDotted()) {
295                        if (styleBuilder.length() > 0) {
296                                styleBuilder.append(",");
297                        }
298                        styleBuilder.append("dotted");
299                } else if (connection.isBold()) {
300                        if (styleBuilder.length() > 0) {
301                                styleBuilder.append(",");
302                        }
303                        styleBuilder.append("bold");
304                }
305                int thickness = connection.getThickness();
306                if (thickness > 0) {
307                        if (styleBuilder.length() > 0) {
308                                styleBuilder.append(",");
309                        }
310                        styleBuilder.append("thickness=").append(thickness);                    
311                }
312                String style = styleBuilder.length() > 0 ? "[" + styleBuilder + "]" : "";
313                
314                ret
315                        .append(sourceId)
316                        .append(" ")
317                        .append(Context.singleton("style", style).interpolateToString(connection.getType()))
318                        .append(" ")
319                        .append(targetId);                      
320                
321                EList<EObject> description = connection.getDescription();
322                if (!description.isEmpty()) {
323                        ret.append(" : ");
324                        ret.append(render(description));
325                }
326                
327                ret.append(System.lineSeparator());
328                
329                for (Note note: connection.getNotes()) {
330                        ret.append("note on link").append(System.lineSeparator());
331                        ret.append(renderNote(note));
332                        ret.append("end note").append(System.lineSeparator());
333                }
334                
335                return ret.toString();
336        }
