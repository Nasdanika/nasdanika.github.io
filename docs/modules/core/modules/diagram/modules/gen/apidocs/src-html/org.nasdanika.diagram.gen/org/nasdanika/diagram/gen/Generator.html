001package org.nasdanika.diagram.gen;
002
003import java.io.File;
004import java.io.IOException;
005import java.nio.file.Files;
006
007import org.eclipse.emf.common.util.URI;
008import org.eclipse.emf.ecore.resource.Resource;
009import org.nasdanika.common.DefaultConverter;
010import org.nasdanika.common.DiagramGenerator;
011import org.nasdanika.common.Util;
012import org.nasdanika.diagram.Diagram;
013import org.nasdanika.drawio.Document;
014import org.nasdanika.ncore.Marker;
015
016/**
017 * Generates diagram HTML delegating to different generators based on diagram type.
018 * @author Pavel
019 *
020 */
021public class Generator {
022        
023        public static Generator INSTANCE = new Generator();
024
025        public static final String PLANTUML_SCHEMA = "plantuml:";
026        public static final String DRAWIO_SCHEMA = "drawio:";
027//      public static final String MERMAID_SCHEMA = "mermaid:";
028
029        public String generate(Diagram diagram) throws Exception {
030                String type = diagram.getType();
031                if (type.startsWith(PLANTUML_SCHEMA)) {
032                        String dialect = type.substring(PLANTUML_SCHEMA.length()).toLowerCase();
033                        return createPlantumlGenerator().generateDiagram(diagram, dialect);
034                }
035                
036//              if (type.startsWith(MERMAID_SCHEMA)) {
037//                      String dialect = type.substring(MERMAID_SCHEMA.length()).toLowerCase();
038//                      return createMermaidGenerator().generateDiagram(diagram, dialect);
039//              }
040                
041                if (type.startsWith(DRAWIO_SCHEMA)) {
042                        String diagramURI = type.substring(DRAWIO_SCHEMA.length());
043                        URI uri = URI.createURI(diagramURI);
044                        
045                        URI markerBase = null;
046                        for (Marker marker: diagram.getMarkers()) { 
047                                if (marker != null && !Util.isBlank(marker.getLocation())) { // Finds a first marker with location
048                                        markerBase = URI.createURI(marker.getLocation());
049                                        break;
050                                }                               
051                        }
052
053                        URI resourceBase = null;
054                        Resource resource = diagram.eResource();
055                        if (resource != null) {
056                                resourceBase = resource.getURI();
057                        }
058                        
059                        if (markerBase != null && markerBase.hasAbsolutePath() && diagramURI.startsWith("./")) {
060                                uri = uri.resolve(markerBase);
061                        } else if (resourceBase != null) {
062                                uri = uri.resolve(resourceBase);
063                        }
064                        
065                        if (uri.isFile()) {
066                                String fileStr = uri.toFileString();
067                                File diagramFile = new File(fileStr);
068                                if (diagramFile.exists()) {
069                                        // TODO - merging - read, merge models, write if result is not null.
070                                } else {
071                                        File container = diagramFile.getParentFile();
072                                        if (container.isDirectory() || container.mkdirs()) {
073                                                Document document = createDrawioGenerator().generateModel(diagram);
074                                                Files.write(diagramFile.toPath(), DefaultConverter.INSTANCE.toByteArray(document.save(true)));
075                                        }
076                                }
077                        }                                               
078                        
079                        return generateDrawioHtml(uri);
080                }
081                
082                if (type.equals("drawio")) {
083                        return createDrawioGenerator().generateDiagram(diagram);
084                }
085                
086                throw new IllegalArgumentException("Unsupported diagram type: " + type);
087        }
088        
089        /**
090         * Generates draw.io HTML from diagram resource.
091         * @param diagram
092         * @return
093         * @throws IOException
094         */
095        public String generateDrawioHtml(URI diagram) throws Exception {
096                return getDiagramGenerator().generateDrawioDiagram(DefaultConverter.INSTANCE.stringContent(diagram));
097        }
098        
099        protected DiagramGenerator getDiagramGenerator() {
100                return DiagramGenerator.INSTANCE;
101        }
102        
103        protected PlantumlGenerator createPlantumlGenerator() {
104                return new PlantumlGenerator(getDiagramGenerator());
105        }
106        
107        protected DrawioGenerator createDrawioGenerator() {
108                return new DrawioGenerator(getDiagramGenerator());
109        }
110
111}
