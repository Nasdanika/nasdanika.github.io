001package org.nasdanika.diagram.gen;
002
003import java.io.File;
004import java.io.IOException;
005import java.net.URL;
006import java.nio.file.Files;
007
008import org.eclipse.emf.common.util.URI;
009import org.eclipse.emf.ecore.resource.Resource;
010import org.nasdanika.common.DefaultConverter;
011import org.nasdanika.common.DiagramGenerator;
012import org.nasdanika.common.Util;
013import org.nasdanika.common.DiagramGenerator.Dialect;
014import org.nasdanika.diagram.Diagram;
015import org.nasdanika.ncore.Marker;
016
017/**
018 * Generates diagram HTML delegating to different generators based on diagram type.
019 * @author Pavel
020 *
021 */
022public class Generator {
023        
024        public static Generator INSTANCE = new Generator();
025
026        private static final String PLANTUML_SCHEMA = "plantuml:";
027        private static final String DRAWIO_SCHEMA = "drawio:";
028
029        public String generate(Diagram diagram) throws Exception {
030                String type = diagram.getType();
031                if (type == null) {
032                        type = "plantuml:uml";
033                }
034                if (type.startsWith(PLANTUML_SCHEMA)) {
035                        Dialect dialect = Dialect.valueOf(type.substring(PLANTUML_SCHEMA.length()).toUpperCase());
036                        return createPlantumlGenerator().generateDiagram(diagram, dialect);
037                }
038                
039                if (type.startsWith(DRAWIO_SCHEMA)) {
040                        String diagramURI = type.substring(DRAWIO_SCHEMA.length());
041                        URI uri = URI.createURI(diagramURI);
042                        
043                        URI markerBase = null;
044                        Marker marker = diagram.getMarker();
045                        if (marker != null && !Util.isBlank(marker.getLocation())) {
046                                markerBase = URI.createURI(marker.getLocation());
047                        }
048
049                        URI resourceBase = null;
050                        Resource resource = diagram.eResource();
051                        if (resource != null) {
052                                resourceBase = resource.getURI();
053                        }
054                        
055                        if (markerBase != null && markerBase.hasAbsolutePath() && diagramURI.startsWith("./")) {
056                                uri = uri.resolve(markerBase);
057                        } else if (resourceBase != null) {
058                                uri = uri.resolve(resourceBase);
059                        }
060                        
061                        if (uri.isFile()) {
062                                String fileStr = uri.toFileString();
063                                File diagramFile = new File(fileStr);
064                                if (diagramFile.exists()) {
065                                        // TODO - merging - read, merge models, write if result is not null.
066                                } else {
067                                        File container = diagramFile.getParentFile();
068                                        if (container.isDirectory() || container.mkdirs()) {
069                                                Files.write(diagramFile.toPath(), DefaultConverter.INSTANCE.toByteArray(createDrawioGenerator().generateModel(diagram)));
070                                        }
071                                }
072                        }                                               
073                        
074                        URL url = new URL(uri.toString());                      
075                        return generateDrawioHtml(url);
076                }
077                
078                if (type.equals("drawio")) {
079                        return createDrawioGenerator().generateDiagram(diagram);
080                }
081                
082                throw new IllegalArgumentException("Unsupported diagram type: " + type);
083        }
084        
085        /**
086         * Generates draw.io HTML from diagram resource.
087         * @param diagram
088         * @return
089         * @throws IOException
090         */
091        public String generateDrawioHtml(URL diagram) throws Exception {
092                return getDiagramGenerator().generateDrawioDiagram(DefaultConverter.INSTANCE.stringContent(diagram));
093        }
094        
095        protected DiagramGenerator getDiagramGenerator() {
096                return DiagramGenerator.INSTANCE;
097        }
098        
099        protected PlantumlGenerator createPlantumlGenerator() {
100                return new PlantumlGenerator(getDiagramGenerator());
101        }
102        
103        protected DrawioGenerator createDrawioGenerator() {
104                return new DrawioGenerator(getDiagramGenerator());
105        }
106
107}
