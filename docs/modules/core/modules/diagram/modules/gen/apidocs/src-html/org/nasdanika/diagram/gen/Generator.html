001package org.nasdanika.diagram.gen;
002
003import java.io.File;
004import java.io.IOException;
005import java.net.URL;
006import java.nio.file.Files;
007
008import org.eclipse.emf.common.util.URI;
009import org.eclipse.emf.ecore.resource.Resource;
010import org.nasdanika.common.DefaultConverter;
011import org.nasdanika.common.DiagramGenerator;
012import org.nasdanika.common.DiagramGenerator.Dialect;
013import org.nasdanika.diagram.Diagram;
014
015/**
016 * Generates diagram HTML delegating to different generators based on diagram type.
017 * @author Pavel
018 *
019 */
020public class Generator {
021        
022        public static Generator INSTANCE = new Generator();
023
024        private static final String PLANTUML_SCHEMA = "plantuml:";
025        private static final String DRAWIO_SCHEMA = "drawio:";
026
027        public String generate(Diagram diagram) throws Exception {
028                String type = diagram.getType();
029                if (type == null) {
030                        type = "plantuml:uml";
031                }
032                if (type.startsWith(PLANTUML_SCHEMA)) {
033                        Dialect dialect = Dialect.valueOf(type.substring(PLANTUML_SCHEMA.length()).toUpperCase());
034                        return createPlantumlGenerator().generateDiagram(diagram, dialect);
035                }
036                
037                if (type.startsWith(DRAWIO_SCHEMA)) {
038                        URI uri = URI.createURI(type.substring(DRAWIO_SCHEMA.length()));
039                        Resource dResource = diagram.eResource();
040                        if (dResource != null) {
041                                URI rURI = dResource.getURI();
042                                if (rURI != null) {
043                                        uri = uri.resolve(rURI);
044                                }
045                        }
046                        if (uri.isFile()) {
047                                String fileStr = uri.toFileString();
048                                File diagramFile = new File(fileStr);
049                                if (diagramFile.exists()) {
050                                        // TODO - merging - read, merge models, write if result is not null.
051                                } else {
052                                        File container = diagramFile.getParentFile();
053                                        if (container.isDirectory() || container.mkdirs()) {
054                                                Files.write(diagramFile.toPath(), DefaultConverter.INSTANCE.toByteArray(createDrawioGenerator().generateModel(diagram)));
055                                        }
056                                }
057                        }                                               
058                        
059                        URL url = new URL(uri.toString());
060                        
061                        return generateDrawioHtml(url);
062                }
063                
064                if (type.equals("drawio")) {
065                        return createDrawioGenerator().generateDiagram(diagram);
066                }
067                
068                throw new IllegalArgumentException("Unsupported diagram type: " + type);
069        }
070        
071        /**
072         * Generates draw.io HTML from diagram resource.
073         * @param diagram
074         * @return
075         * @throws IOException
076         */
077        public String generateDrawioHtml(URL diagram) throws Exception {
078                return getDiagramGenerator().generateDrawioDiagram(DefaultConverter.INSTANCE.stringContent(diagram));
079        }
080        
081        protected DiagramGenerator getDiagramGenerator() {
082                return DiagramGenerator.INSTANCE;
083        }
084        
085        protected PlantumlGenerator createPlantumlGenerator() {
086                return new PlantumlGenerator(getDiagramGenerator());
087        }
088        
089        protected DrawioGenerator createDrawioGenerator() {
090                return new DrawioGenerator(getDiagramGenerator());
091        }
092
093}
