001package org.nasdanika.diagram.gen;
002
003import java.io.File;
004import java.io.IOException;
005import java.nio.file.Files;
006
007import org.eclipse.emf.common.util.URI;
008import org.eclipse.emf.ecore.resource.Resource;
009import org.nasdanika.common.DefaultConverter;
010import org.nasdanika.common.DiagramGenerator;
011import org.nasdanika.common.DiagramGenerator.Dialect;
012import org.nasdanika.common.Util;
013import org.nasdanika.diagram.Diagram;
014import org.nasdanika.ncore.Marker;
015
016/**
017 * Generates diagram HTML delegating to different generators based on diagram type.
018 * @author Pavel
019 *
020 */
021public class Generator {
022        
023        public static Generator INSTANCE = new Generator();
024
025        public static final String PLANTUML_SCHEMA = "plantuml:";
026        public static final String DRAWIO_SCHEMA = "drawio:";
027
028        public String generate(Diagram diagram) throws Exception {
029                String type = diagram.getType();
030                if (type == null) {
031                        type = "plantuml:uml";
032                }
033                if (type.startsWith(PLANTUML_SCHEMA)) {
034                        Dialect dialect = Dialect.valueOf(type.substring(PLANTUML_SCHEMA.length()).toUpperCase());
035                        return createPlantumlGenerator().generateDiagram(diagram, dialect);
036                }
037                
038                if (type.startsWith(DRAWIO_SCHEMA)) {
039                        String diagramURI = type.substring(DRAWIO_SCHEMA.length());
040                        URI uri = URI.createURI(diagramURI);
041                        
042                        URI markerBase = null;
043                        Marker marker = diagram.getMarker();
044                        if (marker != null && !Util.isBlank(marker.getLocation())) {
045                                markerBase = URI.createURI(marker.getLocation());
046                        }
047
048                        URI resourceBase = null;
049                        Resource resource = diagram.eResource();
050                        if (resource != null) {
051                                resourceBase = resource.getURI();
052                        }
053                        
054                        if (markerBase != null && markerBase.hasAbsolutePath() && diagramURI.startsWith("./")) {
055                                uri = uri.resolve(markerBase);
056                        } else if (resourceBase != null) {
057                                uri = uri.resolve(resourceBase);
058                        }
059                        
060                        if (uri.isFile()) {
061                                String fileStr = uri.toFileString();
062                                File diagramFile = new File(fileStr);
063                                if (diagramFile.exists()) {
064                                        // TODO - merging - read, merge models, write if result is not null.
065                                } else {
066                                        File container = diagramFile.getParentFile();
067                                        if (container.isDirectory() || container.mkdirs()) {
068                                                Files.write(diagramFile.toPath(), DefaultConverter.INSTANCE.toByteArray(createDrawioGenerator().generateModel(diagram)));
069                                        }
070                                }
071                        }                                               
072                        
073                        return generateDrawioHtml(uri);
074                }
075                
076                if (type.equals("drawio")) {
077                        return createDrawioGenerator().generateDiagram(diagram);
078                }
079                
080                throw new IllegalArgumentException("Unsupported diagram type: " + type);
081        }
082        
083        /**
084         * Generates draw.io HTML from diagram resource.
085         * @param diagram
086         * @return
087         * @throws IOException
088         */
089        public String generateDrawioHtml(URI diagram) throws Exception {
090                return getDiagramGenerator().generateDrawioDiagram(DefaultConverter.INSTANCE.stringContent(diagram));
091        }
092        
093        protected DiagramGenerator getDiagramGenerator() {
094                return DiagramGenerator.INSTANCE;
095        }
096        
097        protected PlantumlGenerator createPlantumlGenerator() {
098                return new PlantumlGenerator(getDiagramGenerator());
099        }
100        
101        protected DrawioGenerator createDrawioGenerator() {
102                return new DrawioGenerator(getDiagramGenerator());
103        }
104
105}
