001package org.nasdanika.engineering.util;
002
003import java.util.Collection;
004import java.util.Collections;
005
006import org.eclipse.emf.common.util.URI;
007import org.eclipse.emf.ecore.EObject;
008import org.nasdanika.common.Context;
009import org.nasdanika.common.ProgressMonitor;
010import org.nasdanika.common.Supplier;
011
012/**
013 * Loads root element from a resource URI passed to constructor.
014 * @author Pavel
015 *
016 */
017public class EngineeringYamlSupplier extends EngineeringYamlLoadingExecutionParticipant implements Supplier<EObject> {
018
019        private URI uri;
020        
021        public EngineeringYamlSupplier(URI uri, Context context) {
022                super(context);
023                this.uri = uri;
024        }
025
026        @Override
027        public EObject execute(ProgressMonitor progressMonitor) throws Exception {
028                return resourceSet.getResource(uri, false).getContents().get(0);
029        }
030
031        @Override
032        protected Collection<URI> getResources() {
033                return Collections.singletonList(uri);
034        }
035
036}
