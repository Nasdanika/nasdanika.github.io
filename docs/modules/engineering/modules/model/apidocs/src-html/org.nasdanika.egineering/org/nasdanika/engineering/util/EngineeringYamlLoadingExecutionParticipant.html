001package org.nasdanika.engineering.util;
002
003import java.util.Collection;
004
005import org.eclipse.emf.ecore.EPackage;
006import org.nasdanika.common.Context;
007import org.nasdanika.engineering.EngineeringPackage;
008import org.nasdanika.engineering.journey.JourneyPackage;
009import org.nasdanika.flow.util.FlowYamlLoadingExecutionParticipant;
010import org.nasdanika.html.model.app.AppPackage;
011import org.nasdanika.html.model.bootstrap.BootstrapPackage;
012import org.nasdanika.html.model.html.HtmlPackage;
013
014/**
015 * {@link YamlLoadingSupplier} for Engineering {@link EPackage}s.
016 * Registers exec- loader. 
017 * @author Pavel
018 *
019 */
020public abstract class EngineeringYamlLoadingExecutionParticipant extends FlowYamlLoadingExecutionParticipant {
021
022        public EngineeringYamlLoadingExecutionParticipant(Context context) {
023                super(context);
024        }
025
026        @Override
027        protected Collection<EPackage> getEPackages() {
028                Collection<EPackage> ret = super.getEPackages(); 
029                ret.add(HtmlPackage.eINSTANCE);
030                ret.add(BootstrapPackage.eINSTANCE);
031                ret.add(AppPackage.eINSTANCE);
032                ret.add(EngineeringPackage.eINSTANCE);
033                ret.add(JourneyPackage.eINSTANCE);
034                return ret;
035        }
036
037}
