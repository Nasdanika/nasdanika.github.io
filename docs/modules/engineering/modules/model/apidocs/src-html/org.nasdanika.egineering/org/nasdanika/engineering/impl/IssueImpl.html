425        @Override
426        public double getCompletion() {
427                IssueStatus status = getStatus();
428                if (status != null && status.isDone()) {
429                        return 1;
430                }
431                double rate = getAssignee().getRate();
432                double remaining = getRemainingCost() + rate * getRemainingEffort();
433                double total = getCost() + rate * getEffort();
434                for (Issue child: getChildren()) {
435                        double cc = child.getCompletion();
436                        double ctc = child.getTotalCost();
437                        total += ctc;
438                        if (Double.isFinite(cc)) {
439                                remaining += (1 - cc) * ctc; 
440                        }
441                }
442                return total == 0 ? Double.NaN : Math.max(0, total - remaining) / total;
443        }
