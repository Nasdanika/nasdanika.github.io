045        @Override
046        protected Action buildAction(
047                        Action action,
048                        BiConsumer<EObject,Action> registry, 
049                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
050                        ProgressMonitor progressMonitor) throws Exception {
051                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
052                
053                createIssuesAction(action, registry, resolveConsumer, progressMonitor);         
054                createAllIssuesAction(action, registry, resolveConsumer, progressMonitor);              
055                createPrinciplesActions(action, registry, resolveConsumer, progressMonitor);
056                createDiscussionAction(action, registry, resolveConsumer, progressMonitor);             
057                
058                createOutboundConnectionsActions(action, registry, resolveConsumer, progressMonitor);
059                
060                return action;
061        }

063        protected void createOutboundConnectionsActions(
064                        Action action,
065                        BiConsumer<EObject,Action> registry, 
066                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
067                        ProgressMonitor progressMonitor) throws Exception {
068                
069                EList<Action> groupAnonymous = action.getAnonymous();
070                for (Connection outboundConnection: getTarget().getOutboundConnections()) {

072                }
073        }

076                        Action action,
077                        BiConsumer<EObject,Action> registry, 
078                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer,

095                }
096                
097        }

134        private static int compareActions(Action a, Action b) {
135                if (a == null) {
136                        if (b == null) {
137                                return 0;
138                        }
139                        return 1;
140                }
141                if (b == null) {
142                        return -1;
143                }
144                
145                String aText = a.getText();
146                String bText = b.getText();             
147                if (Util.isBlank(aText)) {
148                        return Util.isBlank(bText) ? 0 : 1;
149                }
150                if (Util.isBlank(bText)) {
151                        return -1;
152                }
153                return aText.compareTo(bText);
154        }

156        @Override
157        protected void resolve(
158                        Action action, 
159                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
160                        ProgressMonitor progressMonitor) throws Exception {
161                super.resolve(action, context, progressMonitor);
162                
163                // Outbound connections
164                EList<Connection> outboundConnections = getTarget().getOutboundConnections();
165                if (!outboundConnections.isEmpty()) {
166                        Action outboundConnectionsAction = AppFactory.eINSTANCE.createAction();
167                        outboundConnectionsAction.setText("Outbound connections");
168                        action.getSections().add(outboundConnectionsAction);
169                        
170                        Table outboundConnectionsTable = buildTable(
171                                        outboundConnections, 
172                                        action, 
173                                        EngineeringPackage.Literals.CONNECTION_SOURCE__OUTBOUND_CONNECTIONS, 
174                                        context, 
175                                        progressMonitor, 
176                                        createColumnBuilder("Name"),
177                                        createColumnBuilder(EngineeringPackage.Literals.CONNECTION__TARGET),
178                                        createColumnBuilder(NcorePackage.Literals.MODEL_ELEMENT__DESCRIPTION));
179                        
180                        outboundConnectionsAction.getContent().add(outboundConnectionsTable);                   
181                }
182                                
183                // Inbound connections
184                Comparator<Connection> inboundConnectionComparator = (a, b) -> {
185                        if (a == b) {
186                                return 0;
187                        }
188                        
189                        // Connections
190                        int ret = compareActions(context.getAction(a), context.getAction(b));
191                        return ret == 0 ? compareActions(context.getAction(a.eContainer()), context.getAction(b.eContainer())) : ret;
192                };
193                List<Connection> inboundConnections = getTarget().getInboundConnections().stream().sorted(inboundConnectionComparator).collect(Collectors.toList());
194                if (!inboundConnections.isEmpty()) {
195                        Action inboundConnectionsAction = AppFactory.eINSTANCE.createAction();
196                        inboundConnectionsAction.setText("Inbound connections");
197                        action.getSections().add(inboundConnectionsAction);
198                        
199                        Table inboundConnectionsTable = buildTable(
200                                        inboundConnections, 
201                                        action, 
202                                        EngineeringPackage.Literals.CONNECTION_TARGET__INBOUND_CONNECTIONS, 
203                                        context, 
204                                        progressMonitor, 
205                                        createColumnBuilder("Name"),
206                                        createColumnBuilder(EcorePackage.Literals.EOBJECT___ECONTAINER, "Source"),
207                                        createColumnBuilder(NcorePackage.Literals.MODEL_ELEMENT__DESCRIPTION));
208                        
209                        inboundConnectionsAction.getContent().add(inboundConnectionsTable);                     
210                }
211                
212                T engineeredElement = getTarget();
213                EList<Issue> issues = engineeredElement.getIssues();
214                if (!issues.isEmpty()) {
215                        String issuesGroupUUID = action.getUuid() + "-issues";
216                        Optional<Action> issuesActionOptional = action.getNavigation().stream()
217                                        .filter(Action.class::isInstance)                                       
218                                        .map(Action.class::cast)
219                                        .filter(a -> issuesGroupUUID.equals(a.getUuid()))
220                                        .findFirst();
221                        
222                        Action issuesAction = issuesActionOptional.get();
223                        issuesAction.getContent().add(renderList(issues, true, createIssueChildrenProvider(), issuesAction, EngineeringPackage.Literals.ENGINEERED_ELEMENT__ISSUES, context, progressMonitor)); // Table?
224                }
225                
226                EList<Issue> allIssues = engineeredElement.getAllIssues();
227                if (!allIssues.isEmpty()) {
228                        String allIssuesGroupUUID = action.getUuid() + "-all-issues";
229                        Optional<Action> allIssuesActionOptional = action.getNavigation().stream()
230                                        .filter(Action.class::isInstance)                                       
231                                        .map(Action.class::cast)
232                                        .filter(a -> allIssuesGroupUUID.equals(a.getUuid()))
233                                        .findFirst();
234                        Action allIssuesAction = allIssuesActionOptional.get();
235                        Tag allIssuesTable = buildDynamicTable(
236                                        allIssues, 
237                                        action, 
238                                        EngineeringPackage.Literals.ENGINEERED_ELEMENT__ALL_ISSUES,
239                                        "engineered-element--all-issues",
240                                        null,
241                                        context, 
242                                        progressMonitor,
243                                        createDynamicColumnBuilder("Issue", true),
244                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__TARGET, true),
245                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__ASSIGNEE, true),
246                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__CATEGORIES, true),
247                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__PRIORITY, true),
248                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__SEVERITY, true),
249                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__STATUS, true),
250                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__WORKABLE, true),
251                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__BENEFIT, false),
252                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__COST, false),
253                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__EFFORT, false),
254                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__REMAINING_COST, false),
255                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__REMAINING_EFFORT, false),
256                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION, true),
257                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ENGINEERED_CAPABILITY__RELEASES, true),
258                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__CONTRIBUTES_TO, true),
259                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__INCREMENT, true),
260                                        createDynamicColumnBuilder(NcorePackage.Literals.PERIOD__START, true),
261                                        createDynamicColumnBuilder(NcorePackage.Literals.PERIOD__END, true),
262                                        createDynamicColumnBuilder(NcorePackage.Literals.PERIOD__DURATION, false));
263                        
264                        allIssuesAction.getContent().add(allIssuesTable);
265                }
266                
267                EList<Principle> principles = getTarget().getPrinciples();
268                if (!principles.isEmpty()) {
269                        String principlesGroupUUID = action.getUuid() + "-principles";
270                        Optional<Action> principlesActionOptional = action.getNavigation().stream()
271                                        .filter(Action.class::isInstance)
272                                        .map(Action.class::cast)
273                                        .filter(a -> principlesGroupUUID.equals(a.getUuid()))
274                                        .findFirst();
275                        
276                        Action principlesAction = principlesActionOptional.get();
277                        principlesAction.getContent().add(renderList(principles, true, createPrincipleChildrenListProvider(), action, EngineeringPackage.Literals.ENGINEERED_ELEMENT__PRINCIPLES, context, progressMonitor));
278                }
279        }

281        protected ContentProvider<Principle> createPrincipleChildrenListProvider() {
282                ContentProvider<Principle> goalChildrenListProvider = new ContentProvider<Principle>() {
283
284                        @Override
285                        public List<EObject> createContent(
286                                        Principle element, 
287                                        Action base, 
288                                        ETypedElement typedElement,
289                                        org.nasdanika.html.emf.EObjectActionResolver.Context context, 
290                                        ProgressMonitor progressMonitor) throws Exception {
291                                
292                                EList<Principle> children = element.getChildren();
293                                if (children.isEmpty()) {
294                                        return null;
295                                }
296                                return Collections.singletonList(renderList(children, true, this, base, EngineeringPackage.Literals.PRINCIPLE__CHILDREN, context, progressMonitor));
297                        }
298                        
299                };
300                return goalChildrenListProvider;
301        }

333                        group.setText("Principles");

340                        }
341                }
