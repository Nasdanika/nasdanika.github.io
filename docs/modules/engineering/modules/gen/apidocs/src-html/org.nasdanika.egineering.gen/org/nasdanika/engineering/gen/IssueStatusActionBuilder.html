001package org.nasdanika.engineering.gen;
002
003import java.util.List;
004
005import org.eclipse.emf.ecore.ETypedElement;
006import org.eclipse.emf.ecore.EcorePackage;
007import org.nasdanika.common.Context;
008import org.nasdanika.common.ProgressMonitor;
009import org.nasdanika.engineering.EngineeringPackage;
010import org.nasdanika.engineering.Issue;
011import org.nasdanika.engineering.IssueStatus;
012import org.nasdanika.html.model.app.Action;
013import org.nasdanika.html.model.app.AppFactory;
014import org.nasdanika.html.model.bootstrap.Table;
015import org.nasdanika.ncore.NcorePackage;
016
017public class IssueStatusActionBuilder extends NamedElementActionBuilder<IssueStatus> {
018        
019        public IssueStatusActionBuilder(IssueStatus target, Context context) {
020                super(target, context);         
021        }
022        
023        @Override
024        protected List<ETypedElement> getProperties() {
025                List<ETypedElement> properties = super.getProperties();
026                properties.add(EngineeringPackage.Literals.ISSUE_STATUS__DONE);
027                return properties;
028        }
029                
030        @Override
031        protected void resolve(
032                        Action action, 
033                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
034                        ProgressMonitor progressMonitor) throws Exception {
035                super.resolve(action, context, progressMonitor);
036                
037                List<Issue> issues = getTarget().getIssues();
038                if (!issues.isEmpty()) {
039                        Table issuesTable = buildTable(
040                                        issues, 
041                                        action, 
042                                        EngineeringPackage.Literals.ISSUE_CATEGORY__ISSUES, 
043                                        context, 
044                                        progressMonitor,
045                                        createColumnBuilder("Issue"),
046                                        createColumnBuilder(EcorePackage.Literals.EOBJECT___ECONTAINER, "Module"),
047                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__INCREMENT),
048                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__ASSIGNEE),
049                                        createColumnBuilder(NcorePackage.Literals.PERIOD__START),
050                                        createColumnBuilder(NcorePackage.Literals.PERIOD__END),
051                                        createColumnBuilder(NcorePackage.Literals.PERIOD__DURATION),
052                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
053                        
054                        Action section = AppFactory.eINSTANCE.createAction();
055                        section.setText("Issues");
056                        section.getContent().add(issuesTable);
057                
058                        action.getSections().add(section);
059                }                       
060        }
061        
062}
