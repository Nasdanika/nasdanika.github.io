001package org.nasdanika.engineering.gen;
002
003import java.util.List;
004import java.util.Optional;
005import java.util.function.BiConsumer;
006
007import org.eclipse.emf.common.util.EList;
008import org.eclipse.emf.ecore.EObject;
009import org.eclipse.emf.ecore.ETypedElement;
010import org.nasdanika.common.Context;
011import org.nasdanika.common.ProgressMonitor;
012import org.nasdanika.engineering.Capability;
013import org.nasdanika.engineering.EngineeringPackage;
014import org.nasdanika.engineering.Issue;
015import org.nasdanika.html.model.app.Action;
016import org.nasdanika.html.model.app.AppFactory;
017import org.nasdanika.html.model.bootstrap.Table;
018
019public class CapabilityActionBuilder<T extends Capability> extends NamedElementActionBuilder<T> {
020        
021        public CapabilityActionBuilder(T target, Context context) {
022                super(target, context);         
023        }
024        
025        @Override
026        protected Action buildAction(
027                        Action action,
028                        BiConsumer<EObject,Action> registry, 
029                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
030                        ProgressMonitor progressMonitor) throws Exception {
031                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
032                
033                createRequiredByAction(action, registry, resolveConsumer, progressMonitor);             
034                
035                return action;
036        }
037        
038        @Override
039        protected List<ETypedElement> getProperties() {
040                List<ETypedElement> properties = super.getProperties();
041                properties.add(EngineeringPackage.Literals.CAPABILITY__AVAILABLE);
042                return properties;
043        }
044        
045        protected void createRequiredByAction(
046                        Action action,
047                        BiConsumer<EObject,Action> registry, 
048                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
049                        ProgressMonitor progressMonitor) throws Exception {
050                
051                List<Issue> requiredBy = getTarget().getRequiredBy();
052                if (!requiredBy.isEmpty()) {
053                        Action group = AppFactory.eINSTANCE.createAction();
054                        group.setText("Required by");
055                        group.setUuid(action.getUuid() + "-required-by");
056                        group.setLocation("required-by.html");
057                        action.getSections().add(group);
058                }               
059        }
060                
061        @Override
062        protected void resolve(
063                        Action action, 
064                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
065                        ProgressMonitor progressMonitor) throws Exception {
066                super.resolve(action, context, progressMonitor);
067                
068                EList<Issue> requiredBy = getTarget().getRequiredBy();
069                if (!requiredBy.isEmpty()) {
070                        String requiredByGroupUUID = action.getUuid() + "-required-by";
071                        Optional<Action> requiredByOptional = action.getSections().stream()
072                                        .filter(a -> requiredByGroupUUID.equals(a.getUuid()))
073                                        .findFirst();
074                        Action issuesAction = requiredByOptional.get();
075                        Table requiredByTable = buildTable(
076                                        requiredBy, 
077                                        action, 
078                                        EngineeringPackage.Literals.CAPABILITY__REQUIRED_BY, 
079                                        context, 
080                                        progressMonitor,
081                                        createColumnBuilder("Issue"),
082                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__TARGET),
083                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__ASSIGNEE),
084                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__CATEGORIES),
085                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__PRIORITY),
086                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__SEVERITY),
087                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__STATUS),
088                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__WORKABLE),
089                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__BENEFIT),
090                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__COST),
091                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__EFFORT),
092                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__REMAINING_COST),
093                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__REMAINING_EFFORT),
094                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
095                        
096                        issuesAction.getContent().add(requiredByTable);
097                }
098                
099        }
100                
101}
