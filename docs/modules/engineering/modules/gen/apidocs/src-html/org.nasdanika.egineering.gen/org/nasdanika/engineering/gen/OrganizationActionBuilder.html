001package org.nasdanika.engineering.gen;
002
003import java.util.Collections;
004import java.util.List;
005import java.util.Optional;
006import java.util.function.BiConsumer;
007import java.util.stream.Collectors;
008
009import org.eclipse.emf.common.util.EList;
010import org.eclipse.emf.ecore.EObject;
011import org.eclipse.emf.ecore.ETypedElement;
012import org.nasdanika.common.Context;
013import org.nasdanika.common.ProgressMonitor;
014import org.nasdanika.engineering.Engineer;
015import org.nasdanika.engineering.EngineeringPackage;
016import org.nasdanika.engineering.Organization;
017import org.nasdanika.html.model.app.Action;
018import org.nasdanika.html.model.app.AppFactory;
019import org.nasdanika.ncore.util.NamedElementComparator;
020
021public class OrganizationActionBuilder<T extends Organization> extends EngineerActionBuilder<T> {
022        
023        public OrganizationActionBuilder(T target, Context context) {
024                super(target, context); 
025        }
026        
027        @Override
028        protected Action buildAction(
029                        Action action,
030                        BiConsumer<EObject,Action> registry, 
031                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
032                        ProgressMonitor progressMonitor) throws Exception {
033                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
034                createEngineersActions(action, registry, resolveConsumer, progressMonitor);
035                return action;
036        }
037        
038        /**
039         * Creates a list of actions for personas and adds them to the list of group's anonymous actions. 
040         * @param progressMonitor
041         * @return An empty list if there are no personas. A singleton list containing a grouping action containing persona actions otherwise.
042         * @throws Exception 
043         */
044        protected void createEngineersActions(
045                        Action action,
046                        BiConsumer<EObject,Action> registry, 
047                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
048                        ProgressMonitor progressMonitor) throws Exception {
049                
050                List<Engineer> engineers = getTarget().getEngineers();
051                if (!engineers.isEmpty()) {
052                        Action group = AppFactory.eINSTANCE.createAction();
053                        group.setText("Engineers");
054                        group.setUuid(action.getUuid() + "-engineers");
055                        group.setLocation("engineers.html");
056                        // TODO - icon, ...
057                        EList<Action> groupAnonymous = group.getAnonymous();
058                        for (Engineer engineer: engineers.stream().sorted(NamedElementComparator.INSTANCE).collect(Collectors.toList())) {
059                                groupAnonymous.add(createChildAction(engineer, registry, resolveConsumer, progressMonitor));
060                        }
061                
062                        action.getNavigation().add(group);
063                }
064        }
065                
066        @Override
067        protected void resolve(
068                        Action action, 
069                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
070                        ProgressMonitor progressMonitor) throws Exception {
071                super.resolve(action, context, progressMonitor);
072                
073                Organization semanticElement = getTarget();
074                EList<Engineer> engineers = semanticElement.getEngineers();
075                if (!engineers.isEmpty()) {
076                        String personaGroupUUID = action.getUuid() + "-engineers";
077                        Optional<Action> personasActionOptional = action.getNavigation().stream()
078                                        .filter(Action.class::isInstance)                                       
079                                        .map(Action.class::cast)
080                                        .filter(a -> personaGroupUUID.equals(a.getUuid()))
081                                        .findFirst();
082                        
083                        Action personasAction = personasActionOptional.get();
084                        personasAction.getContent().add(renderList(engineers, true, createEngineerChildrenProvider(), personasAction, EngineeringPackage.Literals.ENGINEER__PERSONAS, context, progressMonitor)); // Table?
085                }
086                
087        }
088        
089        private ContentProvider<Engineer> createEngineerChildrenProvider() {
090                return new ContentProvider<Engineer>() {
091        
092                        @Override
093                        public List<EObject> createContent(
094                                        Engineer element, 
095                                        Action base, 
096                                        ETypedElement typedElement,
097                                        org.nasdanika.html.emf.EObjectActionResolver.Context context, 
098                                        ProgressMonitor progressMonitor) throws Exception {
099                                
100                                if (element instanceof Organization) {
101                                        EList<Engineer> engineers = ((Organization) element).getEngineers();
102                                        if (!engineers.isEmpty()) {
103                                                return Collections.singletonList(renderList(engineers, true, this, base, EngineeringPackage.Literals.ORGANIZATION__ENGINEERS, context, progressMonitor));
104                                        }
105                                }
106                                return null;
107                        }
108                        
109                };
110        }       
111        
112}
