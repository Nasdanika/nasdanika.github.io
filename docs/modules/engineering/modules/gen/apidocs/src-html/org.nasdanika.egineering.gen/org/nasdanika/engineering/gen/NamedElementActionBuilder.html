001package org.nasdanika.engineering.gen;
002
003import java.util.function.BiConsumer;
004
005import org.eclipse.emf.ecore.EObject;
006import org.nasdanika.common.Context;
007import org.nasdanika.common.ProgressMonitor;
008import org.nasdanika.engineering.NamedElement;
009import org.nasdanika.html.model.app.Action;
010
011public class NamedElementActionBuilder<T extends NamedElement> extends ModelElementActionBuilder<T> {
012        
013        public NamedElementActionBuilder(T target, Context context) {
014                super(target, context);         
015        }
016        
017        @Override
018        protected Action buildAction(
019                        Action action,
020                        BiConsumer<EObject,Action> registry, 
021                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
022                        ProgressMonitor progressMonitor) throws Exception {
023                Action ret = super.buildAction(action, registry, resolveConsumer, progressMonitor);             
024                
025                ret.setText(getTarget().getName()); // Escape?
026                return ret;
027        }
028
029        @Override
030        public String name() {
031                return getTarget().getName();
032        }
033        
034}
