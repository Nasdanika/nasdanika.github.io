001package org.nasdanika.engineering.gen;
002
003import java.util.function.BiConsumer;
004import java.util.function.Consumer;
005
006import org.eclipse.emf.common.util.EList;
007import org.eclipse.emf.ecore.EObject;
008import org.eclipse.emf.ecore.ETypedElement;
009import org.nasdanika.common.Context;
010import org.nasdanika.common.ProgressMonitor;
011import org.nasdanika.engineering.EngineeringPackage;
012import org.nasdanika.engineering.Forum;
013import org.nasdanika.engineering.Topic;
014import org.nasdanika.html.bootstrap.Text.Alignment;
015import org.nasdanika.html.emf.ColumnBuilder;
016import org.nasdanika.html.model.app.Action;
017import org.nasdanika.html.model.app.AppFactory;
018import org.nasdanika.html.model.bootstrap.Appearance;
019import org.nasdanika.html.model.bootstrap.BootstrapFactory;
020import org.nasdanika.html.model.bootstrap.Table;
021import org.nasdanika.html.model.bootstrap.TableCell;
022import org.nasdanika.html.model.bootstrap.Text;
023
024public class ForumActionBuilder<T extends Forum> extends NamedElementActionBuilder<T> {
025        
026        public ForumActionBuilder(T target, Context context) {
027                super(target, context);         
028        }
029        
030        @Override
031        protected Action buildAction(
032                        Action action,
033                        BiConsumer<EObject, Action> registry,
034                        Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer,
035                        ProgressMonitor progressMonitor) throws Exception {
036                
037                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
038                
039                EList<Action> groupAnonymous = action.getAnonymous();
040                for (Forum forum: getTarget().getDiscussion()) {
041                        groupAnonymous.add(createChildAction(forum, registry, resolveConsumer, progressMonitor));
042                }               
043                for (Topic topic: getTarget().getTopics()) {
044                        groupAnonymous.add(createChildAction(topic, registry, resolveConsumer, progressMonitor));
045                }               
046                
047                return action;
048        }
049                
050        @Override
051        protected void resolve(
052                        Action action, 
053                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
054                        ProgressMonitor progressMonitor) throws Exception {
055                super.resolve(action, context, progressMonitor);
056                
057                T forum = getTarget();
058                
059                // Rendering topics and sub-forums under discussion action if it is present (EngineeredElement) or under this action (Forum) 
060                String discussionActionUUID = action.getUuid() + "-discussion";
061                Action discussionAction = action.getNavigation().stream()
062                                .filter(Action.class::isInstance)                                       
063                                .map(Action.class::cast)
064                                .filter(a -> discussionActionUUID.equals(a.getUuid()))
065                                .findFirst()
066                                .orElse(action);
067
068                EList<Forum> forums = forum.getDiscussion();
069                if (!forums.isEmpty()) {
070                        Action forumsAction = AppFactory.eINSTANCE.createAction();
071                        forumsAction.setText("Forums");
072                        discussionAction.getSections().add(forumsAction);
073                        
074                        ColumnBuilder<Forum> topicsColumnBuilder = new ColumnBuilder<Forum>() {
075
076                                @Override
077                                public void buildHeader(
078                                                TableCell header, 
079                                                Action base, 
080                                                ETypedElement typedElement,
081                                                org.nasdanika.html.emf.EObjectActionResolver.Context context, 
082                                                ProgressMonitor progressMonitor) throws Exception {
083
084                                        header.getContent().add(createText("Topics"));
085                                }
086
087                                @Override
088                                public void buildCell(
089                                                Forum rowElement, 
090                                                TableCell cell, 
091                                                Action base, 
092                                                ETypedElement typedElement,
093                                                org.nasdanika.html.emf.EObjectActionResolver.Context context, 
094                                                ProgressMonitor progressMonitor) throws Exception {
095
096                                        cell.getContent().add(createText(String.valueOf(rowElement.getTopics().size())));
097                                        Appearance cellAppearance = cell.getAppearance();
098                                        if (cellAppearance == null) {
099                                                cellAppearance = BootstrapFactory.eINSTANCE.createAppearance();
100                                                cell.setAppearance(cellAppearance);
101                                        }
102                                        Text textAppearance = cellAppearance.getText();
103                                        if (textAppearance == null) {
104                                                textAppearance = BootstrapFactory.eINSTANCE.createText();
105                                                cellAppearance.setText(textAppearance);
106                                        }
107                                        textAppearance.setAlignment(Alignment.CENTER);
108                                }
109                        };
110                        
111                        Table forumsTable = buildTable(
112                                        forums, 
113                                        action, 
114                                        EngineeringPackage.Literals.FORUM__DISCUSSION, 
115                                        context, 
116                                        progressMonitor,
117                                        createColumnBuilder("Forum"),
118                                        topicsColumnBuilder
119                                        // last activity
120                                        );
121                        
122                        forumsAction.getContent().add(forumsTable);
123                }
124                
125                EList<Topic> topics = forum.getTopics();
126                if (!topics.isEmpty()) {
127                        Action topicsAction = AppFactory.eINSTANCE.createAction();
128                        topicsAction.setText("Topics");
129                        discussionAction.getSections().add(topicsAction);
130                        
131                        ColumnBuilder<Topic> messagesColumnBuilder = new ColumnBuilder<Topic>() {
132
133                                @Override
134                                public void buildHeader(
135                                                TableCell header, 
136                                                Action base, 
137                                                ETypedElement typedElement,
138                                                org.nasdanika.html.emf.EObjectActionResolver.Context context, 
139                                                ProgressMonitor progressMonitor) throws Exception {
140
141                                        header.getContent().add(createText("Messages"));
142                                }
143
144                                @Override
145                                public void buildCell(
146                                                Topic rowElement, 
147                                                TableCell cell, 
148                                                Action base, 
149                                                ETypedElement typedElement,
150                                                org.nasdanika.html.emf.EObjectActionResolver.Context context, 
151                                                ProgressMonitor progressMonitor) throws Exception {
152
153                                        cell.getContent().add(createText(String.valueOf(rowElement.getMessages().size())));
154                                        Appearance cellAppearance = cell.getAppearance();
155                                        if (cellAppearance == null) {
156                                                cellAppearance = BootstrapFactory.eINSTANCE.createAppearance();
157                                                cell.setAppearance(cellAppearance);
158                                        }
159                                        Text textAppearance = cellAppearance.getText();
160                                        if (textAppearance == null) {
161                                                textAppearance = BootstrapFactory.eINSTANCE.createText();
162                                                cellAppearance.setText(textAppearance);
163                                        }
164                                        textAppearance.setAlignment(Alignment.CENTER);
165                                }
166                        };
167                        
168                        Table topicsTable = buildTable(
169                                        topics, 
170                                        action, 
171                                        EngineeringPackage.Literals.FORUM__TOPICS, 
172                                        context, 
173                                        progressMonitor,
174                                        createColumnBuilder("Topic"),
175                                        messagesColumnBuilder
176                                        // last activity
177                                        );
178                        
179                        topicsAction.getContent().add(topicsTable);
180                }
181                
182        }
183        
184}
