042        @Override
043        protected Action buildAction(
044                        Action action,
045                        BiConsumer<EObject,Action> registry, 
046                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
047                        ProgressMonitor progressMonitor) throws Exception {
048                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
049                
050                createDomainActions(action, registry, resolveConsumer, progressMonitor);                
051                createModuleActions(action, registry, resolveConsumer, progressMonitor);                
052                createIncrementActions(action, registry, resolveConsumer, progressMonitor);             
053                createPersonaActions(action, registry, resolveConsumer, progressMonitor);
054                createIssueGroupingActions(action, registry, resolveConsumer, progressMonitor);
055                createObjectiveActions(action, registry, resolveConsumer, progressMonitor);
056                createAssignmentsAction(action, registry, resolveConsumer, progressMonitor);            
057                createEngineeredElementStatusAction(action, registry, resolveConsumer, progressMonitor);                
058                
059                return action;
060        }

070

072        }

074        protected void createEngineeredElementStatusAction(
075                        Action action,
076                        BiConsumer<EObject,Action> registry, 
077                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
078                        ProgressMonitor progressMonitor) throws Exception {
079                
080                T engineer = getTarget();
081                EList<EngineeredElementStatus> engineeredElementStatuses = engineer.getEngineeredElementStatuses();
082                if (engineeredElementStatuses.isEmpty()) {
083                        return;
084                }
085
086                Action group = AppFactory.eINSTANCE.createAction();
087                group.setText("Engineered Element Statuses");
088                group.setUuid(action.getUuid() + "-engineered-element-statuses");
089                group.setLocation("engineered-element-statuses.html");
090                EList<Action> groupAnonymous = group.getAnonymous();
091                for (EngineeredElementStatus status: engineeredElementStatuses) {
092                        groupAnonymous.add(createChildAction(status, registry, resolveConsumer, progressMonitor));
093                }
094        
095                action.getNavigation().add(group);
096        }

116                        Action action,

118                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer,

133        /**
134         * Creates a list of actions for modules. 
135         * @param progressMonitor
136         * @return An empty list if there are no modules. A singleton list containing a grouping action containing module actions otherwise.
137         * @throws Exception 
138         */
139        protected void createModuleActions(
140                        Action action,
141                        BiConsumer<EObject,Action> registry, 
142                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
143                        ProgressMonitor progressMonitor) throws Exception {
144                List<org.nasdanika.engineering.Module> modules = getTarget().getModules();
145                if (!modules.isEmpty()) {
146                        Action group = AppFactory.eINSTANCE.createAction();
147                        Predicate<org.nasdanika.engineering.Module> isProduct = Product.class::isInstance;                      
148                        group.setText(modules.stream().anyMatch(isProduct.negate()) ? "Modules" : "Products");
149                        group.setIcon("fas fa-cubes");
150                        EList<EObject> children = group.getChildren();
151                        for (org.nasdanika.engineering.Module module: modules) {
152                                children.add(createChildAction(module, registry, resolveConsumer, progressMonitor));
153                        }
154        
155                        action.getChildren().add(group);
156                }
157        }

228        protected void createIssueGroupingActions(
229                        Action action,
230                        BiConsumer<EObject,Action> registry, 
231                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
232                        ProgressMonitor progressMonitor) throws Exception {
233                
234                T engineer = getTarget();
235                EList<IssueCategory> issueCategories = engineer.getIssueCategories();
236                EList<IssuePriority> issuePriorities = engineer.getIssuePriorities();
237                EList<IssueSeverity> issueSeverities = engineer.getIssueSeverities();
238                EList<IssueStatus> issueStatuses = engineer.getIssueStatuses();
239                
240                if (issueCategories.isEmpty() 
241                                && issuePriorities.isEmpty()
242                                && issueSeverities.isEmpty()
243                                && issueStatuses.isEmpty()) {
244                        return;
245                }
246
247                Action root = AppFactory.eINSTANCE.createAction();
248                root.setText("Issue");
249                root.setUuid(action.getUuid() + "-issue");
250                action.getNavigation().add(root);
251                
252                if (!issueCategories.isEmpty()) {
253                        Action group = AppFactory.eINSTANCE.createAction();
254                        group.setText("Categories");
255                        group.setUuid(root.getUuid() + "-categories");
256                        group.setLocation("issue-categories.html");
257                        EList<Action> groupAnonymous = group.getAnonymous();
258                        for (IssueCategory issueCategory: issueCategories) {
259                                groupAnonymous.add(createChildAction(issueCategory, registry, resolveConsumer, progressMonitor));
260                        }
261                
262                        root.getChildren().add(group);
263                }
264                
265                if (!issuePriorities.isEmpty()) {
266                        Action group = AppFactory.eINSTANCE.createAction();
267                        group.setText("Priorities");
268                        group.setUuid(root.getUuid() + "-priorities");
269                        group.setLocation("issue-priorities.html");
270                        EList<Action> groupAnonymous = group.getAnonymous();
271                        for (IssuePriority issuePriority: issuePriorities) {
272                                groupAnonymous.add(createChildAction(issuePriority, registry, resolveConsumer, progressMonitor));
273                        }
274                
275                        root.getChildren().add(group);
276                }
277                
278                if (!issueSeverities.isEmpty()) {
279                        Action group = AppFactory.eINSTANCE.createAction();
280                        group.setText("Severities");
281                        group.setUuid(root.getUuid() + "-severities");
282                        group.setLocation("issue-severities.html");
283                        EList<Action> groupAnonymous = group.getAnonymous();
284                        for (IssueSeverity issueSeverity: issueSeverities) {
285                                groupAnonymous.add(createChildAction(issueSeverity, registry, resolveConsumer, progressMonitor));
286                        }
287                
288                        root.getChildren().add(group);
289                }
290                
291                if (!issueStatuses.isEmpty()) {
292                        Action group = AppFactory.eINSTANCE.createAction();
293                        group.setText("Statuses");
294                        group.setUuid(root.getUuid() + "-statuses");
295                        group.setLocation("issue-statuses.html");
296                        EList<Action> groupAnonymous = group.getAnonymous();
297                        for (IssueStatus issueStatus: issueStatuses) {
298                                groupAnonymous.add(createChildAction(issueStatus, registry, resolveConsumer, progressMonitor));
299                        }
300                
301                        root.getChildren().add(group);
302                }
303        }

305        @Override
306        protected void resolve(
307                        Action action, 
308                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
309                        ProgressMonitor progressMonitor) throws Exception {
310                super.resolve(action, context, progressMonitor);
311                
312                T engineer = getTarget();
313                
314                EList<Objective> objectives = engineer.getObjectives();
315                if (!objectives.isEmpty()) {
316                        String objectivesGroupUUID = action.getUuid() + "-objectives";
317                        Optional<Action> objectivesActionOptional = action.getNavigation().stream()
318                                        .filter(Action.class::isInstance)                                       
319                                        .map(Action.class::cast)
320                                        .filter(a -> objectivesGroupUUID.equals(a.getUuid()))
321                                        .findFirst();
322                        
323                        Action objectivesAction = objectivesActionOptional.get();
324                        objectivesAction.getContent().add(renderList(objectives, true, null, objectivesAction, EngineeringPackage.Literals.ENGINEER__OBJECTIVES, context, progressMonitor)); // Table?
325                }
326                
327                EList<Persona> personas = engineer.getPersonas();
328                if (!personas.isEmpty()) {
329                        String personaGroupUUID = action.getUuid() + "-personas";
330                        Optional<Action> personasActionOptional = action.getNavigation().stream()
331                                        .filter(Action.class::isInstance)                                       
332                                        .map(Action.class::cast)
333                                        .filter(a -> personaGroupUUID.equals(a.getUuid()))
334                                        .findFirst();
335                        
336                        Action personasAction = personasActionOptional.get();
337                        personasAction.getContent().add(renderList(personas, true, null, personasAction, EngineeringPackage.Literals.ENGINEER__PERSONAS, context, progressMonitor)); // Table?
338                }
339                
340                String issueGroupUUID = action.getUuid() + "-issue";
341                Optional<Action> issueGroupOptional = action.getNavigation().stream()
342                                .filter(Action.class::isInstance)                                       
343                                .map(Action.class::cast)
344                                .filter(a -> issueGroupUUID.equals(a.getUuid()))
345                                .findFirst();
346                
347                if (issueGroupOptional.isPresent()) {
348                        Action issueGroup = issueGroupOptional.get();
349                        
350                        EList<IssueCategory> issueCategories = engineer.getIssueCategories();
351                        if (!issueCategories.isEmpty()) {
352                                String issueCategoriesUUID = issueGroup.getUuid() + "-categories";
353                                Optional<Action> issueCategoriesActionOptional = issueGroup.getChildren().stream()
354                                                .filter(Action.class::isInstance)                                       
355                                                .map(Action.class::cast)
356                                                .filter(a -> issueCategoriesUUID.equals(a.getUuid()))
357                                                .findFirst();
358                                                                
359                                Action issueCategoriesAction = issueCategoriesActionOptional.get();
360                                issueCategoriesAction.getContent().add(renderList(issueCategories, false, createIssueCategoryChildrenProvider(), issueCategoriesAction, EngineeringPackage.Literals.ENGINEER__ISSUE_CATEGORIES, context, progressMonitor)); // Table?
361                        }
362                        
363                        EList<IssuePriority> issuePriorities = engineer.getIssuePriorities();
364                        if (!issuePriorities.isEmpty()) {
365                                String issuePrioritiesUUID = issueGroup.getUuid() + "-priorities";
366                                Optional<Action> issuePrioritiesActionOptional = issueGroup.getChildren().stream()
367                                                .filter(Action.class::isInstance)                                       
368                                                .map(Action.class::cast)
369                                                .filter(a -> issuePrioritiesUUID.equals(a.getUuid()))
370                                                .findFirst();
371                                
372                                Action issuePrioritiesAction = issuePrioritiesActionOptional.get();
373                                issuePrioritiesAction.getContent().add(renderList(issuePriorities, false, null, issuePrioritiesAction, EngineeringPackage.Literals.ENGINEER__ISSUE_PRIORITIES, context, progressMonitor)); // Table?
374                        }
375                        
376                        EList<IssueSeverity> issueSeverities = engineer.getIssueSeverities();
377                        if (!issueSeverities.isEmpty()) {
378                                String issueSeveritiesUUID = issueGroup.getUuid() + "-severities";
379                                Optional<Action> issueSeveritiesActionOptional = issueGroup.getChildren().stream()
380                                                .filter(Action.class::isInstance)                                       
381                                                .map(Action.class::cast)
382                                                .filter(a -> issueSeveritiesUUID.equals(a.getUuid()))
383                                                .findFirst();
384                                
385                                Action issueSeveritiesAction = issueSeveritiesActionOptional.get();
386                                issueSeveritiesAction.getContent().add(renderList(issueSeverities, false, null, issueSeveritiesAction, EngineeringPackage.Literals.ENGINEER__ISSUE_SEVERITIES, context, progressMonitor)); // Table?
387                        }
388                        
389                        EList<IssueStatus> issueStatuses = engineer.getIssueStatuses();
390                        if (!issueStatuses.isEmpty()) {
391                                String issueStatusesUUID = issueGroup.getUuid() + "-statuses";
392                                Optional<Action> issueStatusesActionOptional = issueGroup.getChildren().stream()
393                                                .filter(Action.class::isInstance)                                       
394                                                .map(Action.class::cast)
395                                                .filter(a -> issueStatusesUUID.equals(a.getUuid()))
396                                                .findFirst();
397                                
398                                Action issueStatusesAction = issueStatusesActionOptional.get();
399                                issueStatusesAction.getContent().add(renderList(issueStatuses, false, null, issueStatusesAction, EngineeringPackage.Literals.ENGINEER__ISSUE_STATUSES, context, progressMonitor)); // Table?
400                        }
401                }
402                
403                // EngineeredElement statuses           
404                EList<EngineeredElementStatus> engineeredElementStatuses = engineer.getEngineeredElementStatuses();
405                if (!engineeredElementStatuses.isEmpty()) {
406                        String engineeredElementStatusesUUID = action.getUuid() + "-engineered-element-statuses";
407                        Optional<Action> engineeredElementStatusesActionOptional = action.getNavigation().stream()
408                                        .filter(Action.class::isInstance)                                       
409                                        .map(Action.class::cast)
410                                        .filter(a -> engineeredElementStatusesUUID.equals(a.getUuid()))
411                                        .findFirst();
412                                                        
413                        Action engineeredElementStatusesAction = engineeredElementStatusesActionOptional.get();
414                        engineeredElementStatusesAction.getContent().add(renderList(engineeredElementStatuses, false, createEngineeredElementStatusChildrenProvider(), engineeredElementStatusesAction, EngineeringPackage.Literals.ENGINEER__ENGINEERED_ELEMENT_STATUSES, context, progressMonitor)); // Table?
415                }
416                
417                // Assignments          
418                EList<Endeavor> assignments = getTarget().getAssignments();
419                if (!assignments.isEmpty()) {
420                        String assignmentsGroupUUID = action.getUuid() + "-assignments";
421                        Optional<Action> assignmentsActionOptional = action.getNavigation().stream()
422                                        .filter(Action.class::isInstance)                                       
423                                        .map(Action.class::cast)
424                                        .filter(a -> assignmentsGroupUUID.equals(a.getUuid()))
425                                        .findFirst();
426                        Action assignmentsAction = assignmentsActionOptional.get();
427                        Map<EClass, List<Endeavor>> groupedAssignments = Util.groupBy(assignments, EObject::eClass);
428                        
429                        List<Endeavor> features = groupedAssignments.get(EngineeringPackage.Literals.FEATURE);
430                        if (features != null && !features.isEmpty()) {
431                                Table featuresTable = buildTable(
432                                                features, 
433                                                action, 
434                                                EngineeringPackage.Literals.ENGINEER__ASSIGNMENTS, 
435                                                context, 
436                                                progressMonitor,
437                                                createColumnBuilder("Feature"),
438                                                createColumnBuilder(EcorePackage.Literals.EOBJECT___ECONTAINER, "Product"),
439                                                createColumnBuilder(EngineeringPackage.Literals.ENGINEERED_CAPABILITY__RELEASES),
440                                                createColumnBuilder(NcorePackage.Literals.PERIOD__START),
441                                                createColumnBuilder(NcorePackage.Literals.PERIOD__END),
442                                                createColumnBuilder(NcorePackage.Literals.PERIOD__DURATION),
443                                                createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
444                                
445                                Action section = AppFactory.eINSTANCE.createAction();
446                                section.setText("Features");
447                                section.getContent().add(featuresTable);
448                        
449                                assignmentsAction.getSections().add(section);
450                        }
451                        
452                        List<Endeavor> increments = groupedAssignments.get(EngineeringPackage.Literals.INCREMENT);
453                        if (increments != null && !increments.isEmpty()) {
454                                Table incrementsTable = buildTable(
455                                                increments, 
456                                                action, 
457                                                EngineeringPackage.Literals.ENGINEER__ASSIGNMENTS, 
458                                                context, 
459                                                progressMonitor,
460                                                createColumnBuilder("Increment"),
461                                                createColumnBuilder(NcorePackage.Literals.PERIOD__START),
462                                                createColumnBuilder(NcorePackage.Literals.PERIOD__END),
463                                                createColumnBuilder(NcorePackage.Literals.PERIOD__DURATION),
464                                                createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
465                                
466                                Action section = AppFactory.eINSTANCE.createAction();
467                                section.setText("Increments");
468                                section.getContent().add(incrementsTable);
469                        
470                                assignmentsAction.getSections().add(section);
471                        }
472                        
473                        List<Endeavor> issues = groupedAssignments.get(EngineeringPackage.Literals.ISSUE);
474                        if (issues != null && !issues.isEmpty()) {
475                                Table issuesTable = buildTable(
476                                                issues, 
477                                                action, 
478                                                EngineeringPackage.Literals.ENGINEER__ASSIGNMENTS, 
479                                                context, 
480                                                progressMonitor,
481                                                createColumnBuilder("Issue"),
482                                                createColumnBuilder(EcorePackage.Literals.EOBJECT___ECONTAINER, "Module"),
483                                                createColumnBuilder(EngineeringPackage.Literals.ISSUE__INCREMENT),
484                                                createColumnBuilder(NcorePackage.Literals.PERIOD__START),
485                                                createColumnBuilder(NcorePackage.Literals.PERIOD__END),
486                                                createColumnBuilder(NcorePackage.Literals.PERIOD__DURATION),
487                                                createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
488                                
489                                Action section = AppFactory.eINSTANCE.createAction();
490                                section.setText("Issues");
491                                section.getContent().add(issuesTable);
492                        
493                                assignmentsAction.getSections().add(section);
494                        }
495                        
496                        List<Endeavor> releases = groupedAssignments.get(EngineeringPackage.Literals.RELEASE);
497                        if (releases != null && !releases.isEmpty()) {
498                                Table releasesTable = buildTable(
499                                                releases, 
500                                                action, 
501                                                EngineeringPackage.Literals.ENGINEER__ASSIGNMENTS, 
502                                                context, 
503                                                progressMonitor,
504                                                createColumnBuilder("Release"),
505                                                createColumnBuilder(EcorePackage.Literals.EOBJECT___ECONTAINER, "Product"),
506                                                createColumnBuilder(EngineeringPackage.Literals.RELEASE__INCREMENT),
507                                                createColumnBuilder(NcorePackage.Literals.PERIOD__START),
508                                                createColumnBuilder(NcorePackage.Literals.PERIOD__END),
509                                                createColumnBuilder(NcorePackage.Literals.PERIOD__DURATION),
510                                                createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
511                                
512                                Action section = AppFactory.eINSTANCE.createAction();
513                                section.setText("Releases");
514                                section.getContent().add(releasesTable);
515                        
516                                assignmentsAction.getSections().add(section);
517                        }                       
518                }               
519        }

521        private ContentProvider<EngineeredElementStatus> createEngineeredElementStatusChildrenProvider() {
522                return new ContentProvider<EngineeredElementStatus>() {
523
524                        @Override
525                        public List<EObject> createContent(
526                                        EngineeredElementStatus element, 
527                                        Action base, 
528                                        ETypedElement typedElement,
529                                        org.nasdanika.html.emf.EObjectActionResolver.Context context, 
530                                        ProgressMonitor progressMonitor) throws Exception {
531                                
532                                EList<EngineeredElementStatus> children = element.getChildren();
533                                if (children.isEmpty()) {
534                                        return null;
535                                }
536                                return Collections.singletonList(renderList(children, true, this, base, EngineeringPackage.Literals.ENGINEERED_ELEMENT_STATUS__CHILDREN, context, progressMonitor));
537                        }
538                        
539                };
540        }
