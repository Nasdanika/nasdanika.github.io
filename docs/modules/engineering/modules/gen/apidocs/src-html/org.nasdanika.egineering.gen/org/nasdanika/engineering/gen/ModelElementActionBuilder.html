040public class ModelElementActionBuilder<T extends ModelElement> extends EObjectActionBuilder<T> {
041        
042        /**
043         * Descriptions shorter than this value are put on the top of the tabs, longer
044         * ones end up in their own tab. 
045         */
046        protected int descriptionTabLengthThreshold = 2500;
047
048        public ModelElementActionBuilder(T value, Context context) {
049                super(value, context);          
050        }
051        
052        @Override
053        protected Action buildAction(
054                        Action action,
055                        BiConsumer<EObject,Action> registry, 
056                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
057                        ProgressMonitor progressMonitor) throws Exception {
058                Action ret = super.buildAction(action, registry, resolveConsumer, progressMonitor);             
059                T eObj = getTarget();
060                URI uri = NcoreUtil.getUri(eObj);
061                String id = uri == null ? eObj.getUuid() : uri.toString();
062                String digest = Hex.encodeHexString(MessageDigest.getInstance("SHA-256").digest(id.getBytes(StandardCharsets.UTF_8)));
063                ret.setId(digest);
064                
065                String description = eObj.getDescription();
066                ret.setDescription(description);
067
068                BiSupplier<EObject, String> cPath = NcoreUtil.containmentPath(eObj);
069                if (cPath == null || Util.isBlank(cPath.getSecond())) {
070                        ret.setLocation("${base-uri}index.html");
071                } else {
072                        ret.setLocation(cPath.getSecond() + "/index.html");
073                }
074                
075                ret.setSectionStyle(SectionStyle.HEADER);
076                
077                EList<Action> sections = ret.getSections();
078                for (Document section: eObj.getSections()) {
079                        sections.add(createChildAction(section, registry, resolveConsumer, progressMonitor));                   
080                }
081                
082                createResourcesAction(ret, registry, resolveConsumer, progressMonitor);                         
083                return ret;
084        }
085
086        protected void createResourcesAction(
087                        Action action,
088                        BiConsumer<EObject,Action> registry, 
089                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
090                        ProgressMonitor progressMonitor) throws Exception {
091                
092                EList<NamedElement> resources = getTarget().getResources();
093                if (!resources.isEmpty()) {
094                        Action group = AppFactory.eINSTANCE.createAction();
095                        group.setText("Resources");
096                        group.setUuid(action.getUuid() + "-resources");
097                        group.setLocation("resources.html");
098                        EList<Action> groupAnonymous = group.getAnonymous();
099                        for (NamedElement resource: resources) {
100                                groupAnonymous.add(createChildAction(resource, registry, resolveConsumer, progressMonitor));
101                        }
102                
103                        action.getNavigation().add(group);
104                }       
105        }

117        @Override
118        protected void resolve(Action action, org.nasdanika.html.emf.EObjectActionResolver.Context context,     ProgressMonitor progressMonitor) throws Exception {
119                super.resolve(action, context, progressMonitor);
120                
121                // Adding documentation here so it appears under the properties table
122                T semanticElement = getTarget();
123                EList<EObject> documentation = semanticElement.getDocumentation();
124                if (documentation.isEmpty()) {
125                        // Using description as documentation only if documentation is empty.
126                        String description = semanticElement.getDescription();
127                        addContent(action, description);
128                } else {
129                        action.getContent().addAll(EcoreUtil.copyAll(documentation)); // TODO - wrap into a group in order to inject uri's properties, optionally save into its own resource.
130                }               
131                
132                // Representations
133                for (Representation representation: semanticElement.getRepresentations()) {
134                        if (representation.getElements().isEmpty()) {
135                                populateRepresentation(representation, action, context, progressMonitor);
136                        }                       
137                        Action representationAction;
138                        if (Util.isBlank(representation.getName())) {
139                                representationAction = action;
140                        } else {
141                                representationAction = AppFactory.eINSTANCE.createAction();
142                                representationAction.setText(representation.getName());
143                                action.getSections().add(representationAction); // TODO - support of navigation/navigation-modal - get from properties.
144                        }
145                        String rDescr = representation.getDescription();
146                        if (Util.isBlank(rDescr)) {
147                                addContent(representationAction, createGenerator().generate(representation));                           
148                        } else {
149                                Table table = BootstrapFactory.eINSTANCE.createTable();
150                                action.getContent().add(table);
151                                table.setBordered(true);
152                                TableSection body = BootstrapFactory.eINSTANCE.createTableSection();
153                                table.setBody(body);
154                                table.getAttributes().put("style", createText("width:auto"));                           
155                                
156                                TableRow diagramRow = BootstrapFactory.eINSTANCE.createTableRow();
157                                body.getRows().add(diagramRow);
158                                TableCell diagramCell = BootstrapFactory.eINSTANCE.createTableCell();
159                                diagramRow.getCells().add(diagramCell);                         
160                                diagramCell.getContent().add(createText(createGenerator().generate(representation)));
161                                
162                                TableRow descriptionRow = BootstrapFactory.eINSTANCE.createTableRow();
163                                body.getRows().add(descriptionRow);
164                                TableCell descriptionCell = BootstrapFactory.eINSTANCE.createTableCell();
165                                descriptionRow.getCells().add(descriptionCell);                         
166                                descriptionCell.getContent().add(createText(rDescr));                                                           
167                        }
168                }
169                
170                EList<NamedElement> resources = semanticElement.getResources();
171                if (!resources.isEmpty()) {
172                        String resourcesGroupUUID = action.getUuid() + "-resources";
173                        Optional<Action> resourcesActionOptional = action.getNavigation().stream()
174                                        .filter(Action.class::isInstance)                                       
175                                        .map(Action.class::cast)
176                                        .filter(a -> resourcesGroupUUID.equals(a.getUuid()))
177                                        .findFirst();
178                        
179                        Action resourcesAction = resourcesActionOptional.get();
180                        resourcesAction.getContent().add(renderList(resources, true, createResourceChildrenProvider(), resourcesAction, EngineeringPackage.Literals.MODEL_ELEMENT__RESOURCES, context, progressMonitor)); // Table?
181                }
182                                
183        }
184        
185        private ContentProvider<NamedElement> createResourceChildrenProvider() {
186                return new ContentProvider<NamedElement>() {
187        
188                        @Override
189                        public List<EObject> createContent(
190                                        NamedElement element, 
191                                        Action base, 
192                                        ETypedElement typedElement,
193                                        org.nasdanika.html.emf.EObjectActionResolver.Context context, 
194                                        ProgressMonitor progressMonitor) throws Exception {
195        
196                                if (element instanceof Directory) {
197                                        EList<NamedElement> children = ((Directory) element).getElements();
198                                        if (!children.isEmpty()) {
199                                                return Collections.singletonList(renderList(children, true, this, base, EngineeringPackage.Literals.DIRECTORY__ELEMENTS, context, progressMonitor));
200                                        }
201                                }
202                                return null;
203                        }
204                        
205                };
206        }

208        /**
209         * Populates empty representations. An empty representation indicates that it has to be auto-populated. 
210         * Non-empty representations indicate that they were pre-populated, e.g. manually, and should not be auto-populated. 
211         * @param representation
212         */
213        protected void populateRepresentation(
214                        Representation representation, 
215                        Action action, 
216                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
217                        ProgressMonitor progressMonitor) throws Exception {
218                RepresentationGeneratorAdapter adapter = EObjectAdaptable.adaptTo(representation, RepresentationGeneratorAdapter.class);
219                if (adapter != null) {
220                        adapter.generate(action, context, progressMonitor);
221                }
222        }
223        
224        @Override
225        protected EObject renderValue(
226                        Action base, 
227                        ETypedElement typedElement, 
228                        Object value,
229                        org.nasdanika.html.emf.EObjectActionResolver.Context context, ProgressMonitor progressMonitor)
230                        throws Exception {
231
232                if ((typedElement == EngineeringPackage.Literals.ENDEAVOR__COMPLETION || typedElement == EngineeringPackage.Literals.KEY_RESULT__COMPLETION) && value instanceof Double) {
233                        double completion = (Double) value;
234                        if (completion != Double.NaN && completion > 0.001) {
235                                int percentage = (int) (100.0 * completion);
236                                org.nasdanika.html.model.html.Tag progressBar = HtmlFactory.eINSTANCE.createTag();
237                                progressBar.setName("div");
238                                progressBar.getAttributes().put("class", createText("progress-bar"));
239                                progressBar.getAttributes().put("style", createText("width:" + percentage + "%"));
240                                progressBar.getContent().add(createText(percentage + "%"));
241                                
242                                org.nasdanika.html.model.html.Tag progress = HtmlFactory.eINSTANCE.createTag();
243                                progress.setName("div");
244                                progress.getAttributes().put("class", createText("progress"));
245                                progress.getContent().add(progressBar);
246                                return progress;
247                        }
248                        return null;
249                }               
250                
251                return super.renderValue(base, typedElement, value, context, progressMonitor);
252        }
