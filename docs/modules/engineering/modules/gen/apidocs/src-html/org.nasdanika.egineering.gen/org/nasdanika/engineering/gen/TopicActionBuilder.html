001package org.nasdanika.engineering.gen;
002
003import org.eclipse.emf.ecore.EObject;
004import org.eclipse.emf.ecore.util.EcoreUtil;
005import org.nasdanika.common.Context;
006import org.nasdanika.common.ProgressMonitor;
007import org.nasdanika.common.Util;
008import org.nasdanika.engineering.Engineer;
009import org.nasdanika.engineering.EngineeringPackage;
010import org.nasdanika.engineering.Message;
011import org.nasdanika.engineering.Topic;
012import org.nasdanika.html.bootstrap.Size;
013import org.nasdanika.html.model.app.Action;
014import org.nasdanika.html.model.bootstrap.Appearance;
015import org.nasdanika.html.model.bootstrap.BootstrapElement;
016import org.nasdanika.html.model.bootstrap.BootstrapFactory;
017import org.nasdanika.html.model.bootstrap.Card;
018import org.nasdanika.html.model.bootstrap.Spacing;
019
020public class TopicActionBuilder extends MessageActionBuilder<Topic> {
021        
022        public TopicActionBuilder(Topic target, Context context) {
023                super(target, context);         
024        }
025                
026        @Override
027        protected void resolve(
028                        Action action, 
029                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
030                        ProgressMonitor progressMonitor) throws Exception {
031                super.resolve(action, context, progressMonitor);
032                
033                for (Message message: getTarget().getMessages()) {
034                        Card messageCard = BootstrapFactory.eINSTANCE.createCard();
035                        Appearance appearance = BootstrapFactory.eINSTANCE.createAppearance();
036                        messageCard.setAppearance(appearance);
037                        Spacing mTop = BootstrapFactory.eINSTANCE.createSpacing();
038                        mTop.setY(true);
039                        mTop.setSize(Size.S1);
040                        appearance.getMargin().add(mTop);
041                        
042                        if (!Util.isBlank(message.getName())) {
043                                BootstrapElement header = BootstrapFactory.eINSTANCE.createBootstrapElement();
044                                header.getContent().add(createText(message.getName()));
045                                messageCard.setHeader(header);
046                        }
047
048                        BootstrapElement body = BootstrapFactory.eINSTANCE.createBootstrapElement();
049                        messageCard.setBody(body);
050                        
051                        if (!Util.isBlank(message.getDescription())) {
052                                body.getContent().add(createText(message.getDescription()));
053                        }
054                        
055                        for (EObject documentation: message.getDocumentation()) {
056                                body.getContent().add(EcoreUtil.copy(documentation));
057                        }
058                        
059                        Engineer author = message.getAuthor();
060                        if (author != null || message.getDate() != null) {
061                                BootstrapElement footer = BootstrapFactory.eINSTANCE.createBootstrapElement();
062                                if (author != null) {
063                                        footer.getContent().add(renderValue(action, EngineeringPackage.Literals.MESSAGE__AUTHOR, author, context, progressMonitor));
064                                }
065                                if (message.getDate() != null) {
066                                        if (author != null) {
067                                                footer.getContent().add(createText(", "));
068                                        }
069                                        footer.getContent().add(renderValue(action, EngineeringPackage.Literals.MESSAGE__DATE, message.getDate(), context, progressMonitor));
070                                }
071                                messageCard.setFooter(footer);
072                        }
073                        
074                        
075                        action.getContent().add(messageCard);
076                }
077                
078        }
079        
080}
