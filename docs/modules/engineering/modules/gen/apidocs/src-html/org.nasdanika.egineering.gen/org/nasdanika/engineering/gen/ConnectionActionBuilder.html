001package org.nasdanika.engineering.gen;
002
003import java.util.List;
004import java.util.function.BiConsumer;
005import java.util.function.Consumer;
006
007import org.eclipse.emf.ecore.EObject;
008import org.eclipse.emf.ecore.ETypedElement;
009import org.nasdanika.common.Context;
010import org.nasdanika.common.ProgressMonitor;
011import org.nasdanika.common.Util;
012import org.nasdanika.engineering.Connection;
013import org.nasdanika.engineering.EngineeringPackage;
014import org.nasdanika.html.model.app.Action;
015
016public class ConnectionActionBuilder extends EngineeredElementActionBuilder<Connection> {
017        
018        public ConnectionActionBuilder(Connection target, Context context) {
019                super(target, context);         
020        }
021        
022        @Override
023        protected Action buildAction(
024                        Action action, 
025                        BiConsumer<EObject, Action> registry,
026                        Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer,
027                        ProgressMonitor progressMonitor) throws Exception {
028                Action ret = super.buildAction(action, registry, resolveConsumer, progressMonitor);
029                if (Util.isBlank(ret.getText())) {
030                        ret.setText("(" + getTarget().getPath() + ")");
031                }
032                return ret;
033        }
034        
035        @Override
036        protected List<ETypedElement> getProperties() {
037                List<ETypedElement> properties = super.getProperties();
038                properties.add(EngineeringPackage.Literals.CONNECTION__TARGET);
039                
040                return properties;
041        }
042        
043}
