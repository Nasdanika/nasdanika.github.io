001package org.nasdanika.engineering.gen;
002
003import java.util.List;
004
005import org.eclipse.emf.ecore.EcorePackage;
006import org.nasdanika.common.Context;
007import org.nasdanika.common.ProgressMonitor;
008import org.nasdanika.engineering.EngineeringPackage;
009import org.nasdanika.engineering.Issue;
010import org.nasdanika.engineering.IssuePriority;
011import org.nasdanika.html.model.app.Action;
012import org.nasdanika.html.model.app.AppFactory;
013import org.nasdanika.html.model.bootstrap.Table;
014import org.nasdanika.ncore.NcorePackage;
015
016public class IssuePriorityActionBuilder extends NamedElementActionBuilder<IssuePriority> {
017        
018        public IssuePriorityActionBuilder(IssuePriority target, Context context) {
019                super(target, context);         
020        }
021        
022        @Override
023        protected void resolve(
024                        Action action, 
025                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
026                        ProgressMonitor progressMonitor) throws Exception {
027                super.resolve(action, context, progressMonitor);
028                
029                List<Issue> issues = getTarget().getIssues();
030                if (!issues.isEmpty()) {
031                        Table issuesTable = buildTable(
032                                        issues, 
033                                        action, 
034                                        EngineeringPackage.Literals.ISSUE_CATEGORY__ISSUES, 
035                                        context, 
036                                        progressMonitor,
037                                        createColumnBuilder("Issue"),
038                                        createColumnBuilder(EcorePackage.Literals.EOBJECT___ECONTAINER, "Module"),
039                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__INCREMENT),
040                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__ASSIGNEE),
041                                        createColumnBuilder(NcorePackage.Literals.PERIOD__START),
042                                        createColumnBuilder(NcorePackage.Literals.PERIOD__END),
043                                        createColumnBuilder(NcorePackage.Literals.PERIOD__DURATION),
044                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
045                        
046                        Action section = AppFactory.eINSTANCE.createAction();
047                        section.setText("Issues");
048                        section.getContent().add(issuesTable);
049                
050                        action.getSections().add(section);
051                }                       
052        }
053        
054}
