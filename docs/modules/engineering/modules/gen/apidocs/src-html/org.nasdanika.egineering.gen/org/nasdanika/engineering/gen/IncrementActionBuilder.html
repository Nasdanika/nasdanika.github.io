023
024public class IncrementActionBuilder extends NamedElementActionBuilder<Increment> {
025        
026        public IncrementActionBuilder(Increment target, Context context) {
027                super(target, context);         
028        }
029        
030        @Override
031        protected List<ETypedElement> getProperties() {
032                List<ETypedElement> properties = super.getProperties();
033                
034                // Endeavor's
035                properties.add(EngineeringPackage.Literals.ENDEAVOR__ASSIGNEE);
036                properties.add(EngineeringPackage.Literals.ENDEAVOR__COMPLETION);
037                properties.add(EngineeringPackage.Literals.ENDEAVOR__TOTAL_COST);
038                properties.add(EngineeringPackage.Literals.ENDEAVOR__BENEFIT);
039                properties.add(EngineeringPackage.Literals.ENDEAVOR__AFFECTS);
040                
041                return properties;
042        }
043                
044        @Override
045        protected Action buildAction(
046                        Action action,
047                        BiConsumer<EObject,Action> registry, 
048                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
049                        ProgressMonitor progressMonitor) throws Exception {
050                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
051                action.setIcon("far fa-calendar-alt");
052                
053                createChildrenActions(action, registry, resolveConsumer, progressMonitor);
054                createAlignsActions(action, registry, resolveConsumer, progressMonitor);
055                createReleasesAction(action, registry, resolveConsumer, progressMonitor);               
056                createIssuesAction(action, registry, resolveConsumer, progressMonitor);         
057                createAllIssuesAction(action, registry, resolveConsumer, progressMonitor);              
058                createAllObjectiveActions(action, registry, resolveConsumer, progressMonitor);
059                
060                return action;
061        }
062        
063        protected void createAllObjectiveActions(
064                        Action action,
065                        BiConsumer<EObject,Action> registry, 
066                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
067                        ProgressMonitor progressMonitor) throws Exception {
068                
069                if (!getTarget().getAllObjectives().isEmpty()) {
070                        Action group = AppFactory.eINSTANCE.createAction();
071                        group.setText("Objectives");
072                        group.setUuid(action.getUuid() + "-all-objectives");
073                        // TODO - icon, ...
074                        EList<Action> groupAnonymous = group.getAnonymous();
075                        for (Objective objective: getTarget().getObjectives()) {
076                                groupAnonymous.add(createChildAction(objective, registry, resolveConsumer, progressMonitor));
077                        }
078                
079                        action.getSections().add(group);
080                }
081        }
082        
083        protected void createAlignsActions(
084                        Action action, 
085                        BiConsumer<EObject,Action> registry, 
086                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
087                        ProgressMonitor progressMonitor) throws Exception {
088        
089                List<Alignment> alignments = getTarget().getAligns();
090                if (!alignments.isEmpty()) {
091                        Action group = AppFactory.eINSTANCE.createAction();
092                        group.setText("Aligns");

094                        action.getSections().add(group);
095                }
096                
097        }
098

114        protected void createIssuesAction(
115                        Action action,
116                        BiConsumer<EObject,Action> registry, 
117                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer,

127        }
128        
129        protected void createAllIssuesAction(
130                        Action action,
131                        BiConsumer<EObject,Action> registry, 
132                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
133                        ProgressMonitor progressMonitor) throws Exception {
134                
135                List<Issue> allIssues = getTarget().getAllIssues();
136                if (!allIssues.isEmpty()) {
137                        Action group = AppFactory.eINSTANCE.createAction();
138                        group.setText("All Issues");

145        
146        @Override
147        protected void resolve(
148                        Action action, 
149                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
150                        ProgressMonitor progressMonitor) throws Exception {
151                super.resolve(action, context, progressMonitor);
152                
153                // Endeavor's - aligns, all objectives (objectives), all issues, capacity (TODO)
154                EList<Alignment> alignments = getTarget().getAligns();
155                if (!alignments.isEmpty()) {
156                        String alignsGroupUUID = action.getUuid() + "-aligns";
157                        Optional<Action> alignmentsActionOptional = action.getSections().stream()
158                                        .filter(a -> alignsGroupUUID.equals(a.getUuid()))
159                                        .findFirst();
160                        
161                        Table alignmentsTable = buildTable(
162                                        alignments, 
163                                        action, 
164                                        EngineeringPackage.Literals.ALIGNABLE__ALIGNS, 
165                                        context, 
166                                        progressMonitor, 
167                                        createColumnBuilder(EngineeringPackage.Literals.ALIGNMENT__AIM),
168                                        createColumnBuilder(NcorePackage.Literals.MODEL_ELEMENT__DESCRIPTION),
169                                        createColumnBuilder(EngineeringPackage.Literals.ALIGNMENT__WEIGHT));
170                        alignmentsActionOptional.get().getContent().add(alignmentsTable);
171                }
172                
173                EList<Objective> allObjectives = getTarget().getAllObjectives();
174                if (!allObjectives.isEmpty()) {
175                        String allObjectivesGroupUUID = action.getUuid() + "-all-objectives";
176                        Optional<Action> allObjectivesActionOptional = action.getSections().stream()
177                                        .filter(a -> allObjectivesGroupUUID.equals(a.getUuid()))
178                                        .findFirst();
179                        
180                        allObjectivesActionOptional.get().getContent().add(renderList(allObjectives, true, null, action, EngineeringPackage.Literals.ENDEAVOR__ALL_OBJECTIVES, context, progressMonitor)); // Table?
181                }
182                                
183                // Releases
184                EList<Release> releases = getTarget().getReleases();
185                if (!releases.isEmpty()) {
186                        String releasesGroupUUID = action.getUuid() + "-releases";
187                        Optional<Action> releasesActionOptional = action.getSections().stream()
188                                        .filter(Action.class::isInstance)                                       
189                                        .map(Action.class::cast)
190                                        .filter(a -> releasesGroupUUID.equals(a.getUuid()))
191                                        .findFirst();
192                        Action releasesAction = releasesActionOptional.get();
193                        Table allIssuesTable = buildTable(
194                                        releases, 
195                                        action, 
196                                        EngineeringPackage.Literals.INCREMENT__RELEASES, 
197                                        context, 
198                                        progressMonitor,
199                                        createColumnBuilder(EcorePackage.Literals.EOBJECT___ECONTAINER, "Product"),
200                                        createColumnBuilder("Release"),
201                                        createColumnBuilder(NcorePackage.Literals.PERIOD__START),                                       
202                                        createColumnBuilder(NcorePackage.Literals.PERIOD__END),                                 
203                                        createColumnBuilder(NcorePackage.Literals.PERIOD__DURATION),                                    
204                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
205                        
206                        releasesAction.getContent().add(allIssuesTable);
207                }
208                
209                // Issues
210                EList<Issue> issues = getTarget().getIssues();
211                if (!issues.isEmpty()) {
212                        String issuesGroupUUID = action.getUuid() + "-issues";
213                        Optional<Action> issuesActionOptional = action.getSections().stream()
214                                        .filter(Action.class::isInstance)                                       
215                                        .map(Action.class::cast)
216                                        .filter(a -> issuesGroupUUID.equals(a.getUuid()))
217                                        .findFirst();
218                        Action issuesAction = issuesActionOptional.get();
219                        Table issuesTable = buildTable(
220                                        issues, 
221                                        action, 
222                                        EngineeringPackage.Literals.INCREMENT__ISSUES, 
223                                        context, 
224                                        progressMonitor,
225                                        createColumnBuilder("Issue"),
226                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__TARGET),
227                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__ASSIGNEE),
228                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__CATEGORIES),
229                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__PRIORITY),
230                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__SEVERITY),
231                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__STATUS),
232                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__WORKABLE),
233                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__BENEFIT),
234                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__COST),
235                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__EFFORT),
236                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__REMAINING_COST),
237                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__REMAINING_EFFORT),
238                                        createColumnBuilder(EngineeringPackage.Literals.ENGINEERED_CAPABILITY__RELEASES),
239                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__CONTRIBUTES_TO),                                 
240                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
241                        
242                        issuesAction.getContent().add(issuesTable);
243                }
244                
245                // All Issues
246                EList<Issue> allIssues = getTarget().getAllIssues();
247                if (!allIssues.isEmpty()) {
248                        String allIssuesGroupUUID = action.getUuid() + "-all-issues";
249                        Optional<Action> issuesActionOptional = action.getNavigation().stream()
250                                        .filter(Action.class::isInstance)                                       
251                                        .map(Action.class::cast)
252                                        .filter(a -> allIssuesGroupUUID.equals(a.getUuid()))
253                                        .findFirst();
254                        Action allIssuesAction = issuesActionOptional.get();
255                        Table allIssuesTable = buildTable(
256                                        allIssues, 
257                                        action, 
258                                        EngineeringPackage.Literals.ENGINEERED_ELEMENT__ALL_ISSUES, 
259                                        context, 
260                                        progressMonitor,
261                                        createColumnBuilder("Issue"),
262                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__TARGET),
263                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__ASSIGNEE),
264                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__CATEGORIES),
265                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__PRIORITY),
266                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__SEVERITY),
267                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__STATUS),
268                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__WORKABLE),
269                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__BENEFIT),
270                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__COST),
271                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__EFFORT),
272                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__REMAINING_COST),
273                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__REMAINING_EFFORT),
274                                        createColumnBuilder(EngineeringPackage.Literals.ENGINEERED_CAPABILITY__RELEASES),
275                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__CONTRIBUTES_TO),                                 
276                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
277                        
278                        allIssuesAction.getContent().add(allIssuesTable);
279                }
280                
281        }       
282
283        /**
284         * Creates a list of actions for sub-packages. 
285         * @param progressMonitor
286         * @return An empty list if there are no sub-packages. A singleton list containing a grouping action containing sub-package actions otherwise.
287         * @throws Exception 
288         */
289        protected void createChildrenActions(
290                        Action action,
291                        BiConsumer<EObject,Action> registry, 
292                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
293                        ProgressMonitor progressMonitor) throws Exception {
294                List<Increment> children = getTarget().getChildren();
295                if (!children.isEmpty()) {
296                        EList<EObject> actionChildren = action.getChildren();
297                        for (Increment child: children) {
298                                actionChildren.add(createChildAction(child, registry, resolveConsumer, progressMonitor));
299                        }
300                }               
301        }
