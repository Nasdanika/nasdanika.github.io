001package org.nasdanika.engineering.gen;
002
003import java.util.Collections;
004import java.util.List;
005import java.util.function.BiConsumer;
006
007import org.eclipse.emf.common.util.EList;
008import org.eclipse.emf.ecore.EObject;
009import org.eclipse.emf.ecore.ETypedElement;
010import org.nasdanika.common.Context;
011import org.nasdanika.common.ProgressMonitor;
012import org.nasdanika.engineering.EngineeredElement;
013import org.nasdanika.engineering.EngineeredElementStatus;
014import org.nasdanika.engineering.EngineeringPackage;
015import org.nasdanika.html.model.app.Action;
016import org.nasdanika.html.model.app.AppFactory;
017import org.nasdanika.html.model.html.Tag;
018
019public class EngineeredElementStatusActionBuilder extends NamedElementActionBuilder<EngineeredElementStatus> {
020        
021        public EngineeredElementStatusActionBuilder(EngineeredElementStatus target, Context context) {
022                super(target, context);         
023        }
024        
025        @Override
026        protected Action buildAction(
027                        Action action,
028                        BiConsumer<EObject,Action> registry, 
029                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
030                        ProgressMonitor progressMonitor) throws Exception {
031                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
032                
033                action.getSections().addAll(createChildrenActions(registry, resolveConsumer, progressMonitor));
034                
035                return action;
036        }
037        
038        protected List<Action> createChildrenActions(
039                        BiConsumer<EObject,Action> registry, 
040                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
041                        ProgressMonitor progressMonitor) throws Exception {
042                
043                List<EngineeredElementStatus> children = getTarget().getChildren();
044                if (children.isEmpty()) {
045                        return Collections.emptyList();
046                }
047                Action group = AppFactory.eINSTANCE.createAction();
048                group.setText("Sub-statuses");
049                EList<Action> groupAnonymous = group.getAnonymous();
050                for (EngineeredElementStatus subStatus: children) {
051                        groupAnonymous.add(createChildAction(subStatus, registry, resolveConsumer, progressMonitor));
052                }
053                
054                return Collections.singletonList(group);
055        }
056        
057        @Override
058        protected void resolve(
059                        Action action, 
060                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
061                        ProgressMonitor progressMonitor) throws Exception {
062                super.resolve(action, context, progressMonitor);
063                
064                EList<EngineeredElementStatus> children = getTarget().getChildren();
065                if (!children.isEmpty()) {
066                        Action childrenAction = (Action) action.getSections().get(0);
067                        ContentProvider<EngineeredElementStatus> engineeredElementStatusChildrenListProvider = new ContentProvider<EngineeredElementStatus>() {
068
069                                @Override
070                                public List<EObject> createContent(
071                                                EngineeredElementStatus element, 
072                                                Action base, 
073                                                ETypedElement typedElement,
074                                                org.nasdanika.html.emf.EObjectActionResolver.Context context, 
075                                                ProgressMonitor progressMonitor) throws Exception {
076                                        
077                                        EList<EngineeredElementStatus> children = element.getChildren();
078                                        if (children.isEmpty()) {
079                                                return null;
080                                        }
081                                        return Collections.singletonList(renderList(children, true, this, base, EngineeringPackage.Literals.ENGINEERED_ELEMENT_STATUS__CHILDREN, context, progressMonitor));
082                                }
083                                
084                        };
085                        
086                        childrenAction.getContent().add(renderList(children, true, engineeredElementStatusChildrenListProvider, action, EngineeringPackage.Literals.ENGINEER__ENGINEERED_ELEMENT_STATUSES, context, progressMonitor));
087                }
088                
089                List<EngineeredElement> elements = getTarget().getElements();
090                if (!elements.isEmpty()) {
091                        Tag elementsList = renderList(elements, true, null, action, EngineeringPackage.Literals.ENGINEERED_ELEMENT_STATUS__ELEMENTS, context, progressMonitor);
092                        Action section = AppFactory.eINSTANCE.createAction();
093                        section.setText("Elements");
094                        section.getContent().add(elementsList);
095                
096                        action.getSections().add(section);
097                }                       
098        }
099        
100}
