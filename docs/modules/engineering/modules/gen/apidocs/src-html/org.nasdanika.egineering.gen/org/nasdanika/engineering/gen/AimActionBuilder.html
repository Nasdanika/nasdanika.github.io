001package org.nasdanika.engineering.gen;
002
003import java.util.List;
004import java.util.Optional;
005import java.util.function.BiConsumer;
006
007import org.eclipse.emf.common.util.EList;
008import org.eclipse.emf.ecore.EObject;
009import org.eclipse.emf.ecore.EcorePackage;
010import org.nasdanika.common.Context;
011import org.nasdanika.common.ProgressMonitor;
012import org.nasdanika.engineering.Aim;
013import org.nasdanika.engineering.Alignment;
014import org.nasdanika.engineering.EngineeringPackage;
015import org.nasdanika.html.model.app.Action;
016import org.nasdanika.html.model.app.AppFactory;
017import org.nasdanika.html.model.bootstrap.Table;
018import org.nasdanika.ncore.NcorePackage;
019
020public class AimActionBuilder<T extends Aim> extends NamedElementActionBuilder<T> {
021        
022        public AimActionBuilder(T target, Context context) {
023                super(target, context);         
024        }
025                
026        @Override
027        protected Action buildAction(
028                        Action action,
029                        BiConsumer<EObject,Action> registry, 
030                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
031                        ProgressMonitor progressMonitor) throws Exception {
032                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
033                
034                createAlignsActions(action, registry, resolveConsumer, progressMonitor);
035                createAlignmentsActions(action, registry, resolveConsumer, progressMonitor);
036                
037                return action;
038        }
039        
040        protected void createAlignmentsActions(
041                        Action action, 
042                        BiConsumer<EObject,Action> registry, 
043                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
044                        ProgressMonitor progressMonitor) throws Exception {
045                
046                List<Alignment> alignments = getTarget().getAlignments();
047                if (!alignments.isEmpty()) {
048                        Action group = AppFactory.eINSTANCE.createAction();
049                        group.setText("Alignments");
050                        group.setUuid(action.getUuid() + "-alignments");
051                        action.getSections().add(group);
052                }
053        }
054        
055        protected void createAlignsActions(
056                        Action action, 
057                        BiConsumer<EObject,Action> registry, 
058                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
059                        ProgressMonitor progressMonitor) throws Exception {
060                
061                List<Alignment> alignments = getTarget().getAligns();
062                if (!alignments.isEmpty()) {
063                        Action group = AppFactory.eINSTANCE.createAction();
064                        group.setText("Aligns");
065                        group.setUuid(action.getUuid() + "-aligns");
066                        action.getSections().add(group);
067                }
068        }
069        
070        @Override
071        protected void resolve(
072                        Action action, 
073                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
074                        ProgressMonitor progressMonitor) throws Exception {
075                super.resolve(action, context, progressMonitor);
076                
077                EList<Alignment> aligns = getTarget().getAligns();
078                if (!aligns.isEmpty()) {
079                        String alignsGroupUUID = action.getUuid() + "-aligns";
080                        Optional<Action> alignmentsActionOptional = action.getSections().stream()
081                                        .filter(a -> alignsGroupUUID.equals(a.getUuid()))
082                                        .findFirst();
083                        
084                        Table alignmentsTable = buildTable(
085                                        aligns, 
086                                        action, 
087                                        EngineeringPackage.Literals.ALIGNABLE__ALIGNS, 
088                                        context, 
089                                        progressMonitor, 
090                                        createColumnBuilder(EngineeringPackage.Literals.ALIGNMENT__AIM),
091                                        createColumnBuilder(NcorePackage.Literals.MODEL_ELEMENT__DESCRIPTION),
092                                        createColumnBuilder(EngineeringPackage.Literals.ALIGNMENT__WEIGHT));
093                        alignmentsActionOptional.get().getContent().add(alignmentsTable);
094                }
095                
096                EList<Alignment> alignments = getTarget().getAlignments();
097                if (!alignments.isEmpty()) {
098                        String alignmentsGroupUUID = action.getUuid() + "-alignments";
099                        Optional<Action> alignmentsActionOptional = action.getSections().stream()
100                                        .filter(a -> alignmentsGroupUUID.equals(a.getUuid()))
101                                        .findFirst();
102                        
103                        Table alignmentsTable = buildTable(
104                                        alignments, 
105                                        action, 
106                                        EngineeringPackage.Literals.AIM__ALIGNMENTS, 
107                                        context, 
108                                        progressMonitor, 
109                                        createColumnBuilder(EcorePackage.Literals.EOBJECT___ECONTAINER, "Element"),
110                                        createColumnBuilder(NcorePackage.Literals.MODEL_ELEMENT__DESCRIPTION),
111                                        createColumnBuilder(EngineeringPackage.Literals.ALIGNMENT__WEIGHT));
112                        alignmentsActionOptional.get().getContent().add(alignmentsTable);
113                }
114        }
115                
116}
