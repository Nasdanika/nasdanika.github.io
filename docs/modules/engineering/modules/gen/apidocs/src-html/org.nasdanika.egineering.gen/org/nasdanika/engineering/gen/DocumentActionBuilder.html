001package org.nasdanika.engineering.gen;
002
003import org.nasdanika.common.Context;
004import org.nasdanika.common.ProgressMonitor;
005import org.nasdanika.engineering.Document;
006import org.nasdanika.html.model.app.Action;
007import org.nasdanika.html.model.app.AppFactory;
008import org.nasdanika.html.model.bootstrap.Table;
009
010public class DocumentActionBuilder extends EngineeredElementActionBuilder<Document> {
011        
012        public DocumentActionBuilder(Document target, Context context) {
013                super(target, context);         
014        }
015                
016        @Override
017        protected Table createPropertiesTable(
018                        Action action, 
019                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
020                        ProgressMonitor progressMonitor) throws Exception {             
021                Table propertiesTable = super.createPropertiesTable(action, context, progressMonitor);
022                if (propertiesTable != null) {
023                        Action propertiesAction = AppFactory.eINSTANCE.createAction();
024                        propertiesAction.setText("Properties");
025                        propertiesAction.setLocation(action.getUuid()+"-properties.html");
026                        propertiesAction.getContent().add(propertiesTable);
027                        action.getNavigation().add(propertiesAction);
028                }
029                return null;
030        }
031        
032}
