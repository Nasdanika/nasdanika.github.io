001package org.nasdanika.engineering.gen;
002
003import java.util.List;
004import java.util.Optional;
005import java.util.function.BiConsumer;
006
007import org.eclipse.emf.common.util.EList;
008import org.eclipse.emf.ecore.EObject;
009import org.eclipse.emf.ecore.ETypedElement;
010import org.nasdanika.common.Context;
011import org.nasdanika.common.ProgressMonitor;
012import org.nasdanika.engineering.Alignment;
013import org.nasdanika.engineering.EngineeredCapability;
014import org.nasdanika.engineering.EngineeringPackage;
015import org.nasdanika.engineering.Issue;
016import org.nasdanika.engineering.Objective;
017import org.nasdanika.html.model.app.Action;
018import org.nasdanika.html.model.app.AppFactory;
019import org.nasdanika.html.model.bootstrap.Table;
020import org.nasdanika.ncore.NcorePackage;
021
022public class EngineeredCapabilityActionBuilder<T extends EngineeredCapability> extends CapabilityActionBuilder<T> {
023        
024        public EngineeredCapabilityActionBuilder(T target, Context context) {
025                super(target, context);         
026        }
027        
028        @Override
029        protected List<ETypedElement> getProperties() {
030                List<ETypedElement> properties = super.getProperties();
031                
032                // Endeavor's
033                properties.add(EngineeringPackage.Literals.ENDEAVOR__ASSIGNEE);
034                properties.add(EngineeringPackage.Literals.ENDEAVOR__COMPLETION);
035                properties.add(EngineeringPackage.Literals.ENDEAVOR__TOTAL_COST);
036                properties.add(EngineeringPackage.Literals.ENDEAVOR__BENEFIT);
037                properties.add(EngineeringPackage.Literals.ENDEAVOR__AFFECTS);
038
039                // Engineered Capability
040                properties.add(EngineeringPackage.Literals.ENGINEERED_CAPABILITY__RELEASES);
041                
042                return properties;
043        }
044                
045        @Override
046        protected Action buildAction(
047                        Action action,
048                        BiConsumer<EObject,Action> registry, 
049                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
050                        ProgressMonitor progressMonitor) throws Exception {
051                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
052                
053                createAlignsActions(action, registry, resolveConsumer, progressMonitor);
054                createAllIssuesAction(action, registry, resolveConsumer, progressMonitor);              
055                createAllObjectiveActions(action, registry, resolveConsumer, progressMonitor);
056                
057                return action;
058        }
059        
060        protected void createAllObjectiveActions(
061                        Action action,
062                        BiConsumer<EObject,Action> registry, 
063                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
064                        ProgressMonitor progressMonitor) throws Exception {
065                
066                if (!getTarget().getAllObjectives().isEmpty()) {
067                        Action group = AppFactory.eINSTANCE.createAction();
068                        group.setText("Objectives");
069                        group.setUuid(action.getUuid() + "-all-objectives");
070                        // TODO - icon, ...
071                        EList<Action> groupAnonymous = group.getAnonymous();
072                        for (Objective objective: getTarget().getObjectives()) {
073                                groupAnonymous.add(createChildAction(objective, registry, resolveConsumer, progressMonitor));
074                        }
075                
076                        action.getSections().add(group);
077                }
078        }
079        
080        protected void createAlignsActions(
081                        Action action, 
082                        BiConsumer<EObject,Action> registry, 
083                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
084                        ProgressMonitor progressMonitor) throws Exception {
085        
086                List<Alignment> alignments = getTarget().getAligns();
087                if (!alignments.isEmpty()) {
088                        Action group = AppFactory.eINSTANCE.createAction();
089                        group.setText("Aligns");
090                        group.setUuid(action.getUuid() + "-aligns");
091                        action.getSections().add(group);
092                }               
093                
094        }
095        
096        protected void createAllIssuesAction(
097                        Action action,
098                        BiConsumer<EObject,Action> registry, 
099                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
100                        ProgressMonitor progressMonitor) throws Exception {
101                
102                List<Issue> allIssues = getTarget().getAllIssues();
103                if (!allIssues.isEmpty()) {
104                        Action group = AppFactory.eINSTANCE.createAction();
105                        group.setText("All Issues");
106                        group.setUuid(action.getUuid() + "-all-issues");
107                        group.setLocation("all-issues.html");
108                        action.getNavigation().add(group);
109                }
110                
111        }
112        
113        @Override
114        protected void resolve(
115                        Action action, 
116                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
117                        ProgressMonitor progressMonitor) throws Exception {
118                super.resolve(action, context, progressMonitor);
119                
120                // Endeavor's - aligns, all objectives (objectives), all issues, capacity (TODO)
121                EList<Alignment> alignments = getTarget().getAligns();
122                if (!alignments.isEmpty()) {
123                        String alignsGroupUUID = action.getUuid() + "-aligns";
124                        Optional<Action> alignmentsActionOptional = action.getSections().stream()
125                                        .filter(a -> alignsGroupUUID.equals(a.getUuid()))
126                                        .findFirst();
127                        
128                        Table alignmentsTable = buildTable(
129                                        alignments, 
130                                        action, 
131                                        EngineeringPackage.Literals.ALIGNABLE__ALIGNS, 
132                                        context, 
133                                        progressMonitor, 
134                                        createColumnBuilder(EngineeringPackage.Literals.ALIGNMENT__AIM),
135                                        createColumnBuilder(NcorePackage.Literals.MODEL_ELEMENT__DESCRIPTION),
136                                        createColumnBuilder(EngineeringPackage.Literals.ALIGNMENT__WEIGHT));
137                        alignmentsActionOptional.get().getContent().add(alignmentsTable);
138                }
139                
140                EList<Objective> allObjectives = getTarget().getAllObjectives();
141                if (!allObjectives.isEmpty()) {
142                        String allObjectivesGroupUUID = action.getUuid() + "-all-objectives";
143                        Optional<Action> allObjectivesActionOptional = action.getSections().stream()
144                                        .filter(a -> allObjectivesGroupUUID.equals(a.getUuid()))
145                                        .findFirst();
146                        
147                        allObjectivesActionOptional.get().getContent().add(renderList(allObjectives, true, null, action, EngineeringPackage.Literals.ENDEAVOR__ALL_OBJECTIVES, context, progressMonitor)); // Table?
148                }
149                
150                // All Issues
151                EList<Issue> allIssues = getTarget().getAllIssues();
152                if (!allIssues.isEmpty()) {
153                        String allIssuesGroupUUID = action.getUuid() + "-all-issues";
154                        Optional<Action> issuesActionOptional = action.getNavigation().stream()
155                                        .filter(Action.class::isInstance)                                       
156                                        .map(Action.class::cast)
157                                        .filter(a -> allIssuesGroupUUID.equals(a.getUuid()))
158                                        .findFirst();
159                        Action allIssuesAction = issuesActionOptional.get();
160                        Table allIssuesTable = buildTable(
161                                        allIssues, 
162                                        action, 
163                                        EngineeringPackage.Literals.ENGINEERED_ELEMENT__ALL_ISSUES, 
164                                        context, 
165                                        progressMonitor,
166                                        createColumnBuilder("Issue"),
167                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__TARGET),
168                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__ASSIGNEE),
169                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__CATEGORIES),
170                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__PRIORITY),
171                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__SEVERITY),
172                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__STATUS),
173                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__WORKABLE),
174                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__BENEFIT),
175                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__COST),
176                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__EFFORT),
177                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__REMAINING_COST),
178                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__REMAINING_EFFORT),
179                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
180                        
181                        allIssuesAction.getContent().add(allIssuesTable);
182                }
183                
184        }       
185
