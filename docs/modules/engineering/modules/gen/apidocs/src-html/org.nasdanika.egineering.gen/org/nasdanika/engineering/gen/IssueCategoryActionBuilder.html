001package org.nasdanika.engineering.gen;
002
003import java.util.Collections;
004import java.util.List;
005import java.util.function.BiConsumer;
006
007import org.eclipse.emf.common.util.EList;
008import org.eclipse.emf.ecore.EObject;
009import org.eclipse.emf.ecore.ETypedElement;
010import org.eclipse.emf.ecore.EcorePackage;
011import org.nasdanika.common.Context;
012import org.nasdanika.common.ProgressMonitor;
013import org.nasdanika.engineering.EngineeringPackage;
014import org.nasdanika.engineering.Issue;
015import org.nasdanika.engineering.IssueCategory;
016import org.nasdanika.html.model.app.Action;
017import org.nasdanika.html.model.app.AppFactory;
018import org.nasdanika.html.model.bootstrap.Table;
019import org.nasdanika.ncore.NcorePackage;
020
021public class IssueCategoryActionBuilder extends AimActionBuilder<IssueCategory> {
022        
023        public IssueCategoryActionBuilder(IssueCategory target, Context context) {
024                super(target, context);         
025        }
026        
027        @Override
028        protected List<ETypedElement> getProperties() {
029                List<ETypedElement> properties = super.getProperties();
030                properties.add(EngineeringPackage.Literals.ISSUE_CATEGORY__WEIGHT);
031                return properties;
032        }
033        
034        @Override
035        protected Action buildAction(
036                        Action action,
037                        BiConsumer<EObject,Action> registry, 
038                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
039                        ProgressMonitor progressMonitor) throws Exception {
040                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
041                
042                action.getSections().addAll(createChildrenActions(registry, resolveConsumer, progressMonitor));
043                
044                return action;
045        }
046        
047        protected List<Action> createChildrenActions(
048                        BiConsumer<EObject,Action> registry, 
049                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
050                        ProgressMonitor progressMonitor) throws Exception {
051                
052                List<IssueCategory> children = getTarget().getChildren();
053                if (children.isEmpty()) {
054                        return Collections.emptyList();
055                }
056                Action group = AppFactory.eINSTANCE.createAction();
057                group.setText("Sub-categories");
058                EList<Action> groupAnonymous = group.getAnonymous();
059                for (IssueCategory subCategory: children) {
060                        groupAnonymous.add(createChildAction(subCategory, registry, resolveConsumer, progressMonitor));
061                }
062                
063                return Collections.singletonList(group);
064        }
065        
066        @Override
067        protected void resolve(
068                        Action action, 
069                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
070                        ProgressMonitor progressMonitor) throws Exception {
071                super.resolve(action, context, progressMonitor);
072                
073                EList<IssueCategory> children = getTarget().getChildren();
074                if (!children.isEmpty()) {
075                        Action childrenAction = (Action) action.getSections().get(0);
076                        ContentProvider<IssueCategory> issueCategoryChildrenListProvider = new ContentProvider<IssueCategory>() {
077
078                                @Override
079                                public List<EObject> createContent(
080                                                IssueCategory element, 
081                                                Action base, 
082                                                ETypedElement typedElement,
083                                                org.nasdanika.html.emf.EObjectActionResolver.Context context, 
084                                                ProgressMonitor progressMonitor) throws Exception {
085                                        
086                                        EList<IssueCategory> children = element.getChildren();
087                                        if (children.isEmpty()) {
088                                                return null;
089                                        }
090                                        return Collections.singletonList(renderList(children, true, this, base, EngineeringPackage.Literals.ISSUE_CATEGORY__CHILDREN, context, progressMonitor));
091                                }
092                                
093                        };
094                        
095                        childrenAction.getContent().add(renderList(children, true, issueCategoryChildrenListProvider, action, EngineeringPackage.Literals.ENGINEER__ISSUE_CATEGORIES, context, progressMonitor));
096                }
097                
098                List<Issue> issues = getTarget().getIssues();
099                if (!issues.isEmpty()) {
100                        Table issuesTable = buildTable(
101                                        issues, 
102                                        action, 
103                                        EngineeringPackage.Literals.ISSUE_CATEGORY__ISSUES, 
104                                        context, 
105                                        progressMonitor,
106                                        createColumnBuilder("Issue"),
107                                        createColumnBuilder(EcorePackage.Literals.EOBJECT___ECONTAINER, "Module"),
108                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__INCREMENT),
109                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__ASSIGNEE),
110                                        createColumnBuilder(NcorePackage.Literals.PERIOD__START),
111                                        createColumnBuilder(NcorePackage.Literals.PERIOD__END),
112                                        createColumnBuilder(NcorePackage.Literals.PERIOD__DURATION),
113                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
114                        
115                        Action section = AppFactory.eINSTANCE.createAction();
116                        section.setText("Issues");
117                        section.getContent().add(issuesTable);
118                
119                        action.getSections().add(section);
120                }                       
121        }
122        
123}
