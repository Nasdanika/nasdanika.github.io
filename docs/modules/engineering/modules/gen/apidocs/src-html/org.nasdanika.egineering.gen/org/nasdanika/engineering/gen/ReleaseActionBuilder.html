001package org.nasdanika.engineering.gen;
002
003import java.util.List;
004import java.util.Map;
005
006import org.eclipse.emf.common.util.EList;
007import org.eclipse.emf.ecore.EClass;
008import org.eclipse.emf.ecore.EObject;
009import org.eclipse.emf.ecore.ETypedElement;
010import org.eclipse.emf.ecore.EcorePackage;
011import org.nasdanika.common.Context;
012import org.nasdanika.common.ProgressMonitor;
013import org.nasdanika.common.Util;
014import org.nasdanika.engineering.EngineeredCapability;
015import org.nasdanika.engineering.EngineeringPackage;
016import org.nasdanika.engineering.Release;
017import org.nasdanika.html.model.app.Action;
018import org.nasdanika.html.model.app.AppFactory;
019import org.nasdanika.html.model.bootstrap.Table;
020import org.nasdanika.ncore.NcorePackage;
021
022public class ReleaseActionBuilder extends EngineeredCapabilityActionBuilder<Release> {
023        
024        public ReleaseActionBuilder(Release target, Context context) {
025                super(target, context);         
026        }
027        
028        @Override
029        protected List<ETypedElement> getProperties() {
030                List<ETypedElement> properties = super.getProperties();
031                properties.add(EngineeringPackage.Literals.RELEASE__INCREMENT);
032                return properties;
033        }
034        
035        @Override
036        protected void resolve(
037                        Action action, 
038                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
039                        ProgressMonitor progressMonitor) throws Exception {
040                super.resolve(action, context, progressMonitor);
041                
042                // Capabilities         
043                EList<EngineeredCapability> capabilities = getTarget().getCapabilities();
044                if (!capabilities.isEmpty()) {
045                        Map<EClass, List<EngineeredCapability>> groupedCapabilities = Util.groupBy(capabilities, EObject::eClass);
046                        
047                        List<EngineeredCapability> releases = groupedCapabilities.get(EngineeringPackage.Literals.RELEASE);
048                        if (releases != null && !releases.isEmpty()) {
049                                Table releasesTable = buildTable(
050                                                releases, 
051                                                action, 
052                                                EngineeringPackage.Literals.ENGINEER__ASSIGNMENTS, 
053                                                context, 
054                                                progressMonitor,
055                                                createColumnBuilder("Release"),
056                                                createColumnBuilder(EcorePackage.Literals.EOBJECT___ECONTAINER, "Product"),
057                                                createColumnBuilder(EngineeringPackage.Literals.RELEASE__INCREMENT),
058                                                createColumnBuilder(NcorePackage.Literals.PERIOD__START),
059                                                createColumnBuilder(NcorePackage.Literals.PERIOD__END),
060                                                createColumnBuilder(NcorePackage.Literals.PERIOD__DURATION),
061                                                createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
062                                
063                                Action section = AppFactory.eINSTANCE.createAction();
064                                section.setText("Releases");
065                                section.getContent().add(releasesTable);
066                        
067                                action.getSections().add(section);
068                        }
069                        
070                        List<EngineeredCapability> features = groupedCapabilities.get(EngineeringPackage.Literals.FEATURE);
071                        if (features != null && !features.isEmpty()) {
072                                Table featuresTable = buildTable(
073                                                features, 
074                                                action, 
075                                                EngineeringPackage.Literals.ENGINEER__ASSIGNMENTS, 
076                                                context, 
077                                                progressMonitor,
078                                                createColumnBuilder("Feature"),
079                                                createColumnBuilder(EcorePackage.Literals.EOBJECT___ECONTAINER, "Product"),
080                                                createColumnBuilder(NcorePackage.Literals.PERIOD__START),
081                                                createColumnBuilder(NcorePackage.Literals.PERIOD__END),
082                                                createColumnBuilder(NcorePackage.Literals.PERIOD__DURATION),
083                                                createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
084                                
085                                Action section = AppFactory.eINSTANCE.createAction();
086                                section.setText("Features");
087                                section.getContent().add(featuresTable);
088                        
089                                action.getSections().add(section);
090                        }
091                        
092                        List<EngineeredCapability> issues = groupedCapabilities.get(EngineeringPackage.Literals.ISSUE);
093                        if (issues != null && !issues.isEmpty()) {
094                                Table issuesTable = buildTable(
095                                                issues, 
096                                                action, 
097                                                EngineeringPackage.Literals.ENGINEER__ASSIGNMENTS, 
098                                                context, 
099                                                progressMonitor,
100                                                createColumnBuilder("Issue"),
101                                                createColumnBuilder(EcorePackage.Literals.EOBJECT___ECONTAINER, "Module"),
102                                                createColumnBuilder(EngineeringPackage.Literals.ISSUE__INCREMENT),
103                                                createColumnBuilder(NcorePackage.Literals.PERIOD__START),
104                                                createColumnBuilder(NcorePackage.Literals.PERIOD__END),
105                                                createColumnBuilder(NcorePackage.Literals.PERIOD__DURATION),
106                                                createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
107                                
108                                Action section = AppFactory.eINSTANCE.createAction();
109                                section.setText("Issues");
110                                section.getContent().add(issuesTable);
111                        
112                                action.getSections().add(section);
113                        }                       
114                }               
115        }               
116                
117}
