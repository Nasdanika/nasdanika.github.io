001package org.nasdanika.engineering.gen;
002
003import java.util.Collections;
004import java.util.List;
005import java.util.function.BiConsumer;
006
007import org.eclipse.emf.common.util.EList;
008import org.eclipse.emf.ecore.EObject;
009import org.eclipse.emf.ecore.ETypedElement;
010import org.nasdanika.common.Context;
011import org.nasdanika.common.ProgressMonitor;
012import org.nasdanika.common.Util;
013import org.nasdanika.engineering.EngineeringPackage;
014import org.nasdanika.engineering.Interface;
015import org.nasdanika.html.model.app.Action;
016import org.nasdanika.html.model.app.AppFactory;
017
018public class ModuleActionBuilder<T extends org.nasdanika.engineering.Module> extends EngineeredElementActionBuilder<T> {
019        
020        public ModuleActionBuilder(T target, Context context) {
021                super(target, context);         
022        }
023                
024        @Override
025        protected Action buildAction(
026                        Action action,
027                        BiConsumer<EObject,Action> registry, 
028                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
029                        ProgressMonitor progressMonitor) throws Exception {
030
031                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
032                if (Util.isBlank(action.getIcon())) {
033                        action.setIcon("fas fa-cube");          
034                }
035                
036                createModuleActions(action, registry, resolveConsumer, progressMonitor);
037                createInterfaceActions(action, registry, resolveConsumer, progressMonitor);
038                
039                return action;
040        }
041        
042        @Override
043        protected List<ETypedElement> getProperties() {
044                List<ETypedElement> properties = super.getProperties();
045                properties.add(EngineeringPackage.Literals.MODULE__DEPENDENCIES);
046                properties.add(EngineeringPackage.Literals.MODULE__DEPENDANTS);
047                
048                return properties;
049        }
050        
051        /**
052         * Creates a list of actions for sub-packages. 
053         * @param progressMonitor
054         * @return An empty list if there are no sub-packages. A singleton list containing a grouping action containing sub-package actions otherwise.
055         * @throws Exception 
056         */
057        protected void createModuleActions(
058                        Action action,
059                        BiConsumer<EObject,Action> registry, 
060                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
061                        ProgressMonitor progressMonitor) throws Exception {
062                EList<EObject> children = action.getChildren();
063                for (org.nasdanika.engineering.Module module: getTarget().getModules()) {
064                        children.add(createChildAction(module, registry, resolveConsumer, progressMonitor));
065                }
066        }
067                
068        protected void createInterfaceActions(
069                        Action action,
070                        BiConsumer<EObject,Action> registry, 
071                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
072                        ProgressMonitor progressMonitor) throws Exception {
073                
074                EList<Action> groupAnonymous = action.getAnonymous();
075                for (Interface moduleInterface: getTarget().getInterfaces()) {
076                        groupAnonymous.add(createChildAction(moduleInterface, registry, resolveConsumer, progressMonitor));
077                }               
078        }
079                
080        @Override
081        protected void resolve(
082                        Action action, 
083                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
084                        ProgressMonitor progressMonitor) throws Exception {
085                super.resolve(action, context, progressMonitor);
086
087                // TODO - indented table with numbers of inbound connections?
088                // Interfaces
089                EList<Interface> interfaces = getTarget().getInterfaces();
090                if (!interfaces.isEmpty()) {
091                        Action interfacesAction = AppFactory.eINSTANCE.createAction();
092                        interfacesAction.setText("Interfaces");
093                        action.getSections().add(interfacesAction);
094                        ContentProvider<Interface> interfaceChildrenListProvider = new ContentProvider<Interface>() {
095
096                                @Override
097                                public List<EObject> createContent(
098                                                Interface element, 
099                                                Action base, 
100                                                ETypedElement typedElement,
101                                                org.nasdanika.html.emf.EObjectActionResolver.Context context, 
102                                                ProgressMonitor progressMonitor) throws Exception {
103                                        
104                                        EList<Interface> children = element.getChildren();
105                                        if (children.isEmpty()) {
106                                                return null;
107                                        }
108                                        return Collections.singletonList(renderList(children, true, this, base, EngineeringPackage.Literals.INTERFACE__CHILDREN, context, progressMonitor));
109                                }
110                                
111                        };
112                        
113                        interfacesAction.getContent().add(renderList(interfaces, true, interfaceChildrenListProvider, action, EngineeringPackage.Literals.MODULE__INTERFACES, context, progressMonitor));
114                        
115                }
116                                
117        }       
118        
119}
