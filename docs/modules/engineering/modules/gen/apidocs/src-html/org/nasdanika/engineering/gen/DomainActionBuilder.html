001package org.nasdanika.engineering.gen;
002
003import java.util.function.BiConsumer;
004
005import org.eclipse.emf.common.util.EList;
006import org.eclipse.emf.ecore.EObject;
007import org.nasdanika.common.Context;
008import org.nasdanika.common.ProgressMonitor;
009import org.nasdanika.engineering.Domain;
010import org.nasdanika.engineering.EngineeredElement;
011import org.nasdanika.html.model.app.Action;
012
013public class DomainActionBuilder extends EngineeredElementActionBuilder<Domain> {
014        
015        public DomainActionBuilder(Domain target, Context context) {
016                super(target, context);         
017        }
018                
019        @Override
020        protected Action buildAction(
021                        Action action,
022                        BiConsumer<EObject,Action> registry, 
023                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
024                        ProgressMonitor progressMonitor) throws Exception {
025                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);         
026                createElementActions(action, registry, resolveConsumer, progressMonitor);               
027                return action;
028        }
029        
030        protected void createElementActions(
031                        Action action,
032                        BiConsumer<EObject,Action> registry, 
033                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
034                        ProgressMonitor progressMonitor) throws Exception {
035                EList<EObject> children = action.getChildren();
036                for (EngineeredElement element: getTarget().getElements()) {
037                        children.add(createChildAction(element, registry, resolveConsumer, progressMonitor));
038                }
039        }       
040        
041}
