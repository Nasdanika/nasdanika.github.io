023
024public class IncrementActionBuilder extends NamedElementActionBuilder<Increment> {
025        
026        public IncrementActionBuilder(Increment target, Context context) {
027                super(target, context);         
028        }
029        
030        @Override
031        protected List<ETypedElement> getProperties() {
032                List<ETypedElement> properties = super.getProperties();
033                
034                // Endeavor's
035                properties.add(EngineeringPackage.Literals.ENDEAVOR__ASSIGNEE);
036                properties.add(EngineeringPackage.Literals.ENDEAVOR__COMPLETION);
037                properties.add(EngineeringPackage.Literals.ENDEAVOR__TOTAL_COST);
038                properties.add(EngineeringPackage.Literals.ENDEAVOR__BENEFIT);
039                
040                return properties;
041        }
042                
043        @Override
044        protected Action buildAction(
045                        Action action,
046                        BiConsumer<EObject,Action> registry, 
047                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
048                        ProgressMonitor progressMonitor) throws Exception {
049                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
050                action.setIcon("far fa-calendar-alt");
051                
052                createChildrenActions(action, registry, resolveConsumer, progressMonitor);
053                createAlignsActions(action, registry, resolveConsumer, progressMonitor);
054                createReleasesAction(action, registry, resolveConsumer, progressMonitor);               
055                createIssuesAction(action, registry, resolveConsumer, progressMonitor);         
056                createAllIssuesAction(action, registry, resolveConsumer, progressMonitor);              
057                createAllObjectiveActions(action, registry, resolveConsumer, progressMonitor);
058                
059                return action;
060        }
061        
062        protected void createAllObjectiveActions(
063                        Action action,
064                        BiConsumer<EObject,Action> registry, 
065                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
066                        ProgressMonitor progressMonitor) throws Exception {
067                
068                if (!getTarget().getAllObjectives().isEmpty()) {
069                        Action group = AppFactory.eINSTANCE.createAction();
070                        group.setText("Objectives");
071                        group.setUuid(action.getUuid() + "-all-objectives");
072                        // TODO - icon, ...
073                        EList<Action> groupAnonymous = group.getAnonymous();
074                        for (Objective objective: getTarget().getObjectives()) {
075                                groupAnonymous.add(createChildAction(objective, registry, resolveConsumer, progressMonitor));
076                        }
077                
078                        action.getSections().add(group);
079                }
080        }
081        
082        protected void createAlignsActions(
083                        Action action, 
084                        BiConsumer<EObject,Action> registry, 
085                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
086                        ProgressMonitor progressMonitor) throws Exception {
087        
088                List<Alignment> alignments = getTarget().getAligns();
089                if (!alignments.isEmpty()) {
090                        Action group = AppFactory.eINSTANCE.createAction();
091                        group.setText("Aligns");
092                        group.setUuid(action.getUuid() + "-aligns");
093                        action.getSections().add(group);
094                }
095                
096        }
097

104                List<Release> releases = getTarget().getReleases();

112        
113        protected void createIssuesAction(
114                        Action action,
115                        BiConsumer<EObject,Action> registry, 
116                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
117                        ProgressMonitor progressMonitor) throws Exception {
118                
119                List<Issue> issues = getTarget().getIssues();

125                }
126        }
127        
128        protected void createAllIssuesAction(
129                        Action action,
130                        BiConsumer<EObject,Action> registry, 
131                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
132                        ProgressMonitor progressMonitor) throws Exception {
133                
134                List<Issue> allIssues = getTarget().getAllIssues();
135                if (!allIssues.isEmpty()) {
136                        Action group = AppFactory.eINSTANCE.createAction();
137                        group.setText("All Issues");

142                
143        }
144        
145        @Override
146        protected void resolve(
147                        Action action, 
148                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
149                        ProgressMonitor progressMonitor) throws Exception {
150                super.resolve(action, context, progressMonitor);
151                
152                // Endeavor's - aligns, all objectives (objectives), all issues, capacity (TODO)
153                EList<Alignment> alignments = getTarget().getAligns();
154                if (!alignments.isEmpty()) {
155                        String alignsGroupUUID = action.getUuid() + "-aligns";
156                        Optional<Action> alignmentsActionOptional = action.getSections().stream()
157                                        .filter(a -> alignsGroupUUID.equals(a.getUuid()))
158                                        .findFirst();
159                        
160                        Table alignmentsTable = buildTable(
161                                        alignments, 
162                                        action, 
163                                        EngineeringPackage.Literals.ALIGNABLE__ALIGNS, 
164                                        context, 
165                                        progressMonitor, 
166                                        createColumnBuilder(EngineeringPackage.Literals.ALIGNMENT__AIM),
167                                        createColumnBuilder(NcorePackage.Literals.MODEL_ELEMENT__DESCRIPTION),
168                                        createColumnBuilder(EngineeringPackage.Literals.ALIGNMENT__WEIGHT));
169                        alignmentsActionOptional.get().getContent().add(alignmentsTable);
170                }
171                
172                EList<Objective> allObjectives = getTarget().getAllObjectives();
173                if (!allObjectives.isEmpty()) {
174                        String allObjectivesGroupUUID = action.getUuid() + "-all-objectives";
175                        Optional<Action> allObjectivesActionOptional = action.getSections().stream()
176                                        .filter(a -> allObjectivesGroupUUID.equals(a.getUuid()))
177                                        .findFirst();
178                        
179                        allObjectivesActionOptional.get().getContent().add(renderList(allObjectives, true, null, action, EngineeringPackage.Literals.ENDEAVOR__ALL_OBJECTIVES, context, progressMonitor)); // Table?
180                }
181                                
182                // Releases
183                EList<Release> releases = getTarget().getReleases();
184                if (!releases.isEmpty()) {
185                        String releasesGroupUUID = action.getUuid() + "-releases";
186                        Optional<Action> releasesActionOptional = action.getSections().stream()
187                                        .filter(Action.class::isInstance)                                       
188                                        .map(Action.class::cast)
189                                        .filter(a -> releasesGroupUUID.equals(a.getUuid()))
190                                        .findFirst();
191                        Action releasesAction = releasesActionOptional.get();
192                        Table allIssuesTable = buildTable(
193                                        releases, 
194                                        action, 
195                                        EngineeringPackage.Literals.INCREMENT__RELEASES, 
196                                        context, 
197                                        progressMonitor,
198                                        createColumnBuilder(EcorePackage.Literals.EOBJECT___ECONTAINER, "Product"),
199                                        createColumnBuilder("Release"),
200                                        createColumnBuilder(NcorePackage.Literals.PERIOD__START),                                       
201                                        createColumnBuilder(NcorePackage.Literals.PERIOD__END),                                 
202                                        createColumnBuilder(NcorePackage.Literals.PERIOD__DURATION),                                    
203                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
204                        
205                        releasesAction.getContent().add(allIssuesTable);
206                }
207                
208                // Issues
209                EList<Issue> issues = getTarget().getIssues();
210                if (!issues.isEmpty()) {
211                        String issuesGroupUUID = action.getUuid() + "-issues";
212                        Optional<Action> issuesActionOptional = action.getSections().stream()
213                                        .filter(Action.class::isInstance)                                       
214                                        .map(Action.class::cast)
215                                        .filter(a -> issuesGroupUUID.equals(a.getUuid()))
216                                        .findFirst();
217                        Action issuesAction = issuesActionOptional.get();
218                        Table issuesTable = buildTable(
219                                        issues, 
220                                        action, 
221                                        EngineeringPackage.Literals.INCREMENT__ISSUES, 
222                                        context, 
223                                        progressMonitor,
224                                        createColumnBuilder("Issue"),
225                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__TARGET),
226                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__ASSIGNEE),
227                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__CATEGORIES),
228                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__PRIORITY),
229                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__SEVERITY),
230                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__STATUS),
231                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__WORKABLE),
232                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__BENEFIT),
233                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__COST),
234                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__EFFORT),
235                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__REMAINING_COST),
236                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__REMAINING_EFFORT),
237                                        createColumnBuilder(EngineeringPackage.Literals.ENGINEERED_CAPABILITY__RELEASES),
238                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__CONTRIBUTES_TO),                                 
239                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
240                        
241                        issuesAction.getContent().add(issuesTable);
242                }
243                
244                // All Issues
245                EList<Issue> allIssues = getTarget().getAllIssues();
246                if (!allIssues.isEmpty()) {
247                        String allIssuesGroupUUID = action.getUuid() + "-all-issues";
248                        Optional<Action> issuesActionOptional = action.getNavigation().stream()
249                                        .filter(Action.class::isInstance)                                       
250                                        .map(Action.class::cast)
251                                        .filter(a -> allIssuesGroupUUID.equals(a.getUuid()))
252                                        .findFirst();
253                        Action allIssuesAction = issuesActionOptional.get();
254                        Table allIssuesTable = buildTable(
255                                        allIssues, 
256                                        action, 
257                                        EngineeringPackage.Literals.ENGINEERED_ELEMENT__ALL_ISSUES, 
258                                        context, 
259                                        progressMonitor,
260                                        createColumnBuilder("Issue"),
261                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__TARGET),
262                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__ASSIGNEE),
263                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__CATEGORIES),
264                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__PRIORITY),
265                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__SEVERITY),
266                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__STATUS),
267                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__WORKABLE),
268                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__BENEFIT),
269                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__COST),
270                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__EFFORT),
271                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__REMAINING_COST),
272                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__REMAINING_EFFORT),
273                                        createColumnBuilder(EngineeringPackage.Literals.ENGINEERED_CAPABILITY__RELEASES),
274                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__CONTRIBUTES_TO),                                 
275                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
276                        
277                        allIssuesAction.getContent().add(allIssuesTable);
278                }
279                
280        }       
281
282        /**
283         * Creates a list of actions for sub-packages. 
284         * @param progressMonitor
285         * @return An empty list if there are no sub-packages. A singleton list containing a grouping action containing sub-package actions otherwise.
286         * @throws Exception 
287         */
288        protected void createChildrenActions(
289                        Action action,
290                        BiConsumer<EObject,Action> registry, 
291                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
292                        ProgressMonitor progressMonitor) throws Exception {
293                List<Increment> children = getTarget().getChildren();
294                if (!children.isEmpty()) {
295                        EList<EObject> actionChildren = action.getChildren();
296                        for (Increment child: children) {
297                                actionChildren.add(createChildAction(child, registry, resolveConsumer, progressMonitor));
298                        }
299                }               
300        }
