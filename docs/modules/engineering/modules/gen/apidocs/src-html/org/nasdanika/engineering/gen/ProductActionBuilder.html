001package org.nasdanika.engineering.gen;
002
003import java.util.Collections;
004import java.util.List;
005import java.util.Optional;
006import java.util.function.BiConsumer;
007import java.util.function.Consumer;
008
009import org.eclipse.emf.common.util.EList;
010import org.eclipse.emf.ecore.EObject;
011import org.eclipse.emf.ecore.ETypedElement;
012import org.nasdanika.common.Context;
013import org.nasdanika.common.ProgressMonitor;
014import org.nasdanika.common.Util;
015import org.nasdanika.engineering.EngineeringPackage;
016import org.nasdanika.engineering.Feature;
017import org.nasdanika.engineering.Product;
018import org.nasdanika.engineering.Release;
019import org.nasdanika.html.model.app.Action;
020import org.nasdanika.html.model.app.AppFactory;
021
022public class ProductActionBuilder extends ModuleActionBuilder<Product> {
023        
024        public ProductActionBuilder(Product target, Context context) {
025                super(target, context);         
026        }
027        
028        @Override
029        protected Action buildAction(
030                        Action action,
031                        BiConsumer<EObject, Action> registry,
032                        Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer,
033                        ProgressMonitor progressMonitor) throws Exception {
034                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
035                if (Util.isBlank(action.getIcon())) {
036                        action.setIcon("fas fa-box");
037                }
038                
039                createReleaseActions(action, registry, resolveConsumer, progressMonitor);               
040                createFeatureActions(action, registry, resolveConsumer, progressMonitor);
041                
042                return action;
043        }
044        
045        protected void createReleaseActions(
046                        Action action,
047                        BiConsumer<EObject,Action> registry, 
048                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
049                        ProgressMonitor progressMonitor) throws Exception {
050                
051                List<Release> releases = getTarget().getReleases();
052                if (!releases.isEmpty()) {
053                        Action group = AppFactory.eINSTANCE.createAction();
054                        group.setText("Releases");
055                        group.setUuid(action.getUuid() + "-releases");
056                        group.setLocation("releases.html");
057                        // TODO - icon, ...
058                        EList<Action> groupAnonymous = group.getAnonymous();
059                        for (Release release: releases) {
060                                groupAnonymous.add(createChildAction(release, registry, resolveConsumer, progressMonitor));
061                        }
062                
063                        action.getNavigation().add(group);
064                }
065        }
066        
067        protected void createFeatureActions(
068                        Action action,
069                        BiConsumer<EObject,Action> registry, 
070                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
071                        ProgressMonitor progressMonitor) throws Exception {
072                
073                List<Feature> features = getTarget().getFeatures();
074                if (!features.isEmpty()) {
075                        Action group = AppFactory.eINSTANCE.createAction();
076                        group.setText("Features");
077                        group.setUuid(action.getUuid() + "-features");
078                        group.setLocation("features.html");
079                        // TODO - icon, ...
080                        EList<Action> groupAnonymous = group.getAnonymous();
081                        for (Feature feature: features) {
082                                groupAnonymous.add(createChildAction(feature, registry, resolveConsumer, progressMonitor));
083                        }
084                
085                        action.getNavigation().add(group);
086                }
087        }
088                
089        @Override
090        protected void resolve(
091                        Action action, 
092                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
093                        ProgressMonitor progressMonitor) throws Exception {
094                super.resolve(action, context, progressMonitor);
095                
096                Product product = getTarget();
097                
098                EList<Release> releases = product.getReleases();
099                if (!releases.isEmpty()) {
100                        String releaseGroupUUID = action.getUuid() + "-releases";
101                        Optional<Action> releasesActionOptional = action.getNavigation().stream()
102                                        .filter(Action.class::isInstance)                                       
103                                        .map(Action.class::cast)
104                                        .filter(a -> releaseGroupUUID.equals(a.getUuid()))
105                                        .findFirst();
106                        
107                        Action releasesAction = releasesActionOptional.get();
108                        releasesAction.getContent().add(renderList(releases, false, null, releasesAction, EngineeringPackage.Literals.PRODUCT__RELEASES, context, progressMonitor)); // Table?
109                }
110                
111                EList<Feature> features = product.getFeatures();
112                if (!features.isEmpty()) {
113                        String featuresGroupUUID = action.getUuid() + "-features";
114                        Optional<Action> featuresActionOptional = action.getNavigation().stream()
115                                        .filter(Action.class::isInstance)                                       
116                                        .map(Action.class::cast)
117                                        .filter(a -> featuresGroupUUID.equals(a.getUuid()))
118                                        .findFirst();
119                        
120                        Action featuresAction = featuresActionOptional.get();
121                        featuresAction.getContent().add(renderList(features, false, createFeatureChildrenProvider(), featuresAction, EngineeringPackage.Literals.PRODUCT__FEATURES, context, progressMonitor)); // Table?
122                }               
123        }
124
125        private ContentProvider<Feature> createFeatureChildrenProvider() {
126                return new ContentProvider<Feature>() {
127
128                        @Override
129                        public List<EObject> createContent(
130                                        Feature element, 
131                                        Action base, 
132                                        ETypedElement typedElement,
133                                        org.nasdanika.html.emf.EObjectActionResolver.Context context, 
134                                        ProgressMonitor progressMonitor) throws Exception {
135                                
136                                EList<Feature> children = element.getChildren();
137                                if (children.isEmpty()) {
138                                        return null;
139                                }
140                                return Collections.singletonList(renderList(children, false, this, base, EngineeringPackage.Literals.FEATURE__CHILDREN, context, progressMonitor));
141                        }
142                        
143                };
144        }
