001package org.nasdanika.engineering.gen;
002
003import java.util.Collections;
004import java.util.List;
005import java.util.function.BiConsumer;
006import java.util.function.Consumer;
007
008import org.eclipse.emf.common.util.EList;
009import org.eclipse.emf.ecore.EObject;
010import org.eclipse.emf.ecore.ETypedElement;
011import org.nasdanika.common.Context;
012import org.nasdanika.common.ProgressMonitor;
013import org.nasdanika.engineering.Directory;
014import org.nasdanika.engineering.EngineeringPackage;
015import org.nasdanika.engineering.NamedElement;
016import org.nasdanika.html.model.app.Action;
017import org.nasdanika.html.model.app.AppFactory;
018import org.nasdanika.html.model.bootstrap.Table;
019
020public class DirectoryActionBuilder extends EngineeredElementActionBuilder<Directory> {
021        
022        public DirectoryActionBuilder(Directory target, Context context) {
023                super(target, context);
024        }
025        
026        @Override
027        protected Action buildAction(
028                        Action action, 
029                        BiConsumer<EObject, Action> registry,
030                        Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer,
031                        ProgressMonitor progressMonitor) throws Exception {
032                Action ret = super.buildAction(action, registry, resolveConsumer, progressMonitor);
033                
034                EList<Action> anonymous = ret.getAnonymous();
035                for (NamedElement element: getTarget().getElements()) {
036                        anonymous.add(createChildAction(element, registry, resolveConsumer, progressMonitor));
037                }
038                
039                return ret;
040        }
041        
042        @Override
043        protected Table createPropertiesTable(
044                        Action action, 
045                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
046                        ProgressMonitor progressMonitor) throws Exception {             
047                Table propertiesTable = super.createPropertiesTable(action, context, progressMonitor);
048                if (propertiesTable != null) {
049                        Action propertiesAction = AppFactory.eINSTANCE.createAction();
050                        propertiesAction.setText("Properties");
051                        propertiesAction.setLocation(action.getUuid()+"-properties.html");
052                        propertiesAction.getContent().add(propertiesTable);
053                        action.getNavigation().add(propertiesAction);
054                }
055                return null;
056        }
057        
058        @Override
059        protected void resolve(
060                        Action action, 
061                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
062                        ProgressMonitor progressMonitor) throws Exception {
063                super.resolve(action, context, progressMonitor);
064        
065                action.getContent().add(renderList(getTarget().getElements(), true, createElementChildrenProvider(), action, EngineeringPackage.Literals.MODEL_ELEMENT__RESOURCES, context, progressMonitor)); // Table?                
066        }
067        
068        
069        private ContentProvider<NamedElement> createElementChildrenProvider() {
070                return new ContentProvider<NamedElement>() {
071        
072                        @Override
073                        public List<EObject> createContent(
074                                        NamedElement element, 
075                                        Action base, 
076                                        ETypedElement typedElement,
077                                        org.nasdanika.html.emf.EObjectActionResolver.Context context, 
078                                        ProgressMonitor progressMonitor) throws Exception {
079        
080                                if (element instanceof Directory) {
081                                        EList<NamedElement> children = ((Directory) element).getElements();
082                                        if (!children.isEmpty()) {
083                                                return Collections.singletonList(renderList(children, true, this, base, EngineeringPackage.Literals.DIRECTORY__ELEMENTS, context, progressMonitor));
084                                        }
085                                }
086                                return null;
087                        }
088                        
089                };
090        }       
091        
092        
093}
