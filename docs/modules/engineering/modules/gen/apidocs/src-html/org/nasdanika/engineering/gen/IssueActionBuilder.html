001package org.nasdanika.engineering.gen;
002
003import java.util.Collections;
004import java.util.List;
005import java.util.Optional;
006import java.util.function.BiConsumer;
007import java.util.function.Consumer;
008
009import org.eclipse.emf.common.util.EList;
010import org.eclipse.emf.ecore.EObject;
011import org.eclipse.emf.ecore.ETypedElement;
012import org.nasdanika.common.Context;
013import org.nasdanika.common.ProgressMonitor;
014import org.nasdanika.engineering.Capability;
015import org.nasdanika.engineering.EngineeringPackage;
016import org.nasdanika.engineering.Issue;
017import org.nasdanika.engineering.Note;
018import org.nasdanika.html.model.app.Action;
019import org.nasdanika.html.model.app.AppFactory;
020import org.nasdanika.html.model.bootstrap.Table;
021import org.nasdanika.ncore.NcorePackage;
022
023public class IssueActionBuilder extends EngineeredCapabilityActionBuilder<Issue> {
024        
025        public IssueActionBuilder(Issue target, Context context) {
026                super(target, context);         
027        }
028        
029        @Override
030        protected List<ETypedElement> getProperties() {
031                List<ETypedElement> properties = super.getProperties();
032                properties.add(EngineeringPackage.Literals.ISSUE__INCREMENT);
033                properties.add(EngineeringPackage.Literals.ISSUE__CATEGORIES);
034                properties.add(EngineeringPackage.Literals.ISSUE__PRIORITY);
035                properties.add(EngineeringPackage.Literals.ISSUE__SEVERITY);
036                properties.add(EngineeringPackage.Literals.ISSUE__STATUS);
037                properties.add(EngineeringPackage.Literals.ISSUE__CONTRIBUTES_TO);
038                properties.add(EngineeringPackage.Literals.ISSUE__COST);
039                properties.add(EngineeringPackage.Literals.ISSUE__EFFORT);
040                properties.add(EngineeringPackage.Literals.ISSUE__REMAINING_COST);
041                properties.add(EngineeringPackage.Literals.ISSUE__REMAINING_EFFORT);
042                properties.add(EngineeringPackage.Literals.ISSUE__TARGET);
043                properties.add(EngineeringPackage.Literals.ISSUE__WORKABLE);
044                properties.add(EngineeringPackage.Literals.ISSUE__REQUIRES);
045                return properties;
046        }
047
048        @Override
049        protected Action buildAction(
050                        Action action,
051                        BiConsumer<EObject, Action> registry,
052                        Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer,
053                        ProgressMonitor progressMonitor) throws Exception {
054                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
055                
056                createRequiresAction(action, registry, resolveConsumer, progressMonitor);               
057                createChildrenActions(action, registry, resolveConsumer, progressMonitor);              
058                createNotesActions(action, registry, resolveConsumer, progressMonitor);
059                
060                return action;
061        }
062        
063        protected void createChildrenActions(
064                        Action action,
065                        BiConsumer<EObject,Action> registry, 
066                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
067                        ProgressMonitor progressMonitor) throws Exception {
068                
069                List<Issue> children = getTarget().getChildren();
070                if (!children.isEmpty()) {
071                        Action group = AppFactory.eINSTANCE.createAction();
072                        group.setText("Children");
073                        group.setUuid(action.getUuid() + "-children");
074                        EList<Action> groupAnonymous = group.getAnonymous();
075                        for (Issue child: children) {
076                                groupAnonymous.add(createChildAction(child, registry, resolveConsumer, progressMonitor));
077                        }
078                
079                        action.getSections().add(group);
080                }
081        }
082        
083        protected void createNotesActions(
084                        Action action,
085                        BiConsumer<EObject,Action> registry, 
086                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
087                        ProgressMonitor progressMonitor) throws Exception {
088                
089                List<Note> notes = getTarget().getNotes();
090                if (!notes.isEmpty()) {
091                        Action group = AppFactory.eINSTANCE.createAction();
092                        group.setText("Notes");
093                        group.setUuid(action.getUuid() + "-notes");
094                        action.getSections().add(group);
095                }
096        }
097                
098        @Override
099        protected void resolve(
100                        Action action, 
101                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
102                        ProgressMonitor progressMonitor) throws Exception {
103                super.resolve(action, context, progressMonitor);
104                
105                Issue issue = getTarget();
106                
107                EList<Capability> requires = getTarget().getRequires();
108                if (!requires.isEmpty()) {
109                        String allIssuesGroupUUID = action.getUuid() + "-requires";
110                        Optional<Action> requiredByOptional = action.getSections().stream()
111                                        .filter(a -> allIssuesGroupUUID.equals(a.getUuid()))
112                                        .findFirst();
113                        Action issuesAction = requiredByOptional.get();
114                        Table requiredByTable = buildTable(
115                                        requires, 
116                                        action, 
117                                        EngineeringPackage.Literals.ISSUE__REQUIRES, 
118                                        context, 
119                                        progressMonitor,
120                                        createColumnBuilder("Capability"),
121                                        createColumnBuilder(EngineeringPackage.Literals.CAPABILITY__AVAILABLE),
122                                        createColumnBuilder(NcorePackage.Literals.PERIOD__START),
123                                        createColumnBuilder(NcorePackage.Literals.PERIOD__END));
124                        
125                        issuesAction.getContent().add(requiredByTable);
126                }
127                                
128                EList<Issue> children = issue.getChildren();
129                if (!children.isEmpty()) {
130                        String childrenGroupUUID = action.getUuid() + "-children";
131                        Optional<Action> childrenActionOptional = action.getSections().stream()
132                                        .filter(a -> childrenGroupUUID.equals(a.getUuid()))
133                                        .findFirst();
134                        
135                        ContentProvider<Issue> issueChildrenListProvider = new ContentProvider<Issue>() {
136
137                                @Override
138                                public List<EObject> createContent(
139                                                Issue element, 
140                                                Action base, 
141                                                ETypedElement typedElement,
142                                                org.nasdanika.html.emf.EObjectActionResolver.Context context, 
143                                                ProgressMonitor progressMonitor) throws Exception {
144                                        
145                                        EList<Issue> children = element.getChildren();
146                                        if (children.isEmpty()) {
147                                                return null;
148                                        }
149                                        return Collections.singletonList(renderList(children, true, this, base, EngineeringPackage.Literals.ISSUE__CHILDREN, context, progressMonitor));
150                                }
151                                
152                        };
153                        
154                        Action childrenAction = childrenActionOptional.get();
155                        childrenAction.getContent().add(renderList(children, false, issueChildrenListProvider, action, EngineeringPackage.Literals.ISSUE__CHILDREN, context, progressMonitor)); // Table?
156                }
157                
158                EList<Note> notes = issue.getNotes();
159                if (!notes.isEmpty()) {
160                        String featuresGroupUUID = action.getUuid() + "-notes";
161                        Optional<Action> featuresActionOptional = action.getSections().stream()
162                                        .filter(a -> featuresGroupUUID.equals(a.getUuid()))
163                                        .findFirst();
164                        
165                        Action featuresAction = featuresActionOptional.get();
166                        Table notesTable = buildTable(
167                                        notes, 
168                                        action, 
169                                        EngineeringPackage.Literals.ISSUE__NOTES, 
170                                        context, 
171                                        progressMonitor, 
172                                        createColumnBuilder(NcorePackage.Literals.MODEL_ELEMENT__DESCRIPTION),
173                                        createColumnBuilder(EngineeringPackage.Literals.NOTE__AUTHOR),
174                                        createColumnBuilder(EngineeringPackage.Literals.NOTE__DATE),
175                                        createColumnBuilder(EngineeringPackage.Literals.NOTE__STATUS),
176                                        createColumnBuilder(EngineeringPackage.Literals.NOTE__EFFORT),
177                                        createColumnBuilder(EngineeringPackage.Literals.NOTE__COST),
178                                        createColumnBuilder(EngineeringPackage.Literals.NOTE__REMAINING_EFFORT),
179                                        createColumnBuilder(EngineeringPackage.Literals.NOTE__REMAINING_COST));
180                        featuresAction.getContent().add(notesTable);
181                }               
182        }
183
184        protected void createRequiresAction(
185                        Action action,
186                        BiConsumer<EObject,Action> registry, 
187                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
188                        ProgressMonitor progressMonitor) throws Exception {
189                
190                List<Capability> requires = getTarget().getRequires();
191                if (!requires.isEmpty()) {
192                        Action group = AppFactory.eINSTANCE.createAction();
193                        group.setText("Requires");
194                        group.setUuid(action.getUuid() + "-requires");
195                        group.setLocation("requires.html");
196                        action.getSections().add(group);
197                }               
198        }
