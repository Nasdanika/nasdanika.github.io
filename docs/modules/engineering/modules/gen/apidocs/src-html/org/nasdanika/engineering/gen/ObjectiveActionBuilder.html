001package org.nasdanika.engineering.gen;
002
003import java.util.Collections;
004import java.util.List;
005import java.util.Optional;
006import java.util.function.BiConsumer;
007
008import org.eclipse.emf.common.util.EList;
009import org.eclipse.emf.ecore.EObject;
010import org.eclipse.emf.ecore.ETypedElement;
011import org.nasdanika.common.Context;
012import org.nasdanika.common.ProgressMonitor;
013import org.nasdanika.engineering.EngineeringPackage;
014import org.nasdanika.engineering.KeyResult;
015import org.nasdanika.engineering.Objective;
016import org.nasdanika.html.model.app.Action;
017import org.nasdanika.html.model.app.AppFactory;
018
019public class ObjectiveActionBuilder extends KeyResultActionBuilder<Objective> {
020        
021        public ObjectiveActionBuilder(Objective target, Context context) {
022                super(target, context);         
023        }
024        
025        @Override
026        protected List<ETypedElement> getProperties() {
027                List<ETypedElement> properties = super.getProperties();
028                properties.add(EngineeringPackage.Literals.OBJECTIVE__PARENT);
029                properties.add(EngineeringPackage.Literals.OBJECTIVE__ENDEAVOR);
030                
031                return properties;
032        }
033        
034        @Override
035        protected Action buildAction(
036                        Action action,
037                        BiConsumer<EObject,Action> registry, 
038                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
039                        ProgressMonitor progressMonitor) throws Exception {
040                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
041                
042                createSubObjectiveActions(action, registry, resolveConsumer, progressMonitor);
043                createKeyResultActions(action, registry, resolveConsumer, progressMonitor);
044                
045                return action;
046        }
047        
048        protected void createSubObjectiveActions(
049                        Action action, 
050                        BiConsumer<EObject,Action> registry, 
051                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
052                        ProgressMonitor progressMonitor) throws Exception {
053                
054                if (!getTarget().getSubObjectives().isEmpty()) {
055                        Action group = AppFactory.eINSTANCE.createAction();
056                        group.setText("Children");
057                        group.setUuid(action.getUuid() + "-sub-objectives");
058                        action.getSections().add(group);
059                        EList<Action> groupAnonymous = group.getAnonymous();
060                        for (Objective objective: getTarget().getChildren()) {
061                                groupAnonymous.add(createChildAction(objective, registry, resolveConsumer, progressMonitor));
062                        }
063                }
064        }
065        
066        protected void createKeyResultActions(
067                        Action action, 
068                        BiConsumer<EObject,Action> registry, 
069                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
070                        ProgressMonitor progressMonitor) throws Exception {
071                
072                EList<KeyResult> keyResults = getTarget().getKeyResults();
073                if (!keyResults.isEmpty()) {
074                        Action group = AppFactory.eINSTANCE.createAction();
075                        group.setText("Key Results");
076                        group.setUuid(action.getUuid() + "-key-results");
077                        action.getSections().add(group);
078                        EList<Action> groupAnonymous = group.getAnonymous();
079                        for (KeyResult keyResult: keyResults) {
080                                groupAnonymous.add(createChildAction(keyResult, registry, resolveConsumer, progressMonitor));
081                        }
082                }
083        }
084        
085        @Override
086        protected void resolve(
087                        Action action, 
088                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
089                        ProgressMonitor progressMonitor) throws Exception {
090                super.resolve(action, context, progressMonitor);
091                
092                EList<Objective> subObjectives = getTarget().getSubObjectives();
093                if (!subObjectives.isEmpty()) {
094                        String subObjectivesGroupUUID = action.getUuid() + "-sub-objectives";
095                        Optional<Action> subObjectivesActionOptional = action.getSections().stream()
096                                        .filter(a -> subObjectivesGroupUUID.equals(a.getUuid()))
097                                        .findFirst();
098                        subObjectivesActionOptional.get().getContent().add(renderList(subObjectives, true, createObjectiveSubObjectiveListProvider(), action, EngineeringPackage.Literals.OBJECTIVE__SUB_OBJECTIVES, context, progressMonitor));
099                }
100                
101                EList<KeyResult> keyResults = getTarget().getKeyResults();
102                if (!keyResults.isEmpty()) {
103                        String keyResultsGroupUUID = action.getUuid() + "-key-results";
104                        Optional<Action> keyResultsActionOptional = action.getSections().stream()
105                                        .filter(a -> keyResultsGroupUUID.equals(a.getUuid()))
106                                        .findFirst();
107                        keyResultsActionOptional.get().getContent().add(renderList(keyResults, true, null, action, EngineeringPackage.Literals.OBJECTIVE__KEY_RESULTS, context, progressMonitor));
108                }
109        }
110
111        protected ContentProvider<Objective> createObjectiveSubObjectiveListProvider() {
112                ContentProvider<Objective> objectiveSubObjectiveListProvider = new ContentProvider<Objective>() {
113
114                        @Override
115                        public List<EObject> createContent(
116                                        Objective element, 
117                                        Action base, 
118                                        ETypedElement typedElement,
119                                        org.nasdanika.html.emf.EObjectActionResolver.Context context, 
120                                        ProgressMonitor progressMonitor) throws Exception {
121                                
122                                EList<Objective> subObjectives = element.getSubObjectives();
123                                if (subObjectives.isEmpty()) {
124                                        return null;
125                                }
126                                return Collections.singletonList(renderList(subObjectives, true, this, base, EngineeringPackage.Literals.OBJECTIVE__SUB_OBJECTIVES, context, progressMonitor));
127                        }
128                        
129                };
130                return objectiveSubObjectiveListProvider;
131        }
132        
133}
