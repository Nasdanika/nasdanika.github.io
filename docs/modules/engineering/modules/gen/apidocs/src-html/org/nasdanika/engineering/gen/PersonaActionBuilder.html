001package org.nasdanika.engineering.gen;
002
003import java.util.Collections;
004import java.util.List;
005import java.util.Optional;
006import java.util.function.BiConsumer;
007
008import org.eclipse.emf.common.util.EList;
009import org.eclipse.emf.ecore.EObject;
010import org.eclipse.emf.ecore.ETypedElement;
011import org.nasdanika.common.Context;
012import org.nasdanika.common.ProgressMonitor;
013import org.nasdanika.engineering.EngineeringPackage;
014import org.nasdanika.engineering.Goal;
015import org.nasdanika.engineering.Persona;
016import org.nasdanika.html.model.app.Action;
017import org.nasdanika.html.model.app.AppFactory;
018
019public class PersonaActionBuilder<T extends Persona> extends EngineeredElementActionBuilder<T> {
020        
021        public PersonaActionBuilder(T target, Context context) {
022                super(target, context);         
023        }
024                
025        @Override
026        protected Action buildAction(
027                        Action action,
028                        BiConsumer<EObject,Action> registry, 
029                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
030                        ProgressMonitor progressMonitor) throws Exception {
031                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
032                
033                createGoalActions(action, registry, resolveConsumer, progressMonitor);
034                
035                return action;
036        }
037        
038        @Override
039        protected List<ETypedElement> getProperties() {
040                List<ETypedElement> properties = super.getProperties();
041                properties.add(EngineeringPackage.Literals.PERSONA__EXTENDS);
042                properties.add(EngineeringPackage.Literals.PERSONA__EXTENSIONS);
043                properties.add(EngineeringPackage.Literals.PERSONA__REPRESENTATIVES);
044                return properties;
045        }
046        
047        protected void createGoalActions(
048                        Action action, 
049                        BiConsumer<EObject,Action> registry, 
050                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
051                        ProgressMonitor progressMonitor) throws Exception {
052                
053                List<Goal> goals = getTarget().getGoals();
054                if (!goals.isEmpty()) {
055                        Action group = AppFactory.eINSTANCE.createAction();
056                        group.setText("Goals");
057                        group.setUuid(action.getUuid() + "-goals");
058                        action.getSections().add(group);
059                        EList<Action> groupAnonymous = group.getAnonymous();
060                        for (Goal goal: goals) {
061                                groupAnonymous.add(createChildAction(goal, registry, resolveConsumer, progressMonitor));
062                        }
063                }
064        }
065        
066        @Override
067        protected void resolve(
068                        Action action, 
069                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
070                        ProgressMonitor progressMonitor) throws Exception {
071                super.resolve(action, context, progressMonitor);
072                
073                EList<Goal> goals = getTarget().getGoals();
074                if (!goals.isEmpty()) {
075                        String goalsGroupUUID = action.getUuid() + "-goals";
076                        Optional<Action> goalsActionOptional = action.getSections().stream()
077                                        .filter(a -> goalsGroupUUID.equals(a.getUuid()))
078                                        .findFirst();
079                        
080                        Action goalsAction = goalsActionOptional.get();
081                        goalsAction.getContent().add(renderList(goals, true, createGoalChildrenListProvider(), action, EngineeringPackage.Literals.PERSONA__GOALS, context, progressMonitor));
082                }
083        }
084
085        protected ContentProvider<Goal> createGoalChildrenListProvider() {
086                ContentProvider<Goal> goalChildrenListProvider = new ContentProvider<Goal>() {
087
088                        @Override
089                        public List<EObject> createContent(
090                                        Goal element, 
091                                        Action base, 
092                                        ETypedElement typedElement,
093                                        org.nasdanika.html.emf.EObjectActionResolver.Context context, 
094                                        ProgressMonitor progressMonitor) throws Exception {
095                                
096                                EList<Goal> children = element.getChildren();
097                                if (children.isEmpty()) {
098                                        return null;
099                                }
100                                return Collections.singletonList(renderList(children, true, this, base, EngineeringPackage.Literals.GOAL__CHILDREN, context, progressMonitor));
101                        }
102                        
103                };
104                return goalChildrenListProvider;
105        }
106                
107}
