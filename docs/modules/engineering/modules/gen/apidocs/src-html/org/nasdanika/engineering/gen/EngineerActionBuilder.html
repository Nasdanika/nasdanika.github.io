041        @Override
042        protected Action buildAction(
043                        Action action,
044                        BiConsumer<EObject,Action> registry, 
045                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
046                        ProgressMonitor progressMonitor) throws Exception {
047                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
048                
049                createDomainActions(action, registry, resolveConsumer, progressMonitor);                
050                createModuleActions(action, registry, resolveConsumer, progressMonitor);                
051                createIncrementActions(action, registry, resolveConsumer, progressMonitor);             
052                createPersonaActions(action, registry, resolveConsumer, progressMonitor);
053                createIssueGroupingActions(action, registry, resolveConsumer, progressMonitor);
054                createObjectiveActions(action, registry, resolveConsumer, progressMonitor);
055                createAssignmentsAction(action, registry, resolveConsumer, progressMonitor);            
056                
057                return action;
058        }

065                properties.add(EngineeringPackage.Literals.ENGINEER__REPRESENTS);
066
067                return properties;
068        }

087        protected void createDomainActions(
088                        Action action,
089                        BiConsumer<EObject,Action> registry, 
090                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
091                        ProgressMonitor progressMonitor) throws Exception {
092                List<Domain> domains = getTarget().getDomains();
093                if (!domains.isEmpty()) {
094                        Action group = AppFactory.eINSTANCE.createAction();
095                        group.setText("Domains");
096                        EList<EObject> children = group.getChildren();
097                        for (Domain domain: domains) {
098                                children.add(createChildAction(domain, registry, resolveConsumer, progressMonitor));
099                        }
100        
101                        action.getChildren().add(group);
102                }
103        }

105        /**
106         * Creates a list of actions for modules. 
107         * @param progressMonitor
108         * @return An empty list if there are no modules. A singleton list containing a grouping action containing module actions otherwise.
109         * @throws Exception 
110         */
111        protected void createModuleActions(
112                        Action action,
113                        BiConsumer<EObject,Action> registry, 
114                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
115                        ProgressMonitor progressMonitor) throws Exception {
116                List<org.nasdanika.engineering.Module> modules = getTarget().getModules();
117                if (!modules.isEmpty()) {
118                        Action group = AppFactory.eINSTANCE.createAction();
119                        Predicate<org.nasdanika.engineering.Module> isProduct = Product.class::isInstance;                      
120                        group.setText(modules.stream().anyMatch(isProduct.negate()) ? "Modules" : "Products");
121                        group.setIcon("fas fa-cubes");
122                        EList<EObject> children = group.getChildren();
123                        for (org.nasdanika.engineering.Module module: modules) {
124                                children.add(createChildAction(module, registry, resolveConsumer, progressMonitor));
125                        }
126        
127                        action.getChildren().add(group);
128                }
129        }

200        protected void createIssueGroupingActions(
201                        Action action,
202                        BiConsumer<EObject,Action> registry, 
203                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
204                        ProgressMonitor progressMonitor) throws Exception {
205                
206                T engineer = getTarget();
207                EList<IssueCategory> issueCategories = engineer.getIssueCategories();
208                EList<IssuePriority> issuePriorities = engineer.getIssuePriorities();
209                EList<IssueSeverity> issueSeverities = engineer.getIssueSeverities();
210                EList<IssueStatus> issueStatuses = engineer.getIssueStatuses();
211                
212                if (issueCategories.isEmpty() 
213                                && issuePriorities.isEmpty()
214                                && issueSeverities.isEmpty()
215                                && issueStatuses.isEmpty()) {
216                        return;
217                }
218
219                Action root = AppFactory.eINSTANCE.createAction();
220                root.setText("Issue");
221                root.setUuid(action.getUuid() + "-issue");
222                action.getNavigation().add(root);
223                
224                if (!issueCategories.isEmpty()) {
225                        Action group = AppFactory.eINSTANCE.createAction();
226                        group.setText("Categories");
227                        group.setUuid(root.getUuid() + "-categories");
228                        group.setLocation("issue-categories.html");
229                        EList<Action> groupAnonymous = group.getAnonymous();
230                        for (IssueCategory issueCategory: issueCategories) {
231                                groupAnonymous.add(createChildAction(issueCategory, registry, resolveConsumer, progressMonitor));
232                        }
233                
234                        root.getChildren().add(group);
235                }
236                
237                if (!issuePriorities.isEmpty()) {
238                        Action group = AppFactory.eINSTANCE.createAction();
239                        group.setText("Priorities");
240                        group.setUuid(root.getUuid() + "-priorities");
241                        group.setLocation("issue-priorities.html");
242                        EList<Action> groupAnonymous = group.getAnonymous();
243                        for (IssuePriority issuePriority: issuePriorities) {
244                                groupAnonymous.add(createChildAction(issuePriority, registry, resolveConsumer, progressMonitor));
245                        }
246                
247                        root.getChildren().add(group);
248                }
249                
250                if (!issueSeverities.isEmpty()) {
251                        Action group = AppFactory.eINSTANCE.createAction();
252                        group.setText("Severities");
253                        group.setUuid(root.getUuid() + "-severities");
254                        group.setLocation("issue-severities.html");
255                        EList<Action> groupAnonymous = group.getAnonymous();
256                        for (IssueSeverity issueSeverity: issueSeverities) {
257                                groupAnonymous.add(createChildAction(issueSeverity, registry, resolveConsumer, progressMonitor));
258                        }
259                
260                        root.getChildren().add(group);
261                }
262                
263                if (!issueStatuses.isEmpty()) {
264                        Action group = AppFactory.eINSTANCE.createAction();
265                        group.setText("Statuses");
266                        group.setUuid(root.getUuid() + "-statuses");
267                        group.setLocation("issue-statuses.html");
268                        EList<Action> groupAnonymous = group.getAnonymous();
269                        for (IssueStatus issueStatus: issueStatuses) {
270                                groupAnonymous.add(createChildAction(issueStatus, registry, resolveConsumer, progressMonitor));
271                        }
272                
273                        root.getChildren().add(group);
274                }
275        }

277        @Override
278        protected void resolve(
279                        Action action, 
280                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
281                        ProgressMonitor progressMonitor) throws Exception {
282                super.resolve(action, context, progressMonitor);
283                
284                T engineer = getTarget();
285                
286                EList<Objective> objectives = engineer.getObjectives();
287                if (!objectives.isEmpty()) {
288                        String objectivesGroupUUID = action.getUuid() + "-objectives";
289                        Optional<Action> objectivesActionOptional = action.getNavigation().stream()
290                                        .filter(Action.class::isInstance)                                       
291                                        .map(Action.class::cast)
292                                        .filter(a -> objectivesGroupUUID.equals(a.getUuid()))
293                                        .findFirst();
294                        
295                        Action objectivesAction = objectivesActionOptional.get();
296                        objectivesAction.getContent().add(renderList(objectives, true, null, objectivesAction, EngineeringPackage.Literals.ENGINEER__OBJECTIVES, context, progressMonitor)); // Table?
297                }
298                
299                EList<Persona> personas = engineer.getPersonas();
300                if (!personas.isEmpty()) {
301                        String personaGroupUUID = action.getUuid() + "-personas";
302                        Optional<Action> personasActionOptional = action.getNavigation().stream()
303                                        .filter(Action.class::isInstance)                                       
304                                        .map(Action.class::cast)
305                                        .filter(a -> personaGroupUUID.equals(a.getUuid()))
306                                        .findFirst();
307                        
308                        Action personasAction = personasActionOptional.get();
309                        personasAction.getContent().add(renderList(personas, true, null, personasAction, EngineeringPackage.Literals.ENGINEER__PERSONAS, context, progressMonitor)); // Table?
310                }
311                
312                String issueGroupUUID = action.getUuid() + "-issue";
313                Optional<Action> issueGroupOptional = action.getNavigation().stream()
314                                .filter(Action.class::isInstance)                                       
315                                .map(Action.class::cast)
316                                .filter(a -> issueGroupUUID.equals(a.getUuid()))
317                                .findFirst();
318                
319                if (issueGroupOptional.isPresent()) {
320                        Action issueGroup = issueGroupOptional.get();
321                        
322                        EList<IssueCategory> issueCategories = engineer.getIssueCategories();
323                        if (!issueCategories.isEmpty()) {
324                                String issueCategoriesUUID = issueGroup.getUuid() + "-categories";
325                                Optional<Action> issueCategoriesActionOptional = issueGroup.getChildren().stream()
326                                                .filter(Action.class::isInstance)                                       
327                                                .map(Action.class::cast)
328                                                .filter(a -> issueCategoriesUUID.equals(a.getUuid()))
329                                                .findFirst();
330                                                                
331                                Action issueCategoriesAction = issueCategoriesActionOptional.get();
332                                issueCategoriesAction.getContent().add(renderList(issueCategories, false, createIssueCategoryChildrenProvider(), issueCategoriesAction, EngineeringPackage.Literals.ENGINEER__ISSUE_CATEGORIES, context, progressMonitor)); // Table?
333                        }
334                        
335                        EList<IssuePriority> issuePriorities = engineer.getIssuePriorities();
336                        if (!issuePriorities.isEmpty()) {
337                                String issuePrioritiesUUID = issueGroup.getUuid() + "-priorities";
338                                Optional<Action> issuePrioritiesActionOptional = issueGroup.getChildren().stream()
339                                                .filter(Action.class::isInstance)                                       
340                                                .map(Action.class::cast)
341                                                .filter(a -> issuePrioritiesUUID.equals(a.getUuid()))
342                                                .findFirst();
343                                
344                                Action issuePrioritiesAction = issuePrioritiesActionOptional.get();
345                                issuePrioritiesAction.getContent().add(renderList(issuePriorities, false, null, issuePrioritiesAction, EngineeringPackage.Literals.ENGINEER__ISSUE_PRIORITIES, context, progressMonitor)); // Table?
346                        }
347                        
348                        EList<IssueSeverity> issueSeverities = engineer.getIssueSeverities();
349                        if (!issueSeverities.isEmpty()) {
350                                String issueSeveritiesUUID = issueGroup.getUuid() + "-severities";
351                                Optional<Action> issueSeveritiesActionOptional = issueGroup.getChildren().stream()
352                                                .filter(Action.class::isInstance)                                       
353                                                .map(Action.class::cast)
354                                                .filter(a -> issueSeveritiesUUID.equals(a.getUuid()))
355                                                .findFirst();
356                                
357                                Action issueSeveritiesAction = issueSeveritiesActionOptional.get();
358                                issueSeveritiesAction.getContent().add(renderList(issueSeverities, false, null, issueSeveritiesAction, EngineeringPackage.Literals.ENGINEER__ISSUE_SEVERITIES, context, progressMonitor)); // Table?
359                        }
360                        
361                        EList<IssueStatus> issueStatuses = engineer.getIssueStatuses();
362                        if (!issueStatuses.isEmpty()) {
363                                String issueStatusesUUID = issueGroup.getUuid() + "-statuses";
364                                Optional<Action> issueStatusesActionOptional = issueGroup.getChildren().stream()
365                                                .filter(Action.class::isInstance)                                       
366                                                .map(Action.class::cast)
367                                                .filter(a -> issueStatusesUUID.equals(a.getUuid()))
368                                                .findFirst();
369                                
370                                Action issueStatusesAction = issueStatusesActionOptional.get();
371                                issueStatusesAction.getContent().add(renderList(issueStatuses, false, null, issueStatusesAction, EngineeringPackage.Literals.ENGINEER__ISSUE_STATUSES, context, progressMonitor)); // Table?
372                        }
373                }               
374                
375                // Assignments          
376                EList<Endeavor> assignments = getTarget().getAssignments();
377                if (!assignments.isEmpty()) {
378                        String assignmentsGroupUUID = action.getUuid() + "-assignments";
379                        Optional<Action> assignmentsActionOptional = action.getNavigation().stream()
380                                        .filter(Action.class::isInstance)                                       
381                                        .map(Action.class::cast)
382                                        .filter(a -> assignmentsGroupUUID.equals(a.getUuid()))
383                                        .findFirst();
384                        Action assignmentsAction = assignmentsActionOptional.get();
385                        Map<EClass, List<Endeavor>> groupedAssignments = Util.groupBy(assignments, EObject::eClass);
386                        
387                        List<Endeavor> features = groupedAssignments.get(EngineeringPackage.Literals.FEATURE);
388                        if (features != null && !features.isEmpty()) {
389                                Table featuresTable = buildTable(
390                                                features, 
391                                                action, 
392                                                EngineeringPackage.Literals.ENGINEER__ASSIGNMENTS, 
393                                                context, 
394                                                progressMonitor,
395                                                createColumnBuilder("Feature"),
396                                                createColumnBuilder(EcorePackage.Literals.EOBJECT___ECONTAINER, "Product"),
397                                                createColumnBuilder(EngineeringPackage.Literals.ENGINEERED_CAPABILITY__RELEASES),
398                                                createColumnBuilder(NcorePackage.Literals.PERIOD__START),
399                                                createColumnBuilder(NcorePackage.Literals.PERIOD__END),
400                                                createColumnBuilder(NcorePackage.Literals.PERIOD__DURATION),
401                                                createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
402                                
403                                Action section = AppFactory.eINSTANCE.createAction();
404                                section.setText("Features");
405                                section.getContent().add(featuresTable);
406                        
407                                assignmentsAction.getSections().add(section);
408                        }
409                        
410                        List<Endeavor> increments = groupedAssignments.get(EngineeringPackage.Literals.INCREMENT);
411                        if (increments != null && !increments.isEmpty()) {
412                                Table incrementsTable = buildTable(
413                                                increments, 
414                                                action, 
415                                                EngineeringPackage.Literals.ENGINEER__ASSIGNMENTS, 
416                                                context, 
417                                                progressMonitor,
418                                                createColumnBuilder("Increment"),
419                                                createColumnBuilder(NcorePackage.Literals.PERIOD__START),
420                                                createColumnBuilder(NcorePackage.Literals.PERIOD__END),
421                                                createColumnBuilder(NcorePackage.Literals.PERIOD__DURATION),
422                                                createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
423                                
424                                Action section = AppFactory.eINSTANCE.createAction();
425                                section.setText("Increments");
426                                section.getContent().add(incrementsTable);
427                        
428                                assignmentsAction.getSections().add(section);
429                        }
430                        
431                        List<Endeavor> issues = groupedAssignments.get(EngineeringPackage.Literals.ISSUE);
432                        if (issues != null && !issues.isEmpty()) {
433                                Table issuesTable = buildTable(
434                                                issues, 
435                                                action, 
436                                                EngineeringPackage.Literals.ENGINEER__ASSIGNMENTS, 
437                                                context, 
438                                                progressMonitor,
439                                                createColumnBuilder("Issue"),
440                                                createColumnBuilder(EcorePackage.Literals.EOBJECT___ECONTAINER, "Module"),
441                                                createColumnBuilder(EngineeringPackage.Literals.ISSUE__INCREMENT),
442                                                createColumnBuilder(NcorePackage.Literals.PERIOD__START),
443                                                createColumnBuilder(NcorePackage.Literals.PERIOD__END),
444                                                createColumnBuilder(NcorePackage.Literals.PERIOD__DURATION),
445                                                createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
446                                
447                                Action section = AppFactory.eINSTANCE.createAction();
448                                section.setText("Issues");
449                                section.getContent().add(issuesTable);
450                        
451                                assignmentsAction.getSections().add(section);
452                        }
453                        
454                        List<Endeavor> releases = groupedAssignments.get(EngineeringPackage.Literals.RELEASE);
455                        if (releases != null && !releases.isEmpty()) {
456                                Table releasesTable = buildTable(
457                                                releases, 
458                                                action, 
459                                                EngineeringPackage.Literals.ENGINEER__ASSIGNMENTS, 
460                                                context, 
461                                                progressMonitor,
462                                                createColumnBuilder("Release"),
463                                                createColumnBuilder(EcorePackage.Literals.EOBJECT___ECONTAINER, "Product"),
464                                                createColumnBuilder(EngineeringPackage.Literals.RELEASE__INCREMENT),
465                                                createColumnBuilder(NcorePackage.Literals.PERIOD__START),
466                                                createColumnBuilder(NcorePackage.Literals.PERIOD__END),
467                                                createColumnBuilder(NcorePackage.Literals.PERIOD__DURATION),
468                                                createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
469                                
470                                Action section = AppFactory.eINSTANCE.createAction();
471                                section.setText("Releases");
472                                section.getContent().add(releasesTable);
473                        
474                                assignmentsAction.getSections().add(section);
475                        }                       
476                }               
477        }

479        private ContentProvider<IssueCategory> createIssueCategoryChildrenProvider() {
480                return new ContentProvider<IssueCategory>() {
481
482                        @Override
483                        public List<EObject> createContent(
484                                        IssueCategory element, 
485                                        Action base, 
486                                        ETypedElement typedElement,
487                                        org.nasdanika.html.emf.EObjectActionResolver.Context context, 
488                                        ProgressMonitor progressMonitor) throws Exception {
489                                
490                                EList<IssueCategory> children = element.getChildren();
491                                if (children.isEmpty()) {
492                                        return null;
493                                }
494                                return Collections.singletonList(renderList(children, true, this, base, EngineeringPackage.Literals.ISSUE_CATEGORY__CHILDREN, context, progressMonitor));
495                        }
496                        
497                };
498        }
