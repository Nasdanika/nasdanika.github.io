042        @Override
043        protected Action buildAction(
044                        Action action,
045                        BiConsumer<EObject,Action> registry, 
046                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
047                        ProgressMonitor progressMonitor) throws Exception {
048                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
049                
050                createDomainActions(action, registry, resolveConsumer, progressMonitor);                
051                createModuleActions(action, registry, resolveConsumer, progressMonitor);                
052                createIncrementActions(action, registry, resolveConsumer, progressMonitor);             
053                createPersonaActions(action, registry, resolveConsumer, progressMonitor);
054                createIssueGroupingActions(action, registry, resolveConsumer, progressMonitor);
055                createObjectiveActions(action, registry, resolveConsumer, progressMonitor);
056                createAssignmentsAction(action, registry, resolveConsumer, progressMonitor);            
057                createEngineeredElementStatusAction(action, registry, resolveConsumer, progressMonitor);                
058                
059                return action;
060        }

072        protected void createEngineeredElementStatusAction(
073                        Action action,
074                        BiConsumer<EObject,Action> registry, 
075                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
076                        ProgressMonitor progressMonitor) throws Exception {
077                
078                T engineer = getTarget();
079                EList<EngineeredElementStatus> engineeredElementStatuses = engineer.getEngineeredElementStatuses();
080                if (engineeredElementStatuses.isEmpty()) {
081                        return;
082                }
083
084                Action group = AppFactory.eINSTANCE.createAction();
085                group.setText("Engineered Element Statuses");
086                group.setUuid(action.getUuid() + "-engineered-element-statuses");
087                group.setLocation("engineered-element-statuses.html");
088                EList<Action> groupAnonymous = group.getAnonymous();
089                for (EngineeredElementStatus status: engineeredElementStatuses) {
090                        groupAnonymous.add(createChildAction(status, registry, resolveConsumer, progressMonitor));
091                }
092        
093                action.getNavigation().add(group);
094        }

114                        Action action,
115                        BiConsumer<EObject,Action> registry, 
116                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer,

131        /**
132         * Creates a list of actions for modules. 
133         * @param progressMonitor
134         * @return An empty list if there are no modules. A singleton list containing a grouping action containing module actions otherwise.
135         * @throws Exception 
136         */
137        protected void createModuleActions(
138                        Action action,
139                        BiConsumer<EObject,Action> registry, 
140                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
141                        ProgressMonitor progressMonitor) throws Exception {
142                List<org.nasdanika.engineering.Module> modules = getTarget().getModules();
143                if (!modules.isEmpty()) {
144                        Action group = AppFactory.eINSTANCE.createAction();
145                        Predicate<org.nasdanika.engineering.Module> isProduct = Product.class::isInstance;                      
146                        group.setText(modules.stream().anyMatch(isProduct.negate()) ? "Modules" : "Products");
147                        group.setIcon("fas fa-cubes");
148                        EList<EObject> children = group.getChildren();
149                        for (org.nasdanika.engineering.Module module: modules) {
150                                children.add(createChildAction(module, registry, resolveConsumer, progressMonitor));
151                        }
152        
153                        action.getChildren().add(group);
154                }
155        }

226        protected void createIssueGroupingActions(
227                        Action action,
228                        BiConsumer<EObject,Action> registry, 
229                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
230                        ProgressMonitor progressMonitor) throws Exception {
231                
232                T engineer = getTarget();
233                EList<IssueCategory> issueCategories = engineer.getIssueCategories();
234                EList<IssuePriority> issuePriorities = engineer.getIssuePriorities();
235                EList<IssueSeverity> issueSeverities = engineer.getIssueSeverities();
236                EList<IssueStatus> issueStatuses = engineer.getIssueStatuses();
237                
238                if (issueCategories.isEmpty() 
239                                && issuePriorities.isEmpty()
240                                && issueSeverities.isEmpty()
241                                && issueStatuses.isEmpty()) {
242                        return;
243                }
244
245                Action root = AppFactory.eINSTANCE.createAction();
246                root.setText("Issue");
247                root.setUuid(action.getUuid() + "-issue");
248                action.getNavigation().add(root);
249                
250                if (!issueCategories.isEmpty()) {
251                        Action group = AppFactory.eINSTANCE.createAction();
252                        group.setText("Categories");
253                        group.setUuid(root.getUuid() + "-categories");
254                        group.setLocation("issue-categories.html");
255                        EList<Action> groupAnonymous = group.getAnonymous();
256                        for (IssueCategory issueCategory: issueCategories) {
257                                groupAnonymous.add(createChildAction(issueCategory, registry, resolveConsumer, progressMonitor));
258                        }
259                
260                        root.getChildren().add(group);
261                }
262                
263                if (!issuePriorities.isEmpty()) {
264                        Action group = AppFactory.eINSTANCE.createAction();
265                        group.setText("Priorities");
266                        group.setUuid(root.getUuid() + "-priorities");
267                        group.setLocation("issue-priorities.html");
268                        EList<Action> groupAnonymous = group.getAnonymous();
269                        for (IssuePriority issuePriority: issuePriorities) {
270                                groupAnonymous.add(createChildAction(issuePriority, registry, resolveConsumer, progressMonitor));
271                        }
272                
273                        root.getChildren().add(group);
274                }
275                
276                if (!issueSeverities.isEmpty()) {
277                        Action group = AppFactory.eINSTANCE.createAction();
278                        group.setText("Severities");
279                        group.setUuid(root.getUuid() + "-severities");
280                        group.setLocation("issue-severities.html");
281                        EList<Action> groupAnonymous = group.getAnonymous();
282                        for (IssueSeverity issueSeverity: issueSeverities) {
283                                groupAnonymous.add(createChildAction(issueSeverity, registry, resolveConsumer, progressMonitor));
284                        }
285                
286                        root.getChildren().add(group);
287                }
288                
289                if (!issueStatuses.isEmpty()) {
290                        Action group = AppFactory.eINSTANCE.createAction();
291                        group.setText("Statuses");
292                        group.setUuid(root.getUuid() + "-statuses");
293                        group.setLocation("issue-statuses.html");
294                        EList<Action> groupAnonymous = group.getAnonymous();
295                        for (IssueStatus issueStatus: issueStatuses) {
296                                groupAnonymous.add(createChildAction(issueStatus, registry, resolveConsumer, progressMonitor));
297                        }
298                
299                        root.getChildren().add(group);
300                }
301        }

303        @Override
304        protected void resolve(
305                        Action action, 
306                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
307                        ProgressMonitor progressMonitor) throws Exception {
308                super.resolve(action, context, progressMonitor);
309                
310                T engineer = getTarget();
311                
312                EList<Objective> objectives = engineer.getObjectives();
313                if (!objectives.isEmpty()) {
314                        String objectivesGroupUUID = action.getUuid() + "-objectives";
315                        Optional<Action> objectivesActionOptional = action.getNavigation().stream()
316                                        .filter(Action.class::isInstance)                                       
317                                        .map(Action.class::cast)
318                                        .filter(a -> objectivesGroupUUID.equals(a.getUuid()))
319                                        .findFirst();
320                        
321                        Action objectivesAction = objectivesActionOptional.get();
322                        objectivesAction.getContent().add(renderList(objectives, true, null, objectivesAction, EngineeringPackage.Literals.ENGINEER__OBJECTIVES, context, progressMonitor)); // Table?
323                }
324                
325                EList<Persona> personas = engineer.getPersonas();
326                if (!personas.isEmpty()) {
327                        String personaGroupUUID = action.getUuid() + "-personas";
328                        Optional<Action> personasActionOptional = action.getNavigation().stream()
329                                        .filter(Action.class::isInstance)                                       
330                                        .map(Action.class::cast)
331                                        .filter(a -> personaGroupUUID.equals(a.getUuid()))
332                                        .findFirst();
333                        
334                        Action personasAction = personasActionOptional.get();
335                        personasAction.getContent().add(renderList(personas, true, null, personasAction, EngineeringPackage.Literals.ENGINEER__PERSONAS, context, progressMonitor)); // Table?
336                }
337                
338                String issueGroupUUID = action.getUuid() + "-issue";
339                Optional<Action> issueGroupOptional = action.getNavigation().stream()
340                                .filter(Action.class::isInstance)                                       
341                                .map(Action.class::cast)
342                                .filter(a -> issueGroupUUID.equals(a.getUuid()))
343                                .findFirst();
344                
345                if (issueGroupOptional.isPresent()) {
346                        Action issueGroup = issueGroupOptional.get();
347                        
348                        EList<IssueCategory> issueCategories = engineer.getIssueCategories();
349                        if (!issueCategories.isEmpty()) {
350                                String issueCategoriesUUID = issueGroup.getUuid() + "-categories";
351                                Optional<Action> issueCategoriesActionOptional = issueGroup.getChildren().stream()
352                                                .filter(Action.class::isInstance)                                       
353                                                .map(Action.class::cast)
354                                                .filter(a -> issueCategoriesUUID.equals(a.getUuid()))
355                                                .findFirst();
356                                                                
357                                Action issueCategoriesAction = issueCategoriesActionOptional.get();
358                                issueCategoriesAction.getContent().add(renderList(issueCategories, false, createIssueCategoryChildrenProvider(), issueCategoriesAction, EngineeringPackage.Literals.ENGINEER__ISSUE_CATEGORIES, context, progressMonitor)); // Table?
359                        }
360                        
361                        EList<IssuePriority> issuePriorities = engineer.getIssuePriorities();
362                        if (!issuePriorities.isEmpty()) {
363                                String issuePrioritiesUUID = issueGroup.getUuid() + "-priorities";
364                                Optional<Action> issuePrioritiesActionOptional = issueGroup.getChildren().stream()
365                                                .filter(Action.class::isInstance)                                       
366                                                .map(Action.class::cast)
367                                                .filter(a -> issuePrioritiesUUID.equals(a.getUuid()))
368                                                .findFirst();
369                                
370                                Action issuePrioritiesAction = issuePrioritiesActionOptional.get();
371                                issuePrioritiesAction.getContent().add(renderList(issuePriorities, false, null, issuePrioritiesAction, EngineeringPackage.Literals.ENGINEER__ISSUE_PRIORITIES, context, progressMonitor)); // Table?
372                        }
373                        
374                        EList<IssueSeverity> issueSeverities = engineer.getIssueSeverities();
375                        if (!issueSeverities.isEmpty()) {
376                                String issueSeveritiesUUID = issueGroup.getUuid() + "-severities";
377                                Optional<Action> issueSeveritiesActionOptional = issueGroup.getChildren().stream()
378                                                .filter(Action.class::isInstance)                                       
379                                                .map(Action.class::cast)
380                                                .filter(a -> issueSeveritiesUUID.equals(a.getUuid()))
381                                                .findFirst();
382                                
383                                Action issueSeveritiesAction = issueSeveritiesActionOptional.get();
384                                issueSeveritiesAction.getContent().add(renderList(issueSeverities, false, null, issueSeveritiesAction, EngineeringPackage.Literals.ENGINEER__ISSUE_SEVERITIES, context, progressMonitor)); // Table?
385                        }
386                        
387                        EList<IssueStatus> issueStatuses = engineer.getIssueStatuses();
388                        if (!issueStatuses.isEmpty()) {
389                                String issueStatusesUUID = issueGroup.getUuid() + "-statuses";
390                                Optional<Action> issueStatusesActionOptional = issueGroup.getChildren().stream()
391                                                .filter(Action.class::isInstance)                                       
392                                                .map(Action.class::cast)
393                                                .filter(a -> issueStatusesUUID.equals(a.getUuid()))
394                                                .findFirst();
395                                
396                                Action issueStatusesAction = issueStatusesActionOptional.get();
397                                issueStatusesAction.getContent().add(renderList(issueStatuses, false, null, issueStatusesAction, EngineeringPackage.Literals.ENGINEER__ISSUE_STATUSES, context, progressMonitor)); // Table?
398                        }
399                }
400                
401                // EngineeredElement statuses           
402                EList<EngineeredElementStatus> engineeredElementStatuses = engineer.getEngineeredElementStatuses();
403                if (!engineeredElementStatuses.isEmpty()) {
404                        String engineeredElementStatusesUUID = action.getUuid() + "-engineered-element-statuses";
405                        Optional<Action> engineeredElementStatusesActionOptional = action.getNavigation().stream()
406                                        .filter(Action.class::isInstance)                                       
407                                        .map(Action.class::cast)
408                                        .filter(a -> engineeredElementStatusesUUID.equals(a.getUuid()))
409                                        .findFirst();
410                                                        
411                        Action engineeredElementStatusesAction = engineeredElementStatusesActionOptional.get();
412                        engineeredElementStatusesAction.getContent().add(renderList(engineeredElementStatuses, false, createEngineeredElementStatusChildrenProvider(), engineeredElementStatusesAction, EngineeringPackage.Literals.ENGINEER__ENGINEERED_ELEMENT_STATUSES, context, progressMonitor)); // Table?
413                }
414                
415                // Assignments          
416                EList<Endeavor> assignments = getTarget().getAssignments();
417                if (!assignments.isEmpty()) {
418                        String assignmentsGroupUUID = action.getUuid() + "-assignments";
419                        Optional<Action> assignmentsActionOptional = action.getNavigation().stream()
420                                        .filter(Action.class::isInstance)                                       
421                                        .map(Action.class::cast)
422                                        .filter(a -> assignmentsGroupUUID.equals(a.getUuid()))
423                                        .findFirst();
424                        Action assignmentsAction = assignmentsActionOptional.get();
425                        Map<EClass, List<Endeavor>> groupedAssignments = Util.groupBy(assignments, EObject::eClass);
426                        
427                        List<Endeavor> features = groupedAssignments.get(EngineeringPackage.Literals.FEATURE);
428                        if (features != null && !features.isEmpty()) {
429                                Table featuresTable = buildTable(
430                                                features, 
431                                                action, 
432                                                EngineeringPackage.Literals.ENGINEER__ASSIGNMENTS, 
433                                                context, 
434                                                progressMonitor,
435                                                createColumnBuilder("Feature"),
436                                                createColumnBuilder(EcorePackage.Literals.EOBJECT___ECONTAINER, "Product"),
437                                                createColumnBuilder(EngineeringPackage.Literals.ENGINEERED_CAPABILITY__RELEASES),
438                                                createColumnBuilder(NcorePackage.Literals.PERIOD__START),
439                                                createColumnBuilder(NcorePackage.Literals.PERIOD__END),
440                                                createColumnBuilder(NcorePackage.Literals.PERIOD__DURATION),
441                                                createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
442                                
443                                Action section = AppFactory.eINSTANCE.createAction();
444                                section.setText("Features");
445                                section.getContent().add(featuresTable);
446                        
447                                assignmentsAction.getSections().add(section);
448                        }
449                        
450                        List<Endeavor> increments = groupedAssignments.get(EngineeringPackage.Literals.INCREMENT);
451                        if (increments != null && !increments.isEmpty()) {
452                                Table incrementsTable = buildTable(
453                                                increments, 
454                                                action, 
455                                                EngineeringPackage.Literals.ENGINEER__ASSIGNMENTS, 
456                                                context, 
457                                                progressMonitor,
458                                                createColumnBuilder("Increment"),
459                                                createColumnBuilder(NcorePackage.Literals.PERIOD__START),
460                                                createColumnBuilder(NcorePackage.Literals.PERIOD__END),
461                                                createColumnBuilder(NcorePackage.Literals.PERIOD__DURATION),
462                                                createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
463                                
464                                Action section = AppFactory.eINSTANCE.createAction();
465                                section.setText("Increments");
466                                section.getContent().add(incrementsTable);
467                        
468                                assignmentsAction.getSections().add(section);
469                        }
470                        
471                        List<Endeavor> issues = groupedAssignments.get(EngineeringPackage.Literals.ISSUE);
472                        if (issues != null && !issues.isEmpty()) {
473                                Table issuesTable = buildTable(
474                                                issues, 
475                                                action, 
476                                                EngineeringPackage.Literals.ENGINEER__ASSIGNMENTS, 
477                                                context, 
478                                                progressMonitor,
479                                                createColumnBuilder("Issue"),
480                                                createColumnBuilder(EcorePackage.Literals.EOBJECT___ECONTAINER, "Module"),
481                                                createColumnBuilder(EngineeringPackage.Literals.ISSUE__INCREMENT),
482                                                createColumnBuilder(NcorePackage.Literals.PERIOD__START),
483                                                createColumnBuilder(NcorePackage.Literals.PERIOD__END),
484                                                createColumnBuilder(NcorePackage.Literals.PERIOD__DURATION),
485                                                createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
486                                
487                                Action section = AppFactory.eINSTANCE.createAction();
488                                section.setText("Issues");
489                                section.getContent().add(issuesTable);
490                        
491                                assignmentsAction.getSections().add(section);
492                        }
493                        
494                        List<Endeavor> releases = groupedAssignments.get(EngineeringPackage.Literals.RELEASE);
495                        if (releases != null && !releases.isEmpty()) {
496                                Table releasesTable = buildTable(
497                                                releases, 
498                                                action, 
499                                                EngineeringPackage.Literals.ENGINEER__ASSIGNMENTS, 
500                                                context, 
501                                                progressMonitor,
502                                                createColumnBuilder("Release"),
503                                                createColumnBuilder(EcorePackage.Literals.EOBJECT___ECONTAINER, "Product"),
504                                                createColumnBuilder(EngineeringPackage.Literals.RELEASE__INCREMENT),
505                                                createColumnBuilder(NcorePackage.Literals.PERIOD__START),
506                                                createColumnBuilder(NcorePackage.Literals.PERIOD__END),
507                                                createColumnBuilder(NcorePackage.Literals.PERIOD__DURATION),
508                                                createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION));
509                                
510                                Action section = AppFactory.eINSTANCE.createAction();
511                                section.setText("Releases");
512                                section.getContent().add(releasesTable);
513                        
514                                assignmentsAction.getSections().add(section);
515                        }                       
516                }               
517        }

519        private ContentProvider<EngineeredElementStatus> createEngineeredElementStatusChildrenProvider() {
520                return new ContentProvider<EngineeredElementStatus>() {
521
522                        @Override
523                        public List<EObject> createContent(
524                                        EngineeredElementStatus element, 
525                                        Action base, 
526                                        ETypedElement typedElement,
527                                        org.nasdanika.html.emf.EObjectActionResolver.Context context, 
528                                        ProgressMonitor progressMonitor) throws Exception {
529                                
530                                EList<EngineeredElementStatus> children = element.getChildren();
531                                if (children.isEmpty()) {
532                                        return null;
533                                }
534                                return Collections.singletonList(renderList(children, true, this, base, EngineeringPackage.Literals.ENGINEERED_ELEMENT_STATUS__CHILDREN, context, progressMonitor));
535                        }
536                        
537                };
538        }
