001package org.nasdanika.engineering.gen;
002
003import java.util.List;
004import java.util.function.BiConsumer;
005
006import org.eclipse.emf.common.util.EList;
007import org.eclipse.emf.ecore.EObject;
008import org.eclipse.emf.ecore.ETypedElement;
009import org.nasdanika.common.Context;
010import org.nasdanika.common.ProgressMonitor;
011import org.nasdanika.common.Util;
012import org.nasdanika.engineering.EngineeringPackage;
013import org.nasdanika.html.model.app.Action;
014
015public class ModuleActionBuilder<T extends org.nasdanika.engineering.Module> extends EngineeredElementActionBuilder<T> {
016        
017        public ModuleActionBuilder(T target, Context context) {
018                super(target, context);         
019        }
020                
021        @Override
022        protected Action buildAction(
023                        Action action,
024                        BiConsumer<EObject,Action> registry, 
025                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
026                        ProgressMonitor progressMonitor) throws Exception {
027
028                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
029                if (Util.isBlank(action.getIcon())) {
030                        action.setIcon("fas fa-cube");          
031                }
032                
033                createModuleActions(action, registry, resolveConsumer, progressMonitor);
034                
035                return action;
036        }
037        
038        @Override
039        protected List<ETypedElement> getProperties() {
040                List<ETypedElement> properties = super.getProperties();
041                properties.add(EngineeringPackage.Literals.MODULE__DEPENDENCIES);
042                properties.add(EngineeringPackage.Literals.MODULE__DEPENDANTS);
043                
044                return properties;
045        }
046        
047        /**
048         * Creates a list of actions for sub-packages. 
049         * @param progressMonitor
050         * @return An empty list if there are no sub-packages. A singleton list containing a grouping action containing sub-package actions otherwise.
051         * @throws Exception 
052         */
053        protected void createModuleActions(
054                        Action action,
055                        BiConsumer<EObject,Action> registry, 
056                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
057                        ProgressMonitor progressMonitor) throws Exception {
058                EList<EObject> children = action.getChildren();
059                for (org.nasdanika.engineering.Module module: getTarget().getModules()) {
060                        children.add(createChildAction(module, registry, resolveConsumer, progressMonitor));
061                }
062        }       
063        
064}
