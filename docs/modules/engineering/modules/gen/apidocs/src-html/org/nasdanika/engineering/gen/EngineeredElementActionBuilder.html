001package org.nasdanika.engineering.gen;
002
003import java.util.Collections;
004import java.util.List;
005import java.util.Optional;
006import java.util.function.BiConsumer;
007
008import org.eclipse.emf.common.util.EList;
009import org.eclipse.emf.ecore.EObject;
010import org.eclipse.emf.ecore.ETypedElement;
011import org.nasdanika.common.Context;
012import org.nasdanika.common.ProgressMonitor;
013import org.nasdanika.engineering.EngineeredElement;
014import org.nasdanika.engineering.EngineeringPackage;
015import org.nasdanika.engineering.Forum;
016import org.nasdanika.engineering.Issue;
017import org.nasdanika.engineering.Principle;
018import org.nasdanika.engineering.Topic;
019import org.nasdanika.html.model.app.Action;
020import org.nasdanika.html.model.app.AppFactory;
021import org.nasdanika.html.model.html.Tag;
022import org.nasdanika.ncore.NcorePackage;
023
024public class EngineeredElementActionBuilder<T extends EngineeredElement> extends ForumActionBuilder<T> {
025        
026        public EngineeredElementActionBuilder(T target, Context context) {
027                super(target, context);         
028        }
029        
030        protected List<ETypedElement> getProperties() {
031                List<ETypedElement> properties = super.getProperties();
032                properties.add(EngineeringPackage.Literals.ENGINEERED_ELEMENT__OWNERS);
033                properties.add(EngineeringPackage.Literals.ENGINEERED_ELEMENT__EXPERTS);
034                return properties;
035        }       
036        
037        @Override
038        protected Action buildAction(
039                        Action action,
040                        BiConsumer<EObject,Action> registry, 
041                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
042                        ProgressMonitor progressMonitor) throws Exception {
043                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
044                
045                createIssuesAction(action, registry, resolveConsumer, progressMonitor);         
046                createAllIssuesAction(action, registry, resolveConsumer, progressMonitor);              
047                createPrinciplesActions(action, registry, resolveConsumer, progressMonitor);
048                createDiscussionAction(action, registry, resolveConsumer, progressMonitor);             
049                
050                return action;
051        }
052        
053        protected void createIssuesAction(
054                        Action action,
055                        BiConsumer<EObject,Action> registry, 
056                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
057                        ProgressMonitor progressMonitor) throws Exception {
058                
059                List<Issue> issues = getTarget().getIssues();
060                if (!issues.isEmpty()) {
061                        Action group = AppFactory.eINSTANCE.createAction();
062                        group.setText("Issues");
063                        group.setUuid(action.getUuid() + "-issues");
064                        group.setLocation("issues.html");
065                        //group.setNotification(String.valueOf(issues.size()));
066                        // TODO - icon, ...
067                        EList<Action> groupAnonymous = group.getAnonymous();
068                        for (Issue issue: issues) {
069                                groupAnonymous.add(createChildAction(issue, registry, resolveConsumer, progressMonitor));
070                        }
071                
072                        action.getNavigation().add(group);
073                }
074                
075        }
076        
077        protected void createAllIssuesAction(
078                        Action action,
079                        BiConsumer<EObject,Action> registry, 
080                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
081                        ProgressMonitor progressMonitor) throws Exception {
082                
083                List<Issue> allIssues = getTarget().getAllIssues();
084                if (!allIssues.isEmpty()) {
085                        Action group = AppFactory.eINSTANCE.createAction();
086                        group.setText("All Issues");
087                        group.setUuid(action.getUuid() + "-all-issues");
088                        group.setLocation("all-issues.html");
089                        action.getNavigation().add(group);
090                }
091                
092        }
093        
094        protected void createDiscussionAction(
095                        Action action,
096                        BiConsumer<EObject,Action> registry, 
097                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
098                        ProgressMonitor progressMonitor) throws Exception {
099                
100                List<Forum> discussion = getTarget().getDiscussion();
101                EList<Topic> topics = getTarget().getTopics();
102                if (!discussion.isEmpty() || !topics.isEmpty()) {
103                        Action group = AppFactory.eINSTANCE.createAction();
104                        group.setText("Discussion");
105                        group.setIcon("far fa-comments");
106                        group.setUuid(action.getUuid() + "-discussion");
107                        group.setLocation("discussion.html");
108                        action.getNavigation().add(group);
109                }               
110        }
111                
112        @Override
113        protected void resolve(
114                        Action action, 
115                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
116                        ProgressMonitor progressMonitor) throws Exception {
117                super.resolve(action, context, progressMonitor);
118                
119                T engineeredElement = getTarget();
120                EList<Issue> issues = engineeredElement.getIssues();
121                if (!issues.isEmpty()) {
122                        String issuesGroupUUID = action.getUuid() + "-issues";
123                        Optional<Action> issuesActionOptional = action.getNavigation().stream()
124                                        .filter(Action.class::isInstance)                                       
125                                        .map(Action.class::cast)
126                                        .filter(a -> issuesGroupUUID.equals(a.getUuid()))
127                                        .findFirst();
128                        
129                        Action issuesAction = issuesActionOptional.get();
130                        issuesAction.getContent().add(renderList(issues, true, createIssueChildrenProvider(), issuesAction, EngineeringPackage.Literals.ENGINEERED_ELEMENT__ISSUES, context, progressMonitor)); // Table?
131                }
132                
133                EList<Issue> allIssues = engineeredElement.getAllIssues();
134                if (!allIssues.isEmpty()) {
135                        String allIssuesGroupUUID = action.getUuid() + "-all-issues";
136                        Optional<Action> allIssuesActionOptional = action.getNavigation().stream()
137                                        .filter(Action.class::isInstance)                                       
138                                        .map(Action.class::cast)
139                                        .filter(a -> allIssuesGroupUUID.equals(a.getUuid()))
140                                        .findFirst();
141                        Action allIssuesAction = allIssuesActionOptional.get();
142                        Tag allIssuesTable = buildDynamicTable(
143                                        allIssues, 
144                                        action, 
145                                        EngineeringPackage.Literals.ENGINEERED_ELEMENT__ALL_ISSUES,
146                                        "engineered-element--all-issues",
147                                        null,
148                                        context, 
149                                        progressMonitor,
150                                        createDynamicColumnBuilder("Issue", true),
151                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__TARGET, true),
152                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__ASSIGNEE, true),
153                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__CATEGORIES, true),
154                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__PRIORITY, true),
155                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__SEVERITY, true),
156                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__STATUS, true),
157                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__WORKABLE, true),
158                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__BENEFIT, false),
159                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__COST, false),
160                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__EFFORT, false),
161                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__REMAINING_COST, false),
162                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__REMAINING_EFFORT, false),
163                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION, true),
164                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ENGINEERED_CAPABILITY__RELEASES, true),
165                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__CONTRIBUTES_TO, true),
166                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__INCREMENT, true),
167                                        createDynamicColumnBuilder(NcorePackage.Literals.PERIOD__START, true),
168                                        createDynamicColumnBuilder(NcorePackage.Literals.PERIOD__END, true),
169                                        createDynamicColumnBuilder(NcorePackage.Literals.PERIOD__DURATION, false));
170                        
171                        allIssuesAction.getContent().add(allIssuesTable);
172                }
173                
174                EList<Principle> principles = getTarget().getPrinciples();
175                if (!principles.isEmpty()) {
176                        String principlesGroupUUID = action.getUuid() + "-principles";
177                        Optional<Action> principlesActionOptional = action.getNavigation().stream()
178                                        .filter(Action.class::isInstance)
179                                        .map(Action.class::cast)
180                                        .filter(a -> principlesGroupUUID.equals(a.getUuid()))
181                                        .findFirst();
182                        
183                        Action principlesAction = principlesActionOptional.get();
184                        principlesAction.getContent().add(renderList(principles, true, createPrincipleChildrenListProvider(), action, EngineeringPackage.Literals.ENGINEERED_ELEMENT__PRINCIPLES, context, progressMonitor));
185                }
186        }
187
188        protected ContentProvider<Principle> createPrincipleChildrenListProvider() {
189                ContentProvider<Principle> goalChildrenListProvider = new ContentProvider<Principle>() {
190
191                        @Override
192                        public List<EObject> createContent(
193                                        Principle element, 
194                                        Action base, 
195                                        ETypedElement typedElement,
196                                        org.nasdanika.html.emf.EObjectActionResolver.Context context, 
197                                        ProgressMonitor progressMonitor) throws Exception {
198                                
199                                EList<Principle> children = element.getChildren();
200                                if (children.isEmpty()) {
201                                        return null;
202                                }
203                                return Collections.singletonList(renderList(children, true, this, base, EngineeringPackage.Literals.PRINCIPLE__CHILDREN, context, progressMonitor));
204                        }
205                        
206                };
207                return goalChildrenListProvider;
208        }
209        
210        private ContentProvider<Issue> createIssueChildrenProvider() {
211                return new ContentProvider<Issue>() {
212        
213                        @Override
214                        public List<EObject> createContent(
215                                        Issue element, 
216                                        Action base, 
217                                        ETypedElement typedElement,
218                                        org.nasdanika.html.emf.EObjectActionResolver.Context context, 
219                                        ProgressMonitor progressMonitor) throws Exception {
220                                
221                                EList<Issue> children = element.getChildren();
222                                if (children.isEmpty()) {
223                                        return null;
224                                }
225                                return Collections.singletonList(renderList(children, true, this, base, EngineeringPackage.Literals.ISSUE__CHILDREN, context, progressMonitor));
226                        }
227                        
228                };
229        }       
230        
231        protected void createPrinciplesActions(
232                        Action action, 
233                        BiConsumer<EObject,Action> registry, 
234                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
235                        ProgressMonitor progressMonitor) throws Exception {
236                
237                List<Principle> principles = getTarget().getPrinciples();
238                if (!principles.isEmpty()) {
239                        Action group = AppFactory.eINSTANCE.createAction();
240                        group.setText("Principles");
241                        group.setUuid(action.getUuid() + "-principles");
242                        group.setLocation("principles.html");
243                        action.getNavigation().add(group);
244                        EList<Action> groupAnonymous = group.getAnonymous();
245                        for (Principle principle: principles) {
246                                groupAnonymous.add(createChildAction(principle, registry, resolveConsumer, progressMonitor));
247                        }
248                }
249        }
