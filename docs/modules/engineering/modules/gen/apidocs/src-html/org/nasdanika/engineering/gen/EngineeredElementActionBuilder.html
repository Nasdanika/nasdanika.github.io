026
027public class EngineeredElementActionBuilder<T extends EngineeredElement> extends ForumActionBuilder<T> {
028        
029        public EngineeredElementActionBuilder(T target, Context context) {
030                super(target, context);         
031        }
032        
033        protected List<ETypedElement> getProperties() {
034                List<ETypedElement> properties = super.getProperties();
035                properties.add(EngineeringPackage.Literals.ENGINEERED_ELEMENT__STATUS);
036                properties.add(EngineeringPackage.Literals.ENGINEERED_ELEMENT__OWNERS);
037                properties.add(EngineeringPackage.Literals.ENGINEERED_ELEMENT__EXPERTS);
038                return properties;
039        }       
040        
041        @Override
042        protected Action buildAction(
043                        Action action,
044                        BiConsumer<EObject,Action> registry, 
045                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
046                        ProgressMonitor progressMonitor) throws Exception {
047                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
048                
049                createIssuesAction(action, registry, resolveConsumer, progressMonitor);         
050                createAllIssuesAction(action, registry, resolveConsumer, progressMonitor);              
051                createPrinciplesActions(action, registry, resolveConsumer, progressMonitor);
052                createDiscussionAction(action, registry, resolveConsumer, progressMonitor);             
053                
054                createOutboundConnectionsActions(action, registry, resolveConsumer, progressMonitor);
055                
056                return action;
057        }
058        
059        protected void createOutboundConnectionsActions(
060                        Action action,
061                        BiConsumer<EObject,Action> registry, 
062                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
063                        ProgressMonitor progressMonitor) throws Exception {
064                
065                EList<Action> groupAnonymous = action.getAnonymous();
066                for (Connection outboundConnection: getTarget().getOutboundConnections()) {
067                        groupAnonymous.add(createChildAction(outboundConnection, registry, resolveConsumer, progressMonitor));
068                }
069        }
070        
071        protected void createIssuesAction(
072                        Action action,
073                        BiConsumer<EObject,Action> registry, 
074                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
075                        ProgressMonitor progressMonitor) throws Exception {
076                
077                List<Issue> issues = getTarget().getIssues();
078                if (!issues.isEmpty()) {
079                        Action group = AppFactory.eINSTANCE.createAction();
080                        group.setText("Issues");
081                        group.setUuid(action.getUuid() + "-issues");
082                        group.setLocation("issues.html");
083                        //group.setNotification(String.valueOf(issues.size()));
084                        // TODO - icon, ...
085                        EList<Action> groupAnonymous = group.getAnonymous();
086                        for (Issue issue: issues) {
087                                groupAnonymous.add(createChildAction(issue, registry, resolveConsumer, progressMonitor));
088                        }
089                
090                        action.getNavigation().add(group);
091                }
092                
093        }
094        
095        protected void createAllIssuesAction(
096                        Action action,
097                        BiConsumer<EObject,Action> registry, 
098                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
099                        ProgressMonitor progressMonitor) throws Exception {
100                
101                List<Issue> allIssues = getTarget().getAllIssues();
102                if (!allIssues.isEmpty()) {
103                        Action group = AppFactory.eINSTANCE.createAction();
104                        group.setText("All Issues");
105                        group.setUuid(action.getUuid() + "-all-issues");
106                        group.setLocation("all-issues.html");
107                        action.getNavigation().add(group);
108                }
109                
110        }
111        
112        protected void createDiscussionAction(
113                        Action action,
114                        BiConsumer<EObject,Action> registry, 
115                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
116                        ProgressMonitor progressMonitor) throws Exception {
117                
118                List<Forum> discussion = getTarget().getDiscussion();
119                EList<Topic> topics = getTarget().getTopics();
120                if (!discussion.isEmpty() || !topics.isEmpty()) {
121                        Action group = AppFactory.eINSTANCE.createAction();
122                        group.setText("Discussion");
123                        group.setIcon("far fa-comments");
124                        group.setUuid(action.getUuid() + "-discussion");
125                        group.setLocation("discussion.html");
126                        action.getNavigation().add(group);
127                }               
128        }
129                
130        @Override
131        protected void resolve(
132                        Action action, 
133                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
134                        ProgressMonitor progressMonitor) throws Exception {
135                super.resolve(action, context, progressMonitor);
136                
137                // Outbound connections
138                EList<Connection> outboundConnections = getTarget().getOutboundConnections();
139                if (!outboundConnections.isEmpty()) {
140                        Action outboundConnectionsAction = AppFactory.eINSTANCE.createAction();
141                        outboundConnectionsAction.setText("Outbound connections");
142                        action.getSections().add(outboundConnectionsAction);
143                        
144                        Table outboundConnectionsTable = buildTable(
145                                        outboundConnections, 
146                                        action, 
147                                        EngineeringPackage.Literals.CONNECTION_SOURCE__OUTBOUND_CONNECTIONS, 
148                                        context, 
149                                        progressMonitor, 
150                                        createColumnBuilder("Name"),
151                                        createColumnBuilder(EngineeringPackage.Literals.CONNECTION__TARGET),
152                                        createColumnBuilder(NcorePackage.Literals.MODEL_ELEMENT__DESCRIPTION));
153                        
154                        outboundConnectionsAction.getContent().add(outboundConnectionsTable);                   
155                }
156                                
157                // Inbound connections
158                EList<Connection> inboundConnections = getTarget().getInboundConnections();
159                if (!inboundConnections.isEmpty()) {
160                        Action inboundConnectionsAction = AppFactory.eINSTANCE.createAction();
161                        inboundConnectionsAction.setText("Inbound connections");
162                        action.getSections().add(inboundConnectionsAction);
163                        
164                        Table inboundConnectionsTable = buildTable(
165                                        inboundConnections, 
166                                        action, 
167                                        EngineeringPackage.Literals.CONNECTION_TARGET__INBOUND_CONNECTIONS, 
168                                        context, 
169                                        progressMonitor, 
170                                        createColumnBuilder("Name"),
171                                        createColumnBuilder(EcorePackage.Literals.EOBJECT___ECONTAINER, "Source"),
172                                        createColumnBuilder(NcorePackage.Literals.MODEL_ELEMENT__DESCRIPTION));
173                        
174                        inboundConnectionsAction.getContent().add(inboundConnectionsTable);                     
175                }
176                
177                T engineeredElement = getTarget();
178                EList<Issue> issues = engineeredElement.getIssues();
179                if (!issues.isEmpty()) {
180                        String issuesGroupUUID = action.getUuid() + "-issues";
181                        Optional<Action> issuesActionOptional = action.getNavigation().stream()
182                                        .filter(Action.class::isInstance)                                       
183                                        .map(Action.class::cast)
184                                        .filter(a -> issuesGroupUUID.equals(a.getUuid()))
185                                        .findFirst();
186                        
187                        Action issuesAction = issuesActionOptional.get();
188                        issuesAction.getContent().add(renderList(issues, true, createIssueChildrenProvider(), issuesAction, EngineeringPackage.Literals.ENGINEERED_ELEMENT__ISSUES, context, progressMonitor)); // Table?
189                }
190                
191                EList<Issue> allIssues = engineeredElement.getAllIssues();
192                if (!allIssues.isEmpty()) {
193                        String allIssuesGroupUUID = action.getUuid() + "-all-issues";
194                        Optional<Action> allIssuesActionOptional = action.getNavigation().stream()
195                                        .filter(Action.class::isInstance)                                       
196                                        .map(Action.class::cast)
197                                        .filter(a -> allIssuesGroupUUID.equals(a.getUuid()))
198                                        .findFirst();
199                        Action allIssuesAction = allIssuesActionOptional.get();
200                        Tag allIssuesTable = buildDynamicTable(
201                                        allIssues, 
202                                        action, 
203                                        EngineeringPackage.Literals.ENGINEERED_ELEMENT__ALL_ISSUES,
204                                        "engineered-element--all-issues",
205                                        null,
206                                        context, 
207                                        progressMonitor,
208                                        createDynamicColumnBuilder("Issue", true),
209                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__TARGET, true),
210                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__ASSIGNEE, true),
211                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__CATEGORIES, true),
212                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__PRIORITY, true),
213                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__SEVERITY, true),
214                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__STATUS, true),
215                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__WORKABLE, true),
216                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__BENEFIT, false),
217                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__COST, false),
218                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__EFFORT, false),
219                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__REMAINING_COST, false),
220                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__REMAINING_EFFORT, false),
221                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION, true),
222                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ENGINEERED_CAPABILITY__RELEASES, true),
223                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__CONTRIBUTES_TO, true),
224                                        createDynamicColumnBuilder(EngineeringPackage.Literals.ISSUE__INCREMENT, true),
225                                        createDynamicColumnBuilder(NcorePackage.Literals.PERIOD__START, true),
226                                        createDynamicColumnBuilder(NcorePackage.Literals.PERIOD__END, true),
227                                        createDynamicColumnBuilder(NcorePackage.Literals.PERIOD__DURATION, false));
228                        
229                        allIssuesAction.getContent().add(allIssuesTable);
230                }
231                
232                EList<Principle> principles = getTarget().getPrinciples();
233                if (!principles.isEmpty()) {
234                        String principlesGroupUUID = action.getUuid() + "-principles";
235                        Optional<Action> principlesActionOptional = action.getNavigation().stream()
236                                        .filter(Action.class::isInstance)
237                                        .map(Action.class::cast)
238                                        .filter(a -> principlesGroupUUID.equals(a.getUuid()))
239                                        .findFirst();
240                        
241                        Action principlesAction = principlesActionOptional.get();
242                        principlesAction.getContent().add(renderList(principles, true, createPrincipleChildrenListProvider(), action, EngineeringPackage.Literals.ENGINEERED_ELEMENT__PRINCIPLES, context, progressMonitor));
243                }
244        }
245
246        protected ContentProvider<Principle> createPrincipleChildrenListProvider() {
247                ContentProvider<Principle> goalChildrenListProvider = new ContentProvider<Principle>() {
248
249                        @Override
250                        public List<EObject> createContent(
251                                        Principle element, 
252                                        Action base, 
253                                        ETypedElement typedElement,
254                                        org.nasdanika.html.emf.EObjectActionResolver.Context context, 
255                                        ProgressMonitor progressMonitor) throws Exception {
256                                
257                                EList<Principle> children = element.getChildren();
258                                if (children.isEmpty()) {
259                                        return null;
260                                }
261                                return Collections.singletonList(renderList(children, true, this, base, EngineeringPackage.Literals.PRINCIPLE__CHILDREN, context, progressMonitor));
262                        }
263                        
264                };
265                return goalChildrenListProvider;
266        }
267        
268        private ContentProvider<Issue> createIssueChildrenProvider() {
269                return new ContentProvider<Issue>() {
270        
271                        @Override
272                        public List<EObject> createContent(
273                                        Issue element, 
274                                        Action base, 
275                                        ETypedElement typedElement,
276                                        org.nasdanika.html.emf.EObjectActionResolver.Context context, 
277                                        ProgressMonitor progressMonitor) throws Exception {
278                                
279                                EList<Issue> children = element.getChildren();
280                                if (children.isEmpty()) {
281                                        return null;
282                                }
283                                return Collections.singletonList(renderList(children, true, this, base, EngineeringPackage.Literals.ISSUE__CHILDREN, context, progressMonitor));
284                        }
285                        
286                };
287        }       
288        
289        protected void createPrinciplesActions(
290                        Action action, 
291                        BiConsumer<EObject,Action> registry, 
292                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
293                        ProgressMonitor progressMonitor) throws Exception {
294                
295                List<Principle> principles = getTarget().getPrinciples();
296                if (!principles.isEmpty()) {
297                        Action group = AppFactory.eINSTANCE.createAction();
298                        group.setText("Principles");
299                        group.setUuid(action.getUuid() + "-principles");
300                        group.setLocation("principles.html");
301                        action.getNavigation().add(group);
302                        EList<Action> groupAnonymous = group.getAnonymous();
303                        for (Principle principle: principles) {
304                                groupAnonymous.add(createChildAction(principle, registry, resolveConsumer, progressMonitor));
305                        }
306                }
307        }
