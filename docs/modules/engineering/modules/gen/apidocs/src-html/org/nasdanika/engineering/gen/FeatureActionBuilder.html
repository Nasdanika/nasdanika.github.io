001package org.nasdanika.engineering.gen;
002
003import java.util.Collections;
004import java.util.List;
005import java.util.Optional;
006import java.util.function.BiConsumer;
007
008import org.eclipse.emf.common.util.EList;
009import org.eclipse.emf.ecore.EObject;
010import org.eclipse.emf.ecore.ETypedElement;
011import org.nasdanika.common.Context;
012import org.nasdanika.common.ProgressMonitor;
013import org.nasdanika.engineering.EngineeringPackage;
014import org.nasdanika.engineering.Feature;
015import org.nasdanika.engineering.Issue;
016import org.nasdanika.html.model.app.Action;
017import org.nasdanika.html.model.app.AppFactory;
018import org.nasdanika.html.model.bootstrap.Table;
019import org.nasdanika.ncore.NcorePackage;
020
021public class FeatureActionBuilder extends EngineeredCapabilityActionBuilder<Feature> {
022        
023        public FeatureActionBuilder(Feature target, Context context) {
024                super(target, context);         
025        }
026        
027        @Override
028        protected Action buildAction(
029                        Action action,
030                        BiConsumer<EObject,Action> registry, 
031                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
032                        ProgressMonitor progressMonitor) throws Exception {
033                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
034                
035                action.getSections().addAll(createChildrenActions(registry, resolveConsumer, progressMonitor));
036                createIssuesAction(action, registry, resolveConsumer, progressMonitor);         
037                
038                return action;
039        }
040        
041        protected void createIssuesAction(
042                        Action action,
043                        BiConsumer<EObject,Action> registry, 
044                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
045                        ProgressMonitor progressMonitor) throws Exception {
046                
047                List<Issue> issues = getTarget().getIssues();
048                if (!issues.isEmpty()) {
049                        Action group = AppFactory.eINSTANCE.createAction();
050                        group.setText("Issues");
051                        group.setUuid(action.getUuid() + "-issues");
052                        group.setLocation("issues.html");
053                        //group.setNotification(String.valueOf(issues.size()));
054                        // TODO - icon, ...
055                
056                        action.getNavigation().add(group);
057                }
058                
059        }
060        
061        protected List<Action> createChildrenActions(
062                        BiConsumer<EObject,Action> registry, 
063                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
064                        ProgressMonitor progressMonitor) throws Exception {
065                
066                List<Feature> children = getTarget().getChildren();
067                if (children.isEmpty()) {
068                        return Collections.emptyList();
069                }
070                Action group = AppFactory.eINSTANCE.createAction();
071                group.setText("Children");
072                EList<Action> groupAnonymous = group.getAnonymous();
073                for (Feature child: children) {
074                        groupAnonymous.add(createChildAction(child, registry, resolveConsumer, progressMonitor));
075                }
076                
077                return Collections.singletonList(group);
078        }
079        
080        @Override
081        protected void resolve(
082                        Action action, 
083                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
084                        ProgressMonitor progressMonitor) throws Exception {
085                super.resolve(action, context, progressMonitor);
086                
087                EList<Feature> children = getTarget().getChildren();
088                if (!children.isEmpty()) {
089                        Action childrenAction = (Action) action.getSections().get(0);
090                        ContentProvider<Feature> issueCategoryChildrenListProvider = new ContentProvider<Feature>() {
091
092                                @Override
093                                public List<EObject> createContent(
094                                                Feature element, 
095                                                Action base, 
096                                                ETypedElement typedElement,
097                                                org.nasdanika.html.emf.EObjectActionResolver.Context context, 
098                                                ProgressMonitor progressMonitor) throws Exception {
099                                        
100                                        EList<Feature> children = element.getChildren();
101                                        if (children.isEmpty()) {
102                                                return null;
103                                        }
104                                        return Collections.singletonList(renderList(children, false, this, base, EngineeringPackage.Literals.FEATURE__CHILDREN, context, progressMonitor));
105                                }
106                                
107                        };
108                        
109                        childrenAction.getContent().add(renderList(children, true, issueCategoryChildrenListProvider, action, EngineeringPackage.Literals.FEATURE__CHILDREN, context, progressMonitor));
110                }
111                
112                EList<Issue> issues = getTarget().getIssues();
113                if (!issues.isEmpty()) {
114                        String issuesGroupUUID = action.getUuid() + "-issues";
115                        Optional<Action> issuesActionOptional = action.getNavigation().stream()
116                                        .filter(Action.class::isInstance)                                       
117                                        .map(Action.class::cast)
118                                        .filter(a -> issuesGroupUUID.equals(a.getUuid()))
119                                        .findFirst();
120                        Action issuesAction = issuesActionOptional.get();
121                        Table issuesTable = buildTable(
122                                        issues, 
123                                        action, 
124                                        EngineeringPackage.Literals.FEATURE__ISSUES, 
125                                        context, 
126                                        progressMonitor,
127                                        createColumnBuilder("Issue"),
128                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__TARGET),
129                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__ASSIGNEE),
130                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__CATEGORIES),
131                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__PRIORITY),
132                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__SEVERITY),
133                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__STATUS),
134                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__WORKABLE),
135                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__BENEFIT),
136                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__COST),
137                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__EFFORT),
138                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__REMAINING_COST),
139                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__REMAINING_EFFORT),
140                                        createColumnBuilder(EngineeringPackage.Literals.ENDEAVOR__COMPLETION),
141                                        createColumnBuilder(EngineeringPackage.Literals.ISSUE__INCREMENT),
142                                        createColumnBuilder(NcorePackage.Literals.PERIOD__START),
143                                        createColumnBuilder(NcorePackage.Literals.PERIOD__END),
144                                        createColumnBuilder(NcorePackage.Literals.PERIOD__DURATION));
145                        
146                        issuesAction.getContent().add(issuesTable);
147                }
148                
149        }       
150        
151}
