001package org.nasdanika.engineering.gen;
002
003import java.util.Collections;
004import java.util.List;
005import java.util.Optional;
006import java.util.function.BiConsumer;
007
008import org.eclipse.emf.common.util.EList;
009import org.eclipse.emf.ecore.EObject;
010import org.eclipse.emf.ecore.ETypedElement;
011import org.nasdanika.common.Context;
012import org.nasdanika.common.ProgressMonitor;
013import org.nasdanika.engineering.EngineeringPackage;
014import org.nasdanika.engineering.Goal;
015import org.nasdanika.html.model.app.Action;
016import org.nasdanika.html.model.app.AppFactory;
017
018public class GoalActionBuilder extends AimActionBuilder<Goal> {
019        
020        public GoalActionBuilder(Goal target, Context context) {
021                super(target, context);         
022        }
023        
024        @Override
025        protected Action buildAction(
026                        Action action,
027                        BiConsumer<EObject,Action> registry, 
028                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
029                        ProgressMonitor progressMonitor) throws Exception {
030                action = super.buildAction(action, registry, resolveConsumer, progressMonitor);
031                
032                createChildrenActions(action, registry, resolveConsumer, progressMonitor);
033                
034                return action;
035        }
036        
037        protected void createChildrenActions(
038                        Action action, 
039                        BiConsumer<EObject,Action> registry, 
040                        java.util.function.Consumer<org.nasdanika.common.Consumer<org.nasdanika.html.emf.EObjectActionResolver.Context>> resolveConsumer, 
041                        ProgressMonitor progressMonitor) throws Exception {
042                
043                List<Goal> children = getTarget().getChildren();
044                if (!children.isEmpty()) {
045                        Action group = AppFactory.eINSTANCE.createAction();
046                        group.setText("Children");
047                        group.setUuid(action.getUuid() + "-children");
048                        action.getSections().add(group);
049                        EList<Action> groupAnonymous = group.getAnonymous();
050                        for (Goal goal: children) {
051                                groupAnonymous.add(createChildAction(goal, registry, resolveConsumer, progressMonitor));
052                        }
053                }
054        }
055        
056        @Override
057        protected void resolve(
058                        Action action, 
059                        org.nasdanika.html.emf.EObjectActionResolver.Context context,
060                        ProgressMonitor progressMonitor) throws Exception {
061                super.resolve(action, context, progressMonitor);
062                
063                EList<Goal> children = getTarget().getChildren();
064                if (!children.isEmpty()) {
065                        String childrenGroupUUID = action.getUuid() + "-children";
066                        Optional<Action> childrenActionOptional = action.getSections().stream()
067                                        .filter(a -> childrenGroupUUID.equals(a.getUuid()))
068                                        .findFirst();
069                        childrenActionOptional.get().getContent().add(renderList(children, true, createGoalChildrenListProvider(), action, EngineeringPackage.Literals.PERSONA__GOALS, context, progressMonitor));
070                }
071        }
072
073        protected ContentProvider<Goal> createGoalChildrenListProvider() {
074                ContentProvider<Goal> goalChildrenListProvider = new ContentProvider<Goal>() {
075
076                        @Override
077                        public List<EObject> createContent(
078                                        Goal element, 
079                                        Action base, 
080                                        ETypedElement typedElement,
081                                        org.nasdanika.html.emf.EObjectActionResolver.Context context, 
082                                        ProgressMonitor progressMonitor) throws Exception {
083                                
084                                EList<Goal> children = element.getChildren();
085                                if (children.isEmpty()) {
086                                        return null;
087                                }
088                                return Collections.singletonList(renderList(children, true, this, base, EngineeringPackage.Literals.GOAL__CHILDREN, context, progressMonitor));
089                        }
090                        
091                };
092                return goalChildrenListProvider;
093        }
094        
095}
