027/**
028 * Represents the root mxfile element.
029 * 
030 * @author Pavel
031 *
032 */
033public interface Document extends Element {
034        
035        /**
036         * @return a list of pages. This list supports add() method to insert pages from other documents. The add() method creates a clone of the page passed as a parameter. 
037         */
038        List<Page> getPages();
039        
040        /**
041         * @return Creates a new page and adds it to the list of pages.
042         */
043        Page createPage();
044        
045        /**
046         * Loads document from an XML string.
047         * @param docStr
048         * @param uri Optional document URI for resolution of relative URI's.
049         * @return

051         * @throws SAXException 
052         * @throws ParserConfigurationException 
053         */

055                return new DocumentImpl(docStr, uri);
056        }
057
058        /**
059         * 
060         * @param reader
061         * @param uri Optional document URI for resolution of relative URI's.
062         * @return
063         * @throws IOException
064         * @throws ParserConfigurationException
065         * @throws SAXException
066         */
067        static Document load(Reader reader, URI uri) throws IOException, ParserConfigurationException, SAXException {
068                return new DocumentImpl(reader, uri);
069        }
070        
071        /**
072         * 
073         * @param compressed
074         * @param uri Optional document URI for resolution of relative URI's.
075         * @return
076         * @throws ParserConfigurationException
077         */
078        static Document create(boolean compressed, URI uri) throws ParserConfigurationException {
079                return new DocumentImpl(compressed, uri);
080        }
081
082        /**
083         * Loads document using UTF-8 encoding.
084         * @param in
085         * @param uri Optional document URI for resolution of relative URI's.
086         * @return
087         * @throws IOException
088         * @throws SAXException 
089         * @throws ParserConfigurationException 
090         */
091        static Document load(InputStream in, URI uri) throws IOException, ParserConfigurationException, SAXException {
092                return new DocumentImpl(in, uri);
093        }
094        
095        /**
096         * @param uri
097         * @return true if the argument URI is drawio data URI, i.e. is starts with data:drawio[;base64],
098         */
099        static boolean isDataURI(URI uri) {
100                if (uri != null && "data".equals(uri.scheme())) {
101                        String uriStr = uri.toString();
102                        String DRAWIO = "data:drawio";
103                        return uriStr.startsWith(DRAWIO +";base64,") || uriStr.startsWith(DRAWIO +",");
104                }
105                return false;
106        }
107
108        /**
109         * Loads document from an URI - supports data: schema.
110         * @param source
111         * @return
112         * @throws IOException
113         * @throws ParserConfigurationException
114         * @throws SAXException
115         */
116        static Document load(URI source) throws IOException, ParserConfigurationException, SAXException {
117                if (isDataURI(source)) {
118                        String uriStr = source.toString();
119                        uriStr = uriStr.substring("data:drawio".length());                              
120                        String BASE64 = ";base64,";
121                        if (uriStr.startsWith(BASE64)) {
122                                uriStr = new String(Base64.decodeBase64(URLDecoder.decode(uriStr.substring(BASE64.length()), StandardCharsets.UTF_8)), StandardCharsets.UTF_8);
123                        } else {
124                                uriStr = URLDecoder.decode(uriStr.substring(1), StandardCharsets.UTF_8);
125                        }
126                        JSONObject payload = new JSONObject(new JSONTokener(uriStr));
127                        return load(payload.getString("document"), payload.has("uri") ? URI.createURI(payload.getString("uri")) : null);
128                }
129                        
130                return load(new URL(source.toString()));
131        }
132        
133        /**
134         * Loads document from URL by opening a stream.
135         * @param source
136         * @return
137         * @throws IOException
138         * @throws ParserConfigurationException
139         * @throws SAXException
140         */
141        static Document load(URL source) throws IOException, ParserConfigurationException, SAXException {
142                try (InputStream in = source.openStream()) {
143                        return load(in, URI.createURI(source.toString()));
144                }
145        }
146        
147        /**
148         * Loads documents stored in PNG metadata using UTF-8 encoding.
149         * @param in
150         * @return
151         * @throws IOException
152         * @throws SAXException 
153         * @throws ParserConfigurationException 
154         */
155        static List<Document> loadFromPngMetadata(InputStream in, URI uri) throws IOException, ParserConfigurationException, SAXException {
156                ImageReader imageReader = ImageIO.getImageReadersByFormatName("png").next();
157                imageReader.setInput(ImageIO.createImageInputStream(in));
158                IIOMetadata metadata = imageReader.getImageMetadata(0);
159                List<Document> ret = new ArrayList<>();
160                for (int imageCount = imageReader.getNumImages(true), minIndex = imageReader.getMinIndex(), idx = 0; idx < imageCount; ++idx) {
161                        IIOMetadata imageMetadata = imageReader.getImageMetadata(minIndex + idx);
162                        if (imageMetadata != null) {
163                                for (String formatName: imageMetadata.getMetadataFormatNames()) {
164                                        Node tree = metadata.getAsTree(formatName);
165                                        NodeList childNodes = tree.getChildNodes();
166                                        for (int i = 0; i < childNodes.getLength(); ++i) {
167                                                Node child = childNodes.item(i);
168                                                if (child instanceof org.w3c.dom.Element && "Text".equals(((org.w3c.dom.Element) child).getTagName())) {
169                                                        NodeList grandChildNodes = child.getChildNodes();
170                                                        for (int j = 0; j < grandChildNodes.getLength(); ++j) {
171                                                                Node grandChild = grandChildNodes.item(j);
172                                                                if (grandChild instanceof org.w3c.dom.Element 
173                                                                                && "TextEntry".equals(((org.w3c.dom.Element) grandChild).getTagName())
174                                                                                && ((org.w3c.dom.Element) grandChild).hasAttribute("keyword")
175                                                                                && "mxfile".equals(((org.w3c.dom.Element) grandChild).getAttribute("keyword"))) {
176                                                                        String value = ((org.w3c.dom.Element) grandChild).getAttribute("value");
177                                                                        ret.add(load((URLDecoder.decode(value, StandardCharsets.UTF_8)), uri));                                                                 
178                                                                }
179                                                        }
180                                                }
181                                        }
182                                }
183                        }
184                }
185                return ret;
186        }
187        
188        /**
189         * Loads documents stored in PNG metadata using UTF-8 encoding.
190         * @param source
191         * @return
192         * @throws IOException
193         * @throws ParserConfigurationException
194         * @throws SAXException
195         */
196        static List<Document> loadFromPngMetadata(URL source) throws IOException, ParserConfigurationException, SAXException {
197                try (InputStream in = source.openStream()) {
198                        return loadFromPngMetadata(in, URI.createURI(source.toString()));
199                }
200        }
201        
202        /**
203         * @param compress If null keeps the original compression setting. If not null forces either compressed (TRUE) or uncompressed (FALSE).
204         * This can be used to generate HTML tags with compressed document while keeping the source document uncompressed so it can be manually inspected, edited, diffed, and merged.
205         * @param viewer URL of the viewer script. Default value is used if null.  
206         * @return div tag with the embedded document which can be inserted into HTML pages.
207         */
208        String toHtml(Boolean compress, String viewer) throws TransformerException, IOException;
209
210        /**
211         * Encodes document as data: URI.
212         * @return
213         */
214        URI toDataURI(Boolean compress) throws TransformerException, IOException;
215
216        /**
217         * Saves the document to String.
218         * @param compress If null keeps the original compression setting. If not null forces either compressed (TRUE) or uncompressed (FALSE).
219         * This can be used to generate HTML tags with compressed document while keeping the source document uncompressed so it can be manually inspected, edited, diffed, and merged.  
220         * @return Document string.
221         */
222        String save(Boolean compress) throws TransformerException, IOException;
