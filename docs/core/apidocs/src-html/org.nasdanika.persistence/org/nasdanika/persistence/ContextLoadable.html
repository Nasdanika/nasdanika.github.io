001package org.nasdanika.persistence;
002
003import java.util.List;
004
005import org.eclipse.emf.common.util.URI;
006import org.nasdanika.common.Context;
007import org.nasdanika.common.ProgressMonitor;
008
009/**
010 * Objects which know how to load themselves from configuration may implement this interface.
011 * Also this interface may be used as an adapter interface. In this case an object to be loaded is first created by the object loader based on 
012 * type passed to {@link ObjectLoader}'s create() method and then is adapted to this interface to be loaded.
013 * @author Pavel
014 *
015 */
016public interface ContextLoadable {
017        
018        void load(Context context, ObjectLoader loader, Object config, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers);      
019
020}
