001package org.nasdanika.persistence;
002
003import java.util.function.Consumer;
004
005/**
006 * Resolves references when loading persistent objects.
007 * @author Pavel Vlasov
008 * 
009 * @param <T> Persistent object type.
010 * @param <R> Reference type.
011 */
012public interface Resolver<T,R> {
013        
014        /**
015         * Resolves the reference.
016         * @param source Source object.
017         * @param reference Reference.
018         * @param targetConsumer ConsumerFactory of resolved target object. This consumer may be invoked as part of resolve() method
019         * or later, once all objects are loaded, in the case of a reference to a not yet loaded object.
020         */
021        void resolve(T source, R reference, Consumer<T> targetConsumer);
022
023}
