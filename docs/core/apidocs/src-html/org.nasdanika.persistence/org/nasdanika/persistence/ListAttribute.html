001package org.nasdanika.persistence;
002
003import java.util.ArrayList;
004import java.util.Collection;
005import java.util.Collections;
006import java.util.List;
007
008import org.eclipse.emf.common.util.URI;
009import org.nasdanika.common.ProgressMonitor;
010
011/**
012 * If config is a list loads each element using createElement(), otherwise creates a singleton loaded with createElement()
013 * @author Pavel
014 *
015 * @param <T>
016 */
017public class ListAttribute<T> extends Attribute<List<T>> {
018
019        public ListAttribute(Object key, 
020                        boolean isDefault, 
021                        boolean required, 
022                        List<T> defaultValue, 
023                        String description, 
024                        Object... exclusiveWith) {
025                super(key, isDefault, required, defaultValue, description, exclusiveWith);
026        }
027        
028        @Override
029        public List<T> create(ObjectLoader loader, Object config, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) {
030                if (config instanceof Collection) {
031                        List<T> ret = new ArrayList<>();
032                        int idx = 0;
033                        for (Object element: (Collection<?>) config) {
034                                ret.addAll(createElements(loader, element, base, progressMonitor, Util.getMarkers((Collection<?>) config, idx++)));
035                        }
036                        return ret;
037                }
038                
039                return createElements(loader, config, base, progressMonitor, markers);
040        }
041        
042        /**
043         * Creates list elements from a config element. May create zero or more elements. 
044         * This implementation delegates to createElement(), i.e. creates a single element. 
045         * @param loader
046         * @param element
047         * @param base
048         * @param progressMonitor
049         * @param marker
050         * @return
051         */
052        protected List<T> createElements(ObjectLoader loader, Object element, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) { 
053                return Collections.singletonList(createElement(loader, element, base, progressMonitor, markers)); 
054        }
055        
056
057        /**
058         * Creates element list.
059         * @param loader
060         * @param element
061         * @param base
062         * @param progressMonitor
063         * @param marker
064         * @return
065         */
066        @SuppressWarnings("unchecked")
067        protected T createElement(ObjectLoader loader, Object element, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) { 
068                return (T) element; 
069        }
070        
071}
