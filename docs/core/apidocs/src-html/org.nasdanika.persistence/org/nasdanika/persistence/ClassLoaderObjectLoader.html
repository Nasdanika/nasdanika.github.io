001package org.nasdanika.persistence;
002
003import java.lang.reflect.Constructor;
004import java.lang.reflect.InvocationTargetException;
005import java.net.URL;
006import java.util.List;
007
008import org.eclipse.emf.common.util.URI;
009import org.nasdanika.common.ExecutionException;
010import org.nasdanika.common.ProgressMonitor;
011
012/**
013 * Creates object instances with {@link ClassLoader}. Type shall resolve to fully qualified class name.
014 * Classes shall have a 5-argument constructors taking {@link ObjectLoader}, config object, {@link URL}, {@link ProgressMonitor}, and {@link Marker}. 
015 * @author Pavel
016 *
017 */
018public class ClassLoaderObjectLoader implements ObjectLoader {
019        
020        private ClassLoader classLoader;
021        private java.util.function.Function<String, String> resolver;
022        private ObjectLoader chain;
023
024        /**
025         * 
026         * @param classLoader {@link ClassLoader} for loading classes.
027         * @param resolver Resolver of type to fully qualified class name. If null, then type is used AS-IS. E.g. a map::get for a map of logical names to fully qualified class names.
028         * @param chain Object loader to fall back to if resolver is not null and returned null. 
029         */
030        public ClassLoaderObjectLoader(ClassLoader classLoader, java.util.function.Function<String,String> resolver, ObjectLoader chain) {
031                this.resolver = resolver;
032                this.classLoader = classLoader;
033                this.chain = chain;
034        }
035
036        @Override
037        public Object create(ObjectLoader loader, String type, Object config, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) {
038                String fqn = resolver == null ? type : resolver.apply(type);
039                if (fqn == null) {
040                        if (resolver == null) {
041                                throw new NullPointerException("Type name is null");
042                        }
043                        if (chain == null) {
044                                throw new IllegalArgumentException("Type was resolved to null and there is no chain loader: " + type);
045                        }
046                        
047                        return chain.create(loader, type, config, base, progressMonitor, markers);
048                }
049                
050                try {
051                        Class<?> clazz = classLoader.loadClass(fqn);            
052                        Constructor<?> constructor = clazz.getConstructor(ObjectLoader.class, Object.class, URL.class, ProgressMonitor.class, Marker.class);            
053                        return constructor.newInstance(loader, config, base, progressMonitor, markers);
054                } catch (ClassNotFoundException | NoSuchMethodException | SecurityException | InstantiationException | IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
055                        throw new ExecutionException(e);
056                }
057        }
058
059}
