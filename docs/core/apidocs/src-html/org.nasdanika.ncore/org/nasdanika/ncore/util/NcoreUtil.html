075        /**
076         * Flattened inheritance hierarchy from the argument class to all of its supertypes in the order of inheritance.
077         * @param eClass
078         * @return
079         */
080        public static List<EClass> lineage(EClass eClass) {
081                List<EClass> ret = new ArrayList<>();
082                ret.add(eClass);
083                for (EClass sc: eClass.getESuperTypes()) {
084                        for (EClass scLineage: lineage(sc)) {
085                                if (!ret.contains(scLineage)) {
086                                        ret.add(scLineage);
087                                }
088                        }
089                }
090                return ret;
091        }

093        /**
094         * Finds objects referencing the argument object (target) via the specified reference.
095         * @param <T>
096         * @param target Referenced object
097         * @param eReference Reference
098         * @return Object referencing this one via the specified reference.
099         */
100        public static <T extends EObject> EList<T> getReferrers(EObject target, EReference eReference) {
101                EList<T> ret = new BasicInternalEList<>(EObject.class);
102                
103                Resource res = target.eResource(); 
104                TreeIterator<?> cit = null;
105                if (res == null) {
106                        EObject root = target;
107                        EObject rc;
108                        while ((rc = root.eContainer()) != null) {
109                                root = rc;
110                        }
111                        if (root != null) {
112                                NcoreUtil.collect(root, target, eReference, ret);
113                                cit = root.eAllContents();
114                        }                       
115                } else {
116                        ResourceSet rSet = res.getResourceSet();
117                        cit = rSet == null ? res.getAllContents() : rSet. getAllContents();
118                }
119                if (cit != null) {
120                        while (cit.hasNext()) {
121                                NcoreUtil.collect(cit.next(), target, eReference, ret);
122                        }                       
123                }
124                return ret;
125        }

127        @SuppressWarnings("unchecked")
128        private static <T extends EObject> void collect(
129                        Object source, 
130                        EObject target, 
131                        EReference eReference,
132                        EList<T> accumulator) {
133                if (eReference.getEContainingClass().isInstance(source)) {
134                        Object value = ((EObject) source).eGet(eReference);
135                        if (eReference.isMany()) {
136                                if (((Collection<?>) value).contains(target)) {
137                                        accumulator.add((T) source);
138                                }
139                        } else if (value == target) {
140                                accumulator.add((T) source);
141                        }                                       
142                }
143        }

160        /**
161         * Containment path relative to the object container. If the container is a map entry its key is added to path and the entry container is skipped. This happens recursively. 
162         * @param eObject
163         * @return null if there is no container. BiSupplier with first returning the first container which is not EMap entry and reference name concatenated with object key for {@link EMap}'s and with object index for many-references.
164         * Containment index is constructed taking eKeys into account.
165         */
166        public static BiSupplier<EObject, String> containmentPath(EObject eObj) {
167                EObject container = eObj.eContainer();
168                if (container == null) {
169                        return null;
170                }
171                EReference eContainmentFeature = eObj.eContainmentFeature();
172                
173                LinkedList<String> containmentIndex = new LinkedList<>();
174                while (NcoreUtil.isEMapEntry(container)) {
175                        Object key = ((Map.Entry<?,?>) container).getKey();
176                        if (key == null) {
177                                return null;
178                        }
179                        String idx = NcoreUtil.containmentIndex(eObj);
180                        if (idx != null) {
181                                containmentIndex.addFirst(idx);
182                        }
183                        containmentIndex.addFirst(key.toString());
184                        
185                        eContainmentFeature = container.eContainmentFeature();
186                        container = container.eContainer();
187                }
188                
189                if (containmentIndex.isEmpty()) {
190                        String idx = NcoreUtil.containmentIndex(eObj);
191                        if (idx != null) {
192                                containmentIndex.add(idx);
193                        }
194                } 
195        
196                containmentIndex.addFirst(NcoreUtil.getFeatureKey(container.eClass(), eContainmentFeature));
197                return BiSupplier.of(container, String.join("/", containmentIndex));                            
198        }

200        /**
201         * Index part of the containment path. Null for single references. EKeys for containments with EKeys. Position otherwise.
202         * @param eObj
203         * @return
204         */
205        private static String containmentIndex(EObject eObj) {
206                EObject container = eObj.eContainer();
207                if (container == null) {
208                        return null;
209                }
210                EReference eContainmentFeature = eObj.eContainmentFeature();
211                if (eContainmentFeature.isMany()) {
212                        EList<EAttribute> eKeys = eContainmentFeature.getEKeys();
213                        String position = String.valueOf(((List<?>) container.eGet(eContainmentFeature)).indexOf(eObj));
214                        if (eKeys.isEmpty()) {
215                                return position;
216                        }
217                        StringBuilder pathBuilder = new StringBuilder();
218                        for (EAttribute eKey: eKeys) {
219                                Object value = eObj.eGet(eKey);
220                                if (value == null || (value instanceof String && Util.isBlank((String) value))) {
221                                        return pathBuilder.length() == 0 ? position : pathBuilder.toString();
222                                }
223                                if (pathBuilder.length() > 0) {
224                                        pathBuilder.append("/");
225                                }
226                                pathBuilder.append(value);
227                        }
228                        return pathBuilder.toString();
229                }
230                return null;
231        }

233        /**
234                 * Uses {@link ModelElement}.getUris() if the argument is an instance of ModelElement to compute URI's.
235                 * Computes {@link URI}'s from the containment reference otherwise.
236                 * @param eObj
237                 * @return
238                 */
239                public static List<URI> getUris(EObject eObj) {
240                        if (eObj == null || isEMapEntry(eObj)) {
241                                return Collections.emptyList(); // EMap entries do not have URI's.
242                        }
243
244                        List<URI> ownURIs = new ArrayList<>();
245                        if (eObj instanceof ModelElement) {
246                                for (String uri: ((ModelElement) eObj).getUris()) {
247                                        if (!Util.isBlank(uri)) {
248                                                ownURIs.add(URI.createURI(uri));
249                                        }
250                                }
251                        }
252                        
253                        BiSupplier<EObject, String> containmentPath = containmentPath(eObj);
254                        if (containmentPath == null) {
255                                return ownURIs;
256                        }
257                        
258                        EObject container = containmentPath.getFirst();
259                        
260                        // Computing all permutations
261                        Collection<URI> ret = new HashSet<>();
262                        
263                        List<URI> relativeOwnURIs = new ArrayList<>();
264                        for (URI ownURI: ownURIs) {
265                                if (ownURI.isRelative()) {
266                                        relativeOwnURIs.add(ownURI);
267                                } else {
268                                        ret.add(ownURI); // Absolute or opaque
269                                }
270                        }
271                        
272                        // Resolving containment URI's
273                        for (URI cURI: getUris(container)) {
274                                if (!cURI.isRelative() && cURI.isHierarchical()) {
275                                        String cLastSegment = cURI.lastSegment();
276                                        if (cLastSegment == null || cLastSegment.length() > 0) {
277                                                cURI = cURI.appendSegment("");
278                                        }
279                                        for (URI relativeOwnURI: relativeOwnURIs) {
280                                                URI resolved = resolve(relativeOwnURI,cURI);
281                                                if (resolved != null) {
282                                                        ret.add(resolved);
283                                                }                                               
284                                        }
285                                }
286
287                                if (ownURIs.isEmpty()) { // Resorting to containment URI's only if there are no own URI's.
288                                        URI containmentURI = URI.createURI(containmentPath.getSecond());                                
289                                        URI resolved = resolve(containmentURI,cURI);
290                                        if (resolved != null) {
291                                                ret.add(resolved);
292                                        }
293                                }
294                        }
295                        
296                        return ret.stream().distinct().collect(Collectors.toList());
297                }

299                private static URI resolve(URI uri, URI base) {         
300                        if (uri.isRelative() && base.isHierarchical()) {
301                                if (!base.isRelative()) { 
302                                        return uri.resolve(base);
303                                }
304                                // Resolving containment path URI against container URI by combining all segments and treating the first first segment as scheme, second as authority and the rest as segments.                 
305                                List<String> allParts = new ArrayList<>();
306                                if (base.scheme() != null) {
307                                        allParts.add(base.scheme());
308                                }
309                                if (base.authority() != null) {
310                                        allParts.add(base.authority());
311                                }
312                                allParts.addAll(base.segmentsList());
313                                
314                                if (uri.scheme() != null) {
315                                        allParts.add(uri.scheme());
316                                }
317                                if (uri.authority() != null) {
318                                        allParts.add(uri.authority());
319                                }
320                                allParts.addAll(uri.segmentsList());
321                                
322                                if (allParts.size() == 0) {
323                                        return uri;
324                                }
325                                
326                                if (allParts.size() == 1) {
327                                        return URI.createURI(allParts.get(0));
328                                }
329
330                                if (allParts.size() == 2 && allParts.get(1).length() == 0) {
331                                        return URI.createURI(String.join("/", allParts) + (uri == null ? "" : uri));                            
332                                }
333                                
334                                URI allPartsURI = URI.createHierarchicalURI(allParts.get(0), allParts.get(1), null, allParts.subList(2, allParts.size()).toArray(new String[allParts.size() - 2]), null, null);
335                                return uri == null ? allPartsURI : uri.resolve(allPartsURI);
336                        }
337
338                        return null;
339                }

342         * @param eObj
343         * @return true if the argument is an entry for EMap
344         */
345        public static boolean isEMapEntry(EObject eObj) {
346                if (eObj instanceof Map.Entry) {
347                        EReference cf = eObj.eContainmentFeature();
348                        return cf != null && cf.isMany() && cf.getEType().getInstanceClass() == Map.Entry.class;
349                }
350                return false;
351        }

353        public static EObject commonAncestor(EObject a, EObject b) {
354                if (a == null || b == null) {
355                        return null;
356                }
357                if (EcoreUtil.isAncestor(a, b)) {
358                        return a.eContainer();
359                }
360                if (EcoreUtil.isAncestor(b, a)) {
361                        b.eContainer();
362                }
363                for (EObject ancestor = a.eContainer(); ancestor != null; ancestor = ancestor.eContainer()) {
364                        if (EcoreUtil.isAncestor(ancestor, b)) {
365                                return ancestor;
366                        }
367                }
368                return null;
369        }

393        /**
394         * Feature key is used for feature loading from YAML and other key-value sources and also in construction of containment path by containmentPath() method and, as such, EObject {@link URI} by getUri() method.
395         * This method traverses the class inheritance hierarchy and uses feature-keys class-level annotations to retrieve feature key. If such an annotation is not found then feature-level feature-key annotation is used.
396         * If none is found in the entire hierarchy then feature name is converted to kebab case is returned. 
397         * @param eClass Context class for which feature key is retrieved.
398         * @param feature Feature
399         * @return
400         */
401        public static String getFeatureKey(EClass eClass, EStructuralFeature feature) {
402                String featureKey = NcoreUtil.getNasdanikaAnnotationDetail(feature, FEATURE_KEY);
403                if (!Util.isBlank(featureKey)) {
404                        return featureKey;
405                }
406                
407                for (EClass ec: NcoreUtil.lineage(eClass)) {
408                        String featureKeys = NcoreUtil.getNasdanikaAnnotationDetail(ec, FEATURE_KEYS);
409                        if (!Util.isBlank(featureKeys)) {
410                                Yaml yaml = new Yaml();
411                                Map<String,Object> featureKeyMap = yaml.load(featureKeys);
412                                Object fKey = featureKeyMap.get(feature.getName());
413                                if (fKey instanceof String) {
414                                        return (String) fKey;
415                                }
416                        }
417                }
418                return Util.camelToKebab(feature.getName());
419        }

421        /**
422         * 
423         * @param eReference
424         * @return "Logical" opposite - either the "physical" opposite returned by {@link EReference}.getEOpposite() or a reference with a name specified in "opposite" Nasdanika annotation
425         * or a reference which specifies this reference in its "opposite" Nasdanika annotation.
426         */
427        public static EReference getOpposite(EReference eReference) {
428                EReference opposite = eReference.getEOpposite();
429                if (opposite != null) {
430                        return opposite;
431                }
432                String oName = getNasdanikaAnnotationDetail(eReference, "opposite");
433                EClass refType = eReference.getEReferenceType();
434                for (EReference ref: refType.getEAllReferences()) {
435                        if (ref.getName().equals(oName)) {
436                                return ref;
437                        }
438                        if (ref.getEReferenceType() == eReference.getEContainingClass() && eReference.getName().equals(getNasdanikaAnnotationDetail(ref, "opposite"))) {
439                                return ref;
440                        }
441                }
442                return null;
443        }

445        /**
446         * Computes end and duration from start and previously loaded things.
447         * @param period
448         * @param newStart
449         */
450        public static void onStart(Period period, Temporal newStart) {
451                if (newStart != null) {
452                        LoadTracker loadTracker = (LoadTracker) EcoreUtil.getRegisteredAdapter(period, LoadTracker.class);
453                        if (loadTracker != null && loadTracker.isLoading(NcorePackage.Literals.PERIOD__START)) {
454                                if (loadTracker.isLoaded(NcorePackage.Literals.PERIOD__END)) {
455                                        if (!loadTracker.isLoaded(NcorePackage.Literals.PERIOD__DURATION)) {
456                                                period.setDuration(period.getEnd().minus(newStart));
457                                        }
458                                } else if (loadTracker.isLoaded(NcorePackage.Literals.PERIOD__DURATION)) {
459                                        period.setEnd(newStart.plus(period.getDuration()));
460                                }
461                        }
462                }
463        }

465        public static void onEnd(Period period, Temporal newEnd) {
466                if (newEnd != null) {
467                        LoadTracker loadTracker = (LoadTracker) EcoreUtil.getRegisteredAdapter(period, LoadTracker.class);
468                        if (loadTracker != null && loadTracker.isLoading(NcorePackage.Literals.PERIOD__END)) {
469                                if (loadTracker.isLoaded(NcorePackage.Literals.PERIOD__START)) {
470                                        if (!loadTracker.isLoaded(NcorePackage.Literals.PERIOD__DURATION)) {
471                                                period.setDuration(newEnd.minus(period.getStart()));
472                                        }
473                                } else if (loadTracker.isLoaded(NcorePackage.Literals.PERIOD__DURATION)) {
474                                        period.setStart(newEnd.minus(period.getDuration()));
475                                }
476                        }
477                }
478        }

480        public static void onDuration(Period period, Duration newDuration) {
481                if (newDuration != null) {
482                        LoadTracker loadTracker = (LoadTracker) EcoreUtil.getRegisteredAdapter(period, LoadTracker.class);
483                        if (loadTracker != null && loadTracker.isLoading(NcorePackage.Literals.PERIOD__DURATION)) {
484                                if (loadTracker.isLoaded(NcorePackage.Literals.PERIOD__START)) {
485                                        if (!loadTracker.isLoaded(NcorePackage.Literals.PERIOD__END)) {
486                                                period.setEnd(period.getStart().plus(newDuration));
487                                        }
488                                } else if (loadTracker.isLoaded(NcorePackage.Literals.PERIOD__END)) {
489                                        period.setStart(period.getEnd().minus(newDuration));
490                                }
491                        }
492                }
493        }

495        /**
496         * @param eReference
497         * @return Target EClass of association class reference.
498         */
499        public static EClass getAssociationTarget(EReference eReference) {
500                String association = NcoreUtil.getNasdanikaAnnotationDetail(eReference, "association");
501                if (Util.isBlank(association)) {
502                        return null;
503                }
504                EClass refType = eReference.getEReferenceType();
505                EStructuralFeature associationFeature = refType.getEStructuralFeature(association);
506                if (associationFeature instanceof EReference) {
507                        return ((EReference) associationFeature).getEReferenceType();
508                }
509                return null;            
510        }
