001package org.nasdanika.graph.processor.emf;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.util.List;
006import java.util.Map;
007import java.util.Objects;
008import java.util.function.Supplier;
009import java.util.stream.Collectors;
010import java.util.stream.Stream;
011
012import org.eclipse.emf.common.util.URI;
013import org.eclipse.emf.ecore.EObject;
014import org.eclipse.emf.ecore.resource.impl.ResourceImpl;
015import org.nasdanika.common.ProgressMonitor;
016import org.nasdanika.graph.Element;
017import org.nasdanika.graph.processor.ConnectionProcessorConfig;
018import org.nasdanika.graph.processor.NodeProcessorConfig;
019import org.nasdanika.graph.processor.ProcessorConfig;
020import org.nasdanika.graph.processor.ProcessorFactory;
021import org.nasdanika.graph.processor.ProcessorInfo;
022
023
024/**
025 * Base class for resource which load their contents by creating a {@link Stream} of {@link Element}'s from a resource {@link URI} and then uses {@link ProcessorFactory} to create {@link Supplier}'s of {@link EObject} with {@link ProcessorFactory}. 
026 * @author Pavel
027 *
028 */
029public abstract class GraphProcessorResource<P, T extends EObject> extends ResourceImpl {
030        
031        protected abstract ProcessorFactory<P, ?, ?> getProcessorFactory();
032        
033        protected GraphProcessorResource(URI uri) {
034                super(uri);
035        }
036
037        /**
038         * Loads {@link Element}s from the input stream to pass to 
039         * @param inputStream
040         * @param options
041         * @return
042         * @throws IOException
043         */
044        protected abstract Stream<? extends Element> loadElements(InputStream inputStream, Map<?, ?> options) throws IOException;
045        
046        @Override
047        protected void doLoad(InputStream inputStream, Map<?, ?> options) throws IOException {
048                List<? extends Element> elements = loadElements(inputStream, options).collect(Collectors.toList());
049                Map<Element, ProcessorInfo<P>> registry = getProcessorFactory().createProcessors(elements.stream(), getProgressMonitor());
050                List<T> roots = getSemanticElements(registry).filter(this::isRoot).collect(Collectors.toList()); // TODO .forEach(getContents()::add) - fails with concurrent modification exception
051                getContents().addAll(roots);
052                for (Element element: elements) {
053                        ProcessorInfo<P> info = registry.get(element);
054                        ProcessorConfig<P> config = info.getConfig();
055                        if (config instanceof NodeProcessorConfig) {
056                                eAdapters().add(new NodeProcessorConfigAdapter<>((NodeProcessorConfig<P, ?, ?>) config));
057                        } else if (config instanceof ConnectionProcessorConfig) {
058                                eAdapters().add(new ConnectionProcessorConfigAdapter<>((ConnectionProcessorConfig<P, ?, ?>) config));                           
059                        } else {
060                                eAdapters().add(new ProcessorConfigAdapter<>(config));                          
061                        }
062                }
063        }
064
065        protected abstract ProgressMonitor getProgressMonitor();
066        
067        @SuppressWarnings("unchecked")
068        protected Stream<T> getSemanticElements(P processor) {
069                if (processor instanceof SemanticProcessor) {
070                        return ((SemanticProcessor<T>) processor).getSemanticElements().stream();
071                }
072                return Stream.empty();
073        };
074
075        /**
076         * Retrieves semantic elements {@link EObject}s from the registry.
077         * @param registry
078         * @return
079         */
080        protected Stream<T> getSemanticElements(Map<Element, ProcessorInfo<P>> registry) {
081                return registry.values().stream().map(pi -> pi.getProcessor()).filter(Objects::nonNull).flatMap(this::getSemanticElements);
082        }
083        
084        /**
085         * @param semanticElement 
086         * @return true if the argument semantic element shall be added to the resource contents.
087         * This implementation returns true if the semantic element is not null and is not contained by another semantic element, i.e. its eContainer() returns null.
088         */
089        protected boolean isRoot(EObject semanticElement) {
090                return semanticElement != null && semanticElement.eContainer() == null;
091        }
092                        
093}
