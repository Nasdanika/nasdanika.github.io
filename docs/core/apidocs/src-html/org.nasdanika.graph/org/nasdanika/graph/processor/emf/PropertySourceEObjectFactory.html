052        protected String getPropertyValue(org.nasdanika.graph.Element element, String propertyName) {
053                if (!org.nasdanika.common.Util.isBlank(propertyName) && element instanceof PropertySource) {
054                        @SuppressWarnings("unchecked")
055                        PropertySource<Object,?> propertySource = (PropertySource<Object,?>) element;
056                        for (String propertyPrefix: getPropertyPrefixes()) {
057                                Object propertyValue = propertySource.getProperty(propertyPrefix + propertyName);
058                                if (propertyValue instanceof String && !org.nasdanika.common.Util.isBlank((String) propertyValue)) {
059                                        return (String) propertyValue;
060                                }
061                        }
062                }
063                return null;
064        }

084        @Override
085        protected Collection<T> createSemanticElements(ProcessorConfig<P> config, ProgressMonitor progressMonitor) {
086                String spec = getPropertyValue(config.getElement(), getSpecPropertyName());
087                String specFormat = getPropertyValue(config.getElement(), getSpecFormatPropertyName());
088                if (!org.nasdanika.common.Util.isBlank(spec)) {
089                        return load(spec, specFormat, getBaseURI(), config, getLoadingContext(config, progressMonitor), progressMonitor);
090                }
091                                
092                // Spec URI
093                String specUriStr = getPropertyValue(config.getElement(), getSpecUriPropertyName());
094                if (!org.nasdanika.common.Util.isBlank(specUriStr)) {
095                        URI specURI = URI.createURI(specUriStr);
096                        if (specURI.isRelative()) {
097                                URI baseURI = getBaseURI();
098                                if (baseURI != null && !baseURI.isRelative() && baseURI.isHierarchical()) {
099                                        specURI = specURI.resolve(baseURI);
100                                }
101                        }
102                        return load(specURI, specFormat, config, getLoadingContext(config, progressMonitor), progressMonitor);
103                }               
104                
105                // Semantic URI
106                String semanticUriStr = getPropertyValue(config.getElement(), getSemanticUriPropertyName());
107                if (!org.nasdanika.common.Util.isBlank(semanticUriStr)) {
108                        URI semanticURI = URI.createURI(semanticUriStr);
109                        if (semanticURI.isRelative()) {
110                                URI baseURI = getBaseURI();
111                                if (baseURI != null && !baseURI.isRelative() && baseURI.isHierarchical()) {
112                                        semanticURI = semanticURI.resolve(baseURI);
113                                }
114                        }
115                        return load(semanticURI, specFormat, config, getLoadingContext(config, progressMonitor), progressMonitor);
116                }
117                
118                
119                return Collections.emptyList();
120        }

493                return "child-references";

496        @Override
497        protected EReference getChildReference(ProcessorConfig<P> config, T semanticElement, Element child,     ProcessorInfo<P> childProcessorInfo, T semanticChild) {
498                if (config == null) {
499                        return null;
500                }
501                String value = getPropertyValue(config.getElement(), getChildReferencesPropertyName());
502                if (org.nasdanika.common.Util.isBlank(value)) {
503                        return null;
504                }
505                Yaml yaml = new Yaml();
506                Object yamlValue = yaml.load(value);
507                if (yamlValue instanceof Map) {
508                        Map<?, ?> yamlMap = (Map<?,?>) yamlValue;
509                        for (Entry<?, ?> me: yamlMap.entrySet()) {
510                                Object key = me.getKey();
511                                if (key instanceof String) {
512                                        Object eValue = me.getValue();
513                                        if (eValue instanceof String) {
514                                                ExpressionParser parser = new SpelExpressionParser();
515                                                Expression exp = parser.parseExpression((String) eValue);
516                                                EvaluationContext evaluationContext = createEvaluationContext();
517                                                evaluationContext.setVariable("parentConfig", config);
518                                                evaluationContext.setVariable("parent", semanticElement);
519                                                evaluationContext.setVariable("parentElement", config.getElement());
520                                                evaluationContext.setVariable("element", child);
521                                                evaluationContext.setVariable("config", childProcessorInfo.getConfig());
522                                                try {
523                                                        if (exp.getValue(evaluationContext, childProcessorInfo.getProcessor(), Boolean.class)) {
524                                                                EClass eClass = semanticElement.eClass();
525                                                                EStructuralFeature ref = eClass.getEStructuralFeature((String) key);
526                                                                if (ref instanceof EReference) {
527                                                                        return (EReference) ref;
528                                                                }
529                                                                throw new ConfigurationException("Reference '" + key + "' not found in " + eClass.getName() + " " + eClass.getEPackage().getNsURI(), getMarked(config));
530                                                        }
531                                                } catch (EvaluationException e) {
532                                                        // NOP - continue
533                                                }
534                                        } else {
535                                                throw new IllegalArgumentException(getChildReferencesPropertyName() + " map values shall be strings, got " + eValue);                                           
536                                        }
537                                } else {
538                                        throw new IllegalArgumentException(getChildReferencesPropertyName() + " map keys shall be strings, got " + key);
539                                }
540                        }                                               
541                }
542                throw new IllegalArgumentException(getChildReferencesPropertyName() + " value shall be a YAML map, got " + yamlValue);
543        }

573        @Override
574        protected EReference getRegistryReference(ProcessorConfig<P> config, P processor, T semanticElement, Element registryElement, ProcessorInfo<P> registryElementProcessorInfo, T registrySemanticElement) {               
575                if (registrySemanticElement == null) {
576                        return null;
577                }
578                
579                String value = getPropertyValue(config.getElement(), getRegistryReferencesPropertyName());
580                if (org.nasdanika.common.Util.isBlank(value)) {
581                        return null;
582                }
583                Yaml yaml = new Yaml();
584                Object yamlValue = yaml.load(value);
585                if (yamlValue instanceof Map) {
586                        Map<?, ?> yamlMap = (Map<?,?>) yamlValue;
587                        for (Entry<?, ?> me: yamlMap.entrySet()) {
588                                Object key = me.getKey();
589                                if (key instanceof String) {
590                                        Object eValue = me.getValue();
591                                        if (eValue instanceof String) {
592                                                ExpressionParser parser = new SpelExpressionParser();
593                                                Expression exp = parser.parseExpression((String) eValue);
594                                                EvaluationContext evaluationContext = createEvaluationContext();
595                                                evaluationContext.setVariable("config", config);
596                                                evaluationContext.setVariable("element", config.getElement());
597                                                evaluationContext.setVariable("registryElement", registryElement);
598                                                evaluationContext.setVariable("registryConfig", registryElementProcessorInfo.getConfig());
599                                                evaluationContext.setVariable("semanticElement", semanticElement);
600                                                try {
601                                                        if (exp.getValue(evaluationContext, registrySemanticElement, Boolean.class)) {
602                                                                EClass eClass = semanticElement.eClass();
603                                                                EStructuralFeature ref = eClass.getEStructuralFeature((String) key);
604                                                                if (ref instanceof EReference) {
605                                                                        return (EReference) ref;
606                                                                }
607                                                                throw new ConfigurationException("Reference '" + key + "' not found in " + eClass.getName() + " " + eClass.getEPackage().getNsURI(), getMarked(config));
608                                                        }
609                                                } catch (EvaluationException e) {
610                                                        // NOP - continue
611                                                }
612                                        } else {
613                                                throw new IllegalArgumentException(getRegistryReferencesPropertyName() + " map values shall be strings, got " + eValue);                                                
614                                        }
615                                } else {
616                                        throw new IllegalArgumentException(getRegistryReferencesPropertyName() + " map keys shall be strings, got " + key);
617                                }
618                        }                                               
619                }
620                throw new IllegalArgumentException(getRegistryReferencesPropertyName() + " value shall be a YAML map, got " + yamlValue);
621        }
