052        public ReflectiveProcessorFactory(IntrospectionLevel introspectionLevel, Object... targets) {
053                this.introspectionLevel = introspectionLevel == null ? IntrospectionLevel.NONE : introspectionLevel;
054                this.targets = targets;

057        public Map<Element, ProcessorInfo<P>> createProcessors(Element element, IntrospectionLevel introspectionLevel, ProgressMonitor progressMonitor, Object... registryTargets) {
058                Map<Element, ProcessorInfo<P>> registry = ProcessorFactory.super.createProcessors(progressMonitor, element);
059                for (Object registryTarget: registryTargets) {
060                        // TODO  - progress moinitor                    
061                        wireRegistryEntry(registryTarget, introspectionLevel).accept(registry);
062                        wireRegistry(registryTarget, introspectionLevel).accept(registry);
063                }
064                return registry;
065        }

067        @Override
068        public ProcessorInfo<P> createProcessor(ProcessorConfig<P> config, ProgressMonitor progressMonitor) {
069                if (introspectionLevel != IntrospectionLevel.NONE) {
070                        for (Object target: targets) {
071                                // TODO - split
072                                ProcessorInfo<P> elementProcessorInfo = createProcessor(target, config, introspectionLevel, progressMonitor);
073                                if (elementProcessorInfo != null) {
074                                        return elementProcessorInfo;
075                                }
076                        }
077                }
078                return ProcessorFactory.super.createProcessor(config, progressMonitor);                 
079        }

081        @SuppressWarnings("unchecked")
082        protected ProcessorInfo<P> createProcessor(Object target, ProcessorConfig<P> config, IntrospectionLevel introspectionLevel, ProgressMonitor progressMonitor) {  
083                if (target != null && (target.getClass().getAnnotation(Factory.class) == null || matchPredicate(config.getElement(), target.getClass().getAnnotation(Factory.class).value()))) {
084                        // TODO - progress steps.
085                        Optional<Method> match = getMethods(target.getClass(), introspectionLevel)
086                                .filter(m -> matchFactoryMethod(config, m))
087                                .sorted(this::compareProcessorMethods)
088                                .findFirst();
089                        
090                        if (match.isPresent()) {
091                                Method method = match.get();
092                                Object processor;
093                                if (method.getParameterCount() == 0) {
094                                        processor = invokeMethod(target, method);
095                                } else {
096                                        processor = invokeMethod(target, method, config);
097                                }
098                                if (processor != null) {
099                                        Processor elementProcessorAnnotation = method.getAnnotation(Processor.class);
100                                        IntrospectionLevel processorIntrospectionLevel = elementProcessorAnnotation.introspect();
101                                        if (processorIntrospectionLevel == IntrospectionLevel.NONE) {
102                                                return ProcessorFactory.super.createProcessor(config, progressMonitor);
103                                        }
104                                        
105                                        boolean hideWired = elementProcessorAnnotation.hideWired();
106                                        Map<Element, ProcessorInfo<P>> unwiredChildProcessorsInfo = wireChildProcessor(processor, config.getChildProcessorsInfo(), processorIntrospectionLevel);
107                                        wireChildProcessors(processor, hideWired ? unwiredChildProcessorsInfo : config.getChildProcessorsInfo(), processorIntrospectionLevel);
108
109                                        Collection<Throwable> failures = new ArrayList<>();
110                                        BiFunction<Void, Throwable, ProcessorInfo<P>> failureHandler  = (result, failure) -> {
111                                                if (failure != null) {
112                                                        failures.add(failure);
113                                                }
114                                                return null;
115                                        };
116
117                                        Consumer<ProcessorInfo<P>> pc = wireParentProcessor(processor, processorIntrospectionLevel);
118                                        if (pc != null) {
119                                                config.getParentProcessorInfo().thenAccept(pc).handle(failureHandler);
120                                        }
121                                        wireProcessorElement(processor, config.getElement(), processorIntrospectionLevel);
122                                        
123                                        config.getRegistry().thenAccept(wireRegistryEntry(processor, processorIntrospectionLevel)).handle(failureHandler);
124                                        
125                                        Consumer<Map<Element, ProcessorInfo<P>>> rc = wireRegistry(processor, processorIntrospectionLevel);
126                                        if (rc != null) {
127                                                config.getRegistry().thenAccept(rc).handle(failureHandler);
128                                        }
129                                        
130                                        ProcessorConfig<P> unwiredConfig;
131                                        
132                                        if (config instanceof NodeProcessorConfig) {
133                                                NodeProcessorConfig<P, H, E> nodeProcessorConfig = (NodeProcessorConfig<P, H, E>) config;
134                                                Map<Connection, CompletionStage<E>> unwiredIncomingEndpoints = wireIncomingEndpoint(processor, nodeProcessorConfig.getIncomingEndpoints(), processorIntrospectionLevel, failureHandler);
135                                                wireIncomingEndpoints(processor, hideWired ? unwiredIncomingEndpoints : nodeProcessorConfig.getIncomingEndpoints(), processorIntrospectionLevel);
136                                                
137                                                Map<Connection, Consumer<H>> unwiredIncomingHandlerConsumers = wireIncomingHandler(processor, nodeProcessorConfig.getIncomingHandlerConsumers(), processorIntrospectionLevel);
138                                                wireIncomingHandlerConsumers(processor, hideWired ? unwiredIncomingHandlerConsumers : nodeProcessorConfig.getIncomingHandlerConsumers(), processorIntrospectionLevel);
139                                                
140                                                Map<Connection, CompletionStage<E>> unwiredOutgoingEndpoints = wireOutgoingEndpoint(processor, nodeProcessorConfig.getOutgoingEndpoints(), processorIntrospectionLevel, failureHandler);
141                                                wireOutgoingEndpoints(processor, hideWired ? unwiredOutgoingEndpoints : nodeProcessorConfig.getOutgoingEndpoints(), processorIntrospectionLevel);
142                                                
143                                                Map<Connection, Consumer<H>> unwiredOutgoingHandlerConsumers = wireOutgoingHandler(processor, nodeProcessorConfig.getOutgoingHandlerConsumers(), processorIntrospectionLevel);
144                                                wireOutgoingHandlerConsumers(processor, hideWired ? unwiredOutgoingHandlerConsumers : nodeProcessorConfig.getOutgoingHandlerConsumers(), processorIntrospectionLevel);
145                                                
146                                                unwiredConfig = new NodeProcessorConfig<P, H, E>() {
147        
148                                                        @Override
149                                                        public Map<Element, ProcessorInfo<P>> getChildProcessorsInfo() {
150                                                                return unwiredChildProcessorsInfo;
151                                                        }
152        
153                                                        @Override
154                                                        public CompletionStage<ProcessorInfo<P>> getParentProcessorInfo() {
155                                                                return config.getParentProcessorInfo();
156                                                        }
157        
158                                                        @Override
159                                                        public CompletionStage<Map<Element, ProcessorInfo<P>>> getRegistry() {
160                                                                return config.getRegistry();
161                                                        }
162        
163                                                        @Override
164                                                        public Node getElement() {
165                                                                return (Node) config.getElement();
166                                                        }
167        
168                                                        @Override
169                                                        public Map<Connection, CompletionStage<E>> getIncomingEndpoints() {
170                                                                return unwiredIncomingEndpoints;
171                                                        }
172        
173                                                        @Override
174                                                        public Map<Connection, Consumer<H>> getIncomingHandlerConsumers() {
175                                                                return unwiredIncomingHandlerConsumers;
176                                                        }
177        
178                                                        @Override
179                                                        public Map<Connection, CompletionStage<E>> getOutgoingEndpoints() {
180                                                                return unwiredOutgoingEndpoints;
181                                                        }
182        
183                                                        @Override
184                                                        public Map<Connection, Consumer<H>> getOutgoingHandlerConsumers() {
185                                                                return unwiredOutgoingHandlerConsumers;
186                                                        }
187                                                };
188                                        } else if (config instanceof ConnectionProcessorConfig) {
189                                                ConnectionProcessorConfig<P, H, E> connectionProcessorConfig = (ConnectionProcessorConfig<P, H, E>) config;
190                                                boolean wiredSourceEndpoint = wireSourceEndpoint(processor, connectionProcessorConfig.getSourceEndpoint(), processorIntrospectionLevel, failureHandler);
191                                                boolean wiredSourceHandler = wireSourceHandler(processor, connectionProcessorConfig, processorIntrospectionLevel);
192                                                boolean wiredTargetEndpoint = wireTargetEndpoint(processor, connectionProcessorConfig.getTargetEndpoint(), processorIntrospectionLevel, failureHandler);
193                                                boolean wiredTargetHandler = wireTargetHandler(processor, connectionProcessorConfig, processorIntrospectionLevel);
194                                                
195                                                unwiredConfig = new ConnectionProcessorConfig<P, H, E>() {
196        
197                                                        @Override
198                                                        public Map<Element, ProcessorInfo<P>> getChildProcessorsInfo() {
199                                                                return unwiredChildProcessorsInfo;
200                                                        }
201                                                        
202                                                        @Override
203                                                        public CompletionStage<ProcessorInfo<P>> getParentProcessorInfo() {
204                                                                return config.getParentProcessorInfo();
205                                                        }
206        
207                                                        @Override
208                                                        public CompletionStage<Map<Element, ProcessorInfo<P>>> getRegistry() {
209                                                                return config.getRegistry();
210                                                        }
211        
212                                                        @Override
213                                                        public Connection getElement() {
214                                                                return (Connection) config.getElement();
215                                                        }
216        
217                                                        @Override
218                                                        public CompletionStage<E> getSourceEndpoint() {
219                                                                return wiredSourceEndpoint ? null : connectionProcessorConfig.getSourceEndpoint();
220                                                        }
221        
222                                                        @Override
223                                                        public void setSourceHandler(H sourceHandler) {
224                                                                if (wiredSourceHandler) {
225                                                                        throw new IllegalStateException("Source handler is already wired for " + getElement());
226                                                                }
227                                                                connectionProcessorConfig.setSourceHandler(sourceHandler);
228                                                        }
229        
230                                                        @Override
231                                                        public CompletionStage<E> getTargetEndpoint() {
232                                                                return wiredTargetEndpoint ? null : connectionProcessorConfig.getTargetEndpoint();
233                                                        }
234        
235                                                        @Override
236                                                        public void setTargetHandler(H targetHandler) {
237                                                                if (wiredTargetHandler) {
238                                                                        throw new IllegalStateException("Target handler is already wired for " + getElement());
239                                                                }
240                                                                connectionProcessorConfig.setTargetHandler(targetHandler);
241                                                        }
242                                                };
243                                        } else {
244                                                unwiredConfig = new ProcessorConfig<P>() {
245        
246                                                        @Override
247                                                        public Map<Element, ProcessorInfo<P>> getChildProcessorsInfo() {
248                                                                return unwiredChildProcessorsInfo;
249                                                        }
250                                                        
251                                                        @Override
252                                                        public CompletionStage<ProcessorInfo<P>> getParentProcessorInfo() {
253                                                                return config.getParentProcessorInfo();
254                                                        }
255        
256                                                        @Override
257                                                        public CompletionStage<Map<Element, ProcessorInfo<P>>> getRegistry() {
258                                                                return config.getRegistry();
259                                                        }
260        
261                                                        @Override
262                                                        public Element getElement() {
263                                                                return config.getElement();
264                                                        }
265        
266                                                };
267                                        }
268
269                                        return ProcessorInfo.of(hideWired ? unwiredConfig : config, (P) processor, () -> failures);
270                                }
271                        }
272                        
273                        // Factories
274                        List<Entry<Object, IntrospectionLevel>> factories = getFieldsAndMethods(target.getClass(), introspectionLevel)
275                                        .filter(ae -> ae.getAnnotation(Factory.class) != null)
276                                        .filter(ae -> matchPredicate(config.getElement(), ae.getAnnotation(Factory.class).value()))
277                                        .filter(ae -> mustGet(ae, null, "Methods annotated with Factory shall have no parameters: " + ae))
278                                        .sorted((a, b) -> b.getAnnotation(Factory.class).priority() - a.getAnnotation(Factory.class).priority())
279                                        .map(ae -> Map.entry(get(target, ae), ae.getAnnotation(Factory.class).introspect()))
280                                        .collect(Collectors.toList());
281                        
282                        for (Entry<Object, IntrospectionLevel> factory: factories) {
283                                // TODO - sub-monitors
284                                ProcessorInfo<P> elementProcessorInfo = createProcessor(factory.getKey(), config, factory.getValue(), progressMonitor);
285                                if (elementProcessorInfo != null) {
286                                        return elementProcessorInfo;
287                                }
288                        }
289                        
290                        factories = getFieldsAndMethods(target.getClass(), introspectionLevel)
291                                        .filter(ae -> ae.getAnnotation(Factories.class) != null)
292                                        .filter(ae -> matchPredicate(config.getElement(), ae.getAnnotation(Factories.class).value()))
293                                        .filter(ae -> mustGet(ae, Collection.class, "Methods and fields annotated with Factories shall be of type/return Collection. Methods shall have no parameters: " + ae))
294                                        .sorted((a, b) -> b.getAnnotation(Factories.class).priority() - a.getAnnotation(Factories.class).priority())
295                                        .flatMap(ae -> ((Collection<Object>) get(target, ae)).stream().map(e -> Map.entry(e, ae.getAnnotation(Factories.class).introspect())))
296                                        .collect(Collectors.toList());
297                        
298                        for (Entry<Object, IntrospectionLevel> factory: factories) {
299                                // TODO - sub-monitors
300                                ProcessorInfo<P> elementProcessorInfo = createProcessor(factory.getKey(), config, factory.getValue(), progressMonitor);
301                                if (elementProcessorInfo != null) {
302                                        return elementProcessorInfo;
303                                }
304                        }                       
305                        
306                }
307                return null;
308        }

310        protected int compareProcessorMethods(Method a, Method b) {
311                int priorityCmp = b.getAnnotation(Processor.class).priority() - a.getAnnotation(Processor.class).priority();
312                if (priorityCmp != 0) {
313                        return priorityCmp;
314                }
315                
316                Class<Element> aType = a.getAnnotation(Processor.class).type();
317                Class<Element> bType = b.getAnnotation(Processor.class).type();
318                if (!Objects.equals(aType, bType)) {
319                        if (aType.isAssignableFrom(bType)) {
320                                // b is more specific
321                                return 1;
322                        }
323                        if (bType.isAssignableFrom(aType)) {
324                                // a is more specific
325                                return -1;
326                        }
327                }
328                
329                // Taking config is more specific 
330                int paramCountCmp = b.getParameterCount() - a.getParameterCount();
331                if (paramCountCmp != 0) {
332                        return paramCountCmp;
333                }
334                
335                return a.getName().compareTo(b.getName());
336        }

338        protected Consumer<Map<Element, ProcessorInfo<P>>> wireRegistryEntry(Object processor, IntrospectionLevel processorIntrospectionLevel) {
339                List<AccessibleObject> registryEntrySetters = getFieldsAndMethods(processor.getClass(), processorIntrospectionLevel)
340                        .filter(ae -> ae.getAnnotation(RegistryEntry.class) != null)
341                        .filter(ae -> mustSet(ae, null, "Fields/methods annotated with RegistryEntry must have (parameter) type assignable from the processor type or ProcessorConfig if config is set to true: " + ae))
342                        .collect(Collectors.toList());
343                
344                return registry -> {
345                        registryEntrySetters.forEach(setter -> {
346                                for (Entry<Element, ProcessorInfo<P>> re: registry.entrySet()) {
347                                        RegistryEntry registryEntryAnnotation = setter.getAnnotation(RegistryEntry.class);
348                                        if (matchPredicate(re.getKey(), registryEntryAnnotation.value())) {
349                                                set(processor, setter, registryEntryAnnotation.config() ? re.getValue().getConfig() : re.getValue().getProcessor());
350                                        };                                              
351                                }                               
352                        });                                             
353                };
354        }

426        /**
427         * Matches processor field or method and incoming connection.
428         * @return
429         */
430        protected boolean matchIncomingHandler(AnnotatedElement handlerMember, Connection incomingConnection) {
431                IncomingHandler incomingHandlerAnnotation = handlerMember.getAnnotation(IncomingHandler.class);
432                if (incomingHandlerAnnotation == null) {
433                        return false;
434                }
435                
436                if (handlerMember instanceof Method) {
437                        Method handlerMethod = (Method) handlerMember;
438                        int pc = handlerMethod.getParameterCount();
439                        if (pc > 1) {
440                                throw new NasdanikaException("A method annotated with IncomingHandler shall have zero or one parameter: " + handlerMethod);
441                        }
442                        if (pc == 1 && !handlerMethod.getParameterTypes()[0].isInstance(incomingConnection)) {
443                                throw new NasdanikaException("A single parameter type of a method annotated with IncomingHandler shall be assignable from Connection: " + handlerMethod);                               
444                        }
445                }
446                                
447                return matchPredicate(incomingConnection, incomingHandlerAnnotation.value());
448        }

451        @SuppressWarnings("unchecked")
452        protected Map<Connection, Consumer<H>> wireIncomingHandler(
453                        Object processor, 
454                        Map<Connection, Consumer<H>> incomingHandlerConsumers,
455                        IntrospectionLevel processorIntrospectionLevel) {
456                
457                Map<Connection, Consumer<H>> ret = new LinkedHashMap<>(incomingHandlerConsumers);
458
459                // Streaming fields and methods and then flat mapping them to all permutations with incoming handler consumers.
460                // then filtering using matchIncomingHandler, sorting by priority, finding first, wiring it and removing from ret.
461                getFieldsAndMethods(processor.getClass(), processorIntrospectionLevel)
462                        .filter(m -> !Modifier.isAbstract(((Member) m).getModifiers()))
463                        .flatMap(ae -> incomingHandlerConsumers.entrySet().stream().map(ihce -> Map.entry(ae, ihce)))
464                        .filter(e -> matchIncomingHandler(e.getKey(), e.getValue().getKey()))
465                        .sorted((a, b) -> b.getKey().getAnnotation(IncomingHandler.class).priority() - a.getKey().getAnnotation(IncomingHandler.class).priority())
466                        .findFirst().ifPresent(e -> {
467                                AccessibleObject handlerMember = e.getKey();
468                                Entry<Connection, Consumer<H>> incomingHandlerConsumerEntry = e.getValue();
469                                if (handlerMember instanceof Field) {                                   
470                                        incomingHandlerConsumerEntry.getValue().accept((H) getFieldValue(processor, (Field) handlerMember));
471                                } else {
472                                        Method handlerSupplierMethod = (Method) handlerMember;
473                                        Object incomingHandler = handlerSupplierMethod.getParameterCount() == 0 ? invokeMethod(processor, handlerSupplierMethod) : invokeMethod(processor, handlerSupplierMethod, incomingHandlerConsumerEntry.getKey());
474                                        incomingHandlerConsumerEntry.getValue().accept((H) incomingHandler);
475                                }
476                                ret.remove(incomingHandlerConsumerEntry.getKey());
477                        });
478                                
479                return ret;
480        }

497        protected boolean matchIncomingEndpoint(AnnotatedElement endpointMember, Connection incomingConnection) {
498                IncomingEndpoint incomingEndpointAnnotation = endpointMember.getAnnotation(IncomingEndpoint.class);
499                if (incomingEndpointAnnotation == null) {
500                        return false;
501                }
502                
503                if (endpointMember instanceof Method) {
504                        Method endpointMethod = (Method) endpointMember;
505                        int pc = endpointMethod.getParameterCount();
506                        if (pc == 0 || pc > 2) {
507                                throw new NasdanikaException("A method annotated with IncomingEndpoint shall have one or two parameters: " + endpointMethod);
508                        }
509                        if (!endpointMethod.getParameterTypes()[0].isInstance(incomingConnection)) {
510                                throw new NasdanikaException("The first parameter type of a method annotated with IncomingEndpoint shall be assignable from Connection: " + endpointMethod);                            
511                        }
512                }
513                                
514                return matchPredicate(incomingConnection, incomingEndpointAnnotation.value());
515        }

517        protected Map<Connection, CompletionStage<E>> wireIncomingEndpoint(
518                        Object processor, 
519                        Map<Connection, CompletionStage<E>> incomingEndpoints,
520                        IntrospectionLevel processorIntrospectionLevel,
521                        BiFunction<Void, Throwable, ProcessorInfo<P>> failureHandler) {
522                                                
523                Map<Connection, CompletionStage<E>> ret = new LinkedHashMap<>(incomingEndpoints);
524
525                // Streaming fields and methods and then flat mapping them to all permutations with incoming endpoints.
526                // then filtering using matchIncomingEndpoint, sorting by priority, finding first, wiring it and removing from ret.
527                getFieldsAndMethods(processor.getClass(), processorIntrospectionLevel)
528                        .filter(m -> !Modifier.isAbstract(((Member) m).getModifiers()))
529                        .flatMap(ae -> incomingEndpoints.entrySet().stream().map(iee -> Map.entry(ae, iee)))
530                        .filter(e -> matchIncomingEndpoint(e.getKey(), e.getValue().getKey()))
531                        .sorted((a, b) -> b.getKey().getAnnotation(IncomingEndpoint.class).priority() - a.getKey().getAnnotation(IncomingEndpoint.class).priority())
532                        .findFirst().ifPresent(e -> {
533                                AccessibleObject endpointMember = e.getKey();
534                                Entry<Connection, CompletionStage<E>> incomingEndpointEntry = e.getValue();
535                                incomingEndpointEntry.getValue().thenAccept(incomingEndpoint -> {
536                                        if (endpointMember instanceof Field) {
537                                                setFieldValue(processor, (Field) endpointMember, incomingEndpoint);
538                                        } else {
539                                                Method endpointMethod = (Method) endpointMember;
540                                                if (endpointMethod.getParameterCount() == 1) {
541                                                        invokeMethod(processor, endpointMethod, incomingEndpoint);
542                                                } else {
543                                                        invokeMethod(processor, endpointMethod, incomingEndpointEntry.getKey(), incomingEndpoint);                                              
544                                                }
545                                        }                                       
546                                }).handle(failureHandler);
547                                ret.remove(incomingEndpointEntry.getKey());
548                        });
549                                
550                return ret;
551        }

553        protected void wireIncomingEndpoints(
554                        Object processor, 
555                        Map<Connection, CompletionStage<E>> incomingEndpoints,
556                        IntrospectionLevel processorIntrospectionLevel) {
557                
558                getFieldsAndMethods(processor.getClass(), processorIntrospectionLevel)
559                                .filter(ae -> ae.getAnnotation(IncomingEndpoints.class) != null)

561                                .forEach(setter -> set(processor, setter, incomingEndpoints));

565         * Matches processor field or method and outgoing connection.
566         * @return
567         */
568        protected boolean matchOutgoingHandler(AnnotatedElement handlerMember, Connection outgoingConnection) {
569                OutgoingHandler outgoingHandlerAnnotation = handlerMember.getAnnotation(OutgoingHandler.class);
570                if (outgoingHandlerAnnotation == null) {
571                        return false;
572                }
573                
574                if (handlerMember instanceof Method) {
575                        Method handlerMethod = (Method) handlerMember;
576                        int pc = handlerMethod.getParameterCount();
577                        if (pc > 1) {
578                                throw new NasdanikaException("A method annotated with OutgoingHandler shall have zero or one parameter: " + handlerMethod);
579                        }
580                        if (pc == 1 && !handlerMethod.getParameterTypes()[0].isInstance(outgoingConnection)) {
581                                throw new NasdanikaException("A single parameter type of a method annotated with OutgoingHandler shall be assignable from Connection: " + handlerMethod);                               
582                        }
583                }
584                                
585                return matchPredicate(outgoingConnection, outgoingHandlerAnnotation.value());
586        }

588        @SuppressWarnings("unchecked")
589        protected Map<Connection, Consumer<H>> wireOutgoingHandler(
590                        Object processor, 
591                        Map<Connection, Consumer<H>> outgoingHandlerConsumers,
592                        IntrospectionLevel processorIntrospectionLevel) {
593
594                Map<Connection, Consumer<H>> ret = new LinkedHashMap<>(outgoingHandlerConsumers);
595
596                // Streaming fields and methods and then flat mapping them to all permutations with outgoing handler consumers.
597                // then filtering using matchOutgoingHandler, sorting by priority, finding first, wiring it and removing from ret.
598                getFieldsAndMethods(processor.getClass(), processorIntrospectionLevel)
599                        .filter(m -> !Modifier.isAbstract(((Member) m).getModifiers()))
600                        .flatMap(ae -> outgoingHandlerConsumers.entrySet().stream().map(ihce -> Map.entry(ae, ihce)))
601                        .filter(e -> matchOutgoingHandler(e.getKey(), e.getValue().getKey()))
602                        .sorted((a, b) -> b.getKey().getAnnotation(OutgoingHandler.class).priority() - a.getKey().getAnnotation(OutgoingHandler.class).priority())
603                        .findFirst().ifPresent(e -> {
604                                AccessibleObject handlerMember = e.getKey();
605                                Entry<Connection, Consumer<H>> outgoingHandlerConsumerEntry = e.getValue();
606                                if (handlerMember instanceof Field) {                                   
607                                        outgoingHandlerConsumerEntry.getValue().accept((H) getFieldValue(processor, (Field) handlerMember));
608                                } else {
609                                        Method handlerSupplierMethod = (Method) handlerMember;
610                                        Object incomingHandler = handlerSupplierMethod.getParameterCount() == 0 ? invokeMethod(processor, handlerSupplierMethod) : invokeMethod(processor, handlerSupplierMethod, outgoingHandlerConsumerEntry.getKey());
611                                        outgoingHandlerConsumerEntry.getValue().accept((H) incomingHandler);
612                                }
613                                ret.remove(outgoingHandlerConsumerEntry.getKey());
614                        });
615                                
616                return ret;
617        }

654        protected Map<Connection, CompletionStage<E>> wireOutgoingEndpoint(
655                        Object processor, 
656                        Map<Connection, CompletionStage<E>> outgoingEndpoints,
657                        IntrospectionLevel processorIntrospectionLevel,
658                        BiFunction<Void, Throwable, ProcessorInfo<P>> failureHandler) {
659                                                
660                Map<Connection, CompletionStage<E>> ret = new LinkedHashMap<>(outgoingEndpoints);
661
662                // Streaming fields and methods and then flat mapping them to all permutations with outgoing endpoints.
663                // then filtering using matchOutgoingEndpoint, sorting by priority, finding first, wiring it and removing from ret.
664                getFieldsAndMethods(processor.getClass(), processorIntrospectionLevel)
665                        .filter(m -> !Modifier.isAbstract(((Member) m).getModifiers()))
666                        .flatMap(ae -> outgoingEndpoints.entrySet().stream().map(iee -> Map.entry(ae, iee)))
667                        .filter(e -> matchOutgoingEndpoint(e.getKey(), e.getValue().getKey()))
668                        .sorted((a, b) -> b.getKey().getAnnotation(OutgoingEndpoint.class).priority() - a.getKey().getAnnotation(OutgoingEndpoint.class).priority())
669                        .findFirst().ifPresent(e -> {
670                                AccessibleObject endpointMember = e.getKey();
671                                Entry<Connection, CompletionStage<E>> outgoingEndpointEntry = e.getValue();
672                                outgoingEndpointEntry.getValue().thenAccept(outgoingEndpoint -> {
673                                        if (endpointMember instanceof Field) {
674                                                setFieldValue(processor, (Field) endpointMember, outgoingEndpoint);
675                                        } else {
676                                                Method endpointMethod = (Method) endpointMember;
677                                                if (endpointMethod.getParameterCount() == 1) {
678                                                        invokeMethod(processor, endpointMethod, outgoingEndpoint);
679                                                } else {
680                                                        invokeMethod(processor, endpointMethod, outgoingEndpointEntry.getKey(), outgoingEndpoint);                                              
681                                                }
682                                        }                                       
683                                }).handle(failureHandler);
684                                ret.remove(outgoingEndpointEntry.getKey());
685                        });
686                                
687                return ret;
688        }

782        protected boolean matchFactoryMethod(ProcessorConfig<P> elementProcessorConfig, Method method) {
783                if (Modifier.isAbstract(method.getModifiers())) {
784                        return false;
785                }
786                
787                Processor elementProcessorAnnotation = method.getAnnotation(Processor.class);
788                if (elementProcessorAnnotation == null) {
789                        return false;
790                }
791                
792                Element element = elementProcessorConfig.getElement();
793                if (!elementProcessorAnnotation.type().isInstance(element)) {
794                        return false;
795                }
796                
797                if (method.getParameterCount() > 1) {
798                        return false;
799                }
800                
801                if (method.getParameterCount() == 1 && !method.getParameterTypes()[0].isInstance(elementProcessorConfig)) {
802                        return false;
803                }
804                
805                return matchPredicate(elementProcessorConfig.getElement(), elementProcessorAnnotation.value());
806        }

808        /**
809         * Parses and evaluates expression using <a href="https://docs.spring.io/spring-framework/docs/current/reference/html/core.html#expressions">Spring Expression Language</a> 
810         * @param obj
811         * @param expr 
812         * @return true if expression is blank or evaluates to true, false if the expression evaluates to false or throws EvaluationException.
813         */
814        protected boolean matchPredicate(Object obj, String expr) {
815                if (Util.isBlank(expr)) {
816                        return true;
817                }
818                
819                ExpressionParser parser = new SpelExpressionParser();
820                Expression exp = parser.parseExpression(expr);
821                EvaluationContext evaluationContext = getEvaluationContext();
822                try {                   
823                        return evaluationContext == null ? exp.getValue(obj, Boolean.class) : exp.getValue(evaluationContext, obj, Boolean.class);
824                } catch (EvaluationException e) {
825                        return false;
826                }
827        }

829        /**
830         * @return A stream of methods. Accessible methods for the introspection level ACCESSIBLE and declared methods from the class and all super classes and implemented interfaces for the introspection level DECLARED.
831         */
832        protected static Stream<Method> getMethods(Class<?> clazz, IntrospectionLevel introspectionLevel) {
833                if (clazz == null || introspectionLevel == null || introspectionLevel == IntrospectionLevel.NONE || Object.class.equals(clazz)) {
834                        return Stream.empty();
835                }
836                
837                if (introspectionLevel == IntrospectionLevel.ACCESSIBLE) {
838                        return Arrays.stream(clazz.getMethods());
839                }
840                
841                Stream<Method> declared = Arrays.stream(clazz.getDeclaredMethods());
842                
843                Class<?> superClass = clazz.getSuperclass();
844                declared = Stream.concat(declared, getMethods(superClass, introspectionLevel));
845                
846                for (Class<?> implementedInterface: clazz.getInterfaces()) {
847                        declared = Stream.concat(declared, getMethods(implementedInterface, introspectionLevel));
848                        
849                }
850                return declared;
851        }
852        
853        /**
854         * @return A stream of fields. Accessible fields for the introspection level ACCESSIBLE and declared fields from the class and all super classes and implemented interfaces for the introspection level DECLARED.
855         */
856        protected static Stream<Field> getFields(Class<?> clazz, IntrospectionLevel introspectionLevel) {
857                if (clazz == null || introspectionLevel == null || introspectionLevel == IntrospectionLevel.NONE || Object.class.equals(clazz)) {
858                        return Stream.empty();
859                }
860                
861                if (introspectionLevel == IntrospectionLevel.ACCESSIBLE) {
862                        return Arrays.stream(clazz.getFields());
863                }
864                
865                Stream<Field> declared = Arrays.stream(clazz.getDeclaredFields());
866                
867                Class<?> superClass = clazz.getSuperclass();
868                declared = Stream.concat(declared, getFields(superClass, introspectionLevel));
869                
870                for (Class<?> implementedInterface: clazz.getInterfaces()) {
871                        declared = Stream.concat(declared, getFields(implementedInterface, introspectionLevel));
872                        
873                }
874                return declared;
875        }
876        
877        protected static Stream<AccessibleObject> getFieldsAndMethods(Class<?> clazz, IntrospectionLevel introspectionLevel) {
878                return Stream.concat(getMethods(clazz, introspectionLevel), getFields(clazz, introspectionLevel));
879        }

881        /**
882         * Retrieves field value. May need to be overridden if the target class is not exported by its module.
883         * @param target
884         * @param field
885         * @return
886         */
887        protected Object getFieldValue(Object target, Field field) {
888                boolean canAccess = field.canAccess(target);
889                try {
890                        if (!canAccess) {
891                                field.setAccessible(true);
892                        }
893                        return  field.get(target);
894                } catch (IllegalArgumentException | IllegalAccessException e) {
895                        throw new NasdanikaException("Cannot access field " + field + " of " + target + ": " + e, e);
896                } finally {
897                        if (!canAccess) {
898                                field.setAccessible(false);
899                        }
900                }       
901        }

928         * @param target
929         * @param method
930         * @param args
931         * @return
932         */
933        protected Object invokeMethod(Object target, Method method, Object... args) {
934                boolean canAccess = method.canAccess(target);
935                try {
936                        if (!canAccess) {
937                                method.setAccessible(true);
938                        }
939                        return method.invoke(target, args);
940                } catch (IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
941                        throw new NasdanikaException("Error invoking " + method + " of " + target + ": " + e, e);
942                } finally {
943                        if (!canAccess) {
944                                method.setAccessible(false);
945                        }
946                }

949        protected boolean canSet(AnnotatedElement element, Class<?> type) {
950                if (element instanceof Field) {
951                        return type == null || ((Field) element).getType().isAssignableFrom(type);
952                }
953                if (element instanceof Method) {
954                        Method method = (Method) element;
955                        return !Modifier.isAbstract(((Member) method).getModifiers()) && method.getParameterCount() == 1 && (type == null || method.getParameterTypes()[0].isAssignableFrom(type));
956                }
957                return false;
958        }
959        
960        protected boolean mustSet(AnnotatedElement element, Class<?> type, String message) {
961                if (canSet(element, type)) {
962                        return true;
963                }
964                throw new NasdanikaException(message);
965        }

967        protected void set(Object target, AnnotatedElement element, Object value) {
968                if (element instanceof Field) {
969                        setFieldValue(target, (Field) element, value);
970                } else if (element instanceof Method) {
971                        invokeMethod(target, (Method) element, value);
972                } else {
973                        throw new IllegalArgumentException("Cannot set value into " + element);
974                }
975        }

977        protected Object get(Object target, AnnotatedElement element) {
978                if (element instanceof Field) {
979                        return getFieldValue(target, (Field) element);
980                } 
981                if (element instanceof Method) {
982                        return invokeMethod(target, (Method) element);
983                }
984                
985                throw new IllegalArgumentException("Cannot get value from " + element);
986        }

988        protected boolean canGet(AnnotatedElement element, Class<?> type) {
989                if (element instanceof Field) {
990                        return type == null || type.isAssignableFrom(((Field) element).getType());
991                }
992                if (element instanceof Method) {
993                        Method method = (Method) element;                                       
994                        return !Modifier.isAbstract(((Member) method).getModifiers()) && method.getParameterCount() == 0 && (type == null || type.isAssignableFrom(method.getReturnType()));
995                }
996                return false;
997        }
998        
999        protected boolean mustGet(AnnotatedElement element, Class<?> type, String message) {
1000                if (canGet(element, type)) {
1001                        return true;
1002                }
1003                throw new NasdanikaException(message);
1004        }
