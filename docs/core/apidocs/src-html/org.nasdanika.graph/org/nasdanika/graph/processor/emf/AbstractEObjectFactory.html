025/**
026 * Creates {@link EObject}s from graph elements and wires them together using diagram element properties.
027 * Uses {@link ProcessorInfo} as handler and enpoint type.
028 * @author Pavel
029 *
030 */
031public abstract class AbstractEObjectFactory<T extends EObject, P extends SemanticProcessor<T>> implements NopEndpointProcessorFactory<P, ProcessorInfo<P>> {
032        
033        @SuppressWarnings("unchecked")
034        @Override
035        public ProcessorInfo<P> createProcessor(ProcessorConfig<P> config, ProgressMonitor progressMonitor) {
036                Collection<T> semanticElements = createSemanticElements(config, progressMonitor);
037                P processor = createProcessor(config, semanticElements, progressMonitor);
038                Collection<Throwable> failures = new ArrayList<>();
039                BiFunction<Void, Throwable, ProcessorInfo<P>> failureHandler  = (result, failure) -> {
040                        if (failure != null) {
041                                failures.add(failure);
042                        }
043                        return null;
044                };
045                ProcessorInfo<P> processorInfo = ProcessorInfo.of(config, processor, () -> {
046                        return failures;
047                });
048                if (processor == null) {
049                        if (config instanceof ConnectionProcessorConfig) {
050                                // Pass-through wiring source endpoint to target handler and target endpoint to source handler
051                                ConnectionProcessorConfig<P, ProcessorInfo<P>, ProcessorInfo<P>> connectionConfig = (ConnectionProcessorConfig<P, ProcessorInfo<P>, ProcessorInfo<P>>) config;
052                                CompletionStage<ProcessorInfo<P>> sourceEndpoint = connectionConfig.getSourceEndpoint();
053                                if (sourceEndpoint != null) {
054                                        sourceEndpoint.thenAccept(connectionConfig::setTargetHandler).handle(failureHandler);
055                                }
056                                
057                                CompletionStage<ProcessorInfo<P>> targetEndpoint = connectionConfig.getTargetEndpoint();
058                                if (targetEndpoint != null) {
059                                        targetEndpoint.thenAccept(connectionConfig::setSourceHandler).handle(failureHandler);
060                                }
061                        }
062                } else {
063                        // Wiring                       
064
065                        // Parent
066                        config.getParentProcessorInfo().thenAccept(parentInfo -> setParent(config, processor, parentInfo, progressMonitor, failures::add)).handle(failureHandler);
067                        
068                        // Children
069                        setChildren(config, processor, config.getChildProcessorsInfo(), progressMonitor);
070                        
071                        // Registry
072                        config.getRegistry().thenAccept(registry -> setRegistry(config, processor, registry, progressMonitor)).handle(failureHandler);
073                        
074                        if (config instanceof NodeProcessorConfig) {
075                                NodeProcessorConfig<P, ProcessorInfo<P>, ProcessorInfo<P>> nodeConfig = (NodeProcessorConfig<P, ProcessorInfo<P>, ProcessorInfo<P>>) config;
076                                for (T semanticElement: processor.getSemanticElements()) {
077                                        semanticElement.eAdapters().add(new NodeProcessorConfigAdapter<>(nodeConfig));
078                                }
079                                
080                                // Incoming 
081                                for (Entry<Connection, CompletionStage<ProcessorInfo<P>>> ie: nodeConfig.getIncomingEndpoints().entrySet()) {
082                                        ie.getValue().thenAccept(incomingInfo -> setIncoming(nodeConfig, processor, ie.getKey(), incomingInfo, progressMonitor)).handle(failureHandler);
083                                }
084                                
085                                // Outgoing
086                                for (Entry<Connection, CompletionStage<ProcessorInfo<P>>> oe: nodeConfig.getOutgoingEndpoints().entrySet()) {
087                                        oe.getValue().thenAccept(outgoingInfo -> setOutgoing(nodeConfig, processor, oe.getKey(), outgoingInfo, progressMonitor)).handle(failureHandler);
088                                }
089                                
090                                nodeConfig.getIncomingHandlerConsumers().forEach((k,v) -> v.accept(processorInfo));
091                                nodeConfig.getOutgoingHandlerConsumers().forEach((k,v) -> v.accept(processorInfo));                             
092                        } else if (config instanceof ConnectionProcessorConfig) {
093                                ConnectionProcessorConfig<P, ProcessorInfo<P>, ProcessorInfo<P>> connectionConfig = (ConnectionProcessorConfig<P, ProcessorInfo<P>, ProcessorInfo<P>>) config;
094                                for (T semanticElement: processor.getSemanticElements()) {
095                                        semanticElement.eAdapters().add(new ConnectionProcessorConfigAdapter<>(connectionConfig));
096                                }
097                                
098                                // Source
099                                connectionConfig.getSourceEndpoint().thenAccept(sourceInfo -> setSource(connectionConfig, processor, sourceInfo, progressMonitor)).handle(failureHandler);
100                                
101                                // Target
102                                connectionConfig.getTargetEndpoint().thenAccept(targetInfo -> setTarget(connectionConfig, processor, targetInfo, progressMonitor)).handle(failureHandler);
103
104                                connectionConfig.setSourceHandler(processorInfo);
105                                connectionConfig.setTargetHandler(processorInfo);                               
106                        } else {
107                                for (T semanticElement: processor.getSemanticElements()) {
108                                        semanticElement.eAdapters().add(new ProcessorConfigAdapter<P, ProcessorConfig<P>>(config));
109                                }
110                        }
111                        
112                        for (T semanticElement: processor.getSemanticElements()) {
113                                LinkedResourcesAdapter linkedResourcesAdapter = (LinkedResourcesAdapter) EcoreUtil.getRegisteredAdapter(semanticElement, LinkedResourcesAdapter.class);
114                                if (linkedResourcesAdapter != null) {
115                                        for (Resource linkedResource: linkedResourcesAdapter.getLinkedResources()) {
116                                                linkResource(config, processor, linkedResource, progressMonitor);
117                                        }
118                                }
119                        }
120                }               
121                
122                return processorInfo;
123        }

125        /**
126         * Creates a semantic element for a diagram element. May return null if there is no semantic mapping for a given diagram element.
127         * @param config
128         * @return
129         */     
130        protected abstract Collection<T> createSemanticElements(ProcessorConfig<P> config, ProgressMonitor progressMonitor);    
131
132        /**
133         * Creates a processor from config and a collection of semantic elements. May return null.
134         * @param config
135         * @param semanticElements
136         * @param progressMonitor
137         * @return
138         */
139        protected abstract P createProcessor(ProcessorConfig<P> config, Collection<T> semanticElements, ProgressMonitor progressMonitor);

141        /**
142         * Links resource. This implementation calls setChildren() for resource roots.
143         * @param config
144         * @param semanticElement
145         * @param linkedResource
146         * @param progressMonitor
147         */
148        @SuppressWarnings("unchecked")
149        protected void linkResource(ProcessorConfig<P> config, P processor, Resource linkedResource, ProgressMonitor progressMonitor) {
150                for (EObject linkedRoot: new ArrayList<>(linkedResource.getContents())) {
151                        ProcessorConfig<P> linkedRootConfig = (ProcessorConfig<P>) EcoreUtil.getRegisteredAdapter(linkedRoot, ProcessorConfig.class);
152                        P linkedRootProcessor = createProcessor(config, Collections.singleton((T) linkedRoot), progressMonitor);
153                        setChildren(config, processor, Collections.singletonMap(linkedRootConfig == null ? null : linkedRootConfig.getElement(), ProcessorInfo.of(linkedRootConfig, linkedRootProcessor, null)), progressMonitor);
154                }
155        }

166        /**
167         * Sets element children.
168         * @param config
169         * @param semanticElement
170         * @param children
171         */
172        @SuppressWarnings("unchecked")
173        protected void setChildren(ProcessorConfig<P> config, P processor, Map<Element, ProcessorInfo<P>> children, ProgressMonitor progressMonitor) {
174                for (T semanticElement: processor.getSemanticElements()) {
175                        for (Entry<Element, ProcessorInfo<P>> ce: children.entrySet()) {
176                                P childProcessor = ce.getValue().getProcessor();
177                                if (childProcessor != null) {
178                                        for (T semanticChild: childProcessor.getSemanticElements()) {
179                                                EReference childReference = getChildReference(config, semanticElement, ce.getKey(), ce.getValue(), semanticChild);
180                                                if (childReference != null) {
181                                                        if (childReference.isMany()) {
182                                                                ((Collection<EObject>) semanticElement.eGet(childReference)).add(semanticChild);
183                                                        } else {
184                                                                semanticElement.eSet(childReference, semanticChild);
185                                                        }
186                                                }
187                                        }
188                                }                                                                                                                                       
189                        }
190                }
191        }

202        /**
203         * Sets registry. This method does nothing.
204         * @param config
205         * @param semanticElement
206         * @param registry
207         */
208        @SuppressWarnings("unchecked")
209        protected void setRegistry(ProcessorConfig<P> config, P processor, Map<Element, ProcessorInfo<P>> registry, ProgressMonitor progressMonitor) {
210                if (processor != null) {
211                        for (T semanticElement: processor.getSemanticElements()) {
212                                for (Entry<Element, ProcessorInfo<P>> re: registry.entrySet()) {
213                                        P registryEntryProcessor = re.getValue().getProcessor();
214                                        if (registryEntryProcessor != null) {
215                                                for (T registrySemanticElement: registryEntryProcessor.getSemanticElements()) {
216                                                        if (registrySemanticElement != null) {
217                                                                EReference registryReference = getRegistryReference(config, processor, semanticElement, re.getKey(), re.getValue(), registrySemanticElement);
218                                                                if (registryReference != null) {
219                                                                        if (registryReference.isMany()) {
220                                                                                ((Collection<EObject>) semanticElement.eGet(registryReference)).add(registrySemanticElement);
221                                                                        } else {
222                                                                                semanticElement.eSet(registryReference, registrySemanticElement);
223                                                                        }
224                                                                }
225                                                        }
226                                                }
227                                        }
228                                }
229                        }
230                }

257        /**
258         * Sets parent. If parent is null this method sets itself to be called for the grand parent etc. 
259         * If child reference property is not null then the semantic element is injected into the parent's references specified in 
260         * the child reference property. Otherwise, if parent reference property is not null then the parent is injected into the semantic element's 
261         * reference specified in the parent reference property.
262         * @param semanticElement
263         * @param parentProcessorInfo
264         */
265        @SuppressWarnings("unchecked")
266        protected void setParent(ProcessorConfig<P> config, P processor, ProcessorInfo<P> parentProcessorInfo, ProgressMonitor progressMonitor, Consumer<Throwable> failureConsumer) {
267                P parentProcessor = parentProcessorInfo.getProcessor();
268                if (parentProcessor == null) {
269                        // No processor, need to go up to the parent.
270                        ProcessorConfig<P> parentConfig = parentProcessorInfo.getConfig();
271                        if (parentConfig != null) {
272                                parentConfig.getParentProcessorInfo().thenAccept(grandParentProcessorInfo -> setParent(config, processor, grandParentProcessorInfo, progressMonitor, failureConsumer)).handle((result, failure) -> {
273                                        if (failure != null) {
274                                                failureConsumer.accept(failure);
275                                        }
276                                        return null; 
277                                } );
278                        }
279                } else {
280                        for (T semanticElement: processor.getSemanticElements()) {
281                                for (T parentSemanticElement: parentProcessor.getSemanticElements()) {
282                                        EReference childReference = getChildReference(config, processor, semanticElement, parentProcessorInfo, parentSemanticElement);
283                                        if (childReference == null) {
284                                                EReference parentReference = getParentReference(config, processor, semanticElement, parentProcessorInfo, parentSemanticElement);
285                                                if (parentReference != null) {
286                                                        if (parentReference.isMany()) {
287                                                                ((Collection<EObject>) semanticElement.eGet(parentReference)).add(parentSemanticElement);
288                                                        } else {
289                                                                semanticElement.eSet(parentReference, parentSemanticElement);
290                                                        }                                       
291                                                }
292                                        } else {
293                                                if (childReference.isMany()) {
294                                                        ((Collection<EObject>) parentSemanticElement.eGet(childReference)).add(semanticElement);
295                                                } else {
296                                                        parentSemanticElement.eSet(childReference, semanticElement);
297                                                }
298                                        }
299                                        
300                                }
301                        }
302                }                                                                                               
303        }

305        /**
306         * This method calls protected setParent() method to establish parent/child relationships between semantic elements
307         * using data from the graph elements from which these semantic elements were created.
308         * This method uses {@link ProcessorConfigAdapter}s to retrieve graph configs to pass to setParent()  
309         * @param child
310         * @param parent
311         */
312        @SuppressWarnings("unchecked")
313        public void setParent(T child, T parent, ProgressMonitor progressMonitor) {
314                if (child != null && parent != null) {
315                        ProcessorConfig<P> childConfig = (ProcessorConfig<P>) EcoreUtil.getRegisteredAdapter(child, ProcessorConfig.class);
316                        ProcessorConfig<P> parentConfig = (ProcessorConfig<P>) EcoreUtil.getRegisteredAdapter(parent, ProcessorConfig.class);
317                        P childProcessor = createProcessor(childConfig, Collections.singleton(child), progressMonitor);
318                        P parentProcessor = createProcessor(parentConfig, Collections.singleton(parent), progressMonitor);
319                        setParent(childConfig, childProcessor, ProcessorInfo.of(parentConfig, parentProcessor, null), progressMonitor, f -> {
320                                f.printStackTrace();
321                        }); // Should be no failures here
322                }
323        }

331        /**
332         * Sets connection semantic element source if source reference property is not null.  
333         */
334        @SuppressWarnings("unchecked")
335        protected void setSource(ConnectionProcessorConfig<P, ProcessorInfo<P>, ProcessorInfo<P>> config, P processor, ProcessorInfo<P> sourceProcessorInfo, ProgressMonitor progressMonitor) {
336                P sourceProcessor = sourceProcessorInfo.getProcessor();         
337                if (sourceProcessor != null && processor != null) {
338                        for (T semanticElement: processor.getSemanticElements()) {
339                                for (T sourceSemanticElement: sourceProcessor.getSemanticElements()) {
340                                        EReference sourceReference = getSourceReference(config, processor, semanticElement, sourceProcessorInfo, sourceSemanticElement);
341                                        if (sourceReference != null) {
342                                                if (sourceReference.isMany()) {
343                                                        ((Collection<EObject>) semanticElement.eGet(sourceReference)).add(sourceSemanticElement);
344                                                } else {
345                                                        semanticElement.eSet(sourceReference, sourceSemanticElement);
346                                                }
347                                        }
348                                }
349                        }
350                }                                                                                               
351        }
