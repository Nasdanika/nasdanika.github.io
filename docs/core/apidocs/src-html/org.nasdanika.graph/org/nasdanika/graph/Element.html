001package org.nasdanika.graph;
002
003import java.lang.reflect.InvocationTargetException;
004import java.lang.reflect.Method;
005import java.lang.reflect.Modifier;
006import java.util.Arrays;
007import java.util.Collection;
008import java.util.Map;
009import java.util.Optional;
010import java.util.function.BiFunction;
011import java.util.function.Consumer;
012import java.util.stream.Stream;
013
014import org.nasdanika.common.NasdanikaException;
015import org.nasdanika.common.Util;
016import org.springframework.expression.EvaluationException;
017import org.springframework.expression.Expression;
018import org.springframework.expression.ExpressionParser;
019import org.springframework.expression.spel.standard.SpelExpressionParser;
020
021/**
022 * Graph element accepts visitors and supports streaming. 
023 * @author Pavel
024 *
025 */
026public interface Element {
027        
028        /**
029         * Accepts the visitor in children first way.
030         * @param visitor
031         */
032        default void accept(Consumer<? super Element> visitor) {
033                accept((e, cr) -> { visitor.accept(e); return null; });
034        }
035        
036        /**
037         * Accepts a result producing visitor {@link BiFunction}. Passes to it a map of results collected from children, nulls are not included.   
038         * @param <T>
039         * @param visitor
040         * @return result returned by the visitor.
041         */
042        <T> T accept(BiFunction<? super Element, Map<? extends Element, T>, T> visitor);
043        
044        /**
045         * Creates a visitor dispatching elements to methods annotated with {@link Handler}.
046         * Methods may have one or two parameters taking Element and a map of children.
047         */
048        default Object dispatch(Object... targets) {
049                return dispatch(Arrays.asList(targets));
050        }
051        
052        default Object dispatch(Collection<Object> targets) {
053                return accept((element, childResults) -> {
054                        for (Object target: targets) {
055                                Optional<Method> handlerOptional = Arrays.stream(target.getClass().getMethods())
056                                        .filter(m -> !Modifier.isAbstract(m.getModifiers()))    
057                                        .filter(m -> m.getAnnotation(Handler.class) != null)
058                                        .filter(m -> m.getParameterCount() == 1 || m.getParameterCount() == 2)
059                                        .filter(m -> m.getParameters()[0].getType().isInstance(element))
060                                        .filter(m -> m.getParameterCount() == 1 ? true : m.getParameters()[1].getType().isInstance(childResults))
061                                        .filter(m -> matchPredicate(element, m.getAnnotation(Handler.class).value()))
062                                        .sorted((a, b) -> b.getAnnotation(Handler.class).priority() - a.getAnnotation(Handler.class).priority())
063                                        .findFirst();
064                                
065                                if (!handlerOptional.isEmpty()) {
066                                        Method handler = handlerOptional.get();
067                                        try {
068                                                if (handler.getParameterCount() == 1) {
069                                                        return handler.invoke(target, element);
070                                                }
071                                                return handler.invoke(target, element, childResults);                                   
072                                        } catch (IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
073                                                throw new NasdanikaException("Error invoking handler: " + e, e);
074                                        }
075                                }
076                        }
077                        return null;
078                });
079        }
080        
081        /**
082         * @return Stream containing this element and its children.
083         */
084        default Stream<Element> stream() {
085                BiFunction<Element, Map<? extends Element, Stream<Element>>, Stream<Element>> visitor = (element, childMappings) -> {
086                        return childMappings.values().stream().reduce(Stream.of(element), (a,b) -> Stream.concat(a, b));
087                };
088                return accept(visitor);
089        }; 
090        
091        /**
092         * Parses and evaluates expression using <a href="https://docs.spring.io/spring-framework/docs/current/reference/html/core.html#expressions">Spring Expression Language</a> 
093         * @param obj
094         * @param expr 
095         * @return true if expression is blank or evaluates to true, false if the expression evaluates to false or throws EvaluationException.
096         */
097        private boolean matchPredicate(Object obj, String expr) {
098                if (Util.isBlank(expr)) {
099                        return true;
100                }
101                
102                ExpressionParser parser = new SpelExpressionParser();
103                Expression exp = parser.parseExpression(expr);                  
104                try {
105                        return exp.getValue(obj, Boolean.class);
106                } catch (EvaluationException e) {
107                        return false;
108                }
109        }
110        
111
112}
