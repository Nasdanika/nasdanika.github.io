001package org.nasdanika.resources;
002
003import java.io.FilterInputStream;
004import java.io.IOException;
005import java.io.InputStream;
006import java.util.Collections;
007import java.util.HashMap;
008import java.util.Map;
009import java.util.function.BiFunction;
010import java.util.function.Predicate;
011import java.util.zip.ZipEntry;
012import java.util.zip.ZipInputStream;
013import java.util.zip.ZipOutputStream;
014
015import org.nasdanika.common.ProgressMonitor;
016
017/**
018 * Container of {@link Entity}ies.
019 * @author Pavel
020 *
021 */
022public interface EntityContainer<C,E extends Entity<C,E>> extends Container<E> {
023        
024        
025        /**
026         * Loads entities from {@link ZipInputStream}
027         * @param zipInputStream
028         * @param filter
029         * @param decoder Decodes input stream to entity state.
030         * @param progressMonitor
031         * @throws IOException
032         */
033        default void load(
034                        ZipInputStream zipInputStream, 
035                        Predicate<String> filter,
036                        BiFunction<String,InputStream,C> decoder,
037                        ProgressMonitor progressMonitor) throws IOException {
038
039                try {
040                        ZipEntry zipEntry;
041                        while ((zipEntry = zipInputStream.getNextEntry()) != null) {
042                                try (ProgressMonitor entryMonitor = progressMonitor.split("Entry "+zipEntry.getName(), 1, zipEntry)) {
043                                        String entryName = zipEntry.getName();
044                                        if (filter == null || filter.test(entryName)) {
045                                                if (entryName.endsWith("/")) {
046                                                        if (getContainer(entryName.substring(0, entryName.length() - 1), entryMonitor.split("Getting container "+entryName, 1)) == null) {
047                                                                throw new IOException("Container with path "+entryName+" cannot be created");
048                                                        }
049                                                } else {
050                                                        E entity = get(entryName, entryMonitor.split("Getting entity "+entryName, 1, this));
051                                                        if (entity == null) {
052                                                                throw new IllegalArgumentException("Cannot get entity "+entryName+" ("+this+")");
053                                                        }
054                                                        // Overriding close because set state closes the source stream when it is drained.
055                                                        InputStream entryStream = new FilterInputStream(zipInputStream) { 
056
057                                                                public void close() {
058                                                                        
059                                                                }
060                                                                
061                                                        };
062                                                        
063                                                        C state = decoder.apply(entryName, entryStream);
064                                                        entity.setState(state, entryMonitor.split("Setting state for "+entryName, 1, entity, state));
065                                                }
066                                        }
067                                        zipInputStream.closeEntry();
068                                }
069                        }
070                } finally {
071                        progressMonitor.close();
072                        zipInputStream.close();
073                }
074                
075        }
076        
077        /**
078         * Stores entities to {@link ZipOutputStream}
079         * @param zipOutputStream
080         * @param prefix
081         * @param encoder Encodes entity state to {@link InputStream}
082         * @param progressMonitor
083         * @throws IOException
084         */
085        default void store(
086                        ZipOutputStream zipOutputStream, 
087                        String prefix, 
088                        BiFunction<String,C,InputStream> encoder,
089                        ProgressMonitor progressMonitor) throws IOException {
090                
091                Copier<E, InputStream> contentSerializer = (path, source, existingElement, monitor) -> encoder.apply(path, source.getState(monitor));
092                store(zipOutputStream, prefix, contentSerializer, progressMonitor);
093        }
094        
095        /**
096         * Adapts to container of entity state.
097         * @return
098         */
099        default Container<C> stateAdapter() {
100                return new Container<C>() {
101
102                        @SuppressWarnings("unchecked")
103                        @Override
104                        public Object find(String path, ProgressMonitor monitor) {
105                                Object ret = EntityContainer.this.find(path, monitor.split("Finiding", 1, this));
106                                if (ret instanceof EntityContainer) {
107                                        return ((EntityContainer<C,E>) ret).stateAdapter();
108                                }
109                                return ret == null ? null : ((E) ret).getState(monitor.split("Getting state", 1, this));
110                        }
111
112                        @Override
113                        public C get(String path, ProgressMonitor monitor) {
114                                E entity = EntityContainer.this.get(path, monitor.split("Getting enitity "+path, 1, this));
115                                return entity == null ? null : entity.getState(monitor.split("Getting state of "+path, 1, this));
116                        }
117
118                        @Override
119                        public void put(String path, C element, ProgressMonitor monitor) throws IllegalArgumentException {
120                                E entity = EntityContainer.this.get(path, monitor.split("Getting enitity "+path, 1, this));
121                                entity.setState(element, monitor.split("Setting state of "+path, 1, this));
122                        }
123
124                        @SuppressWarnings("unchecked")
125                        @Override
126                        public Object delete(String path, ProgressMonitor monitor) {
127                                Object deleted = EntityContainer.this.delete(path, monitor.split("Deleting "+path, 1, this));
128                                if (deleted instanceof Container) {
129                                        return ((EntityContainer<C,E>) deleted).stateAdapter();
130                                }
131                                return deleted == null ? null : ((E) deleted).getState(monitor.split("Getting state of "+path, 1, this));
132                        }
133
134                        @SuppressWarnings("unchecked")
135                        @Override
136                        public Container<C> getContainer(String path, ProgressMonitor monitor) {
137                                Container<E> ret = EntityContainer.this.getContainer(path, monitor);
138                                return ret instanceof EntityContainer ? ((EntityContainer<C,E>) ret).stateAdapter() : null;
139                        }
140
141                        @SuppressWarnings("unchecked")
142                        @Override
143                        public Map<String, Object> getChildren(ProgressMonitor monitor) {
144                                Map<String, Object> ret = new HashMap<>();
145                                EntityContainer.this.getChildren(monitor.split("Getting children", 1, this)).forEach((path, child) -> {
146                                        if (child instanceof EntityContainer) {
147                                                ret.put(path, ((EntityContainer<C,E>) ret).stateAdapter());
148                                        } else {
149                                                ret.put(path, ((E) child).getState(monitor.split("Getting state", 1, child)));
150                                        }                                       
151                                }); 
152                                return Collections.unmodifiableMap(ret);
153                        }
154
155                        @SuppressWarnings("unchecked")
156                        @Override
157                        public Container<C> getParent() {
158                                Container<E> parent = EntityContainer.this.getParent();
159                                return parent instanceof EntityContainer ? ((EntityContainer<C,E>) parent).stateAdapter() : null;
160                        }
161
162                        @Override
163                        public String getName() {
164                                return EntityContainer.this.getName();
165                        }
166
167                        @Override
168                        public void copy(Container<? super C> container, String path, ProgressMonitor monitor) {
169                                throw new UnsupportedOperationException("Implement when this one is thrown");
170                        }
171
172                        @Override
173                        public void move(Container<? super C> container, String path, ProgressMonitor monitor) {
174                                throw new UnsupportedOperationException("Implement when this one is thrown");
175                        }
176                        
177                        @Override
178                        public String toString() {
179                                return getClass().getName()+"("+getPath()+")";
180                        }
181                        
182                };
183        }
