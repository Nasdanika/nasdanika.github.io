003/**

006 * @author Pavel
007 *
008 * @param <E>
009 */
010public interface TypedEntity<T> extends Entity<T,TypedEntity<T>>, TypedResource<T> {
011        
012        /**
013         * Narrowing the return type.
014         */
015        @Override
016        TypedEntityContainer<T> getParent();
017
018}
