038/**
039 * This factory uses {@link DrawioResource} with {@link DrawioEObjectFactory}. 
040 * It injects name, description, and markers in configureSemanticElement(). It 
041 * @author Pavel
042 *
043 * @param <T>
044 */
045public abstract class NcoreDrawioResourceFactory<T extends EObject> extends ResourceFactoryImpl {
046        
047        private class NcoreDrawioResource extends DrawioResource<SemanticProcessor<T>, T> implements AbstractEObjectFactoryProcessorResource<T> {
048
049                private DrawioEObjectFactory<T, SemanticProcessor<T>> processorFactory;
050                private ProgressMonitor progressMonitor;
051
052                protected NcoreDrawioResource(URI uri, DrawioEObjectFactory<T, SemanticProcessor<T>> processorFactory, ProgressMonitor progressMonitor) {
053                        super(uri);
054                        this.processorFactory = processorFactory;
055                        this.progressMonitor = progressMonitor;
056                }
057
058                @SuppressWarnings("unchecked")
059                @Override
060                public void setParent(T parent) {
061                        for (EObject root: new ArrayList<>(getContents())) {
062                                processorFactory.setParent((T) root, parent, getProgressMonitor());
063                        }
064                }
065
066                @Override
067                protected DrawioEObjectFactory<T, SemanticProcessor<T>> getProcessorFactory() {
068                        return processorFactory;
069                }
070                
071                @Override
072                protected ProgressMonitor getProgressMonitor() {
073                        return progressMonitor;
074                };
075                
076        }
077        
078        @Override
079        public Resource createResource(URI uri) {
080                return new NcoreDrawioResource(uri, createProcessorFactory(uri), getProgressMonitor(uri));
081        }
082                
083        protected DrawioEObjectFactory<T, SemanticProcessor<T>> createProcessorFactory(URI uri) {
084                return new DrawioEObjectFactory<T, SemanticProcessor<T>>() {
085
086                        @Override
087                        protected ResourceSet getResourceSet() {
088                                return NcoreDrawioResourceFactory.this.getResourceSet();
089                        }
090
091                        @Override
092                        protected SemanticProcessor<T> createProcessor(ProcessorConfig<SemanticProcessor<T>> config, Collection<T> semanticElements, ProgressMonitor progressMonitor) {
093                                return NcoreDrawioResourceFactory.this.createProcessor(uri, config, semanticElements, progressMonitor);
094                        }
095
096                        @Override
097                        protected URI getBaseURI() {
098                                return uri;
099                        }
100                        
101                        @Override
102                        protected Collection<T> createSemanticElements(ProcessorConfig<SemanticProcessor<T>> config, ProgressMonitor progressMonitor) {
103                                Collection<T> semanticElements = super.createSemanticElements(config, progressMonitor);
104                                Collection<T> configuredSemanticElements = NcoreDrawioResourceFactory.this.configureSemanticElements(uri, config, semanticElements, progressMonitor);
105                                return configuredSemanticElements;
106                        }
107                        
108                        /**
109                         * Checking for failures, injecting representations after all processors have been created. 
110                         */
111                        @Override
112                        public Map<Element, ProcessorInfo<SemanticProcessor<T>>> createProcessors(Stream<? extends Element> elements, ProgressMonitor progressMonitor) {
113                                Map<Element, ProcessorInfo<SemanticProcessor<T>>> registry = super.createProcessors(elements, progressMonitor);
114                                
115                                List<Throwable> failures = registry.values().stream().flatMap(pi -> pi.getFailures().stream()).collect(Collectors.toList());
116                                if (!failures.isEmpty()) {
117                                        NasdanikaException ne = new NasdanikaException("Theres's been " + failures.size() +  " failures during processor creation: " + failures);
118                                        for (Throwable failure: failures) {
119                                                ne.addSuppressed(failure);
120                                        }
121                                        throw ne;
122                                }                               
123                                
124                                String representationPropertyName = getRepresentationPropertyName(uri);
125                                                                
126                                registry.entrySet().forEach(re -> configureRegistryEntry(re.getKey(), re.getValue(), getPropertyValue(re.getKey(), representationPropertyName)));
127                                return registry;
128                        }
129                        
130                };
131        }
132
133        /**
134         * Injects encoded document into semantic elements of model elements with representation property.
135         * Sorts references.
136         * @param element
137         * @param info
138         */
139        protected void configureRegistryEntry(Element element, ProcessorInfo<SemanticProcessor<T>> info, String representationKey) {
140                if (info != null && !Util.isBlank(representationKey)) {
141                        SemanticProcessor<T> processor = info.getProcessor();
142                        if (processor != null && element instanceof ModelElement) {
143                                Collection<T> semanticElements = processor.getSemanticElements();
144                                if (semanticElements != null && !semanticElements.isEmpty()) {
145                                        for (T semanticElement: semanticElements) {
146                                                if (semanticElement instanceof org.nasdanika.ncore.ModelElement) {
147                                                        Document document = ((ModelElement) element).getModel().getPage().getDocument();
148                                                        try {
149                                                                ((org.nasdanika.ncore.ModelElement) semanticElement).getRepresentations().put(representationKey, document.toDataURI(true).toString());
150                                                        } catch (TransformerException | IOException e) {
151                                                                throw new NasdanikaException("Error encoding document: " + e, e);
152                                                        }
153                                                }
154                                        }
155                                }
156                        }
157                }
158                
159                // TODO - sort - YAML map of references to comparators. Comparator either String - comparator name or a map - name to configuration. 
160                // Use ECollections to sort. Expression (SPEL) comparator.
161
162        }

168        protected SemanticProcessor<T> createProcessor(URI uri, ProcessorConfig<SemanticProcessor<T>> config, Collection<T> semanticElements, ProgressMonitor progressMonitor) {                
169                if (semanticElements == null || semanticElements.isEmpty()) {
170                        return null; // For pass-through to parent.
171                }
172                
173                // To avoid concurrent modification exception for situations when processor is backed by a resource and semantic elements get reparented and move out of the resource. 
174                Collection<T> semanticElementsSnapshot = Collections.unmodifiableCollection(new ArrayList<>(semanticElements));
175                
176                return new SemanticProcessor<T>() {
177
178                        @Override
179                        public Collection<T> getSemanticElements() {
180                                return semanticElementsSnapshot;
181                        }
182                        
183                };
184        }       
185
186        protected MarkerFactory getMarkerFactory() {
187                return MarkerFactory.INSTANCE;
188        }
189                
190        protected Collection<T> configureSemanticElements(
191                        URI uri,
192                        ProcessorConfig<SemanticProcessor<T>> config, 
193                        Collection<T> semanticElements, 
194                        ProgressMonitor progressMonitor) {
195
196                if (config == null) {
197                        return semanticElements;
198                }
199                Element element = config.getElement();
200                if (semanticElements == null) {
201                        return semanticElements;
202                }
203                for (T semanticElement: semanticElements) {
204                        if (uri != null && semanticElement instanceof Marked && element instanceof Marked) {
205                                for (org.nasdanika.persistence.Marker elementMarker: ((Marked) element).getMarkers()) {
206                                        org.nasdanika.ncore.Marker marker = getMarkerFactory().createMarker(elementMarker.getLocation(), progressMonitor);
207                                        marker.setPosition(elementMarker.getPosition());
208                                        ((Marked) semanticElement).getMarkers().add(marker);
209                                }
210                        }
211                        
212                        if (element instanceof ModelElement) {
213                                ModelElement modelElement = (ModelElement) element;
214                                String tooltip = modelElement.getTooltip();
215                                if (!Util.isBlank(tooltip)) {
216                                        if (semanticElement instanceof org.nasdanika.ncore.ModelElement && !semanticElement.eIsSet(NcorePackage.Literals.MODEL_ELEMENT__DESCRIPTION)) {
217                                                String firstTooltipSentence = org.nasdanika.common.Util.firstPlainTextSentence(tooltip, 50, 250);
218                                                ((org.nasdanika.ncore.ModelElement) semanticElement).setDescription(firstTooltipSentence);                                              
219                                        }
220                                        if (semanticElement instanceof Documented && ((Documented) semanticElement).getDocumentation().isEmpty()) {
221                                                ((Documented) semanticElement).getDocumentation().add(NcoreUtil.wrapString("<pre style='white-space:pre-wrap'>" + System.lineSeparator() + tooltip + System.lineSeparator() + "</pre>"));
222                                        }
223                                }
224                                
225                                String label = modelElement.getLabel();
226                                if (!Util.isBlank(label) && semanticElement instanceof NamedElement && !semanticElement.eIsSet(NcorePackage.Literals.NAMED_ELEMENT__NAME)) {
227                                        ((NamedElement) semanticElement).setName(Jsoup.parse(label).text());                            
228                                }
229                                
230                                String id = modelElement.getId();
231                                if (!Util.isBlank(id) && semanticElement instanceof org.nasdanika.ncore.Composite && !semanticElement.eIsSet(NcorePackage.Literals.COMPOSITE__ID)) {
232                                        ((org.nasdanika.ncore.Composite) semanticElement).setId(id);                            
233                                }
234                                
235                                if (semanticElement instanceof org.nasdanika.ncore.ModelElement) {
236                                        String semanticUuid = ((org.nasdanika.ncore.ModelElement) semanticElement).getUuid();
237                                        if (!Util.isBlank(semanticUuid)) {
238                                                modelElement.setProperty("semantic-uuid", semanticUuid);
239                                        }
240                                }
241                        }
242                        
243                }
244                
245                return semanticElements;                
246        }
247        
248        /**
249         * If this property is set and the semantic element is {@link org.nasdanika.ncore.ModelElement} then {@link Document} is encoded and stored into the representation with the specified name.
250         * @param uri Resource {@link URI}
251         * @return
252         */
253        protected String getRepresentationPropertyName(URI uri) {
254                return "representation";
255        }
