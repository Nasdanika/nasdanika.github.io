001package org.nasdanika.emf;
002
003import java.util.Locale;
004
005import org.eclipse.emf.ecore.EModelElement;
006import org.nasdanika.common.Context;
007
008/**
009 * Resource locator which adds ``_<default locale language>`` suffix and falls back to the chain passed to the constructor. 
010 * @author Pavel
011 *
012 */
013public class LocaleLanguageResourceLocator extends EModelElementAnnotationResourceLocator {
014        
015        public LocaleLanguageResourceLocator(EModelElement modelElement, Locale locale, Context chain) {                
016                super(modelElement, "urn:org.nasdanika", key -> key+"_"+locale.getLanguage(), chain);
017        }
018        
019}
