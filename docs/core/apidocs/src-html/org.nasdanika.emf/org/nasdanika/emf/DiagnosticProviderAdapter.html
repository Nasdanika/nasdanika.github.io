001package org.nasdanika.emf;
002
003import java.util.Collection;
004import java.util.Collections;
005import java.util.function.BiFunction;
006import java.util.function.Function;
007
008import org.eclipse.emf.common.notify.Notifier;
009import org.eclipse.emf.common.notify.impl.AdapterImpl;
010import org.eclipse.emf.common.util.Diagnostic;
011import org.eclipse.emf.ecore.EStructuralFeature;
012
013public class DiagnosticProviderAdapter extends AdapterImpl implements DiagnosticProvider {
014        
015        private Function<Notifier, Collection<Diagnostic>> targetDiagnosticSupplier; 
016        private BiFunction<Notifier, EStructuralFeature, Collection<Diagnostic>> featureDiagnosticSupplier;     
017
018        public DiagnosticProviderAdapter(
019                        Notifier target, 
020                        Function<Notifier, Collection<Diagnostic>> targetDiagnosticSupplier, 
021                        BiFunction<Notifier, EStructuralFeature, Collection<Diagnostic>> featureDiagnosticSupplier) {
022                setTarget(target);
023                this.targetDiagnosticSupplier = targetDiagnosticSupplier;
024                this.featureDiagnosticSupplier = featureDiagnosticSupplier;
025        }
026        
027        @Override
028        public boolean isAdapterForType(Object type) {
029                return type == DiagnosticProvider.class;
030        }
031
032        @Override
033        public Collection<Diagnostic> getDiagnostic() {
034                return targetDiagnosticSupplier == null ? Collections.emptyList() : targetDiagnosticSupplier.apply(getTarget());
035        }
036
037        @Override
038        public Collection<Diagnostic> getFeatureDiagnostic(EStructuralFeature feature) {
039                return featureDiagnosticSupplier == null ? Collections.emptyList() : featureDiagnosticSupplier.apply(getTarget(), feature);
040        }
041
042}
