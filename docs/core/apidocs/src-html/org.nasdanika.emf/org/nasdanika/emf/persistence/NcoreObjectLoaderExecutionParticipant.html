001package org.nasdanika.emf.persistence;
002
003import java.util.ArrayList;
004import java.util.Collection;
005
006import org.eclipse.emf.ecore.EPackage;
007import org.eclipse.emf.ecore.resource.ResourceSet;
008import org.nasdanika.common.Context;
009import org.nasdanika.ncore.NcorePackage;
010
011/**
012 * Registers {@link NcorePackage} with the {@link ResourceSet}.
013 * @author Pavel
014 *
015 */
016public abstract class NcoreObjectLoaderExecutionParticipant extends ObjectLoaderExecutionParticipant {
017
018        public NcoreObjectLoaderExecutionParticipant(Context context) {
019                super(context);
020        }
021
022        @Override
023        protected Collection<EPackage> getEPackages() {
024                Collection<EPackage> ret = new ArrayList<>(); 
025                ret.add(NcorePackage.eINSTANCE);
026                return ret;
027        }
028        
029}
