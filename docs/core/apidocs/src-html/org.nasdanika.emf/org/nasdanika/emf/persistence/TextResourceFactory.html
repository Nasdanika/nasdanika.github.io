001package org.nasdanika.emf.persistence;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.io.OutputStream;
006import java.net.URLDecoder;
007import java.net.URLEncoder;
008import java.nio.charset.StandardCharsets;
009import java.util.Map;
010
011import org.eclipse.emf.common.util.URI;
012import org.eclipse.emf.ecore.EObject;
013import org.eclipse.emf.ecore.resource.Resource;
014import org.eclipse.emf.ecore.resource.impl.ResourceFactoryImpl;
015import org.eclipse.emf.ecore.resource.impl.ResourceImpl;
016import org.nasdanika.common.DefaultConverter;
017import org.nasdanika.common.URIEncodable;
018import org.nasdanika.ncore.NcoreFactory;
019
020/**
021 * Loads text files as resources with a single {@link org.nasdanika.ncore.String} object.
022 * @author Pavel
023 */
024public class TextResourceFactory extends ResourceFactoryImpl {
025        
026        private class TextResource extends ResourceImpl implements URIEncodable {
027                
028                TextResource(URI uri) {
029                        super(uri);
030                }
031                
032                public void load(java.util.Map<?,?> options) throws IOException {
033                        if (isDataURI(getURI())) {
034                                String uriStr = getURI().toString();
035                                uriStr = uriStr.substring("data:text,".length());                               
036                                uriStr = URLDecoder.decode(uriStr, StandardCharsets.UTF_8);
037                                org.nasdanika.ncore.String str = NcoreFactory.eINSTANCE.createString();
038                                str.setValue(uriStr);
039                                getContents().clear();
040                                getContents().add(str);
041                        } else {
042                                super.load(options);
043                        }                       
044                };
045                
046                @Override
047                protected void doLoad(InputStream inputStream, Map<?, ?> options) throws IOException {
048                        String text = DefaultConverter.INSTANCE.toString(inputStream);
049                        org.nasdanika.ncore.String str = NcoreFactory.eINSTANCE.createString();
050                        str.setValue(text);
051                        getContents().clear();
052                        getContents().add(str);
053                }
054
055                @Override
056                public URI encode() {
057                        StringBuilder payloadBuilder = new StringBuilder();
058                        for (EObject eObj: getContents()) {
059                                if (eObj instanceof org.nasdanika.ncore.String) {
060                                        String text = ((org.nasdanika.ncore.String) eObj).getValue();
061                                        if (text != null) {
062                                                payloadBuilder.append(text);
063                                        }
064                                } else {
065                                        throw new UnsupportedOperationException("Unsupported contents: " + eObj);
066                                }
067                        }
068                        String uriStr = "data:text," + URLEncoder.encode(payloadBuilder.toString(), StandardCharsets.UTF_8);
069                        return URI.createURI(uriStr);
070                }
071
072                @Override
073                protected void doSave(OutputStream outputStream, Map<?, ?> options) throws IOException {
074                        try {
075                                for (EObject eObj: getContents()) {
076                                        if (eObj instanceof org.nasdanika.ncore.String) {
077                                                String text = ((org.nasdanika.ncore.String) eObj).getValue();
078                                                if (text != null) {
079                                                        outputStream.write(text.getBytes(StandardCharsets.UTF_8));
080                                                }
081                                        } else {
082                                                throw new UnsupportedOperationException("Unsupported contents: " + eObj);
083                                        }
084                                }
085                        } finally {
086                                outputStream.close();
087                        }
088                }
089                
090        }
091        
092        @Override
093        public Resource createResource(URI uri) {
094                return new TextResource(uri);
095        }
096        
097        /**
098         * @param uri
099         * @return true if the argument URI is drawio data URI, i.e. is starts with data:drawio[;base64],
100         */
101        public static boolean isDataURI(URI uri) {
102                return uri != null && "data".equals(uri.scheme()) && uri.toString().startsWith("data:text,");
103        }       
104
105}
