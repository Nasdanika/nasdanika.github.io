042/**
043 * Loads {@link EObject} {@link EStructuralFeature}s from configuration.
044 * @author Pavel
045 *
046 */
047public class EObjectSupplierFactory extends SupplierFactoryFeatureObject<EObject> {
048        
049        private Map<String, EStructuralFeature> featureMap = new LinkedHashMap<>();
050
051        private EClass eClass;
052        private EObjectLoader loader;
053                
054        private java.util.function.Function<EClass, EObject> constructor = this::instantiate;
055
056        /**
057         * 
058         * @param loader
059         * @param eClass
060         * @param keyProvider
061         * @param constructor Constructs a new EObject instance for EClass. If null, EPackage factory is used to create a new object.
062         */
063        public EObjectSupplierFactory(
064                        EObjectLoader loader, 
065                        EClass eClass, 
066                        BiFunction<EClass,ENamedElement,String> keyProvider,
067                        java.util.function.Function<EClass, EObject> constructor) {
068
069                this.loader = loader;
070                this.eClass = eClass;   // TODO - handling prototype - if there is an annotation - chain - may need to handle @ super?
071                if (keyProvider == null) {
072                        keyProvider = EObjectLoader.LOAD_KEY_PROVIDER; 
073                }               
074                for (EStructuralFeature feature: eClass.getEAllStructuralFeatures()) {
075                        if (!feature.isChangeable() && !"true".equals(NcoreUtil.getNasdanikaAnnotationDetail(feature, EObjectLoader.IS_COMPUTED))) {
076                                continue;
077                        }
078                        String featureKey = keyProvider.apply(eClass, feature);
079                        if (featureKey == null) {
080                                continue;
081                        }
082                        featureMap.put(featureKey, feature);
083                        addFeature(wrapFeature(featureKey, feature, loader, keyProvider));
084                }
085                if (constructor != null) {
086                        this.constructor = constructor;
087                }
088        }
089        
090        /**
091         * Wraps {@link EStructuralFeature} into {@link Feature}
092         * @param featureKey
093         * @param feature
094         * @return
095         */
096        protected Feature<?> wrapFeature(
097                        String featureKey, 
098                        EStructuralFeature feature, 
099                        EObjectLoader loader, 
100                        BiFunction<EClass,ENamedElement,String> keyProvider) {
101                
102                return wrapFeature(featureKey, feature, loader, keyProvider, EObjectLoader.isDefaultFeature(eClass, feature));
103        }
104
105        /**
106         * Wraps {@link EStructuralFeature} into {@link Feature}. 
107         * Static version is needed for {@link EMap} support. 
108         * @param featureKey
109         * @param feature
110         * @return
111         */
112        @SuppressWarnings({ "rawtypes", "unchecked" })
113        public Feature<?> wrapFeature(
114                        String featureKey, 
115                        EStructuralFeature feature, 
116                        EObjectLoader loader, 
117                        BiFunction<EClass,ENamedElement,String> keyProvider,
118                        boolean isDefault) {
119                
120                Object[] exclusiveWith = EObjectLoader.getExclusiveWith(eClass, feature, keyProvider);
121                
122                EModelElementDocumentation documentation = EmfUtil.getDocumentation(feature);
123                if (feature instanceof EAttribute) {
124                        if (feature.isMany()) {
125                                return new ListSupplierFactoryAttribute<>(new org.nasdanika.persistence.ReferenceList<>(featureKey, isDefault, feature.isRequired(), null, documentation == null ? null : documentation.getDocumentation(), exclusiveWith), true);
126                        }
127                        
128                        EClassifier featureType = feature.getEType();
129                        Class<?> featureClass = featureType.getInstanceClass();
130                        boolean interpolate = "true".equals(NcoreUtil.getNasdanikaAnnotationDetail(feature, "interpolate", "true"));
131                        if (String.class == featureClass) {
132                                org.nasdanika.persistence.Reference delegate = new org.nasdanika.persistence.Reference(featureKey, isDefault, feature.isRequired(), null, documentation == null ? null : documentation.getDocumentation(), exclusiveWith) {
133                                        
134                                        @Override
135                                        public Object create(ObjectLoader loader, Object config, URI base, ProgressMonitor progressMonitor,     List<? extends Marker> markers) {
136                                                Object ret = super.create(loader, config, base, progressMonitor, markers);
137                                                if (base != null && !base.isRelative() && base.isHierarchical() && ret instanceof String && "true".equals(NcoreUtil.getNasdanikaAnnotationDetail(feature, EObjectLoader.IS_RESOLVE_URI))) {
138                                                        return URI.createURI((String) ret).resolve(base).toString();
139                                                }
140                                                return ret;
141                                        }
142                                        
143                                };
144                                return new StringSupplierFactoryAttribute(delegate, interpolate);
145                        }
146                        
147                        if (featureClass.isEnum()) {
148                                StringSupplierFactoryAttribute stringAttribute = new StringSupplierFactoryAttribute(new org.nasdanika.persistence.Reference(featureKey, isDefault, feature.isRequired(), null, documentation == null ? null : documentation.getDocumentation(), exclusiveWith), interpolate);
149                                return new EnumSupplierFactoryAttribute(stringAttribute, featureClass, null);
150                        }
151                        
152                        if (EClass.class == featureClass) {
153                                StringSupplierFactoryAttribute stringAttribute = new StringSupplierFactoryAttribute(new org.nasdanika.persistence.Reference(featureKey, isDefault, feature.isRequired(), null, documentation == null ? null : documentation.getDocumentation(), exclusiveWith), interpolate);
154                                FunctionFactory<String,EClass> functionFactory = new FunctionFactory<String, EClass>() {
155
156                                        @Override
157                                        public Function<String, EClass> create(Context ctx) {
158                                                return new Function<String, EClass>() {
159
160                                                        @Override
161                                                        public double size() {
162                                                                return 1;
163                                                        }
164
165                                                        @Override
166                                                        public String name() {
167                                                                return "Resolving type to EClass";
168                                                        }
169
170                                                        @Override
171                                                        public EClass execute(String type, ProgressMonitor progressMonitor) {
172                                                                BiSupplier<EClass, BiFunction<EClass,ENamedElement, String>> result = loader.resolveEClass(type);
173                                                                if (result == null) {
174                                                                        throw new ConfigurationException("Cannot resolve " + type+ " to EClass", EObjectSupplierFactory.this.getMarkers());
175                                                                }
176                                                                return result.getFirst();
177                                                        }
178                                                        
179                                                };
180                                        }
181                                        
182                                };
183                                return new FunctionSupplierFactoryAttribute(stringAttribute, functionFactory);
184                        }
185                        
186                        Function converter = null; // TODO: From annotations for, say, dates - parse pattern - SimpleDateFormat?
187                        return new TypedSupplierFactoryAttribute(featureClass, new org.nasdanika.persistence.Reference(featureKey, isDefault, feature.isRequired(), null, documentation == null ? null : documentation.getDocumentation(), exclusiveWith), interpolate, converter);                     
188                }
189                
190                if (feature instanceof EReference) {
191                        EReference eReference = (EReference) feature;
192                        if (feature.isMany()) {
193                                // EMap
194                                if (feature.getEType().getInstanceClass() == Map.Entry.class) {
195                                        return new MapSupplierFactoryAttribute<>(new ReferenceMap<>(
196                                                        featureKey, 
197                                                        isDefault, 
198                                                        feature.isRequired(), 
199                                                        null, 
200                                                        documentation == null ? null : documentation.getDocumentation(),
201                                                        eClass,
202                                                        eReference,
203                                                        loader,
204                                                        true,
205                                                        keyProvider,
206                                                        exclusiveWith), true);
207                                }
208                                return new ListSupplierFactoryAttribute<>(new ReferenceList<>(
209                                                featureKey, 
210                                                isDefault, 
211                                                feature.isRequired(), 
212                                                null, 
213                                                documentation == null ? null : documentation.getDocumentation(),
214                                                eClass,
215                                                eReference,
216                                                loader,
217                                                true,
218                                                keyProvider,
219                                                exclusiveWith), true);
220                        }
221                        
222                        return new DelegatingSupplierFactoryFeature<>(new Reference<>(
223                                        featureKey, 
224                                        isDefault, 
225                                        feature.isRequired(), 
226                                        null, 
227                                        documentation == null ? null : documentation.getDocumentation(),
228                                        eClass,
229                                        eReference,
230                                        loader,
231                                        true,                                   
232                                        keyProvider,
233                                        exclusiveWith));
234                        
235                }
236                
237                throw new UnsupportedOperationException("Unusupported feature type: " + feature);
238        }

240        @Override
241        protected Function<Map<Object, Object>, EObject> createResultFunction(Context context) {
242                return new Function<Map<Object,Object>, EObject>() {
243                        
244                        @Override
245                        public double size() {
246                                return 1;
247                        }
248                        
249                        @Override
250                        public String name() {
251                                return "Creating " + eClass.getName();
252                        }
253                        
254                        @SuppressWarnings("unchecked")
255                        @Override
256                        public EObject execute(Map<Object, Object> data, ProgressMonitor progressMonitor) {
257                                EObject ret = constructor.apply(eClass);
258                                if (ret == null) {
259                                        ret = instantiate(eClass);
260                                }
261                                List<? extends Marker> markers = getMarkers();
262                                loader.mark(ret, markers, progressMonitor);
263                                Map<EStructuralFeature, Object> loadedFeatures = new HashMap<>();
264                                EStructuralFeature[] loadingFeature = { null };
265                                ret.eAdapters().add(new LoadTrackerAdapter() {
266
267                                        @Override
268                                        public boolean isLoaded(EStructuralFeature feature) {
269                                                return loadedFeatures.containsKey(feature);
270                                        }
271
272                                        @Override
273                                        public boolean isLoading(EStructuralFeature feature) {
274                                                return feature == loadingFeature[0];
275                                        }
276                                        
277                                        @Override
278                                        public Object get(EStructuralFeature feature) {
279                                                return loadedFeatures.get(feature);
280                                        }
281                                        
282                                });
283                                for (Feature<?> feature: features) {
284                                        if (feature.isLoaded()) {
285                                                try {
286                                                        EStructuralFeature structuralFeature = featureMap.get(feature.getKey());
287                                                        loadingFeature[0] = structuralFeature;
288                                                        Object value = feature.get(data);
289                                                        if (structuralFeature.isChangeable()) {
290                                                                // Unwrapping singleton lists
291                                                                if (structuralFeature instanceof EReference) {
292                                                                        if (structuralFeature.isMany()) {
293                                                                                if (value instanceof List) {
294                                                                                        List<Object> theValue = new ArrayList<>();
295                                                                                        for (Object element: (List<?>) value) {
296                                                                                                if (element instanceof List) {
297                                                                                                        theValue.addAll((List<Object>) element);
298                                                                                                } else {
299                                                                                                        theValue.add(element);
300                                                                                                }                                                                                               
301                                                                                        }
302                                                                                        setFeature(ret, structuralFeature, theValue);
303                                                                                } else if (value instanceof Map) {
304                                                                                        EClass refType = ((EReference) structuralFeature).getEReferenceType();
305                                                                                        boolean isManyValue = false;
306                                                                                        Class<?> refTypeInstanceClass = refType.getInstanceClass();
307                                                                                        if (refTypeInstanceClass != null && Map.Entry.class.isAssignableFrom(refTypeInstanceClass)) {
308                                                                                                EStructuralFeature valueFeature = refType.getEStructuralFeature("value");
309                                                                                                isManyValue = valueFeature != null && valueFeature.isMany();
310                                                                                        }
311                                                                                        Map<Object,Object> theValue = new LinkedHashMap<>();
312                                                                                        for (Entry<Object, Object> entry: ((Map<Object,Object>) value).entrySet()) {
313                                                                                                Object entryValue = entry.getValue();
314                                                                                                if (entryValue instanceof List) {
315                                                                                                        List<?> entryValueList = (List<?>) entryValue;
316                                                                                                        if (isManyValue) {
317                                                                                                                List<Object> theEntryValueList = new ArrayList<>();
318                                                                                                                for (Object entryValueElement: (List<?>) entryValueList) {
319                                                                                                                        if (entryValueElement instanceof List) {
320                                                                                                                                theEntryValueList.addAll((List<Object>) entryValueElement);
321                                                                                                                        } else {
322                                                                                                                                theEntryValueList.add(entryValueElement);
323                                                                                                                        }                                                                                               
324                                                                                                                }       
325                                                                                                                theValue.put(entry.getKey(), theEntryValueList);
326                                                                                                        } else {
327                                                                                                                if (entryValueList.size() == 1) {
328                                                                                                                        theValue.put(entry.getKey(), entryValueList.get(0));                                                                                                            
329                                                                                                                } else if (!entryValueList.isEmpty()) {
330                                                                                                                        throw new ConfigurationException("Map entry value list size is more than one: "+ entryValue, markers);
331                                                                                                                }
332                                                                                                        }
333                                                                                                } else {
334                                                                                                        theValue.put(entry.getKey(), entryValue);
335                                                                                                }                                                                                               
336                                                                                        }
337                                                                                        setFeature(ret, structuralFeature, theValue);                                                                                   
338                                                                                } else {
339                                                                                        setFeature(ret, structuralFeature, value);                                                                                      
340                                                                                }
341                                                                        } else {
342                                                                                if (value instanceof List && ((List<?>) value).size() == 1) {
343                                                                                        setFeature(ret, structuralFeature, ((List<?>) value).get(0));
344                                                                                } else {
345                                                                                        setFeature(ret, structuralFeature, value);
346                                                                                }
347                                                                        }
348                                                                } else {
349                                                                        setFeature(ret, structuralFeature, value);
350                                                                }
351                                                        }
352                                                        loadedFeatures.put(structuralFeature, value);
353                                                } catch (ConfigurationException e) {
354                                                        throw e;
355                                                } catch (Exception e) {
356                                                        throw markers == null || markers.isEmpty() ? new ExecutionException(e) : new ConfigurationException("Error setting feature " + feature.getKey() + " in " + ret + ": " + e, e, markers);
357                                                } finally {
358                                                        loadingFeature[0] = null;
359                                                }
360                                        }
361                                }
362                                return ret;
363                        }
364                };
365        }
366        
367        protected EObject instantiate(EClass eClazz) {
368                return eClazz.getEPackage().getEFactoryInstance().create(eClazz);
369        }       
370
371        /**
372         * Sets structural feature. This implementation calls eObj.eSet().  
373         * @param eObj
374         * @param structuralFeature
375         * @param value
376         */
377        protected void setFeature(EObject eObj, EStructuralFeature structuralFeature, Object value) {
378                eObj.eSet(structuralFeature, value);
379        }
