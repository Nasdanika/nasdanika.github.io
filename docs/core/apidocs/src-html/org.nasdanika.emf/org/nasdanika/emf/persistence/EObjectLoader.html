132         * 
133         * @param eClass
134         * @param feature
135         * @return Features in the argument class the argument feature is exclusive with.
136         */
137        public static Object[] getExclusiveWith(EClass eClass, EStructuralFeature feature, BiFunction<EClass,ENamedElement,String> keyProvider) {
138                Set<String> ret = new HashSet<>();
139                for (EStructuralFeature sf: eClass.getEAllStructuralFeatures()) {
140                        String exclusiveWithStr = NcoreUtil.getNasdanikaAnnotationDetail(sf, "exclusive-with"); // A space-separated list of load keys which are mutually exclusive with the annotated feature.
141                        if (!Util.isBlank(exclusiveWithStr)) {
142                                for (String ew: exclusiveWithStr.split("\\s")) {
143                                        if (sf == feature) {
144                                                ret.add(ew);
145                                        } else if (ew.equals(keyProvider.apply(eClass, feature))) {
146                                                ret.add(keyProvider.apply(eClass, sf));
147                                        }
148                                        
149                                }
150                        }
151                }
152
153                return ret.toArray();
154        }

162        public static boolean isDefaultFeature(EClass eClass, EStructuralFeature feature) {
163                if ("true".equals(NcoreUtil.getNasdanikaAnnotationDetail(feature, IS_DEFAULT_FEATURE))) {
164                        return true;
165                }
166                String dfName = NcoreUtil.getNasdanikaAnnotationDetail(eClass, IS_DEFAULT_FEATURE);
167                if (!Util.isBlank(dfName)) {
168                        return feature.getName().equals(dfName);
169                }
170                for (EClass st: eClass.getEAllSuperTypes()) {
171                        dfName = NcoreUtil.getNasdanikaAnnotationDetail(st, IS_DEFAULT_FEATURE);
172                        if (!Util.isBlank(dfName)) {
173                                return feature.getName().equals(dfName);
174                        }                       
175                }
176                return false;
177        }

199        /**
200         * Default implementation of load key provider - gets load key from the Nasdanika annotation (urn:org.nasdanika) <code>load-key</code> details if it is present.
201         * For features delegates to {@link NcoreUtil} getFeatureKey.
202         * Otherwise converts {@link ENamedElement} element name from camel case to lower-cased kebab case. E.g. firstName is converted first-name.
203         */
204        public static final BiFunction<EClass,ENamedElement,String> LOAD_KEY_PROVIDER = (eClass,ne) -> {
205                EAnnotation na = NcoreUtil.getNasdanikaAnnotation(ne);
206                if (na != null) {
207                        if (na.getDetails().containsKey(IS_LOADABLE) && "false".equals(na.getDetails().get(IS_LOADABLE))) {
208                                return null;
209                        }
210                        if (na.getDetails().containsKey(LOAD_KEY)) {
211                                return na.getDetails().get(LOAD_KEY);
212                        }
213                }
214
215                if (eClass != null && ne instanceof EStructuralFeature) {
216                        EStructuralFeature feature = (EStructuralFeature) ne;
217                        String loadKey = getLoadKey(eClass, feature);           
218                        return loadKey == null ? NcoreUtil.getFeatureKey(eClass, feature) : loadKey;
219                }
220                
221                return Util.camelToKebab(ne.getName());
222        };

247        public EObjectLoader(ObjectLoader chain, BiFunction<EClass,ENamedElement,String> keyProvider, EPackage... ePackages) {
248                super(chain);
249                this.keyProvider = keyProvider == null ? LOAD_KEY_PROVIDER : keyProvider;
250                if (ePackages.length > 0) {
251                        Map<String,EPackageEntry> registry = new LinkedHashMap<>();
252                        for (EPackage ePackage: ePackages) {
253                                Entry<String, EPackageEntry> re = createRegistryEntry(ePackage);
254                                if (re != null) {
255                                        registry.put(re.getKey(), re.getValue());
256                                }
257                        }
258                        registrySupplier = registry::entrySet;
259                }
260        }

289        private Map.Entry<String, EPackageEntry> createRegistryEntry(EPackage ePackage, BiFunction<EClass,ENamedElement,String> keyProvider) {
290                if (keyProvider == null) {
291                        keyProvider = this.keyProvider;
292                }
293                String packageKey = keyProvider.apply(null, ePackage);
294                if (Util.isBlank(packageKey)) {
295                        return null;
296                }
297                return Map.entry(packageKey + "-", new EPackageEntry(ePackage, keyProvider == null ? this.keyProvider : keyProvider));
298        }

332        /**
333         * Resolves EClass and key provider from type name.
334         * @param type
335         * @return
336         */
337        public BiSupplier<EClass,BiFunction<EClass,ENamedElement,String>> resolveEClass(String type) {          
338                for (Entry<String, EPackageEntry> re: getRegistry()) {
339                        if (type.startsWith(re.getKey())) {
340                                EPackageEntry ePackageEntry = re.getValue();
341                                for (EClassifier eClassifier: ePackageEntry.ePackage.getEClassifiers()) {
342                                        if (eClassifier instanceof EClass && "true".equals(NcoreUtil.getNasdanikaAnnotationDetail(eClassifier, IS_LOADABLE, "true"))) {
343                                                String eClassifierKey = ePackageEntry.keyProvider.apply(null, eClassifier);
344                                                if (eClassifierKey != null && type.equals(re.getKey() + eClassifierKey)) {
345                                                        return new BiSupplier<EClass, BiFunction<EClass,ENamedElement,String>>() {
346
347                                                                @Override
348                                                                public EClass getFirst() {
349                                                                        return (EClass) eClassifier;
350                                                                }
351
352                                                                @Override
353                                                                public BiFunction<EClass,ENamedElement, String> getSecond() {
354                                                                        return ePackageEntry.keyProvider;
355                                                                }
356                                                        }; 
357                                                }
358                                        }
359                                }
360                        }
361                }
362                return null;            
363        }

365        /**
366         * Creates an instance of EClass or a supplier factory
367         * @param loader
368         * @param eClass
369         * @param config
370         * @param base
371         * @param progressMonitor
372         * @param marker
373         * @param ePackageEntry
374         * @param eClassifier
375         * @return
376         */
377        public Object create(
378                        ObjectLoader loader, 
379                        EClass eClass, 
380                        Object config, 
381                        URI base, 
382                        ProgressMonitor progressMonitor, 
383                        List<? extends Marker> markers, 
384                        BiFunction<EClass,ENamedElement,String> keyProvider,
385                        java.util.function.Function<EClass, EObject> constructor) {
386                
387                // Nulls are nulls
388                if (config == null) {
389                        return null;
390                }
391                
392                // Proxy
393                Object proxy = createProxy(eClass, config, base, markers, progressMonitor);
394                if (proxy != null) {
395                        return configure(proxy, loader, config, base, null, progressMonitor);
396                }
397                
398                Class<?> instanceClass = eClass.getInstanceClass();
399                EFactory factory = eClass.getEPackage().getEFactoryInstance();
400                
401                if (Loadable.class.isAssignableFrom(instanceClass)) {
402                        EObject eObject = factory.create(eClass);
403                        ((Loadable) eObject).load(loader, config, base, progressMonitor, markers);
404                        return configure(eObject, loader, config, base, null, progressMonitor);
405                }
406                if (ContextLoadable.class.isAssignableFrom(instanceClass)) {
407                        return new SupplierFactory<EObject>() {
408
409                                @Override
410                                public Supplier<EObject> create(Context context) {
411                                        return new Supplier<EObject>() {
412
413                                                @Override
414                                                public double size() {
415                                                        return 1;
416                                                }
417
418                                                @Override
419                                                public String name() {
420                                                        return "Creating " + eClass.getName();
421                                                }
422
423                                                @Override
424                                                public EObject execute(ProgressMonitor pm) {
425                                                        EObject eObject = factory.create(eClass);
426                                                        ((ContextLoadable) eObject).load(context, loader, config, base, pm, markers);
427                                                        return configure(eObject, loader, config, base, context, pm);
428                                                }
429                                                
430                                        };
431                                }                                                                       
432                        };
433                }                                                       
434                EObjectSupplierFactory eObjectSupplierFactory = createEObjectSupplierFactory(eClass, keyProvider, constructor);
435                eObjectSupplierFactory.load(loader, config, base, progressMonitor, markers);
436                FunctionFactory<EObject, EObject> configureFactory = ctx -> new org.nasdanika.common.Function<EObject, EObject>() {
437
438                        @Override
439                        public double size() {
440                                return 1;
441                        }
442
443                        @Override
444                        public String name() {
445                                return "Configure";
446                        }
447
448                        @Override
449                        public EObject execute(EObject eObject, ProgressMonitor pm) {
450                                return configure(eObject, loader, config, base, ctx, pm);
451                        }
452                        
453                };
454                return eObjectSupplierFactory.then(configureFactory);
455        }

457        /**
458         * Override to configure an object after creation. This method can replace the object.
459         * @param obj Created object to configure.
460         * @param loader Root loader.
461         * @param config Object config.
462         * @param base Base URI.
463         * @param context Context. Not null if created by a supplier.
464         * @param progressMonitor Progress monitor.
465         * @return Configured object. This implementation loads linked resources.
466         */
467        @SuppressWarnings("unchecked")
468        protected <T> T configure(
469                        T obj,
470                        ObjectLoader loader, 
471                        Object config, 
472                        URI base, 
473                        Context context,
474                        ProgressMonitor progressMonitor) {
475                
476                // Loading linked resources if resourceSet is not null.
477                if (resourceSet != null) {
478                        if (obj instanceof EObject) {
479                                EObject eObj = (EObject) obj;
480                                EClass eClass = eObj.eClass();
481                                for (EAttribute eAttr: eClass.getEAllAttributes()) {
482                                        if (isLinkedResourceAttribute(eAttr)) {
483                                                linkResources(eObj, eObj, eAttr, loader, config, base, context, progressMonitor);
484                                        }
485                                }
486                                
487                                // EMaps with EString value attributes annotated with content-type = resource-uri.
488                                for (EReference eRef: eClass.getEAllReferences()) {
489                                        EClassifier refType = eRef.getEType();
490                                        Class<?> refClass = refType.getInstanceClass();
491                                        if (eRef.isMany() && refType instanceof EClass && refClass != null && Map.Entry.class.isAssignableFrom(refClass)) {
492                                                EStructuralFeature valueFeature = ((EClass) refType).getEStructuralFeature("value");
493                                                if (valueFeature instanceof EAttribute && isLinkedResourceAttribute((EAttribute) valueFeature)) {
494                                                        for (EObject entry: (Collection<EObject>) eObj.eGet(eRef)) {
495                                                                linkResources(eObj, entry, (EAttribute) valueFeature, loader, config, base, context, progressMonitor);                                                  
496                                                        }
497                                                }
498                                        }
499                                }
500                        }
501                }
502                
503                return obj;
504        }
505        
506        protected boolean isLinkedResourceAttribute(EAttribute eAttribute) {
507                return eAttribute != null && eAttribute.getEType() == EcorePackage.Literals.ESTRING && "resource-uri".equals(NcoreUtil.getNasdanikaAnnotationDetail(eAttribute, "content-type")); 
508        }

510        /**
511         * Loads and links resources specified in the valueObject's linkedResourceAttribute to the semanticElement.
512         * @param semanticElement Loaded object to link the resource to.
513         * @param valueObject Value object containing linked resource attribute, can be different from the semanticElement in case of EMap's.
514         * @param linkedResourceAttribute Attribute containing resource URI.
515         * @param loader 
516         * @param config
517         * @param base
518         * @param context
519         * @param progressMonitor
520         */
521        @SuppressWarnings("unchecked")
522        protected void linkResources(
523                        EObject semanticElement, 
524                        EObject valueObject, 
525                        EAttribute linkedResourceAttribute,
526                        ObjectLoader loader, 
527                        Object config, 
528                        URI base, 
529                        Context context,
530                        ProgressMonitor progressMonitor) {
531                
532                if (linkedResourceAttribute.isMany()) {
533                        ListIterator<String> it = ((List<String>) valueObject.eGet(linkedResourceAttribute)).listIterator();
534                        while (it.hasNext()) {
535                                it.set(linkResource(semanticElement, it.next(), loader, config, base, context, progressMonitor));
536                        }
537                } else {
538                        valueObject.eSet(linkedResourceAttribute, linkResource(semanticElement, (String) valueObject.eGet(linkedResourceAttribute), loader, config, base, context, progressMonitor));
539                }               
540        }

542        /**
543         * Loads and links resources specified in the valueObject's linkedResourceAttribute to the semanticElement.
544         * @param semanticElement Loaded object to link the resource to.
545         * @param resourceUriStr
546         * @param loader
547         * @param config
548         * @param base
549         * @param context
550         * @param progressMonitor
551         * @return
552         */
553        @SuppressWarnings("unchecked")
554        protected String linkResource(
555                        EObject semanticElement,
556                        String resourceUriStr,
557                        ObjectLoader loader, 
558                        Object config, 
559                        URI base, 
560                        Context context,
561                        ProgressMonitor progressMonitor) {
562                
563                URI resourceURI = URI.createURI(resourceUriStr);
564                if (resourceURI.isRelative() && base != null && !base.isRelative() && base.isHierarchical()) {
565                        resourceURI = resourceURI.resolve(base);
566                }
567                Resource linkedResource = resourceSet.getResource(resourceURI, true);
568                
569                // Adding to linked resources adapter for setting children in createSemanticElement                             
570                LinkedResourcesAdapter linkedResourcesAdapter = (LinkedResourcesAdapter) EcoreUtil.getRegisteredAdapter(semanticElement, LinkedResourcesAdapter.class);
571                if (linkedResourcesAdapter == null) {
572                        linkedResourcesAdapter = new LinkedResourcesAdapter();
573                        semanticElement.eAdapters().add(linkedResourcesAdapter);
574                }
575                linkedResourcesAdapter.getLinkedResources().add(linkedResource);
576                
577                if (linkedResource instanceof AbstractEObjectFactoryProcessorResource) {
578                        ((AbstractEObjectFactoryProcessorResource<EObject>) linkedResource).setParent(semanticElement);
579                }
580                
581                return encodeLinkedResource(semanticElement, resourceUriStr, linkedResource, loader, config, base, context, progressMonitor);
582        }

586         * This implementation replaces resourceUriStr value with encoded Drawio document so it can be used to generate interactive diagrams in documentation. 
587         * @param semanticElement Loaded object to link the resource to.
588         * @param resourceUriStr

590         * @param loader
591         * @param config
592         * @param base
593         * @param context

597        protected String encodeLinkedResource(

599                        String resourceUriStr,
600                        Resource linkedResource,
601                        ObjectLoader loader,

609                }
610                
611                return resourceUriStr;  
612        }

632         * @return Proxy object or null if config is not a proxy config.
633         */
634        public EObject createProxy(EClass eClass, Object config, URI base, List<? extends Marker> markers, ProgressMonitor progressMonitor) {
635                if (config instanceof Map) {
636                        @SuppressWarnings("unchecked")
637                        Map<Object,Object> configMap = (Map<Object,Object>) config;
638                        if (configMap.size() == 1 
639                                        && configMap.containsKey(HREF_KEY)) {
640                                EObject eObject = eClass.getEPackage().getEFactoryInstance().create(eClass);
641                                if (eObject instanceof MinimalEObjectImpl) {
642                                        Object href = configMap.get(HREF_KEY);                                  
643                                        URI proxyURI;
644                                        if (href instanceof URI) {
645                                                proxyURI = (URI) href;
646                                        } else if (href instanceof String) {
647                                                proxyURI = URI.createURI((String) configMap.get(HREF_KEY));
648                                        } else {
649                                                return null;
650                                        }
651                                        
652                                        String proxyURIStr = proxyURI.toString();
653                                        if (proxyURIStr.startsWith(LATE_PROXY_RESOLUTION_URI_PREFIX)) {
654                                                proxyURI = URI.createURI(proxyURIStr.substring(LATE_PROXY_RESOLUTION_URI_PREFIX.length()));
655                                        } else if (base != null) {
656                                                proxyURI = proxyURI.resolve(base);
657                                        }
658                                        ((MinimalEObjectImpl) eObject).eSetProxyURI(proxyURI);
659                                        mark(eObject, markers, progressMonitor);
660                                        
661                                        return eObject;
662                                }
663                        }
664                }
665                return null;
666        }

668        /**
669         * If marker is not null adds marked adapter. If eObject is instance of {@link Marked} creates and sets {@link org.nasdanika.ncore.Marker}.
670         * @param eObject
671         * @param marker
672         */
673        public void mark(EObject eObject, List<? extends Marker> markers, ProgressMonitor progressMonitor) {
674                if (markers != null) {
675                        eObject.eAdapters().add(new MarkedAdapter(markers));
676                        if (eObject instanceof Marked) {                
677                                for (Marker marker: markers) {
678                                        org.nasdanika.ncore.Marker mMarker = markerFactory.createMarker(marker.getLocation(), progressMonitor);
679                                        mMarker.setPosition(marker.getPosition());
680                                        ((Marked) eObject).getMarkers().add(0, mMarker);
681                                }
682                        }
683                }
684        }

686        /**
687         * Resolves proxy using the loader's resource set.
688         * Does not suppress exceptions. 
689         */
690        public EObject resolve(EObject proxy) {
691                if (!proxy.eIsProxy()) {
692                        return proxy;
693                }
694
695                URI proxyURI = ((InternalEObject) proxy).eProxyURI();
696                EObject ret = resourceSet.getEObject(proxyURI, true);
697                return ret == null || ret == proxy ? ret : resolve(ret);
698        }

701                return resourceSet;

704        /**
705         * @return Load key from <code>load-keys</code> annotation. 
706         */
707        public static String getLoadKey(EClass eClass, EStructuralFeature feature) {
708                for (EClass ec: NcoreUtil.lineage(eClass)) {
709                        String loadKeys = NcoreUtil.getNasdanikaAnnotationDetail(ec, LOAD_KEYS);
710                        if (!Util.isBlank(loadKeys)) {
711                                Yaml yaml = new Yaml();
712                                Map<String,Object> loadKeyMap = yaml.load(loadKeys);
713                                Object loadKey = loadKeyMap.get(feature.getName());
714                                if (loadKey instanceof String) {
715                                        return (String) loadKey;
716                                }
717                        }
718                }
719                return null;
720        }
