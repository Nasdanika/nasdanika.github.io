001package org.nasdanika.emf.persistence;
002
003import java.util.HashMap;
004import java.util.Map;
005import java.util.concurrent.atomic.AtomicLong;
006import java.util.function.BiFunction;
007
008import org.eclipse.emf.common.notify.Notification;
009import org.eclipse.emf.common.notify.impl.AdapterImpl;
010import org.eclipse.emf.common.notify.impl.NotificationImpl;
011import org.eclipse.emf.ecore.EObject;
012import org.eclipse.emf.ecore.EStructuralFeature;
013
014/**
015 * Caches computed feature values. Clears the cache on notifications.
016 * @author Pavel
017 *
018 */
019public class FeatureCacheAdapter extends AdapterImpl implements FeatureCache {
020        
021        private static AtomicLong calls = new AtomicLong();
022        private static AtomicLong misses = new AtomicLong();
023        private static AtomicLong computeTime = new AtomicLong();
024        
025        public static long getCalls() {
026                return calls.get();
027        }
028
029        public static long getMisses() {
030                return misses.get();
031        }
032
033        public static long getComputeTime() {
034                return computeTime.get();
035        }
036
037        /**
038         * Identity based notification to clear all caches.
039         */
040        public static final Notification CLEAR_CACHE = new NotificationImpl(Notification.EVENT_TYPE_COUNT + 1, null, null);
041        
042        private Map<EStructuralFeature, Object> cache = new HashMap<>();
043                
044        @Override
045        public boolean isAdapterForType(Object type) {
046                return FeatureCache.class == type;
047        }
048        
049        // TODO - hit/miss ratio
050
051        @SuppressWarnings("unchecked")
052        @Override
053        public <F extends EStructuralFeature, T> T get(F feature, BiFunction<EObject, F, T> computer) {
054                calls.incrementAndGet();
055                return (T) cache.computeIfAbsent(feature, f -> {
056                        misses.incrementAndGet();
057                        long start = System.nanoTime();
058                        T result = computer.apply((EObject) getTarget(), (F) f);
059                        computeTime.addAndGet(System.nanoTime() - start);
060                        return result;
061                });
062        }
063        
064        @Override
065        public void notifyChanged(Notification msg) {
066                super.notifyChanged(msg);
067                if (msg == CLEAR_CACHE) {
068                        cache.clear();
069                }
070        }
071
072}
