001package org.nasdanika.common;
002
003/**
004 * Delegates to the target {@link ProgressMonitor}. 
005 * Subclass and override methods as needed. 
006 * @author Pavel
007 *
008 */
009public class FilterProgressMonitor implements ProgressMonitor {
010        
011        protected ProgressMonitor target;
012
013        public FilterProgressMonitor(ProgressMonitor target) {
014                this.target = target;
015        }
016
017        @Override
018        public void close() {
019                target.close();
020        }
021
022        @Override
023        public boolean isCancelled() {
024                return target.isCancelled();
025        }
026
027        @Override
028        public ProgressMonitor split(String taskName, double size, Object... data) {
029                return target.split(taskName, size, data); 
030        }
031
032        @Override
033        public void worked(Status status, double work, String progressMessage, Object... data) {
034                target.worked(status, work, progressMessage, data);
035        }
036
037        @Override
038        public ProgressMonitor setWorkRemaining(double size) {
039                target.setWorkRemaining(size);
040                return this;
041        }
042
043}
