001package org.nasdanika.common;
002
003import org.apache.commons.text.StringEscapeUtils;
004import org.json.JSONObject;
005
006public class DiagramGeneratorImpl implements DiagramGenerator {
007        
008        @Override
009        public String generateDiagram(String spec, String dialect) {
010                if (DRAWIO_DIALECT.equals(dialect)) {
011                        JSONObject data = new JSONObject();
012                        data.put("highlight", "#0000ff");
013                        data.put("nav", true);
014                        data.put("resize",true);
015                        data.put("toolbar", "zoom layers tags lightbox");
016                        data.put("edit","_blank");
017                        
018                        data.put("xml", spec);
019                        
020                        String diagramDiv = "<div class=\"mxgraph\" style=\"max-width:100%;border:1px solid transparent;\" data-mxgraph=\"" + StringEscapeUtils.escapeHtml4(data.toString()) + "\"></div>";
021                        String drawioViewer = getDrawioViewer();
022                        if (Util.isBlank(drawioViewer)) {
023                                return diagramDiv;
024                        }
025                        String script = "<script type=\"text/javascript\" src=\"" + drawioViewer + "\"></script>";
026                        return diagramDiv + System.lineSeparator() + script;            
027                }
028                
029                if (MERMAID_DIALECT.equals(dialect)) {
030                        String mermaidDiv = "<div class=\"mermaid\">"  + System.lineSeparator() + spec + System.lineSeparator() + "</div>";
031                        String mermaidInitialize = getMermaidInitialize();
032                        if (Util.isBlank(mermaidInitialize)) {
033                                return mermaidDiv;
034                        }
035                        return mermaidDiv
036                                        + System.lineSeparator()
037                                        + "<script>"
038                                        + System.lineSeparator()
039                                        + mermaidInitialize
040                                        + System.lineSeparator()
041                                        + "</script>";                                  
042                }
043                
044                throw new UnsupportedOperationException("Unsupported dialect: " + dialect);
045        }
046
047        protected String getDrawioViewer() {
048                return "https://cdn.jsdelivr.net/gh/Nasdanika/drawio@dev/src/main/webapp/js/viewer-static.min.js";
049        }
050        
051        protected String getMermaidInitialize() {
052                return "mermaid.initialize({ startOnLoad: true, securityLevel: 'loose' });";
053        }
054
055        @Override
056        public boolean isSupported(String dialect) {
057                return DRAWIO_DIALECT.equals(dialect) || MERMAID_DIALECT.equals(dialect);
058        }
059}
