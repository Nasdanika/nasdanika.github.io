001package org.nasdanika.common;
002
003/**
004 * Something that can be composed to create an aggregate. E.g. {@link AccessController}s can be composed with AND operation,
005 * {@link Context}s can be composed by chaining.
006 * @author Pavel
007 *
008 * @param <T>
009 */
010public interface Composeable<T> {
011        
012        /**
013         * @param <T> Instances of T shall implement {@link Composeable}.
014         * @return Composing operator which can be use in reducing streams of {@link Composeable}s to a single composeable.
015         */
016        @SuppressWarnings("unchecked")
017        static <T> T composer(T a, T b) {       
018                if (a == null) {
019                        return b;
020                }
021                if (b == null) {
022                        return a;
023                }
024                return ((Composeable<T>) a).compose(b);
025        }
026
027        /**
028         * Returns a composition of this instance and the other with this instance being 
029         * the first if the order of composition is important.
030         * @param other
031         * @return
032         */
033        T compose(T other);
034        
035}
