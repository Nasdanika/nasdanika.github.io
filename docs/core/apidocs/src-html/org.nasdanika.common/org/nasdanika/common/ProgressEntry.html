001package org.nasdanika.common;
002
003import java.util.ArrayList;
004import java.util.Arrays;
005import java.util.List;
006import java.util.Map;
007
008import org.json.JSONArray;
009import org.json.JSONObject;
010
011/**
012 * Progress entry storing progress information for subsequent processing.
013 * @author Pavel
014 *
015 */
016public class ProgressEntry extends ProgressRecorder {
017
018        /**
019         * 
020         * @param totalWork
021         */
022        ProgressEntry(ProgressRecorder parent, String name, double totalWork, List<Object> data) {
023                super(parent);
024                this.name = name;               
025                this.totalWork = totalWork;
026                if (data != null) {
027                        this.data.addAll(data);
028                }
029                this.start = System.currentTimeMillis();
030        }
031        /**
032         * 
033         * @param totalWork
034         */
035        public ProgressEntry(String name, long totalWork, Object... data) {
036                this(null, name, totalWork, Arrays.asList(data));
037        }
038        
039        private String name;
040        
041        public String getName() {
042                return name;
043        }
044        
045        private List<Object> data = new ArrayList<>();
046        
047        public List<Object> getData() {
048                return data;
049        }
050        
051        private double totalWork;
052                        
053        /**
054         * @return Entry size passed to split() or constructor, not total work reported by worked() calls.
055         */
056        public double getTotalWork() {
057                return totalWork;
058        }
059        
060        private long start;
061        
062        private long finish;
063        
064        /**
065         * @return Time when this entry was created.
066         */
067        public long getStart() {
068                return start;
069        }
070        
071        /**
072         * @return Time when this entry was closed.
073         */
074        public long getFinish() {
075                return finish;
076        }
077        
078        /**
079         * Outputs to JSON.
080         * @return
081         */
082        @Override
083        public JSONObject toJSON(int depth, boolean withData) {
084                JSONObject ret = super.toJSON(depth, withData);
085                ret.put("name", getName());
086                ret.put("totalWork", totalWork);
087                if (start > 0) {
088                        ret.put("start", start);
089                }
090                if (finish > 0) {
091                        ret.put("finish", finish);
092                }
093                
094                if (withData && getData() != null && getData().size() > 0) {
095                        JSONArray jd = new JSONArray();
096                        for (Object d: getData()) {
097                                jd.put(detailToJSON(d));
098                        }
099                        ret.put("data", jd);                                    
100                }                                               
101                
102                return ret;
103        }
104        
105        /**
106         * Outputs to Map, which can be then used to output to YAML.
107         * @param depth Output depth, 0 means infinite depth.
108         * @return
109         */
110        @Override
111        public Map<String, Object> toMap(int depth, boolean withData) {
112                Map<String,Object> ret = super.toMap(depth, withData);
113                ret.put("name", getName());
114                ret.put("totalWork", totalWork);
115                if (start > 0) {
116                        ret.put("start", start);
117                }
118                if (finish > 0) {
119                        ret.put("finish", finish);
120                }
121                if (start > 0 && finish > 0) {
122                        ret.put("duration", finish - start);
123                }
124                
125                if (withData) {
126                        ret.put("data", getData());
127                }
128                
129                return ret;
130        }
131        
132        @Override
133        public void close() {
134                finish = System.currentTimeMillis();
135                super.close();
136        }
137
138}
