001package org.nasdanika.common;
002
003import java.util.ArrayList;
004import java.util.Arrays;
005import java.util.LinkedHashMap;
006import java.util.List;
007import java.util.Map;
008import java.util.concurrent.CancellationException;
009import java.util.stream.Collectors;
010
011import org.json.JSONArray;
012import org.json.JSONObject;
013
014/**
015 * Progress recorder storing progress information for subsequent processing or replaying.
016 * @author Pavel
017 *
018 */
019public class ProgressRecorder implements ProgressMonitor {
020        
021        private ProgressRecorder parent;
022
023        /**
024         * 
025         * @param totalWork
026         */
027        protected ProgressRecorder(ProgressRecorder parent) {
028                this.parent = parent;
029        }
030        /**
031         * 
032         * @param totalWork
033         */
034        public ProgressRecorder() {
035                this(null);
036        }
037
038        /**
039         * Supplier step reported by worked() methods.
040         * @author Pavel
041         *
042         */
043        public interface Step extends Diagnostic {
044                
045                long getTime();
046                
047                double getWorked();
048                
049        }
050        
051        private boolean closed;
052        
053        private List<Object> entries = new ArrayList<>();
054
055        private boolean cancelled;
056        
057        protected void cancel() {
058                cancelled = true;
059        }
060        
061        /**
062         * @return Steps and child entries created by split().
063         */
064        public List<Object> getEntries() {
065                return entries;
066        }
067        
068        /**
069         * @return Child entries created by split().
070         */
071        public List<ProgressEntry> getChildren() {
072                return entries.stream().filter(e -> e instanceof ProgressEntry).map(e -> (ProgressEntry) e).collect(Collectors.toList());
073        }       
074        
075        public List<Step> getSteps() {
076                return entries.stream().filter(e -> e instanceof Step).map(e -> (Step) e).collect(Collectors.toList());
077        }
078        
079        @Override
080        public void close() {
081                closed = true;
082        }
083
084        @Override
085        public boolean isCancelled() {
086                return parent == null ? cancelled : parent.isCancelled();
087        }
088
089        @Override
090        public ProgressMonitor split(String taskName, double size, Object... data) {
091                if (isCancelled()) {
092                        throw new CancellationException();
093                }
094                if (closed) {
095                        throw new IllegalStateException("Monitor is closed: "+hashCode());
096                }
097                ProgressRecorder child = new ProgressEntry(this, taskName, size, Arrays.asList(data));
098                entries.add(child);
099                return child;
100        }
101
102        @Override
103        public void worked(Status status, double work, String progressMessage, Object... data) {
104                if (closed) {
105//                      throw new IllegalStateException("Monitor is closed");
106                        progressMessage += " (Warning - Reporting progress to a closed monitor)";
107                }
108                String theProgressMessage = progressMessage;
109                
110                class BasicStep extends BasicDiagnostic implements Step {
111
112                        public BasicStep() {
113                                super(status, theProgressMessage, data);
114                        }
115
116                        long now = System.currentTimeMillis();                  
117
118                        @Override
119                        public long getTime() {
120                                return now;
121                        }
122
123                        @Override
124                        public double getWorked() {
125                                return work;
126                        }
127                        
128                }
129                
130                entries.add(new BasicStep());
131                if (status == Status.CANCEL) {
132                        if (parent == null) {
133                                cancelled = true;
134                        } else {
135                                parent.cancel();
136                        }
137                }
138        }
139        
140        /**
141         * Outputs to JSON.
142         * @return
143         */
144        public JSONObject toJSON(int depth, boolean withData) {
145                JSONObject ret = new JSONObject();
146
147                if (depth != 1) {
148                        List<ProgressEntry> children = getChildren();
149                        if (!children.isEmpty()) {
150                                JSONArray jChildren = new JSONArray();
151                                int cDepth = depth - 1;
152                                children.forEach(child -> jChildren.put(child.toJSON(cDepth, withData)));
153                                ret.put("children", jChildren);
154                        }
155                }
156                
157                List<Step> steps = getSteps();
158                if (!steps.isEmpty()) {
159                        JSONArray jSteps = new JSONArray();
160                        steps.forEach(we -> {
161                                JSONObject jwe = new JSONObject();
162                                jwe.put("status", we.getStatus().name());
163                                jwe.put("worked", we.getWorked());
164                                jwe.put("time", we.getTime());
165                                jwe.put("message", we.getMessage());
166                                if (withData && we.getData().size() > 0) {
167                                        JSONArray jd = new JSONArray();
168                                        for (Object d: we.getData()) {
169                                                jd.put(detailToJSON(d));
170                                        }
171                                        jwe.put("data", jd);                                    
172                                }
173                                jSteps.put(jwe);
174                        });
175                        ret.put("steps", jSteps);               
176                }
177                
178                return ret;
179        }
180        
181        /**
182         * Converts detail to JSON. This implementation delegates to the parent or calls {@link DefaultConverter#}.INSTANCE.convert
183         * if parent is null.
184         * @param detail
185         * @return
186         */
187        protected Object detailToJSON(Object detail) {
188                return parent == null ? String.valueOf(detail) /* DefaultConverter.INSTANCE.convert(detail, JSONObject.class) */ : parent.detailToJSON(detail);
189        }
190
191        @Override
192        public String toString() {
193                return toJSON(0, true).toString(4);
194        }
195        
196        /**
197         * Outputs to Map, which can be then used to output to YAML.
198         * @param depth output depth. 0 means infinite.
199         * @return 
200         */
201        public Map<String, Object> toMap(int depth, boolean withData) {
202                Map<String,Object> ret = new LinkedHashMap<>();
203
204                if (depth != 1) {
205                        List<ProgressEntry> children = getChildren();
206                        if (!children.isEmpty()) {
207                                List<Map<String, Object>> mChildren = new ArrayList<>();
208                                int cDepth = depth - 1;
209                                children.forEach(child -> mChildren.add(child.toMap(cDepth, withData)));
210                                ret.put("children", mChildren);
211                        }
212                }
213                
214                List<Step> steps = getSteps();
215                if (!steps.isEmpty()) {
216                        List<Map<String, Object>> mSteps = new ArrayList<>();
217                        steps.forEach(we -> {
218                                Map<String, Object> mwe = new LinkedHashMap<>();
219                                mwe.put("status", we.getStatus().name());
220                                mwe.put("worked", we.getWorked());
221                                mwe.put("time", we.getTime());
222                                mwe.put("message", we.getMessage());
223                                if (withData && we.getData().size() > 0) {
224                                        mwe.put("data", we.getData());                                  
225                                }
226                        });
227                        ret.put("steps", mSteps);               
228                }                       
229                
230                return ret;
231        }
232        
233        /**
234         * Replays recorded progress to the target monitor.  
235         * @param target
236         */
237        public void replay(ProgressMonitor monitor) {
238                for (Object entry: entries) {
239                        if (entry instanceof Step) {
240                                Step step = (Step) entry;
241                                monitor.worked(step.getStatus(),  step.getWorked(),  step.getMessage(), step.getData());
242                        } else {
243                                @SuppressWarnings("resource")
244                                ProgressEntry child = (ProgressEntry) entry;
245                                try (ProgressMonitor subMonitor = monitor.split(child.getName(), child.getTotalWork(), child.getData())) {
246                                        child.replay(subMonitor);
247                                }
248                        }
249                }
250        }
251        @Override
252        public ProgressRecorder setWorkRemaining(double size) {
253                // TODO - record too?
254                return this;
255        }
