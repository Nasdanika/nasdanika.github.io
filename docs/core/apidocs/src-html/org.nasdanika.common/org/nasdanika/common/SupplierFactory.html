001package org.nasdanika.common;
002
003import java.util.function.BiFunction;
004
005/**
006 * @author Pavel Vlasov
007 *
008 * @param <T>
009 */
010public interface SupplierFactory<T> extends ExecutionParticipantFactory<Supplier<T>>, java.util.function.BiFunction<Context,ProgressMonitor,T> {
011                
012        static <T> SupplierFactory<T> empty() {
013                return context -> Supplier.empty();
014        }       
015        
016        /**
017         * Provides {@link SupplierFactory} for a specific type.
018         * @author Pavel
019         *
020         */
021        interface Provider {
022                <T> SupplierFactory<T> getFactory(Class<T> type);
023        }
024                
025        default <V> SupplierFactory<V> then(FunctionFactory<? super T,V> then) {
026                return new SupplierFactory<V>() {
027                        
028                        @Override
029                        public Supplier<V> create(Context context) {
030                                return SupplierFactory.this.create(context).then(then.create(context));
031                        }
032                        
033                };
034        }
035        
036        default CommandFactory then(ConsumerFactory<? super T> then) {
037                return new CommandFactory() {
038                        
039                        @Override
040                        public Command create(Context context) {
041                                return SupplierFactory.this.create(context).then(then.create(context));
042                        }
043                        
044                };
045        }
046                
047        @Override
048        default T apply(Context context, ProgressMonitor progressMonitor) {
049                return create(context).apply(progressMonitor);
050        }
051        
052        default <V> FunctionFactory<V,BiSupplier<V,T>> asFunctionFactory() { 
053                return new FunctionFactory<V, BiSupplier<V,T>>() {
054
055                        @Override
056                        public Function<V, BiSupplier<V, T>> create(Context context) {                          
057                                return SupplierFactory.this.create(context).asFunction();
058                        }
059                };
060        }
061        
062        static <T> SupplierFactory<T> from(BiFunction<Context, ProgressMonitor, T> biFunction, String name, double size) {
063                return new SupplierFactory<T>() {
064
065                        @Override
066                        public Supplier<T> create(Context context) {
067                                return Supplier.fromFunction(pm -> biFunction.apply(context, pm), name, size);
068                        }
069                        
070                };
071        }
072        
073        static <T> SupplierFactory<T> from(T value, String name) {
074                return from((c,p) -> value, name, 0);           
075        }       
076        
077        /**
078         * If source is adaptable to SupplierFactory then adapter is returned. Otherwise it is wrapped into a supplier factory
079         * @param <T>
080         * @param value
081         * @param name
082         * @return
083         */
084        @SuppressWarnings("unchecked")
085        static <T> SupplierFactory<T> adapt(Object source, String name) {
086                SupplierFactory<?> ret = Adaptable.adaptTo(source, SupplierFactory.class);
087                if (ret == null) {
088                        ret = from((c,p) -> source, name, 0);
089                }
090                return (SupplierFactory<T>) ret;
091        }       
092        
093        /**
094         * @param contextSupplierFactory Factory which creates a context to be passed to this factory to create supplier.
095         * @return Supplier factory which creates a context using the context supplier factory and then uses that context and this factory to create a supplier.
096         */
097        default SupplierFactory<T> contextify(SupplierFactory<Context> contextSupplierFactory) {
098                return new SupplierFactory<T>() {
099
100                        @Override
101                        public Supplier<T> create(Context context) {
102                                return new ContextifiedExecutionParticipant.ContextifiedSupplier<T>(contextSupplierFactory.create(context), SupplierFactory.this);
103                        }
104                        
105                };
106        }       
107        
108}
