068        /**
069         * Converts name to label text by breaking the name by camel case, capitalizing the first word and lowsercasing the rest.
070         * @return
071         */
072        public static String nameToLabel(String name) {
073                String[] cca = StringUtils.splitByCharacterTypeCamelCase(name);
074                cca[0] = StringUtils.capitalize(cca[0]);
075                for (int i=1; i<cca.length; ++i) {
076                        cca[i] = cca[i].toLowerCase();
077                }
078                return String.join(" ", cca);
079        }

082         * Converts string in camel case, e.g. <code>firstName</code> to kebab case, e.g. <code>first-name</code>
083         * @return
084         */
085        public static String camelToKebab(String str) {
086                String[] ns = StringUtils.splitByCharacterTypeCamelCase(str);
087                for (int i = 0; i < ns.length; ++i) {
088                        ns[i] = ns[i].toLowerCase();
089                }
090                return String.join("-", ns);
091        }

098         * Creates a function for context mapping which recognizes path navigation with ..
099         * @param prefix
100         * @return
101         */
102        public static java.util.function.Function<String,String> hierarchicalMapper(String prefix) {            
103                return key -> {
104                        StringTokenizer st = new StringTokenizer(prefix + key, "/", true);
105                        LinkedList<String> collector = new LinkedList<>();
106                        while (st.hasMoreTokens()) {
107                                String token = st.nextToken();
108                                if ("..".equals(token) && collector.size() > 1 && !"..".equals(collector.get(collector.size() - 2))) {
109                                        collector.removeLast();
110                                        collector.removeLast();
111                                        if (st.hasMoreTokens()) {
112                                                st.nextToken(); 
113                                        }
114                                } else {
115                                        collector.add(token);
116                                }
117                        }
118                        StringBuilder sb = new StringBuilder();
119                        collector.forEach(sb::append);
120                        return sb.toString();                   
121                };
122        }

124        public static InputStream join(Stream<InputStream> streams) throws IOException {
125                ByteArrayOutputStream baos = new ByteArrayOutputStream();
126                streams.forEach(in -> {
127                        if (in != null) {                               
128                                try (BufferedInputStream bin = new BufferedInputStream(in)) {
129                                        int b;
130                                        while ((b = bin.read()) != -1) {
131                                                baos.write(b);
132                                        }                                               
133                                } catch (IOException e) {
134                                        throw new NasdanikaException("Error joining streams: " + e, e);
135                                }
136                        }
137                });
138                baos.close();
139                return new ByteArrayInputStream(baos.toByteArray());
140        }

142        public static InputStream join(InputStream... streams) throws IOException {
143                if (streams.length == 1) {
144                        return streams[0];
145                }
146                ByteArrayOutputStream baos = new ByteArrayOutputStream();
147                for (InputStream in: streams) {
148                        if (in != null) {
149                                try (BufferedInputStream bin = new BufferedInputStream(in)) {
150                                        int b;
151                                        while ((b = bin.read()) != -1) {
152                                                baos.write(b);
153                                        }                                               
154                                }
155                        }
156                }
157                baos.close();
158                return new ByteArrayInputStream(baos.toByteArray());
159        }

161        public static InputStream toStream(Context context, String text) throws IOException {
162                Charset charset = context.get(Charset.class, StandardCharsets.UTF_8);
163                ByteArrayOutputStream baos = new ByteArrayOutputStream();
164                if (text != null) {
165                        try (OutputStreamWriter writer = new OutputStreamWriter(baos, charset)) {
166                                writer.write(text);
167                        }
168                }
169                baos.close();
170                return new ByteArrayInputStream(baos.toByteArray());
171        }

223        public static String toString(Context context, InputStream in) throws IOException {
224                if (in == null) {
225                        return "";
226                }
227                StringWriter sw = new StringWriter();
228                Charset charset = context.get(Charset.class, StandardCharsets.UTF_8);
229                try (Reader reader = new InputStreamReader(new BufferedInputStream(in), charset)) {
230                        int ch;
231                        while ((ch = reader.read()) != -1) {
232                                sw.write(ch);
233                        }
234                }
235                sw.close();
236                return sw.toString();
237        }

460        /**
461         * Extracts first sentence from text
462         * @param context
463         * @param text
464         * @return
465         */
466        public static String firstSentence(String text, int minSentenceLength, int maxSentenceLength, String... abbreviations) {
467                if (text == null || text.length() < minSentenceLength) {
468                        return text;
469                }
470                Matcher matcher = SENTENCE_PATTERN.matcher(text);               
471                Z: while (matcher.find()) {
472                        String group = matcher.group();
473                        for (String abbr: abbreviations) {
474                                if (group.trim().endsWith(abbr)) {
475                                        continue Z;
476                                }
477                        }
478                        if (matcher.end() > minSentenceLength && matcher.end() < maxSentenceLength) {
479                                return text.substring(0, matcher.end());
480                        }
481                }
482                
483                return text.length() < maxSentenceLength ? text : text.substring(0, maxSentenceLength)+"...";
484        }

499        /**
500         * Executes full supplier lifecycle - diagnose, execute, commit/rollback, close.
501         * @param monitor
502         * @param diagnosticConsumer TODO
503         * @param context
504         * @param component
505         * @return
506         */
507        public static <T> T call(Supplier<T> supplier, ProgressMonitor monitor, Consumer<Diagnostic> diagnosticConsumer, Status... failOnStatuses) {
508                try (ProgressMonitor progressMonitor = monitor.setWorkRemaining(3).split("Calling supplier", 3)) {
509                        Diagnostic diagnostic = supplier.splitAndDiagnose(progressMonitor);
510                        if (diagnosticConsumer != null) {
511                                diagnosticConsumer.accept(diagnostic);
512                        }
513                        Status status = diagnostic.getStatus();
514                        if (failOnStatuses.length == 0) {
515                                if (status == Status.FAIL) {
516                                        throw new DiagnosticException(diagnostic);
517                                }
518                        } else {
519                                for (Status failStatus: failOnStatuses) {
520                                        if (status == failStatus) {
521                                                throw new DiagnosticException(diagnostic);
522                                        }                                       
523                                }
524                        }
525                        
526                        try {
527                                T result = supplier.splitAndExecute(progressMonitor);
528                                supplier.splitAndCommit(progressMonitor);
529                                return result;
530                        } catch (Exception e) {
531                                supplier.splitAndRollback(progressMonitor);
532                                throw e;
533                        }
534                } finally {
535                        supplier.close();
536                }
537        }

622        /**
623         * Wraps object into a supplier factory. Handles adapters, and collections.
624         * @param obj
625         * @return
626         */
627        @SuppressWarnings("unchecked")
628        public static <T> SupplierFactory<T> asSupplierFactory(Object obj, Class<T> type, FunctionFactory<List<T>, T> joinFactory) {
629                if (obj instanceof Collection && joinFactory != null) {
630                        ListCompoundSupplierFactory<T> ret = new ListCompoundSupplierFactory<>("Supplier collection");
631                        for (Object e: (Collection<?>) obj) {
632                                SupplierFactory<T> esp = type == InputStream.class ? (SupplierFactory<T>) asInputStreamSupplierFactory(e) : asSupplierFactory(e, type, joinFactory);
633                                if (esp != null) {
634                                        ret.add(esp);
635                                }
636                        }
637                        return ret.then(joinFactory);
638                }
639                
640                if (obj instanceof SupplierFactory) {           
641                        return (SupplierFactory<T>) obj;
642                }
643                
644                if (obj instanceof SupplierFactory.Provider && type != null) {
645                        SupplierFactory<T> factory = ((SupplierFactory.Provider) obj).getFactory(type);
646                        if (factory != null) {
647                                return factory;
648                        }
649                }
650                
651                if (obj instanceof Adaptable) {
652                        Adaptable adaptable = (Adaptable) obj;
653                        if (type != null) {
654                                SupplierFactory.Provider provider = adaptable.adaptTo(SupplierFactory.Provider.class);
655                                if (provider != null) {
656                                        SupplierFactory<T> factory = provider.getFactory(type);
657                                        if (factory != null) {
658                                                return factory;
659                                        }
660                                }
661                        }
662                        
663                        SupplierFactory<T> adapter = adaptable.adaptTo(SupplierFactory.class);
664                        if (adapter != null) {
665                                return adapter;
666                        }
667                }
668                
669                if (obj instanceof EObject) {
670                        Adapter adapter = EcoreUtil.getRegisteredAdapter((EObject) obj, SupplierFactory.class);
671                        if (adapter instanceof SupplierFactory) {
672                                return (SupplierFactory<T>) adapter;
673                        }
674                }
675                
676                return null;
677        }

963        /**
964         * Wraps texts so its line width grows proportionally as the number of lines increases. 
965         * @param text
966         * @param initialLineWidth
967         * @param separator
968         * @return
969         */
970        public static String wrap(String text, int initialLineWidth, int widthIncrementPerLine, String separator) {
971                if (Util.isBlank(text) || text.length() < initialLineWidth) { 
972                        return text;
973                }
974                double lines = quadraticRoot(widthIncrementPerLine, initialLineWidth - widthIncrementPerLine, -text.length()); 
975                return WordUtils.wrap(text, initialLineWidth + (int) (widthIncrementPerLine * Math.ceil(lines)), separator, false);
976        }

978        /**
979         * Inserts separators into a list of elements so when these elements are rendered as text the text is wrapped in such a way that
980         * line width grows proportionally as the number of lines increases. 
981         * @param elements Elements to wrap 
982         * @param initialLineWidth
983         * @param separatorSupplier creates separators to insert into the resulting list.
984         * @return
985         */
986        public static <T> List<T> wrap(
987                        List<T> elements, 
988                        ToIntFunction<T> elementLengthComputer,  
989                        int initialLineWidth, 
990                        int widthIncrementPerLine, 
991                        java.util.function.Supplier<T> wordSeparatorSupplier,                   
992                        java.util.function.Supplier<T> lineSeparatorSupplier) {
993                int length = elements.stream().mapToInt(elementLengthComputer).sum();           
994                int wrapLength = initialLineWidth;
995                if (length > initialLineWidth) { 
996                        wrapLength += (int) (widthIncrementPerLine * Math.ceil(quadraticRoot(widthIncrementPerLine, initialLineWidth - widthIncrementPerLine, -length)));
997                }
998                List<T> ret = new ArrayList<T>();
999                int lineLength = 0;
1000                for (T element: elements) {
1001                        if (lineLength > wrapLength) {
1002                                ret.add(lineSeparatorSupplier.get());
1003                                lineLength = 0;
1004                        } 
1005                        if (lineLength > 0) {
1006                                ret.add(wordSeparatorSupplier.get());
1007                        }
1008                        ret.add(element);
1009                        lineLength += elementLengthComputer.applyAsInt(element);
1010                }
1011                return ret;
1012        }

1014        private static double quadraticRoot(double a, double b, double c) { 
1015                return (-b + Math.sqrt(b*b - 4 * a * c)) / (2 * a);
1016        }

1021         * Java 8 uses package-list, Java 11 uses element-list.
1022         */
1023        private static final String[] PACKAGE_LIST_LOCATIONS = { "package-list", "element-list" };

1025        private static final String MODULE_PREFIX = "module:";

1027        /**
1028         * @param urls Mapping of online URL's (to build links) to offline URL's (to load package lists from).  
1029         * @return Function which given a fully qualified class name returns an anchor linking to Javadoc page for the class. The argument may contain a hash symbol (fragment). 
1030         * Everything after the hash gets transfered to the computed URL providing ability to address fields and methods.
1031         * 
1032         */
1033        public static java.util.function.Function<String, String> createJavadocResolver(Map<URL,URL> urls, ProgressMonitor progressMonitor) {
1034                // package -> Base URL.
1035                Map<String,String> unsortedPackageMap = new HashMap<>();                
1036                progressMonitor.setWorkRemaining(urls.size());
1037                Z: for (Map.Entry<URL, URL> urlEntry: urls.entrySet()) {
1038                        for (String packageListLocation: PACKAGE_LIST_LOCATIONS) {
1039                                try (BufferedReader br = new BufferedReader(new InputStreamReader(new URL(urlEntry.getValue(), packageListLocation).openStream()))) {
1040                                        String module = null;
1041                                        String line;
1042                                        while ((line = br.readLine()) != null) {
1043                                                if (line.startsWith(MODULE_PREFIX)) {
1044                                                        module = line.substring(MODULE_PREFIX.length()); 
1045                                                } else {
1046                                                        String packageLocation = urlEntry.getKey().toString() + (module == null ? "index.html?" : module + "/");
1047                                                        unsortedPackageMap.put(line.trim(), packageLocation);
1048                                                }
1049                                        }
1050                                        progressMonitor.worked(Status.SUCCESS, 1, "Loaded package list from " + urlEntry.getValue() + " for " + urlEntry.getKey());
1051                                        continue Z;
1052                                } catch (Exception e) {
1053                                        progressMonitor.worked(Status.ERROR, 1, "Could not download package list from " + urlEntry.getValue() + " - " + e);
1054                                }
1055                        }
1056                }
1057                
1058                Map<String,String> sortedPackageMap = new LinkedHashMap<>();
1059                unsortedPackageMap.keySet().stream().sorted((k1,k2) -> k2.length() - k1.length()).forEach(k -> sortedPackageMap.put(k, unsortedPackageMap.get(k)));
1060                
1061                return new java.util.function.Function<String, String>() {
1062                        
1063                        private String getRef(String spec) {
1064                                int hashIdx = spec.indexOf("#");
1065                                if (hashIdx != -1) {
1066                                        spec = spec.substring(0, hashIdx);
1067                                }
1068                                for (Entry<String, String> location: sortedPackageMap.entrySet()) {
1069                                        String key = location.getKey();
1070                                        String value = location.getValue();
1071                                        if (key.equals(spec)) { // Package
1072                                                return value + spec.replace('.', '/') + "/" + PACKAGE_SUMMARY_HTML;
1073                                        }
1074                                        
1075                                        int idx = spec.lastIndexOf('.');
1076                                        if (idx != -1 && spec.substring(0, idx).equals(key)) { // Class
1077                                                return value + spec.replace('.', '/').replace('$', '.') + ".html";                              
1078                                        }
1079                                }
1080                                return null;                            
1081                        }
1082
1083                        @Override
1084                        public String apply(String key) {
1085                                if (key == null) {
1086                                        return key;
1087                                }
1088                                
1089                                int spaceIdx = key.indexOf(' ');
1090                                String spec = spaceIdx == -1 ? key : key.substring(0, spaceIdx);
1091                                String text = spaceIdx == -1 ? null : key.substring(spaceIdx + 1);
1092                                
1093                                String href = getRef(spec);
1094                                if (href == null) {
1095                                        return spec;
1096                                }
1097                                                                
1098                                int hashIdx = spec.indexOf("#");
1099                                if (hashIdx != -1) {
1100                                        String fragment = spec.substring(hashIdx+1);
1101                                        int firstParenthesisIdx = fragment.indexOf("(");
1102                                        if (firstParenthesisIdx > 0 && fragment.endsWith(")")) {
1103                                                // Convert (type[,type]) to -type-type-                                         
1104                                                StringBuilder fragmentBuilder = new StringBuilder(fragment.substring(0, firstParenthesisIdx)).append("-");
1105                                                String[] pTypes = fragment.substring(firstParenthesisIdx+1, fragment.length()-1).split(",");
1106                                                for (String pType: pTypes) {
1107                                                        pType = pType.trim();
1108                                                        int bIdx = pType.indexOf("[]");
1109                                                        if (bIdx == -1) {
1110                                                                fragmentBuilder.append(pType);
1111                                                        } else {
1112                                                                fragmentBuilder.append(pType.substring(0, bIdx));
1113                                                                for (int bc = bIdx; bc < pType.length(); bc+=2) {
1114                                                                        fragmentBuilder.append(":A");
1115                                                                }
1116                                                        }
1117                                                        fragmentBuilder.append("-");
1118                                                }
1119                                                fragment = fragmentBuilder.toString();
1120                                        } 
1121                                        href += "#" + fragment;
1122                                }
1123                                return new StringBuilder("<a href='")
1124                                                .append(href)
1125                                                .append("'>")
1126                                                .append(Util.isBlank(text) ? spec.replace('#', '.').replace('$', '.') : text)
1127                                                .append("</a>").toString();
1128                        }
1129                        
1130                };
1131                
1132        }
