001package org.nasdanika.common;
002
003import java.util.function.Function;
004
005/**
006 * A context which support setting properties and registering services.
007 * @author Pavel
008 *
009 */
010public interface MutableContext extends Context {
011                
012        /**
013         * Sets property value.
014         * @param key
015         * @param value
016         */
017        void put(String key, Object value);
018        
019        /**
020         * Registers a service. Multiple registrations are allowed.
021         * @param <T>
022         * @param type
023         * @param service
024         */
025        <T> void register(Class<T> type, T service);
026        
027        /**
028         * Registers a service computer.
029         * @param <T>
030         * @param type
031         * @param serviceComputer
032         */
033        <T> void register(Class<T> type, ServiceComputer<? super T> serviceComputer);
034                
035        /**
036         * Unregisters a service.
037         * @param <T>
038         * @param type
039         * @param service
040         */
041        <T> void unregister(Class<T> type, T service);
042                
043        /**
044         * Registers a service computer.
045         * @param <T>
046         * @param type
047         * @param serviceComputer
048         */
049        <T> void unregister(Class<T> type, ServiceComputer<? super T> serviceComputer); 
050        
051        /**
052         * @param mapper Key mapping function. 
053         * @return A context which maps the key using the mapper function in <code>get(String></code> and <code>get(String,Class)</code> methods. Returns this context if mapper is null.
054         * Service registration and unregistration methods are passed to this context without mapping.
055         */
056        @Override
057        default MutableContext map(Function<String,String> mapper) {
058                if (mapper == null) {
059                        return this;
060                }
061                
062                return new MutableContext() {
063
064                        @Override
065                        public Object get(String key) {
066                                return MutableContext.this.get(mapper.apply(key));
067                        }
068
069                        @Override
070                        public <T> T get(Class<T> type) {
071                                return MutableContext.this.get(type);
072                        }
073
074                        @Override
075                        public <T> T get(String key, Class<T> type) {
076                                return MutableContext.this.get(mapper.apply(key), type);
077                        }
078
079                        @Override
080                        public void put(String key, Object value) {
081                                MutableContext.this.put(mapper.apply(key), value);                              
082                        }
083
084                        @Override
085                        public <T> void register(Class<T> type, T service) {
086                                MutableContext.this.register(type, service);                                                            
087                        }
088
089                        @Override
090                        public <T> void register(Class<T> type, ServiceComputer<? super T> serviceComputer) {
091                                MutableContext.this.register(type, serviceComputer);                                                            
092                        }
093
094                        @Override
095                        public <T> void unregister(Class<T> type, T service) {
096                                MutableContext.this.unregister(type, service);                                                          
097                        }
098
099                        @Override
100                        public <T> void unregister(Class<T> type, ServiceComputer<? super T> serviceComputer) {
101                                MutableContext.this.unregister(type, serviceComputer);                                                          
102                        }
103                        
104                };
105        }
106        
107        
108}
