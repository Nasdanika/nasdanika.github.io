001package org.nasdanika.common;
002
003import java.util.Collection;
004import java.util.IdentityHashMap;
005import java.util.Map;
006
007public class ElementIdentityMapCompoundFunction<T,R> extends ElementIdentityMapCompoundExecutionParticipant<Function<T,R>,R> implements Function<T,java.util.function.Function<Function<T,R>,R>>  {
008
009        public ElementIdentityMapCompoundFunction(String name) {
010                super(name);
011        }
012
013        public ElementIdentityMapCompoundFunction(String name, Collection<Function<T,R>> functions) {
014                super(name);
015                functions.forEach(this::put);
016        }
017
018        @Override
019        public java.util.function.Function<Function<T, R>, R> execute(T arg, ProgressMonitor progressMonitor) {
020                progressMonitor.setWorkRemaining(size());
021                Map<Function<T,R>,R> result = new IdentityHashMap<>();
022                for (Function<T,R> e: elements) {
023                        result.put(e, e.splitAndExecute(arg, progressMonitor));                 
024                }
025                return result::get; 
026        }       
027
028}
