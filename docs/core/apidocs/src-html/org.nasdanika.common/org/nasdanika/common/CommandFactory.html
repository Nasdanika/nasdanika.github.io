001package org.nasdanika.common;
002
003/**
004 * @author Pavel Vlasov
005 *
006 * @param <T>
007 */
008public interface CommandFactory extends ExecutionParticipantFactory<Command> {
009                
010        /**
011         * @param contextSupplierFactory Factory which creates a context to be passed to this factory to create supplier.
012         * @return Command factory which creates a context using the context supplier factory and then uses that context and this factory to create a command.
013         */
014        default CommandFactory contextify(SupplierFactory<Context> contextSupplierFactory) {
015                return new CommandFactory() {
016
017                        @Override
018                        public Command create(Context context) {
019                                return new ContextifiedExecutionParticipant.ContextifiedCommand(contextSupplierFactory.create(context), CommandFactory.this);
020                        }
021                        
022                };
023        }
024        
025        
026}
