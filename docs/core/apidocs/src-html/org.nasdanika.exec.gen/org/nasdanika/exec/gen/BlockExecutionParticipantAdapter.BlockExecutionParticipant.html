001package org.nasdanika.exec.gen;
002
003import java.util.concurrent.CancellationException;
004
005import org.eclipse.emf.common.notify.impl.AdapterImpl;
006import org.nasdanika.common.BasicDiagnostic;
007import org.nasdanika.common.Context;
008import org.nasdanika.common.Diagnostic;
009import org.nasdanika.common.ExecutionParticipant;
010import org.nasdanika.common.ProgressMonitor;
011import org.nasdanika.common.Reference;
012import org.nasdanika.common.Status;
013import org.nasdanika.common.Supplier;
014import org.nasdanika.exec.Block;
015
016/**
017 * Base class for Block adapters.
018 * @author Pavel
019 *
020 */
021public class BlockExecutionParticipantAdapter extends AdapterImpl {
022        
023        public BlockExecutionParticipantAdapter(Block block) {
024                setTarget(block);
025        }
026        
027        public static final String ERROR_KEY = "error";
028        public static final String ERROR_TYPE_KEY = "error/type";
029        public static final String ERROR_MESSAGE_KEY = "error/message"; 
030        
031        /**
032         * Base class for command, consumer, and supplier participants.
033         * @author Pavel
034         *
035         * @param <E>
036         */
037        protected static abstract class BlockExecutionParticipant<E extends ExecutionParticipant> implements ExecutionParticipant {
038                
039                protected E tryParticipant;
040                protected E catchParticipant;
041                protected E finallyParticipant;
042                
043                BlockExecutionParticipant(E tryParticipant, E catchParticipant, E finallyParticipant) {
044                        this.tryParticipant = tryParticipant;
045                        this.catchParticipant = catchParticipant;
046                        this.finallyParticipant = finallyParticipant;
047                }
048
049                @Override
050                public String name() {
051                        return "Block";
052                }       
053                
054                @Override
055                public double size() {
056                        double ret = tryParticipant.size();
057                        if (catchParticipant != null) {
058                                ret += catchParticipant.size();
059                        }
060                        if (finallyParticipant != null) {
061                                ret += finallyParticipant.size();
062                        }
063                        return ret;
064                }
065
066                @Override
067                public Diagnostic diagnose(ProgressMonitor progressMonitor) {           
068                        if (progressMonitor.isCancelled()) {
069                                progressMonitor.worked(1, "Cancelled");
070                                return new BasicDiagnostic(Status.CANCEL, "Progress monitor is cancelled", this);
071                        }
072                        BasicDiagnostic ret = new BasicDiagnostic(Status.SUCCESS, name());
073                        progressMonitor.setWorkRemaining(size());
074                        ret.add(tryParticipant.splitAndDiagnose(progressMonitor));
075                        if (catchParticipant != null) {
076                                ret.add(catchParticipant.splitAndDiagnose(progressMonitor));
077                        }
078                        if (finallyParticipant != null) {
079                                ret.add(finallyParticipant.splitAndDiagnose(progressMonitor));
080                        }
081                        return ret;
082                }
083                
084                @Override
085                public void commit(ProgressMonitor progressMonitor) {
086                        if (progressMonitor.isCancelled()) {
087                                progressMonitor.worked(1, "Cancelled");
088                                throw new CancellationException();
089                        }
090                        progressMonitor.setWorkRemaining(size());
091                        tryParticipant.splitAndCommit(progressMonitor);
092                        if (catchParticipant != null) {
093                                catchParticipant.splitAndCommit(progressMonitor);
094                        }
095                        if (finallyParticipant != null) {
096                                finallyParticipant.splitAndCommit(progressMonitor);
097                        }
098                }
099                
100                @Override
101                public boolean rollback(ProgressMonitor progressMonitor) {
102                        if (progressMonitor.isCancelled()) {
103                                progressMonitor.worked(1, "Cancelled");
104                                throw new CancellationException();
105                        }
106                        progressMonitor.setWorkRemaining(size());
107                        boolean result = true;
108                        if (finallyParticipant != null) {
109                                result = finallyParticipant.splitAndRollback(progressMonitor) && result;
110                        }
111                        if (catchParticipant != null) {
112                                result = catchParticipant.splitAndRollback(progressMonitor) && result;
113                        }
114                        result = tryParticipant.splitAndRollback(progressMonitor) && result;
115                        return result;
116                }
117                
118                @Override
119                public void close() {
120                        tryParticipant.close();
121                        if (catchParticipant != null) {
122                                catchParticipant.close();
123                        }
124                        if (finallyParticipant != null) {
125                                finallyParticipant.close();
126                        }
127                }
128                
129                @Override
130                public String toString() {
131                        return super.toString() + " " + name() + " " + size();
132                }
133
134        }
135                
136        /**
137         * Supplier of context with {@link Exception} "service" indicating that the catch block shall be executed. 
138         * @param context
139         * @param eRef
140         * @return
141         */
142        protected org.nasdanika.common.Supplier<Context> createCatchContextSupplier(Context context, Reference<Exception> eRef) {
143                return new Supplier<Context>() {
144        
145                        @Override
146                        public double size() {
147                                return 1;
148                        }
149        
150                        @Override
151                        public String name() {
152                                return "Injecting error info";
153                        }
154        
155                        @Override
156                        public Context execute(ProgressMonitor progressMonitor) {
157                                return new Context() {
158        
159                                        @Override
160                                        public Object get(String key) {
161                                                Exception exception = eRef.get();
162        
163                                                if (ERROR_KEY.equals(key)) {
164                                                        return exception;
165                                                }
166                                                if (exception != null) {
167                                                        if (ERROR_TYPE_KEY.equals(key)) {
168                                                                return exception.getClass().getName();
169                                                        }
170                                                        if (ERROR_MESSAGE_KEY.equals(key)) {
171                                                                return exception.getMessage();
172                                                        }
173                                                }
174                                                
175                                                return null;
176                                        }
177        
178                                        @SuppressWarnings("unchecked")
179                                        @Override
180                                        public <T> T get(Class<T> type) {
181                                                return type == Exception.class ? (T) eRef.get() : null;
182                                        }
183                                        
184                                }.compose(context);
185                                
186                        }
187                        
188                };
189                
190        }
191
192}
