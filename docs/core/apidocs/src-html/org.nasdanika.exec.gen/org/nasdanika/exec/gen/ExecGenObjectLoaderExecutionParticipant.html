001package org.nasdanika.exec.gen;
002
003import org.eclipse.emf.ecore.EPackage;
004import org.eclipse.emf.ecore.resource.ResourceSet;
005import org.nasdanika.common.Context;
006import org.nasdanika.common.ProgressMonitor;
007import org.nasdanika.exec.util.ExecObjectLoaderExecutionParticipant;
008import org.nasdanika.persistence.ObjectLoaderResourceFactory;
009
010/**
011 * {@link YamlLoadingSupplier} for Exec {@link EPackage}s.
012 * Registers exec- loader. 
013 * @author Pavel
014 *
015 */
016public abstract class ExecGenObjectLoaderExecutionParticipant extends ExecObjectLoaderExecutionParticipant {
017
018        public ExecGenObjectLoaderExecutionParticipant(Context context) {
019                super(context);
020        }
021        
022        @Override
023        protected ObjectLoaderResourceFactory createObjectLoaderResorceFactory(ResourceSet resourceSet, ProgressMonitor progressMonitor) {
024                resourceSet.getAdapterFactories().add(new ExecutionParticpantAdapterFactory());
025                return super.createObjectLoaderResorceFactory(resourceSet, progressMonitor);
026        }
027        
028}
