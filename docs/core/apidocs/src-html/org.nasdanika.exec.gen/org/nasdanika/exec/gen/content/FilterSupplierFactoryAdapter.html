001package org.nasdanika.exec.gen.content;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.util.Objects;
006
007import org.eclipse.emf.common.notify.impl.AdapterImpl;
008import org.eclipse.emf.ecore.EObject;
009import org.nasdanika.common.Context;
010import org.nasdanika.common.ExecutionException;
011import org.nasdanika.common.Function;
012import org.nasdanika.common.FunctionFactory;
013import org.nasdanika.common.ProgressMonitor;
014import org.nasdanika.common.Supplier;
015import org.nasdanika.common.SupplierFactory;
016import org.nasdanika.common.Util;
017import org.nasdanika.emf.EObjectAdaptable;
018import org.nasdanika.exec.content.Filter;
019
020public abstract class FilterSupplierFactoryAdapter<T extends Filter> extends AdapterImpl implements SupplierFactory<InputStream> {
021        
022        protected FilterSupplierFactoryAdapter(T filter) {
023                setTarget(filter);
024        }
025        
026        @SuppressWarnings("unchecked")
027        @Override
028        public T getTarget() {
029                return (T) super.getTarget();
030        }
031        
032        private FunctionFactory<InputStream,InputStream> filterFactory = ctx -> new Function<InputStream, InputStream>() {
033
034                @Override
035                public double size() {
036                        return 1;
037                }
038
039                @Override
040                public String name() {
041                        return FilterSupplierFactoryAdapter.this.getClass().getName();
042                }
043
044                @Override
045                public InputStream execute(InputStream input, ProgressMonitor progressMonitor) {
046                        try {
047                                return Util.filter(ctx, input, str -> FilterSupplierFactoryAdapter.this.filter(ctx, str));
048                        } catch (IOException e) {
049                                throw new ExecutionException(e, this);
050                        }
051                }
052                
053        };
054                
055        @Override
056        public Supplier<InputStream> create(Context context) {
057                EObject source = ((Filter) getTarget()).getSource();
058                SupplierFactory<InputStream> ssf = Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(source, InputStream.class), "Cannot adapt to SupplierFactory: " + source);
059                return ssf.then(filterFactory).create(context);
060        }
061        
062        protected abstract String filter(Context context, String input);        
063
064}
