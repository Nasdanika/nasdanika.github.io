001package org.nasdanika.emf.persistence;
002
003import java.util.Iterator;
004
005import org.apache.poi.ss.usermodel.Cell;
006import org.apache.poi.ss.usermodel.CellValue;
007import org.apache.poi.ss.usermodel.FormulaEvaluator;
008import org.apache.poi.ss.usermodel.Row;
009import org.apache.poi.ss.usermodel.Sheet;
010import org.apache.poi.xssf.usermodel.XSSFFormulaEvaluator;
011import org.apache.poi.xssf.usermodel.XSSFWorkbook;
012import org.eclipse.emf.common.notify.Adapter;
013import org.eclipse.emf.common.util.URI;
014import org.eclipse.emf.ecore.resource.Resource;
015import org.eclipse.emf.ecore.resource.impl.ResourceFactoryImpl;
016
017/**
018 * Creates {@link ExcelResource}
019 * @author Pavel
020 *
021 */
022public class ExcelResourceFactory extends ResourceFactoryImpl {
023        
024        public interface UpdateAdapter extends Adapter {
025                
026                /**
027                 * Updates the workbook.
028                 * @param workbook
029                 */
030                void update(XSSFWorkbook workbook);
031                
032        }
033        
034        @Override
035        public Resource createResource(URI uri) {
036                return new ExcelResource(uri) {
037                        
038                        @Override
039                        void loadWorkbook(XSSFWorkbook workbook, XSSFFormulaEvaluator createFormulaEvaluator) {
040                                ExcelResourceFactory.this.loadWorkbook(this, workbook, createFormulaEvaluator);                         
041                        }
042                        
043                };
044        }
045        
046        protected void loadWorkbook(Resource resource, XSSFWorkbook workbook, FormulaEvaluator formulaEvaluator) {
047                Iterator<Sheet> sheetIterator = workbook.sheetIterator();
048                while (sheetIterator.hasNext()) {
049                        loadSheet(resource, sheetIterator.next(), formulaEvaluator);
050                }
051        };
052        
053        protected void loadSheet(Resource resource, Sheet sheet, FormulaEvaluator formulaEvaluator) {
054                Iterator<Row> rowIterator = sheet.rowIterator();
055                while (rowIterator.hasNext()) {
056                        loadRow(resource, rowIterator.next(), formulaEvaluator);
057                }
058        };
059        
060        protected void loadRow(Resource resource, Row row, FormulaEvaluator formulaEvaluator) {
061                // NOP          
062        };
063        
064        protected Object getCellValue(Cell cell, FormulaEvaluator formulaEvaluator) {
065                if (cell == null) {
066                        return null;
067                }
068                switch (cell.getCellType()) {
069                case NUMERIC:
070                        return cell.getNumericCellValue();
071                case STRING:
072                        return cell.getStringCellValue();
073                case BOOLEAN:
074                        return cell.getBooleanCellValue();
075                case BLANK:
076                        return null;
077                case ERROR:
078                        return cell.getErrorCellValue();
079                case FORMULA:
080                        CellValue cellValue = formulaEvaluator.evaluate(cell);
081                        if (cellValue == null) {
082                                return null;
083                        }
084                        switch (cellValue.getCellType()) {
085                        case NUMERIC:
086                                return cellValue.getNumberValue();
087                        case STRING:
088                                return cellValue.getStringValue();
089                        case BOOLEAN:
090                                return cellValue.getBooleanValue();
091                        case BLANK:
092                                return null;
093                        case ERROR:
094                                return cellValue.getErrorValue();
095                        default:
096                                throw new UnsupportedOperationException("Unsupported formula result type: " + cellValue.getCellType());                         
097                        }
098                default:
099                        throw new UnsupportedOperationException("Unsupported cell type: " + cell.getCellType());
100                }
101        }       
102                
103}
