001package org.nasdanika.emf.localization;
002
003import org.eclipse.emf.ecore.EModelElement;
004import org.nasdanika.emf.EModelElementAnnotationResourceLocator;
005
006/**
007 * Resource locator which adds ``_ru`` suffix and falls back to UI.RU localization. 
008 * @author Pavel
009 *
010 */
011public class RussianResourceLocator extends EModelElementAnnotationResourceLocator {
012        
013        public RussianResourceLocator(EModelElement modelElement) {
014                super(modelElement, "urn:org.nasdanika", key -> key+"_ru", UI.RU);
015        }
016        
017}
