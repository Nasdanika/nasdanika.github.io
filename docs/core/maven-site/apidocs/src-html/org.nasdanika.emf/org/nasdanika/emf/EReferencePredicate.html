001package org.nasdanika.emf;
002
003import org.eclipse.emf.ecore.EObject;
004import org.eclipse.emf.ecore.EReference;
005
006/**
007 * {@link EObject}s can implement this interface to filter out 
008 * objects and their {@link EReference}s to which it can be added. 
009 * @author Pavel
010 *
011 */
012public interface EReferencePredicate {
013        
014        /**
015         * @param source EObject to test for eligibility to add target object to its eReference.
016         * @param eReference EReference to which the target object is tested for addition.
017         * @param target Target object being tested for addition to the source object {@link EReference}.
018         * @param direct If true, an object is considered for direct addition and may be a subject for capacity limitations, e.g. single reference. 
019         * If false, the target is considered for type compatibility by filters. E.g. if reference A can contain only instances of B "semantically" but can also contain a predicate P for conditional
020         * inclusion of B, then P may call accept() with direct set to false to select semantic candidates which are compatible with the reference A.
021         * @return
022         */
023        boolean accept(EObject source, EReference eReference, EObject target, boolean direct);
024
025}
