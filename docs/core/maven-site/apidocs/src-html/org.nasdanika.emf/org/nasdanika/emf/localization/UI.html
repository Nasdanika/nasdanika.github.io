001package org.nasdanika.emf.localization;
002
003import org.nasdanika.common.ResourceLocator;
004
005/**
006 * Holds built-in localizations
007 * @author Pavel
008 *
009 */
010public enum UI implements ResourceLocator, PropertyKeys {
011        
012        RU() {
013
014                @Override
015                public Object get(String key) {
016                        switch (key) {
017                        case UI_DESCRIPTION:
018                                return "Описание";
019                        case UI_SUMMARY:
020                                return "Краткое описание";
021                        case UI_NAME:
022                                return "Наименование";
023                        case UI_TYPE:
024                                return "Тип";
025                        case UI_CARDINALITY:
026                                return "Мощность";
027                        case UI_PACKAGE:
028                                return "Пакет";
029                        case UI_SUPERTYPES:
030                                return "Супертипы";
031                        case UI_SUBTYPES:
032                                return "Подтипы";                                               
033                        case UI_DIAGRAM:
034                                return "Диаграмма";                                             
035                        case UI_CONTENTS:
036                                return "Содержание";                                            
037                        case UI_NAMESPACE_URI:
038                                return "URI пространства имён";                                         
039                        case UI_OPPOSITE:
040                                return "Противоположная сссылка";                                               
041                        case UI_ABSTRACT:
042                                return "Абстрактный";                                           
043                        case UI_INTERFACE:
044                                return "Интерфейс";                                             
045                        default:
046                                return null;
047                        }
048                }
049                
050        };
051        
052        @Override
053        public <T> T get(Class<T> type) {
054                return null;
055        }
056
057}
