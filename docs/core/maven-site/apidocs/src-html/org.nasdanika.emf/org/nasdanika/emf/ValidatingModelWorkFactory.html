001package org.nasdanika.emf;
002
003import java.util.Map;
004
005import org.eclipse.emf.common.util.Diagnostic;
006import org.eclipse.emf.common.util.DiagnosticException;
007import org.eclipse.emf.common.util.URI;
008import org.eclipse.emf.ecore.EPackage;
009import org.eclipse.emf.ecore.resource.ResourceSet;
010import org.eclipse.emf.ecore.util.Diagnostician;
011import org.nasdanika.common.Context;
012import org.nasdanika.common.ExecutionException;
013import org.nasdanika.common.Supplier;
014
015/**
016 * This class loads a model, validates it and then returns work for executing generation. 
017 * @author Pavel
018 *
019 */
020public class ValidatingModelWorkFactory<T> extends ModelWorkFactory<T> {
021
022        public ValidatingModelWorkFactory(ResourceSet resourceSet, URI modelUri) {
023                super(resourceSet, modelUri);
024        }
025
026        public ValidatingModelWorkFactory(String platformPluginUri, EPackage... ePackages) {
027                super(platformPluginUri, ePackages);
028        }
029
030        public ValidatingModelWorkFactory(URI modelUri, EPackage... ePackages) {
031                super(modelUri, ePackages);
032        }
033        
034        public ValidatingModelWorkFactory(ResourceSet resourceSet, String platformPluginUri) {
035                super(resourceSet, platformPluginUri);
036        }
037        
038        @Override
039        public Supplier<T> create(Context context) {
040                Diagnostician diagnostician = new Diagnostician() {
041                        
042                        public Map<Object,Object> createDefaultContext() {
043                                Map<Object, Object> ctx = super.createDefaultContext();
044                                ctx.put(Context.class, context);
045                                return ctx;
046                        };
047                        
048                };                              
049                Diagnostic validationResult = diagnostician.validate(root);
050                if (validationResult.getSeverity() == Diagnostic.ERROR) {
051                        throw new ExecutionException(new DiagnosticException(validationResult));
052                }
053                return super.create(context);
054        }
055
056}
