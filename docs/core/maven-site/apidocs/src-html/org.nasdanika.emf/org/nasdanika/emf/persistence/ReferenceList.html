001package org.nasdanika.emf.persistence;
002
003import java.util.ArrayList;
004import java.util.List;
005import java.util.Map;
006import java.util.Map.Entry;
007import java.util.function.BiFunction;
008
009import org.eclipse.emf.common.util.URI;
010import org.eclipse.emf.ecore.EAttribute;
011import org.eclipse.emf.ecore.EClass;
012import org.eclipse.emf.ecore.ENamedElement;
013import org.eclipse.emf.ecore.EReference;
014import org.eclipse.emf.ecore.EStructuralFeature;
015import org.nasdanika.common.ProgressMonitor;
016import org.nasdanika.persistence.ConfigurationException;
017import org.nasdanika.persistence.ListAttribute;
018import org.nasdanika.persistence.MarkedArrayList;
019import org.nasdanika.persistence.MarkedLinkedHashMap;
020import org.nasdanika.persistence.Marker;
021import org.nasdanika.persistence.ObjectLoader;
022
023/**
024 * Loads a list of references using loader.
025 * @author Pavel
026 *
027 */
028public class ReferenceList<T> extends ListAttribute<T> {
029
030        private ReferenceFactory referenceFactory;
031        private List<String> keys = new ArrayList<>();
032        private BiFunction<EClass, ENamedElement, String> keyProvider;
033        
034        /**
035         * 
036         * @param key
037         * @param isDefault
038         * @param required
039         * @param defaultValue
040         * @param description
041         * @param resolver
042         * @param referenceType Reference type if the reference is homogenous, i.e. its type is known beforehand.
043         * @param exclusiveWith
044         */
045        public ReferenceList(
046                        Object key, 
047                        boolean isDefault, 
048                        boolean required, 
049                        List<T> defaultValue, 
050                        String description, 
051                        EClass eClass,
052                        EReference eReference,
053                        EObjectLoader resolver,
054                        boolean referenceSupplierFactory,
055                        BiFunction<EClass,ENamedElement,String> keyProvider,
056                        Object... exclusiveWith) {
057                super(key, isDefault, required, defaultValue, description, exclusiveWith);
058                this.referenceFactory = new ReferenceFactory(eClass, eReference, null, resolver, referenceSupplierFactory, keyProvider);
059                this.keyProvider = keyProvider;
060                for (EAttribute rKey: eReference.getEKeys()) {
061                        keys.add(keyProvider.apply(eClass, rKey));
062                }
063        }
064        
065        @Override
066        public List<T> create(ObjectLoader loader, Object config, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) {
067                if (config instanceof Map) {
068                        if (!keys.isEmpty()) {
069                                String valueFeature = EObjectLoader.getValueFeature(referenceFactory.getEReference());
070                                if (keys.size() == 1) {
071                                        // Converting entry set to a list of maps
072                                        String keyName = keys.get(0);
073                                        MarkedArrayList<Map<?,?>> entryList = new MarkedArrayList<>();
074                                        entryList.mark(markers);
075                                        for (Entry<?, ?> entry: ((Map<?,?>) config).entrySet()) {
076                                                Object value = entry.getValue();
077                                                List<? extends Marker> valueMarkers = null;
078                                                if (config instanceof MarkedLinkedHashMap) {
079                                                        valueMarkers = ((MarkedLinkedHashMap<?, ?>) config).getEntryMarkers(entry.getKey());
080                                                }
081                                                
082                                                if (referenceFactory.isHomogenous()) {
083                                                        // Config shall be a map with potentially multiple entries, type is already known
084                                                        MarkedLinkedHashMap<Object, Object> entryConfig = new MarkedLinkedHashMap<>();
085                                                        entryConfig.mark(valueMarkers);
086                                                        entryConfig.put(keyName, entry.getKey());
087                                                        entryConfig.markEntry(keyName, valueMarkers);
088                                                        if (valueFeature == null) {
089                                                                if (value instanceof Map) {
090                                                                        entryConfig.putAll((Map<?,?>) value);
091                                                                } else {
092                                                                        EStructuralFeature defaultFeature = referenceFactory.effectiveDefaultFeature(value);
093                                                                        if (defaultFeature == null) {
094                                                                                throw new ConfigurationException("Configuration shall be a map: " + value, valueMarkers);
095                                                                        }
096                                                                        entryConfig.put(keyProvider.apply(null, defaultFeature), value);
097                                                                }
098                                                        } else {
099                                                                entryConfig.put(valueFeature, value);
100                                                        }
101                                                        entryList.add(entryConfig);
102                                                } else {
103                                                        // Config (value) must be a singleton map with type as a key. We need to inject things into the value.
104                                                        if (value instanceof Map && ((Map<?,?>) value).size() == 1) {
105                                                                for (Entry<?, ?> valueEntry: ((Map<?,?>) value).entrySet()) {
106                                                                        MarkedLinkedHashMap<Object, Object> entryConfig = new MarkedLinkedHashMap<>();
107                                                                        entryConfig.mark(valueMarkers);
108                                                                        entryConfig.put(keyName, entry.getKey());
109                                                                        entryConfig.markEntry(keyName, valueMarkers);
110                                                                        if (valueFeature == null) {
111                                                                                Object valueEntryValue = valueEntry.getValue();
112                                                                                if (valueEntryValue instanceof Map) {
113                                                                                        entryConfig.putAll((Map<?,?>) valueEntryValue);
114                                                                                } else {
115                                                                                        // TODO - Default feature                                                                                       
116                                                                                        throw new ConfigurationException("Configuration shall be a map: " + valueEntry.getValue(), valueMarkers); // TODO - marker 
117                                                                                }
118                                                                        } else {
119                                                                                entryConfig.put(valueFeature, valueEntry.getValue());
120                                                                        }
121                                                                        
122                                                                        MarkedLinkedHashMap<Object, Object> singleton = new MarkedLinkedHashMap<>();
123                                                                        singleton.put(valueEntry.getKey(), entryConfig);
124                                                                        singleton.mark(valueMarkers);
125                                                                        singleton.markEntry(valueEntry.getKey(), valueMarkers);
126                                                                        entryList.add(singleton); 
127                                                                }
128                                                        } else {
129                                                                throw new ConfigurationException("Configuration shall be a singleton map: " + value, valueMarkers);                                                             
130                                                        }
131                                                }
132                                                entryList.getElementMarkers().add(valueMarkers);
133                                        }
134                                        return super.create(loader, entryList, base, progressMonitor, markers);                 
135                                } 
136                                
137                                throw new UnsupportedOperationException("Multiple e-keys are not supported yet");
138                        }
139                }
140                return super.create(loader, config, base, progressMonitor, markers);
141        }
142        
143        @SuppressWarnings("unchecked")
144        @Override
145        protected List<T> createElements(ObjectLoader loader, Object element, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) {
146                return (List<T>) referenceFactory.create(loader, element, base, progressMonitor, markers);
147        }
148
149}
