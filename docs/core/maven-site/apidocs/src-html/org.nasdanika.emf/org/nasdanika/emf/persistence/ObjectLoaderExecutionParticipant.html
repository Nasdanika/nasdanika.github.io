001package org.nasdanika.emf.persistence;
002
003import java.util.Map;
004
005import org.eclipse.emf.ecore.EPackage;
006import org.eclipse.emf.ecore.resource.Resource;
007import org.eclipse.emf.ecore.resource.ResourceSet;
008import org.nasdanika.common.Context;
009import org.nasdanika.common.ProgressMonitor;
010import org.nasdanika.persistence.ObjectLoader;
011import org.nasdanika.persistence.ObjectLoaderResourceFactory;
012
013/**
014 * Registers {@link YamlResourceFactory} with the resource set.
015 * Also registers {@link EPackage}'s
016 * @author Pavel
017 *
018 */
019public abstract class ObjectLoaderExecutionParticipant extends LoadingExecutionParticipant {
020
021        public ObjectLoaderExecutionParticipant(Context context) {
022                super(context);
023        }
024
025        @Override
026        protected ResourceSet createResourceSet(ProgressMonitor progressMonitor) {
027                ResourceSet ret = super.createResourceSet(progressMonitor);
028                
029                ObjectLoaderResourceFactory objectLoaderResourceFactory = createObjectLoaderResorceFactory(ret, progressMonitor);
030                
031                Map<String, Object> extensionToFactoryMap = ret.getResourceFactoryRegistry().getExtensionToFactoryMap();
032                extensionToFactoryMap.put("yml", objectLoaderResourceFactory);
033                extensionToFactoryMap.put("json", objectLoaderResourceFactory);
034                
035                ret.getResourceFactoryRegistry().getProtocolToFactoryMap().put("data", objectLoaderResourceFactory);
036                
037                return ret;
038        }
039
040        protected ObjectLoaderResourceFactory createObjectLoaderResorceFactory(ResourceSet resourceSet, ProgressMonitor progressMonitor) {              
041                return new ObjectLoaderResourceFactory() {
042                        
043                        @Override
044                        protected ObjectLoader getObjectLoader(Resource resource) {
045                                return getObjectLLoader(resourceSet, resource, progressMonitor);
046                        }
047                        
048                        @Override
049                        protected Context getContext(Resource resource) {
050                                return context;
051                        }
052                        
053                        @Override
054                        protected ProgressMonitor getProgressMonitor(Resource resource) {
055                                return progressMonitor;
056                        }
057                        
058                };
059        }
060
061        protected ObjectLoader getObjectLLoader(ResourceSet resourceSet, Resource resource, ProgressMonitor progressMonitor) {
062                EObjectLoader eObjectLoader = new EObjectLoader(null, null, resourceSet);
063                eObjectLoader.setMarkerFactory(new GitMarkerFactory());
064                return eObjectLoader;
065        }
066
067}
