039/**
040 * This factory uses {@link DrawioResource} with {@link DrawioEObjectFactory}. 
041 * It injects name, description, and markers in configureSemanticElement(). 
042 * @author Pavel
043 *
044 * @param <T>
045 */
046public abstract class NcoreDrawioResourceFactory<T extends EObject> extends ResourceFactoryImpl {
047        
048        public static final String SEMANTIC_UUID_KEY = "semantic-uuid"; 
049        
050        private class NcoreDrawioResource extends DrawioResource<SemanticProcessor<T>, T> implements AbstractEObjectFactoryProcessorResource<T> {
051
052                private DrawioEObjectFactory<T, SemanticProcessor<T>> processorFactory;
053                private ProgressMonitor progressMonitor;
054
055                protected NcoreDrawioResource(URI uri, DrawioEObjectFactory<T, SemanticProcessor<T>> processorFactory, ProgressMonitor progressMonitor) {
056                        super(uri);
057                        this.processorFactory = processorFactory;
058                        this.progressMonitor = progressMonitor;
059                }
060
061                @SuppressWarnings("unchecked")
062                @Override
063                public void setParent(T parent) {
064                        for (EObject root: new ArrayList<>(getContents())) {
065                                processorFactory.setParent((T) root, parent, getProgressMonitor());
066                        }
067                }
068
069                @Override
070                protected DrawioEObjectFactory<T, SemanticProcessor<T>> getProcessorFactory() {
071                        return processorFactory;
072                }
073                
074                @Override
075                protected ProgressMonitor getProgressMonitor() {
076                        return progressMonitor;
077                };
078                
079        }
080        
081        @Override
082        public Resource createResource(URI uri) {
083                return new NcoreDrawioResource(uri, createProcessorFactory(uri), getProgressMonitor(uri));
084        }
085        
086        /**
087         * Allows to override the default behavior of using spec, spec-uri, and semantic-uri properties for creation of semantic elements.
088         * @param config
089         * @param progressMonitor
090         * @param fallback
091         * @return
092         */
093        protected Collection<T> createSemanticElements(
094                        ProcessorConfig<SemanticProcessor<T>> config, 
095                        ProgressMonitor progressMonitor, 
096                        BiFunction<ProcessorConfig<SemanticProcessor<T>>, ProgressMonitor, Collection<T>> defaultFactory) {
097                return defaultFactory.apply(config, progressMonitor);
098        }
099                
100        protected DrawioEObjectFactory<T, SemanticProcessor<T>> createProcessorFactory(URI uri) {
101                return new DrawioEObjectFactory<T, SemanticProcessor<T>>() {
102
103                        @Override
104                        protected ResourceSet getResourceSet() {
105                                return NcoreDrawioResourceFactory.this.getResourceSet();
106                        }
107
108                        @Override
109                        protected SemanticProcessor<T> createProcessor(ProcessorConfig<SemanticProcessor<T>> config, Collection<T> semanticElements, ProgressMonitor progressMonitor) {
110                                return NcoreDrawioResourceFactory.this.createProcessor(uri, config, semanticElements, progressMonitor);
111                        }
112
113                        @Override
114                        protected URI getBaseURI() {
115                                return uri;
116                        }
117                        
118                        @Override
119                        protected Collection<T> createSemanticElements(ProcessorConfig<SemanticProcessor<T>> config, ProgressMonitor progressMonitor) {
120                                Collection<T> semanticElements = NcoreDrawioResourceFactory.this.createSemanticElements(config, progressMonitor, super::createSemanticElements);
121                                Collection<T> configuredSemanticElements = NcoreDrawioResourceFactory.this.configureSemanticElements(uri, config, semanticElements, progressMonitor);
122                                return configuredSemanticElements;
123                        }
124                        
125                        /**
126                         * Checking for failures, injecting representations after all processors have been created. 
127                         */
128                        @Override
129                        public Map<Element, ProcessorInfo<SemanticProcessor<T>>> createProcessors(Stream<? extends Element> elements, ProgressMonitor progressMonitor) {
130                                Map<Element, ProcessorInfo<SemanticProcessor<T>>> registry = super.createProcessors(elements, progressMonitor);
131                                
132                                List<Throwable> failures = registry.values().stream().flatMap(pi -> pi.getFailures().stream()).collect(Collectors.toList());
133                                if (!failures.isEmpty()) {
134                                        NasdanikaException ne = new NasdanikaException("Theres's been " + failures.size() +  " failures during processor creation: " + failures);
135                                        for (Throwable failure: failures) {
136                                                ne.addSuppressed(failure);
137                                        }
138                                        throw ne;
139                                }                               
140                                
141                                String representationPropertyName = getRepresentationPropertyName(uri);
142                                                                
143                                registry.entrySet().forEach(re -> configureRegistryEntry(re.getKey(), re.getValue(), getPropertyValue(re.getKey(), representationPropertyName)));
144                                return registry;
145                        }
146                        
147                };
148        }
149
150        /**
151         * Injects encoded document into semantic elements of model elements with representation property.
152         * Sorts references.
153         * @param element
154         * @param info
155         */
156        protected void configureRegistryEntry(Element element, ProcessorInfo<SemanticProcessor<T>> info, String representationKey) {
157                if (info != null && !Util.isBlank(representationKey)) {
158                        SemanticProcessor<T> processor = info.getProcessor();
159                        if (processor != null && element instanceof ModelElement) {
160                                Collection<T> semanticElements = processor.getSemanticElements();
161                                if (semanticElements != null && !semanticElements.isEmpty()) {
162                                        for (T semanticElement: semanticElements) {
163                                                if (semanticElement instanceof org.nasdanika.ncore.ModelElement) {
164                                                        Document document = ((ModelElement) element).getModel().getPage().getDocument();
165                                                        try {
166                                                                ((org.nasdanika.ncore.ModelElement) semanticElement).getRepresentations().put(representationKey, document.toDataURI(true).toString());
167                                                        } catch (TransformerException | IOException e) {
168                                                                throw new NasdanikaException("Error encoding document: " + e, e);
169                                                        }
170                                                }
171                                        }
172                                }
173                        }
174                }
175                
176                // TODO - sort - YAML map of references to comparators. Comparator either String - comparator name or a map - name to configuration. 
177                // Use ECollections to sort. Expression (SPEL) comparator.
178
179        }

185        protected SemanticProcessor<T> createProcessor(URI uri, ProcessorConfig<SemanticProcessor<T>> config, Collection<T> semanticElements, ProgressMonitor progressMonitor) {                
186                if (semanticElements == null || semanticElements.isEmpty()) {
187                        return null; // For pass-through to parent.
188                }
189                
190                // To avoid concurrent modification exception for situations when processor is backed by a resource and semantic elements get reparented and move out of the resource. 
191                Collection<T> semanticElementsSnapshot = Collections.unmodifiableCollection(new ArrayList<>(semanticElements));
192                
193                return new SemanticProcessor<T>() {
194
195                        @Override
196                        public Collection<T> getSemanticElements() {
197                                return semanticElementsSnapshot;
198                        }
199                        
200                };
201        }       
202
203        protected MarkerFactory getMarkerFactory() {
204                return MarkerFactory.INSTANCE;
205        }
206                
207        protected Collection<T> configureSemanticElements(
208                        URI uri,
209                        ProcessorConfig<SemanticProcessor<T>> config, 
210                        Collection<T> semanticElements, 
211                        ProgressMonitor progressMonitor) {
212
213                if (config == null) {
214                        return semanticElements;
215                }
216                Element element = config.getElement();
217                if (semanticElements == null) {
218                        return semanticElements;
219                }
220                for (T semanticElement: semanticElements) {
221                        if (uri != null && semanticElement instanceof Marked && element instanceof Marked) {
222                                for (org.nasdanika.persistence.Marker elementMarker: ((Marked) element).getMarkers()) {
223                                        org.nasdanika.ncore.Marker marker = getMarkerFactory().createMarker(elementMarker.getLocation(), progressMonitor);
224                                        marker.setPosition(elementMarker.getPosition());
225                                        ((Marked) semanticElement).getMarkers().add(marker);
226                                }
227                        }
228                        
229                        if (element instanceof ModelElement) {
230                                ModelElement modelElement = (ModelElement) element;
231                                String tooltip = modelElement.getTooltip();
232                                if (!Util.isBlank(tooltip)) {
233                                        if (semanticElement instanceof org.nasdanika.ncore.ModelElement && !semanticElement.eIsSet(NcorePackage.Literals.MODEL_ELEMENT__DESCRIPTION)) {
234                                                String firstTooltipSentence = org.nasdanika.common.Util.firstPlainTextSentence(tooltip, 50, 250);
235                                                ((org.nasdanika.ncore.ModelElement) semanticElement).setDescription(firstTooltipSentence);                                              
236                                        }
237                                        if (semanticElement instanceof Documented && ((Documented) semanticElement).getDocumentation().isEmpty()) {
238                                                ((Documented) semanticElement).getDocumentation().add(NcoreUtil.wrapString("<pre style='white-space:pre-wrap'>" + System.lineSeparator() + tooltip + System.lineSeparator() + "</pre>"));
239                                        }
240                                }
241                                
242                                String label = modelElement.getLabel();
243                                if (!Util.isBlank(label) && semanticElement instanceof NamedElement && !semanticElement.eIsSet(NcorePackage.Literals.NAMED_ELEMENT__NAME)) {
244                                        ((NamedElement) semanticElement).setName(Jsoup.parse(label).text());                            
245                                }
246                                
247                                String id = modelElement.getId();
248                                if (!Util.isBlank(id) && semanticElement instanceof org.nasdanika.ncore.Composite && !semanticElement.eIsSet(NcorePackage.Literals.COMPOSITE__ID)) {
249                                        ((org.nasdanika.ncore.Composite) semanticElement).setId(id);                            
250                                }
251                                
252                                if (semanticElement instanceof org.nasdanika.ncore.ModelElement) {
253                                        String semanticUuid = ((org.nasdanika.ncore.ModelElement) semanticElement).getUuid();
254                                        if (!Util.isBlank(semanticUuid)) {
255                                                modelElement.setProperty(SEMANTIC_UUID_KEY, semanticUuid);
256                                        }
257                                }
258                        }
259                        
260                }
261                
262                return semanticElements;                
263        }
264        
265        /**
266         * If this property is set and the semantic element is {@link org.nasdanika.ncore.ModelElement} then {@link Document} is encoded and stored into the representation with the specified name.
267         * @param uri Resource {@link URI}
268         * @return
269         */
270        protected String getRepresentationPropertyName(URI uri) {
271                return "representation";
272        }
