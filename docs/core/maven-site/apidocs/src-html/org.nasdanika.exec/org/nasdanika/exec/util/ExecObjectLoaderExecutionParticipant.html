001package org.nasdanika.exec.util;
002
003import java.util.Collection;
004
005import org.eclipse.emf.ecore.EPackage;
006import org.nasdanika.common.Context;
007import org.nasdanika.emf.persistence.NcoreObjectLoaderExecutionParticipant;
008import org.nasdanika.exec.ExecPackage;
009import org.nasdanika.exec.content.ContentPackage;
010import org.nasdanika.exec.resources.ResourcesPackage;
011
012/**
013 * Registers Exec {@link EPackage}s. 
014 * @author Pavel
015 *
016 */
017public abstract class ExecObjectLoaderExecutionParticipant extends NcoreObjectLoaderExecutionParticipant {
018
019        public ExecObjectLoaderExecutionParticipant(Context context) {
020                super(context);
021        }
022
023        @Override
024        protected Collection<EPackage> getEPackages() {
025                Collection<EPackage> ret = super.getEPackages(); 
026                ret.add(ExecPackage.eINSTANCE);
027                ret.add(ContentPackage.eINSTANCE);
028                ret.add(ResourcesPackage.eINSTANCE);
029                return ret;
030        }
031        
032}
