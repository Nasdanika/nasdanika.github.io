001package org.nasdanika.exec.gen.resources;
002
003import java.util.Objects;
004import java.util.concurrent.CancellationException;
005
006import org.eclipse.emf.ecore.EObject;
007import org.nasdanika.common.CollectionCompoundConsumerFactory;
008import org.nasdanika.common.Consumer;
009import org.nasdanika.common.ConsumerFactory;
010import org.nasdanika.common.Context;
011import org.nasdanika.common.FilterConsumer;
012import org.nasdanika.common.Function;
013import org.nasdanika.common.FunctionFactory;
014import org.nasdanika.common.ProgressMonitor;
015import org.nasdanika.emf.EObjectAdaptable;
016import org.nasdanika.exec.resources.Container;
017import org.nasdanika.resources.BinaryEntityContainer;
018
019public class ContainerConsumerFactoryAdapter extends ResourceConsumerFactoryAdapter<Container> {
020
021        public ContainerConsumerFactoryAdapter(Container container) {
022                super(container);
023        }
024        
025        @Override
026        public boolean isAdapterForType(Object type) {
027                return type == ConsumerFactory.class;
028        }       
029
030        @Override
031        public Consumer<BinaryEntityContainer> create(Context context) {
032                return createContainerFactory().then(createContentsFactory()).create(context);
033        }
034
035        protected FunctionFactory<BinaryEntityContainer, BinaryEntityContainer> createContainerFactory() {
036                Container semanticElement = (Container) getTarget();
037                
038                return context -> new Function<BinaryEntityContainer, BinaryEntityContainer>() {
039                        
040                        private String finalName = finalName(context.interpolateToString(semanticElement.getName()));
041
042                        @Override
043                        public double size() {
044                                return 1;
045                        }
046
047                        @Override
048                        public String name() {
049                                return "Create container " + finalName;
050                        }
051
052                        @Override
053                        public BinaryEntityContainer execute(BinaryEntityContainer container, ProgressMonitor progressMonitor) {
054                                if (progressMonitor.isCancelled()) {
055                                        return null;
056                                }
057                                
058                                BinaryEntityContainer ret = Objects.requireNonNull(container.getContainer(finalName, progressMonitor), "Cannot create container " + finalName + " in " + container);
059                                switch (semanticElement.getReconcileAction()) {
060                                case OVERWRITE:
061                                        if (ret.exists(progressMonitor)) {
062                                                ret.delete(progressMonitor);
063                                        }
064                                case MERGE:
065                                case APPEND:
066                                        return ret;
067                                case CANCEL:
068                                        if (ret.exists(progressMonitor)) {
069                                                throw new CancellationException("Cancelling generation - container '" + finalName + "' already exists in " + container);
070                                        }
071                                        return ret;
072                                case KEEP:
073                                        if (ret.exists(progressMonitor)) {
074                                                return null; // Indicates that elements factory shall not do anything.
075                                        }
076                                        return ret;
077                                }
078                                return ret;
079                        }
080                        
081                };
082        }
083
084        protected ConsumerFactory<BinaryEntityContainer> createContentsFactory() {
085                Container container = (Container) getTarget();
086                
087                CollectionCompoundConsumerFactory<BinaryEntityContainer> contentsFactory = new CollectionCompoundConsumerFactory<>("Contents");
088                for (EObject ce: container.getContents()) {
089                        contentsFactory.add(Objects.requireNonNull(EObjectAdaptable.adaptToConsumerFactory(ce, BinaryEntityContainer.class), "Cannot adapt to ConsumerFactory"));
090                }
091                
092                return new ConsumerFactory<BinaryEntityContainer>() {           
093                
094                        @Override
095                        public Consumer<BinaryEntityContainer> create(Context context) {
096                                Consumer<BinaryEntityContainer> contentsConsumer = contentsFactory.create(context);
097                                
098                                return new FilterConsumer<BinaryEntityContainer>(contentsConsumer) {
099                                        
100                                        @Override
101                                        public void execute(BinaryEntityContainer container, ProgressMonitor progressMonitor) {
102                                                if (container != null) {
103                                                        contentsConsumer.execute(container, progressMonitor);
104                                                }                                               
105                                        }
106                                        
107                                };
108                        }
109                        
110                };
111        }
112
113}
