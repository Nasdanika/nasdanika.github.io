040public class CallSupplierFactoryAdapter extends AdapterImpl implements SupplierFactory<InputStream> {
041        
042        public CallSupplierFactoryAdapter(Call call) {
043                setTarget(call);
044        }
045        
046        @Override
047        public boolean isAdapterForType(Object type) {
048                return type == SupplierFactory.class;
049        }
050        
051        @Override
052        public Supplier<InputStream> create(Context context) {
053                MapCompoundSupplierFactory<EStructuralFeature, Object> featureMapFactory = new MapCompoundSupplierFactory<>("Feature map factory");
054                Call call = (Call) getTarget();
055                EList<EObject> init = call.getInit();
056                if (!init.isEmpty()) {
057                        ListCompoundSupplierFactory<Object> initFactory = new ListCompoundSupplierFactory<>("Init"); 
058                        for (EObject initArg: init) {
059                                initFactory.add(Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(initArg, Object.class), "Cannot adapt to SupplierFactory: " + initArg));
060                        }
061                        featureMapFactory.put(ExecPackage.Literals.CALL__INIT, initFactory);
062                }
063                
064                EList<EObject> args = call.getArguments();
065                if (!args.isEmpty()) {
066                        ListCompoundSupplierFactory<Object> argFactory = new ListCompoundSupplierFactory<>("Arguments"); 
067                        for (EObject arg: args) {
068                                argFactory.add(Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(arg, Object.class), "Cannot adapt to SupplierFactory: " + arg));
069                        }
070                        featureMapFactory.put(ExecPackage.Literals.CALL__ARGUMENTS, argFactory);
071                }
072                
073                EMap<String, EObject> properties = call.getProperties();
074                if (!properties.isEmpty()) {
075                        MapCompoundSupplierFactory<String,Object> propertiesFactory = new MapCompoundSupplierFactory<>("Properties");
076                        for (Entry<String, EObject> pe: properties.entrySet()) {
077                                propertiesFactory.put(pe.getKey(), Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(pe.getValue(), Object.class), "Cannot adapt to SupplierFactory: " + pe.getValue()));
078                        }
079                        featureMapFactory.put(ExecPackage.Literals.CALL__PROPERTIES, propertiesFactory);
080                }
081                
082                return featureMapFactory.then(this::createResultFunction).create(context);
083        }
084        
085        protected Function<Map<EStructuralFeature,Object>, InputStream> createResultFunction(Context context) {
086                Call call = (Call) getTarget();
087                
088                return new Function<Map<EStructuralFeature,Object>, InputStream>() {
089                        
090                        private Class<?> clazz;
091                        private Constructor<?> constructor;
092                        private Object instance;                
093                        private Method method;
094
095                        @Override
096                        public double size() {
097                                return 1;
098                        }
099
100                        @Override
101                        public String name() {
102                                String description = call.getDescription();
103                                return Util.isBlank(description) ? "Call" : "Call " + description;
104                        }
105                        
106                        @Override
107                        public Diagnostic diagnose(ProgressMonitor progressMonitor) {
108                                String property = call.getProperty();
109                                if (!Util.isBlank(property)) {
110                                        instance = context.get(property);
111                                        if (instance == null) {
112                                                return new BasicDiagnostic(Status.ERROR, "Call property not found: " + property, call);
113                                        }
114                                        if (!call.getInit().isEmpty()) {
115                                                return new BasicDiagnostic(Status.ERROR, "Cannot specify constructor arguments for property", call);                                                            
116                                        }
117                                        if (!call.getProperties().isEmpty()) {
118                                                return new BasicDiagnostic(Status.ERROR, "Cannot specify properties for property", call);                                                               
119                                        }
120                                } else {
121                                        ClassLoader classLoader = context.get(ClassLoader.class, getClass().getClassLoader());
122                                        String service = call.getService();
123                                        if (!Util.isBlank(service)) {
124                                                try {
125                                                        instance = context.get(classLoader.loadClass(service));
126                                                        if (instance == null) {
127                                                                return new BasicDiagnostic(Status.ERROR, "Call service not found: " + service, call);
128                                                        }
129                                                        if (!call.getInit().isEmpty()) {
130                                                                return new BasicDiagnostic(Status.ERROR, "Cannot specify constructor arguments for service", call);                                                             
131                                                        }
132                                                        if (!call.getProperties().isEmpty()) {
133                                                                return new BasicDiagnostic(Status.ERROR, "Cannot specify properties for service", call);                                                                
134                                                        }
135                                                } catch (ClassNotFoundException e) {
136                                                        return new BasicDiagnostic(Status.ERROR, "Service class not found: " + service, call);
137                                                }                                               
138                                        } else {
139                                                String type = call.getType();
140                                                if (Util.isBlank(type)) {
141                                                        return new BasicDiagnostic(Status.ERROR, "Call must specify one of 'class', 'property', or 'service'", call);                                                   
142                                                }
143                                                try {
144                                                        clazz = classLoader.loadClass(type);
145                                                        if (!BiConsumer.class.isAssignableFrom(clazz) && !call.getProperties().isEmpty()) {
146                                                                return new BasicDiagnostic(Status.ERROR, "In order to inject properties " + clazz.getName() + " must implement " + BiConsumer.class, call);                                                                                                                     
147                                                        }
148                                                        if ((SupplierFactory.class.isAssignableFrom(clazz) || Supplier.class.isAssignableFrom(clazz))
149                                                                        && call.getInit().isEmpty()
150                                                                        && call.getProperties().isEmpty()
151                                                                        && Util.isBlank(call.getMethod())) {
152                                                                // Special case - creating an instance here so can be diagnosed
153                                                                try {
154                                                                        Constructor<?> constr = clazz.getDeclaredConstructor();
155                                                                        Object obj = constr.newInstance();
156                                                                        instance = SupplierFactory.class.isAssignableFrom(clazz) ? ((SupplierFactory<?>) obj).create(context) : (Supplier<?>) obj;
157                                                                        clazz = null;
158                                                                        return ((Supplier<?>) instance).diagnose(progressMonitor);
159                                                                } catch (Exception e) {
160                                                                        return new BasicDiagnostic(Status.ERROR, "Cannot instantiate " + clazz, call, e);                                                                                                                       
161                                                                        
162                                                                }                                                               
163                                                        }
164                                                } catch (ClassNotFoundException e) {
165                                                        return new BasicDiagnostic(Status.ERROR, "Class not found: " + type, call);
166                                                }                                                                                               
167                                        }
168                                }
169                                String methodName = call.getMethod();
170                                if (Util.isBlank(methodName)) {
171                                        if (!call.getArguments().isEmpty()) {
172                                                return new BasicDiagnostic(Status.ERROR, "Cannot specify arguments without method", call);                                                              
173                                        }
174                                        for (Constructor<?> constructor: clazz.getConstructors()) {
175                                                if (constructor.getParameterCount() == call.getInit().size()) {
176                                                        this.constructor = constructor;
177                                                        break;
178                                                }
179                                        }               
180                                        if (constructor == null) {
181                                                return new BasicDiagnostic(Status.ERROR, "There is no constructor for class '" + call.getType() + "' which takes " + call.getInit().size() + " arguments", call);
182                                        }
183                                } else {
184                                        if (clazz == null) {
185                                                for (Method method: instance.getClass().getMethods()) {
186                                                        if (method.getName().equals(methodName) && method.getParameterCount() == call.getArguments().size() && !Modifier.isStatic(method.getModifiers())) {
187                                                                this.method = method;
188                                                                break;
189                                                        }
190                                                }               
191                                                if (method == null) {
192                                                        return new BasicDiagnostic(Status.ERROR, "Method '" + methodName +"' with " + call.getArguments().size() + " arguments not found in " + instance, call);                                                                                                                                                                                
193                                                }
194                                        } else {
195                                                for (Method method: clazz.getMethods()) {
196                                                        if (method.getName().equals(methodName) && method.getParameterCount() == call.getArguments().size()) {
197                                                                if (Modifier.isStatic(method.getModifiers())) {
198                                                                        if (!call.getInit().isEmpty()) {
199                                                                                return new BasicDiagnostic(Status.ERROR, "Cannot specify constructor arguments for a static method", call);                                                             
200                                                                        }                                                               
201                                                                        if (!call.getProperties().isEmpty()) {
202                                                                                return new BasicDiagnostic(Status.ERROR, "Cannot specify properties for a static method", call);                                                                
203                                                                        }
204                                                                } else if (clazz != null) {
205                                                                        for (Constructor<?> constructor: clazz.getConstructors()) {
206                                                                                if (constructor.getParameterCount() == call.getInit().size()) {
207                                                                                        this.constructor = constructor;
208                                                                                        break;
209                                                                                }
210                                                                        }               
211                                                                        if (constructor == null) {
212                                                                                return new BasicDiagnostic(Status.ERROR, "There is no constructor for class '" + call.getType() + "' which takes " + call.getInit().size() + " arguments", call);
213                                                                        }
214                                                                }
215                                                                this.method = method;
216                                                                break;
217                                                        }
218                                                }               
219                                                if (method == null) {
220                                                        return new BasicDiagnostic(Status.ERROR, "Method '" + methodName +"' with " + call.getArguments().size() + " arguments not found in " + clazz, call);                                                                                                                                                                           
221                                                }                                               
222                                        }
223                                }
224                                return Function.super.diagnose(progressMonitor);
225                        }

227                        @SuppressWarnings("unchecked")
228                        @Override
229                        public InputStream execute(Map<EStructuralFeature,Object> data, ProgressMonitor progressMonitor) {
230                                List<Object> initList = (List<Object>) data.get(ExecPackage.Literals.CALL__INIT);
231                                
232                                Converter converter = context.get(Converter.class, DefaultConverter.INSTANCE);
233                                Object obj;
234                                if (clazz == null) {
235                                        obj = instance;
236                                } else if (constructor == null) {
237                                        obj = null;
238                                } else {
239                                        Object[] constructorArguments = coerce(context, initList, constructor.getParameterTypes(), progressMonitor);
240                                        try {
241                                                obj = constructor.newInstance(constructorArguments);
242                                        } catch (InstantiationException | IllegalAccessException | IllegalArgumentException     | InvocationTargetException e) {
243                                                throw new ExecutionException(e, this);
244                                        }
245                                        Map<String,Object> properties = (Map<String,Object>) data.get(ExecPackage.Literals.CALL__PROPERTIES);                                   
246                                        if (properties != null) {
247                                                for (Entry<String, Object> pe: properties.entrySet()) {
248                                                        ((BiConsumer<String,Object>) obj).accept(pe.getKey(), pe.getValue());
249                                                }
250                                        }
251                                }
252
253                                Object result;
254                                if (method == null) {
255                                        result = obj;
256                                } else {
257                                        List<Object> argList = (List<Object>) data.get(ExecPackage.Literals.CALL__ARGUMENTS);
258                                        Object[] methodArguments = coerce(context, argList, method.getParameterTypes(), progressMonitor);
259                                        try {
260                                                result = method.invoke(obj, methodArguments);
261                                        } catch (IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
262                                                throw new ExecutionException(e, this);
263                                        }                                       
264                                }
265
266                                if (result instanceof SupplierFactory) {
267                                        result = ((SupplierFactory<?>) result).create(context); 
268                                } 
269                                
270                                if (result instanceof Supplier) {
271                                        try {
272                                                result = Util.call((Supplier<?>) result, progressMonitor, null);
273                                        } catch (Exception e) {
274                                                throw new ExecutionException(e, this);
275                                        }
276                                }
277                                
278                                if (result == null) {
279                                        return null;
280                                }
281
282                                InputStream ret = converter.convert(result, InputStream.class);
283                                if (ret == null) {
284                                        throw new ConfigurationException("Cannot convert '" + obj + "' to InputStream", EObjectAdaptable.adaptTo((EObject) getTarget(), Marked.class));
285                                }
286                                return ret;
287                        }
288                };
289        }

291        /**
292         * Converts arguments to parameter types where possible.
293         * @param arguments
294         * @param parameterTypes
295         * @return
296         */
297        private Object[] coerce(Context context, List<Object> arguments, Class<?>[] parameterTypes, ProgressMonitor progressMonitor) {
298                if (arguments == null) {
299                        return null;
300                }
301                Object[] ret = arguments.toArray();
302                for (int i = 0; i < ret.length; ++i) {
303                        ret[i] = coerce(context, ret[i], parameterTypes[i], progressMonitor);
304                }
305                return ret;
306        }
307        
308        /**
309         * Converts argument to parameter type where possible.
310         * @param argument
311         * @param parameterType
312         * @return
313         */
314        private Object coerce(Context context, Object argument, Class<?> parameterType, ProgressMonitor progressMonitor) {
315                if (argument == null || parameterType.isInstance(argument)) {
316                        return argument;
317                }
318                if (parameterType == String.class && argument instanceof InputStream) {
319                        try {
320                                return Util.toString(context, (InputStream ) argument);
321                        } catch (IOException e) {
322                                throw new ExecutionException(e);
323                        }
324                }
325                Converter converter = context.get(Converter.class, DefaultConverter.INSTANCE);
326                Object ret = converter.convert(argument, parameterType);
327                if (ret == null) {
328                        throw new ConfigurationException("Cannot convert " + argument + " to " + parameterType, EObjectAdaptable.adaptTo((EObject) getTarget(), Marked.class));
329                }
330                return ret;
331        }
