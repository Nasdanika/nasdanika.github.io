001package org.nasdanika.exec.gen;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.util.HashMap;
006import java.util.Map;
007import java.util.Map.Entry;
008import java.util.Objects;
009
010import org.eclipse.emf.common.notify.impl.AdapterImpl;
011import org.eclipse.emf.common.util.EMap;
012import org.eclipse.emf.ecore.EObject;
013import org.nasdanika.common.BiSupplier;
014import org.nasdanika.common.DefaultConverter;
015import org.nasdanika.common.ExecutionException;
016import org.nasdanika.common.Function;
017import org.nasdanika.common.FunctionFactory;
018import org.nasdanika.common.MapCompoundSupplierFactory;
019import org.nasdanika.common.ProgressMonitor;
020import org.nasdanika.common.SupplierFactory;
021import org.nasdanika.common.Util;
022import org.nasdanika.emf.EObjectAdaptable;
023import org.nasdanika.exec.Eval;
024
025public class EvalSupplierFactoryProviderAdapter extends AdapterImpl implements SupplierFactory.Provider {
026        
027        private static final String CONTEXT_BINDING = "context";
028        private static final String PROGRESS_MONITOR_BINDING = "progressMonitor";
029        
030        public EvalSupplierFactoryProviderAdapter(Eval eval) {
031                setTarget(eval);
032        }
033        
034        @Override
035        public boolean isAdapterForType(Object type) {
036                return type == SupplierFactory.class;
037        }
038        
039        @SuppressWarnings("unchecked")
040        @Override
041        public <T> SupplierFactory<T> getFactory(Class<T> type) {
042                Eval eval = (Eval) getTarget();
043                
044                MapCompoundSupplierFactory<String, Object> bindingsFactory = new MapCompoundSupplierFactory<String, Object>("Bindings");
045                EMap<String, EObject> bindings = eval.getBindings();
046                for (Entry<String, EObject> ce: bindings.entrySet()) {
047                        EObject value = ce.getValue();
048                        SupplierFactory<?> sf = Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(value, Object.class), "Cannot adapt " + value + " to SupplierFactory");                          
049                        bindingsFactory.put(ce.getKey(), sf.then(Util.OBJECT_TO_STRING_FACTORY));
050                }
051                
052                SupplierFactory<InputStream> scriptFactory = Objects.requireNonNull(EObjectAdaptable.adaptToSupplierFactory(eval.getScript(), InputStream.class), "Cannot adapt " + eval.getScript() + " to SupplierFactory");          
053                
054                
055                
056                if (type == null || type == Object.class) {
057                        FunctionFactory<BiSupplier<InputStream, Map<String,Object>>, Object> functionFactory = context -> new Function<BiSupplier<InputStream, Map<String,Object>>, Object>() {
058
059                                @Override
060                                public double size() {
061                                        return 1;
062                                }
063
064                                @Override
065                                public String name() {
066                                        return "eval";
067                                }
068
069                                @Override
070                                public Object execute(BiSupplier<InputStream, Map<String,Object>> input, ProgressMonitor progressMonitor) {
071                                        Map<String,Object> bindings = new HashMap<>(input.getSecond());
072                                        bindings.put(CONTEXT_BINDING, context);
073                                        bindings.put(PROGRESS_MONITOR_BINDING, progressMonitor);                                        
074                                        throw new UnsupportedOperationException("TODO - implement with Spring Expression language or drop eval");
075//                                      try {
076//                                              return Util.eval(input.getFirst(), bindings);
077//                                      } catch (IOException e) {
078//                                              throw new ExecutionException(e, this);
079//                                      }
080                                }
081                        };
082                        
083                        return (SupplierFactory<T>) scriptFactory.then(bindingsFactory.asFunctionFactory()).then(functionFactory);
084                }
085                
086                if (type == Boolean.class) {
087                        FunctionFactory<Object, Boolean> booleanFactory = context -> new Function<Object,Boolean>() {
088
089                                @Override
090                                public double size() {
091                                        return 1;
092                                }
093
094                                @Override
095                                public String name() {
096                                        return "boolean";
097                                }
098
099                                @Override
100                                public Boolean execute(Object obj, ProgressMonitor progressMonitor) {
101                                        return Boolean.TRUE.equals(obj);
102                                }
103                        };
104                        
105                        return (SupplierFactory<T>) getFactory(Object.class).then(booleanFactory);                      
106                }
107                
108                if (type == InputStream.class) {
109                        FunctionFactory<Object, InputStream> booleanFactory = context -> new Function<Object,InputStream>() {
110
111                                @Override
112                                public double size() {
113                                        return 1;
114                                }
115
116                                @Override
117                                public String name() {
118                                        return "stream";
119                                }
120
121                                @Override
122                                public InputStream execute(Object obj, ProgressMonitor progressMonitor) {
123                                        InputStream ret = DefaultConverter.INSTANCE.convert(obj, InputStream.class);
124                                        if (obj != null && ret == null) {
125                                                try {
126                                                        ret = Util.toStream(context, obj.toString());
127                                                } catch (IOException e) {
128                                                        throw new ExecutionException(e, this);
129                                                }
130                                        }
131                                        return ret;
132                                }
133                        };
134                        
135                        return (SupplierFactory<T>) getFactory(Object.class).then(booleanFactory);                      
136                }
137                
138                throw new IllegalArgumentException("Unsupported type: " + type);
139        }
140
141}
