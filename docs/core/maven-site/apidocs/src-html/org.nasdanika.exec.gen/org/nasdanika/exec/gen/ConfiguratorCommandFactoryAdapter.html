001package org.nasdanika.exec.gen;
002
003import java.util.Objects;
004
005import org.nasdanika.common.Command;
006import org.nasdanika.common.CommandFactory;
007import org.nasdanika.common.Context;
008import org.nasdanika.common.SupplierFactory;
009import org.nasdanika.emf.EObjectAdaptable;
010import org.nasdanika.exec.Configurator;
011
012public class ConfiguratorCommandFactoryAdapter extends ConfiguratorExecutionParticipantAdapter implements CommandFactory {
013        
014        public ConfiguratorCommandFactoryAdapter(Configurator configurator) {
015                super(configurator);
016        }
017        
018        @Override
019        public boolean isAdapterForType(Object type) {
020                return type == SupplierFactory.class;
021        }
022        
023        @Override
024        public Command create(Context context) {
025                SupplierFactory<Context> contextSupplierFactory = createContextSupplierFactory();
026                Configurator configurator = (Configurator) getTarget();
027                CommandFactory commandFactory = Objects.requireNonNull(EObjectAdaptable.adaptTo(configurator.getTarget(), CommandFactory.class), "Cannot adapt " + configurator.getTarget() + " to CommandFactory");            
028                if (contextSupplierFactory == null) {
029                        return commandFactory.create(context);
030                }
031                
032                return commandFactory.contextify(contextSupplierFactory).create(context);
033        }       
034
035}
