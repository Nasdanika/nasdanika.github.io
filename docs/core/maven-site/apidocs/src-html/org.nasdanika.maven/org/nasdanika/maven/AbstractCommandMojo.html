012import org.apache.maven.plugin.logging.Log;
013import org.apache.maven.plugins.annotations.Parameter;

027/**
028 * Base class for plugins which execute {@link Command}'s. 
029 * This class creates a {@link Context} and a {@link ProgressMonitor} and then executes a command returned from createCommand() method.
030 */
031public abstract class AbstractCommandMojo extends AbstractMojo {
032
033        /**
034         * File to output progress. If not set, progress is output to the log.
035         */
036        @Parameter      
037        private File progressOutput;    
038
039        /**
040         * File to output diagnostic. If not set, diagnostic is output to the log.
041         */
042        @Parameter      
043        private File diagnosticOutput;  
044        
045        /**
046         * Statuses on which to fail execution. FAIL if empty. Available statuses: FAIL, ERROR, WARNING, SUCCESS, INFO, CANCEL
047         */
048    @Parameter
049    private Status[] failStatuses;
050    
051    /**
052     * URL's of YAML or JSON resources which are loaded as contexts. if the URL ends with .yaml or .yml then it is treated as YAML, as a JSON object otherwise. 
053     */
054    @Parameter
055    private List<String> contexts;
056        
057//      @Parameter(name = "json-progress")      
058//      private boolean jsonProgress;   
059        
060        @Parameter(defaultValue = "${project}", required = true, readonly = true)
061        protected MavenProject project; 
062        
063        
064        /**
065         * Creates a context with {@link MavenProject} service.
066         * Then composes it with contexts loaded from context 
067         * @return
068         * @throws IOException 
069         * @throws MojoExecutionException 
070         */
071        @SuppressWarnings("unchecked")
072        protected Context createContext() throws IOException, MojoExecutionException {
073                Context context = Context.singleton(MavenProject.class, project);
074                
075                Map<Object, Object> pc = getPluginContext();
076                if (pc != null) {
077                        Context wpc = Context.wrap(pc::get); 
078                        context = context.compose(wpc);
079                }
080                
081                if (contexts != null && !contexts.isEmpty()) {
082                        File baseDir = project.getBasedir();
083                        URI baseDirURI = URI.createFileURI(baseDir.getAbsolutePath()).appendSegment("");
084                        
085                        for (String ctxLocation: contexts) {
086                                URI ctxURI = URI.createURI(ctxLocation).resolve(baseDirURI);
087                                String lastSegment = ctxURI.lastSegment().toLowerCase();
088                                if (lastSegment.endsWith(".yml") || lastSegment.endsWith(".yaml")) {
089                                        Object yamlObj = DefaultConverter.INSTANCE.loadYAML(ctxURI);
090                                        if (yamlObj instanceof Map) {
091                                                Map<Object, Object> yamlMap = (Map<Object, Object>) yamlObj;                                            
092                                                context = context.compose(Context.wrap(context.interpolate(yamlMap)::get));
093                                        } else {
094                                                String message = "Context is not a YAML map: " + ctxURI;
095                                                getLog().error(message);
096                                                throw new MojoExecutionException(message); 
097                                        }
098                                } else {
099                                        JSONObject jsonObject = DefaultConverter.INSTANCE.loadJSONObject(ctxURI);
100                                        Map<String, Object> jsonMap = jsonObject.toMap();
101                                        context = context.compose(Context.wrap(context.interpolate(jsonMap)::get));
102                                }
103                        }
104                }
105                
106                return context;
107        }
108        
109        /**
110         * Creates a progress monitor outputting either to the log or to a progress output file.
111         * @return
112         * @throws FileNotFoundException
113         */
114        protected ProgressMonitor createProgressMonitor() throws FileNotFoundException {
115                if (progressOutput == null) {
116                        return new LogProgressMonitor(getLog(), 0, 2);
117                }
118                
119//              if (jsonProgress) {
120//                      // JSON output progress monitor
121//                      
122//              }
123                File parent = progressOutput.getParentFile();
124                if (parent != null) {
125                        parent.mkdirs();
126                }
127                
128                return new PrintStreamProgressMonitor(new PrintStream(progressOutput), 0, 2, true);
129        }
130        
131        /**
132         * @return Command to be executed
133         */
134        protected Command createCommand(Context context) {
135                return new Command() {
136
137                        @Override
138                        public double size() {
139                                return 1;
140                        }
141
142                        @Override
143                        public String name() {
144                                return AbstractCommandMojo.this.getClass().getName();
145                        }
146
147                        @Override
148                        public void execute(ProgressMonitor progressMonitor) {
149                                AbstractCommandMojo.this.execute(context, progressMonitor);                             
150                        }
151                        
152                };
153        }
154        
155        protected abstract void execute(Context context, ProgressMonitor progressMonitor);
156
157        public void execute() throws MojoExecutionException {
158                Status[] failStatusesArray =  failStatuses == null ? new Status[0] : failStatuses;
159                try {
160                        logDiagnostic(Util.call(createCommand(createContext()), createProgressMonitor(), failStatusesArray));                   
161                } catch (DiagnosticException e) {
162                        MojoExecutionException mojoExecutionException = new MojoExecutionException("Diagnostic exception: " + e, e);
163                        try {
164                                logDiagnostic(e.getDiagnostic());
165                        } catch (FileNotFoundException e1) {
166                                mojoExecutionException.addSuppressed(e1);
167                                throw mojoExecutionException;
168                        }
169                        throw mojoExecutionException;
170                } catch (IOException e) {
171                        throw new MojoExecutionException(e);
172                }
173        }
174        
175        protected void logDiagnostic(Diagnostic diagnostic) throws FileNotFoundException {
176                if (diagnosticOutput == null) {
177                        log(diagnostic, 0, getLog());
178                } else {
179                        File parent = diagnosticOutput.getParentFile();
180                        if (parent != null) {
181                                parent.mkdirs();
182                        }
183                        try (PrintStream out = new PrintStream(diagnosticOutput)) {
184                                diagnostic.dump(out, 0);
185                        }
186                }               
187        }       
188        
189        private static void log(Diagnostic diagnostic, int indent, Log log) {
190                StringBuilder messageBuilder = new StringBuilder();
191                for (int i=0; i < indent; ++i) {
192                        messageBuilder.append("  ");
193                }
194                                
195                messageBuilder.append(diagnostic.getMessage());
196
197                List<Object> data = diagnostic.getData();
198                if (data != null) {
199                        messageBuilder.append(" data=");
200                        messageBuilder.append(diagnostic.getData());
201                }
202                
203                Status status = diagnostic.getStatus();
204                if (status == Status.ERROR) {
205                        log.error(messageBuilder);
206                        if (data != null) {
207                                for (Object de: data) {
208                                        if (de instanceof Throwable) {
209                                                log.error((Throwable) de);
210                                        }
211                                }
212                        }
213                } else if (status == Status.WARNING) {
214                        log.warn(messageBuilder);
215                        if (data != null) {
216                                for (Object de: data) {
217                                        if (de instanceof Throwable) {
218                                                log.warn((Throwable) de);
219                                        }
220                                }
221                        }
222                } else if (status == Status.INFO) {
223                        log.info(messageBuilder);
224                        if (data != null) {
225                                for (Object de: data) {
226                                        if (de instanceof Throwable) {
227                                                log.info((Throwable) de);
228                                        }
229                                }
230                        }
231                } else {
232                        if (status != null) {
233                                messageBuilder.insert(0, "[" + status.name() + "] ");
234                        }
235                        log.debug(messageBuilder);
236                        if (data != null) {
237                                for (Object de: data) {
238                                        if (de instanceof Throwable) {
239                                                log.debug((Throwable) de);
240                                        }
241                                }
242                        }
243                }
244                
245            List<Diagnostic> children = diagnostic.getChildren();
246                if (children != null) {
247                children.forEach(c -> log(c, indent + 1, log));
248            }
249        }
