001package org.nasdanika.maven;
002
003import java.util.Arrays;
004import java.util.concurrent.CancellationException;
005
006import org.apache.maven.plugin.logging.Log;
007import org.nasdanika.common.ProgressMonitor;
008import org.nasdanika.common.Status;
009
010/**
011 * Progress monitor reporting to {@link Log}.
012 * @author Pavel
013 *
014 */
015public class LogProgressMonitor implements ProgressMonitor {
016
017        private Log log;
018        private String indent;
019        protected int indentIncrement = 2;
020        private boolean cancelled;
021        
022        /**
023         * Constructs a progress monitor for a given log.
024         */
025        public LogProgressMonitor(Log log, int indent, int indentIncrement) {
026                this.log = log;
027                StringBuilder indentBuilder = new StringBuilder();
028                for (int i = 0; i < indent; ++i) {
029                        indentBuilder.append(' ');
030                }
031                this.indent = indentBuilder.toString();
032        }
033
034        @Override
035        public boolean isCancelled() {
036                return cancelled;
037        }
038
039        @Override
040        public ProgressMonitor split(String taskName, double size, Object... data) {
041                if (isCancelled()) {
042                        throw new CancellationException();
043                }
044                log.info(indent + "  " + taskName + " ("+size+")");
045                if (data != null) {
046                        for (Object d: data) {
047                                log.info(formatDetail(d, indent + "    "));                     
048                        }
049                }
050                return new LogProgressMonitor(log, indent.length() + indentIncrement, indentIncrement) {
051                        
052                        @Override
053                        public boolean isCancelled() {
054                                return LogProgressMonitor.this.isCancelled();
055                        }
056                        
057                };
058        }
059        
060        /**
061         * Formats detail element for output. This implementation concatenates the indent with the detail.
062         * @param detail
063         * @param indent
064         * @return
065         */
066        protected String formatDetail(Object detail, String indent) {
067                return indent + detail;
068        }
069
070        @Override
071        public void worked(Status status, double work, String progressMessage, Object... data) {
072                switch (status) {
073                case ERROR:
074                case FAIL:
075                        log.error(indent + "  [" + status + " " + work + "] " + progressMessage);
076                        if (data.length > 0) {
077                                log.error(": " + Arrays.deepToString(data));
078                        }
079                        break;
080                case WARNING:
081                        log.warn(indent + "  [" + status + " " + work + "] " + progressMessage);
082                        if (data.length > 0) {
083                                log.warn(": " + Arrays.deepToString(data));
084                        }
085                        break;
086                case CANCEL:
087                case INFO:
088                case SUCCESS:
089                default:
090                        log.info(indent + "  [" + status + " " + work + "] " + progressMessage);
091                        if (data.length > 0) {
092                                log.info(": " + Arrays.deepToString(data));
093                        }
094                        if (status == Status.CANCEL) {
095                                cancelled = true;
096                        }
097                }
098        }
099
100        @Override
101        public ProgressMonitor setWorkRemaining(double size) {
102                // NOP
103                return this;
104        }
105
106        @Override
107        public void close() {
108                // NOP          
109        }
110
111}
