001package org.nasdanika.common;
002
003/**
004 * Something that can diagnose/validate T
005 * @author Pavel
006 *
007 */
008public interface Diagnostician<T> {
009        
010        /**
011         * Diagnoses the first argument in the context of the second.
012         * @param obj Object to be diagnosed/validated.
013         * @param context Source of contextual information.
014         * @return Diagnostic results.
015         */
016        Diagnostic diagnose(T obj, Context context);
017
018}
