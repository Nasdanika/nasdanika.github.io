087        @ConverterMethod
088        public String toString(Reader reader) throws IOException {
089                StringWriter sw = new StringWriter();
090                try (BufferedReader br = new BufferedReader(reader)) {
091                        int ch;
092                        while ((ch = br.read()) != -1) {
093                                sw.write(ch);
094                        }
095                }
096                sw.close();
097                return sw.toString();
098        }

259        @ConverterMethod
260        public Instant toInstant(String text) {
261                if (text == null) {
262                        return null;
263                }
264                if (text.contains("Z")) {
265                        return Instant.parse(text);
266                }
267                // Parsing via Date for convenience
268                Date date = convert(text, Date.class);
269                return date == null ? null : date.toInstant();
270        }

341        @ConverterMethod
342        public String toString(Node node) throws TransformerException, IOException {
343            DOMSource source = new DOMSource(node);
344            StringWriter out = new StringWriter();
345                TransformerFactory tFactory = TransformerFactory.newInstance();
346            Transformer transformer = tFactory.newTransformer();
347            transformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
348        transformer.setOutputProperty(OutputKeys.INDENT, "yes");
349                transformer.setOutputProperty(OutputKeys.ENCODING, "UTF-8");
350            
351            try (out) {
352                transformer.transform(source, new StreamResult(out));
353            }
354                return out.toString();
355        }

443        /**
444         * Outputs EObject as a Map of {@link EStructuralFeature}s names to their values. Such a map can then be used to output to YAML or JSON.
445         * @param eObj EObject
446         * @param referenceProvider Providers or values for non-containment references. If null such references are not output to the map.
447         * @return
448         */
449        @SuppressWarnings("unchecked")
450        public Map<String, Object> toMap(EObject eObj, java.util.function.Function<EObject, Object> referenceProvider) {
451                if (eObj == null) {
452                        return null;
453                }
454                Map<String, Object> ret = new LinkedHashMap<>();
455                for (EStructuralFeature sf: eObj.eClass().getEAllStructuralFeatures()) {
456                        if (eObj.eIsSet(sf)) {
457                                if (sf instanceof EAttribute) {
458                                        if (sf.isMany()) {
459                                                Collection<Object> vList = new ArrayList<>();
460                                                for (Object av: (Collection<Object>) eObj.eGet(sf)) {
461                                                        vList.add(av);
462                                                }
463                                                if (!vList.isEmpty()) {
464                                                        ret.put(sf.getName(), vList);
465                                                }
466                                        } else {
467                                                ret.put(sf.getName(), eObj.eGet(sf));
468                                        }
469                                } else {
470                                        EReference eRef = (EReference) sf;
471                                        if (eRef.isContainment()) {
472                                                if (eRef.isMany()) {
473                                                        Collection<Map<String, Object>> vList = new ArrayList<>();
474                                                        for (EObject rv: (Collection<EObject>) eObj.eGet(sf)) {
475                                                                vList.add(toMap(rv, referenceProvider));
476                                                        }
477                                                        if (!vList.isEmpty()) {
478                                                                ret.put(sf.getName(), vList);
479                                                        }
480                                                } else {
481                                                        ret.put(sf.getName(), toMap((EObject) eObj.eGet(sf), referenceProvider));
482                                                }
483                                        } else if (referenceProvider != null) {
484                                                if (eRef.isMany()) {
485                                                        Collection<Object> vList = new ArrayList<>();
486                                                        for (EObject rv: (Collection<EObject>) eObj.eGet(sf)) {
487                                                                vList.add(referenceProvider.apply(rv));
488                                                        }
489                                                        if (!vList.isEmpty()) {
490                                                                ret.put(sf.getName(), vList);
491                                                        }
492                                                } else {
493                                                        ret.put(sf.getName(), referenceProvider.apply((EObject) eObj.eGet(sf)));
494                                                }                                               
495                                        }
496                                }
497                        }                       
498                }
499                return ret;
500        }
