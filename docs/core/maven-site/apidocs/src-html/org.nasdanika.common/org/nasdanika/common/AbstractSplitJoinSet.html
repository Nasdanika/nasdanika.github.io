001package org.nasdanika.common;
002
003import java.util.Collection;
004import java.util.Iterator;
005import java.util.Set;
006
007public abstract class AbstractSplitJoinSet<S,C,E> extends AbstractSplitJoinCollection<S, C, E> implements Set<E> {
008        
009        @Override
010        public boolean add(E e) {
011                if (contains(e)) {
012                        return false;
013                }
014                return super.add(e);
015        }
016        
017    public boolean equals(Object o) {
018        if (o == this) {
019            return true;
020        }
021
022        if (!(o instanceof Set)) {
023            return false;
024        }
025        
026        Set<?> s = (Set<?>) o;
027        return s.size() == size() && containsAll(s);
028    }
029
030    public int hashCode() {
031        int ret = 0;
032        for (E e: this) {
033                if (e != null) {
034                        ret ^= e.hashCode();
035                }
036        }
037        return ret;
038    }
039
040    public boolean removeAll(Collection<?> c) {
041        if (c == null) {
042                return false;
043        }
044        boolean ret = false;
045        if (size() < c.size()) {
046                Iterator<?> i = iterator();
047            while (i.hasNext()) {
048                if (c.contains(i.next())) {
049                    i.remove();
050                    ret = true;
051                }
052            }
053        } else {
054            for (Object e : c) {
055                ret = remove(e) | ret;
056            }
057        }
058        return ret;
059    }   
060
061}
