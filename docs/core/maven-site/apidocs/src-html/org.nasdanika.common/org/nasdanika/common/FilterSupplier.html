001package org.nasdanika.common;
002
003public class FilterSupplier<T> extends FilterExecutionParticipant<Supplier<T>> implements Supplier<T> { 
004
005        public FilterSupplier(Supplier<T> target) {
006                super(target);
007        }
008
009        @Override
010        public T execute(ProgressMonitor progressMonitor) {
011                return target.execute(progressMonitor);
012        }
013
014}
