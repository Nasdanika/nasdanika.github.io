001package org.nasdanika.common;
002
003import java.io.PrintStream;
004import java.util.Arrays;
005import java.util.concurrent.CancellationException;
006
007/**
008 * Progress monitor reporting to a {@link PrintStream}, e.g. ``System.out``.
009 * This monitor indents sub-tasks and worked messages. It can be thought of as a hierarchical logger.
010 * @author Pavel
011 *
012 */
013public class PrintStreamProgressMonitor implements ProgressMonitor {
014
015        private PrintStream out;
016        private boolean closeStream;
017        private String indent;
018        protected int indentIncrement = 2;
019        private boolean cancelled;
020        
021        /**
022         * Constructs a progress monitor for a given print stream.
023         * @param out
024         * @param indent Indent in spaces for this monitor.
025         * @param indentIncrement Increment to add to the indent of this monitor when constructing a sub-monitor.
026         * @param closeStream if true the monitor closes the stream in its close() method.
027         */
028        public PrintStreamProgressMonitor(PrintStream out, int indent, int indentIncrement, boolean closeStream) {
029                this.out = out;
030                StringBuilder indentBuilder = new StringBuilder();
031                for (int i = 0; i < indent; ++i) {
032                        indentBuilder.append(' ');
033                }
034                this.indent = indentBuilder.toString();
035                this.closeStream = closeStream;
036        }
037        
038        /**
039         * Constructs a progres monitor outputting to System.out with indent 0, indentIncrement 2 and not closing the stream.
040         */
041        public PrintStreamProgressMonitor() {
042                this(System.out, 0, 2, false);
043        }
044        
045        @Override
046        public void close() {
047                if (closeStream) {
048                        out.close();
049                }
050        }
051
052        @Override
053        public boolean isCancelled() {
054                return cancelled;
055        }
056
057        @Override
058        public ProgressMonitor split(String taskName, double size, Object... data) {
059                if (isCancelled()) {
060                        throw new CancellationException();
061                }
062                out.println(indent+"  "+taskName+" ("+size+")");
063                if (data != null) {
064                        for (Object d: data) {
065                                out.println(formatDetail(d, indent + "    "));                  
066                        }
067                }
068                return new PrintStreamProgressMonitor(out, indent.length()+indentIncrement, indentIncrement, false) {
069                        
070                        @Override
071                        public boolean isCancelled() {
072                                return PrintStreamProgressMonitor.this.isCancelled();
073                        }
074                        
075                };
076        }
077        
078        /**
079         * Formats detail element for output. This implementation concatenates the indent with the detail.
080         * @param detail
081         * @param indent
082         * @return
083         */
084        protected String formatDetail(Object detail, String indent) {
085                return indent + detail;
086        }
087
088        @Override
089        public void worked(Status status, double work, String progressMessage, Object... data) {
090                out.print(indent+"  ["+status+" "+work+"] "+progressMessage);
091                if (data.length > 0) { 
092                        out.print(": "+Arrays.deepToString(data));
093                }
094                out.println();
095                if (status == Status.CANCEL) {
096                        cancelled = true;
097                }
098        }
099
100        @Override
101        public ProgressMonitor setWorkRemaining(double size) {
102                // NOP
103                return this;
104        }
105
106}
