001package org.nasdanika.common;
002
003import java.util.ArrayList;
004import java.util.Arrays;
005import java.util.Collection;
006import java.util.List;
007import java.util.concurrent.Callable;
008import java.util.concurrent.ExecutionException;
009import java.util.concurrent.ExecutorService;
010import java.util.concurrent.Future;
011
012public class CompoundCommand extends ListCompoundExecutionParticipant<Command> implements Command  {
013        
014        private ExecutorService executorService;
015
016        public CompoundCommand(String name, ExecutorService executorService, Collection<Command> commands) {
017                super(name);
018                this.executorService = executorService;
019                for (Command command: commands) {
020                        add(command);
021                }
022        }
023        
024        public CompoundCommand(String name, ExecutorService executorService, Command... commands) {
025                this(name, executorService, Arrays.asList(commands));
026        }
027
028        @Override
029        public void execute(ProgressMonitor progressMonitor) {
030                progressMonitor.setWorkRemaining(size());
031                if (executorService == null || elements.size() < 2) {
032                        for (Command e: getElements()) {
033                                e.splitAndExecute(progressMonitor);                     
034                        }
035                } else {
036                        
037                        try {
038                                List<Future<Void>> invokeAll = executorService.invokeAll(getTasks(elements.subList(1, elements.size()), progressMonitor));
039                                elements.get(0).splitAndExecute(progressMonitor); // Execute the first element in the current thread while other elements are executed in other threads.
040                                for (Future<?> future: invokeAll) {
041                                        future.get();
042                                };                      
043                        } catch (InterruptedException | ExecutionException e) {
044                                throw new org.nasdanika.common.ExecutionException(e, this);
045                        }
046                }
047        }
048        
049        private static List<Callable<Void>> getTasks(List<Command> elements, ProgressMonitor progressMonitor) {
050                List<Callable<Void>> ret = new ArrayList<>();
051                for (Command e: elements) {
052                        ret.add(new Callable<Void>() {
053
054                                @Override
055                                public Void call() {
056                                        e.splitAndExecute(progressMonitor);
057                                        return null;
058                                }
059                                
060                        });
061                }               
062                return ret;
063        }
064
065}
