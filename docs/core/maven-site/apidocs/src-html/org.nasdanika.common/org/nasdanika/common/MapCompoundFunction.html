001package org.nasdanika.common;
002
003import java.util.LinkedHashMap;
004import java.util.Map;
005import java.util.Map.Entry;
006
007public class MapCompoundFunction<K,T,R> 
008        extends MapCompoundExecutionParticipant<K, Function<? super T, ? extends R>> 
009        implements Function<T,Map<K,R>>  {
010
011        public MapCompoundFunction(String name) {
012                super(name);
013        }
014
015        public MapCompoundFunction(String name, Map<K,Function<? super T, ? extends R>> functions) {
016                super(name);
017                functions.forEach(this::put);
018        }
019
020        @Override
021        public Map<K,R> execute(T arg, ProgressMonitor progressMonitor) {
022                progressMonitor.setWorkRemaining(size());
023                Map<K,R> result = new LinkedHashMap<>();
024                for (Entry<K, Function<? super T, ? extends R>> e: elements.entrySet()) {
025                        Function<? super T, ? extends R> value = e.getValue();
026                        if (value != null) {
027                                result.put(e.getKey(), value.splitAndExecute(arg, progressMonitor));                    
028                        }
029                }
030                return result;
031        }       
032
033}
