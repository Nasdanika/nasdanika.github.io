001package org.nasdanika.common;
002
003public class ContextifiedExecutionParticipant<T extends ExecutionParticipant>  implements ExecutionParticipant {
004        
005        protected Supplier<Context> contextSupplier;
006        private ExecutionParticipantFactory<T> targetFactory;
007        
008        protected T target;
009
010        public ContextifiedExecutionParticipant(Supplier<Context> contextSupplier, ExecutionParticipantFactory<T> targetFactory) {
011                this.contextSupplier = contextSupplier;
012                this.targetFactory = targetFactory;
013        }
014
015        @Override
016        public double size() {
017                return contextSupplier.size() * 2 + 1; // Get target, target op, context op. Scaling target size to 1.
018        }
019
020        @Override
021        public String name() {
022                return target == null ? getClass().getName() : "Contextified "+target.name();
023        }
024        
025        /**
026         * Create context, create target. Uses context size from the progress monitor. 
027         * @param progressMonitor
028         * @return
029         */
030        protected synchronized T getTarget(ProgressMonitor progressMonitor) {
031                if (target == null) {
032                        target = targetFactory.create(contextSupplier.splitAndExecute(progressMonitor));
033                } else {
034                        progressMonitor.worked(contextSupplier.size(), "Target was already created");
035                }
036                return target;
037        }
038        
039        @Override
040        public Diagnostic diagnose(ProgressMonitor progressMonitor) {
041                Diagnostic cd = contextSupplier.splitAndDiagnose(progressMonitor);
042                if (cd.getStatus() == Status.ERROR) {
043                        return cd;
044                }
045                
046                try {
047                        return getTarget(progressMonitor).splitAndDiagnose(1, progressMonitor);
048                } catch (Exception e) {
049                        return new BasicDiagnostic(Status.ERROR, "Exception creating target: "+e, e);
050                }
051        }
052        
053        @Override
054        public void commit(ProgressMonitor progressMonitor) {
055                getTarget(progressMonitor).splitAndCommit(1, progressMonitor);
056                contextSupplier.splitAndCommit(progressMonitor);
057        }
058        
059        @Override
060        public boolean rollback(ProgressMonitor progressMonitor) {
061                boolean result = getTarget(progressMonitor).splitAndRollback(1, progressMonitor);
062                return contextSupplier.splitAndRollback(progressMonitor) && result;
063        }
064        
065        @Override
066        public void close() {
067                if (target != null) {
068                        target.close();
069                }
070                contextSupplier.close();
071        }
072
073        // ==========================================
074        
075        
076        public static class ContextifiedSupplier<T> extends ContextifiedExecutionParticipant<Supplier<T>> implements Supplier<T> {
077
078                public ContextifiedSupplier(Supplier<Context> contextSupplier, SupplierFactory<T> supplierFactory) {
079                        super(contextSupplier, supplierFactory);
080                }
081
082                @Override
083                public T execute(ProgressMonitor progressMonitor) {
084                        return getTarget(progressMonitor).execute(progressMonitor.scale(contextSupplier.size() + 1));
085                }
086                
087        }
088        
089        public static class ContextifiedFunction<T,R> extends ContextifiedExecutionParticipant<Function<T,R>> implements Function<T,R> {
090
091                public ContextifiedFunction(Supplier<Context> contextSupplier, FunctionFactory<T,R> functionFactory) {
092                        super(contextSupplier, functionFactory);
093                }
094                
095                @Override
096                public R execute(T arg, ProgressMonitor progressMonitor) {
097                        return getTarget(progressMonitor).execute(arg, progressMonitor.scale(contextSupplier.size() + 1));
098                }
099                
100        }
101        
102        public static class ContextifiedConsumer<T> extends ContextifiedExecutionParticipant<Consumer<T>> implements Consumer<T> {
103
104                public ContextifiedConsumer(Supplier<Context> contextSupplier, ConsumerFactory<T> consumerFactory) {
105                        super(contextSupplier, consumerFactory);
106                }
107
108                @Override
109                public void execute(T arg, ProgressMonitor progressMonitor) {
110                        getTarget(progressMonitor).execute(arg, progressMonitor.scale(contextSupplier.size() + 1));
111                }
112                
113        }
114        
115        public static class ContextifiedCommand extends ContextifiedExecutionParticipant<Command> implements Command {
116
117                public ContextifiedCommand(Supplier<Context> contextSupplier, CommandFactory commandFactory) {
118                        super(contextSupplier, commandFactory);
119                }
120
121                @Override
122                public void execute(ProgressMonitor progressMonitor) {
123                        getTarget(progressMonitor).execute(progressMonitor.scale(contextSupplier.size() + 1));
124                }
125                
126        }
127
