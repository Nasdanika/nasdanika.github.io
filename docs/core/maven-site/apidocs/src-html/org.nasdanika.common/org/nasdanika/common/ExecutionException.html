001package org.nasdanika.common;
002
003/**
004 * Wrapper exception for checked exceptions thrown in methods of {@link ExecutionParticipant}s.
005 * @author Pavel
006 *
007 */
008@SuppressWarnings("serial")
009public class ExecutionException extends NasdanikaException {
010        
011        private ExecutionParticipant executionParticipant;
012
013        public ExecutionException(String message) {
014                super(message);
015        }
016
017        public ExecutionException(Throwable cause) {
018                super(cause);
019        }
020
021        public ExecutionException(String message, Throwable cause) {
022                super(message, cause);
023        }
024
025        public ExecutionException(String message, Throwable cause, boolean enableSuppression, boolean writableStackTrace) {
026                super(message, cause, enableSuppression, writableStackTrace);
027        }
028
029        public ExecutionException(String message, ProgressRecorder progressRecorder) {
030                super(message, progressRecorder);
031        }
032
033        public ExecutionException(Throwable cause, ProgressRecorder progressRecorder) {
034                super(cause, progressRecorder);
035        }
036
037        public ExecutionException(String message, Throwable cause, ProgressRecorder progressRecorder) {
038                super(message, cause, progressRecorder);
039        }
040
041        public ExecutionException(String message, ExecutionParticipant executionParticipant) {
042                this(message);
043                this.executionParticipant = executionParticipant;
044        }
045
046        public ExecutionException(Throwable cause, ExecutionParticipant executionParticipant) {
047                this(cause);
048                this.executionParticipant = executionParticipant;
049        }
050
051        public ExecutionException(String message, Throwable cause, ExecutionParticipant executionParticipant) {
052                this(message, cause);
053                this.executionParticipant = executionParticipant;
054        }
055
056        public ExecutionException(String message, Throwable cause, boolean enableSuppression, boolean writableStackTrace, ExecutionParticipant executionParticipant) {
057                this(message, cause, enableSuppression, writableStackTrace);
058                this.executionParticipant = executionParticipant;
059        }
060
061        public ExecutionException(String message, ProgressRecorder progressRecorder, ExecutionParticipant executionParticipant) {
062                this(message, progressRecorder);
063                this.executionParticipant = executionParticipant;
064        }
065
066        public ExecutionException(Throwable cause, ProgressRecorder progressRecorder, ExecutionParticipant executionParticipant) {
067                this(cause, progressRecorder);
068                this.executionParticipant = executionParticipant;
069        }
070
071        public ExecutionException(String message, Throwable cause, ProgressRecorder progressRecorder, ExecutionParticipant executionParticipant) {
072                this(message, cause, progressRecorder);
073                this.executionParticipant = executionParticipant;
074        }
075        
076        public ExecutionParticipant getExecutionParticipant() {
077                return executionParticipant;
078        }
079
080}
