001package org.nasdanika.common;
002
003/**
004 * Implementations of this interface are invoked by the computing context to 
005 * obtain property values to return to the client.
006 * @author Pavel
007 *
008 */
009public interface PropertyComputer {
010        
011        /**
012         * Computes property value.
013         * @param <T>
014         * @param context
015         * @param key Property computer registration key.
016         * @param path Key path after the registration key or null if there is no path.
017         * @param type
018         * @return
019         */
020        <T> T compute(Context context, String key, String path, Class<T> type);
021
022}
