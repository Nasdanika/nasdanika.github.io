001package org.nasdanika.common;
002
003/**
004 * NOP implementation which tracks cancellation status and returns a new {@link NullProgressMonitor} from ``split()``.
005 * @author Pavel
006 *
007 */
008public class NullProgressMonitor implements ProgressMonitor {
009        
010        public NullProgressMonitor() {}
011
012        @Override
013        public void close() {}
014        
015        private boolean cancelled;
016
017        @Override
018        public boolean isCancelled() {
019                return cancelled;
020        }
021
022        @Override
023        public ProgressMonitor split(String taskName, double size, Object... data) {
024                return this;
025        }
026
027        @Override
028        public void worked(Status status, double work, String progressMessage, Object... data) {
029                if (status == Status.CANCEL) {
030                        cancelled = true;
031                }
032        }
033
034        @Override
035        public ProgressMonitor setWorkRemaining(double size) {
036                // NOP  
037                return this;
038        }
039
040}
