001package org.nasdanika.ncore.util;
002
003import java.util.Optional;
004
005import org.eclipse.emf.common.util.URI;
006import org.eclipse.emf.ecore.EObject;
007
008/**
009 * Interface to find EObjects. 
010 * Can be implemented by resources which facade systems with a large number of objects where iteration is inefficient.
011 * @author Pavel
012 *
013 */
014public interface EObjectLocator {
015        
016        /**
017         * Finds EObject by URI.
018         * @param uri
019         * @return Optional if this locator has an authoritative answer regarding existence or non-existence of an EObject matching the URI.
020         * Null if this locator doesn't have an authoritative answer - it doesn't know, but the EObject is not in the locator for sure, so the iterator can be pruned. 
021         */
022        Optional<EObject> get(URI uri);
023
024}
