001package org.nasdanika.ncore.util;
002
003import java.util.Optional;
004
005import org.eclipse.emf.common.notify.Notifier;
006import org.eclipse.emf.common.util.TreeIterator;
007import org.eclipse.emf.common.util.URI;
008import org.eclipse.emf.ecore.EObject;
009import org.eclipse.emf.ecore.resource.impl.ResourceSetImpl;
010import org.nasdanika.common.Util;
011import org.nasdanika.ncore.ModelElement;
012
013public class NcoreResourceSet extends ResourceSetImpl {
014
015        @Override
016        public EObject getEObject(URI uri, boolean loadOnDemand) {              
017                TreeIterator<Notifier> cit = getAllContents();
018                while (cit.hasNext()) {
019                        Notifier next = cit.next();
020                        if (next instanceof EObjectLocator) {
021                                cit.prune();
022                                Optional<EObject> ret = ((EObjectLocator) next).get(uri);
023                                if (ret != null) {
024                                        return ret.orElse(null);
025                                }
026                        }
027                        if (next instanceof EObject) {
028                                EObject nextEObject = (EObject) next;
029                                if (matchURI(nextEObject, uri)) {
030                                        return nextEObject;
031                                }
032                        }
033                }
034                
035                return super.getEObject(uri, loadOnDemand);
036        }
037        
038        /**
039         * Matches {@link EObject} to {@link URI}. This implementation uses ModelElement.getUri() to 
040         * match containment path URI's and ModelElement.getUuid() to match URI's with <code>uuid</code> scheme.  
041         * uses  
042         * @param eObj
043         * @param uri
044         * @return
045         */
046        protected boolean matchURI(EObject eObj, URI uri) {
047                if (NcoreUtil.isEMapEntry(eObj)) {
048                        return false; // Ignoring EMap entries.
049                }
050                for (URI objUri: NcoreUtil.getUris(eObj)) {
051                        if (objUri != null && objUri.equals(uri)) {
052                                return true;
053                        }
054                }
055                if (eObj instanceof ModelElement) {
056                        String uuid = ((ModelElement) eObj).getUuid();
057                        if (!Util.isBlank(uuid)) {
058                                URI uuidUri = URI.createURI("uuid:" + uuid);
059                                if (uuidUri.equals(uri)) {
060                                        return true;
061                                }
062                        }
063                }
064                return false;
065        }
066
067}
