001package org.nasdanika.persistence;
002
003import org.nasdanika.common.Util;
004
005/**
006 * Position in a text file - yaml, json, ...
007 * @author Pavel Vlasov
008 *
009 */
010public interface Marker {
011        
012        /**
013         * Position within resource, e.g line number, line and column number, URI fragment, some kind of path, character/byte offset, sheet row and column in Excel documents, ....
014         * @return
015         */
016        String getPosition();
017        
018        /**
019         * Location of resource. E.g. file path or URL.
020         * @return
021         */
022        String getLocation();
023        
024        static String toString(Marker marker) {
025                if (marker == null) {
026                        return "";
027                }
028                return (Util.isBlank(marker.getLocation()) ? "" : marker.getLocation()) + " " + (Util.isBlank(marker.getPosition()) ? "" : marker.getPosition());
029        }       
030
031}
