001package org.nasdanika.persistence;
002
003import java.util.List;
004import java.util.Map;
005import java.util.function.Function;
006
007import org.eclipse.emf.common.util.URI;
008import org.nasdanika.common.ProgressMonitor;
009
010/**
011 * If config is a map loads each entry value creating elements using value factory and then loading them.
012 * @author Pavel
013 *
014 * @param <V>
015 */
016public class FeatureObjectMapAttribute<K, V extends FeatureObject> extends MapAttribute<K,V> {
017
018
019        private Function<K, V> valueFactory;
020
021        /**
022         * 
023         * @param key
024         * @param valueFactory Factory of values which takes key as an argument.
025         * @param isDefault
026         * @param required
027         * @param defaultValue
028         * @param description
029         * @param exclusiveWith
030         */
031        public FeatureObjectMapAttribute(
032                        Object key, 
033                        Function<K,V> valueFactory,
034                        boolean isDefault, 
035                        boolean required, 
036                        Map<K,V> defaultValue, 
037                        String description, 
038                        Object... exclusiveWith) {
039                super(key, isDefault, required, defaultValue, description, exclusiveWith);
040                this.valueFactory = valueFactory;
041        }
042
043        /**
044         * Creates map value.
045         * @param loader
046         * @param element
047         * @param base
048         * @param progressMonitor
049         * @param marker
050         * @return
051         */
052        @Override
053        protected V createValue(ObjectLoader loader, K key, Object value, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers) { 
054                V ret = valueFactory.apply(key);
055                ret.load(loader, value, base, progressMonitor, markers);
056                return ret;
057        }
058        
059}
