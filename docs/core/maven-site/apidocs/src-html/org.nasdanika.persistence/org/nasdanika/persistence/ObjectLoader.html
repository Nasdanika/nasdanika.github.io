028/**
029 * Loads objects from configuration, e.g. YAML or JSON.
030 * Objects are loaded as follows: 
031 * 
032 * * If config value is a map with a single element, then its key is an object type to be instantiated. Type is passed to create() method and is implementation-specific. E.g. a factory may use aliases/logical names for a 
033 * pre-defined set of supported objects/components.  
034 * * If config value is a multi-value map, then the result would be a map with each value instantiated as explained here.
035 * * If config value is a list then the result would be a list with each element instantiated as explained here.
036 * * Otherwise the value is returned as-is. 
037 * 
038 * {@link ExecutionParticipant}'s factory loading methods cast instantiated objects to a specific execution participant type factory and wrap values. Lists, and maps are wrapped into compound execution participants of requested type.
039 *   
040 * @author Pavel
041 *
042 */
043public interface ObjectLoader {
044        
045        /**
046         * Creates an object of requested type with a given config.
047         * @param ObjectLoader for creating object parts. This factory or parent/root loader for chained loaders.
048         * @param type
049         * @param config
050         * @param base Base URL for resolving references.
051         * @param progressMonitor 
052         * @param marker Optional source marker for troubleshooting.  
053         * @return
054         */
055        Object create(ObjectLoader loader, String type, Object config, URI base, ProgressMonitor progressMonitor, List<? extends Marker> markers);
056        
057        @SuppressWarnings("unchecked")
058        default Object load(Object spec, URI base, ProgressMonitor progressMonitor) {
059                // map
060                if (spec instanceof Map) {
061                        Map<String,Object> map = (Map<String,Object>) spec;
062                        // single
063                        if (map.size() == 1) {
064                                for (Entry<String, Object> e: map.entrySet()) {
065                                        return create(this, e.getKey(), e.getValue(), base, progressMonitor, Util.getMarkers(map, e.getKey()));
066                                }
067                        }
068                        // otherwise
069                        Map<String,Object> ret = new LinkedHashMap<>();
070                        for (Entry<String, Object> e: map.entrySet()) {
071                                ret.put(e.getKey(), load(e.getValue(), base, progressMonitor));
072                        }
073                        return ret;
074                }
075                
076                // list
077                if (spec instanceof Collection) {
078                        List<Object> ret = new ArrayList<>();
079                        for (Object e: (Collection<?>) spec) {
080                                ret.add(load(e, base, progressMonitor));
081                        }
082                        return ret;
083                }
084                
085                // Scalar - return AS-IS
086                return spec;
087        }
088        
089        default Object loadYaml(String yamlString, URI base, ProgressMonitor progressMonitor) {
090                Yaml yaml = MarkingYamlConstructor.createMarkingYaml(base == null ? null : base.toString());
091                return load(yaml.load(yamlString), base, progressMonitor);
092        }
093        
094        default Object loadYaml(InputStream in, URI base, ProgressMonitor progressMonitor) throws IOException {
095                try (Reader reader = new InputStreamReader(in, getCharset())) {
096                        return loadYaml(reader, base, progressMonitor);
097                }
098        }
099        
100        default Object loadYaml(Reader reader, URI base, ProgressMonitor progressMonitor) {
101                Yaml yaml = MarkingYamlConstructor.createMarkingYaml(base == null ? null : base.toString());
102                return load(yaml.load(reader), base, progressMonitor);
103        }
104        
105        default Object loadYaml(URL url, ProgressMonitor progressMonitor) throws Exception {
106                progressMonitor.worked(1, "Loading YAML from " + url);
107                Yaml yaml = MarkingYamlConstructor.createMarkingYaml(url.toString());
108                try (Reader reader = new InputStreamReader(url.openStream(), getCharset())) {
109                        return load(yaml.load(reader), URI.createURI(url.toString()), progressMonitor);
110                } catch (MarkedYAMLException e) {
111                        throw new ConfigurationException(e.getMessage(), e, new MarkerImpl(url.toString(), e.getProblemMark()));
112                }
113        }
114        
115        default Object loadYaml(File file, ProgressMonitor progressMonitor) throws MalformedURLException, Exception {
116                return loadYaml(file.toURI().toURL(), progressMonitor);
117        }
118                
119        default Object loadJsonObject(String str, URI base, ProgressMonitor progressMonitor) {
120                JSONObject jsonObject = new JSONObject(new JSONTokener(str));
121                return load(jsonObject.toMap(), base, progressMonitor);
122        }
123        
124        default Object loadJsonObject(InputStream in, URI base, ProgressMonitor progressMonitor) throws IOException {
125                try (Reader reader = new InputStreamReader(in, getCharset())) { 
126                        return loadJsonObject(reader, base, progressMonitor);
127                }
128        }
129        
130        default Object loadJsonObject(Reader reader, URI base, ProgressMonitor progressMonitor) {
131                JSONObject jsonObject = new JSONObject(new JSONTokener(reader));
132                return load(jsonObject.toMap(), base, progressMonitor);
133        }
134        
135        default Object loadJsonObject(URL url, ProgressMonitor progressMonitor) throws IOException {
136                progressMonitor.worked(1, "Loading JSON object from " + url);
137                return loadJsonObject(url.openStream(), URI.createURI(url.toString()), progressMonitor);
138        }
139                
140        default Object loadJsonArray(String str, URI base, ProgressMonitor progressMonitor) {
141                JSONArray jsonArray = new JSONArray(new JSONTokener(str));
142                return load(jsonArray.toList(), base, progressMonitor);
143        }
144        
145        default Object loadJsonArray(InputStream in, URI base, ProgressMonitor progressMonitor) throws IOException {
146                try (Reader reader = new InputStreamReader(in, getCharset())) {
147                        return loadJsonArray(reader,  base, progressMonitor);
148                }
149        }
150        
151        default Object loadJsonArray(Reader reader, URI base, ProgressMonitor progressMonitor) {
152                JSONArray jsonArray = new JSONArray(new JSONTokener(reader));
153                return load(jsonArray.toList(), base, progressMonitor);
154        }
155        
156        default Object loadJsonArray(URL url, ProgressMonitor progressMonitor) throws Exception {
157                progressMonitor.worked(1, "Loading JSON array from " + url);
158                return loadJsonArray(url.openStream(), URI.createURI(url.toString()), progressMonitor);
159        }
160        
161        /**
162         * Override to customize charset.
163         * @return
164         */
165        default Charset getCharset() {
166                return StandardCharsets.UTF_8;
167        }
