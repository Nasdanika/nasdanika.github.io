001package org.nasdanika.resources;
002
003import java.io.InputStream;
004import java.net.URL;
005
006/**
007 * 
008 * @author Pavel
009 *
010 */
011public interface Repository {
012        
013        /**
014         * Stores content into a repository. Returns content URL relative to the base URL.
015         * @param content Resource content.
016         * @param base URL which will be referring to the stored resource, e.g. a web page URL which would contain a link to the resource.
017         * @param nameHint Hint for a resource name, e.g. report.pdf
018         * @return URL of the stored resource relative to the base URL.
019         */
020        URL store(InputStream content, URL base, String nameHint);
021
022}
