001package org.nasdanika.graph.processor.emf;
002
003import java.util.concurrent.CompletionStage;
004
005import org.nasdanika.graph.Connection;
006import org.nasdanika.graph.processor.ConnectionProcessorConfig;
007
008public class ConnectionProcessorConfigAdapter<P, H, E> extends ProcessorConfigAdapter<P, ConnectionProcessorConfig<P, H, E>> implements ConnectionProcessorConfig<P, H, E> {
009
010        public ConnectionProcessorConfigAdapter(ConnectionProcessorConfig<P, H, E> delegate) {
011                super(delegate);
012        }
013        
014        @Override
015        public Connection getElement() {
016                return delegate.getElement();
017        }
018
019        @Override
020        public CompletionStage<E> getSourceEndpoint() {
021                return delegate.getSourceEndpoint();
022        }
023
024        @Override
025        public void setSourceHandler(H sourceHandler) {
026                delegate.setSourceHandler(sourceHandler);               
027        }
028
029        @Override
030        public CompletionStage<E> getTargetEndpoint() {
031                return delegate.getTargetEndpoint();
032        }
033
034        @Override
035        public void setTargetHandler(H targetHandler) {
036                delegate.setTargetHandler(targetHandler);
037        }
038
039}
