001package org.nasdanika.graph.processor;
002
003import static java.lang.annotation.ElementType.FIELD;
004import static java.lang.annotation.ElementType.METHOD;
005import static java.lang.annotation.ElementType.TYPE;
006import static java.lang.annotation.RetentionPolicy.RUNTIME;
007
008import java.lang.annotation.Retention;
009import java.lang.annotation.Target;
010
011/**
012 * On a type this annotation is used to filter reflective factory targets using this annotation value as a predicate.
013 * Annotating a class without providing a value makes no sense.
014 * 
015 * For methods and fields this annotation indicates that the method return value or field value shall be used as introspection targets to create element processors.
016 * @author Pavel
017 *
018 */
019@Retention(RUNTIME)
020@Target({ FIELD, METHOD, TYPE })
021public @interface Factory {
022        
023        /**
024         * If not blank, the value shall be a <a href="https://docs.spring.io/spring-framework/docs/5.3.22/reference/html/core.html#expressions">Spring boolean expression</a>
025         * which is evaluated in the context of an element. 
026         * @return
027         */
028        String value() default "";
029        
030        /**
031         * @return {@link IntrospectionLevel}, DECLARED by default.
032         */
033        IntrospectionLevel introspect() default IntrospectionLevel.DECLARED;
034        
035        /**
036         * Matching priority. Defaults to 0. Higher priorities are matched before lower priorities.
037         * @return
038         */
039        int priority() default 0;
040
041}
