001package org.nasdanika.graph.processor;
002
003/**
004 * Level of introspection.
005 * @author Pavel
006 *
007 */
008public enum IntrospectionLevel {
009        
010        /**
011         * No introspection.
012         */
013        NONE,
014        
015        /**
016         * Accessible members
017         */
018        ACCESSIBLE,
019        
020        /**
021         * All declared members including not accessible.
022         * To access no accessible members <code>{@link java.lang.reflect.AcessibleObject}.setAccessible()</code> is called, which may result in {@link SecurityException}.
023         */
024        DECLARED
025
026}
