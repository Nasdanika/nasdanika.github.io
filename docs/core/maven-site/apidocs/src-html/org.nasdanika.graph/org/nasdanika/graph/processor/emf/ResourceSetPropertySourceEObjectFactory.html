029/**
030 * Uses {@link ResourceSet} to load objects from spec-uri.
031 * Text specs get encoded into a data URL by creating a JSON object with spec, format, and base fields. The object is converted to string and base64 encoded.
032 * application/json-eobject-spec is used as media type. 
033 * @author Pavel
034 *
035 * @param <T>
036 */
037public abstract class ResourceSetPropertySourceEObjectFactory<T extends EObject, P extends SemanticProcessor<T>> extends PropertySourceEObjectFactory<T,P> {
038        
039        private static final String EXPR_PREFIX = "expr/";
040        private static final String PROPERTIES_PREFIX = "properties/";
041        
042        /**
043         * Context for spec interpolation.
044         * @return
045         */
046        protected Context getContext() {
047                return Context.EMPTY_CONTEXT;
048        }
049        
050        /**
051         * @return {@link ResourceSet} to load objects. 
052         */
053        protected abstract ResourceSet getResourceSet();
054        
055        /**
056         * Returns an object identified by specURI by calling {@link ResourceSet}.getEObject(). A single resource root object has <code>/</code> URI fragment by default.
057         * @param specURI
058         * @param config
059         * @param progressMonitor
060         * @return
061         */
062        @SuppressWarnings("unchecked")
063        @Override
064        protected Collection<T> load(URI specURI, String specFormat, ProcessorConfig<P> config, Context context, ProgressMonitor progressMonitor) {
065                ResourceSet resourceSet = getResourceSet();
066                String semanticType = getPropertyValue(config.getElement(), getSemanticTypePropertyName());
067                if (!Util.isBlank(semanticType)) {
068                        // Proxy. Try to resolve first in case already available in the resource set.
069                        T semanticElement = (T) resourceSet.getEObject(specURI, false);
070                        if (semanticElement != null) {
071                                return Collections.singleton(semanticElement);
072                        }
073                        Yaml yaml = new Yaml();
074                        Object semanticTypeObj = yaml.load(semanticType);
075                        if (semanticTypeObj instanceof Map) {
076                                Map<?, ?> semanticTypeMap = (Map<?,?>) semanticTypeObj;
077                                Object typeName = semanticTypeMap.get("name");
078                                if (typeName instanceof String) {
079                                        Object typeNamespaceUri = semanticTypeMap.get("namespace-uri");
080                                        if (typeNamespaceUri instanceof String) {
081                                                EPackage ePackage = resourceSet.getPackageRegistry().getEPackage((String) typeNamespaceUri);
082                                                if (ePackage == null) {
083                                                        throw new NasdanikaException("EPackage not found for namespace URI " + typeNamespaceUri);                                                                                               
084                                                }
085                                                EClassifier eClassifier = ePackage.getEClassifier((String) typeName);
086                                                if (eClassifier instanceof EClass) {
087                                                        EObject proxy = ePackage.getEFactoryInstance().create((EClass) eClassifier);
088                                                        if (proxy instanceof InternalEObject) {
089                                                                ((InternalEObject) proxy).eSetProxyURI(specURI);
090                                                                return Collections.singleton((T) proxy);
091                                                        } else {
092                                                                throw new NasdanikaException("Cannot set proxy URI, not an instance of InteralEObject: " + proxy);                                                                                                                                                                                                                      
093                                                        }
094                                                } else {
095                                                        throw new NasdanikaException("EClass '" + typeName + "' not found in " + typeNamespaceUri);                                                                                                                                                     
096                                                }
097                                        } else {
098                                                throw new NasdanikaException("Semantic type map shall have 'namespace-uri' entry with String value: " + semanticType);                                  
099                                        }                                       
100                                } else {
101                                        throw new NasdanikaException("Semantic type map shall have 'name' entry with String value: " + semanticType);                                   
102                                }                               
103                        } else {
104                                throw new NasdanikaException("Semantic type shall be a map, got " + semanticTypeObj.getClass() + " for " + semanticType);
105                        }
106                }
107                
108                if (specURI.hasFragment()) {
109                        T semanticElement = (T) resourceSet.getEObject(specURI, isLoadOnDemand(config.getElement())); 
110                        return semanticElement == null ? Collections.emptySet() : Collections.singleton(semanticElement);
111                }
112                Resource resource = resourceSet.getResource(specURI, isLoadOnDemand(config.getElement()));
113                return resource == null ? Collections.emptyList() : (Collection<T>) resource.getContents();
114        }
115        
116        protected boolean isLoadOnDemand(Element element) {
117                String loadOnDemandVal = getPropertyValue(element, getSemanticLoadOnDemanPropertyName());
118                return Util.isBlank(loadOnDemandVal) || "true".equals(loadOnDemandVal);
119        }
120        
121        protected String getSemanticLoadOnDemanPropertyName() {
122                return "semantic-load-on-demand";
123        }               
124
125        /**
126         * YAML map with name and namespace-uri keys to resolve proxy {@link EClass}.
127         * @return
128         */
129        protected String getSemanticTypePropertyName() {
130                return "semantic-type";
131        }                       
132        
133        // TODO semantic-type, semanticLoadOnDemand, isLoadOnDemand     
134
135        /**
136         * Encodes spec as a data URI and calls load(uri, ...)
137         */
138        @Override
139        protected Collection<T> load(String spec, String specFormat, URI specBase, ProcessorConfig<P> config, Context context, ProgressMonitor progressMonitor) {
140                String interpolatedSpec = createElementContext(config).interpolateToString(spec);
141                URI specURI = ObjectLoaderResource.encode(interpolatedSpec, specFormat, specBase, getElementQualifier(config)); 
142                return load(specURI, specFormat, config, context, progressMonitor);
143        }
144
145        /**
146         * @param config
147         * @return Qualifier to differentiate identical specifications.
148         */
149        protected Object getElementQualifier(ProcessorConfig<P> config) {
150                Element element = config.getElement();
151                return element.getClass() + ":" + element.hashCode();
152        }
153                
154        protected Context createElementContext(ProcessorConfig<P> config) {
155                Context context = getContext();
156                
157                Context propertyAndExpressionComputerContext = new Context() {
158
159                        @SuppressWarnings("unchecked")
160                        @Override
161                        public Object get(String key) {
162                                if (key == null) {
163                                        return null;
164                                }
165                                Element element = config.getElement();
166                                if (key.startsWith(PROPERTIES_PREFIX)) {
167                                        if (element instanceof PropertySource) {
168                                                return ((PropertySource<Object,?>) element).getProperty(key.substring(PROPERTIES_PREFIX.length()));
169                                        }
170                                } else if (key.startsWith(EXPR_PREFIX)) {
171                                        String expr = key.substring(EXPR_PREFIX.length());
172                                        if (!org.nasdanika.common.Util.isBlank(expr)) {
173                                                ExpressionParser parser = new SpelExpressionParser();
174                                                Expression exp = parser.parseExpression(expr);
175                                                EvaluationContext evaluationContext = createEvaluationContext();
176                                                evaluationContext.setVariable("context", context);
177                                                evaluationContext.setVariable("config", config);
178                                                return exp.getValue(evaluationContext, element);                                                
179                                        }                                                                                                       
180                                }
181                                return null;
182                        }
183
184                        @Override
185                        public <V> V get(Class<V> type) {
186                                return null;
187                        }
188                        
189                };
190                
191                return propertyAndExpressionComputerContext.compose(context.computingContext());
192        }
