001package org.nasdanika.graph.processor.emf;
002
003import java.util.Map;
004import java.util.concurrent.CompletionStage;
005
006import org.eclipse.emf.common.notify.impl.AdapterImpl;
007import org.nasdanika.graph.Element;
008import org.nasdanika.graph.processor.ProcessorConfig;
009import org.nasdanika.graph.processor.ProcessorInfo;
010
011public class ProcessorConfigAdapter<P, D extends ProcessorConfig<P>> extends AdapterImpl implements ProcessorConfig<P> {
012
013        protected D delegate;
014
015        public ProcessorConfigAdapter(D delegate) {
016                this.delegate = delegate;
017        }
018
019        @Override
020        public boolean isAdapterForType(Object type) {
021                return ProcessorConfig.class == type;
022        }
023
024        @Override
025        public Element getElement() {
026                return delegate.getElement();
027        }
028
029        @Override
030        public Map<Element, ProcessorInfo<P>> getChildProcessorsInfo() {
031                return delegate.getChildProcessorsInfo();
032        }
033
034        @Override
035        public CompletionStage<ProcessorInfo<P>> getParentProcessorInfo() {
036                return delegate.getParentProcessorInfo();
037        }
038
039        @Override
040        public CompletionStage<Map<Element, ProcessorInfo<P>>> getRegistry() {
041                return delegate.getRegistry();
042        }       
043        
044}
