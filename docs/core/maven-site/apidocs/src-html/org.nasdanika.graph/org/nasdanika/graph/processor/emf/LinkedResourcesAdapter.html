001package org.nasdanika.graph.processor.emf;
002
003import java.util.ArrayList;
004import java.util.Collection;
005
006import org.eclipse.emf.common.notify.impl.AdapterImpl;
007import org.eclipse.emf.ecore.EObject;
008import org.eclipse.emf.ecore.resource.Resource;
009
010/**
011 * Used to associate linked resources with {@link EObject} to pass them from one loading/processing stage to another.
012 * @return
013 */
014public class LinkedResourcesAdapter extends AdapterImpl {
015
016        private Collection<Resource> linkedResources = new ArrayList<>();
017        
018        public LinkedResourcesAdapter() {}
019        
020        public Collection<Resource> getLinkedResources() {
021                return linkedResources;
022        }
023
024        @Override
025        public boolean isAdapterForType(Object type) {
026                return LinkedResourcesAdapter.class == type;
027        }
028
029}
