001package org.nasdanika.drawio.comparators;
002
003import java.util.Comparator;
004import java.util.Objects;
005
006import org.nasdanika.drawio.ModelElement;
007
008/**
009 * Compares {@link ModelElement}s by property value.
010 * @author Pavel
011 *
012 */
013public class PropertyModelElementComparator implements Comparator<ModelElement> {
014        
015        private String property;
016        private boolean descending;
017
018        public PropertyModelElementComparator(String property, boolean descending) {
019                this.property = property;
020                this.descending = descending;
021        }
022        
023        @Override
024        public int compare(ModelElement o1, ModelElement o2) {                          
025                if (Objects.equals(o1, o2)) {
026                        return 0;
027                }
028                
029                if (o1 instanceof ModelElement && o2 instanceof ModelElement) {
030                        String p1 = ((ModelElement) o1).getProperty(property);
031                        String p2 = ((ModelElement) o2).getProperty(property);
032                        
033                        if (org.nasdanika.common.Util.isBlank(p1)) {
034                                if (!org.nasdanika.common.Util.isBlank(p2)) {
035                                        return 1;
036                                }
037                        }
038                        
039                        if (org.nasdanika.common.Util.isBlank(p2)) {
040                                if (!org.nasdanika.common.Util.isBlank(p1)) {
041                                        return -1;
042                                }
043                        }
044                        
045                        if (!org.nasdanika.common.Util.isBlank(p1) && !org.nasdanika.common.Util.isBlank(p2)) {
046                                int cmp = p1.compareTo(p2);
047                                if (cmp != 0) {
048                                        return descending ? -cmp : cmp;
049                                }
050                        }                                               
051                }
052
053                return o1.hashCode() - o2.hashCode();
054        }
055
056}
